28std::pair<RegisterPath, DecoratorType> getPathAndType(RegisterPath path) {
29 DecoratorType type;
30 path.setAltSeparator(".");
31 auto typeName = path.getComponents().back();
32 if(typeName == "casted") {
33 type = DecoratorType::C_style_conversion;
34 }
35 else if(typeName == "limiting") {
36 type = DecoratorType::limiting;
37 }
38 else {
39 throw ChimeraTK::logic_error("Decorator type " + typeName + " not supported");
40 }
41
42 // drop last component
43 path--;
44
45 return {path, type};
46}

50class DecoratorBackend : public ExceptionDummy {
51 public:
52 DecoratorBackend(std::string mapFileName) : ExceptionDummy(mapFileName) {
53 OVERRIDE_VIRTUAL_FUNCTION_TEMPLATE(ExceptionDummy, getRegisterAccessor_impl);
54 }
55
56 // Declare that we are going to override getRegisterAccessor_impl
57 DEFINE_VIRTUAL_FUNCTION_OVERRIDE_VTABLE(ExceptionDummy, getRegisterAccessor_impl,
58 boost::shared_ptr<NDRegisterAccessor<T>>(const RegisterPath&, size_t, size_t, AccessModeFlags));
59
60 template<typename UserType>
61 boost::shared_ptr<NDRegisterAccessor<UserType>> getRegisterAccessor_impl(
62 const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
63 if(flags.has(AccessMode::raw)) throw ChimeraTK::logic_error("Raw accessors not supported");
64
65 auto [path, type] = getPathAndType(registerPathName);
66
67 return getTypeChangingDecorator<UserType>(CALL_BASE_FUNCTION_TEMPLATE(ExceptionDummy, getRegisterAccessor_impl,
68 float, path, numberOfWords, wordOffsetInRegister, flags),
69 type);
70 }
