42 template<typename UserType>
43 class NDRegisterAccessorDecoratorImpl<UserType, UserType> : public NDRegisterAccessor<UserType> {
44 public:
45 NDRegisterAccessorDecoratorImpl(std::string const& name, AccessModeFlags accessModeFlags,
46 std::string const& unit = std::string(TransferElement::unitNotSet),
47 std::string const& description = std::string())
48 : NDRegisterAccessor<UserType>(name, accessModeFlags, unit, description) {
49 FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(getAsCooked_impl);
50 FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(setAsCooked_impl);
51 }
52
53 void doPreRead(TransferType type) override { _target->preRead(type); }
54
55 void doPostRead(TransferType type, bool updateDataBuffer) override {
56 _target->setActiveException(this->_activeException);
57 _target->postRead(type, updateDataBuffer);
58
59 // Decorators have to copy meta data even if updateDataBuffer is false
60 this->_dataValidity = _target->dataValidity();
61 this->_versionNumber = _target->getVersionNumber();
62
63 if(!updateDataBuffer) return;
64 for(size_t i = 0; i < _target->getNumberOfChannels(); ++i) buffer_2D[i].swap(_target->accessChannel(i));
65 }
66
67 void doPreWrite(TransferType type, VersionNumber versionNumber) override {
68 for(size_t i = 0; i < _target->getNumberOfChannels(); ++i) buffer_2D[i].swap(_target->accessChannel(i));
69 _target->setDataValidity(this->_dataValidity);
70 _target->preWrite(type, versionNumber);
71 }
72
73 void doPostWrite(TransferType type, VersionNumber versionNumber) override {
74 // swap back buffers unconditionally (even if postWrite() throws) at the end of this function
75 auto _ = cppext::finally([&] {
76 for(size_t i = 0; i < _target->getNumberOfChannels(); ++i) buffer_2D[i].swap(_target->accessChannel(i));
77 });
78 _target->setActiveException(this->_activeException);
79 _target->postWrite(type, versionNumber);
80 }

82 template<typename COOKED_TYPE>
83 COOKED_TYPE getAsCooked_impl(unsigned int channel, unsigned int sample) {
84 // Swap the user buffer back into the target, so we can call the target's getAsCooked()
85 // Channels might cached. Just swap the channel content.
86 this->buffer_2D[channel].swap(_target->accessChannel(channel));
87 auto retVal = _target->template getAsCooked<COOKED_TYPE>(channel, sample);
88 // Swap the buffer back
89 this->buffer_2D[channel].swap(_target->accessChannel(channel));
90
91 return retVal;
92 }
93
94 template<typename COOKED_TYPE>
95 void setAsCooked_impl(unsigned int channel, unsigned int sample, COOKED_TYPE value) {
96 // Swap the user buffer back into the target, so the target's setAsCooked() puts it in,
97 // then swap the buffer back out of the target.
98 // Channels might cached. Just swap the channel content.
99 this->buffer_2D[channel].swap(_target->accessChannel(channel));
100 _target->template setAsCooked<COOKED_TYPE>(channel, sample, value);
101 this->buffer_2D[channel].swap(_target->accessChannel(channel));
102 }
103
104 protected:
105 using ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D;
106
108 boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> _target;
109 DEFINE_VIRTUAL_FUNCTION_TEMPLATE_VTABLE_FILLER(NDRegisterAccessorDecorator_impl<UserType>, getAsCooked_impl, 2);
110 DEFINE_VIRTUAL_FUNCTION_TEMPLATE_VTABLE_FILLER(NDRegisterAccessorDecorator_impl<UserType>, setAsCooked_impl, 3);
111 };

123 : detail::NDRegisterAccessorDecoratorImpl<UserType, TargetUserType>(
124 target->getName(), target->getAccessModeFlags(), target->getUnit(), target->getDescription()) {
125 _target = target;
126 this->_readQueue = _target->getReadQueue();
127 this->_exceptionBackend = _target->getExceptionBackend();

129 // set ID to match the decorated accessor

131

137 buffer_2D.resize(_target->getNumberOfChannels());
138 for(size_t i = 0; i < _target->getNumberOfChannels(); ++i) buffer_2D[i].resize(_target->getNumberOfSamples());
139 }

202template<typename UserType, typename TargetUserType>
203void ChimeraTK::NDRegisterAccessorDecorator<UserType, TargetUserType>::replaceTransferElement(
204 boost::shared_ptr<ChimeraTK::TransferElement> newElement) {
205 auto casted = boost::dynamic_pointer_cast<ChimeraTK::NDRegisterAccessor<TargetUserType>>(newElement);
206 if(casted && newElement->mayReplaceOther(_target)) {
207 if(_target != newElement) {
208 _target = detail::createCopyDecorator<TargetUserType>(casted);
209 }
210 }
211 else {
212 _target->replaceTransferElement(newElement);
213 }
214 _target->setExceptionBackend(this->_exceptionBackend);
215}
