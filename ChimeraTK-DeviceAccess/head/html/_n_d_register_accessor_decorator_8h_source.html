179 template<typename UserType>
180 void NDRegisterAccessorDecoratorImpl<UserType, UserType>::doPostRead(TransferType type, bool updateDataBuffer) {
181 _target->setActiveException(this->_activeException);
182 _target->postRead(type, updateDataBuffer);
183
184 // Decorators have to copy meta data even if updateDataBuffer is false
185 this->_dataValidity = _target->dataValidity();
186 this->_versionNumber = _target->getVersionNumber();
187
188 if(!updateDataBuffer) {
189 return;
190 }
191 for(size_t i = 0; i < _target->getNumberOfChannels(); ++i) {
192 buffer_2D[i].swap(_target->accessChannel(i));
193 }
194 }
195

198 template<typename UserType>
199 void NDRegisterAccessorDecoratorImpl<UserType, UserType>::doPreWrite(
200 TransferType type, VersionNumber versionNumber) {
201 for(size_t i = 0; i < _target->getNumberOfChannels(); ++i) {
202 buffer_2D[i].swap(_target->accessChannel(i));
203 }
204 _target->setDataValidity(this->_dataValidity);
205 _target->preWrite(type, versionNumber);
206 }

208 /******************************************************************************************************************/
209
210 template<typename UserType>
211 void NDRegisterAccessorDecoratorImpl<UserType, UserType>::doPostWrite(
212 TransferType type, VersionNumber versionNumber) {
213 // swap back buffers unconditionally (even if postWrite() throws) at the end of this function
214 auto _ = cppext::finally([&] {
215 for(size_t i = 0; i < _target->getNumberOfChannels(); ++i) {
216 buffer_2D[i].swap(_target->accessChannel(i));
217 }
218 });
219 _target->setActiveException(this->_activeException);
220 _target->postWrite(type, versionNumber);
221 }

225 template<typename UserType>
226 template<typename COOKED_TYPE>
227 COOKED_TYPE NDRegisterAccessorDecoratorImpl<UserType, UserType>::getAsCooked_impl(
228 unsigned int channel, unsigned int sample) {
229 // Swap the user buffer back into the target, so we can call the target's getAsCooked()
230 // Channels might cached. Just swap the channel content.
231 this->buffer_2D[channel].swap(_target->accessChannel(channel));
232 auto retVal = _target->template getAsCooked<COOKED_TYPE>(channel, sample);
233 // Swap the buffer back
234 this->buffer_2D[channel].swap(_target->accessChannel(channel));
235
236 return retVal;
237 }

241 template<typename UserType>
242 template<typename COOKED_TYPE>
243 void NDRegisterAccessorDecoratorImpl<UserType, UserType>::setAsCooked_impl(
244 unsigned int channel, unsigned int sample, COOKED_TYPE value) {
245 // Swap the user buffer back into the target, so the target's setAsCooked() puts it in,
246 // then swap the buffer back out of the target.
247 // Channels might cached. Just swap the channel content.
248 this->buffer_2D[channel].swap(_target->accessChannel(channel));
249 _target->template setAsCooked<COOKED_TYPE>(channel, sample, value);
250 this->buffer_2D[channel].swap(_target->accessChannel(channel));
251 }

271 template<typename UserType, typename TargetUserType>
272 void NDRegisterAccessorDecorator<UserType, TargetUserType>::initFromTarget(
273 const boost::shared_ptr<ChimeraTK::NDRegisterAccessor<TargetUserType>>& target) {
274 _target = target;
275 this->_readQueue = _target->getReadQueue();
276 this->_exceptionBackend = _target->getExceptionBackend();
277
278 // set ID to match the decorated accessor
279 this->_id = _target->getId();
280
281 // Initialise buffer meta data from target
282 this->_dataValidity = _target->dataValidity();
283 this->_versionNumber = _target->getVersionNumber();
284
285 // initialise buffers
286 buffer_2D.resize(_target->getNumberOfChannels());
287 for(size_t i = 0; i < _target->getNumberOfChannels(); ++i) {
288 buffer_2D[i].resize(_target->getNumberOfSamples());
289 }
290
291 if(target->isReadTransactionInProgress()) {
292 // In case accessor was already used from ReadAnyGroup, it has readTransactionInProgress set. We must copy
293 // state over to decorator, otherwise postRead will be ignored. We do this simply by calling preRead.
294 // It will set the flag and delegate to target->preRead which does nothing.
295 this->preRead(TransferType::read);
296 }
297 if(target->isWriteTransactionInProgress()) {
298 this->preWrite(TransferType::write, this->_versionNumber);
299 }
300 }

419 /********************************************************************************************************************/
420
421 template<typename UserType, typename TargetUserType>
422 boost::shared_ptr<NDRegisterAccessor<UserType>> NDRegisterAccessorDecorator<UserType, TargetUserType>::
423 decorateDeepInside([[maybe_unused]] std::function<boost::shared_ptr<NDRegisterAccessor<UserType>>(
424 const boost::shared_ptr<NDRegisterAccessor<UserType>>&)>
425 factory) {
426 if constexpr(std::is_same_v<UserType, TargetUserType>) {
427 auto res = _target->decorateDeepInside(factory);
428 if(!res) {
429 res = factory(_target);
430 if(res) {
431 initFromTarget(res);
432 }
433 }
434 else {
435 // target was modified. propagate modified state upwards
436 // (e.g. state of future_queue modified by DataConsistencyDecorator)
437 initFromTarget(_target);
438 }
439 return res;
440 }
441 else {
442 // for TargetUserType != UserType the 'inside' decoration makes no sense, so disallow it
443 return {};
444 }
445 }
