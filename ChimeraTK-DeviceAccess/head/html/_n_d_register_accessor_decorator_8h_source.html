175 template<typename UserType>
176 void NDRegisterAccessorDecoratorImpl<UserType, UserType>::doPostRead(TransferType type, bool updateDataBuffer) {
177 _target->setActiveException(this->_activeException);
178 _target->postRead(type, updateDataBuffer);
179
180 // Decorators have to copy meta data even if updateDataBuffer is false
181 this->_dataValidity = _target->dataValidity();
182 this->_versionNumber = _target->getVersionNumber();
183
184 if(!updateDataBuffer) {
185 return;
186 }
187 for(size_t i = 0; i < _target->getNumberOfChannels(); ++i) {
188 buffer_2D[i].swap(_target->accessChannel(i));
189 }
190 }
191

194 template<typename UserType>
195 void NDRegisterAccessorDecoratorImpl<UserType, UserType>::doPreWrite(
196 TransferType type, VersionNumber versionNumber) {
197 for(size_t i = 0; i < _target->getNumberOfChannels(); ++i) {
198 buffer_2D[i].swap(_target->accessChannel(i));
199 }
200 _target->setDataValidity(this->_dataValidity);
201 _target->preWrite(type, versionNumber);
202 }

206 template<typename UserType>
207 void NDRegisterAccessorDecoratorImpl<UserType, UserType>::doPostWrite(
208 TransferType type, VersionNumber versionNumber) {
209 // swap back buffers unconditionally (even if postWrite() throws) at the end of this function
210 auto _ = cppext::finally([&] {
211 for(size_t i = 0; i < _target->getNumberOfChannels(); ++i) {
212 buffer_2D[i].swap(_target->accessChannel(i));
213 }
214 });
215 _target->setActiveException(this->_activeException);
216 _target->postWrite(type, versionNumber);
217 }

221 template<typename UserType>
222 template<typename COOKED_TYPE>
223 COOKED_TYPE NDRegisterAccessorDecoratorImpl<UserType, UserType>::getAsCooked_impl(
224 unsigned int channel, unsigned int sample) {
225 // Swap the user buffer back into the target, so we can call the target's getAsCooked()
226 // Channels might cached. Just swap the channel content.
227 this->buffer_2D[channel].swap(_target->accessChannel(channel));
228 auto retVal = _target->template getAsCooked<COOKED_TYPE>(channel, sample);
229 // Swap the buffer back
230 this->buffer_2D[channel].swap(_target->accessChannel(channel));
231
232 return retVal;
233 }

237 template<typename UserType>
238 template<typename COOKED_TYPE>
239 void NDRegisterAccessorDecoratorImpl<UserType, UserType>::setAsCooked_impl(
240 unsigned int channel, unsigned int sample, COOKED_TYPE value) {
241 // Swap the user buffer back into the target, so the target's setAsCooked() puts it in,
242 // then swap the buffer back out of the target.
243 // Channels might cached. Just swap the channel content.
244 this->buffer_2D[channel].swap(_target->accessChannel(channel));
245 _target->template setAsCooked<COOKED_TYPE>(channel, sample, value);
246 this->buffer_2D[channel].swap(_target->accessChannel(channel));
247 }

251 template<typename UserType>
252 boost::shared_ptr<NDRegisterAccessor<UserType>> NDRegisterAccessorDecoratorImpl<UserType, UserType>::
253 decorateDeepInside([[maybe_unused]] std::function<boost::shared_ptr<NDRegisterAccessor<UserType>>(
254 const boost::shared_ptr<NDRegisterAccessor<UserType>>&)>
255 factory) {
256 auto res = _target->decorateDeepInside(factory);
257 if(!res) {
258 res = factory(_target);
259 if(res) {
260 _target = res;
261 }
262 }
263 return res;
264 }

275 NDRegisterAccessorDecorator<UserType, TargetUserType>::NDRegisterAccessorDecorator(
276 const boost::shared_ptr<ChimeraTK::NDRegisterAccessor<TargetUserType>>& target)
277 : detail::NDRegisterAccessorDecoratorImpl<UserType, TargetUserType>(
278 target->getName(), target->getAccessModeFlags(), target->getUnit(), target->getDescription()) {
279 _target = target;
280 this->_readQueue = _target->getReadQueue();
281 this->_exceptionBackend = _target->getExceptionBackend();
282
283 // set ID to match the decorated accessor
284 this->_id = _target->getId();
285
286 // Initialise buffer meta data from target
287 this->_dataValidity = _target->dataValidity();
288 this->_versionNumber = _target->getVersionNumber();
289
290 // initialise buffers
291 buffer_2D.resize(_target->getNumberOfChannels());
292 for(size_t i = 0; i < _target->getNumberOfChannels(); ++i) {
293 buffer_2D[i].resize(_target->getNumberOfSamples());
294 }
295 }

377 boost::shared_ptr<ChimeraTK::TransferElement> newElement) {
378 auto casted = boost::dynamic_pointer_cast<ChimeraTK::NDRegisterAccessor<TargetUserType>>(newElement);
379 if(casted && newElement->mayReplaceOther(_target)) {
380 if(_target != newElement) {
381 _target = detail::createCopyDecorator<TargetUserType>(casted);
382 }
383 }
384 else {
385 _target->replaceTransferElement(newElement);
386 }
