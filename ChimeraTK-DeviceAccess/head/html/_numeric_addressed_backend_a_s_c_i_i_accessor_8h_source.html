18 class NumericAddressedBackendASCIIAccessor : public NDRegisterAccessor<std::string> {
19 public:
20 NumericAddressedBackendASCIIAccessor(const boost::shared_ptr<DeviceBackend>& dev,
21 const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister,
22 const AccessModeFlags& flags)
23 : NDRegisterAccessor<std::string>(registerPathName, flags),
24 _dev(boost::dynamic_pointer_cast<NumericAddressedBackend>(dev)) {
25 // check for unknown flags
26 flags.checkForUnknownFlags({});
27
28 // check device backend
29 _dev = boost::dynamic_pointer_cast<NumericAddressedBackend>(dev);
30 if(!_dev) {
31 throw ChimeraTK::logic_error("NumericAddressedBackendASCIIAccessor is used with a backend which is not "
32 "a NumericAddressedBackend.");
33 }
34
35 // obtain register information
36 _registerInfo = _dev->getRegisterInfo(registerPathName);
37 assert(!_registerInfo.channels.empty());
38
39 if(_registerInfo.elementPitchBits % 8 != 0) {
40 throw ChimeraTK::logic_error("NumericAddressedBackendASCIIAccessor: Elements must be byte aligned.");
41 }
42
43 if(_registerInfo.channels.size() > 1) {
44 throw ChimeraTK::logic_error("NumericAddressedBackendASCIIAccessor is used with a 2D register.");
45 }
46
47 if(_registerInfo.channels.front().bitOffset > 0) {
48 throw ChimeraTK::logic_error("NumericAddressedBackendASCIIAccessor: Registers must be byte aligned.");
49 }
50
51 if(_registerInfo.channels.front().dataType != NumericAddressedRegisterInfo::Type::ASCII) {
52 throw ChimeraTK::logic_error("NumericAddressedBackendASCIIAccessor: Cannot be used with non-ASCII registers.");
53 }
54
55 // check number of words
56 if(numberOfWords == 0) {
57 numberOfWords = _registerInfo.getNumberOfElements();
58 }
59 if(numberOfWords + wordOffsetInRegister > _registerInfo.getNumberOfElements()) {
60 throw ChimeraTK::logic_error("Requested number of words (" + std::to_string(numberOfWords) + " + " +
61 std::to_string(wordOffsetInRegister) + ") exceeds the size (" +
62 std::to_string(_registerInfo.getNumberOfElements()) + ") of the register '" + registerPathName + "'!");
63 }
64 if(wordOffsetInRegister >= _registerInfo.getNumberOfElements()) {
65 throw ChimeraTK::logic_error("Requested offset (" + std::to_string(wordOffsetInRegister) +
66 ") exceeds the size (" + std::to_string(_registerInfo.getNumberOfElements()) + ") of the register'" +
67 registerPathName + "'!");
68 }
69
70 // change registerInfo (local copy!) to account for given offset and length override
71 _registerInfo.address += wordOffsetInRegister * _registerInfo.elementPitchBits / 8;
72 _registerInfo.nElements = numberOfWords;
73
74 // create low-level transfer element handling the actual data transfer to the hardware with raw data
75 assert(_registerInfo.elementPitchBits % 8 == 0);
76 _rawAccessor = boost::make_shared<NumericAddressedLowLevelTransferElement>(
77 _dev, _registerInfo.bar, _registerInfo.address, _registerInfo.nElements * _registerInfo.elementPitchBits / 8);
78
79 // allocated the buffers
80 NDRegisterAccessor<std::string>::buffer_2D.resize(1);
81 NDRegisterAccessor<std::string>::buffer_2D[0].resize(_registerInfo.nElements);
82
83 FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(getAsCooked_impl);
84 FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(setAsCooked_impl);
85 }
86
87 void doReadTransferSynchronously() override { _rawAccessor->readTransfer(); }

89 bool doWriteTransfer(ChimeraTK::VersionNumber versionNumber) override {
90 assert(!TransferElement::_isInTransferGroup);
91 _rawAccessor->writeTransfer(versionNumber);
92 return false;
93 }

95 void doPostRead(TransferType type, bool hasNewData) override {
96 if(!_dev->isOpen()) return; // do not delegate if exception was thrown by us in doPreWrite
97
98 _rawAccessor->setActiveException(this->_activeException);
99 _rawAccessor->postRead(type, hasNewData);
100
101 if(!hasNewData) return;
102
103 const size_t nCharsPerElement = _registerInfo.elementPitchBits / 8;
104 std::string temp;
105 temp.reserve(nCharsPerElement);
106 // Turn off linter warning. Yes, we are re-interpreting a byte stream as text.
107 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
108 auto* itsrc = reinterpret_cast<char*>(_rawAccessor->begin(_registerInfo.address));
109 for(size_t el = 0; el < _registerInfo.nElements; ++el) {
110 auto len = strnlen(itsrc, nCharsPerElement);
111 temp.resize(len);
112 std::strncpy(temp.data(), itsrc, nCharsPerElement);
113 itsrc += nCharsPerElement;
114 buffer_2D[0][el] = temp;
115 }
116
117 // we don't put the setting of the version number into the PrePostActionImplementor
118 // because it does not need template specialisation, and the implementer does not
119 // know about _versionNumber. It's just easier here.
120 this->_versionNumber = _rawAccessor->getVersionNumber();
121 this->_dataValidity = _rawAccessor->dataValidity();
122 }

124 void doPreWrite(TransferType type, VersionNumber versionNumber) override {
125 if(!_dev->isOpen()) throw ChimeraTK::logic_error("Device not opened.");
126 // raw accessor preWrite must be called before our _prePostActionsImplementor.doPreWrite(), as it needs to
127 // prepare the buffer in case of unaligned access and acquire the lock.
128 _rawAccessor->preWrite(type, versionNumber);
129
130 const size_t nCharsPerElement = _registerInfo.elementPitchBits / 8;
131 // Turn off linter warning. Yes, we are re-interpreting a byte stream as text.
132 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
133 auto* itdst = reinterpret_cast<char*>(_rawAccessor->begin(_registerInfo.address));
134 for(size_t el = 0; el < _registerInfo.nElements; ++el) {
135 std::memset(itdst, 0, nCharsPerElement);
136 std::strncpy(itdst, buffer_2D[0][el].data(), nCharsPerElement);
137 itdst += nCharsPerElement;
138 }
139
140 _rawAccessor->setDataValidity(this->_dataValidity);
141 }

143 void doPreRead(TransferType type) override {
144 if(!_dev->isOpen()) throw ChimeraTK::logic_error("Device not opened.");
145 _rawAccessor->preRead(type);

149 if(!_dev->isOpen()) return; // do not delegate if exception was thrown by us in doPreWrite
150 _rawAccessor->setActiveException(this->_activeException);
151 _rawAccessor->postWrite(type, versionNumber);
152 }

205 void replaceTransferElement(boost::shared_ptr<TransferElement> newElement) override {
206 auto casted = boost::dynamic_pointer_cast<NumericAddressedLowLevelTransferElement>(newElement);
207 if(casted && casted->isMergeable(_rawAccessor)) {
208 size_t newStartAddress = std::min(casted->_startAddress, _rawAccessor->_startAddress);
209 size_t newStopAddress = std::max(
210 casted->_startAddress + casted->_numberOfBytes, _rawAccessor->_startAddress + _rawAccessor->_numberOfBytes);
211 size_t newNumberOfBytes = newStopAddress - newStartAddress;
212 casted->changeAddress(newStartAddress, newNumberOfBytes);
213 _rawAccessor = casted;
214 }
215 _rawAccessor->setExceptionBackend(this->_exceptionBackend);
216 }
