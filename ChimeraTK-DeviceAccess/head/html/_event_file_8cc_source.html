65 #ifdef _DEBUG

77 #ifdef _DEBUG

80 #endif
81  // Only distribute once. If numInterrupts is > 1 we are discarding missed interrupts here.
82  // FIXME: should we have a variable to report this (accessible via RegisterAccessor)?
83  if(numInterrupts != 0) {
