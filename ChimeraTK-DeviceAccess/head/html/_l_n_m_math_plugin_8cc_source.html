58  void MathPlugin::openHook(const boost::shared_ptr<LogicalNameMappingBackend>& backend) {
59  // make sure backend catalogue is updated with target backend information
60  auto catalogue = backend->getRegisterCatalogue();
61 
62  // produce logic_error if MathPlugin has insufficient access rights to parameters
63  if(_isWrite && !_info.isWriteable()) {
64  throw ChimeraTK::logic_error("MathPlugin target not writeable:" + _info.name);
65  }
66  for(const auto& parpair : _parameters) {
67  auto r = catalogue.getRegister(parpair.second);
68 
69  if(!r.isReadable()) {
70  throw ChimeraTK::logic_error("MathPlugin parameter not readable:" + parpair.second);
71  }
72  }
73 
74  // we require that all values used in the formula need to be written after open, before we provide first result
75  {
76  std::unique_lock<std::recursive_mutex> lk(_writeMutex);
77  _mainValueWrittenAfterOpen = false;
78  _allParametersWrittenAfterOpen = false;
79  }
80  }

84  void MathPlugin::postParsingHook(const boost::shared_ptr<const LogicalNameMappingBackend>& backend) {
85  // whether this plugin is write mode depends on catalogue of target device so we need to update catalogue
86  // note, some target devices (e.g. DOOCS backend) provide their catalogue only on open, so it's not final here.
87  auto catalogue = backend->getRegisterCatalogue();
88  if(_isWrite) {
89  // Write direction: check that we have only lnm defined variables as parameters
90  // Current push implementation (via LNMBackendVariableAccessor<UserType>::doPostWrite) is only for variables.
91  // Therefore, we agreed to disallow non-variables, even when push-feature not requested.
92  for(const auto& parpair : _parameters) {
93  std::string pname = RegisterPath(parpair.second); // conversion to RegisterPath and back adds leading '/'
94  if(backend->_variables.find(pname) == backend->_variables.end()) {
95  throw logic_error("no LNM variable defined for parameter " + pname);
96  }
97  // set up connections for push-enabled variables. It is important that we do this already before openHook,
98  // since RegisterAccessor to Variable can be requested earlier!
99  if(_enablePushParameters) {
100  _hasPushParameter = true;
101  backend->_variables[pname].usingFormulas.insert(this);
102  }
103  }
104  if(_enablePushParameters && !_hasPushParameter) {
105  throw ChimeraTK::logic_error("MathPlugin (writing) with push_parameters requested but there is no parameter!");
106  }
107  }
108  }

144 
145  void MathPluginFormulaHelper::updateResult(ChimeraTK::VersionNumber versionNumber) {
146  // Note: _target might be invalid until _mp->_mainValueWrittenAfterOpen == true
147 
148  try {
149  std::unique_lock<std::recursive_mutex> lk(_mp->_writeMutex);
150 
151  auto paramDataValidity = updateParameters();
152 
153  if(!checkAllParametersWritten() || !_mp->_mainValueWrittenAfterOpen) {
154  return;
155  }
156 
157  assert(_mp->_lastMainValue.size() == _target->getNumberOfSamples());
158 
159  computeResult(_mp->_lastMainValue, _target->accessChannel(0));
160  // pass validity to target and delegate preWrite
161  if(paramDataValidity == ChimeraTK::DataValidity::ok && _mp->_lastMainValidity == ChimeraTK::DataValidity::ok) {
162  _target->setDataValidity(ChimeraTK::DataValidity::ok);
163  }
164  else {
165  _target->setDataValidity(ChimeraTK::DataValidity::faulty);
166  }
167 
168  // if versionNumber at target register is already greater, take it instead of supplied versionNumber
169  auto writeVs = std::max<ChimeraTK::VersionNumber>(versionNumber, _target->getVersionNumber());
170  _target->writeDestructively(writeVs);
171  }
172  catch(runtime_error& ex) {
173  // runtime_error from param.readLatest() or target->write()
174  // we could actually even ignore it, since we don't expect exceptions on param.readLatest(), and target->write()
175  // already puts backend into exception state.
176  _target->getExceptionBackend()->setException(ex.what());
177  }
178  }

192  bool MathPluginFormulaHelper::checkAllParametersWritten() {
193  if(_mp->_allParametersWrittenAfterOpen) {
194  return true;
195  }
196 
197  for(const auto& acc : _accessorMap) {
198  // Version number check will notice whether valid data has been provided after open. This works for both push and
199  // poll, since the LNM variables always pass through the version number from the write operation to the read.
200  if(acc.second->getVersionNumber() <= _backend->getVersionOnOpen()) {
201  return false;
202  }
203  }
204  _mp->_allParametersWrittenAfterOpen = true;
205  return _mp->_allParametersWrittenAfterOpen;
206  }

297  template<typename UserType>
298  void MathPluginDecorator<UserType>::doPreWrite(TransferType type, ChimeraTK::VersionNumber versionNumber) {
299  if(!_p->_isWrite) {
300  throw ChimeraTK::logic_error("This register with MathPlugin enabled is not writeable: " + _target->getName());
301  }
302  // LNM backend
303  auto backend = _h->getBackend();
304  if(!backend->isOpen()) {
305  throw ChimeraTK::logic_error("LNM backend not opened!");
306  }
307  backend->checkActiveException();
308 
309  // The readLatest might throw an exception. In this case preWrite() is never delegated and we must not call the
310  // target's postWrite().
311  _skipWriteDelegation = true;
312 
313  auto paramDataValidity = _h->updateParameters();
314 
315  // convert from UserType to double - use the target accessor's buffer as a temporary buffer (this is a bit a hack,
316  // but it is safe to overwrite the buffer and we can avoid the need for an additional permanent buffer which might
317  // not even be used if the register is never written).
318  for(size_t k = 0; k < buffer_2D[0].size(); ++k) {
319  _target->accessData(0, k) = userTypeToNumeric<double>(buffer_2D[0][k]);
320  }
321 
322  // update last written data buffer for other threads if needed
323  if(_p->_hasPushParameter) {
324  // Accquire the lock and hold it until the transaction is completed in postWrite.
325  // This is safe because it is guaranteed by the framework that pre- and post actions are called in pairs.
326  // Do this before the first call to the target, which might create its own locks.
327  _p->_writeMutex.lock();
328  // preWrite() might be called multiple times before postWrite() is called. There are multiple conditions
329  // whether the writeMutex is locked (_hasPushParameters, _skipWriteDelegation, exceptions) so we count
330  // separately how many times the lock has been aquired, so we can release it the exact right amount of times.
331  ++(_p->_writeLockCounter);
332  _p->_lastMainValue = _target->accessChannel(0);
333  _p->_lastMainValidity = _target->dataValidity();
334  _p->_mainValueWrittenAfterOpen = true;
335 
336  if(!_h->checkAllParametersWritten()) {
337  return;
338  }
339  }
340 
341  // There either are only poll-type parameters, or all push-type parameters have been received.
342  // We are OK to go through with the transfer
343  _skipWriteDelegation = false;
344 
345  // evaluate the expression and store into target accessor
346  _h->computeResult(_target->accessChannel(0), _target->accessChannel(0));
347 
348  // pass validity to target and delegate preWrite
349  if(paramDataValidity == ChimeraTK::DataValidity::ok && this->_dataValidity == ChimeraTK::DataValidity::ok) {
350  _target->setDataValidity(ChimeraTK::DataValidity::ok);
351  }
352  else {
353  _target->setDataValidity(ChimeraTK::DataValidity::faulty);
354  }
355  // if versionNumber at target register is already greater, take it instead of supplied versionNumber
356  auto writeVs = std::max<ChimeraTK::VersionNumber>(versionNumber, _target->getVersionNumber());
357  _target->preWrite(type, writeVs);
358  }

380  template<typename UserType>
381  void MathPluginDecorator<UserType>::doPostWrite(TransferType type, ChimeraTK::VersionNumber versionNumber) {
382  // Make sure the mutex is released, even if the delegated postWrite kicks out with an exception.
383  // This has to happen at the very end, after all delegations, such that the target can release all its internal locks first.
384  auto _ = cppext::finally([&] {
385  if(_p->_writeLockCounter > 0) {
386  --(_p->_writeLockCounter);
387  _p->_writeMutex.unlock();
388  }
389  });
390 
391  if(_skipWriteDelegation && (this->_activeException != nullptr)) {
392  // Something has thrown before the target's preWrite was called. Re-throw it here.
393  // Do not unlock the mutex. It never has been locked.
394  std::rethrow_exception(this->_activeException);
395  }
396 
397  if(_skipWriteDelegation) {
398  return; // the trarget preWrite() has not been executed, so stop here
399  }
400 
401  // delegate to the target
402  _target->setActiveException(this->_activeException);
403  _target->postWrite(type, versionNumber);
404  // (the "finally" lambda releasing the lock is executed here latest)
405  } // namespace LNMBackend

442  MathPluginFormulaHelper::MathPluginFormulaHelper(
443  MathPlugin* p, const boost::shared_ptr<LogicalNameMappingBackend>& backend)
444  : _backend(backend), _mp(p) {
445  auto* info = _mp->info();
446  auto length = info->length;
447 
448  _target = backend->getRegisterAccessor_impl<double>(info->getRegisterName(), 0, 0, {}, _mp->_pluginIndex + 1);
449 
450  for(const auto& parpair : _mp->_parameters) {
451  // Even push-type parameters should not be obtained with wait_for_new_data, since the variable accessor will
452  // trigger the math plugin update in its postWrite in that case. Using wait_for_new_data here would leave
453  // exceptions in the queues across recovery which would bring the backend into the exception state again.
454  auto acc = backend->getRegisterAccessor<double>(parpair.second, 0, 0, {});
455  if(acc->getNumberOfChannels() != 1) {
456  throw ChimeraTK::logic_error(
457  "The LogicalNameMapper MathPlugin supports only scalar or 1D array registers. Register name: '" +
458  info->name + "', parameter name: '" + parpair.first + "'");
459  }
460  _accessorMap[parpair.first] = acc;
461  }
462  if(_mp->_hasPushParameter) {
463  _mp->_lastMainValue.resize(length);
464  }
465 
466  // compile formula
467  compileFormula(_mp->_formula, _accessorMap, length);
468  varName = info->name;
469  }

472 
473  void MathPluginFormulaHelper::compileFormula(const std::string& formula,
474 
475  const std::map<std::string, boost::shared_ptr<ChimeraTK::NDRegisterAccessor<double>>>& parameters,
476  size_t nElements) {
477  // create exprtk parser
478  exprtk::parser<double> parser;
479 
480  // add basic constants like pi
481  symbols.add_constants();
482 
483  // Add vector manipulation functions
484  symbols.add_package(vecOpsPkg);
485 
486  // Create vector view for the value and add it to the symbol table. We need to use a vector view instead of adding
487  // the buffer directly as a vector, since our buffers might be swapped and hence the address of the data can
488  // change. The pointer used for the view is for now to a temporary vector and will become invalid once this
489  // function returns. This is acceptable, since before using the view the pointer will be updated to the right
490  // buffer. Using a nullptr instead of the temporary buffer does not work, as the buffer seems to be accessible
491  // during compilation of the formula.
492  std::vector<double> temp(nElements);
493  valueView = std::make_unique<exprtk::vector_view<double>>(exprtk::make_vector_view(temp, nElements));
494  symbols.add_vector("x", *valueView);
495 
496  // iterate parameters, add all but 'formula' parameter as a variable.
497  for(const auto& parpair : parameters) {
498  if(parpair.first == "formula") continue;
499  const auto& acc = parpair.second;
500  acc->setExceptionBackend(_backend);
501  if(acc->getNumberOfChannels() != 1) {
502  throw ChimeraTK::logic_error(
503  "The LogicalNameMapper MathPlugin supports only scalar or 1D array registers. Register name: '" + varName +
504  "', parameter name: '" + parpair.first + "'");
505  }
506  auto view = std::make_unique<exprtk::vector_view<double>>(
507  exprtk::make_vector_view(acc->accessChannel(0), acc->getNumberOfSamples()));
508  symbols.add_vector(parpair.first, *view);
509  params[acc] = std::move(view);
510  }
511 
512  // compile the expression
513  expression.register_symbol_table(symbols);
514  bool success = parser.compile(formula, expression);
515  if(!success) {
516  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + varName +
517  "': failed to compile expression '" + formula + "': " + parser.error());
518  }
519  }

523  template<typename T>
524  void MathPluginFormulaHelper::computeResult(std::vector<double>& x, std::vector<T>& resultBuffer) {
525  // inform the value view of the new data pointer - the buffer might have been swapped
526  valueView->rebase(x.data());
527 
528  // update parameter buffers
529  for(auto& p : params) {
530  p.second->rebase(p.first->accessChannel(0).data());
531  }
532 
533  // evaluate the expression, obtain the result in a way so it also works when using the return statement
534  double valueWhenNotUsingReturn = expression.value();
535  exprtk::results_context<double> results = expression.results();
536 
537  // extract result depending on how it was returned
538  if(results.count() == 0) {
539  // if results.count() is 0, the return statement presumably has not been used
540  if(resultBuffer.size() != 1) {
541  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + varName +
542  "': The expression returns a scalar but " + std::to_string(resultBuffer.size()) + " expected.");
543  }
544  resultBuffer[0] = numericToUserType<T>(valueWhenNotUsingReturn);
545  }
546  else if(results.count() == 1) {
547  // return statement has been used to return exactly one value (note: this value might be an array)
548  exprtk::type_store<double> result = results[0];
549 
550  // make sure we got a numeric result
551  if(result.type != exprtk::type_store<double>::e_scalar && result.type != exprtk::type_store<double>::e_vector) {
552  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + varName +
553  "': The expression did not return a numeric result.");
554  }
555 
556  // create vector view and check that its size matches
557  exprtk::type_store<double>::type_view<double> view(result);
558  if(view.size() != resultBuffer.size()) {
559  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + varName +
560  "': The expression returns " + std::to_string(view.size()) + " elements but " +
561  std::to_string(resultBuffer.size()) + " expected.");
562  }
563 
564  // convert and copy data into target buffer
565  for(size_t k = 0; k < view.size(); ++k) {
566  resultBuffer[k] = numericToUserType<T>(view[k]);
567  }
568  }
569  else {
570  // multiple results in return statement are unexpected
571  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + varName +
572  "': The expression returned " + std::to_string(results.count()) + " results, expect exactly one result.");
573  }
574  }
