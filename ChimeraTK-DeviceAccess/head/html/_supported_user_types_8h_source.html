221 template<typename NUMERIC>
222 NUMERIC detail::userTypeToNumeric_impl<std::string, NUMERIC>::impl(const std::string& value) {
223 // 0xABCD form hex
224 bool isHexidecimal = (value.length() > 1) and ((value[1] == 'x') or (value[1] == 'X')) and (value[0] == '0');
225
226 if constexpr(!std::is_same<NUMERIC, int8_t>::value && !std::is_same<NUMERIC, uint8_t>::value) {
227 NUMERIC v = 0;
228 std::stringstream ss;
229 if(isHexidecimal) {
230 ss << std::hex << value.substr(2);
231 }
232 else {
233 ss.str(value);
234 }
235 ss >> v;
236 return v;
237 }
238 else {
239 // int8_t and uint8_t are interpreted as char resp. unsigned char, in which case the stream operator does the
240 // wrong thing...
241 int64_t temp;
242 try {
243 if(isHexidecimal) {
244 temp = std::stoll(value, nullptr, 16);
245 }
246 else {
247 temp = std::stoll(value);
248 }
249 }
250 catch(...) {
251 // ignore any parsing errors and just return 0 in that case
252 temp = 0;
253 }
254 return detail::userTypeToNumeric_impl<decltype(temp), NUMERIC>::impl(temp);
255 }
256 }

581 [[nodiscard]] inline bool isIntegral() const {
582 switch(_value) {
583 case int8:
584 case uint8:
585 case int16:
586 case uint16:
587 case int32:
588 case uint32:
589 case int64:
590 case uint64:
591 case Boolean:
592 return true;
593 default:
594 return false;
595 }
596 }

604 case int8:

606 case int32:
607 case int64:
608 case float32:
609 case float64:
610 return true;
611 default:
612 return false;
613 }

624 case string:
