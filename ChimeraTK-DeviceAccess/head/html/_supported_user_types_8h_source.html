40 inline std::istream& operator>>(std::istream& is, Boolean& value) {
41 std::string data;
42 is >> data;
43
44 std::transform(data.begin(), data.end(), data.begin(), [](unsigned char c) { return std::tolower(c); });
45
46 if(data == "false" || data == "0" || data.empty()) {
47 value = false;
48 }
49 else {
50 value = true;
51 }
52 return is;
53 }

230 template<typename UserType, typename NUMERIC>
231 UserType detail::numericToUserType_impl<UserType, NUMERIC>::impl(NUMERIC value) {
232 typedef boost::numeric::converter<UserType, NUMERIC, boost::numeric::conversion_traits<UserType, NUMERIC>,
233 boost::numeric::def_overflow_handler, Round<NUMERIC>>
234 converter;
235 // There seems to be no other way to alter the value on negative/positive overflow than using a try-catch-block
236 // here. The overflow_handler is a stateless class and hence can either throw another exception or do nothing.
237 // It does not have any influence on the converted value, and it cannot transport the information out differently.
238 try {
239 return converter::convert(value);
240 }
241 catch(boost::numeric::negative_overflow&) {
242 return std::numeric_limits<UserType>::min();
243 }
244 catch(boost::numeric::positive_overflow&) {
245 return std::numeric_limits<UserType>::max();
246 }
247 }

300 template<typename NUMERIC>
301 NUMERIC detail::userTypeToNumeric_impl<std::string, NUMERIC>::impl(const std::string& value) {
302 // 0xABCD form hex
303 bool isHexidecimal = (value.length() > 1) and ((value[1] == 'x') or (value[1] == 'X')) and (value[0] == '0');
304
305 if constexpr(!std::is_same<NUMERIC, int8_t>::value && !std::is_same<NUMERIC, uint8_t>::value) {
306 NUMERIC v;
307 std::stringstream ss;
308 if(isHexidecimal) {
309 ss << std::hex << value.substr(2);
310 }
311 else {
312 ss.str(value);
313 }
314 ss >> v;
315 return v;
316 }
317 else {
318 // int8_t and uint8_t are interpreted as char resp. unsigned char, in which case the stream operator does the
319 // wrong thing...
320 int64_t temp;
321 try {
322 if(isHexidecimal) {
323 temp = std::stoll(value, nullptr, 16);
324 }
325 else {
326 temp = std::stoll(value);
327 }
328 }
329 catch(...) {
330 // ignore any parsing errors and just return 0 in that case
331 temp = 0;
332 }
333 return detail::userTypeToNumeric_impl<decltype(temp), NUMERIC>::impl(temp);
334 }
335 }
