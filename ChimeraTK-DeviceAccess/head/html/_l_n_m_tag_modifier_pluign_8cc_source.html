11 TagModifierPlugin::TagModifierPlugin(
12 const LNMBackendRegisterInfo& info, size_t pluginIndex, const std::map<std::string, std::string>& parameters)
13 : AccessorPlugin<TagModifierPlugin>(info, pluginIndex) {
14 std::regex re(R"(\s*,\s*)");
15
16 if(parameters.contains("add")) {
17 auto str = parameters.at("add");
18 _tagsToAdd.insert(std::sregex_token_iterator(str.begin(), str.end(), re, -1), std::sregex_token_iterator());
19 }
20
21 if(parameters.contains("remove")) {
22 auto str = parameters.at("remove");
23 _tagsToRemove.insert(std::sregex_token_iterator(str.begin(), str.end(), re, -1), std::sregex_token_iterator());
24 }
25
26 if(parameters.contains("set")) {
27 auto str = parameters.at("set");
28 _tagsToSet.insert(std::sregex_token_iterator(str.begin(), str.end(), re, -1), std::sregex_token_iterator());
29 }
30
31 if(_tagsToAdd.empty() && _tagsToRemove.empty() && _tagsToSet.empty()) {
32 throw ChimeraTK::logic_error(R"(TagModifierPlugin needs at least one of "add","remove" or "set")");
33 }
34
35 if(!_tagsToSet.empty() && !(_tagsToRemove.empty() || _tagsToAdd.empty())) {
36 throw ChimeraTK::logic_error(R"(TagModifierPlugin: "set" is mutual exclusive with "add" and "remove")");
37 }
38 }

42 void TagModifierPlugin::doRegisterInfoUpdate() {
43 if(!_tagsToSet.empty()) {
44 _info.tags = _tagsToSet;
45 }
46 else {
47 _info.tags.insert(_tagsToAdd.begin(), _tagsToAdd.end());
48 std::ranges::set_difference(_info.tags, _tagsToRemove, std::inserter(_tagsToSet, _tagsToSet.end()));
49 _info.tags = _tagsToSet;
50 }
