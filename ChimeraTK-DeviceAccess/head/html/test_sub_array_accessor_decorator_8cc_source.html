19// This decorator does nothing, except swap from its own user buffer into the target user buffer and back.
20// It implements a proper mayReplaceOther for this.
21template<typename UserType>
22class DoNothingDecorator : public NDRegisterAccessorDecorator<UserType, UserType> {
23 using NDRegisterAccessorDecorator<UserType, UserType>::NDRegisterAccessorDecorator;
24
25 // If the targets can replace each other, then the whole thing including the decorator can be replaced.
26 [[nodiscard]] bool mayReplaceOther(const boost::shared_ptr<TransferElement const>& other) const override {
27 auto casted = boost::dynamic_pointer_cast<DoNothingDecorator const>(other);
28 if(!casted) {
29 return false;
30 }
31 if(casted.get() == this) {
32 return false;
33 }
34 return casted->_target->mayReplaceOther(this->_target);
35 }
36};

49 template<typename UserType>
50 boost::shared_ptr<NDRegisterAccessor<UserType>> getRegisterAccessor_impl(
51 const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
52 auto registerInfo = getRegisterInfo(registerPathName);
53
54 if((wordOffsetInRegister == 0) && ((numberOfWords == registerInfo.getNumberOfElements()) || numberOfWords == 0)) {
55 // Put an empty decorator around the full accessor. This introduces an additional copy from the decorator's user
56 // buffer to the target in pre-write (which otherwise is empty and it cannot be tested that it is called
57 // correctly).
58 auto fullAccessor = DummyBackend::getRegisterAccessor_impl<UserType>(
59 registerPathName, registerInfo.getNumberOfElements(), 0 /* offset */, flags);
60 return boost::make_shared<DoNothingDecorator<UserType>>(fullAccessor);
61 }
62
63 return boost::make_shared<ChimeraTK::detail::SubArrayAccessorDecorator<UserType, UserType>>(
64 shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags);
65 }

146 fullAccessor.write();

158 BOOST_TEST(fullAccessor[1] == 11);
159 BOOST_TEST(fullAccessor[2] == 32); // changed
160 BOOST_TEST(fullAccessor[3] == 33); // changed
161 BOOST_TEST(fullAccessor[4] == 34); // changed
162 BOOST_TEST(fullAccessor[5] == 15);
163 BOOST_TEST(fullAccessor[6] == 16);
164 BOOST_TEST(fullAccessor[7] == 17);
165 BOOST_TEST(fullAccessor[8] == 18);
166 BOOST_TEST(fullAccessor[9] == 19); // changed

172

174 device.open();
