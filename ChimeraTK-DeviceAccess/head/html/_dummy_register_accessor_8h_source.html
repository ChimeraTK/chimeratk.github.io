222 template<typename T>
223 class DummyMultiplexedRegisterAccessor {
224 public:
231 DummyMultiplexedRegisterAccessor(DummyBackend* dev, std::string const& module, std::string const& name)
232 : _dev(dev), _path(module + "/" + name) {
233 registerInfo = _dev->_registerMap.getBackendRegister(module + "." + name);
234
235 // create fixed point converters for each channel
236 for(auto& c : registerInfo.channels) {
237 // create fixed point converter for sequence
238 fpc.emplace_back(registerInfo.pathName, c.width, c.nFractionalBits, c.signedFlag);
239 // store offsets and number of bytes per word
240 assert(c.bitOffset % 8 == 0);
241 offsets.push_back(registerInfo.address + c.bitOffset / 8);
242 nbytes.push_back((c.width - 1) / 8 + 1); // width/8 rounded up
243 }
244
245 if(fpc.empty()) {
246 throw ChimeraTK::logic_error("No sequences found for name \"" + name + "\".");
247 }
248
249 // cache some information
250 nElements = registerInfo.nElements;
251 assert(registerInfo.elementPitchBits % 8 == 0);
252 pitch = registerInfo.elementPitchBits / 8;
253 }
254
256 void operator=(const DummyMultiplexedRegisterAccessor& rightHandSide) const = delete;
257
258 // declare that we want the default copy constructor. Needed because we have a custom = operator
259 DummyMultiplexedRegisterAccessor(const DummyMultiplexedRegisterAccessor&) = default;
260
262 unsigned int getNumberOfElements() { return nElements; }
263
265 unsigned int getNumberOfSequences() { return fpc.size(); }
266
272 inline proxies::DummyRegisterSequence<T> operator[](unsigned int sequence) {
273 // todo: Probably ranges are the correct tool in cpp22. In cpp17 this is not available yet. We turn off the
274 // warning not to use reinterpret_cast for the time being
275 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
276 auto* basePtr = reinterpret_cast<std::byte*>(_dev->_barContents[registerInfo.bar].data());
277 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
278 auto* seq = reinterpret_cast<int32_t*>(basePtr + offsets[sequence]);
279 return proxies::DummyRegisterSequence<T>(&(fpc[sequence]), nbytes[sequence], pitch, seq);
280 }
281
283 [[nodiscard]] DummyBackend& getBackend() const { return *_dev; }
284
286 [[nodiscard]] const RegisterPath& getRegisterPath() const { return _path; }
287
288 [[nodiscard]] const NumericAddressedRegisterInfo& getRegisterInfo() { return registerInfo; }
289
290 protected:
292 DummyBackend* _dev;
293
295 NumericAddressedRegisterInfo registerInfo;
296
298 RegisterPath _path;
299
301 std::vector<FixedPointConverter<DEPRECATED_FIXEDPOINT_DEFAULT>> fpc;
302
304 std::vector<uint32_t> offsets;
305
307 std::vector<uint32_t> nbytes;
308
310 int pitch = {0};

313 unsigned int nElements;

336 assert(_backend);
