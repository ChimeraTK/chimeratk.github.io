22 NumericAddressedBackend::NumericAddressedBackend(const std::string& mapFileName,
23 std::unique_ptr<NumericAddressedRegisterCatalogue> registerMapPointer,
24 const std::string& dataConsistencyKeyDescriptor)
25 : _registerMapPointer(std::move(registerMapPointer)), _registerMap(*_registerMapPointer) {
26 FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(getRegisterAccessor_impl);
27 if(!mapFileName.empty()) {
28 MapFileParser parser;
29 std::tie(_registerMap, _metadataCatalogue) = parser.parse(mapFileName);
30 }
31 if(!dataConsistencyKeyDescriptor.empty()) {
32 // parse as JSON
33 try {
34 auto jdescr = nlohmann::json::parse(dataConsistencyKeyDescriptor);
35 for(const auto& el : jdescr.items()) {
36 _registerMap.addDataConsistencyRealm(el.key(), el.value());
37 }
38 }
39 catch(json::parse_error& e) {
40 throw ChimeraTK::logic_error(std::format("Parsing DataConsistencyKeys parameter '{}' results in JSON error: {}",
41 dataConsistencyKeyDescriptor, e.what()));
42 }
43 }
44 }

48 NumericAddressedRegisterInfo NumericAddressedBackend::getRegisterInfo(const RegisterPath& registerPathName) {
49 if(!registerPathName.startsWith(numeric_address::BAR())) {
50 return _registerMap.getBackendRegister(registerPathName);
51 }
53
54 auto components = registerPathName.getComponents();
55 if(components.size() != 3) {
56 throw ChimeraTK::logic_error("Illegal numeric address: '" + (registerPathName) + "'");
57 }
58 auto bar = std::stoi(components[1]);
59 size_t pos = components[2].find_first_of('*');
60 auto address = std::stoi(components[2].substr(0, pos));
61 size_t nBytes;
62 if(pos != std::string::npos) {
63 nBytes = std::stoi(components[2].substr(pos + 1));
64 }
65 else {
66 nBytes = sizeof(int32_t);
67 }
68 auto nElements = nBytes / sizeof(int32_t);
69 if(nBytes == 0 || nBytes % sizeof(int32_t) != 0) {
70 throw ChimeraTK::logic_error("Illegal numeric address: '" + (registerPathName) + "'");
71 }
72 return NumericAddressedRegisterInfo(registerPathName, nElements, address, nBytes, bar);
73 }

118 template<typename UserType>
119 boost::shared_ptr<NDRegisterAccessor<UserType>> NumericAddressedBackend::getRegisterAccessor_impl(
120 const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
121 if(flags.has(AccessMode::wait_for_new_data)) {
122 // get the interrupt information from the map file
123 auto registerInfo = _registerMap.getBackendRegister(registerPathName);
124 if(!registerInfo.getSupportedAccessModes().has(AccessMode::wait_for_new_data)) {
125 throw ChimeraTK::logic_error(
126 "Register " + registerPathName + " does not support AccessMode::wait_for_new_data.");
127 }
128
129 return _asyncDomainsContainer.subscribe<NumericAddressedBackend, std::nullptr_t, UserType>(
130 boost::static_pointer_cast<NumericAddressedBackend>(shared_from_this()),
131 registerInfo.getQualifiedAsyncId().front(), _asyncIsActive, registerPathName, numberOfWords,
132 wordOffsetInRegister, flags);
133 }
134 return getSyncRegisterAccessor<UserType>(registerPathName, numberOfWords, wordOffsetInRegister, flags);
135 }

139 template<typename UserType>
140 boost::shared_ptr<NDRegisterAccessor<UserType>> NumericAddressedBackend::getSyncRegisterAccessor(
141 const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
142 boost::shared_ptr<NDRegisterAccessor<UserType>> accessor;
143 // obtain register info
144 auto registerInfo = getRegisterInfo(registerPathName);
145
146 // 1D or scalar register
147 if(registerInfo.getNumberOfDimensions() <= 1) {
148 if((registerInfo.channels.front().dataType == NumericAddressedRegisterInfo::Type::FIXED_POINT) ||
149 (registerInfo.channels.front().dataType == NumericAddressedRegisterInfo::Type::VOID)) {
150 if(flags.has(AccessMode::raw)) {
151 accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
152 new NumericAddressedBackendRegisterAccessor<UserType, FixedPointConverter, true>(
153 shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
154 }
155 else {
156 accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
157 new NumericAddressedBackendRegisterAccessor<UserType, FixedPointConverter, false>(
158 shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
159 }
160 }
161 else if(registerInfo.channels.front().dataType == NumericAddressedRegisterInfo::Type::IEEE754) {
162 if(flags.has(AccessMode::raw)) {
163 accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
164 new NumericAddressedBackendRegisterAccessor<UserType, IEEE754_SingleConverter, true>(
165 shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
166 }
167 else {
168 accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
169 new NumericAddressedBackendRegisterAccessor<UserType, IEEE754_SingleConverter, false>(
170 shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
171 }
172 }
173 else if(registerInfo.channels.front().dataType == NumericAddressedRegisterInfo::Type::ASCII) {
174 if constexpr(!std::is_same<UserType, std::string>::value) {
175 throw ChimeraTK::logic_error("NumericAddressedBackend: ASCII data must be read with std::string UserType.");
176 }
177 else {
178 accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(new NumericAddressedBackendASCIIAccessor(
179 shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
180 }
181 }
182 else {
183 throw ChimeraTK::logic_error("NumericAddressedBackend: trying to get accessor for unsupported data type");
184 }
185 }
186 // 2D multiplexed register
187 else {
188 flags.checkForUnknownFlags({});
189 if(registerInfo.channels.front().dataType == NumericAddressedRegisterInfo::Type::IEEE754) {
190 accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
191 new NumericAddressedBackendMuxedRegisterAccessor<UserType, IEEE754_SingleConverter>(
192 registerPathName, numberOfWords, wordOffsetInRegister, shared_from_this()));
193 }
194 else {
195 accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
196 new NumericAddressedBackendMuxedRegisterAccessor<UserType, FixedPointConverter>(
197 registerPathName, numberOfWords, wordOffsetInRegister, shared_from_this()));
198 }
199 }
200
201 accessor->setExceptionBackend(shared_from_this());
202 return accessor;
203 }

207 void NumericAddressedBackend::activateAsyncRead() noexcept {
208 _asyncIsActive = true;
209 // Iterating all async domains must happen under the container lock. We prepare a lambda that is executed via
210 // DomainsContainer::forEach().
211 auto activateDomain = [this](size_t key, boost::shared_ptr<async::Domain>& domain) {
212 auto domainImpl = boost::dynamic_pointer_cast<async::DomainImpl<std::nullptr_t>>(domain);
213 assert(domainImpl);
214 auto subscriptionDone = this->activateSubscription(key, domainImpl);
215 // Wait until the backends reports that the subscription is complete (typically set from inside another thread)
216 // before polling the initial values when activating the async domain. This is necessary to make sure we don't
217 // miss an update that came in after polling the initial value.
218 subscriptionDone.wait();
219 domainImpl->activate(nullptr);
220 };
221
222 _asyncDomainsContainer.forEach(activateDomain);
223 }
