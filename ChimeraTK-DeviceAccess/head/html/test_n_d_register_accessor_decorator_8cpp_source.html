34 void doPreRead(TransferType type) override {
35 _transferType = type;
36 if(this->_accessModeFlags.has(AccessMode::wait_for_new_data) && type == TransferType::readNonBlocking) {
37 // if the access mode has wait_for_new_data then readNonBlocking can be called multiple times by readLatest,
38 // without the test re-setting the counter. In this case just check that the number of calls is smaller or euqal
39 // the queue size i.e. called maximum size + 1 times.
40 BOOST_CHECK(_preRead_counter <= (this->_readQueue.size()));
41 }
42 else {
43 // in all other pre-read must be called exactly once
44 BOOST_CHECK_EQUAL(_preRead_counter, 0);
45 }
46 // in all cases, doPreRead and doPostRead must be called in pairs
47 BOOST_CHECK_EQUAL(_preRead_counter, _postRead_counter);
48 BOOST_CHECK(_preWrite_counter == 0);
49 BOOST_CHECK_EQUAL(_readTransfer_counter, 0);
50 BOOST_CHECK(_writeTransfer_counter == 0);
51 BOOST_CHECK(_postWrite_counter == 0);
52 ++_preRead_counter;
53 if(_throwLogicErr) throw ChimeraTK::logic_error("Test");
54 if(_throwRuntimeErrInPre) throw ChimeraTK::runtime_error("Test");
55 if(_throwThreadInterruptedInPre) throw boost::thread_interrupted();
56 }

137 void doPostRead(TransferType type, bool updateDataBuffer) override {
138 // doPreRead and doPortRead must always be called in pairs.
139 // This can happen multiple times in readLatest. The absolute counting is done in doPreRead()
140 BOOST_CHECK_EQUAL(_preRead_counter, _postRead_counter + 1);
141 BOOST_CHECK(_preWrite_counter == 0);
142 if(!_throwLogicErr && !_throwRuntimeErrInPre && !_throwThreadInterruptedInPre) {
143 if(!this->_accessModeFlags.has(AccessMode::wait_for_new_data)) {
144 BOOST_CHECK_EQUAL(_readTransfer_counter, 1);
145 }
146 else {
147 BOOST_CHECK_EQUAL(_readTransfer_counter, 0);
148 }
149 }
150 else {
151 /* Here B.6.1 is tested for read operations */
152 BOOST_CHECK_EQUAL(_readTransfer_counter, 0);
153 }
154 if(_throwLogicErr || _throwRuntimeErrInPre || _throwThreadInterruptedInPre || _throwRuntimeErrInTransfer ||
155 _throwThreadInterruptedInTransfer) {
156 BOOST_CHECK(this->_activeException != nullptr);
157 }
158 /* Check B.7.4 */
159 if(this->_activeException) {
160 BOOST_CHECK(updateDataBuffer == false);
161 }
162 BOOST_CHECK(_writeTransfer_counter == 0);
163 BOOST_CHECK(_postWrite_counter == 0);
164 BOOST_CHECK(_transferType == type);
165 ++_postRead_counter;
166 _hasNewData = updateDataBuffer;
167 if(_throwThreadInterruptedInPost) throw boost::thread_interrupted();
168 }

178 void doPostWrite(TransferType type, VersionNumber versionNumber) override {
179 BOOST_CHECK(_preRead_counter == 0);
180 BOOST_CHECK(_preWrite_counter == 1);
181 BOOST_CHECK_EQUAL(_readTransfer_counter, 0);
182 if(!_throwLogicErr && !_throwRuntimeErrInPre && !_throwThreadInterruptedInPre) {
183 BOOST_CHECK(_writeTransfer_counter == 1);
184 }
185 else {
186 /* Here B.6.1 is tested for write operations */
187 BOOST_CHECK(_writeTransfer_counter == 0);
188 }
189 // Exceptions must be passed on to the level which is throwing it (B.6.3 This actually tests the
190 // NDRegisterAccessorDecorator)
191 if(_throwLogicErr || _throwRuntimeErrInPre || _throwThreadInterruptedInPre || _throwRuntimeErrInTransfer ||
192 _throwThreadInterruptedInTransfer) {
193 BOOST_CHECK(this->_activeException != nullptr);
194 }
195 BOOST_CHECK(versionNumber == _newVersion);
196 BOOST_CHECK(_postRead_counter == 0);
197 BOOST_CHECK(_postWrite_counter == 0);
198 BOOST_CHECK(_transferType == type);
199 ++_postWrite_counter;
200 if(_throwThreadInterruptedInPost) throw boost::thread_interrupted();
201 }

372BOOST_AUTO_TEST_CASE(TestDecorateDeepInside) {
373 std::cout << "TestDecorateDeepInside" << std::endl;
374 auto target = boost::make_shared<DecoratorTestAccessor<int32_t>>(AccessModeFlags({}));
375 auto deco1 = boost::make_shared<TestNDRegisterAccessorDecorator<int32_t>>(target);
376 auto deco2 = boost::make_shared<TestNDRegisterAccessorDecorator<int32_t>>(deco1);
377 auto deco3 = boost::make_shared<TestNDRegisterAccessorDecorator<int32_t>>(deco2);
378
379 // first rest with a factory which never actually decorates, to see if the factory is called in the right sequence
380 std::vector<boost::shared_ptr<NDRegisterAccessor<int32_t>>> paramsSeen;
381 auto deco0 = deco3->decorateDeepInside(
382 [&](const boost::shared_ptr<NDRegisterAccessor<int32_t>>& t) -> boost::shared_ptr<NDRegisterAccessor<int32_t>> {
383 paramsSeen.push_back(t);
384 return {};
385 });
386 BOOST_TEST(paramsSeen.size() == 3);
387 BOOST_TEST(paramsSeen[0] == target);
388 BOOST_TEST(paramsSeen[1] == deco1);
389 BOOST_TEST(paramsSeen[2] == deco2);
390 BOOST_TEST(!deco0);
391
392 // test the actual decoration
393 boost::shared_ptr<TestNDRegisterAccessorDecorator<int32_t>> decoCreated;
394 auto decoReturned = deco3->decorateDeepInside(
395 [&](const boost::shared_ptr<NDRegisterAccessor<int32_t>>& t) -> boost::shared_ptr<NDRegisterAccessor<int32_t>> {
396 decoCreated = boost::make_shared<TestNDRegisterAccessorDecorator<int32_t>>(t);
397 return decoCreated;
398 });
399 BOOST_TEST(decoReturned == decoCreated);
400 BOOST_TEST(decoCreated->_target == target);
401 BOOST_TEST(deco1->_target == decoCreated);
402}
