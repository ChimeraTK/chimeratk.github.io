556 struct DecoratorMapKey {
557 boost::shared_ptr<ChimeraTK::TransferElement> element;
558 const std::type_info& dataType;
559 ChimeraTK::DecoratorType conversionType;
560 bool operator<(const DecoratorMapKey& other) const {
561 if(element < other.element) return true;
562 if(element != other.element) return false;
563 if(&dataType < &other.dataType) return true;
564 if(&dataType != &other.dataType) return false;
565 if(conversionType < other.conversionType) return true;
566 return false;
567 }
568 bool operator==(const DecoratorMapKey& other) const {
569 return element == other.element && dataType == other.dataType && conversionType == other.conversionType;
570 }
571 };

580 template<class UserType>
581 class DecoratorFactory {
582 public:
583 DecoratorFactory(boost::shared_ptr<ChimeraTK::TransferElement> theImpl_, DecoratorType wantedDecoratorType_)
584 : theImpl(theImpl_), wantedDecoratorType(wantedDecoratorType_) {}
585 boost::shared_ptr<ChimeraTK::TransferElement> theImpl;
586 DecoratorType wantedDecoratorType;
587 mutable boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> createdDecorator;
588
589 DecoratorFactory(const DecoratorFactory&) = delete;
590
591 template<typename PAIR>
592 void operator()(PAIR&) const {
593 typedef typename PAIR::first_type TargetImplType;
594 if(typeid(TargetImplType) != theImpl->getValueType()) return;
595
596 if(wantedDecoratorType == DecoratorType::limiting) {
597 createdDecorator.reset(new TypeChangingRangeCheckingDecorator<UserType, TargetImplType>(
598 boost::dynamic_pointer_cast<ChimeraTK::NDRegisterAccessor<TargetImplType>>(theImpl)));
599 }
600 else if(wantedDecoratorType == DecoratorType::C_style_conversion) {
601 createdDecorator.reset(new TypeChangingDirectCastDecorator<UserType, TargetImplType>(
602 boost::dynamic_pointer_cast<ChimeraTK::NDRegisterAccessor<TargetImplType>>(theImpl)));
603 }
604 else {
605 throw ChimeraTK::logic_error("TypeChangingDecorator with range "
606 "limitation is not implemented yet.");
607 }
608 }
609 };

611 // the factory function. You don't have to care about the IMPL_Type when
612 // requesting a decorator. Implementation, declared at the top of this file
613 template<class UserType>
614 boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> getTypeChangingDecorator(
615 const boost::shared_ptr<ChimeraTK::TransferElement>& transferElement, DecoratorType decoratorType) {
616 // check if there already is a decorator for the transfer element with the right type
617 DecoratorMapKey key{transferElement, typeid(UserType), decoratorType};
618 auto decoratorMapEntry = getGlobalDecoratorMap().find(key);
619 if(decoratorMapEntry != getGlobalDecoratorMap().end()) {
620 // There already is a decorator for this transfer element
621 // The decorator has to have a matching type, otherwise we can only throw
622 auto castedType = boost::dynamic_pointer_cast<ChimeraTK::NDRegisterAccessor<UserType>>(decoratorMapEntry->second);
623 assert(castedType);
624 assert(boost::dynamic_pointer_cast<DecoratorTypeHolder>(decoratorMapEntry->second));
625 return castedType;
626 }
627
628 DecoratorFactory<UserType> factory(transferElement, decoratorType);
629 boost::fusion::for_each(ChimeraTK::userTypeMap(), std::ref(factory));
630 if(factory.createdDecorator == nullptr) {
631 throw ChimeraTK::logic_error("ChimeraTK::ControlSystemAdapter: Decorator for TransferElement " +
632 transferElement->getName() + " has been requested for an unknown user type: " + typeid(UserType).name());
633 }
634 getGlobalDecoratorMap()[key] = factory.createdDecorator;
635 return factory.createdDecorator;
636 }
