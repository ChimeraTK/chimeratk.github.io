336 template<typename BackendRegisterInfo>
337 std::unique_ptr<BackendRegisterCatalogueBase> BackendRegisterCatalogue<BackendRegisterInfo>::clone() const {
338 // FIXME: Change BackendRegisterCatalogue to CRTP, i.e. it has the DERRIVED class as template parameter.
339 // Like this the correct catalogue type is already created here and inherriting backends can call this clone,
340 // then cast to the actual type and fill the rest of it's properties. Would get rid of the "fillFromThis()"
341 // workaround and safe clone() implementations in case no data members are added.
342
343 // Create a new instance of a BackendRegisterCatalogue with the correct type.
344 // Inheriting backends will create the derrived type here.
345 std::unique_ptr<BackendRegisterCatalogueBase> c = std::make_unique<BackendRegisterCatalogue<BackendRegisterInfo>>();
346 // Fill the contents of the BackendRegisterCatalogue base class into the target c. This is accessing the
347 // private variables and ensures consistency.
348 fillFromThis(dynamic_cast<BackendRegisterCatalogue<BackendRegisterInfo>*>(c.get()));
349 // Derrived backends will copy/clone their additional data members here, before
350 // returning the unique_ptr. The compiler will return it via copy elision.
351 return c;
352 }

356 template<typename BackendRegisterInfo>
357 void BackendRegisterCatalogue<BackendRegisterInfo>::fillFromThis(
358 BackendRegisterCatalogue<BackendRegisterInfo>* target) const {
359 // FIXME: change this to a single loop which is just filling the new catalogue through the public API
360 for(auto& p : catalogue) {
361 target->catalogue[p.first] = getBackendRegister(p.first);
362 }
363 for(auto& ptr : insertionOrderedCatalogue) {
364 target->insertionOrderedCatalogue.push_back(&target->catalogue[ptr->getRegisterName()]);
365 }

370 template<typename BackendRegisterInfo>
371 void BackendRegisterCatalogue<BackendRegisterInfo>::addRegister(const BackendRegisterInfo& registerInfo) {
372 if(hasRegister(registerInfo.getRegisterName())) {
373 throw ChimeraTK::logic_error("BackendRegisterCatalogue::addRegister(): Register with the name " +
374 registerInfo.getRegisterName() + " already exists!");
375 }
376 catalogue[registerInfo.getRegisterName()] = registerInfo;
377 insertionOrderedCatalogue.push_back(&catalogue[registerInfo.getRegisterName()]);
378 }

382 template<typename BackendRegisterInfo>
383 void BackendRegisterCatalogue<BackendRegisterInfo>::removeRegister(const RegisterPath& name) {
384 // check existence
385 if(!hasRegister(name)) {
386 throw ChimeraTK::logic_error(
387 "BackendRegisterCatalogue::removeRegister(): Register '" + name + "' does not exist.");
388 }
389
390 // remove from insertion-ordered vector
391 auto it = std::find_if(insertionOrderedCatalogue.begin(), insertionOrderedCatalogue.end(),
392 [&](auto reg) { return reg->getRegisterName() == name; });
393 assert(it != insertionOrderedCatalogue.end());
394 insertionOrderedCatalogue.erase(it);
395
396 // remove from catalogue map
397 auto removed = catalogue.erase(name);
398 assert(removed == 1);
399 }

403 template<typename BackendRegisterInfo>
404 void BackendRegisterCatalogue<BackendRegisterInfo>::modifyRegister(const BackendRegisterInfo& registerInfo) {
405 if(!hasRegister(registerInfo.getRegisterName())) {
406 throw ChimeraTK::logic_error("BackendRegisterCatalogue::modifyRegister(): Register '" +
407 registerInfo.getRegisterName() + "' cannot be modified because it does not exist!");
408 }
409 catalogue[registerInfo.getRegisterName()] = registerInfo;
410 // We don't have to touch the insertionOrderedCatalogue because is stores references, and this has not changed.
411 }
