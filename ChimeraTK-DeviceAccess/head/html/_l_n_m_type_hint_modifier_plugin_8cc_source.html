11 TypeHintModifierPlugin::TypeHintModifierPlugin(
12 const LNMBackendRegisterInfo& info, size_t pluginIndex, const std::map<std::string, std::string>& parameters)
13 : AccessorPlugin<TypeHintModifierPlugin>(info, pluginIndex) {
14 try {
15 std::string typeName = parameters.at("type");
16 if(typeName == "integer") typeName = "int32";
17
18 _dataType = DataType(typeName);
19 if(_dataType == DataType::none) {
20 throw ChimeraTK::logic_error(
21 "LogicalNameMappingBackend TypeHintModifierPlugin: Unknown type '" + typeName + "'.");
22 }
23 }
24 catch(std::out_of_range&) {
25 throw ChimeraTK::logic_error("LogicalNameMappingBackend TypeHintModifierPlugin: Missing parameter 'type'.");
26 }
27 }

29 void TypeHintModifierPlugin::doRegisterInfoUpdate() {
30 // keep raw data type and transport layer data type from original entry
31 auto d = _info._dataDescriptor;
32
33 auto newDescriptor = DataDescriptor(_dataType);
34
35 if(newDescriptor.fundamentalType() == DataDescriptor::FundamentalType::numeric) {
36 _info._dataDescriptor = DataDescriptor(newDescriptor.fundamentalType(), newDescriptor.isIntegral(),
37 newDescriptor.isSigned(), newDescriptor.nDigits(),
38 _dataType.isIntegral() ? 0 : newDescriptor.nFractionalDigits(), d.rawDataType(), d.transportLayerDataType());
39 }
40 else {
41 // can't call isIntegral() etc. if not numeric (just provide dummy info for those fields as they are ignored)
42 _info._dataDescriptor = DataDescriptor(
43 newDescriptor.fundamentalType(), false, false, 0, 0, d.rawDataType(), d.transportLayerDataType());
44 }
45
46 d = _info._dataDescriptor;
47 }
