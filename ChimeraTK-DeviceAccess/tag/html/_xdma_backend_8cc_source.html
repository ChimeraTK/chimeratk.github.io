18  void XdmaBackend::open() {
19 #ifdef _DEBUG
20  std::cout << "XDMA: opening dev: " << _devicePath << std::endl;
21 #endif
22  if(_ctrlIntf) {
23  if(isFunctional()) {
24  return;
25  }
26  close();
27  }
28 
29  _ctrlIntf.emplace(_devicePath);
30 
31  // Build vector of DMA channels
32  _dmaChannels.clear();
33  for(size_t i = 0; i < _maxDmaChannels; i++) {
34  try {
35  _dmaChannels.emplace_back(_devicePath, i);
36  }
37  catch(const runtime_error&) {
38  break;
39  }
40  }
41 
42 #ifdef _DEBUG
43  std::cout << "XDMA: opened interface with " << _dmaChannels.size() << " DMA channels and " << _eventFiles.size()
44  << " interrupt sources\n";
45 #endif
46  setOpenedAndClearException();
47  }

59  XdmaIntfAbstract& XdmaBackend::_intfFromBar(uint64_t bar) {
60  if(bar == 0 && _ctrlIntf.has_value()) {
61  return _ctrlIntf.value();
62  }
63  // 13 is magic value for DMA channel (by convention)
64  // We provide N DMA channels starting from there
65  if(bar >= 13) {
66  const size_t dmaChIdx = bar - 13;
67  if(dmaChIdx < _dmaChannels.size()) {
68  return _dmaChannels[dmaChIdx];
69  }
70  }
71  throw ChimeraTK::logic_error("Couldn't find XDMA channel for BAR value " + std::to_string(bar));
72  }

115  std::future<void> XdmaBackend::activateSubscription(
116  uint32_t interruptNumber, boost::shared_ptr<async::DomainImpl<std::nullptr_t>> asyncDomain) {
117  std::promise<void> subscriptionDonePromise;
118  auto subscriptionDoneFuture = subscriptionDonePromise.get_future();
119  if(interruptNumber >= _maxInterrupts) {
120  setException("XDMA interrupt " + std::to_string(interruptNumber) + " out of range, only 0.." +
121  std::to_string(_maxInterrupts - 1) + " available\n");
122  subscriptionDonePromise.set_value();
123  return subscriptionDoneFuture;
124  }
125 
126  if(!_eventFiles[interruptNumber]) {
127  _eventFiles[interruptNumber] = std::make_unique<EventFile>(_devicePath, interruptNumber, asyncDomain);
128  _eventFiles[interruptNumber]->startThread(std::move(subscriptionDonePromise));
129  }
130  else {
131  // thread is already running, just fulfil the promise
132  subscriptionDonePromise.set_value();
133  }
134  return subscriptionDoneFuture;
135  }
