12 TagModifierPlugin::TagModifierPlugin(
13 const LNMBackendRegisterInfo& info, size_t pluginIndex, const std::map<std::string, std::string>& parameters)
14 : AccessorPlugin<TagModifierPlugin>(info, pluginIndex) {
15 const static std::regex re(R"(\s*,\s*)");
16
17 if(parameters.contains("add")) {
18 auto str = parameters.at("add");
19 _tagsToAdd.insert(std::sregex_token_iterator(str.begin(), str.end(), re, -1), std::sregex_token_iterator());
20 }
21
22 if(parameters.contains("remove")) {
23 auto str = parameters.at("remove");
24 _tagsToRemove.insert(std::sregex_token_iterator(str.begin(), str.end(), re, -1), std::sregex_token_iterator());
25 }
26
27 if(parameters.contains("set")) {
28 auto str = parameters.at("set");
29 _tagsToSet.insert(std::sregex_token_iterator(str.begin(), str.end(), re, -1), std::sregex_token_iterator());
30 }
31
32 if(_tagsToAdd.empty() && _tagsToRemove.empty() && _tagsToSet.empty()) {
33 throw ChimeraTK::logic_error(R"(TagModifierPlugin needs at least one of "add","remove" or "set")");
34 }
35
36 if(!_tagsToSet.empty() && !(_tagsToRemove.empty() || _tagsToAdd.empty())) {
37 throw ChimeraTK::logic_error(R"(TagModifierPlugin: "set" is mutual exclusive with "add" and "remove")");
38 }
39 }

43 void TagModifierPlugin::doRegisterInfoUpdate() {
44 if(!_tagsToSet.empty()) {
45 _info.tags = _tagsToSet;
46 }
47 else {
48 _info.tags.insert(_tagsToAdd.begin(), _tagsToAdd.end());
49 std::ranges::set_difference(_info.tags, _tagsToRemove, std::inserter(_tagsToSet, _tagsToSet.end()));
50 _info.tags = _tagsToSet;
51 }
52 }
