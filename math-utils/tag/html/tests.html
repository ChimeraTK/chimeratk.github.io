BOOST_AUTO_TEST_CASE(testStandardDeviationOfCircularQuantitiesAssignAndAdd) {
std::vector<double> vec = {355, 15};
MathUtils::CircularStatistics<double> circStats;
circStats.assign(vec);
circStats.add(5);
BOOST_CHECK_CLOSE(circStats.mean(), 5, 0.0000000000001);
}
BOOST_AUTO_TEST_CASE(testStandardDeviationOfCircularQuantitiesValuesDouble) {
std::vector<double> vec_phase = {0, 0, 90};
MathUtils::CircularStatistics<double> sDcq;
sDcq.assign(vec_phase);
BOOST_CHECK_EQUAL(sDcq.mean(), 26.56505117707799);
BOOST_CHECK_EQUAL(sDcq.meanDirection(), 26.56505117707799);
BOOST_CHECK_EQUAL(sDcq.circularVariance(), 0.2546440075000701);
BOOST_CHECK_EQUAL(sDcq.circularStandardDeviation(), 43.927096375618788);
}
