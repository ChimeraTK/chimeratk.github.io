14 // Have to make it a macro so that it reports exact line numbers when checks fail.
15 #define CHECK_CLOSE_COLLECTION(aa, bb, tolerance) \
16  do { \
17  using std::distance; \
18  auto a = (aa).begin(), ae = (aa).end(); \
19  auto b = (bb).begin(); \
20  BOOST_REQUIRE_EQUAL(std::distance(a, ae), std::distance(b, (bb).end())); \
21  for(; a != ae; ++a, ++b) { \
22  BOOST_CHECK_CLOSE(*a, *b, tolerance); \
23  } \
24  } while(false)
