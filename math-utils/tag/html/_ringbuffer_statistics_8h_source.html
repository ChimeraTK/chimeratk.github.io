43 void resize(size_t newSize, bool keepElements = false) {
44 if(newSize == size) return;
45
46 // If we do not want to keep the old data, drop it, reset the counters and we are done
47 if(!keepElements) {
48 size = newSize;
49 values.clear();
50 values.reserve(size);
51 position = 0;
52
53 return;
54 }
55
56 // If we are not yet in overwriting state it is sufficient to just modify the size of the buffer, move the write
57 // pointer to the end of that block
58 if(values.size() < size) {
59 size = newSize;
60 values.reserve(size);
61
62 return;
63 }
64
65 // we are already in the overwriting phase, we have to shuffle the buffer around. Values right of the current
66 // write position go to the front, values up to the write position go after that.
67 // If the buffer was shrunk, drop as many values from the beginning as necessary to reach the new size.
68
69 decltype(values) v{values.begin() + position, values.end()};
70 std::copy(values.begin(), values.begin() + position, std::back_inserter(v));
71
72 if(newSize < size) {
73 v.erase(v.begin(), v.begin() + (v.size() - newSize));
74
75 // At this point, the buffer is filled, we will start overwriting the oldest == first element in the ring
76 // Also the cached mean is invalid, if any.
77 flagMeanTmp = true;
78 position = 0;
79 }
80 else {
81 // We have more room now, so we can just append values to the end
82 position = v.size();
83 }
84
85 size = newSize;
86 values = std::move(v);
87 }
88

90
98 void add(T value) {
99 if(values.size() < size) {
100 values.push_back(value);
101 }
102 else {
103 values[position] = value;
104 }
105
106 position++;
107 if(position == size) position = 0;
108 flagMeanTmp = true;
109 }
