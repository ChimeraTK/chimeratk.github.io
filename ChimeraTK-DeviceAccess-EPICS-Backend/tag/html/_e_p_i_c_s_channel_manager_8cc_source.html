42  void ChannelManager::channelStateHandler(connection_handler_args args) {
43  auto backend = reinterpret_cast<EpicsBackend*>(ca_puser(args.chid));
44  if(args.op == CA_OP_CONN_UP) {
45  backend->setBackendState(true);
46  std::lock_guard<std::mutex> lock(ChannelManager::getInstance().mapLock);
47  // channel should have been added to the manager - if not let throw here, because this should not happen
48  auto it = ChannelManager::getInstance().findChid(args.chid);
49  it->second._connected = true;
50  // configure channel
51  if(!it->second._configured) {
52  it->second._pv->nElems = ca_element_count(args.chid);
53  it->second._pv->dbfType = ca_field_type(args.chid);
54  it->second._pv->dbrType = dbf_type_to_DBR_TIME(it->second._pv->dbfType);
55  it->second._pv->value =
56  calloc(it->second._pv->nElems, dbr_size_n(it->second._pv->dbrType, it->second._pv->nElems));
57  it->second._configured = true;
58  }
59  }
60  else if(args.op == CA_OP_CONN_DOWN) {
61  backend->setBackendState(false);
62  if(!backend->isOpen()) {
63 #ifdef CHIMERATK_UNITTEST
64  ChannelManager::getInstance().currentState = args.op;
65 #endif
66  return;
67  }
68  std::lock_guard<std::mutex> lock(ChannelManager::getInstance().mapLock);
69  // channel should have been added to the manager - if not let throw here, because this should not happen
70  auto it = ChannelManager::getInstance().findChid(args.chid);
71  it->second._connected = false;
72  if(it->second._asyncReadActivated) {
73  ChannelManager::getInstance().deactivateChannels();
74  }
75 
76  for(auto& accessor : it->second._accessors) {
77  if(!accessor->_hasNotificationsQueue || !accessor->_backend->_asyncReadActivated) {
78  continue;
79  }
80  try {
81  throw ChimeraTK::runtime_error(std::string("Channel for PV ") + it->second._caName + " was disconnected.");
82  }
83  catch(...) {
84  accessor->_notifications.push_overwrite_exception(std::current_exception());
85  }
86  }
87  }
88 #ifdef CHIMERATK_UNITTEST
89  // set state -> it is used in the test to wait for a connect/reconnect
90  std::lock_guard<std::mutex> lock(ChannelManager::getInstance().mapLock);
91  if(args.op == CA_OP_CONN_UP) {
92  if(ChannelManager::getInstance().checkAllConnections(true)) {
93  // only set connected once all are up
94  ChannelManager::getInstance().currentState = args.op;
95  }
96  }
97  else {
98  if(ChannelManager::getInstance().checkAllConnections(false)) {
99  // only set connected once all are down and exceptions have been pushed
100  ChannelManager::getInstance().currentState = args.op;
101  }
102  }
103 #endif
104  }

106  void ChannelManager::handleEvent(evargs args) {
107  auto base = reinterpret_cast<ChannelManager*>(args.usr);
108  auto backend = reinterpret_cast<EpicsBackend*>(ca_puser(args.chid));
109  std::lock_guard<std::mutex> lock(base->mapLock);
110  auto it = base->findChid(args.chid);
111  if(backend->isOpen() && backend->isFunctional()) {
112  if(it->second._asyncReadActivated) {
113  for(auto& accessor : it->second._accessors) {
114  // channel can have accessors without mode wait_for_new_data -> no notification queue
115  if(accessor->_hasNotificationsQueue) {
116  EpicsRawData data(args);
117  accessor->_notifications.push_overwrite(std::move(data));
118  it->second._initialValueReceived = true;
119  }
120  }
121  }
122  }
123  }

141  void ChannelManager::addChannel(const std::string name, EpicsBackend* backend) {
142  if(!channelPresent(name)) {
143  channelMap.insert(std::make_pair(name, ChannelInfo(name)));
144  }
145  auto pv = getPV(name);
146  auto result =
147  ca_create_channel(name.c_str(), ChannelManager::channelStateHandler, backend, default_ca_priority, &pv->chid);
148  if(result != ECA_NORMAL) {
149  std::stringstream ss;
150  ss << "CA error " << ca_message(result) << " occurred while trying to create channel " << name;
151  throw ChimeraTK::runtime_error(ss.str());

205  void ChannelManager::addAccessor(const std::string& name, EpicsBackendRegisterAccessorBase* accessor) {
206  // map locked by calling function
207  if(!channelPresent(name)) {
208  throw ChimeraTK::runtime_error("Tryed to add an accessor without having a map entry!");
209  }
210  channelMap.find(name)->second._accessors.push_back(accessor);
211  if(channelMap.find(name)->second._accessors.size() > 1 && channelMap.find(name)->second._asyncReadActivated) {
212  if(accessor->_hasNotificationsQueue) {
213  accessor->setInitialValue(channelMap.find(name)->second._pv->value, channelMap.find(name)->second._pv->dbrType,
214  channelMap.find(name)->second._pv->nElems);
215  }
216  }
217  }

219  void ChannelManager::removeAccessor(const std::string& name, EpicsBackendRegisterAccessorBase* accessor) {
220  std::lock_guard<std::mutex> lock(ChannelManager::getInstance().mapLock);
221  // check if channel is in map -> map might be already cleared.
222  if(channelMap.count(name)) {
223  auto entry = &channelMap.at(name);
224  if(entry->_accessors.size() > 0) {
225  bool erased = false;
226  for(auto itaccessor = entry->_accessors.begin(); itaccessor != entry->_accessors.end(); ++itaccessor) {
227  if(accessor == *itaccessor) {
228  entry->_accessors.erase(itaccessor);
229  erased = true;
230  break;
231  }
232  }
233  if(!erased) {
234  std::cout << "Failed to erase accessor for pv:" << name << std::endl;
235  }
236  }
237  if(entry->_accessors.size() == 0) {
238  if(!entry->_asyncReadActivated) return;
239  ca_clear_subscription(*entry->_subscriptionId);
240  entry->_asyncReadActivated = false;
241  ca_flush_io();
242  }
243  }
244  }

251  void ChannelManager::activateChannel(ChannelInfo* channel) {
252  if(channel->_asyncReadActivated) return;
253  // only open subscription if accessors are present -> else the initial value will be lost
254  // The handler will be called directly after creating the subscription
255  // E.g. in case of QtHardmon EpicsBackend::activateAsyncRead is called first and accessors are added later
256  if(channel->_accessors.size() == 0) return;
257  channel->_subscriptionId = new evid();
258  auto ret = ca_create_subscription(channel->_pv->dbrType, channel->_pv->nElems, channel->_pv->chid, DBE_VALUE,
259  &ChannelManager::handleEvent, static_cast<ChannelManager*>(this), channel->_subscriptionId);
260  if(ret != ECA_NORMAL) {
261  throw ChimeraTK::runtime_error(std::string("Failed to create subscription for channel: ") + channel->_pv->name);
262  }
263  ca_flush_io();
264  channel->_asyncReadActivated = true;
265  channel->_initialValueReceived = false;
266  std::cout << "Channel " << channel->_caName << " activated for async read." << std::endl;
267  }

298  void ChannelManager::setException(const std::string error) {
299  std::lock_guard<std::mutex> lock(mapLock);
300  ChannelManager::getInstance().deactivateChannels();
301  for(auto& mapItem : channelMap) {
302  // only push exceptions to channels that are still connected
303  // if an exception is see on the first channel it is push to the notification queue and _connected is set false.
304  if(mapItem.second._connected) {
305  mapItem.second._connected = false;
306  for(auto& accessor : mapItem.second._accessors) {
307  try {
308  throw ChimeraTK::runtime_error(error);
309  }
310  catch(...) {
311  if(accessor->_hasNotificationsQueue) {
312  accessor->_notifications.push_overwrite_exception(std::current_exception());
313  }
314  }
315  }
316  }
317  }
