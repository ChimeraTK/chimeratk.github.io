210  template<typename UserType>
211  std::vector<std::vector<UserType>> getRemoteValue() {
212  auto result = IOCLauncher::helper->getValue(pvName());
213  std::vector<std::string> tmp;
214  size_t pos = 0;
215  std::string token;
216  bool firstElement{true};
217  while((pos = result.find(" ")) != std::string::npos) {
218  token = result.substr(0, pos);
219  if(firstElement) {
220  if(std::stoul(token.c_str()) != nElementsPerChannel()) {
221  throw std::runtime_error("Size mismatch when reading array data.");
222  }
223  firstElement = false;
224  }
225  else {
226  tmp.push_back(token);
227  }
228  result.erase(0, pos + 1);
229  }
230  // add last element
231  tmp.push_back(result);
232  std::vector<UserType> values;
233  for(size_t i = 0; i < nElementsPerChannel(); ++i) {
234  values.push_back(ChimeraTK::userTypeToUserType<UserType, std::string>(tmp.at(i)));
235  }
236  return {values};
237  }
