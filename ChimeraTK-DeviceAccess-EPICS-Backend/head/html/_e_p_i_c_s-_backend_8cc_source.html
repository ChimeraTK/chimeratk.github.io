72 void EpicsBackend::open() {
73 if(!isFunctional()) {
74 // after closing a new ca_context is needed (_opened is set also in constructor to signal prepareChannelAccess was
75 // just called before)
76 if(!_freshCreated) {
77 prepareChannelAccess();
78 {
79 std::lock_guard<std::mutex> lock(ChannelManager::getInstance().mapLock);
80 ChannelManager::getInstance().addChannelsFromMap(this);
81 }
82 }
83 else {
84 _freshCreated = false;
85 }
86 size_t n = default_ca_timeout / 0.1; // sleep 100ms per loop, wait default_ca_timeout until giving up
87 for(size_t i = 0; i < n; i++) {
88 {
89 std::lock_guard<std::mutex> lock(ChannelManager::getInstance().mapLock);
90 if(ChannelManager::getInstance().checkAllConnections(true)) {
91 _channelAccessUp = true;
92 break;
93 }
94 }
95 std::this_thread::sleep_for(std::chrono::milliseconds(100));
96 }
97 if(!_channelAccessUp) {
98 throw ChimeraTK::runtime_error("Failed to establish channel access connection.");
99 }
100 if(_asyncReadActivated) {
101 std::lock_guard<std::mutex> lock(ChannelManager::getInstance().mapLock);
102 ChannelManager::getInstance().activateChannels();
103 }
104 _startVersion = {};
105 setOpenedAndClearException();
106 }
107 }

118 void EpicsBackend::activateAsyncRead() noexcept {
119 if(!isFunctional()) return;
120 // activate async read expects an initial value so deactivate channels first to force initial value
121 {
122 std::lock_guard<std::mutex> lock(ChannelManager::getInstance().mapLock);
123 ChannelManager::getInstance().deactivateChannels();
124 ChannelManager::getInstance().activateChannels();
125 }
126 size_t n = default_ca_timeout / 0.1; // sleep 100ms per loop, wait default_ca_timeout until giving up
127 bool allGood = false;
128 for(size_t i = 0; i < n; i++) {
129 if(ChannelManager::getInstance().checkInitialValueReceived()) {
130 allGood = true;
131 break;
132 }
133 std::this_thread::sleep_for(std::chrono::milliseconds(100));
134 }
135 if(!allGood) {
136 std::cerr << "Failed to receive initial value for all subscriptions in activateAsyncRead()." << std::endl;
137 }
138 _asyncReadActivated = true;
139 }

232 void EpicsBackend::configureChannel(EpicsBackendRegisterInfo& info) {
233 std::lock_guard<std::mutex> lock(ChannelManager::getInstance().mapLock);
234 auto pv = ChannelManager::getInstance().getPV(info._caName);
235 if(!ChannelManager::getInstance().isChannelConfigured(info._caName)) {
236 throw ChimeraTK::runtime_error("Trying to read an unconfigured channel.");
237 }
238 info._nElements = pv->nElems;
239 info._dbfType = pv->dbfType;
240
241 if(ca_read_access(pv->chid) != 1) info._isReadable = false;
242 if(ca_write_access(pv->chid) != 1) info._isWritable = false;
243 if(pv->dbfType == DBF_STRING) {
244 info._dataDescriptor = DataDescriptor(DataDescriptor::FundamentalType::string, true, true, 320, 300);
245 }
246 else if(pv->dbfType == DBF_DOUBLE || pv->dbfType == DBF_FLOAT) {
247 info._dataDescriptor = DataDescriptor(DataDescriptor::FundamentalType::numeric, false, true, 320, 300);
248 }
249 else if(pv->dbfType == DBF_INT || pv->dbfType == DBF_LONG || pv->dbfType == DBF_SHORT) {
250 info._dataDescriptor = DataDescriptor(DataDescriptor::FundamentalType::numeric, true, true, 320, 300);
251 }
252 else if(pv->dbfType == DBF_ENUM) {
253 info._dataDescriptor = DataDescriptor(DataDescriptor::FundamentalType::boolean, true, true, 320, 300);
254 }
255 else {
256 std::cerr << "Failed to data descriptor for node: " << info._caName << "." << std::endl;
257 }
258 info._accessModes.add(AccessMode::wait_for_new_data);
259 }

261 void EpicsBackend::fillCatalogueFromMapFile(const std::string& mapfileName) {
262 boost::char_separator<char> sep{"\t ", "", boost::drop_empty_tokens};
263 std::string line;
264 std::ifstream mapfile(mapfileName);
265 if(mapfile.is_open()) {
266 while(std::getline(mapfile, line)) {
267 if(line.empty()) continue;
268 tokenizer tok{line, sep};
269 size_t nTokens = std::distance(tok.begin(), tok.end());
270 if(!(nTokens == 2)) {
271 std::cerr << "Wrong number of tokens (" << nTokens << ") in mapfile " << mapfileName
272 << " line (-> line is ignored): \n " << line << std::endl;
273 continue;
274 }
275 auto it = tok.begin();
276
277 try {
278 std::shared_ptr<std::string> pathStr = std::make_shared<std::string>(*it);
279 RegisterPath path(*(pathStr.get()));
280 it++;
281 std::shared_ptr<std::string> nodeName = std::make_shared<std::string>(*it);
282 addCatalogueEntry(path, nodeName);
283 }
284 catch(std::out_of_range& e) {
285 std::cerr << "Failed reading the following line from mapping file " << mapfileName << "\n " << line
286 << std::endl;
287 }
288 }
289 mapfile.close();
290 }
291 else {
292 throw ChimeraTK::runtime_error(std::string("Failed reading mapfile: ") + mapfileName);
293 }
294
295 if(_catalogue_mutable.getNumberOfRegisters() == 0) {
296 throw ChimeraTK::runtime_error("No registers found in catalogue!");
297 }
298
299 auto result = ca_flush_io();
300 if(result == ECA_TIMEOUT) {
301 throw ChimeraTK::runtime_error("Channel setup failed.");
302 }
303 size_t n = default_ca_timeout / 0.1; // sleep 100ms per loop, wait default_ca_timeout until giving up
304 for(size_t i = 0; i < n; i++) {
305 {
306 std::lock_guard<std::mutex> lock(ChannelManager::getInstance().mapLock);
307 if(ChannelManager::getInstance().checkAllConnections(true)) {
308 _channelAccessUp = true;
309 break;
310 }
311 }
312 std::this_thread::sleep_for(std::chrono::milliseconds(100));
313 }
314 if(!_channelAccessUp) {
315 throw ChimeraTK::runtime_error("Failed to establish channel access connection.");
316 }
317 for(auto& reg : _catalogue_mutable) {
318 configureChannel(reg);
319 }
320 }
