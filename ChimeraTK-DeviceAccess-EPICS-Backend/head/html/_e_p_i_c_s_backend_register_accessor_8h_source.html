27  template<typename DestType, typename SourceType>
28  class EpicsRangeCheckingDataConverter {
30  template<class S>
31  struct Round {
32  static S nearbyint(S s) { return round(s); }
33 
34  typedef boost::mpl::integral_c<std::float_round_style, std::round_to_nearest> round_style;
35  };
36  typedef boost::numeric::converter<DestType, SourceType, boost::numeric::conversion_traits<DestType, SourceType>,
37  boost::numeric::def_overflow_handler, Round<SourceType>>
38  converter;
39 
40  public:
41  DestType convert(SourceType& x) {
42  try {
43  return converter::convert(x);
44  }
45  catch(boost::numeric::positive_overflow&) {
46  return std::numeric_limits<DestType>::max();
47  }
48  catch(boost::numeric::negative_overflow&) {
49  return std::numeric_limits<DestType>::min();
50  }
51  catch(...) {
52  throw ChimeraTK::runtime_error("Data conversion failed for unknown reason.");
53  }
54  }
55  };

209  template<typename EpicsBaseType, typename EpicsType, typename CTKType>
210  EpicsBackendRegisterAccessor<EpicsBaseType, EpicsType, CTKType>::EpicsBackendRegisterAccessor(
211  const RegisterPath& path, boost::shared_ptr<DeviceBackend> backend, const EpicsBackendRegisterInfo& registerInfo,
212  AccessModeFlags flags, size_t numberOfWords, size_t wordOffsetInRegister, bool asyncReadActivated)
213  : EpicsBackendRegisterAccessorBase(
214  boost::dynamic_pointer_cast<EpicsBackend>(backend), registerInfo, numberOfWords, wordOffsetInRegister),
215  NDRegisterAccessor<CTKType>(path, flags) {
216  if constexpr(std::is_array_v<EpicsBaseType>) {
217  if(numberOfWords > 1) {
218  throw ChimeraTK::logic_error("No support for lso array.");
219  }
220  }
221  if(flags.has(AccessMode::raw)) throw ChimeraTK::logic_error("Raw access mode is not supported.");
222  NDRegisterAccessor<CTKType>::buffer_2D.resize(1);
223  this->accessChannel(0).resize(numberOfWords);
224  std::lock_guard<std::mutex> lock(ChannelManager::getInstance().mapLock);
225  auto pv = ChannelManager::getInstance().getPV(_info._caName);
226  if(flags.has(AccessMode::wait_for_new_data)) {
227  _hasNotificationsQueue = true;
228  _notifications = cppext::future_queue<EpicsRawData>(3);
229  _readQueue = _notifications.then<void>(
230  [pv](EpicsRawData& data) { memcpy(pv->value, data.data, data.size); }, std::launch::deferred);
231  }
232  if(pv->nElems != numberOfWords) _isPartial = true;
233  ChannelManager::getInstance().addAccessor(_info._caName, this);
234  if(flags.has(AccessMode::wait_for_new_data) && asyncReadActivated) {
235  ChannelManager::getInstance().activateChannel(_info._caName);
236  }
237  NDRegisterAccessor<CTKType>::_exceptionBackend = backend;
238  }

247  template<typename EpicsBaseType, typename EpicsType, typename CTKType>
248  void EpicsBackendRegisterAccessor<EpicsBaseType, EpicsType, CTKType>::doReadTransferSynchronously() {
249  _backend->checkActiveException();
250  std::lock_guard<std::mutex> lock(ChannelManager::getInstance().mapLock);
251  auto pv = ChannelManager::getInstance().getPV(_info._caName);
252  // one could also use ChannelManager::isChannelConnected -> however we ask explicitly ChannelAccess here
253  if(ca_state(pv->chid) == cs_conn) {
254  if(pv->nElems == 1) {
255  auto result = ca_get(pv->dbrType, pv->chid, pv->value);
256  if(result != ECA_NORMAL) {
257  throw ChimeraTK::runtime_error(std::string("Failed to read pv: ") + pv->name);
258  }
259  result = ca_pend_io(default_ca_timeout);
260  if(result == ECA_TIMEOUT) {
261  throw ChimeraTK::runtime_error(std::string("Read operation timed out for pv: ") + pv->name);
262  }
263  }
264  else {
265  auto result = ca_array_get(pv->dbrType, pv->nElems, pv->chid, pv->value);
266  if(result != ECA_NORMAL) {
267  throw ChimeraTK::runtime_error(std::string("Failed to read pv: ") + pv->name);
268  }
269  result = ca_pend_io(default_ca_timeout);
270  if(result == ECA_TIMEOUT) {
271  throw ChimeraTK::runtime_error(std::string("Read operation timed out for pv: ") + pv->name);
272  }
273  }
274  }
275  else {
276  throw ChimeraTK::runtime_error(
277  std::string("ChannelAccess not connected in doReadTransferSynchronously when writing: ") + _info._name + "(" +
278  pv->name + ")");
279  }
280  }

282  template<typename EpicsBaseType, typename EpicsType, typename CTKType>
283  void EpicsBackendRegisterAccessor<EpicsBaseType, EpicsType, CTKType>::doPostRead(TransferType, bool hasNewData) {
284  if(!hasNewData) return;
285  std::lock_guard<std::mutex> lock(ChannelManager::getInstance().mapLock);
286  auto pv = ChannelManager::getInstance().getPV(_info._caName);
287  EpicsBaseType* tmp = (EpicsBaseType*)dbr_value_ptr(pv->value, pv->dbrType);
288 
289  if constexpr(std::is_array_v<EpicsBaseType>) {
290  // only single element as checked in the constructor
291  dbr_string_t tmpStr;
292  strcpy(tmpStr, (char*)tmp);
293  this->accessData(0) = toCTK.convert(tmpStr);
294  }
295  else {
296  for(size_t i = 0; i < _numberOfWords; i++) {
297  EpicsBaseType value = tmp[_offsetWords + i];
298  // Fill the NDRegisterAccessor buffer
299  this->accessData(i) = toCTK.convert(value);
300  }
301  }
302 
303  EpicsType* tp = (EpicsType*)pv->value;
304  _currentVersion = EPICS::VersionMapper::getInstance().getVersion(tp[0].stamp);
305  if(_currentVersion < _backend->_startVersion) {
306  _currentVersion = _backend->_startVersion;
307  }
308  TransferElement::_versionNumber = _currentVersion;
309  }

311  template<typename EpicsBaseType, typename EpicsType, typename CTKType>
312  bool EpicsBackendRegisterAccessor<EpicsBaseType, EpicsType, CTKType>::doWriteTransfer(
313  VersionNumber /*versionNumber*/) {
314  _backend->checkActiveException();
315  auto pv = ChannelManager::getInstance().getPV(_info._caName);
316  // one could also use ChannelManager::isChannelConnected -> however we ask explicitly ChannelAccess here
317  if(ca_state(pv->chid) == cs_conn) {
318  if(_isPartial) EpicsBackendRegisterAccessor<EpicsBaseType, EpicsType, CTKType>::doReadTransferSynchronously();
319  EpicsBaseType* tmp = (EpicsBaseType*)dbr_value_ptr(pv->value, pv->dbrType);
320  long result;
321  if constexpr(std::is_array_v<EpicsBaseType>) {
322  // only single element as checked in the constructor
323  result = ca_array_put(pv->dbfType, pv->nElems, pv->chid, toEpics.convert(this->accessData(0)).c_str());
324  }
325  else {
326  for(size_t i = 0; i < _numberOfWords; i++) {
327  tmp[_offsetWords + i] = toEpics.convert(this->accessData(i));
328  }
329  result = ca_array_put(pv->dbfType, pv->nElems, pv->chid, tmp);
330  }
331 
332  if(result != ECA_NORMAL) {
333  std::cerr << "Failed to to write pv: " << _info._caName << std::endl;
334  return false;
335  }
336  result = ca_pend_io(default_ca_timeout);
337  if(result == ECA_TIMEOUT) {
338  std::cerr << "Timeout while writing pv: " << _info._caName << std::endl;
339  return false;
340  }
341  }
342  else {
343  throw ChimeraTK::runtime_error(std::string("ChannelAccess not connected in doWriteTransfer when writing: ") +
344  _info._name + "(" + pv->name + ")");
345  }
346  return true;
347  }
