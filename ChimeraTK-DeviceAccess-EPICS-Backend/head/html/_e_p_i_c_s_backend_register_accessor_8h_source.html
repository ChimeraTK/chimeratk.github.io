28 template<typename DestType, typename SourceType>
29 class EpicsRangeCheckingDataConverter {
31 template<class S>
32 struct Round {
33 static S nearbyint(S s) { return round(s); }
34
35 typedef boost::mpl::integral_c<std::float_round_style, std::round_to_nearest> round_style;
36 };
37 typedef boost::numeric::converter<DestType, SourceType, boost::numeric::conversion_traits<DestType, SourceType>,
38 boost::numeric::def_overflow_handler, Round<SourceType>>
39 converter;
40
41 public:
42 DestType convert(SourceType& x) {
43 try {
44 return converter::convert(x);
45 }
46 catch(boost::numeric::positive_overflow&) {
47 return std::numeric_limits<DestType>::max();
48 }
49 catch(boost::numeric::negative_overflow&) {
50 return std::numeric_limits<DestType>::min();
51 }
52 catch(...) {
53 throw ChimeraTK::runtime_error("Data conversion failed for unknown reason.");
54 }
55 }
56 };

210 template<typename EpicsBaseType, typename EpicsType, typename CTKType>
211 EpicsBackendRegisterAccessor<EpicsBaseType, EpicsType, CTKType>::EpicsBackendRegisterAccessor(
212 const RegisterPath& path, boost::shared_ptr<DeviceBackend> backend, const EpicsBackendRegisterInfo& registerInfo,
213 AccessModeFlags flags, size_t numberOfWords, size_t wordOffsetInRegister, bool asyncReadActivated)
214 : EpicsBackendRegisterAccessorBase(
215 boost::dynamic_pointer_cast<EpicsBackend>(backend), registerInfo, numberOfWords, wordOffsetInRegister),
216 NDRegisterAccessor<CTKType>(path, flags) {
217 if constexpr(std::is_array_v<EpicsBaseType>) {
218 if(numberOfWords > 1) {
219 throw ChimeraTK::logic_error("No support for lso array.");
220 }
221 }
222 if(flags.has(AccessMode::raw)) throw ChimeraTK::logic_error("Raw access mode is not supported.");
223 NDRegisterAccessor<CTKType>::buffer_2D.resize(1);
224 this->accessChannel(0).resize(numberOfWords);
225 std::lock_guard<std::mutex> lock(ChannelManager::getInstance().mapLock);
226 auto pv = ChannelManager::getInstance().getPV(_info._caName);
227 if(flags.has(AccessMode::wait_for_new_data)) {
228 _hasNotificationsQueue = true;
229 _notifications = cppext::future_queue<EpicsRawData>(3);
230 _readQueue = _notifications.then<void>(
231 [pv](EpicsRawData& data) { memcpy(pv->value, data.data, data.size); }, std::launch::deferred);
232 }
233 if(pv->nElems != numberOfWords) _isPartial = true;
234 ChannelManager::getInstance().addAccessor(_info._caName, this);
235 if(flags.has(AccessMode::wait_for_new_data) && asyncReadActivated) {
236 ChannelManager::getInstance().activateChannel(_info._caName);
237 }
238 NDRegisterAccessor<CTKType>::_exceptionBackend = backend;
239 }

248 template<typename EpicsBaseType, typename EpicsType, typename CTKType>
249 void EpicsBackendRegisterAccessor<EpicsBaseType, EpicsType, CTKType>::doReadTransferSynchronously() {
250 _backend->checkActiveException();
251 std::lock_guard<std::mutex> lock(ChannelManager::getInstance().mapLock);
252 auto pv = ChannelManager::getInstance().getPV(_info._caName);
253 // one could also use ChannelManager::isChannelConnected -> however we ask explicitly ChannelAccess here
254 if(ca_state(pv->chid) == cs_conn) {
255 if(pv->nElems == 1) {
256 auto result = ca_get(pv->dbrType, pv->chid, pv->value);
257 if(result != ECA_NORMAL) {
258 throw ChimeraTK::runtime_error(std::string("Failed to read pv: ") + pv->name);
259 }
260 result = ca_pend_io(default_ca_timeout);
261 if(result == ECA_TIMEOUT) {
262 throw ChimeraTK::runtime_error(std::string("Read operation timed out for pv: ") + pv->name);
263 }
264 }
265 else {
266 auto result = ca_array_get(pv->dbrType, pv->nElems, pv->chid, pv->value);
267 if(result != ECA_NORMAL) {
268 throw ChimeraTK::runtime_error(std::string("Failed to read pv: ") + pv->name);
269 }
270 result = ca_pend_io(default_ca_timeout);
271 if(result == ECA_TIMEOUT) {
272 throw ChimeraTK::runtime_error(std::string("Read operation timed out for pv: ") + pv->name);
273 }
274 }
275 }
276 else {
277 throw ChimeraTK::runtime_error(
278 std::string("ChannelAccess not connected in doReadTransferSynchronously when writing: ") + _info._name + "(" +
279 pv->name + ")");
280 }
281 }

283 template<typename EpicsBaseType, typename EpicsType, typename CTKType>
284 void EpicsBackendRegisterAccessor<EpicsBaseType, EpicsType, CTKType>::doPostRead(TransferType, bool hasNewData) {
285 if(!hasNewData) return;
286 std::lock_guard<std::mutex> lock(ChannelManager::getInstance().mapLock);
287 auto pv = ChannelManager::getInstance().getPV(_info._caName);
288 EpicsBaseType* tmp = (EpicsBaseType*)dbr_value_ptr(pv->value, pv->dbrType);
289
290 if constexpr(std::is_array_v<EpicsBaseType>) {
291 // only single element as checked in the constructor
292 dbr_string_t tmpStr;
293 strcpy(tmpStr, (char*)tmp);
294 this->accessData(0) = toCTK.convert(tmpStr);
295 }
296 else {
297 for(size_t i = 0; i < _numberOfWords; i++) {
298 EpicsBaseType value = tmp[_offsetWords + i];
299 // Fill the NDRegisterAccessor buffer
300 this->accessData(i) = toCTK.convert(value);
301 }
302 }
303
304 EpicsType* tp = (EpicsType*)pv->value;
305 _currentVersion = EPICS::VersionMapper::getInstance().getVersion(tp[0].stamp);
306 if(_currentVersion < _backend->_startVersion) {
307 _currentVersion = _backend->_startVersion;
308 }
309 TransferElement::_versionNumber = _currentVersion;
310 }

312 template<typename EpicsBaseType, typename EpicsType, typename CTKType>
313 bool EpicsBackendRegisterAccessor<EpicsBaseType, EpicsType, CTKType>::doWriteTransfer(
314 VersionNumber /*versionNumber*/) {
315 _backend->checkActiveException();
316 auto pv = ChannelManager::getInstance().getPV(_info._caName);
317 // one could also use ChannelManager::isChannelConnected -> however we ask explicitly ChannelAccess here
318 if(ca_state(pv->chid) == cs_conn) {
319 if(_isPartial) EpicsBackendRegisterAccessor<EpicsBaseType, EpicsType, CTKType>::doReadTransferSynchronously();
320 long result;
321 if constexpr(std::is_array_v<EpicsBaseType>) {
322 // only single element as checked in the constructor
323 result = ca_array_put(pv->dbfType, pv->nElems, pv->chid, toEpics.convert(this->accessData(0)).c_str());
324 }
325 else {
326 EpicsBaseType* tmp = (EpicsBaseType*)dbr_value_ptr(pv->value, pv->dbrType);
327 for(size_t i = 0; i < _numberOfWords; i++) {
328 tmp[_offsetWords + i] = toEpics.convert(this->accessData(i));
329 }
330 result = ca_array_put(pv->dbfType, pv->nElems, pv->chid, tmp);
331 }
332
333 if(result != ECA_NORMAL) {
334 std::cerr << "Failed to to write pv: " << _info._caName << std::endl;
335 return false;
336 }
337 result = ca_pend_io(default_ca_timeout);
338 if(result == ECA_TIMEOUT) {
339 std::cerr << "Timeout while writing pv: " << _info._caName << std::endl;
340 return false;
341 }
342 }
343 else {
344 throw ChimeraTK::runtime_error(std::string("ChannelAccess not connected in doWriteTransfer when writing: ") +
345 _info._name + "(" + pv->name + ")");
346 }
347 return true;
348 }
