198void amtfDbCav_piezoCap::calculateLeastSquares(const rdataV<std::vector<float>, float> &x, const rdataV<std::vector<float>, float> &y, rdata<float> &a, rdata<float> &b)
199{
200 float sumX = 0;
201 float sumX2 = 0;
202 float sumY = 0;
203 float sumY2 = 0;
204 float sumXY = 0;
205 size_t N = x.size();
206 std::vector<float>::const_iterator tempIterX, tempIterY;
207 const std::vector<float> &vTempX = (std::vector<float>) x;
208 const std::vector<float> &vTempY = (std::vector<float>) y;
209 tempIterX = vTempX.begin();
210 tempIterY = vTempY.begin();
211 for (size_t i = 0; i < N; i++, ++tempIterX, ++tempIterY)
212 {
213 sumX += *tempIterX;
214 sumX2 += *tempIterX * *tempIterX;
215 sumY += *tempIterY;
216 sumY2 += *tempIterY * *tempIterY;
217 sumXY += *tempIterX * *tempIterY;
218
219 }
220
221 float denom = (N * sumX2 - sumX * sumX);
222
223 if (denom == 0) {
224 // nope, we can't solve the problem
225 return;
226 } else {
227 a = (N * sumXY - sumX * sumY ) / denom;
228 b = (sumY * sumX2 -sumX * sumXY) / denom;
229 }
230
231 return;
232}
