50 template<class DATA_TYPE>
51 std::pair<DATA_TYPE, bool> qvariantToStandardDataType(const QVariant& data) {
52  if(data.type() == HexDataType) {
53  if(!data.canConvert<HexData>()) {
54  return {DATA_TYPE(), false};
55  }
56  // The variant has a templated function called value() which returns the
57  // template type, HexData in our case. The HexData class itself has a member
58  // called value, which holds the actual value. Phew...
59  return {data.value<HexData>().value, true};
60  }
61  else {
62  if(!data.canConvert<DATA_TYPE>()) {
63  return {DATA_TYPE(), false};
64  }
65  return {data.value<DATA_TYPE>(), true};
66  }
67 }
