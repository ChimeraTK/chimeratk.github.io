43 template<class DATA_TYPE>
44 std::pair<DATA_TYPE, bool> qvariantToStandardDataType(const QVariant& data) {
45  if(data.type() == HexDataType) {
46  if(!data.canConvert<HexData>()) {
47  return {DATA_TYPE(), false};
48  }
49  // The variant has a templated function called value() which returns the
50  // template type, HexData in our case. The HexData class itself has a member
51  // called value, which holds the actual value. Phew...
52  return {data.value<HexData>().value, true};
53  }
54  else {
55  if(!data.canConvert<DATA_TYPE>()) {
56  return {DATA_TYPE(), false};
57  }
58  return {data.value<DATA_TYPE>(), true};
59  }
60 }
