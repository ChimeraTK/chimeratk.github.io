175 void RegisterTypeAbstractorRawImpl<RAW_DATA_TYPE, COOKED_DATA_TYPE>::updateCachedCookedData() {
176  _cachedCookedDataValid = false; // set to false in case something goes wrong during conversion
177  for(size_t channelIndex = 0; channelIndex < _cachedCookedData.size(); ++channelIndex) {
178  auto& cookedChan = _cachedCookedData[channelIndex];
179  for(size_t elementIndex = 0; elementIndex < cookedChan.size(); ++elementIndex) {
180  cookedChan[elementIndex] = _accessor.template getAsCooked<COOKED_DATA_TYPE>(channelIndex, elementIndex);
181  }
182  }
183  _cachedCookedDataValid = true;
