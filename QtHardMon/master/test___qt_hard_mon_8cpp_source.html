192  void switchRegisterSelection(std::vector<std::string> RegisterToSelect) {
193  registerToBeFound = NULL;
194  QList<QTreeWidgetItem*> registerList = qtHardMon->ui.registerTreeWidget->findItems(
195  RegisterToSelect.back().c_str(), Qt::MatchExactly | Qt::MatchRecursive);
196  RegisterToSelect.pop_back();
197 
198  for(std::vector<std::string>::reverse_iterator nameIter = RegisterToSelect.rbegin();
199  nameIter != RegisterToSelect.rend();
200  ++nameIter) {
201  // Iterate the list until we find the one with the right module
202 
203  for(QList<QTreeWidgetItem*>::iterator registerIter = registerList.begin(); registerIter != registerList.end();
204  ++registerIter) {
205  // if we found the right register select it and quit the loop
206  if((*registerIter)->parent()->text(0) == (*nameIter).c_str()) {
207  registerToBeFound = *registerIter;
208  break;
209  }
210  }
211  }
212  if(!registerToBeFound) {
213  BOOST_FAIL("Register not found...");
214  }
215  else {
216  qtHardMon->ui.registerTreeWidget->setCurrentItem(registerToBeFound);
217  }
218  }
219 };
