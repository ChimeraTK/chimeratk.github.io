54 template<class DATA_TYPE>
55 std::pair<DATA_TYPE, bool> qvariantToStandardDataType(const QVariant& data) {
56  if(data.type() == HexDataType) {
57  if(!data.canConvert<HexData>()) {
58  return {DATA_TYPE(), false};
59  }
60  // The variant has a templated function called value() which returns the
61  // template type, HexData in our case. The HexData class itself has a member
62  // called value, which holds the actual value. Phew...
63  return {data.value<HexData>().value, true};
64  }
65  else {
66  if(!data.canConvert<DATA_TYPE>()) {
67  return {DATA_TYPE(), false};
68  }
69  return {data.value<DATA_TYPE>(), true};
70  }
71 }
