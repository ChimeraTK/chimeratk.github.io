8RegisterAccessorModel::RegisterAccessorModel(
9 QObject* parent, std::shared_ptr<RegisterTypeAbstractor> const& abstractAccessor)
10: QAbstractTableModel(parent), _abstractAccessor(abstractAccessor), _channelNumber(0), _cookedHexColumnIndex(-1),
11 _rawColumnIndex(-1), _rawHexColumnIndex(-1), _nColumns(1) {
12 _isModified.resize(_abstractAccessor->nChannels());
13 for(auto& channelModifiedFlags : _isModified) {
14 channelModifiedFlags.resize(_abstractAccessor->nElements());
15 // Set all entries to modified to flag the 0s of a write-only register (indicate that
16 // these are not the values on the device).
17 // The (automatic) read for normal registers will clear it, so there is no behaviour change
18 // for readable registers.
19 for(auto modifiedFlag : channelModifiedFlags) {
20 modifiedFlag = true;
21 }
22 }
23 // show hexadecimal representation for integer data
24 // notice: DataType::isIntegral can only be true if the data type is not none.
25 // we don't have to check this separately
26 if(_abstractAccessor->isIntegral() && !_abstractAccessor->rawDataType().isIntegral()) { // there will be no raw hex
27 // column
28 _cookedHexColumnIndex = _nColumns++;
29 }
30 if(_abstractAccessor->rawDataType() != DataType::none) {
31 // add a raw column if there is a raw data type
32 _rawColumnIndex = _nColumns++;
33 if(_abstractAccessor->rawDataType().isIntegral()) {
34 // add a raw hex column if the raw data type is an integer
35 _rawHexColumnIndex = _nColumns++;
36 }
37 }
38}

53QVariant RegisterAccessorModel::data(const QModelIndex& modelIndex, int role) const {
54 switch(role) {
55 case Qt::DisplayRole:
56 case Qt::EditRole:
57 if(modelIndex.row() >= PreferencesProviderSingleton::Instance().getValue<int>("maxWords")) {
58 // If the row is larger than maxWords show "truncated"
59 // If the size of the accessor is set correctly, it should be maxWords+1,
60 // so there should be only one line with this content.
61 return QString("truncated");
62 }
63 if(modelIndex.column() == _cookedHexColumnIndex) {
64 return _abstractAccessor->dataAsHex(_channelNumber, modelIndex.row());
65 }
66 else if(modelIndex.column() == _rawColumnIndex) {
67 return _abstractAccessor->rawData(_channelNumber, modelIndex.row());
68 }
69 else if(modelIndex.column() == _rawHexColumnIndex) {
70 return _abstractAccessor->rawDataAsHex(_channelNumber, modelIndex.row());
71 }
72 else {
73 return _abstractAccessor->data(_channelNumber, modelIndex.row());
74 }
75 case Qt::BackgroundRole:
76 if(_isModified[_channelNumber][modelIndex.row()]) { // change background color for modified
77 // rows
78 return QBrush(QColor(255, 100, 100, 255)); // a light red
79 }
80 break;
81 }
82 return QVariant();
83}
