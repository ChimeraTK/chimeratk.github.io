92template<class T>
93T ConfigFileReaderWriter::getValue(std::string const& variableName, T const& defaultValue) const {
94 // get the map using the getMap functions, which are overloaded by type
95 std::map<std::string, T> const* map;
96 getMap(map);
97
98 // Try to find the value. When it's not found the iterator is map->end()
99 typename std::map<std::string, T>::const_iterator valueIter = map->find(variableName);
100 // std::map< std::string, T>::iterator valueIter = map->find(variableName);
101 if(valueIter == map->end()) {
102 // not found, return the default
103 return defaultValue;
104 }
105 else {
106 // the map iter points to a pair. The 'second' entry is the value.
107 return valueIter->second;
108 }
109}
