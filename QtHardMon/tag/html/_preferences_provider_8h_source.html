39T PreferencesProvider::getValue(QString key) {
40 QVariant value = getRawValue(key);
41 std::stringstream temporaryStream;
42
43 switch(static_cast<QMetaType::Type>(value.type())) {
44 case QMetaType::Int:
45 temporaryStream << value.toInt();
46 break;
47 case QMetaType::Float:
48 temporaryStream << value.toFloat();
49 break;
50 case QMetaType::Double:
51 temporaryStream << value.toDouble();
52 break;
53 case QMetaType::QString:
54 temporaryStream << value.toString().toStdString();
55 break;
56 case QMetaType::Bool:
57 temporaryStream << value.toBool();
58 break;
59 default:
60 throw InvalidOperationException("Type not supported. Use getRawValue instead.");
61 }

63 T returnedValue;
