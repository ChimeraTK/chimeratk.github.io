20 /*
21  * Method displays double values in the cell with decimal places. ((eg: 123.6500
22  * if precision required is 4)). The desired number of decimal places
23  * (precision) can be specified by setting
24  * _doubleSpinBoxPrecision
25  */
26 QString CustomDelegates::displayText(const QVariant& value, const QLocale& locale) const {
27  if(value.type() == QVariant::Double) { // frame and return a formatted
28  // QString if handling a cell with
29  // double value
30  return locale.toString((value.toDouble()), 'f', _doubleSpinBoxPrecision);
31  }
32  else if(value.type() == HexDataType) {
33  HexData hexValue = value.value<HexData>(); // get the datatype stored in this cell
34  // QString::number has a bug and gives negative values as 64 bit hex.
35  // So we have to do it manually.
36  int intValue = hexValue.value; // covert the value to 32 bit int
37  std::stringstream s;
38  s << "0x" << std::hex << intValue;
39  return QString(s.str().c_str());
40  }
41  else { // else let the default delegate take care of formatting if
42  // content is not double
43  return QStyledItemDelegate::displayText(value, locale);
44  }
45 }

49  * spinbox supports a precision specified by _doubleSpinBoxPrecision.

51 QWidget* CustomDelegates::createEditor(QWidget* parent_,
52  const QStyleOptionViewItem& option,
53  const QModelIndex& index) const {
54  if(index.data(Qt::EditRole).type() == QVariant::Double) { // create spinbox with custom precision
55  // for cells in the double column
56  QDoubleSpinBox* doubleSpinBox = new QDoubleSpinBox(parent_);
57  doubleSpinBox->setRange(std::numeric_limits<double>::lowest(), std::numeric_limits<double>::max());
58  doubleSpinBox->setDecimals(_doubleSpinBoxPrecision);
59  return doubleSpinBox;
60  }
61  else if(index.data(Qt::EditRole).type() == HexDataType) {
62  HexSpinBox* hexSpinBox = new HexSpinBox(parent_);
63  hexSpinBox->setRange(std::numeric_limits<int>::min(), std::numeric_limits<int>::max());
64  return hexSpinBox;
65  }
66  else {
67  // should work for int, uint, string and bool
68  return QStyledItemDelegate::createEditor(parent_, option, index);
69  }
