34void PropertiesWidget::updateRegisterInfo(const ChimeraTK::RegisterInfo& registerInfo) {
35 // There are node elements ("modules") which don't have register information
36 if(!registerInfo.isValid()) {
37 clearFields();
38 return;
39 }
40
41 setEnabled(true);
42 ui.registerPathDisplay->setText(static_cast<const std::string&>(registerInfo.getRegisterName()).c_str());
43 setShape(
44 registerInfo.getNumberOfDimensions(), registerInfo.getNumberOfChannels(), registerInfo.getNumberOfElements());
45 setType(registerInfo.getDataDescriptor());
46 ui.wait_for_new_dataDisplay->setText(
47 registerInfo.getSupportedAccessModes().has(ChimeraTK::AccessMode::wait_for_new_data) ? "yes" : "no");
48
49 // Try to cast to the old, numeric addressed registerInfo.
50 auto numericAddressedInfo = dynamic_cast<const ChimeraTK::NumericAddressedRegisterInfo*>(&registerInfo.getImpl());
51 // If the cast succeeded fill the numeric addressed and fixed point fields.
52 // This info only exists for this type of registers.
53 if(numericAddressedInfo) {
54 setAddress(numericAddressedInfo->bar, numericAddressedInfo->address,
55 numericAddressedInfo->nElements * numericAddressedInfo->elementPitchBits / 8);
56 setFixedPointInfo(numericAddressedInfo->channels.front().width,
57 numericAddressedInfo->channels.front().nFractionalBits, numericAddressedInfo->channels.front().signedFlag,
58 registerInfo.getNumberOfDimensions());
59 }
60 else {
61 ui.numericalAddressGroupBox->hide();
62 ui.fixedPointGroupBox->hide();
63 }
64}
