75void PlotWindow::plot() {
76#if(USE_QWT)
77 QVector<QPointF> samples;
78
79 PreferencesProvider& preferencesProvider = PreferencesProviderSingleton::Instance();
80
81 // the plot window might be active while no register is selected. Just don't
82 // do anything in this case.
83 if(!(_hardMon->currentAccessorModel_)) {
84 return;
85 }
86
87 // We use the data model here. It automatically always gives the cooked data
88 // and the right size. In addition, if the coversion to double fails, we get
89 // this information and don't display.
90 //@todo Get the data type and show a message why data cannot be plotted
91 // note: In rowCount we can use an invalid model index.
92 for(int row = 0; row <
93 std::min(_hardMon->currentAccessorModel_->rowCount(QModelIndex()), preferencesProvider.getValue<int>("maxWords"));
94 ++row) {
95 auto dataPoint =
96 _hardMon->currentAccessorModel_->data(_hardMon->currentAccessorModel_->index(row, 0 /* =cooked data column*/));
97
98 // If the conversion cannot be executed, the value is 0. No need to do
99 // special handling here. For strings for example canConvert is true, and
100 // "3" actually is converted to 3.0, while "hello world" will result in 0.
101 // All other currently supported user data types should be properly
102 // convertible to double (void is still to come, we will see). Anyway, with
103 // QVariants default behaviour to return 0 we should be fine.
104 double value = dataPoint.value<double>();
105
106 samples.push_back(QPointF(row, value));
107 }
108
109 _myData->setSamples(samples);
110
111 _curve1->setData(_myData);
112
113 _qwtPlot->setAxisAutoScale(QwtPlot::yLeft);
114 _qwtPlot->setAxisAutoScale(QwtPlot::xBottom);
115 _qwtPlot->setAxisFont(QwtPlot::yLeft, this->font());
116 _qwtPlot->setAxisFont(QwtPlot::xBottom, this->font());
117
118 _qwtPlot->replot();
119
120 if(_zoomer) delete _zoomer;
121
122 _zoomer = new QwtPlotZoomer(_qwtPlot->canvas());
123 _zoomer->setMousePattern(QwtEventPattern::MouseSelect2, Qt::RightButton, Qt::ControlModifier);
124 _zoomer->setMousePattern(QwtEventPattern::MouseSelect3, Qt::RightButton);
125
126#endif // USE_QWT
127}
