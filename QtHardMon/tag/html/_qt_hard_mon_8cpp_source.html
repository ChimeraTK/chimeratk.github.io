287void QtHardMon::selectPreviousRegister() {
288 if(!(ui.autoselectPreviousRegisterCheckBox->isChecked())) {
289 // don't re-select if this option is not enabled
290 return;
291 }
292 // Searching a sub-tree does not work in QTreeWidget. So here is the
293 // strategy: First get all register with the right name from the tree,
294 // then pick the one where the full path matches. Ugly as hell, but easier
295 // than writing an recursive search function ourself.
296 auto registerPathComponents = _currentDeviceListItem->lastSelectedRegister.getComponents();
297 if(registerPathComponents.empty()) {
298 // nothing to be done, no previous register
299 return;
300 }
301
302 auto registerName = registerPathComponents.back().c_str();
303 // Get a list of all registers with this name.
304 QList<QTreeWidgetItem*> registerList =
305 ui.registerTreeWidget->findItems(registerName, Qt::MatchExactly | Qt::MatchRecursive);
306
307 // Iterate the list until we find the right one
308 for(auto reg : registerList) {
309 // we know that there are only DeviceElementQTreeItems in the list, so we
310 // can static cast
311 auto deviceElement = static_cast<DeviceElementQTreeItem*>(reg);
312 if(deviceElement->getRegisterPath() == _currentDeviceListItem->lastSelectedRegister) {
313 ui.registerTreeWidget->setCurrentItem(reg);
314 break;
315 }
316 }

355 ui.continuousReadCheckBox->setChecked(false); // stop the continuous reading thread
356 ui.continuousReadCheckBox->setEnabled(false); // prevent from starting the thread
357 // as long as there is no valid accessor
358 // Always clear the old data model. This is needed in all use cases below.
359 ui.propertiesWidget->ui.valuesTableView->setModel(nullptr);
360 delete currentAccessorModel_;
361 ui.lastReadTime->setText("");
362 currentAccessorModel_ = nullptr;
363 bool accessorWritable = false;
364
365 // There is a case when a device entry is clicked in the device list, the slot
366 // is called with a NULL registerItem
367 if(!registerItem) {
368 ui.propertiesWidget->clearFields();

370 }
371
372 // we know that the registerItem is a DeviceElementQTreeItem, so we can static
373 // cast.

375 ui.propertiesWidget->updateRegisterInfo(selectedItem->getRegisterInfo());
376 if(selectedItem->getRegisterInfo().isValid()) {
377 // there is valid register information. Create an accessor

379 try {
380 abstractAccessor = createAbstractAccessor(selectedItem->getRegisterInfo(), currentDevice_);

382 catch(std::exception& e) {
383 showMessageBox(QMessageBox::Critical, QString("QtHardMon : Error"),
384 QString("Could not get register accessor for ") +

386 QString("Info: An exception was thrown:\n") + e.what());

390 // QtHardMon. In this case we don't have an accessor (pointer is null) or a
391 // data model.

395 currentAccessorModel_ = new RegisterAccessorModel(this, abstractAccessor);

398 ui.continuousReadCheckBox->setEnabled(true);

401 }

406 ui.writeButton->setEnabled(selectedItem->getRegisterInfo().isWriteable() || accessorWritable);

417 ui.readButton->setEnabled(false);
418 ui.writeButton->setEnabled(false);

423
424 PreferencesProvider& preferencesProvider = PreferencesProviderSingleton::Instance();
425
426 // Only call autoread for registers (which have a valid accessor model).
427 // Don't call it for modules (where currentAccessorModel_ == nullptr)
428 if(preferencesProvider.getValue<bool>("autoRead") && currentAccessorModel_ && currentAccessorModel_->isReadable()) {
429 // It is allowed to call a blocking read here even with wait_for_new_data because we just created the accessor

432 }

587void QtHardMon::loadConfig(QString const& configFileName) {
588 ConfigFileReaderWriter configReader;
589 try {
590 // even the constructor can throw when opening and reading a file
591 configReader.read(configFileName.toStdString());
592 }
593 catch(std::ifstream::failure& e) {
594 showMessageBox(QMessageBox::Warning, QString("QtHardMon : Warning"),
595 QString("Could not read config file ") + configFileName + ".",
596 QString("Info: An exception was thrown:\n ") + e.what());
597 return;
598 }
599
600 // At this point we considder the config file as successfully loaded, even
601 // though it might not contain any
602 // valid parameters. But we store the file name.
603 configFileName_ = configFileName;
604
605 // show message box with parse errors, but just continue normally
606 if(!configReader.getBadLines().isEmpty()) {
607 QString infoText;
608 for(QStringList::const_iterator badLinesIter = configReader.getBadLines().begin();
609 badLinesIter != configReader.getBadLines().end(); ++badLinesIter) {
610 infoText += *badLinesIter;
611 infoText += "\n";
612 }
613
614 showMessageBox(QMessageBox::Warning, QString("QtHardMon : Warning"),
615 QString("The following lines from the config file ") + configFileName +
616 " are invalid and will be ignored. Please fix your config file.\n",
617 infoText);
618 return;
619 }
620
621 PreferencesProvider& preferencesProvider = PreferencesProviderSingleton::Instance();
622
623 // first handle all settings that do not depend on opening a device map
624
625 preferencesProvider.setValue("floatPrecision",
626 configReader.getValue(PRECISION_INDICATOR_STRING, preferencesProvider.getValue<int>("floatPrecision")));
627
628 // store in a local variable for now
629 int maxWords = configReader.getValue(MAX_WORDS_STRING, preferencesProvider.getValue<int>("maxWords"));
630 // check for validity. Minimum reasonable value is 1.
631 if(maxWords >= 1) {
632 // only after checking set the class wide maxWords variable

674 QApplication::setFont(newFont);
675 }

678 // can just exit the function because
679 // all other variables have already been processed.

681

684 // sense without dmap

700 deviceListItem->lastSelectedRegister = configReader.getValue(deviceRegisterString, std::string());
701 } // for deviceRow

704 std::string currentDeviceString = configReader.getValue(CURRENT_DEVICE_STRING, std::string());

709 }

715 if(matchingDevices.isEmpty()) {

1111void QtHardMon::updateAvgReadInterval(QDateTime timeStamp) {
1112 _lastTimeStamps.push_back(timeStamp.toMSecsSinceEpoch());
1113
1114 size_t n = 10; // number of averaging points
1115 if(_lastTimeStamps.size() == n + 1) { // we are calculating delta_t , so we need one more value
1116 // evaluate and clear the list
1117 double sumDelta_t = 0;
1118 double sumSquareDelta_t = 0;
1119 for(auto it = _lastTimeStamps.begin(); it != (--_lastTimeStamps.end()); /*increment in loop*/) {
1120 auto t1 = *it;
1121 auto t2 = *(++it);
1122 auto delta_t = t2 - t1;
1123 sumDelta_t += delta_t;
1124 sumSquareDelta_t += delta_t * delta_t;
1125 }
1126 for(size_t i = 0; i < n; ++i) { // leave the last element in the list for further calculations
1127 _lastTimeStamps.pop_front();
1128 }
1129 double variance = sqrt((sumSquareDelta_t - sumDelta_t * sumDelta_t / n) / (n - 1));
1130 double mean = sumDelta_t / n;
1131 std::string s = std::to_string(int(std::round(mean))) + " +- " + std::to_string(int(std::round(variance))) + " ms";
1132 ui.avgUpdateInterval->setText(s.c_str());
1133 }
1134}
