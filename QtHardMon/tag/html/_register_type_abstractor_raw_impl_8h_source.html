185template<class RAW_DATA_TYPE, class COOKED_DATA_TYPE>
186void RegisterTypeAbstractorRawImpl<RAW_DATA_TYPE, COOKED_DATA_TYPE>::updateCachedCookedData() {
187 _cachedCookedDataValid = false; // set to false in case something goes wrong during conversion
188 for(size_t channelIndex = 0; channelIndex < _cachedCookedData.size(); ++channelIndex) {
189 auto& cookedChan = _cachedCookedData[channelIndex];
190 for(size_t elementIndex = 0; elementIndex < cookedChan.size(); ++elementIndex) {
191 cookedChan[elementIndex] = _accessor.template getAsCooked<COOKED_DATA_TYPE>(channelIndex, elementIndex);
192 }
193 }
194 _cachedCookedDataValid = true;
195}
