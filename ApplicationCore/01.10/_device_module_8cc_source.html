118  VirtualModule& DeviceModule::virtualiseFromCatalog() const {
119  if(virtualisedModuleFromCatalog_isValid) return virtualisedModuleFromCatalog;
120 
121  virtualisedModuleFromCatalog = VirtualModule(deviceAliasOrURI, "Device module", ModuleType::Device);
122 
123  if(!deviceIsInitialized) {
124  device = Device(deviceAliasOrURI);
125  deviceIsInitialized = true;
126  }
127  // obtain register catalogue
128  auto catalog = device.getRegisterCatalogue();
129  // iterate catalogue, create VariableNetworkNode for all registers starting
130  // with the registerNamePrefix
131  size_t prefixLength = registerNamePrefix.length();
132  for(auto& reg : catalog) {
133  if(std::string(reg.getRegisterName()).substr(0, prefixLength) != std::string(registerNamePrefix)) continue;
134 
135  // ignore 2D registers
136  if(reg.getNumberOfDimensions() > 1) continue;
137 
138  // guess direction and determine update mode
139  VariableDirection direction;
140  UpdateMode updateMode;
141  if(reg.isWriteable()) {
142  direction = {VariableDirection::consuming, false};
143  updateMode = UpdateMode::push;
144  }
145  else {
146  direction = {VariableDirection::feeding, false};
147  if(reg.getSupportedAccessModes().has(AccessMode::wait_for_new_data)) {
148  updateMode = UpdateMode::push;
149  }
150  else {
151  updateMode = UpdateMode::poll;
152  }
153  }
154 
155  // guess type
156  const std::type_info* valTyp{&typeid(AnyType)};
157  auto& dd = reg.getDataDescriptor(); // numeric, string, boolean, nodata, undefined
158  if(dd.fundamentalType() == RegisterInfo::FundamentalType::numeric) {
159  if(dd.isIntegral()) {
160  if(dd.isSigned()) {
161  if(dd.nDigits() > 11) {
162  valTyp = &typeid(int64_t);
163  }
164  else if(dd.nDigits() > 6) {
165  valTyp = &typeid(int32_t);
166  }
167  else if(dd.nDigits() > 4) {
168  valTyp = &typeid(int16_t);
169  }
170  else {
171  valTyp = &typeid(int8_t);
172  }
173  }
174  else {
175  if(dd.nDigits() > 10) {
176  valTyp = &typeid(uint64_t);
177  }
178  else if(dd.nDigits() > 5) {
179  valTyp = &typeid(uint32_t);
180  }
181  else if(dd.nDigits() > 3) {
182  valTyp = &typeid(uint16_t);
183  }
184  else {
185  valTyp = &typeid(uint8_t);
186  }
187  }
188  }
189  else { // fractional
190  valTyp = &typeid(double);
191  }
192  }
193  else if(dd.fundamentalType() == RegisterInfo::FundamentalType::boolean) {
194  valTyp = &typeid(int32_t);
195  }
196  else if(dd.fundamentalType() == RegisterInfo::FundamentalType::string) {
197  valTyp = &typeid(std::string);
198  }
199  else if(dd.fundamentalType() == RegisterInfo::FundamentalType::nodata) {
200  valTyp = &typeid(int32_t);
201  }
202 
203  auto name = std::string(reg.getRegisterName()).substr(prefixLength);
204  auto lastSlash = name.find_last_of("/");
205  std::string dirname, basename;
206  if(lastSlash != std::string::npos) {
207  dirname = name.substr(0, lastSlash);
208  basename = name.substr(lastSlash + 1);
209  }
210  else {
211  dirname = "";
212  basename = name;
213  }
214  VariableNetworkNode node(
215  basename, deviceAliasOrURI, reg.getRegisterName(), updateMode, direction, *valTyp, reg.getNumberOfElements());
216  virtualisedModuleFromCatalog.createAndGetSubmoduleRecursive(dirname).addAccessor(node);
217  }
218 
219  virtualisedModuleFromCatalog_isValid = true;
220  return virtualisedModuleFromCatalog;
221  }

225  void DeviceModule::reportException(std::string errMsg) {
226  try {
227  if(owner->isTestableModeEnabled()) {
228  assert(owner->testableModeTestLock());
229  }
230 
231  // The error queue must only be modified when holding both mutexes (error mutex and testable mode mutex), because
232  // the testable mode counter must always be consistent with the content of the queue.
233  // To avoid deadlocks you must always first aquire the testable mode mutex if you need both.
234  // You can hold the error mutex without holding the testable mode mutex (for instance for checking the error predicate),
235  // but then you must not try to aquire the testable mode mutex!
236  std::unique_lock<std::mutex> errorLock(errorMutex);
237 
238  if(!deviceHasError) { // only report new errors if the device does not have reported errors already
239  if(errorQueue.push(errMsg)) {
240  if(owner->isTestableModeEnabled()) ++owner->testableMode_counter;
241  } // else do nothing. There are plenty of errors reported already: The queue is full.
242  // set the error flag and notify the other threads
243  deviceHasError = true;
244  errorLock.unlock();
245  errorIsReportedCondVar.notify_all();
246  }
247  else {
248  errorLock.unlock();
249  }
250 
251  //Wait until the error condition has been cleared by the exception handling thread.
252  //We must not hold the testable mode mutex while doing so, otherwise the other thread will never run to fulfill the condition
253  owner->testableModeUnlock("waitForDeviceOK");
254  errorLock.lock();
255  while(deviceHasError) {
256  errorIsResolvedCondVar.wait(errorLock);
257  boost::this_thread::interruption_point();
258  }
259  errorLock.unlock();
260  owner->testableModeLock("continueAfterException");
261  }
262  catch(...) {
263  //catch any to notify the waiting thread(s) (exception handling thread), then re-throw
264  errorIsReportedCondVar.notify_all();
265  throw;
266  }
267  }

271  void DeviceModule::handleException() {
272  Application::registerThread("DM_" + getName());
273  std::string error;
274  owner->testableModeLock("Startup");
275  try {
276  while(!device.isOpened()) {
277  try {
278  boost::this_thread::interruption_point();
279  usleep(500000);
280  device.open();
281  if(deviceError.status != 0) {
282  deviceError.status = 0;
283  deviceError.message = "";
284  deviceError.setCurrentVersionNumber({});
285  deviceError.writeAll();
286  }
287  }
288  catch(ChimeraTK::runtime_error& e) {
289  if(deviceError.status != 1) {
290  deviceError.status = 1;
291  deviceError.message = e.what();
292  deviceError.setCurrentVersionNumber({});
293  deviceError.writeAll();
294  }
295  }
296  }
297  // The device was successfully opened, try to initialise it
298  try {
299  for(auto& initHandler : initialisationHandlers) {
300  initHandler(this);
301  }
302  for(auto& te : writeAfterOpen) {
303  te->write();
304  }
305  }
306  catch(ChimeraTK::runtime_error& e) {
307  // we just report the exception and let the exception handling loop do the rest
308  std::lock_guard<std::mutex> locakGuard(errorMutex);
309  deviceHasError = true;
310  if(errorQueue.push(e.what())) {
311  if(owner->isTestableModeEnabled()) ++owner->testableMode_counter;
312  }
313  }
314 
315  while(true) {
316  // We need one interruption point at the beginning of the loop. It is always executed and is not blocked by a lock (except the testable mode lock)
317  boost::this_thread::interruption_point();
318  // release the testable mode mutex for waiting for the exception.
319  owner->testableModeUnlock("Wait for exception");
320  // Do not modify the queue without holding the testable mode lock, because we also consitenly have to modify the counter protected by that mutex.
321  // Just check the condition variable.
322  std::unique_lock<std::mutex> errorLock(errorMutex);
323  while(!deviceHasError) {
324  boost::this_thread::interruption_point(); // Make sure not to start waiting for the condition variable if
325  // interruption was requested.
326  errorIsReportedCondVar.wait(errorLock); // this releases the mutex while waiting
327  boost::this_thread::interruption_point(); // we need an interruption point in the waiting loop
328  }
329 
330  errorLock.unlock(); // We must not hold the error lock when getting the testable mode mutex to avoid deadlocks!
331  owner->testableModeLock("Process exception");
332  errorLock.lock(); // we need both locks to modify the queue, so get it again.
333 
334  auto popResult = errorQueue.pop(error);
335  (void)popResult;
336  assert(popResult); // this if should always be true, otherwise the condition variable logic is wrong
337  if(owner->isTestableModeEnabled()) {
338  assert(owner->testableMode_counter > 0);
339  --owner->testableMode_counter;
340  }
341 
342  // report exception to the control system
343  deviceError.status = 1;
344  deviceError.message = error;
345  deviceError.setCurrentVersionNumber({});
346  deviceError.writeAll();
347 
348  // wait some time until retrying.
349  // Never sleep when holding a lock
350  errorLock.unlock(); // we must not hold the error lock when not having the testable mode mutex
351  owner->testableModeUnlock("Wait for recovery");
352 
353  do {
354  usleep(500000);
355  boost::this_thread::interruption_point();
356  try {
357  // This triggers a recovery (device is already opened).
358  device.open();
359  }
360  catch(ChimeraTK::runtime_error&) {
361  // Recovery failed. Just catch the exception and retry.
362  continue; // should not be necessary because isFunctional() should return false. But no harm in leaving it in.
363  }
364  } while(!device.isFunctional());
365 
366  owner->testableModeLock("Try recovery");
367  errorLock.lock();
368 
369  // FIXME: we have to wait here until the device reports that it is functional again.
370  // Otherwise we spam the error reporting with 2 Hz.
371 
372  try {
373  // re-initialise the device before continuing
374  for(auto& initHandler : initialisationHandlers) {
375  initHandler(this);
376  }
377  { // scope for the lock guard
378  boost::unique_lock<boost::shared_mutex> uniqueLock(recoverySharedMutex);
379  for(auto& te : writeRecoveryOpen) {
380  te->write();
381  }
382  } // end of scope for the lock guard
383  }
384  catch(ChimeraTK::runtime_error& e) {
385  // Report the error. This puts the exception to the queue and we can continue with waiting for the queue.
386  // It is sure we will enter it again because we just pushed to it, so if the device recovers we will notice.
387  // Do not use reportError, which would just do the mutext business in addition to pushing to the queue, but we already have the mutex.
388  if(errorQueue.push(e.what())) {
389  if(owner->isTestableModeEnabled()) ++owner->testableMode_counter;
390  }
391  continue;
392  }
393 
394  // We survived the initialisation (if any). It seems the device is working again.
395  // Empty exception reporting queue.
396  while(errorQueue.pop()) {
397  if(owner->isTestableModeEnabled()) {
398  assert(owner->testableMode_counter > 0);
399  --owner->testableMode_counter;
400  }
401  }
402  // Reset exception state and wait for the next error to be reported.
403  deviceError.status = 0;
404  deviceError.message = "";
405  deviceError.setCurrentVersionNumber({});
406  deviceError.writeAll();
407 
408  deviceHasError = false;
409 
410  // send the trigger that the device is available again
411  deviceBecameFunctional.write();
412 
413  errorLock.unlock();
414  errorIsResolvedCondVar.notify_all();
415  }
416  }
417  catch(...) {
418  // before we leave this thread, we might need to notify other waiting
419  // threads. boost::this_thread::interruption_point() throws an exception
420  // when the thread should be interrupted, so we will end up here
421  errorIsResolvedCondVar.notify_all();
422  throw;
423  }
424  }
