34 /* Module which converts the input data from inches to centimeters - and the
35  * other way round for the return channel. In case of the return channel, the
36  * data is rounded downwards to integer inches and sent again forward. */
37 struct ModuleA : public ctk::ApplicationModule {
38  using ctk::ApplicationModule::ApplicationModule;
39 
40  ctk::ScalarPushInputWB<int> var1{this, "var1", "inches", "A length, for some reason rounded to integer"};
41  ctk::ScalarOutputPushRB<double> var2{this, "var2", "centimeters", "Same length converted to centimeters"};
42 
43  void prepare() override {
44  incrementDataFaultCounter(); // foce all outputs to invalid
45  writeAll(); // write initial values
46  decrementDataFaultCounter(); // validity according to input validity
47  }
48 
49  void mainLoop() override {
50  auto group = readAnyGroup();
51  while(true) {
52  auto var = group.readAny();
53  if(var == var2.getId()) {
54  var1 = std::floor(var2 / 2.54);
55  var1.write();
56  }
57  var2 = var1 * 2.54;
58  var2.write();
59  }
60  }
61 };

65 /* Module which limits a value to stay below a maximum value. */
66 struct ModuleB : public ctk::ApplicationModule {
67  using ctk::ApplicationModule::ApplicationModule;
68 
69  ctk::ScalarPushInputWB<double> var2{this, "var2", "centimeters", "Some length, confined to a configuratble range"};
70  ctk::ScalarPushInput<double> max{this, "max", "centimeters", "Maximum length"};
71  ctk::ScalarOutput<double> var3{this, "var3", "centimeters", "The limited length"};
72 
73  void prepare() override {
74  incrementDataFaultCounter(); // foce all outputs to invalid
75  writeAll(); // write initial values
76  decrementDataFaultCounter(); // validity according to input validity
77  }
78 
79  void mainLoop() override {
80  auto group = readAnyGroup();
81  while(true) {
82  auto var = group.readAny();
83  bool write = var == var2.getId();
84  if(var2 > max) {
85  var2 = static_cast<double>(max);
86  var2.write();
87  write = true;
88  }
89  if(write) { // write only if var2 was received or the value was changed
90  // due to a reduced limit
91  var3 = static_cast<double>(var2);
92  var3.write();
93  }
94  }
95  }
96 };
