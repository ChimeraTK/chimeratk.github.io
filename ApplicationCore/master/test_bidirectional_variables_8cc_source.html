30 /* Module which converts the input data from inches to centimeters - and the
31  * other way round for the return channel. In case of the return channel, the
32  * data is rounded downwards to integer inches and sent again forward. */
33 struct ModuleA : public ctk::ApplicationModule {
34  using ctk::ApplicationModule::ApplicationModule;
35 
36  ctk::ScalarPushInputWB<int> var1{this, "var1", "inches", "A length, for some reason rounded to integer"};
37  ctk::ScalarOutputPushRB<double> var2{this, "var2", "centimeters", "Same length converted to centimeters"};
38 
39  void mainLoop() {
40  auto group = readAnyGroup();
41  while(true) {
42  auto var = group.readAny();
43  if(var == var2.getId()) {
44  var1 = std::floor(var2 / 2.54);
45  var1.write();
46  }
47  var2 = var1 * 2.54;
48  var2.write();
49  }
50  }
51 };

55 /* Module which limits a value to stay below a maximum value. */
56 struct ModuleB : public ctk::ApplicationModule {
57  using ctk::ApplicationModule::ApplicationModule;
58 
59  ctk::ScalarPushInputWB<double> var2{this, "var2", "centimeters", "Some length, confined to a configuratble range"};
60  ctk::ScalarPushInput<double> max{this, "max", "centimeters", "Maximum length"};
61  ctk::ScalarOutput<double> var3{this, "var3", "centimeters", "The limited length"};
62 
63  void mainLoop() {
64  auto group = readAnyGroup();
65  while(true) {
66  auto var = group.readAny();
67  bool write = var == var2.getId();
68  if(var2 > max) {
69  var2 = static_cast<double>(max);
70  var2.write();
71  write = true;
72  }
73  if(write) { // write only if var2 was received or the value was changed
74  // due to a reduced limit
75  var3 = static_cast<double>(var2);
76  var3.write();
77  }
78  }
79  }
80 };

174
