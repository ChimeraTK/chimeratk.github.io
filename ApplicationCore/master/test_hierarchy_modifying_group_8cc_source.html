76 /* Helper for tests: Allows to test the virtual hierarchy quickly, assuming each test case uses a separate tag
77  * and each tag/test case has exactly one variable. */
78 struct TestHelper {
79  template<typename APP>
80  TestHelper(APP& app, const std::string tag) : root(app.findTag(tag)) {
81  current = &root;
82  root.dump();
83  }
84 
85  TestHelper submodule(const std::string& name) {
86  BOOST_REQUIRE(current->getSubmoduleList().size() == 1);
87  BOOST_CHECK(current->getAccessorList().size() == 0);
88  auto child = current->getSubmoduleList().front();
89  BOOST_CHECK(child->getName() == name);
90  return TestHelper(root, child);
91  }
92 
93  void accessor(ctk::VariableNetworkNode node) {
94  BOOST_CHECK(current->getSubmoduleList().size() == 0);
95  BOOST_REQUIRE(current->getAccessorList().size() == 1);
96  BOOST_CHECK(current->getAccessorList().front() == node);
97  }
98 
99  protected:
100  TestHelper(ctk::VirtualModule& _root, ctk::Module* _current) : root(_root), current(_current) {}
101 
102  ctk::VirtualModule root;
103  ctk::Module* current{nullptr};
104 };
