43  ChimeraTK::ReadAnyGroup Module::readAnyGroup() {
44  auto recursiveAccessorList = getAccessorListRecursive();
45 
46  // put push-type transfer elements into a ReadAnyGroup
47  ChimeraTK::ReadAnyGroup group;
48  for(auto& accessor : recursiveAccessorList) {
49  if(accessor.getDirection() == VariableDirection{VariableDirection::feeding, false}) continue;
50  group.add(accessor.getAppAccessorNoType());
51  }
52 
53  group.finalise();
54  return group;
55  }

59  void Module::readAll(bool includeReturnChannels) {
60  auto recursiveAccessorList = getAccessorListRecursive();
61  // first blockingly read all push-type variables
62  for(auto& accessor : recursiveAccessorList) {
63  if(accessor.getMode() != UpdateMode::push) continue;
64  if(includeReturnChannels) {
65  if(accessor.getDirection() == VariableDirection{VariableDirection::feeding, false}) continue;
66  }
67  else {
68  if(accessor.getDirection().dir != VariableDirection::consuming) continue;
69  }
70  accessor.getAppAccessorNoType().read();
71  }
72  // next non-blockingly read the latest values of all poll-type variables
73  for(auto& accessor : recursiveAccessorList) {
74  if(accessor.getMode() == UpdateMode::push) continue;
75  // poll-type accessors cannot have a readback channel
76  if(accessor.getDirection().dir != VariableDirection::consuming) continue;
77  accessor.getAppAccessorNoType().readLatest();
78  }
79  }

152  Module* Module::findApplicationModule() {
153  if(getModuleType() == ModuleType::ApplicationModule) {
154  auto* ret = dynamic_cast<ApplicationModule*>(this);
155  assert(ret != nullptr);
156  return ret;
157  }
158  else if(getModuleType() == ModuleType::Device) {
159  auto* ret = dynamic_cast<DeviceModule*>(this);
160  assert(ret != nullptr);
161  return ret;
162  }
163  else if(getModuleType() == ModuleType::VariableGroup) {
164  auto* owningModule = dynamic_cast<Module*>(getOwner());
165  assert(owningModule != nullptr);
166  return owningModule->findApplicationModule();
167  }
168  else {
169  throw ChimeraTK::logic_error(
170  "EntityOwner::findApplicationModule() called on neither an ApplicationModule nor a VariableGroup.");
171  }
172  }

200  ConfigReader& Module::appConfig() {
201  size_t nConfigReaders = 0;
202  ConfigReader* instance = nullptr;
203  for(auto* mod : Application::getInstance().getSubmoduleListRecursive()) {
204  if(!dynamic_cast<ConfigReader*>(mod)) continue;
205  ++nConfigReaders;
206  instance = dynamic_cast<ConfigReader*>(mod);
207  }
208  if(nConfigReaders != 1) {
209  std::string message = "ApplicationModule::appConfig() called but " + std::to_string(nConfigReaders) +
210  " instances of ChimeraTK::ConfigReader have been found.";
211  // Printing the message as well; there is a situation when running under Boost::Test where this
212  // is caught by Boost and causes a weird destructor message from AppBase.cc instead with no means of
213  // finding out the actual error
214  std::cerr << message << std::endl;
215  throw ChimeraTK::logic_error(message);
216  }
217  return *instance;
218  }
