52  ChimeraTK::ReadAnyGroup Module::readAnyGroup() {
53  auto recursiveAccessorList = getAccessorListRecursive();
54 
55  // put push-type transfer elements into a ReadAnyGroup
56  ChimeraTK::ReadAnyGroup group;
57  for(auto& accessor : recursiveAccessorList) {
58  if(accessor.getDirection() == VariableDirection{VariableDirection::feeding, false}) continue;
59  group.add(accessor.getAppAccessorNoType());
60  }
61 
62  group.finalise();
63  return group;
64  }

68  void Module::readAll(bool includeReturnChannels) {
69  auto recursiveAccessorList = getAccessorListRecursive();
70  // first blockingly read all push-type variables
71  for(auto& accessor : recursiveAccessorList) {
72  if(accessor.getMode() != UpdateMode::push) continue;
73  if(includeReturnChannels) {
74  if(accessor.getDirection() == VariableDirection{VariableDirection::feeding, false}) continue;
75  }
76  else {
77  if(accessor.getDirection().dir != VariableDirection::consuming) continue;
78  }
79  accessor.getAppAccessorNoType().read();
80  }
81  // next non-blockingly read the latest values of all poll-type variables
82  for(auto& accessor : recursiveAccessorList) {
83  if(accessor.getMode() == UpdateMode::push) continue;
84  // poll-type accessors cannot have a readback channel
85  if(accessor.getDirection().dir != VariableDirection::consuming) continue;
86  accessor.getAppAccessorNoType().readLatest();
87  }
88  }

161  const Module& Module::submodule(std::string_view moduleName) const {
162  // strip leading slash if present
163  while(moduleName[0] == '/') moduleName.remove_prefix(1);
164  // an empty string or a "/" refers to the module itself
165  if(moduleName.size() == 0) return *this;
166  // search for first slash
167  size_t slash = moduleName.find_first_of("/");
168  // no slash found: call subscript operator
169  if(slash == std::string::npos) return (*this)[std::string(moduleName)];
170  // slash found: split module name at slash
171  auto upperModuleName = moduleName.substr(0, slash);
172  auto remainingModuleNames = moduleName.substr(slash + 1);
173  return (*this)[std::string(upperModuleName)].submodule(remainingModuleNames);
174  }

185  std::string Module::getVirtualQualifiedName() const {
186  std::string virtualQualifiedName{""};
187  const EntityOwner* currentLevelModule{this};
188 
189  bool rootReached{false};
190  do {
191  if(currentLevelModule == &Application::getInstance()) {
192  break;
193  }
194 
195  auto currentLevelModifier = currentLevelModule->getHierarchyModifier();
196  bool skipNextLevel{false};
197 
198  switch(currentLevelModifier) {
199  case HierarchyModifier::none:
200  virtualQualifiedName = "/" + currentLevelModule->getName() + virtualQualifiedName;
201  break;
202  case HierarchyModifier::hideThis:
203  // Omit name of current level
204  break;
205  case HierarchyModifier::oneLevelUp:
206  virtualQualifiedName = "/" + currentLevelModule->getName() + virtualQualifiedName;
207  skipNextLevel = true;
208  break;
209  case HierarchyModifier::oneUpAndHide:
210  skipNextLevel = true;
211  break;
212  case HierarchyModifier::moveToRoot:
213  virtualQualifiedName = "/" + currentLevelModule->getName() + virtualQualifiedName;
214  rootReached = true;
215  break;
216  }
217 
218  if(skipNextLevel) {
219  auto lastLevelModule = currentLevelModule;
220  currentLevelModule = dynamic_cast<const Module*>(currentLevelModule)->getOwner();
221 
222  if(currentLevelModule == &Application::getInstance()) {
223  throw logic_error(std::string("Module ") + lastLevelModule->getName() +
224  ": cannot have hierarchy modifier 'oneLevelUp' or oneUpAndHide in root of the application.");
225  }
226  }
227 
228  if(!rootReached) {
229  currentLevelModule = dynamic_cast<const Module*>(currentLevelModule)->getOwner();
230  }
231  } while(!rootReached);
232 
233  if(virtualQualifiedName == "") virtualQualifiedName = "/";
234 
235  return virtualQualifiedName;
236  }

240  ApplicationModule* Module::findApplicationModule() {
241  if(getModuleType() == ModuleType::ApplicationModule) {
242  auto* ret = dynamic_cast<ApplicationModule*>(this);
243  assert(ret != nullptr);
244  return ret;
245  }
246  else if(getModuleType() == ModuleType::VariableGroup) {
247  auto* owningModule = dynamic_cast<Module*>(getOwner());
248  assert(owningModule != nullptr);
249  return owningModule->findApplicationModule();
250  }
251  else {
252  throw ChimeraTK::logic_error(
253  "EntityOwner::findApplicationModule() called on neither an ApplicationModule nor a VariableGroup.");
254  }
255  }
