48  auto recursiveAccessorList = getAccessorListRecursive();
49 
50  // put push-type transfer elements into a ReadAnyGroup
51  ChimeraTK::ReadAnyGroup group;

57  group.finalise();

63  void Module::readAll(bool includeReturnChannels) {
64  auto recursiveAccessorList = getAccessorListRecursive();
65  // first blockingly read all push-type variables
66  for(auto accessor : recursiveAccessorList) {
67  if(accessor.getMode() != UpdateMode::push) continue;
68  if(includeReturnChannels) {
69  if(accessor.getDirection() == VariableDirection{VariableDirection::feeding, false}) continue;
70  }
71  else {
72  if(accessor.getDirection().dir != VariableDirection::consuming) continue;
73  }
74  accessor.getAppAccessorNoType().read();
75  }
76  // next non-blockingly read the latest values of all poll-type variables
77  for(auto accessor : recursiveAccessorList) {
78  if(accessor.getMode() == UpdateMode::push) continue;
79  // poll-type accessors cannot have a readback channel
80  if(accessor.getDirection().dir != VariableDirection::consuming) continue;
81  accessor.getAppAccessorNoType().readLatest();
82  }
83  }

156  Module& Module::submodule(const std::string& moduleName) const {
157  size_t slash = moduleName.find_first_of("/");
158  // no slash found: call subscript operator
159  if(slash == std::string::npos) return (*this)[moduleName];
160  // slash found: split module name at slash
161  std::string upperModuleName = moduleName.substr(0, slash);
162  std::string remainingModuleNames = moduleName.substr(slash + 1);
163  return (*this)[upperModuleName].submodule(remainingModuleNames);
164  }

166  std::string Module::getVirtualQualifiedName() const {
167  std::string virtualQualifiedName{""};
168  const EntityOwner* currentLevelModule{this};
169 
170  bool rootReached{false};
171  do {
172  if(currentLevelModule == &Application::getInstance()) {
173  rootReached = true;
174  }
175 
176  auto currentLevelModifier = currentLevelModule->getHierarchyModifier();
177  bool skipNextLevel{false};
178 
179  switch(currentLevelModifier) {
180  case HierarchyModifier::none:
181  virtualQualifiedName = "/" + currentLevelModule->getName() + virtualQualifiedName;
182  break;
183  case HierarchyModifier::hideThis:
184  // Omit name of current level
185  break;
186  case HierarchyModifier::oneLevelUp:
187  virtualQualifiedName = "/" + currentLevelModule->getName() + virtualQualifiedName;
188  skipNextLevel = true;
189  break;
190  case HierarchyModifier::oneUpAndHide:
191  skipNextLevel = true;
192  break;
193  case HierarchyModifier::moveToRoot:
194  virtualQualifiedName =
195  "/" + Application::getInstance().getName() + "/" + currentLevelModule->getName() + virtualQualifiedName;
196  rootReached = true;
197  }
198 
199  if(skipNextLevel) {
200  auto lastLevelModule = currentLevelModule;
201  currentLevelModule = dynamic_cast<const Module*>(currentLevelModule)->getOwner();
202 
203  if(currentLevelModule == &Application::getInstance()) {
204  throw logic_error(std::string("Module ") + lastLevelModule->getName() +
205  ": cannot have hierarchy modifier 'oneLevelUp' or oneUpAndHide in root of the application.");
206  }
207  }
208 
209  if(!rootReached) {
210  currentLevelModule = dynamic_cast<const Module*>(currentLevelModule)->getOwner();
211  }
212  } while(!rootReached);
213 
214  return virtualQualifiedName;
215  }
