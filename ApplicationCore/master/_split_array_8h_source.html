119  template<typename TYPE>
120  void WriteSplitArrayModule<TYPE>::mainLoop() {
121  auto readGroup = readAnyGroup();
122  while(true) {
123  // write the array from the individual elements
124  for(size_t i = 0; i < _nGroups; ++i) {
125  for(size_t k = 0; k < _nElemsPerGroup; ++k) {
126  output[i * _nElemsPerGroup + k] = input[i][k];
127  }
128  }
129  output.write();
130 
131  // wait for new input values (at the end, since we want to process the
132  // initial values first)
133  readGroup.readAny();
134  }
135  }
