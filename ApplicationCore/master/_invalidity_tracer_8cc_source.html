25  void InvalidityTracerVisitor::dispatch(const ApplicationModule& module) {
26  auto validity = module.getDataValidity();
27  // print, if validity is faulty
28  if(validity == DataValidity::faulty) {
29  std::cout << "Module " << module.getQualifiedName() << " has DataValidity::faulty" << std::endl;
30  // dispatch to all accessors of the module
31  for(auto& accessor : module.getAccessorListRecursive()) {
32  dispatch(accessor);
33  }
34  }
35  }

39  void InvalidityTracerVisitor::dispatch(const VariableNetworkNode& node) {
40  // ignore non-application nodes
41  if(node.getType() != NodeType::Application) return;
42 
43  // ignore non-inputs
44  if(node.getDirection().dir != VariableDirection::consuming) return;
45 
46  // get accessor and cast into right type (all application accessors must have the
47  // MetaDataPropagatingRegisterDecorator).
48  auto accessor = boost::dynamic_pointer_cast<MetaDataPropagationFlagProvider>(
49  node.getAppAccessorNoType().getHighLevelImplElement());
50  if(!accessor) {
51  std::cout << "InvalidityTracerVisitor: The following application node does not derive from "
52  "MetaDataPropagationFlagProvider:"
53  << std::endl;
54  node.dump();
55  return;
56  }
57 
58  // check if invalid
59  if(accessor->getLastValidity() == DataValidity::faulty) {
60  std::cout << " -> Input " << node.getQualifiedName() << " has DataValidity::faulty" << std::endl;
61  }
62  }
