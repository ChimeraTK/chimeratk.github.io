130 BOOST_AUTO_TEST_CASE_TEMPLATE(testTriggerDevToApp, T, test_types) {
131  std::cout << "***************************************************************"
132  "******************************************************"
133  << std::endl;
134  std::cout << "==> testTriggerDevToApp<" << typeid(T).name() << ">" << std::endl;
135 
136  ChimeraTK::BackendFactory::getInstance().setDMapFilePath("test.dmap");
137 
138  TestApplication<T> app;
139  auto pvManagers = ctk::createPVManager();
140  app.setPVManager(pvManagers.second);
141 
142  app.testModule.feedingToDevice >> app.dev["MyModule"]("actuator");
143 
144  app.dev["MyModule"]("readBack")[app.testModule.theTrigger] >> app.testModule.consumingPush;
145  app.initialise();
146 
147  app.run();
148  app.testModule.mainLoopStarted.wait(); // make sure the module's mainLoop() is entered
149 
150  // check for initial value. Should be there when entering the main loop.
151  BOOST_CHECK_EQUAL(static_cast<T>(app.testModule.consumingPush), 13);
152 
153  // single theaded test
154  app.testModule.feedingToDevice = 42;
155  app.testModule.feedingToDevice.write();
156  BOOST_CHECK(app.testModule.consumingPush.readNonBlocking() == false);
157  BOOST_CHECK_EQUAL(static_cast<T>(app.testModule.consumingPush), 13);
158  app.testModule.theTrigger.write();
159  app.testModule.consumingPush.read();
160  BOOST_CHECK(app.testModule.consumingPush == 42);
161 
162  // launch read() on the consumer asynchronously and make sure it does not yet
163  // receive anything
164  auto futRead = std::async(std::launch::async, [&app] { app.testModule.consumingPush.read(); });
165  BOOST_CHECK(futRead.wait_for(std::chrono::milliseconds(200)) == std::future_status::timeout);
166 
167  BOOST_CHECK(app.testModule.consumingPush == 42);
168 
169  // write to the feeder
170  app.testModule.feedingToDevice = 120;
171  app.testModule.feedingToDevice.write();
172  BOOST_CHECK(futRead.wait_for(std::chrono::milliseconds(200)) == std::future_status::timeout);
173  BOOST_CHECK(app.testModule.consumingPush == 42);
174 
175  // send trigger
176  app.testModule.theTrigger.write();
177 
178  // check that the consumer now receives the just written value
179  BOOST_CHECK(futRead.wait_for(std::chrono::milliseconds(2000)) == std::future_status::ready);
180  BOOST_CHECK(app.testModule.consumingPush == 120);
181 }
