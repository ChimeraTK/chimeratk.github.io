43  struct SendDataToConsumers {
44  SendDataToConsumers(VersionNumber version, DataValidity triggerValidity)
45  : _version(version), _triggerValidity(triggerValidity) {}
46 
47  template<typename PAIR>
48  void operator()(PAIR& pair) const {
49  auto theMap = pair.second; // map of feeder to FeedingFanOut (i.e. part of
50  // the fanOutMap)
51 
52  // iterate over all feeder/FeedingFanOut pairs
53  for(auto& network : theMap) {
54  auto feeder = network.first;
55  auto fanOut = network.second;
56  fanOut->setDataValidity((_triggerValidity == DataValidity::ok && feeder->dataValidity() == DataValidity::ok) ?
57  DataValidity::ok :
58  DataValidity::faulty);
59  fanOut->accessChannel(0).swap(feeder->accessChannel(0));
60  // don't use write destructively. In case of an exception we still need the data for the next read (see
61  // Exception Handling spec B.2.2.6)
62  bool dataLoss = fanOut->write(_version);
63  if(dataLoss) Application::incrementDataLossCounter(fanOut->getName());
64  // swap the data back to the feeder so we have a valid copy there.
65  fanOut->accessChannel(0).swap(feeder->accessChannel(0));
66  }
67  }
68 
69  VersionNumber _version;
70  DataValidity _triggerValidity;
71  };
72  } // namespace

76  void TriggerFanOut::run() {
77  Application::registerThread("TrFO" + externalTrigger->getName());
78  Application::testableModeLock("start");
79  testableModeReached = true;
80 
81  ChimeraTK::VersionNumber version = Application::getInstance().getStartVersion();
82 
83  // Wait for the initial value of the trigger. There always will be one, and if we don't read it here we would
84  // trigger the loop twice.
85  externalTrigger->read();
86  version = externalTrigger->getVersionNumber();
87 
88  // Wait until the device has been initialised for the first time. This means it
89  // has been opened, and the check in TransferGroup::read() will not throw a logic_error
90  // We don't have to store the lock. Just need it as a synchronisation point.
91  // But we have to increase the testable mode counter because we don't want to fall out of testable mode at this
92  // point already.
93  if(Application::getInstance().testableMode) ++Application::getInstance().testableMode_deviceInitialisationCounter;
94  Application::testableModeUnlock("WaitInitialValueLock");
95  (void)_deviceModule.waitForInitialValues();
96  Application::testableModeLock("Enter while loop");
97  if(Application::getInstance().testableMode) --Application::getInstance().testableMode_deviceInitialisationCounter;
98 
99  while(true) {
100  transferGroup.read();
101  // send the version number to the consumers
102  boost::fusion::for_each(fanOutMap.table, SendDataToConsumers(version, externalTrigger->dataValidity()));
103 
104  // wait for external trigger
105  boost::this_thread::interruption_point();
106  externalTrigger->read();
107  boost::this_thread::interruption_point();
108  version = externalTrigger->getVersionNumber();
109  }
110  }
