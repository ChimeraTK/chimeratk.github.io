45  struct SendDataToConsumers {
46  SendDataToConsumers(VersionNumber version, DataValidity triggerValidity)
47  : _version(version), _triggerValidity(triggerValidity) {}
48 
49  template<typename PAIR>
50  void operator()(PAIR& pair) const {
51  auto theMap = pair.second; // map of feeder to FeedingFanOut (i.e. part of
52  // the fanOutMap)
53 
54  // iterate over all feeder/FeedingFanOut pairs
55  for(auto& network : theMap) {
56  auto feeder = network.first;
57  auto fanOut = network.second;
58  fanOut->setDataValidity((_triggerValidity == DataValidity::ok && feeder->dataValidity() == DataValidity::ok) ?
59  DataValidity::ok :
60  DataValidity::faulty);
61  fanOut->accessChannel(0).swap(feeder->accessChannel(0));
62  // don't use write destructively. In case of an exception we still need the data for the next read (see
63  // Exception Handling spec B.2.2.6)
64  bool dataLoss = fanOut->write(_version);
65  if(dataLoss) Application::incrementDataLossCounter(fanOut->getName());
66  // swap the data back to the feeder so we have a valid copy there.
67  fanOut->accessChannel(0).swap(feeder->accessChannel(0));
68  }
69  }
70 
71  VersionNumber _version;
72  DataValidity _triggerValidity;
73  };

78  void TriggerFanOut::run() {
79  Application::registerThread("TrFO" + externalTrigger->getName());
80  Application::getInstance().getTestableMode().lock("start");
81  _testableModeReached = true;
82 
83  ChimeraTK::VersionNumber version = Application::getInstance().getStartVersion();
84 
85  // Wait for the initial value of the trigger. There always will be one, and if we don't read it here we would
86  // trigger the loop twice.
87  externalTrigger->read();
88  version = externalTrigger->getVersionNumber();
89 
90  // Wait until the device has been initialised for the first time. This means it
91  // has been opened, and the check in TransferGroup::read() will not throw a logic_error
92  // We don't have to store the lock. Just need it as a synchronisation point.
93  // But we have to increase the testable mode counter because we don't want to fall out of testable mode at this
94  // point already.
95  if(Application::getInstance().getTestableMode().isEnabled()) {
96  ++Application::getInstance().getTestableMode().deviceInitialisationCounter;
97  }
98  Application::getInstance().getTestableMode().unlock("WaitInitialValueLock");
99  (void)_deviceModule.waitForInitialValues();
100  Application::getInstance().getTestableMode().lock("Enter while loop");
101  if(Application::getInstance().getTestableMode().isEnabled()) {
102  --Application::getInstance().getTestableMode().deviceInitialisationCounter;
103  }
104 
105  while(true) {
106  transferGroup.read();
107  // send the version number to the consumers
108  boost::fusion::for_each(fanOutMap.table, SendDataToConsumers(version, externalTrigger->dataValidity()));
109 
110  // wait for external trigger
111  boost::this_thread::interruption_point();
112  externalTrigger->read();
113  boost::this_thread::interruption_point();
114  version = externalTrigger->getVersionNumber();
115  }
116  }
