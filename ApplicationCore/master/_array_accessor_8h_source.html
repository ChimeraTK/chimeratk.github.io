164  template<typename UserType>
165  void ArrayAccessor<UserType>::writeIfDifferent(const std::vector<UserType>& newValue) {
166  // Need to get to the MetaDataPropagatingRegisterDecorator to obtain the last written data validity for this PV.
167  // The dynamic_cast is ok, since the MetaDataPropagatingRegisterDecorator is always the outermost accessor, cf.
168  // the data validity propagation specification, Section 2.5.1.
169  auto* targetMetaDataPropagatingDecorator =
170  dynamic_cast<MetaDataPropagatingRegisterDecorator<UserType>*>(this->get());
171  assert(targetMetaDataPropagatingDecorator != nullptr);
172 
173  // In contrast to OneDRegisterAccessor::writeIfDifferent(), we must not set the data validity on the target
174  // accessor, since that would be interpreted by the MetaDataPropagatingRegisterDecorator as an application-induced
175  // forced fault state. This would result in invalidity lock-ups if this happens in a circular network. Hence the
176  // comparison of the data validity must also be done against the validity of the decorator's target accessor which
177  // corresponds to the last written data validity for this PV.
178  if(!std::equal(newValue.begin(), newValue.end(), this->get()->accessChannel(0).begin()) ||
179  this->getVersionNumber() == VersionNumber(nullptr) ||
180  targetMetaDataPropagatingDecorator->getTargetValidity() != this->getOwner()->getDataValidity()) {
181  setAndWrite(newValue);
182  }
183  }
