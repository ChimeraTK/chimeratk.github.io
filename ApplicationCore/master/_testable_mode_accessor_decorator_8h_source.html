53  template<typename UserType>
54  TestableModeAccessorDecorator<UserType>::TestableModeAccessorDecorator(
55  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> accessor, bool handleRead, bool handleWrite,
56  size_t variableIdRead, size_t variableIdWrite)
57  : ChimeraTK::NDRegisterAccessorDecorator<UserType>(accessor), _handleRead(handleRead), _handleWrite(handleWrite),
58  _variableIdRead(variableIdRead), _variableIdWrite(variableIdWrite) {
59  assert(_variableIdRead != 0);
60  assert(_variableIdWrite != 0);
61 
62  // if receiving end, register for testable mode (stall detection)
63  if(this->isReadable() && handleRead) {
64  Application::getInstance().testableMode_processVars[_variableIdRead] = accessor;
65  assert(accessor->getAccessModeFlags().has(AccessMode::wait_for_new_data));
66  }
67 
68  // if this decorating a bidirectional process variable, set the
69  // valueRejectCallback
70  auto bidir = boost::dynamic_pointer_cast<BidirectionalProcessArray<UserType>>(accessor);
71  if(bidir) {
72  bidir->setValueRejectCallback([this] { decrementCounter(); });
73  }
74  else {
75  assert(!(handleRead && handleWrite));
76  }
77  }

81  template<typename UserType>
82  bool TestableModeAccessorDecorator<UserType>::doWriteTransfer(ChimeraTK::VersionNumber versionNumber) {
83  if(!_handleWrite) return _target->writeTransfer(versionNumber);
84 
85  bool dataLost = false;
86  if(!Application::testableModeTestLock()) {
87  // may happen if first write in thread is done before first blocking read
88  Application::testableModeLock("write " + this->getName());
89  }
90  dataLost = _target->writeTransfer(versionNumber);
91  if(!dataLost) {
92  ++Application::getInstance().testableMode_counter;
93  ++Application::getInstance().testableMode_perVarCounter[_variableIdWrite];
94  if(Application::getInstance().enableDebugTestableMode) {
95  std::cout << "TestableModeAccessorDecorator::write[name='" << this->getName() << "', id=" << _variableIdWrite
96  << "]: testableMode_counter increased, now at value "
97  << Application::getInstance().testableMode_counter << std::endl;
98  }
99  }
100  else {
101  if(Application::getInstance().enableDebugTestableMode) {
102  std::cout << "TestableModeAccessorDecorator::write[name='" << this->getName() << "', id=" << _variableIdWrite
103  << "]: testableMode_counter not increased due to lost data" << std::endl;
104  }
105  }
106  return dataLost;
107  }

161  template<typename UserType>
162  void TestableModeAccessorDecorator<UserType>::obtainLockAndDecrementCounter(bool hasNewData) {
163  if(!Application::testableModeTestLock()) Application::testableModeLock("doReadTransfer " + this->getName());
164  if(!hasNewData) return;
165  if(Application::getInstance().testableMode_perVarCounter[_variableIdRead] > 0) {
166  assert(Application::getInstance().testableMode_counter > 0);
167  --Application::getInstance().testableMode_counter;
168  --Application::getInstance().testableMode_perVarCounter[_variableIdRead];
169  if(Application::getInstance().enableDebugTestableMode) {
170  std::cout << "TestableModeAccessorDecorator[name='" << this->getName() << "', id=" << _variableIdRead
171  << "]: testableMode_counter decreased, now at value "
172  << Application::getInstance().testableMode_counter << " / "
173  << Application::getInstance().testableMode_perVarCounter[_variableIdRead] << std::endl;
174  }
175  }
176  else {
177  if(Application::getInstance().enableDebugTestableMode) {
178  std::cout << "TestableModeAccessorDecorator[name='" << this->getName() << "', id=" << _variableIdRead
179  << "]: testableMode_counter NOT decreased, was already at value "
180  << Application::getInstance().testableMode_counter << " / "
181  << Application::getInstance().testableMode_perVarCounter[_variableIdRead] << std::endl;
182  std::cout << Application::getInstance().testableMode_names[_variableIdRead] << std::endl;
183  }
