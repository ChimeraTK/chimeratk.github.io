35 static bool deviceIsInitialised(std::string alias, boost::shared_ptr<ctk::ControlSystemPVManager> csPVManager) {
36  auto dummyDeviceStatus = csPVManager->getProcessArray<int>("/Devices/" + alias + "/status");

84 void testDirectRegister(ctk::TestFacility& test, ChimeraTK::ScalarRegisterAccessor<T> sender,
85  ChimeraTK::ScalarRegisterAccessor<T> receiver, LAMBDA trigger, bool testMinMax = true) {
86  std::cout << "testDirectRegister<" << typeid(T).name() << ">: " << sender.getName() << " -> " << receiver.getName()
87  << std::endl;
88 
89  sender = 42;
90  sender.write();
91  trigger();
92  test.stepApplication();
93  receiver.read();
94  BOOST_CHECK_EQUAL(receiver, 42);
95 
96  if(std::numeric_limits<T>::is_signed) {
97  sender = -120;
98  sender.write();
99  trigger();
100  test.stepApplication();
101  receiver.read();
102  BOOST_CHECK_EQUAL(receiver, -120);
103  }
104 
105  if(testMinMax) {
106  sender = std::numeric_limits<T>::max();
107  sender.write();
108  trigger();
109  test.stepApplication();
110  receiver.read();
111  BOOST_CHECK_EQUAL(receiver, std::numeric_limits<T>::max());
112 
113  sender = std::numeric_limits<T>::min();
114  sender.write();
115  trigger();
116  test.stepApplication();
117  receiver.read();
118  BOOST_CHECK_EQUAL(receiver, std::numeric_limits<T>::min());
119 
120  sender = std::numeric_limits<T>::epsilon();
121  sender.write();
122  trigger();
123  test.stepApplication();
124  receiver.read();
125  BOOST_CHECK_EQUAL(receiver, std::numeric_limits<T>::epsilon());
126  }
