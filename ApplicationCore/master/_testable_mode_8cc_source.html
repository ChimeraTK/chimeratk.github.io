58  void TestableMode::lock(const std::string& name) {
59  // don't do anything if testable mode is not enabled
60  if(not _enabled) {
61  return;
62  }
63 
64  // debug output if enabled (also prevent spamming the same message)
65  if(_enableDebug && _repeatingMutexOwner == 0) { // LCOV_EXCL_LINE
66  // (only cout)
67  std::cout << "Application::testableModeLock(): Thread " << threadName() // LCOV_EXCL_LINE (only cout)
68  << " tries to obtain lock for " << name << std::endl; // LCOV_EXCL_LINE (only cout)
69  } // LCOV_EXCL_LINE (only cout)
70 
71  // if last lock was obtained repeatedly by the same thread, sleep a short time
72  // before obtaining the lock to give the other threads a chance to get the
73  // lock first
74  if(_repeatingMutexOwner > 0) {
75  usleep(10000);
76  }
77 
78  // obtain the lock
79  auto lastSeen_lastOwner = _lastMutexOwner;
80  repeatTryLock:
81  auto success = getLockObject().try_lock_for(std::chrono::seconds(300));
82  if(!success) {
83  auto currentLastOwner = _lastMutexOwner;
84  if(currentLastOwner != lastSeen_lastOwner) {
85  lastSeen_lastOwner = currentLastOwner;
86  goto repeatTryLock;
87  }
88  std::cout << "testableModeLock(): Thread " << threadName() // LCOV_EXCL_LINE
89  << " could not obtain lock for 30 seconds, presumably because " // LCOV_EXCL_LINE
90  << threadName(_lastMutexOwner) << " does not release it." // LCOV_EXCL_LINE
91  << std::endl; // LCOV_EXCL_LINE
92 
93  // throw a specialised exception to make sure whoever catches it really knows what he does...
94  terminateTestStalled(); // LCOV_EXCL_LINE
95  } // LCOV_EXCL_LINE
96 
97  // check if the last owner of the mutex was this thread, which may be a hint
98  // that no other thread is waiting for the lock
99  if(_lastMutexOwner == boost::this_thread::get_id()) {
100  // debug output if enabled
101  if(_enableDebug && _repeatingMutexOwner == 0) { // LCOV_EXCL_LINE
102  // (only cout)
103  std::cout << "testableModeLock(): Thread " << threadName() // LCOV_EXCL_LINE (only cout)
104  << " repeatedly obtained lock successfully for " << name // LCOV_EXCL_LINE (only cout)
105  << ". Further messages will be suppressed." << std::endl; // LCOV_EXCL_LINE (only cout)
106  } // LCOV_EXCL_LINE (only cout)
107 
108  // increase counter for stall detection
109  _repeatingMutexOwner++;
110 
111  // detect stall: if the same thread got the mutex with no other thread
112  // obtaining it in between for one second, we assume no other thread is able
113  // to process data at this time. The test should fail in this case
114  if(_repeatingMutexOwner > 100) {
115  // print an informative message first, which lists also all variables
116  // currently containing unread data.
117  std::cerr << "*** Tests are stalled due to data which has been sent but not received." << std::endl;
118  std::cerr << " The following variables still contain unread values or had data loss due to a queue overflow:"
119  << std::endl;
120  for(auto& pair : _variables) {
121  const auto& variable = pair.second;
122  if(variable.counter > 0) {
123  std::cerr << " - " << variable.name << " [" << variable.processVariable->getId() << "]";
124  // check if process variable still has data in the queue
125  try {
126  if(variable.processVariable->readNonBlocking()) {
127  std::cerr << " (unread data in queue)";
128  }
129  else {
130  std::cerr << " (data loss)";
131  }
132  }
133  catch(std::logic_error&) {
134  // if we receive a logic_error in readNonBlocking() it just means
135  // another thread is waiting on a TransferFuture of this variable,
136  // and we actually were not allowed to read...
137  std::cerr << " (data loss)";
138  }
139  std::cerr << std::endl;
140  }
141  }
142  std::cerr << "(end of list)" << std::endl;
143  // Check for modules waiting for initial values (prints nothing if there are no such modules)
144  Application::getInstance()._circularDependencyDetector.printWaiters();
145  // throw a specialised exception to make sure whoever catches it really knows what he does...
146  terminateTestStalled();
147  }
148  }
149  else {
150  // last owner of the mutex was different: reset the counter and store the
151  // thread id
152  _repeatingMutexOwner = 0;
153  _lastMutexOwner = boost::this_thread::get_id();
154 
155  // debug output if enabled
156  if(_enableDebug) { // LCOV_EXCL_LINE (only cout)
157  std::cout << "TestableMode::lock(): Thread " << threadName() // LCOV_EXCL_LINE (only cout)
158  << " obtained lock successfully for " << name << std::endl; // LCOV_EXCL_LINE (only cout)
159  } // LCOV_EXCL_LINE (only cout)
160  }
161  }

165  void TestableMode::unlock(const std::string& name) {
166  if(not _enabled) {
167  return;
168  }
169  if(_enableDebug &&
170  (not _repeatingMutexOwner // LCOV_EXCL_LINE (only cout)
171  || _lastMutexOwner != boost::this_thread::get_id())) { // LCOV_EXCL_LINE (only cout)
172  std::cout << "TestableMode::unlock(): Thread " << threadName() // LCOV_EXCL_LINE (only cout)
173  << " releases lock for " << name << std::endl; // LCOV_EXCL_LINE (only cout)
174  }
175  // LCOV_EXCL_LINE (only cout)
176  getLockObject().unlock();
177  }

181  void TestableMode::step(bool waitForDeviceInitialisation) {
182  // testableMode.counter must be non-zero, otherwise there is no input for the application to process. It is also
183  // sufficient if testableMode.deviceInitialisationCounter is non-zero, if waitForDeviceInitialisation == true. In
184  // that case we only wait for the device initialisation to be completed.
185  if(_counter == 0 && (!waitForDeviceInitialisation || _deviceInitialisationCounter == 0)) {
186  throw ChimeraTK::logic_error("Application::stepApplication() called despite no input was provided "
187  "to the application to process!");
188  }
189  // let the application run until it has processed all data (i.e. the semaphore
190  // counter is 0)
191  size_t oldCounter = 0;
192  while(_counter > 0 || (waitForDeviceInitialisation && _deviceInitialisationCounter > 0)) {
193  if(_enableDebug && (oldCounter != _counter)) { // LCOV_EXCL_LINE (only cout)
194  std::cout << "Application::stepApplication(): testableMode.counter = " << _counter
195  << std::endl; // LCOV_EXCL_LINE (only cout)
196  oldCounter = _counter; // LCOV_EXCL_LINE (only cout)
197  }
198  unlock("stepApplication");
199  boost::this_thread::yield();
200  lock("stepApplication");
201  }
202  }
