205  template<typename T>
206  void TestFacility::setArrayDefault(const ChimeraTK::RegisterPath& name, const std::vector<T>& value) {
207  if(_app._testFacilityRunApplicationCalled) {
208  throw ChimeraTK::logic_error("TestFacility::setArrayDefault() called after runApplication().");
209  }
210  // check if PV exists
211  if(!_pvManager->hasProcessVariable(name)) {
212  throw ChimeraTK::logic_error("Process variable '" + name + "' does not exist.");
213  }
214 
215  // check if the type is right
216  auto pv = _pvManager->getProcessArray<typename detail::BoolTypeHelper<T>::type>(name);
217  if(pv == nullptr) {
218  auto pvUntyped = _pvManager->getProcessVariable(name);
219  throw ChimeraTK::logic_error("Process variable '" + name + "' requested by the wrong type: " + typeid(T).name() +
220  " != " + pvUntyped->getValueType().name());
221  }
222 
223  // store default value in map
224  auto& tv = boost::fusion::at_key<typename detail::BoolTypeHelper<T>::type>(_defaults.table)[name];
225  if constexpr(!std::is_same<T, bool>::value) {
226  tv = value;
227  }
228  else {
229  tv.resize(value.size());
230  std::transform(value.begin(), value.end(), tv.begin(), [](const bool& v) -> ChimeraTK::Boolean { return v; });
231  }
232  }

236  template<typename T>
237  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<T>> TestFacility::getAccessor(
238  const ChimeraTK::RegisterPath& name) const {
239  // check for existing accessor in cache
240  if(boost::fusion::at_key<T>(_accessorMap.table).count(name) > 0) {
241  return boost::fusion::at_key<T>(_accessorMap.table)[name];
242  }
243 
244  // obtain accessor from ControlSystemPVManager
245  auto pv = _pvManager->getProcessArray<T>(name);
246  if(pv == nullptr) {
247  throw ChimeraTK::logic_error("Process variable '" + name + "' does not exist.");
248  }
249 
250  // decorate with TestableModeAccessorDecorator if variable is sender and
251  // receiver is not poll-type (then no entry in pvIdMapp exists), and store it in cache
252  if(auto it = _app._pvIdMap.find(pv->getUniqueId()); pv->isWriteable() && it != _app._pvIdMap.end()) {
253  boost::fusion::at_key<T>(_accessorMap.table)[name] = _app.getTestableMode().decorate<T>(
254  pv, detail::TestableMode::DecoratorType::WRITE, "ControlSystem:" + name, it->second);
255  }
256  else {
257  boost::fusion::at_key<T>(_accessorMap.table)[name] = pv;
258  }
259 
260  // return the accessor as stored in the cache
261  return boost::fusion::at_key<T>(_accessorMap.table)[name];
262  }
