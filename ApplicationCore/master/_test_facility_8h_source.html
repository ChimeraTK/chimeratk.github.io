39  void runApplication() const {
40  Application::getInstance().testFacilityRunApplicationCalled = true;
41  // send default values for all control system variables
42  for(auto& pv : pvManager->getAllProcessVariables()) {
43  callForTypeNoVoid(pv->getValueType(), [&pv, this](auto arg) {
44  // Applies only to writeable variables. @todo FIXME It should also NOT apply for application-to-controlsystem
45  // variables with a return channel, despite being writeable here!
46  if(!pv->isWriteable()) return;
47  // Safety check against incorrect usage
48  if(pv->getVersionNumber() != VersionNumber(nullptr)) {
49  throw ChimeraTK::logic_error("The variable '" + pv->getName() +
50  "' has been written before TestFacility::runApplication() was called. Instead use "
51  "TestFacility::setScalarDefault() resp. setArrayDefault() to set initial values.");
52  }
53  typedef decltype(arg) T;
54  auto pv_casted = boost::dynamic_pointer_cast<NDRegisterAccessor<T>>(pv);
55  auto table = boost::fusion::at_key<T>(defaults.table);
56  // If default value has been stored, copy the default value to the PV.
57  if(table.find(pv->getName()) != table.end()) {
62  if(pv_casted->getNumberOfSamples() == 1) { // scalar
63  auto accessor = this->getScalar<T>(pv->getName());
64  accessor = table.at(pv->getName())[0];
65  }
66  else { // array
67  auto accessor = this->getArray<T>(pv->getName());
68  accessor = table.at(pv->getName());
69  }
70  // copy value also to undecorated PV
71  pv_casted->accessChannel(0) = table.at(pv->getName());
72  }
73  // Write the initial value. This must be done even if no default value has been stored, since it is expected
74  // by the application.
75  pv_casted->write();
76  });
77  }
78  // start the application
79  Application::getInstance().run();
80  // set thread name
81  Application::registerThread("TestThread");
82  // make sure all initial values have been propagated when in testable mode
83  if(Application::getInstance().isTestableModeEnabled()) {
84  // call stepApplication() only in testable mode and only if the queues are not empty
85  if(Application::getInstance().testableMode_counter != 0 ||
86  Application::getInstance().testableMode_deviceInitialisationCounter != 0) {
87  stepApplication();
88  }
89  }
90 
91  // receive all initial values for the control system variables
92  if(Application::getInstance().isTestableModeEnabled()) {
93  for(auto& pv : pvManager->getAllProcessVariables()) {
94  if(!pv->isReadable()) continue;
95  callForTypeNoVoid(pv->getValueType(), [&](auto t) {
96  typedef decltype(t) UserType;
97  this->getArray<UserType>(pv->getName()).readNonBlocking();
98  });
99  }
100  }
101  }

121  template<typename T>
122  ChimeraTK::ScalarRegisterAccessor<T> getScalar(const ChimeraTK::RegisterPath& name) const {
123  // check for existing accessor in cache
124  if(boost::fusion::at_key<T>(accessorMap.table).count(name) > 0) {
125  return boost::fusion::at_key<T>(accessorMap.table)[name];
126  }
127 
128  // obtain accessor from ControlSystemPVManager
129  auto pv = pvManager->getProcessArray<T>(name);
130  if(pv == nullptr) {
131  throw ChimeraTK::logic_error("Process variable '" + name + "' does not exist.");
132  }
133 
134  // obtain variable id from pvIdMap and transfer it to idMap (required by the
135  // TestableModeAccessorDecorator)
136  size_t varId = Application::getInstance().pvIdMap[pv->getUniqueId()];
137 
138  // decorate with TestableModeAccessorDecorator if variable is sender and
139  // receiver is not poll-type, and store it in cache
140  if(pv->isWriteable() && !Application::getInstance().testableMode_isPollMode[varId]) {
141  auto deco = boost::make_shared<TestableModeAccessorDecorator<T>>(pv, false, true, varId, varId);
142  Application::getInstance().testableMode_names[varId] = "ControlSystem:" + name;
143  boost::fusion::at_key<T>(accessorMap.table)[name] = deco;
144  }
145  else {
146  boost::fusion::at_key<T>(accessorMap.table)[name] = pv;
147  }
148 
149  // return the accessor as stored in the cache
150  return boost::fusion::at_key<T>(accessorMap.table)[name];
151  }

243  throw ChimeraTK::logic_error("Process variable '" + name + "' does not exist.");

245  // store default value in map

252  // Cache (possible decorated) accessors to avoid the need to create accessors multiple times. This would not work
253  // if the accessor is decorated, since the buffer would be lost and thus the current value could no longer be
254  // obtained. This has to be done in dependence of the user type. Since this is a cache and does not change the
255  // logical behaviour of the class, the maps are defined mutable.
256  template<typename UserType>
257  using AccessorMap = std::map<std::string, boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>>>;
258  mutable ChimeraTK::TemplateUserTypeMap<AccessorMap> accessorMap;
259 
260  // default values for process variables
261  template<typename UserType>
262  using Defaults = std::map<std::string, std::vector<UserType>>;
263  ChimeraTK::TemplateUserTypeMap<Defaults> defaults;
