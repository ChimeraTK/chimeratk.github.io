99 // Helper function for testXmlGeneration:
100 // Obtain a value from an XML node
101 std::string getValueFromNode(const xmlpp::Node* node, const std::string& subnodeName) {
102  xmlpp::Node* theChild = nullptr;
103  for(const auto& child : node->get_children()) {
104  if(child->get_name() == subnodeName) theChild = child;
105  }
106  BOOST_REQUIRE(theChild != nullptr); // requested child tag is there
107  auto subChildList = theChild->get_children();
108  if(subChildList.size() == 0) { // special case: no text in the tag -> return empty string
109  return "";
110  }
111  BOOST_REQUIRE_EQUAL(subChildList.size(),
112  1); // child tag contains only text (no further sub-tags)
113  const xmlpp::TextNode* textNode = dynamic_cast<xmlpp::TextNode*>(subChildList.front());
114  BOOST_REQUIRE(textNode != nullptr);
115  return textNode->get_content();
116 }
