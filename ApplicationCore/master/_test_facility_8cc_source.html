23  void TestFacility::runApplication() const {
24  app.testFacilityRunApplicationCalled = true;
25 
26  // send default values for all control system variables
27  for(auto& pv : pvManager->getAllProcessVariables()) {
28  callForType(pv->getValueType(), [&pv, this](auto arg) {
29  using T = decltype(arg);
30 
31  // Applies only to writeable variables.
32  // @todo FIXME It should also NOT apply for application-to-controlsystem variables with a return channel,
33  // despite being writeable here!
34  if(!pv->isWriteable()) return;
35 
36  // Safety check against incorrect usage
37  if(pv->getVersionNumber() != VersionNumber(nullptr)) {
38  throw ChimeraTK::logic_error("The variable '" + pv->getName() +
39  "' has been written before TestFacility::runApplication() was called. Instead use "
40  "TestFacility::setScalarDefault() resp. setArrayDefault() to set initial values.");
41  }
42 
43  // Get the PV accessor
44  auto pv_casted = boost::dynamic_pointer_cast<NDRegisterAccessor<T>>(pv);
45 
46  // If default value has been stored, copy the default value to the PV.
47  if constexpr(!std::is_same<T, ChimeraTK::Void>::value) {
48  auto table = boost::fusion::at_key<T>(defaults.table);
49  if(table.find(pv->getName()) != table.end()) {
54  if(pv_casted->getNumberOfSamples() == 1) { // scalar
55  auto accessor = this->getScalar<T>(pv->getName());
56  accessor = table.at(pv->getName())[0];
57  }
58  else { // array
59  auto accessor = this->getArray<T>(pv->getName());
60  accessor = table.at(pv->getName());
61  }
62  // copy value also to undecorated PV
63  pv_casted->accessChannel(0) = table.at(pv->getName());
64  }
65  }
66 
67  // Write the initial value. This must be done even if no default value has been stored, since it is expected
68  // by the application.
69  pv_casted->write();
70  });
71  }
72  // start the application
73  app.run();
74  // set thread name
75  Application::registerThread("TestThread");
76  // make sure all initial values have been propagated when in testable mode
77  if(app.getTestableMode().enabled) {
78  // call stepApplication() only in testable mode and only if the queues are not empty
79  if(app.getTestableMode().counter != 0 || app.getTestableMode().deviceInitialisationCounter != 0) {
80  stepApplication();
81  }
82  }
83 
84  // receive all initial values for the control system variables
85  if(app.getTestableMode().enabled) {
86  for(auto& pv : pvManager->getAllProcessVariables()) {
87  if(!pv->isReadable()) continue;
88  callForTypeNoVoid(pv->getValueType(), [&](auto t) {
89  using UserType = decltype(t);
90  this->getArray<UserType>(pv->getName()).readNonBlocking();
91  });
92  }
93  }
94  }
