20  void TestFacility::runApplication() const {
21  Application::getInstance().testFacilityRunApplicationCalled = true;
22  // send default values for all control system variables
23  for(auto& pv : pvManager->getAllProcessVariables()) {
24  callForTypeNoVoid(pv->getValueType(), [&pv, this](auto arg) {
25  // Applies only to writeable variables. @todo FIXME It should also NOT apply for application-to-controlsystem
26  // variables with a return channel, despite being writeable here!
27  if(!pv->isWriteable()) return;
28  // Safety check against incorrect usage
29  if(pv->getVersionNumber() != VersionNumber(nullptr)) {
30  throw ChimeraTK::logic_error("The variable '" + pv->getName() +
31  "' has been written before TestFacility::runApplication() was called. Instead use "
32  "TestFacility::setScalarDefault() resp. setArrayDefault() to set initial values.");
33  }
34  typedef decltype(arg) T;
35  auto pv_casted = boost::dynamic_pointer_cast<NDRegisterAccessor<T>>(pv);
36  auto table = boost::fusion::at_key<T>(defaults.table);
37  // If default value has been stored, copy the default value to the PV.
38  if(table.find(pv->getName()) != table.end()) {
43  if(pv_casted->getNumberOfSamples() == 1) { // scalar
44  auto accessor = this->getScalar<T>(pv->getName());
45  accessor = table.at(pv->getName())[0];
46  }
47  else { // array
48  auto accessor = this->getArray<T>(pv->getName());
49  accessor = table.at(pv->getName());
50  }
51  // copy value also to undecorated PV
52  pv_casted->accessChannel(0) = table.at(pv->getName());
53  }
54  // Write the initial value. This must be done even if no default value has been stored, since it is expected
55  // by the application.
56  pv_casted->write();
57  });
58  }
59  // start the application
60  Application::getInstance().run();
61  // set thread name
62  Application::registerThread("TestThread");
63  // make sure all initial values have been propagated when in testable mode
64  if(Application::getInstance().isTestableModeEnabled()) {
65  // call stepApplication() only in testable mode and only if the queues are not empty
66  if(Application::getInstance().testableMode_counter != 0 ||
67  Application::getInstance().testableMode_deviceInitialisationCounter != 0) {
68  stepApplication();
69  }
70  }
71 
72  // receive all initial values for the control system variables
73  if(Application::getInstance().isTestableModeEnabled()) {
74  for(auto& pv : pvManager->getAllProcessVariables()) {
75  if(!pv->isReadable()) continue;
76  callForTypeNoVoid(pv->getValueType(), [&](auto t) {
77  typedef decltype(t) UserType;
78  this->getArray<UserType>(pv->getName()).readNonBlocking();
79  });
80  }
81  }
82  }

98  ChimeraTK::VoidRegisterAccessor TestFacility::getVoid(const ChimeraTK::RegisterPath& name) const {
99  // check for existing accessor in cache
100  if(boost::fusion::at_key<ChimeraTK::Void>(accessorMap.table).count(name) > 0) {
101  return boost::fusion::at_key<ChimeraTK::Void>(accessorMap.table)[name];
102  }
103 
104  // obtain accessor from ControlSystemPVManager
105  auto pv = pvManager->getProcessArray<ChimeraTK::Void>(name);
106  if(pv == nullptr) {
107  throw ChimeraTK::logic_error("Process variable '" + name + "' does not exist.");
108  }
109 
110  // obtain variable id from pvIdMap and transfer it to idMap (required by the
111  // TestableModeAccessorDecorator)
112  size_t varId = Application::getInstance().pvIdMap[pv->getUniqueId()];
113 
114  // decorate with TestableModeAccessorDecorator if variable is sender and
115  // receiver is not poll-type, and store it in cache
116  if(pv->isWriteable() && !Application::getInstance().testableMode_isPollMode[varId]) {
117  auto deco = boost::make_shared<TestableModeAccessorDecorator<ChimeraTK::Void>>(pv, false, true, varId, varId);
118  Application::getInstance().testableMode_names[varId] = "ControlSystem:" + name;
119  boost::fusion::at_key<ChimeraTK::Void>(accessorMap.table)[name] = deco;
120  }
121  else {
122  boost::fusion::at_key<ChimeraTK::Void>(accessorMap.table)[name] = pv;
123  }
124 
125  // return the accessor as stored in the cache
126  return boost::fusion::at_key<ChimeraTK::Void>(accessorMap.table)[name];
127  }
