31  void VariableNetwork::addNode(VariableNetworkNode& a) {
32  assert(!a.hasOwner());
33 
34  // change owner of the node: erase from Application's unconnectedNodeList and
35  // set this as owner
36  a.setOwner(this);
37 
38  // if node is feeding, save as feeder for this network
39  if(a.getDirection().dir == VariableDirection::feeding) {
40  // make sure we only have one feeding node per network
41  if(hasFeedingNode()) {
42  // check if current feeding node is a control system variable: if yes,
43  // switch it to consuming
44  if(getFeedingNode().getType() == NodeType::ControlSystem) {
45  getFeedingNode().setDirection({VariableDirection::consuming, false});
46  }
47  // Current feeder cannot be switch to consumer: throw exception
48  else {
49  std::stringstream msg;
50  msg << "Trying to add a feeding accessor to a network already having a "
51  "feeding accessor."
52  << std::endl;
53  msg << "The network you were trying to add the new accessor to:" << std::endl;
54  dump("", msg);
55  msg << "The node you were trying to add:" << std::endl;
56  a.dump(msg);
57  throw ChimeraTK::logic_error(msg.str());
58  }
59  }
60  // force value type, engineering unit and description of the network if set
61  // in this feeding node
62  if(a.getValueType() != typeid(AnyType)) valueType = &(a.getValueType());
63  if(a.getUnit() != ChimeraTK::TransferElement::unitNotSet) engineeringUnit = a.getUnit();
64  if(a.getDescription() != "") description = a.getDescription();
65  }
66  else {
67  // update value type and engineering unit, if not yet set
68  if(valueType == &typeid(AnyType)) valueType = &(a.getValueType());
69  if(engineeringUnit == ChimeraTK::TransferElement::unitNotSet) engineeringUnit = a.getUnit();
70  if(description == "") description = a.getDescription();
71  }
72  // add node to node list
73  nodeList.push_back(a);
74  }

114  VariableNetwork::TriggerType VariableNetwork::getTriggerType(bool verboseExceptions) const {
115  if(!hasFeedingNode()) return TriggerType::none;
116  const auto& feeder = getFeedingNode();
117  // network has an external trigger
118  if(feeder.hasExternalTrigger()) {
119  if(feeder.getMode() == UpdateMode::push) {
120  std::stringstream msg;
121  msg << "Providing an external trigger to a variable network which is fed "
122  "by a pushing variable is not allowed."
123  << std::endl;
124  if(verboseExceptions) {
125  msg << "The illegal network:" << std::endl;
126  dump("", msg);
127  }
128  throw ChimeraTK::logic_error(msg.str());
129  }
130  return TriggerType::external;
131  }
132  // network is fed by a pushing node
133  if(feeder.getMode() == UpdateMode::push) {
134  return TriggerType::feeder;
135  }
136  // network is fed by a poll-type node: must have exactly one polling consumer
137  size_t nPollingConsumers = count_if(nodeList.begin(), nodeList.end(), [](const VariableNetworkNode& n) {
138  return n.getDirection().dir == VariableDirection::consuming && n.getMode() == UpdateMode::poll;
139  });
140  if(nPollingConsumers != 1) {
141  std::stringstream msg;
142  msg << "In a network with a poll-type feeder and no external trigger, "
143  "there must be exactly one polling consumer. Maybe you forgot to "
144  "add a trigger?"
145  << std::endl;
146  if(verboseExceptions) {
147  msg << "The illegal network:" << std::endl;
148  dump("", msg);
149  }
150  throw ChimeraTK::logic_error(msg.str());
151  }
152  return TriggerType::pollingConsumer;
153  }

157  void VariableNetwork::check() const {
158  // must have consuming nodes
159  if(countConsumingNodes() == 0) {
160  std::stringstream msg;
161  msg << "No consuming nodes connected to this network!" << std::endl;
162  msg << "The illegal network:" << std::endl;
163  dump("", msg);
164  throw ChimeraTK::logic_error(msg.str());
165  }
166 
167  // must have a feeding node
168  if(!hasFeedingNode()) {
169  std::stringstream msg;
170  msg << "No feeding node connected to this network!" << std::endl;
171  msg << "The illegal network:" << std::endl;
172  dump("", msg);
173  throw ChimeraTK::logic_error(msg.str());
174  }
175 
176  // the network's value type must be correctly set
177  if(*valueType == typeid(AnyType)) {
178  std::stringstream msg;
179  msg << "No data type specified for any of the nodes in this network!" << std::endl;
180  msg << "The illegal network:" << std::endl;
181  dump("", msg);
182  throw ChimeraTK::logic_error(msg.str());
183  }
184 
185  // the feeder node must have a non-zero length
186  size_t length = getFeedingNode().getNumberOfElements();
187  if(length == 0) {
188  std::stringstream msg;
189  msg << "The feeding node has zero (or undefined) length!" << std::endl;
190  msg << "The illegal network:" << std::endl;
191  dump("", msg);
192  throw ChimeraTK::logic_error(msg.str());
193  }
194 
195  // all consumers must have the same length as the feeder or a zero length for
196  // trigger receivers
197  for(auto& node : nodeList) {
198  if(node.getType() != NodeType::TriggerReceiver) {
199  if(node.getNumberOfElements() != length) {
200  std::stringstream msg;
201  msg << "The network contains a node with a different length than the "
202  "feeding node!"
203  << std::endl;
204  msg << "The illegal network:" << std::endl;
205  dump("", msg);
206  throw ChimeraTK::logic_error(msg.str());
207  }
208  }
209  else {
210  if(node.getNumberOfElements() != 0) {
211  std::stringstream msg;
212  msg << "The network contains a trigger receiver node with a non-zero "
213  "length!"
214  << std::endl;
215  msg << "The illegal network:" << std::endl;
216  dump("", msg);
217  throw ChimeraTK::logic_error(msg.str());
218  }
219  }
220  }
221 
222  // all nodes must have this network as the owner and a value type equal the
223  // network's value type
224  for(auto& node : nodeList) {
225  assert(&(node.getOwner()) == this);
226  if(node.getValueType() == typeid(AnyType)) node.setValueType(*valueType);
227  if(node.getValueType() != *valueType) {
228  std::stringstream msg;
229  msg << "The network contains variables of different value types, which "
230  "is not supported!"
231  << std::endl;
232  msg << "The illegal network:" << std::endl;
233  dump("", msg);
234  throw ChimeraTK::logic_error(msg.str());
235  }
236  }
237 
238  // if the feeder is an application node, it must be in push mode
239  if(getFeedingNode().getType() == NodeType::Application) {
240  assert(getFeedingNode().getMode() == UpdateMode::push);
241  }
242 
243  // check if trigger is correctly defined (the return type doesn't matter, only
244  // the checks done in the function are needed)
245  getTriggerType();
246 
247  // if (and only if) feeder has return channel, there must be exactly one
248  // consumer with return channel
249  if(getFeedingNode().getDirection().withReturn) {
250  size_t numberOfConsumersWithReturnChannel = 0;
251  for(auto& node : getConsumingNodes()) {
252  if(node.getDirection().withReturn) ++numberOfConsumersWithReturnChannel;
253  }
254  if(numberOfConsumersWithReturnChannel != 1) {
255  std::stringstream msg;
256  msg << "The network has a feeder with return channel but " << numberOfConsumersWithReturnChannel << " consumers"
257  << " with return channel (must have exactly 1)!" << std::endl;
258  msg << "The illegal network:" << std::endl;
259  dump("", msg);
260  throw ChimeraTK::logic_error(msg.str());
261  }
262  }
263  else {
264  for(auto& node : getConsumingNodes()) {
265  if(node.getDirection().withReturn) {
266  std::stringstream msg;
267  msg << "The network has no feeder with return channel but at least one "
268  "consumer with return channel!"
269  << std::endl;
270  msg << "The illegal network:" << std::endl;
271  dump("", msg);
272  throw ChimeraTK::logic_error(msg.str());
273  }
274  }
275  }
276  }

313  bool VariableNetwork::merge(VariableNetwork& other) {
314  // check if merging is possible
315  if(hasFeedingNode() && other.hasFeedingNode()) {
316  if((getFeedingNode().getType() == NodeType::ControlSystem &&
317  other.getFeedingNode().getType() == NodeType::ControlSystem) ||
318  (getFeedingNode().getType() != NodeType::ControlSystem &&
319  other.getFeedingNode().getType() != NodeType::ControlSystem)) {
320  return false;
321  }
322  }
323 
324  // put all consuming nodes of B's owner into A's owner
325  for(auto& node : other.getConsumingNodes()) {
326  other.removeNode(node);
327  addNode(node);
328  }
329 
330  // feeding node: if control system type, convert into consumer. Otherwise just
331  // add it, addNode() will convert the other feeding node if necessary
332  if(other.hasFeedingNode()) {
333  VariableNetworkNode otherFeeder = other.getFeedingNode();
334  other.removeNode(otherFeeder);
335  if(otherFeeder.getType() == NodeType::ControlSystem)
336  otherFeeder.setDirection({VariableDirection::consuming, false});
337  addNode(otherFeeder);
338  }
339 
340  return true;
341  }
