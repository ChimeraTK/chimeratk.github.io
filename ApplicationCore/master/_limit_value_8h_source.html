21  void applyLimit(UserType min, UserType max) {
22  bool wasLimited = isLimited;
23 
24  // clamp input value into given range
25  UserType value = input;
26  if(value > max) {
27  output = max;
28  isLimited = true;
29  }
30  else if(value < min) {
31  output = min;
32  isLimited = true;
33  }
34  else {
35  output = value;
36  isLimited = false;
37  }
38 
39  // write output. isLimited is only written when changed
40  output.write();
41  if(isLimited != wasLimited) isLimited.write();
42  wasLimited = isLimited;
43  }
44  };
