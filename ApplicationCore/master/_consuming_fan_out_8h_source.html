37  void doPostRead(TransferType type, bool updateDataBuffer) override {
38  ChimeraTK::NDRegisterAccessorDecorator<UserType>::doPostRead(type, updateDataBuffer);
39 
40  if(updateDataBuffer) {
41  // We have to keep a copy to write into the slaves. There might
42  // be decorators arount this fanout which swap out buffer_2D, so it is
43  // not available any more for a second read witout updateDataBuffer (exception case).
44  _lastReceivedValue = buffer_2D[0];
45  }
46 
47  // The ConsumingFanOut conceptually never has a wait_fow_new_data flags. Hence each read
48  // operation returns with "new" data, even in case of an exception. So each read
49  // always synchronises all slaves and pushes the content of the data buffer.
50  for(auto& slave : FanOut<UserType>::slaves) { // send out copies to slaves
51  // do not send copy if no data is expected (e.g. trigger)
52  if(slave->getNumberOfSamples() != 0) {
53  slave->accessChannel(0) = _lastReceivedValue;
54  }
55  slave->setDataValidity(this->dataValidity());
56  slave->writeDestructively();
57  }
58  }
