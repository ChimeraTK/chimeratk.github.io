41  ctk::ArrayOutput<int32_t> setpointTable{this, "setpointTable", "bits", tableLength, "..."};

43 
44  void mainLoop() {
45  ctk::ReadAnyGroup tableParametersReadGroup = tableParameters.readAnyGroup();
46  while(true) {
47  tableParametersReadGroup.readAny(); // block until the any table parameter is changed
48 
49  for(size_t i = 0; i < tableLength; ++i) {
50  if(i < tableParameters.pulseLength * samplingFrequency) {
51  setpointTable[i] = tableParameters.setpoint * bitScalingFactor;
52  feedforwardTable[i] = 0.5 * tableParameters.setpoint * bitScalingFactor;
53  }
54  else {
55  setpointTable[i] = 0;
56  feedforwardTable[i] = 0;
57  }
58  }
59

62  }
