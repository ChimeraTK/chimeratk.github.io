26  VariableNetworkNode ControlSystemModule::operator()(
27  const std::string& variableName, const std::type_info& valueType, size_t nElements) const {
28  assert(variableName.find_first_of("/") == std::string::npos);

53  std::list<VariableNetworkNode> ControlSystemModule::getAccessorList() const {
54  std::list<VariableNetworkNode> list;
55  auto& variables = Application::getInstance().controlSystemVariables;
56  for(auto& v : variables) {
57  // check if variable has the right prefix
58  auto idx = v.first.rfind("/"); // position of last slasht
59  if(!v.first.compare(0, idx, variableNamePrefix)) {
60  // All characters until the last slash are equal to our variableNamePrefix: The variable belongs to our module.
61  list.push_back(v.second);
62  }
63  }
64  return list;
65  }
