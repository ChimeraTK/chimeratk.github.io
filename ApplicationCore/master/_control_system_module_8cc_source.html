31  VariableNetworkNode ControlSystemModule::operator()(
32  const std::string& variableName, const std::type_info& valueType, size_t nElements) const {
33  assert(variableName.find_first_of("/") == std::string::npos);
34  auto& variables = Application::getInstance().controlSystemVariables;
35  auto fqn = variableNamePrefix / variableName;
36  if(variables.count(fqn) == 0) {
37  variables[fqn] = {fqn, {VariableDirection::invalid, false}, valueType, nElements};

60  std::list<VariableNetworkNode> ControlSystemModule::getAccessorList() const {
61  std::list<VariableNetworkNode> list;
62  auto& variables = Application::getInstance().controlSystemVariables;
63  for(auto& v : variables) {
64  // check if variable has the right prefix
65  auto idx = v.first.rfind("/"); // position of last slasht
66  if(!v.first.compare(0, idx, variableNamePrefix)) {
67  // All characters until the last slash are equal to our variableNamePrefix: The variable belongs to our module.
68  list.push_back(v.second);
69  }
70  }
71  return list;
72  }
