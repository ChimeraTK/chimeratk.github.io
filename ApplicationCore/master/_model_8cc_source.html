250  ApplicationModuleProxy VariableGroupProxy::getOwningModule() const {
251  Vertex currentVertex{_d->vertex};
252 
253  do {
254  // there must be exactly one owner
255  assert(boost::in_degree(currentVertex, _d->impl->ownershipView) == 1);
256 
257  // Update currentVertex to the owner of the previous currentVertex.
258  currentVertex =
259  boost::source(*boost::in_edges(currentVertex, _d->impl->ownershipView).first, _d->impl->ownershipView);
260 
261  // repeat until the type of the currentVertex is no longer a VariablGroup
262  } while(_d->impl->ownershipView[currentVertex].type == VertexProperties::Type::variableGroup);
263 
264  // The type must now be a ApplicationModule
265  assert(_d->impl->ownershipView[currentVertex].type == VertexProperties::Type::applicationModule);
266 
267  return {currentVertex, _d->impl};
268  }

355  void ProcessVariableProxy::removeNode(const VariableNetworkNode& node) {
356  assert(node.getType() == NodeType::Application || node.getType() == NodeType::Device);
357 
358  // remove node from the list of nodes in the PV's vertex properties
359  auto& nodes = std::get<VertexProperties::ProcessVariableProperties>(_d->impl->graph[_d->vertex].p).nodes;
360  auto it = std::find(nodes.begin(), nodes.end(), node);
361  if(it == nodes.end()) {
362  return;
363  }
364  nodes.erase(it);
365 
366  // remove model relationships between PV and module
367  if(node.getType() == NodeType::Application) {
368  auto* vg = dynamic_cast<VariableGroup*>(node.getOwningModule());
369  assert(vg != nullptr);
370 
371  // remove ownership edge to variable group (if any)
372  auto vgm = vg->getModel();
373  if(vgm.isValid()) {
374  boost::remove_edge(vgm._d->vertex, _d->vertex, _d->impl->graph);
375  }
376 
377  // remove pv access edge, and along side the ownership edge if directly owned by application module
378  auto* am = dynamic_cast<ApplicationModule*>(vg->findApplicationModule());
379  assert(am != nullptr);
380  auto amm = am->getModel();
381 
382  assert(amm.isValid());
383  boost::remove_edge(amm._d->vertex, _d->vertex, _d->impl->graph);
384  boost::remove_edge(_d->vertex, amm._d->vertex, _d->impl->graph);
385  }
386  else if(node.getType() == NodeType::Device) {
387  auto* dm = dynamic_cast<DeviceModule*>(node.getOwningModule());
388  assert(dm != nullptr);
389 
390  // remove ownership edge to variable group (if any)
391  auto dmm = dm->getModel();
392  if(dmm.isValid()) {
393  boost::remove_edge(_d->vertex, dmm._d->vertex, _d->impl->graph);
394  boost::remove_edge(dmm._d->vertex, _d->vertex, _d->impl->graph);
395  }
396  }
397 
398  // if only one incoming edge exists any more, remove the entire variable. the one incoming edge is the parenthood
399  // relation. ownership relations are also incoming, of which we must have zero.
400  if(boost::in_degree(_d->vertex, _d->impl->graph) <= 1 && boost::out_degree(_d->vertex, _d->impl->graph) == 0) {
401  // Note: We cannot really remove the vertex for the variable, since boost::remove_vertex() invalidates all vertex
402  // descriptors, which are the only link between the "real world" and the model.
403  // Instead we completely disconnect it from the rest of the model (in particular the parent directory), so it
404  // usually is no longer found.
405  clear_vertex(_d->vertex, _d->impl->graph);
406  }
407  }

471  template<typename PROXY, typename MODULE, typename PROPS, VertexProperties::Type TYPE>
472  PROXY Impl::generic_add(Vertex owner, MODULE& module) {
473  auto parentDirectory = visit(owner, returnDirectory, getNeighbourDirectory, returnFirstHit(DirectoryProxy{}));
474 
475  // create plain vertex first
476  auto newVertex = boost::add_vertex(graph);
477 
478  // set vertex type and type-dependent properties
479  graph[newVertex].type = TYPE;
480  if constexpr(!std::is_same<MODULE, DeviceModule>::value) {
481  graph[newVertex].p.emplace<PROPS>(PROPS{module.getName(), module});
482  }
483  else {
484  auto alias = module.getDeviceManager().getDeviceAliasOrURI();
485  auto triggerPath = module.getTriggerPath();
486 
487  ProcessVariableProxy trigger; // initially invalid, stays like that if triggerPath.empty()
488 
489  if(!triggerPath.empty()) {
490  auto dir = parentDirectory.addDirectoryRecursive(Utilities::getPathName(triggerPath));
491  trigger = dir.addVariable(Utilities::getUnqualifiedName(triggerPath));
492 
493  // connect trigger vertex with trigger edge
494  auto triggerEdge = boost::add_edge(trigger._d->vertex, newVertex, graph).first;
495  graph[triggerEdge].type = EdgeProperties::Type::trigger;
496  }
497 
498  graph[newVertex].p.emplace<PROPS>(PROPS{alias, trigger, module});
499  }
500 
501  // connect the vertex with an ownership edge
502  auto ownershipEdge = boost::add_edge(owner, newVertex, graph).first; // .second is ignored as it cannot fail
503  graph[ownershipEdge].type = EdgeProperties::Type::ownership;
504 
505  // obtain/create directory corresponding to the fully qualified path of the module
506  DirectoryProxy directory;
507  if constexpr(!std::is_same<MODULE, DeviceModule>::value) {
508  directory = addDirectoryRecursive(parentDirectory._d->vertex, module.getName());
509  }
510  else {
511  // the register content of DeviceModules is connected directly to the parent directory, since the alias/URI shall
512  // not be visible as a sub-directory.
513  directory = parentDirectory;
514  }
515 
516  // connect the vertex with the directory with a neighbourhood edge
517  auto neighbourhoodEdge = boost::add_edge(newVertex, directory._d->vertex, graph).first; // dito
518  graph[neighbourhoodEdge].type = EdgeProperties::Type::neighbourhood;
519 
520  return {newVertex, shared_from_this()};
521  }

549  template<typename MODULE>
550  void Impl::generic_remove(Model::Vertex owner, MODULE& module) {
551  auto modelToRemove = module.getModel();
552 
553  // The model may be invalid e.g. in case of calls to unregisterModule() in move assignment operations. Nothing to be
554  // removed from the model in that case.
555  if(!modelToRemove.isValid()) {
556  return;
557  }
558 
559  // Remove all relationships (edges) of the module and its sub-modules.
560  // Do not remove edges while iterating the model (invalidates iterators), so store them first and remove later.
561  std::list<std::pair<Model::Vertex, Model::Vertex>> removeList;
562  std::set<Module*> moduleList; // needed to remove the VariableNetworkNodes below
563  modelToRemove.visit(
564  [&](auto proxy) {
565  if constexpr(isApplicationModule(proxy)) {
566  moduleList.insert(&proxy.getApplicationModule());
567  }
568 
569  auto vertex = proxy._d->vertex;
570 
571  {
572  auto [b, e] = boost::adjacent_vertices(vertex, graph);
573  for(auto vtx = b; vtx != e; ++vtx) {
574  removeList.push_back({vertex, *vtx});
575  }
576  }
577  {
578  auto [b, e] = boost::inv_adjacent_vertices(vertex, graph);
579  for(auto vtx = b; vtx != e; ++vtx) {
580  removeList.push_back({vertex, *vtx});
581  }
582  }
583  },
584  keepModuleGroups || keepApplicationModules || keepVariableGroups, keepOwnership, depthFirstSearch);
585 
586  // Remove all VariableNetworkNodes from the PVs owned by the module and its sub-modules. We are using the above
587  // collected moduleList here to determine which node is owned by one of the (sub-)modules.
588  // Again, the nodes must not be removed while iterating the model, since ProcessVariableProxy::removeNode() alters
589  // the model and hence invalidates the iterators.
590  std::list<std::pair<ProcessVariableProxy, VariableNetworkNode>> nodeRemoveList;
591  modelToRemove.visit(
592  [&](const ProcessVariableProxy& proxy) {
593  for(const auto& node : proxy.getNodes()) {
594  if(node.getType() != NodeType::Application) continue;
595  auto* am = dynamic_cast<VariableGroup*>(node.getOwningModule())->findApplicationModule();
596  if(moduleList.count(am) == 0) {
597  continue;
598  }
599  nodeRemoveList.emplace_back(proxy, node);
600  }
601  },
602  keepProcessVariables, keepOwnership, depthFirstSearch);
603 
604  // Now execute actual removal of nodes
605  for(auto& p : nodeRemoveList) {
606  p.first.removeNode(p.second);
607  }
608 
609  // Now execute actual removal of edges
610  for(auto p : removeList) {
611  boost::remove_edge(p.first, p.second, graph);
612  boost::remove_edge(p.second, p.first, graph);
613  }
614 
615  // Remove the ownership relationship for the module.
616  auto vertexToRemove = modelToRemove._d->vertex;
617  boost::remove_edge(owner, vertexToRemove, graph);
618  boost::remove_edge(vertexToRemove, owner, graph);
619  }

623  template<typename PROXY>
624  void Impl::addVariableNode(PROXY module, const ProcessVariableProxy& variable, const VariableNetworkNode& node) {
625  // get vertex of for variable
626  auto vertex = variable._d->vertex;
627 
628  // get owning vertex (VariableGroup or ApplicationModule)
629  auto owningVertex = module._d->vertex;
630 
631  // create ownership edge
632  auto newOwnershipEdge = boost::add_edge(owningVertex, vertex, graph).first; // see above
633  graph[newOwnershipEdge].type = EdgeProperties::Type::ownership;
634 
635  // get accessing ApplicationModule or DeviceModule vertex
636  Vertex accessingVertex;
637  if constexpr(std::is_same<PROXY, ApplicationModuleProxy>::value || std::is_same<PROXY, DeviceModuleProxy>::value) {
638  accessingVertex = module._d->vertex;
639  }
640  else {
641  static_assert(std::is_same<PROXY, VariableGroupProxy>::value, "Proxy type cannot have variable nodes.");
642  accessingVertex = module.getOwningModule()._d->vertex;
643  }
644 
645  // add node to variable
646  auto& props = std::get<VertexProperties::ProcessVariableProperties>(graph[vertex].p);
647  props.nodes.emplace_back(node);
648 
649  // add tags
650  const auto& tags = node.getTags();
651  props.tags.insert(tags.begin(), tags.end());
652 
653  // connect the variable vertex with the accessing module, direction depends on access (read/write)
654  Edge newEdge;
655  if(node.getDirection().dir == VariableDirection::feeding) {
656  newEdge = boost::add_edge(accessingVertex, vertex, graph)
657  .first; // .second is ignored: silently reject duplicates CHECKME/FIXME!
658  }
659  else {
660  newEdge = boost::add_edge(vertex, accessingVertex, graph).first; // see above
661  }
662  graph[newEdge].type = EdgeProperties::Type::pvAccess;
663  }

667  ProcessVariableProxy Impl::addVariable(Vertex parent, const std::string& name) {
669 
670  // first check if variable already exists
671  auto existing = visit(parent, returnProcessVariable, adjacentOutSearch, keepParenthood,
672  keepProcessVariables && keepName(name), returnFirstHit(ProcessVariableProxy{}));
673  if(existing.isValid()) {
674  return existing;
675  }
676 
677  // create plain vertex first
678  auto newVertex = boost::add_vertex(graph);
679 
680  // set vertex type and type-dependent properties

721  DirectoryProxy Impl::addDirectoryRecursive(Vertex parent, const std::string& qualifiedPath) {
722  // separate the path into components
723  std::vector<std::string> components;
724  boost::split(components, qualifiedPath, boost::is_any_of("/"));
725 
726  // Start at the parent
727  DirectoryProxy currentDirectory(parent, shared_from_this());
728 
729  for(const auto& component : components) {
730  // special treatment for "."
731  if(component == ".") {
732  continue;
733  }
734 
735  // special treatment for ".."
736  if(component == "..") {
737  if(currentDirectory._d->vertex == *boost::vertices(currentDirectory._d->impl->graph).first) {
738  throw ChimeraTK::logic_error("Path component '..' at root directory level found.");
739  }
740  currentDirectory = currentDirectory.visit(returnDirectory, getParent, returnFirstHit(DirectoryProxy{}));
741  assert(currentDirectory.isValid());
742  continue;
743  }
744 
745  // Special treatment for an extra slash e.g. at the beginning or two consequtive slashes.
746  // Since the slash is the separator, the path component is just empty.
747  if(component.empty()) {
748  currentDirectory = {*(boost::vertices(graph).first), shared_from_this()}; // root directory
749  continue;
750  }
751 
752  // no special treatment necessary: just add the new level (if not yet existing)
753  currentDirectory = currentDirectory.addDirectory(component);
754  }
755 
756  // return the directory (either newly created or previously existing)
757  return currentDirectory;
758  }

762  [[nodiscard]] std::string Impl::getFullyQualifiedPath(Vertex vertex) {
763  return graph[vertex].visit([&](auto props) -> std::string {
764  constexpr bool useDirectoryHierarchy = isDirectory(props) || isVariable(props);
765 
766  Vertex currentVertex = vertex;
767 
768  RegisterPath path;
769 
770  // Add name of current vertex (unless it's a DeviceModule or the Root)
771  if constexpr(hasName(props)) {
772  path = props.name;
773  }
774 
775  // Loop until we have found the root vertex, add all names to the path
776  while(currentVertex != *boost::vertices(graph).first) {
777  // define vistor to be executed for the owner of currentVertex
778  bool found = false;
779  auto vis = [&](auto proxy) {
780  if constexpr(hasName(proxy)) {
781  // prefix name of current object to path, since we are iterating towards the root
782  path = proxy.getName() / path;
783  }
784  currentVertex = proxy._d->vertex;
785  found = true;
786  };
787 
788  // just call the lambda for the owner/parent of currentVertex
789  if constexpr(!useDirectoryHierarchy) {
790  visit(currentVertex, vis, getOwner);
791  }
792  else {
793  visit(currentVertex, vis, getParent);
794  }
795 
796  // abort condition in case we try to get the qualified path for something not connected with the root any more
797  if(!found) {
798  path = "<disjunct>" / path;
799  break;
800  }
801  }
802 
803  // resolve "/", ".." and "."
804  auto components = path.getComponents();
805  path = ""; // build path from components
806  for(const auto& component : components) {
807  if(component == ".") {
808  continue;
809  }
810  if(component == "..") {
811  path--; // remove last component
812  continue;
813  }
814  if(component == "/") {
815  path = ""; // clear entire path
816  continue;
817  }
818  path /= component;
819  }
820 
821  return path;
822  });
823  }
