53  RootProxy::RootProxy(ModuleGroup& app) {
54  // create ProxyData object
55  _d = std::make_shared<ProxyData>();
56  // create the graph
57  _d->impl = std::make_shared<Impl>();
58  // create root vertex
59  _d->vertex = boost::add_vertex(_d->impl->_graph);
60  VertexProperties::RootProperties props{app};
61  _d->impl->_graph[_d->vertex].type = VertexProperties::Type::root;
62  _d->impl->_graph[_d->vertex].p.emplace<VertexProperties::RootProperties>(props);
63 
64  // create neighbourhood edge to itself: the root represents both the Application ModuleGroup and the directory.
65  auto [neighbourhoodEdge, success] = boost::add_edge(_d->vertex, _d->vertex, _d->impl->_graph);
66  assert(success);
67  _d->impl->_graph[neighbourhoodEdge].type = EdgeProperties::Type::neighbourhood;
68  }

294  ApplicationModuleProxy VariableGroupProxy::getOwningModule() const {
295  Vertex currentVertex{_d->vertex};
296 
297  do {
298  // there must be exactly one owner
299  assert(boost::in_degree(currentVertex, _d->impl->_ownershipView) == 1);
300 
301  // Update currentVertex to the owner of the previous currentVertex.
302  currentVertex =
303  boost::source(*boost::in_edges(currentVertex, _d->impl->_ownershipView).first, _d->impl->_ownershipView);
304 
305  // repeat until the type of the currentVertex is no longer a VariablGroup
306  } while(_d->impl->_ownershipView[currentVertex].type == VertexProperties::Type::variableGroup);
307 
308  // The type must now be a ApplicationModule
309  assert(_d->impl->_ownershipView[currentVertex].type == VertexProperties::Type::applicationModule);
310 
311  return _d->impl->_graph[currentVertex].makeProxy<ApplicationModuleProxy>(currentVertex, _d->impl);
312  }

399  void ProcessVariableProxy::removeNode(VariableNetworkNode node) {
400  // Safety check: We must not modify the model while iterating
401  if(_d->impl->_graphVisitingLevel != 0) {
402  throw ChimeraTK::logic_error("Must not alter the model while iterating/visiting!");
403  }
404 
405  assert(node.getType() == NodeType::Application || node.getType() == NodeType::Device);
406  assert(node.getModel().isValid());
407  assert(isValid());
408  assert(node.getModel()._d->vertex == _d->vertex);
409  assert(node.getModel()._d->impl == _d->impl);
410 
411  // remove node from the list of nodes in the PV's vertex properties
412  try {
413  auto& nodes = std::get<VertexProperties::ProcessVariableProperties>(_d->impl->_graph[_d->vertex].p).nodes;
414  auto it = std::find(nodes.begin(), nodes.end(), node);
415  if(it == nodes.end()) {
416  return;
417  }
418  nodes.erase(it);
419 
420  // remove model relationships between PV and module
421  bool ownershipDeleted{false};
422  if(node.getType() == NodeType::Application) {
423  auto* vg = dynamic_cast<VariableGroup*>(node.getOwningModule());
424  assert(vg != nullptr);
425 
426  // remove ownership edge to variable group (if any)
427  auto vgm = vg->getModel();
428  if(vgm.isValid()) {
429  for(const auto& edge :
430  boost::make_iterator_range(boost::edge_range(vgm._d->vertex, _d->vertex, _d->impl->_graph))) {
431  if(_d->impl->_graph[edge].type == EdgeProperties::Type::ownership) {
432  boost::remove_edge(edge, _d->impl->_graph);
433  ownershipDeleted = true;
434  break;
435  }
436  }
437  }
438 
439  // obtain the accessing module
440  auto* am = dynamic_cast<ApplicationModule*>(vg->findApplicationModule());
441  assert(am != nullptr);
442  auto amm = am->getModel();
443  // The owning ApplicationModule might be no longer in the model. This happens when the owning ApplicationModule
444  // has been removed from the model already. In this case, the direct ownership relation as well as the PV access
445  // relation have been removed already when removing the ApplicationModule.
446  if(amm.isValid()) {
447  // remove ownership edge to application module group (if directly owned)
448  if(!ownershipDeleted) {
449  for(const auto& edge :
450  boost::make_iterator_range(boost::edge_range(amm._d->vertex, _d->vertex, _d->impl->_graph))) {
451  if(_d->impl->_graph[edge].type == EdgeProperties::Type::ownership) {
452  boost::remove_edge(edge, _d->impl->_graph);
453  ownershipDeleted = true;
454  break;
455  }
456  }
457  }
458 
459  // the ownership must be removed now
460  assert(ownershipDeleted);
461 
462  // remove pv access edge
463 #ifndef NDEBUG
464  bool pvAccessDeleted{false};
465 #endif
466  if(node.getDirection().dir == VariableDirection::consuming) {
467  for(const auto& edge :
468  boost::make_iterator_range(boost::edge_range(_d->vertex, amm._d->vertex, _d->impl->_graph))) {
469  if(_d->impl->_graph[edge].type == EdgeProperties::Type::pvAccess) {
470  boost::remove_edge(edge, _d->impl->_graph);
471 #ifndef NDEBUG
472  pvAccessDeleted = true;
473 #endif
474  break;
475  }
476  }
477  }
478  else {
479  assert(node.getDirection().dir == VariableDirection::feeding);
480  for(const auto& edge :
481  boost::make_iterator_range(boost::edge_range(amm._d->vertex, _d->vertex, _d->impl->_graph))) {
482  if(_d->impl->_graph[edge].type == EdgeProperties::Type::pvAccess) {
483  boost::remove_edge(edge, _d->impl->_graph);
484 #ifndef NDEBUG
485  pvAccessDeleted = true;
486 #endif
487  break;
488  }
489  }
490  }
491  assert(pvAccessDeleted);
492  }
493  }
494  else if(node.getType() == NodeType::Device) {
495  auto* dm = dynamic_cast<DeviceModule*>(node.getOwningModule());
496  assert(dm != nullptr);
497 
498  // remove ownership and pv access edges to device module
499  auto dmm = dm->getModel();
500  if(dmm.isValid()) {
501  boost::remove_edge(_d->vertex, dmm._d->vertex, _d->impl->_graph);
502  boost::remove_edge(dmm._d->vertex, _d->vertex, _d->impl->_graph);
503  }
504  }
505 
506  // if only one incoming edge exists any more, remove the entire variable. the one incoming edge is the parenthood
507  // relation. ownership relations are also incoming, of which we must have zero.
508  if(boost::in_degree(_d->vertex, _d->impl->_graph) <= 1 && boost::out_degree(_d->vertex, _d->impl->_graph) == 0) {
509  assert(boost::in_degree(_d->vertex, _d->impl->_graph) == 0 ||
510  _d->impl->_graph[*boost::in_edges(_d->vertex, _d->impl->_graph).first].type ==
511  EdgeProperties::Type::parenthood);
512 
513  assert(nodes.empty());
514 
515  boost::clear_vertex(_d->vertex, _d->impl->_graph);
516  boost::remove_vertex(_d->vertex, _d->impl->_graph);
517 
518  _d->vertex = Model::Vertex();
519  _d->impl.reset();
520  }
521  }
522  catch(std::bad_variant_access&) {
523  assert(false);
524  }
525  }

589  template<typename PROXY, typename MODULE, typename PROPS, VertexProperties::Type TYPE>
590  PROXY Impl::genericAdd(Vertex owner, MODULE& module) {
591  // Safety check: We must not modify the model while iterating
592  if(_graphVisitingLevel != 0) {
593  throw ChimeraTK::logic_error("Must not alter the model while iterating/visiting!");
594  }
595 
596  auto parentDirectory = visit(owner, returnDirectory, getNeighbourDirectory, returnFirstHit(DirectoryProxy{}));
597 
598  // create plain vertex first
599  auto* newVertex = boost::add_vertex(_graph);
600 
601  // set vertex type and type-dependent properties
602  _graph[newVertex].type = TYPE;
603  if constexpr(!std::is_same<MODULE, DeviceModule>::value) {
604  _graph[newVertex].p.emplace<PROPS>(PROPS{module.getName(), module});
605  }
606  else {
607  auto alias = module.getDeviceManager().getDeviceAliasOrURI();
608  auto triggerPath = module.getTriggerPath();
609 
610  ProcessVariableProxy trigger; // initially invalid, stays like that if triggerPath.empty()
611 
612  if(!triggerPath.empty()) {
613  auto dir = parentDirectory.addDirectoryRecursive(Utilities::getPathName(triggerPath));
614  trigger = dir.addVariable(Utilities::getUnqualifiedName(triggerPath));
615 
616  // connect trigger vertex with trigger edge
617  auto [triggerEdge, triggerSuccess] = boost::add_edge(trigger._d->vertex, newVertex, _graph);
618  assert(triggerSuccess);
619  _graph[triggerEdge].type = EdgeProperties::Type::trigger;
620  }
621 
622  _graph[newVertex].p.emplace<PROPS>(PROPS{alias, trigger, module});
623  }
624 
625  // connect the vertex with an ownership edge
626  auto [ownershipEdge, ownershipSuccess] = boost::add_edge(owner, newVertex, _graph);
627  assert(ownershipSuccess);
628  _graph[ownershipEdge].type = EdgeProperties::Type::ownership;
629 
630  // obtain/create directory corresponding to the fully qualified path of the module
631  DirectoryProxy directory;
632  if constexpr(!std::is_same<MODULE, DeviceModule>::value) {
633  directory = addDirectoryRecursive(parentDirectory._d->vertex, module.getName());
634  }
635  else {
636  // the register content of DeviceModules is connected directly to the parent directory, since the alias/URI shall
637  // not be visible as a sub-directory.
638  directory = parentDirectory;
639  }
640 
641  // connect the vertex with the directory with a neighbourhood edge
642  auto [neighbourhoodEdge, neighbourhoodSuccess] = boost::add_edge(newVertex, directory._d->vertex, _graph);
643  assert(neighbourhoodSuccess);
644  _graph[neighbourhoodEdge].type = EdgeProperties::Type::neighbourhood;
645 
646  return _graph[newVertex].makeProxy<PROXY>(newVertex, shared_from_this());
647  }

675  template<typename MODULE>
676  void Impl::genericRemove(MODULE& module) {
677  // Safety check: We must not modify the model while iterating
678  if(_graphVisitingLevel != 0) {
679  throw ChimeraTK::logic_error("Must not alter the model while iterating/visiting!");
680  }
681  auto modelToRemove = module.getModel();
682 
683  // The model may be invalid e.g. in case of calls to unregisterModule() in move assignment operations. Nothing to be
684  // removed from the model in that case.
685  if(!modelToRemove.isValid()) {
686  return;
687  }
688 
689  // Remove the vertex representing this module with all edges (ownership, PV access etc.)
690  auto vertexToRemove = modelToRemove._d->vertex;
691  boost::clear_vertex(vertexToRemove, _graph);
692  boost::remove_vertex(vertexToRemove, _graph);
693 
694  module._model = {};
695  }

700  void Impl::addVariableNode(PROXY module, ProcessVariableProxy& variable, VariableNetworkNode& node) {
701  // Safety check: We must not modify the model while iterating
702  if(_graphVisitingLevel != 0) {
703  throw ChimeraTK::logic_error("Must not alter the model while iterating/visiting!");
704  }
705  node.setModel(variable);
706 
707  // get vertex of for variable
708  auto* vertex = variable._d->vertex;
709 
710  // get owning vertex (VariableGroup or ApplicationModule)
711  auto owningVertex = module._d->vertex;
712 
713  // create ownership edge
714  auto [newOwnershipEdge, success] = boost::add_edge(owningVertex, vertex, _graph);
715  assert(success);
716  _graph[newOwnershipEdge].type = EdgeProperties::Type::ownership;
717 
718  // get accessing ApplicationModule or DeviceModule vertex
719  Vertex accessingVertex;
720  if constexpr(std::is_same<PROXY, ApplicationModuleProxy>::value || std::is_same<PROXY, DeviceModuleProxy>::value) {
721  accessingVertex = module._d->vertex;
722  }
723  else {
724  static_assert(std::is_same<PROXY, VariableGroupProxy>::value, "Proxy type cannot have variable nodes.");
725  accessingVertex = module.getOwningModule()._d->vertex;
726  }
727 
728  // add node to variable
729  auto& props = std::get<VertexProperties::ProcessVariableProperties>(_graph[vertex].p);
730  props.nodes.emplace_back(node);
731 
732  // add tags
733  const auto& tags = node.getTags();
734  props.tags.insert(tags.begin(), tags.end());
735 
736  // connect the variable vertex with the accessing module, direction depends on access (read/write)
737  Edge newEdge;
738  if(node.getDirection().dir == VariableDirection::feeding) {
739  std::tie(newEdge, success) = boost::add_edge(accessingVertex, vertex, _graph);
740  }
741  else {
742  std::tie(newEdge, success) = boost::add_edge(vertex, accessingVertex, _graph);
743  }
744  assert(success);
745  _graph[newEdge].type = EdgeProperties::Type::pvAccess;
746  _graph[newEdge].pvAccessWithReturnChannel = node.getDirection().withReturn;
747  }

752  // Safety check: We must not modify the model while iterating
753  if(_graphVisitingLevel != 0) {
754  throw ChimeraTK::logic_error("Must not alter the model while iterating/visiting!");
755  }

757  throw ChimeraTK::logic_error("Variable name '" + name + "' contains illegal characters.");
758  }
759

762  keepProcessVariables && keepName(name), returnFirstHit(ProcessVariableProxy{}));

764  return existing;

819  DirectoryProxy Impl::addDirectoryRecursive(Vertex parent, const std::string& qualifiedPath) {
820  // separate the path into components
821  std::vector<std::string> components;
822  // The analyzer assumes that m_Size can be modified in boost::is_any_of so that there could be a mismatch between
823  // creating a dynamic entry and not freeing it later if m_size suddenly is different from what it was
824  // NOLINTNEXTLINE(clang-analyzer-cplusplus.NewDeleteLeaks)
825  boost::split(components, qualifiedPath, boost::is_any_of("/"));
826 
827  // Start at the parent
828  auto currentDirectory = _graph[parent].makeProxy<DirectoryProxy>(parent, shared_from_this());
829 
830  for(const auto& component : components) {
831  // special treatment for "."
832  if(component == ".") {
833  continue;
834  }
835 
836  // special treatment for ".."
837  if(component == "..") {
838  if(currentDirectory._d->vertex == *boost::vertices(currentDirectory._d->impl->_graph).first) {
839  throw ChimeraTK::logic_error("Path component '..' at root directory level found.");
840  }
841  currentDirectory = currentDirectory.visit(returnDirectory, getParent, returnFirstHit(DirectoryProxy{}));
842  assert(currentDirectory.isValid());
843  continue;
844  }
845 
846  // Special treatment for an extra slash e.g. at the beginning or two consequtive slashes.
847  // Since the slash is the separator, the path component is just empty.
848  if(component.empty()) {
849  currentDirectory = Model::DirectoryProxy(RootProxy::makeRootProxy(shared_from_this())); // root directory
850  continue;
851  }
852 
853  // no special treatment necessary: just add the new level (if not yet existing)
854  currentDirectory = currentDirectory.addDirectory(component);
855  }
856 
857  // return the directory (either newly created or previously existing)
858  return currentDirectory;
859  }

863  [[nodiscard]] std::string Impl::getFullyQualifiedPath(Vertex vertex) {
864  return _graph[vertex].visit([&](auto props) -> std::string {
865  constexpr bool useDirectoryHierarchy = isDirectory(props) || isVariable(props);
866 
867  Vertex currentVertex = vertex;
868 
869  RegisterPath path;
870 
871  // Add name of current vertex (unless it's a DeviceModule or the Root)
872  if constexpr(hasName(props)) {
873  path = props.name;
874  }
875 
876  // Loop until we have found the root vertex, add all names to the path
877  while(currentVertex != *boost::vertices(_graph).first) {
878  // define vistor to be executed for the owner of currentVertex
879  bool found = false;
880  auto vis = [&](auto proxy) {
881  if constexpr(hasName(proxy)) {
882  // prefix name of current object to path, since we are iterating towards the root
883  path = proxy.getName() / path;
884  }
885  currentVertex = proxy._d->vertex;
886  found = true;
887  };
888 
889  // just call the lambda for the owner/parent of currentVertex
890  if constexpr(!useDirectoryHierarchy) {
891  visit(currentVertex, vis, getOwner);
892  }
893  else {
894  visit(currentVertex, vis, getParent);
895  }
896 
897  // abort condition in case we try to get the qualified path for something not connected with the root any more
898  if(!found) {
899  path = "<disjunct>" / path;
900  break;
901  }
902  }
903 
904  // resolve "/", ".." and "."
905  auto components = path.getComponents();
906  path = ""; // build path from components
907  for(const auto& component : components) {
908  if(component == ".") {
909  continue;
910  }
911  if(component == "..") {
912  path--; // remove last component
913  continue;
914  }
915  if(component == "/") {
916  path = ""; // clear entire path
917  continue;
918  }
919  path /= component;
920  }
921 
922  return path;
923  });
924  }
