305  template<typename UserType>
306  VariableNetworkNode VariableNetworkNode::makeConstant(bool makeFeeder, UserType value, size_t length) {
307  VariableNetworkNode node;
308  node.pdata = boost::make_shared<VariableNetworkNode_data>();
309  node.pdata->constNodeCreator.reset(new ConstantAccessorCreatorImpl<UserType>(value, length));
310  node.pdata->type = NodeType::Constant;
311  node.pdata->valueType = &typeid(UserType);
312  node.pdata->nElements = length;
313  node.pdata->name = "*UNNAMED CONSTANT*";
314  if(makeFeeder) {
315  node.pdata->direction = {VariableDirection::feeding, false};
316  node.pdata->mode = UpdateMode::push;
317  }
318  else {
319  node.pdata->direction = {VariableDirection::consuming, false};
320  node.pdata->mode = UpdateMode::poll;
321  }
322  return node;
323  }
