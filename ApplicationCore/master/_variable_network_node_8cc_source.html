138  VariableNetworkNode VariableNetworkNode::operator>>(VariableNetworkNode other) {
139  if(Application::getInstance().initialiseCalled) {
140  throw ChimeraTK::logic_error("Cannot make connections after Application::initialise() has been run.");
141  }
142 
143  if(pdata->direction.dir == VariableDirection::invalid) {
144  if(!other.hasOwner()) {
145  pdata->direction = {VariableDirection::feeding, false};
146  }
147  else {
148  if(other.getOwner().hasFeedingNode()) {
149  pdata->direction = {VariableDirection::consuming, false};
150  if(getType() == NodeType::Device) { // special treatment for Device-type variables:
151  // consumers are push-type
152  pdata->mode = UpdateMode::push;
153  }
154  }
155  else {
156  pdata->direction = {VariableDirection::feeding, false};
157  }
158  }
159  }
160  if(other.pdata->direction.dir == VariableDirection::invalid) {
161  if(!hasOwner()) {
162  other.pdata->direction = {VariableDirection::consuming, false};
163  if(other.getType() == NodeType::Device) { // special treatment for Device-type variables:
164  // consumers are push-type
165  other.pdata->mode = UpdateMode::push;
166  }
167  }
168  else {
169  if(getOwner().hasFeedingNode()) {
170  other.pdata->direction = {VariableDirection::consuming, false};
171  if(other.getType() == NodeType::Device) { // special treatment for Device-type variables:
172  // consumers are push-type
173  other.pdata->mode = UpdateMode::push;
174  }
175  }
176  else {
177  other.pdata->direction = {VariableDirection::feeding, false};
178  }
179  }
180  }
181  Application::getInstance().connect(*this, other);
182  return *this;
183  }

187  VariableNetworkNode VariableNetworkNode::operator[](VariableNetworkNode trigger) {
188  // check if node already has a trigger
189  if(pdata->externalTrigger.getType() != NodeType::invalid) {
190  throw ChimeraTK::logic_error("Only one external trigger per variable network is allowed.");
191  }
192 
193  // force direction of the node we are operating on to be feeding
194  if(pdata->direction.dir == VariableDirection::invalid) pdata->direction = {VariableDirection::feeding, false};
195  assert(pdata->direction.dir == VariableDirection::feeding);
196 
197  // set direction of the triggering node to be feeding, if not yet defined
198  if(trigger.pdata->direction.dir == VariableDirection::invalid)
199  trigger.pdata->direction = {VariableDirection::feeding, false};
200 
201  // check if already existing in map
202  if(pdata->nodeWithTrigger.count(trigger) > 0) {
203  return pdata->nodeWithTrigger[trigger];
204  }
205 
206  // create copy of the node
207  pdata->nodeWithTrigger[trigger].pdata = boost::make_shared<VariableNetworkNode_data>(*pdata);
208 
209  // add ourselves as a trigger receiver to the other network
210  if(!trigger.hasOwner()) {
211  Application::getInstance().createNetwork().addNode(trigger);
212  }
213  trigger.getOwner().addNodeToTrigger(pdata->nodeWithTrigger[trigger]);
214 
215  // set flag and store pointer to other network
216  pdata->nodeWithTrigger[trigger].pdata->externalTrigger = trigger;
217 
218  // return the new node
219  return pdata->nodeWithTrigger[trigger];
220  }

396  VariableNetworkNode::InitialValueMode VariableNetworkNode::hasInitialValue() const {
397  if(getDirection().dir == VariableDirection::feeding) {
398  if(getType() == NodeType::ControlSystem) {
399  return InitialValueMode::Push;
400  }
401  if(getType() == NodeType::Device || getType() == NodeType::Constant) {
402  if(getOwner().getTriggerType() == VariableNetwork::TriggerType::feeder) {
403  return InitialValueMode::Poll;
404  }
405  else {
406  assert(getOwner().getTriggerType() == VariableNetwork::TriggerType::external ||
407  getOwner().getTriggerType() == VariableNetwork::TriggerType::pollingConsumer);
408  return InitialValueMode::Push;
409  }
410  }
411  assert(getType() == NodeType::Application);
412  if(pdata->hasInitialValue) {
413  return InitialValueMode::Push;
414  }
415  else {
416  return InitialValueMode::None;
417  }
418  }
419  else {
420  auto feederInitialValue = getOwner().getFeedingNode().hasInitialValue();
421  if(getOwner().countConsumingNodes() == 1) {
422  return feederInitialValue;
423  }
424  else {
425  if(feederInitialValue != InitialValueMode::None) {
426  // More then one consuming node: FanOut in between. Need to wait for FanOut to process the initial value!
427  // If the FanOut is a ConsumingFanOut, InitialValueMode::Push doesn't hurt as that fan out is always poll-type
428  return InitialValueMode::Push;
429  }
430  else {
431  return InitialValueMode::None;
432  }
433  }
434  }
435  }
