143  VariableNetworkNode VariableNetworkNode::operator>>(VariableNetworkNode other) {
144  if(Application::getInstance().initialiseCalled) {
145  throw ChimeraTK::logic_error("Cannot make connections after Application::initialise() has been run.");
146  }
147 
148  if(pdata->direction.dir == VariableDirection::invalid) {
149  if(!other.hasOwner()) {
150  pdata->direction = {VariableDirection::feeding, false};
151  }
152  else {
153  if(other.getOwner().hasFeedingNode()) {
154  pdata->direction = {VariableDirection::consuming, false};
155  if(getType() == NodeType::Device) { // special treatment for Device-type variables:
156  // consumers are push-type
157  pdata->mode = UpdateMode::push;
158  }
159  }
160  else {
161  pdata->direction = {VariableDirection::feeding, false};
162  }
163  }
164  }
165  if(other.pdata->direction.dir == VariableDirection::invalid) {
166  if(!hasOwner()) {
167  other.pdata->direction = {VariableDirection::consuming, false};
168  if(other.getType() == NodeType::Device) { // special treatment for Device-type variables:
169  // consumers are push-type
170  other.pdata->mode = UpdateMode::push;
171  }
172  }
173  else {
174  if(getOwner().hasFeedingNode()) {
175  other.pdata->direction = {VariableDirection::consuming, false};
176  if(other.getType() == NodeType::Device) { // special treatment for Device-type variables:
177  // consumers are push-type
178  other.pdata->mode = UpdateMode::push;
179  }
180  }
181  else {
182  other.pdata->direction = {VariableDirection::feeding, false};
183  }
184  }
185  }
186  Application::getInstance().connect(*this, other);
187  return *this;
188  }

192  VariableNetworkNode VariableNetworkNode::operator[](VariableNetworkNode trigger) {
193  // check if node already has a trigger
194  if(pdata->externalTrigger.getType() != NodeType::invalid) {
195  throw ChimeraTK::logic_error("Only one external trigger per variable network is allowed.");
196  }
197 
198  // force direction of the node we are operating on to be feeding
199  if(pdata->direction.dir == VariableDirection::invalid) pdata->direction = {VariableDirection::feeding, false};
200  assert(pdata->direction.dir == VariableDirection::feeding);
201 
202  // set direction of the triggering node to be feeding, if not yet defined
203  if(trigger.pdata->direction.dir == VariableDirection::invalid)
204  trigger.pdata->direction = {VariableDirection::feeding, false};
205 
206  // check if already existing in map
207  if(pdata->nodeWithTrigger.count(trigger) > 0) {
208  return pdata->nodeWithTrigger[trigger];
209  }
210 
211  // create copy of the node
212  pdata->nodeWithTrigger[trigger].pdata = boost::make_shared<VariableNetworkNode_data>(*pdata);
213 
214  // add ourselves as a trigger receiver to the other network
215  if(!trigger.hasOwner()) {
216  Application::getInstance().createNetwork().addNode(trigger);
217  }
218  trigger.getOwner().addNodeToTrigger(pdata->nodeWithTrigger[trigger]);
219 
220  // set flag and store pointer to other network
221  pdata->nodeWithTrigger[trigger].pdata->externalTrigger = trigger;
222 
223  // return the new node
224  return pdata->nodeWithTrigger[trigger];
225  }

386  std::list<EntityOwner*> VariableNetworkNode::scanForCircularDepencency() {
387  // We are starting a new scan. Reset the indicator for already found circular dependencies.
388  detail::CircularDependencyDetectionRecursionStopper::startNewScan();
389 
390  // find the feeder of the network
391  auto feeder = getOwner().getFeedingNode();
392  auto feedingModule = feeder.getOwningModule();
393  // CS modules and device modules don't have an owning module. They stop the circle anyway. So if either the feeder or the
394  // receiver (this) don't have an owning module, there is nothing to do here.
395  if(!feedingModule || !getOwningModule()) {
396  return {};
397  }
398  assert(getDirection().dir == VariableDirection::consuming);
399 
400  auto owningModule = getOwningModule();
401  // if the entity owner is a variable group we must go up the hierarchy until we find the applciation module
402  while(owningModule->getModuleType() == EntityOwner::ModuleType::VariableGroup) {
403  auto variableGroup = static_cast<VariableGroup*>(owningModule);
404  owningModule = variableGroup->getOwner();
405  }
406  assert(owningModule->getModuleType() == EntityOwner::ModuleType::ApplicationModule);
407  auto inputModuleList = feedingModule->getInputModulesRecursively({owningModule});
408 
409  auto nInstancesFound = std::count(inputModuleList.begin(), inputModuleList.end(), owningModule);
410  assert(nInstancesFound >= 1); // the start list must not have been deleted in the call
411  // The owning module has been found again when scanning inputs recursively -> There is a circular dependency
412  if(nInstancesFound > 1) {
413  // clean up the circular network we found and return it.
414  inputModuleList.sort();
415  inputModuleList.unique();
416 
417  // Remember that we are part of a circle, and of which circle
418  pdata->circularNetworkHash = boost::hash_range(inputModuleList.begin(), inputModuleList.end());
419  // we already did the assertion that the owning module is an application module above, so we can static cast here
420  auto applicationModule = static_cast<ApplicationModule*>(owningModule);
421  applicationModule->setCircularNetworkHash(pdata->circularNetworkHash);
422 
423  // Find the MetaDataPropagatingRegisterDecorator which is involed and set the _isCurularInput flag
424  auto internalTargetElements = getAppAccessorNoType().getInternalElements();
425  // This is a list of all the nested decorators, so we will find the right point to cast
426  for(auto& elem : internalTargetElements) {
427  auto flagProvider = boost::dynamic_pointer_cast<MetaDataPropagationFlagProvider>(elem);
428  if(flagProvider) {
429  flagProvider->_isCircularInput = true;
430  }
431  }
432 
433  return inputModuleList;
434  }
435 
436  // No circlular network. Return an empty list.
437  return {};
438  }
