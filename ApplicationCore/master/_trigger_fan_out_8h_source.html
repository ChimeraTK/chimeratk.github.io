76  void run() {
77  Application::registerThread("TrFO" + externalTrigger->getName());
78  Application::testableModeLock("start");
79  testableModeReached = true;
80 
81  ChimeraTK::VersionNumber version = Application::getInstance().getStartVersion();
82 
83  // If trigger gets an initial value pushed, read it (otherwise we would trigger twice at application start)
84  auto hasInitialValue = _network.getFeedingNode().getExternalTrigger().hasInitialValue();
85  if(hasInitialValue == VariableNetworkNode::InitialValueMode::Push) {
86  externalTrigger->read();
87  version = externalTrigger->getVersionNumber();
88  }
89 
90  while(true) {
91  // receive data. We need to catch exceptions here, since the ExceptionHandlingDecorator cannot do this for us
92  // inside a TransferGroup, if the exception is thrown inside doReadTransfer() (as it is directly called on the
93  // lowest-level TransferElements inside the group).
94  auto lastValidity = DataValidity::ok;
95  retry:
96  try {
97  if(!_deviceModule.device.isOpened()) {
98  Application::getInstance().testableModeUnlock("waitForDeviceOpen");
99  boost::this_thread::sleep(boost::posix_time::millisec(DeviceOpenTimeout));
100  Application::getInstance().testableModeLock("waitForDeviceOpen");
101  goto retry;
102  }
103  transferGroup.read();
104  }
105  catch(ChimeraTK::runtime_error& e) {
106  // send the data to the consumers
107  if(lastValidity == DataValidity::ok) {
108  lastValidity = DataValidity::faulty;
109  boost::fusion::for_each(fanOutMap.table, SendDataToConsumers(version, lastValidity));
110  }
111  _deviceModule.reportException(e.what());
112  goto retry;
113  }
114  // send the version number to the consumers
115  boost::fusion::for_each(fanOutMap.table, SendDataToConsumers(version));
116  // wait for external trigger (exception handling is done here by the decorator)
117  boost::this_thread::interruption_point();
118  Profiler::stopMeasurement();
119  externalTrigger->read();
120  Profiler::startMeasurement();
121  boost::this_thread::interruption_point();
122  version = externalTrigger->getVersionNumber();
123  }
124  }

133  template<typename PAIR>
134  void operator()(PAIR& pair) const {
135  auto theMap = pair.second; // map of feeder to FeedingFanOut (i.e. part of
136  // the fanOutMap)
137 
138  // iterate over all feeder/FeedingFanOut pairs
139  for(auto& network : theMap) {
140  auto feeder = network.first;
141  auto fanOut = network.second;
142  fanOut->setDataValidity(_validity);
143  fanOut->accessChannel(0).swap(feeder->accessChannel(0));
144  bool dataLoss = fanOut->writeDestructively(_version);
145  if(dataLoss) Application::incrementDataLossCounter();
146  // no need to swap back since we don't need the data
147  }
148  }
