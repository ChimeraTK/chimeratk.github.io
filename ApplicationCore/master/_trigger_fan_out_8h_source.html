70  void run() {
71  Application::registerThread("TrFO" + externalTrigger->getName());
72  Application::testableModeLock("start");
73  testableModeReached = true;
74 
75  ChimeraTK::VersionNumber version = Application::getInstance().getStartVersion();
76 
77  // Wait for the initial value of the trigger. There always will be one, and if we don't read it here we would trigger the loop twice.
78  externalTrigger->read();
79  version = externalTrigger->getVersionNumber();
80 
81  // Wait until the device has been initialised for the first time. This means it
82  // has been opened, and the check in TransferGroup::read() will not throw a logic_error
83  // We don't have to store the lock. Just need it as a synchronisation point.
84  // But we have to increase the testable mode counter because we don't want to fall out of testable mode at this point already.
85  if(Application::getInstance().testableMode) ++Application::getInstance().testableMode_deviceInitialisationCounter;
86  Application::testableModeUnlock("WaitInitialValueLock");
87  (void)_deviceModule.waitForInitialValues();
88  Application::testableModeLock("Enter while loop");
89  if(Application::getInstance().testableMode) --Application::getInstance().testableMode_deviceInitialisationCounter;
90 
91  while(true) {
92  transferGroup.read();
93  // send the version number to the consumers
94  boost::fusion::for_each(fanOutMap.table, SendDataToConsumers(version, externalTrigger->dataValidity()));
95 
96  // wait for external trigger
97  boost::this_thread::interruption_point();
98  Profiler::stopMeasurement();
99  externalTrigger->read();
100  Profiler::startMeasurement();
101  boost::this_thread::interruption_point();
102  version = externalTrigger->getVersionNumber();
103  }
104  }

113  template<typename PAIR>
114  void operator()(PAIR& pair) const {
115  auto theMap = pair.second; // map of feeder to FeedingFanOut (i.e. part of
116  // the fanOutMap)
117 
118  // iterate over all feeder/FeedingFanOut pairs
119  for(auto& network : theMap) {
120  auto feeder = network.first;
121  auto fanOut = network.second;
122  fanOut->setDataValidity((_triggerValidity == DataValidity::ok && feeder->dataValidity() == DataValidity::ok) ?
123  DataValidity::ok :
124  DataValidity::faulty);
125  fanOut->accessChannel(0).swap(feeder->accessChannel(0));
126  // don't use write destructively. In case of an exception we still need the data for the next read (see Exception Handling spec B.2.2.6)
127  bool dataLoss = fanOut->write(_version);
128  if(dataLoss) Application::incrementDataLossCounter(fanOut->getName());
129  // swap the data back to the feeder so we have a valid copy there.
130  fanOut->accessChannel(0).swap(feeder->accessChannel(0));
131  }
132  }
