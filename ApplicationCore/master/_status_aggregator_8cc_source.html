42  void StatusAggregator::scanAndPopulateFromHierarchyLevel(EntityOwner& module, const std::string& namePrefix) {
43  // Search for StatusOutputs to aggregate
44  for(auto& node : module.getAccessorList()) {
45  // Filter required tags
46  // Note: findTag() cannot be used instead, since the search must be done on the original C++ hierarchy. Otherwise
47  // it is not possible to identify which StatusOutputs are aggregated by other StatusAggregators.
48  const auto& tags = node.getTags();
49  if(tags.find(StatusOutput::tagStatusOutput) == tags.end()) {
50  // StatusOutput's reserved tag not present: not a StatusOutput
51  continue;
52  }
53  bool skip = false;
54  for(const auto& tag : _tagsToAggregate) {
55  // Each tag attached to this StatusAggregator must be present at all StatusOutputs to be aggregated
56  if(tags.find(tag) == tags.end()) {
57  skip = true;
58  break;
59  }
60  }
61  if(skip) continue;
62 
63  // All variables with the StatusOutput::tagStatusOutput need to be feeding, otherwise someone has used the tag
64  // wrongly
65  if(node.getDirection().dir != VariableDirection::feeding) {
66  throw ChimeraTK::logic_error("BUG DETECTED: StatusOutput's reserved tag has been found on a consumer.");
67  }
68 
69  // Create matching input for the found StatusOutput of the other StatusAggregator
70  _inputs.emplace_back(this, node.getName(), "", std::unordered_set<std::string>{tagInternalVars});
71  node >> _inputs.back();
72  }
73 
74  // Search for StatusAggregators among submodules
75  const auto& ml = module.getSubmoduleList();
76  for(const auto& submodule : ml) {
77  // do nothing, if it is *this
78  if(submodule == this) continue;
79 
80  auto* aggregator = dynamic_cast<StatusAggregator*>(submodule);
81  if(aggregator != nullptr) {
82  // never aggregate on the same level
83  if(aggregator->getOwner() == getOwner()) {
84  continue;
85  }
86 
87  // if another StatusAggregator has been found, check tags
88  bool skip = false;
89  const auto& tags = aggregator->_tagsToAggregate;
90  for(const auto& tag : _tagsToAggregate) {
91  // Each tag attached to this StatusAggregator must be present at all StatusOutputs to be aggregated
92  if(tags.find(tag) == tags.end()) {
93  skip = true;
94  break;
95  }
96  }
97  if(skip) continue;
98 
99  // aggregate the StatusAggregator's result
100  auto node = VariableNetworkNode(aggregator->_output.status);
101  _inputs.emplace_back(this, node.getName(), "", std::unordered_set<std::string>{tagInternalVars});
102  node >> _inputs.back();
103  return;
104  }
105  }
106 
107  // If no (matching) StatusAggregator is found, recurse into sub-modules
108  for(const auto& submodule : ml) {
109  if(dynamic_cast<StatusAggregator*>(submodule) != nullptr) continue; // StatusAggregators are already handled
110  scanAndPopulateFromHierarchyLevel(*submodule, namePrefix + "/" + submodule->getName());
111  }
112  }

116  int StatusAggregator::getPriority(StatusOutput::Status status) {

118 
119  const std::map<PriorityMode, std::map<Status, int32_t>> map_priorities{
120  {PriorityMode::fwko, {{Status::OK, 1}, {Status::FAULT, 3}, {Status::OFF, 0}, {Status::WARNING, 2}}},
121  {PriorityMode::fwok, {{Status::OK, 0}, {Status::FAULT, 3}, {Status::OFF, 1}, {Status::WARNING, 2}}},
122  {PriorityMode::ofwk, {{Status::OK, 0}, {Status::FAULT, 2}, {Status::OFF, 3}, {Status::WARNING, 1}}},
123  {PriorityMode::fw_warn_mixed, {{Status::OK, -1}, {Status::FAULT, 3}, {Status::OFF, -1}, {Status::WARNING, 2}}}};
124 
125  return map_priorities.at(_mode).at(status);

130  void StatusAggregator::mainLoop() {
131  auto rag = readAnyGroup();
132  DataValidity lastStatusValidity = DataValidity::ok;
133  while(true) {
134  // find highest priority status of all inputs
135  StatusOutput::Status status;
136  bool statusSet = false; // flag whether status has been set from an input already
137  for(auto& input : _inputs) {
138  auto prio = getPriority(input);
139  if(!statusSet || prio > getPriority(status)) {
140  status = input;
141  statusSet = true;
142  }
143  else if(prio == -1) { // -1 means, we need to warn about mixed values
144  if(getPriority(status) == -1 && input != status) {
145  status = StatusOutput::Status::WARNING;
146  }
147  }
148  }
149  assert(statusSet);
150 
151  // write status only if changed, but always write initial value out
152  if(status != _output.status || _output.status.getVersionNumber() == VersionNumber{nullptr} ||
153  getDataValidity() != lastStatusValidity) {
154  _output.status = status;
155  _output.status.write();
156  lastStatusValidity = getDataValidity();
157  }
158 
159  // wait for changed inputs
160  rag.readAny();
161  }
162  }
