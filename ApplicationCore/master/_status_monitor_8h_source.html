354  /*******************************************************************************************************************/
355  template<typename T>
356  void RangeMonitor<T>::mainLoop() {
357  // If there is a change either in value monitored or in requiredValue, the status is re-evaluated
358  ReadAnyGroup group{watch.value, disable.value, warningLowerThreshold.value, warningUpperThreshold.value,
359  faultLowerThreshold.value, faultUpperThreshold.value};
360 
361  while(true) {
362  if(disable.value) {
363  setStatus(StatusOutput::Status::OFF);
364  }
365  // Check for fault limits first. Like this they supersede the warning,
366  // even if they are stricter then the warning limits (mis-configuration)
367  else if(watch.value <= faultLowerThreshold.value || watch.value >= faultUpperThreshold.value) {
368  setStatus(StatusOutput::Status::FAULT);
369  }
370  else if(watch.value <= warningLowerThreshold.value || watch.value >= warningUpperThreshold.value) {
371  setStatus(StatusOutput::Status::WARNING);
372  }
373  else {
374  setStatus(StatusOutput::Status::OK);
375  }
376  group.readAny();
377  }
