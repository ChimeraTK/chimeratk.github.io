58 
59  ctk::ScalarPushInput<int> a{this, "a", "", ""};
60  ctk::ScalarPushInput<int> b{this, "b", "", ""};
61  ctk::ScalarOutput<int> circleResult{this, "circleResult", "", ""};
62 
63  void prepare() override { writeAll(); }
64 
65  void mainLoop() override {
66  // The circular inputs always are both coming as a pair, but we only want to write once.
67  // Hence we only put one of them into the ReadAnyGroup and always read the second one manually if the first one
68  // is read by the group.
69  ctk::ReadAnyGroup rag({a, b, inputGroup.circularInput1});
70 
71  while(true) {
72  auto id = rag.readAny();
73 
74  // A module with circular inputs and readAny must always actively break the circle. Otherwise for each external
75  // input and n-1 internal inputs and additional data element is inserted into the circle, which will let queues
76  // run over and re-trigger the circle all the time.
77  // This is a very typical scenario for circular connections: A module gets some input, triggers a helper module
78  // which calculates a value that is read back by the first module, and then the first module continues without
79  // re-triggering the circle.
80 
81  assert((id == a.getId() || id == b.getId()) || id == inputGroup.circularInput1.getId() ||
82  id == circularInput2.getId());
83 
84  if(id == inputGroup.circularInput1.getId()) {
85  // Read the other circular input as well. They always come in pairs.
86  circularInput2.read();
87  }
88 
89  if(id == a.getId() || id == b.getId()) {
90  circularOutput1 = static_cast<int>(inputGroup.circularInput1) + a;
91  outputGroup.circularOutput2 = static_cast<int>(circularInput2) + b;
92 
93  circularOutput1.write();
94  outputGroup.circularOutput2.write();
95  }
96  else { // new data is from the circular inputs
97  circleResult = static_cast<int>(inputGroup.circularInput1) + circularInput2;
98  circleResult.write();
99  }
100 
101  } // while(true)
102  } // mainLoop
103 }; // ModuleA
