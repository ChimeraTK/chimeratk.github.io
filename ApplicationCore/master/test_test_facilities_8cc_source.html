31 #define CHECK_TIMEOUT(condition, maxMilliseconds) \
32  { \
33  std::chrono::steady_clock::time_point t0 = std::chrono::steady_clock::now(); \
34  while(!(condition)) { \
35  bool timeout_reached = (std::chrono::steady_clock::now() - t0) > std::chrono::milliseconds(maxMilliseconds); \
36  BOOST_CHECK(!timeout_reached); \
37  if(timeout_reached) break; \
38  usleep(1000); \
39  } \
40  }

81  ctk::ScalarPushInput<T> v2{this, "v2", "cm", "Input 2 for testing"};
82  ctk::ScalarPushInput<T> v3{this, "v3", "cm", "Input 3 for testing"};
83  ctk::ScalarPushInput<T> v4{this, "v4", "cm", "Input 4 for testing"};
84  };
85  Inputs inputs{this, "inputs", "A group of inputs"};
86  ctk::ScalarOutput<T> value{this, "value", "cm", "The last value received from any of the inputs"};
87  ctk::ScalarOutput<uint32_t> index{
88  this, "index", "", "The index (1..4) of the input where the last value was received"};
89 
90  void prepare() override {
91  incrementDataFaultCounter(); // foce all outputs to invalid
92  writeAll();
93  decrementDataFaultCounter(); // validity according to input validity
94  }
95 
96  void mainLoop() override {
97  auto group = inputs.readAnyGroup();
98  while(true) {
99  auto justRead = group.readAny();
100  if(inputs.v1.getId() == justRead) {
101  index = 1;
102  value = (T)inputs.v1;
103  }
104  else if(inputs.v2.getId() == justRead) {
105  index = 2;
106  value = (T)inputs.v2;
107  }
108  else if(inputs.v3.getId() == justRead) {
109  index = 3;
110  value = (T)inputs.v3;
111  }
112  else if(inputs.v4.getId() == justRead) {
113  index = 4;
114  value = (T)inputs.v4;
115  }
116  else {
117  index = 0;
118  value = 0;
119  }
120  usleep(10000); // wait some extra time to make sure we are really blocking
121  // the test procedure thread
122  index.write();
123  value.write();
124  }
125  }
