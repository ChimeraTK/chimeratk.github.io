110  VirtualModule EntityOwner::findTag(const std::string& tag) const {
111  // create new module to return
112  VirtualModule module{_name, _description, getModuleType()};
113 
114  // add everything matching the tag to the virtual module and return it
115  if(this == &Application::getInstance()) {
116  // if this module is the top-level application, we need special treatment for HierarchyModifier::moveToRoot
117  findTagAndAppendToModule(module, tag, false, true, false, module);
118  }
119  else {
120  // Not the top-level module: Things that are moved to the top-level are simply discarded
121  VirtualModule discard("discarded", "", ModuleType::Invalid);
122  findTagAndAppendToModule(module, tag, false, true, false, discard);
123  }
124 
125  // remove all empty virtual modules
126  module.stripEmptyChildsRecursive();
127 
128  return module;
129  }

163  // The function adds virtual versions of the EntityOwner itself anf all its children to a virtual module (parent).
164  void EntityOwner::findTagAndAppendToModule(VirtualModule& virtualParent, const std::string& tag,
165  bool eliminateAllHierarchies, bool eliminateFirstHierarchy, bool negate, VirtualModule& root) const {
166  // It might be that it is requested to hide ourselves. In this case we do not add
167  // ourselves but directly put the children into the parent (or grand parent, depending on the hierarchy modifier).
168  // So we store which module to add to (either a new virtual module, the virtual parent or virtual grand parent)
169  VirtualModule* moduleToAddTo;
170 
171  // Check whether we are supposed to hide. This can be the case if
172  // * the hierarchy modifier says so (hideThis or oneUpAndHide -> getEliminateHierarchy is true)
173  // * flatten was called (eliminateAllHierarchies is true)
174  // * the virtual hierarchy level of virtualMe already exists elsewhere and we just add our children to it
175  if(!getEliminateHierarchy() && !eliminateAllHierarchies && !eliminateFirstHierarchy) {
176  // add the virtualMe to the right owner
177  if(_hierarchyModifier == HierarchyModifier::moveToRoot) {
178  moduleToAddTo = &root.createAndGetSubmodule(_name);
179  }
180  else if((_hierarchyModifier == HierarchyModifier::oneLevelUp) ||
181  (_hierarchyModifier == HierarchyModifier::oneUpAndHide)) {
182  auto virtualGrandparent = dynamic_cast<VirtualModule*>(virtualParent.getOwner());
183  if(virtualGrandparent == nullptr) {
184  throw logic_error(std::string("Module ") + virtualParent.getName() +
185  ": cannot have hierarchy modifier 'oneLevelUp' or 'oneUpAndHide' in root of the application." +
186  "\nNon-virtual path to the offending module: " + getQualifiedName());
187  }
188  moduleToAddTo = &virtualGrandparent->createAndGetSubmodule(_name);
189  }
190  else {
191  moduleToAddTo = &virtualParent.createAndGetSubmodule(_name);
192  }
193  }
194  else {
195  // We are hiding. Find the correct ancestor to add to:
196  if(_hierarchyModifier == HierarchyModifier::oneUpAndHide) {
197  // don't just hide but also move one level up -> add to the grandparent
198  moduleToAddTo = dynamic_cast<VirtualModule*>(virtualParent.getOwner());
199  if(moduleToAddTo == nullptr) {
200  throw logic_error(std::string("Module ") + virtualParent.getName() +
201  ": cannot have hierarchy modifier 'oneLevelUp' or 'oneUpAndHide' in root of the application." +
202  "\nNon-virtual path to the offending module: " + getQualifiedName());
203  }
204  }
205  else {
206  // just hide -> add to the parent (could also an already existing object with the same hierarchy level as
207  // virtualMe, where we just add to)
208  moduleToAddTo = &virtualParent;
209  }
210  }
211 
212  // This should have already been caught by previous checks.
213  assert(moduleToAddTo != nullptr);
214 
215  // add nodes to the module if matching the tag
216  std::regex expr(tag);
217  for(auto& node : getAccessorList()) {
218  bool addNode = false;
219  for(auto& nodeTag : node.getTags()) {
220  if(std::regex_match(nodeTag, expr)) {
221  addNode = true;
222  break;
223  }
224  }
225  if(node.getTags().size() == 0)
226  if(std::regex_match("", expr)) addNode = true; // check if empty tag matches, if no tag applied to node
227  if(negate) addNode = !addNode;
228  if(addNode) moduleToAddTo->registerAccessor(node);
229  }
230 
231  // iterate through submodules
232  // nomencature: submodule = EntityOwner in C++ hierarchy, child = VirtualModule in virtual hierarcy
233  for(auto submodule : getSubmoduleList()) {
234  // Duplicate childs will not be created, since createAndGetSubmodule() prevents that.
235  submodule->findTagAndAppendToModule(*moduleToAddTo, tag, eliminateAllHierarchies, false, negate, root);
236  }
237  }
