104  VirtualModule EntityOwner::findTag(const std::string& tag) const {
105  // create new module to return
106  VirtualModule module{_name, _description, getModuleType()};
107 
108  // add everything matching the tag to the virtual module and return it
109  if(this == &Application::getInstance()) {
110  // if this module is the top-level application, we need special treatment for HierarchyModifier::moveToRoot
111  findTagAndAppendToModule(module, tag, false, true, false, module, nullptr);
112  }
113  else {
114  // Not the top-level module: Things that are moved to the top-level are simply discarded
115  VirtualModule discard("discarded", "", ModuleType::Invalid);
116  findTagAndAppendToModule(module, tag, false, true, false, discard, nullptr);
117  }
118 
119  return module;
120  }

143  // The function adds virtual versions of the EntityOwner itself anf all its children to a virtual module (parent).
144  void EntityOwner::findTagAndAppendToModule(VirtualModule& virtualParent, const std::string& tag,
145  bool eliminateAllHierarchies, bool eliminateFirstHierarchy, bool negate, VirtualModule& root,
146  VirtualModule* virtualGrandparent) const {
147  // This function adds the EntityOwner "this" (C++ hierarchy) and its children to the virtual hierarchy.
149  VirtualModule virtualMe{_name, _description, getModuleType()};
150  // It might be that it is requested to hide ourselves. In this case we do not add
151  // ourselves but directly put the children into the parent (or grand parent, depending on the hierarchy modifier).
152  // So we store which module to add to (either virtualMe, the virtual parent or virtual grand parent)
153  VirtualModule* moduleToAddTo;
154  VirtualModule* parentOfModuleToAddTo;
155 
156  bool needToAddMyself = false;
157  // Check whether we are supposed to hide. This can be the case if
158  // * the hierarchy modifier says so (hideThis or oneUpAndHide -> getEliminateHierarchy is true)
159  // * flatten was called (eliminateAllHierarchies is true)
160  // * the virtual hierarchy level of virtualMe already exists elsewhere and we just add our children to it
161  if(!getEliminateHierarchy() && !eliminateAllHierarchies && !eliminateFirstHierarchy) {
162  // We are not hiding ourselves.
163  moduleToAddTo = &virtualMe;
164  parentOfModuleToAddTo = &virtualParent;
165  needToAddMyself = true;
166  }
167  else {
168  // We are hiding. Find the correct ancestor to add to:
169  if(_hierarchyModifier == HierarchyModifier::oneUpAndHide) {
170  // don't just hide but also move one level up -> add to the grandparent
171  moduleToAddTo = virtualGrandparent;
172  parentOfModuleToAddTo = nullptr; //FIXME: oneLevelUp and oneUpAndHide cannit be nested
173  }
174  else {
175  // just hide -> add to the parent (could also an already existing objet with the same hierarchy level as virtualMe, where we just add to)
176  moduleToAddTo = &virtualParent;
177  parentOfModuleToAddTo = virtualGrandparent;
178  }
179  }
180 
181  // add nodes to the module if matching the tag
182  std::regex expr(tag);
183  for(auto node : getAccessorList()) {
184  bool addNode = false;
185  for(auto& nodeTag : node.getTags()) {
186  if(std::regex_match(nodeTag, expr)) {
187  addNode = true;
188  break;
189  }
190  }
191  if(node.getTags().size() == 0)
192  if(std::regex_match("", expr)) addNode = true; // check if empty tag matches, if no tag applied to node
193  if(negate) addNode = !addNode;
194  if(addNode) moduleToAddTo->registerAccessor(node);
195  }
196 
197  // iterate through submodules
198  // nomencature: submodule = EntityOwner in C++ hierarchy, child = VirtualModule in virtual hierarcy
199  for(auto submodule : getSubmoduleList()) {
200  // check if submodule already exists by this name and its hierarchy should
201  // not be eliminated
202  if(!submodule->getEliminateHierarchy() && moduleToAddTo->hasSubmodule(submodule->getName())) {
203  // exists: add to the existing child
204  auto* existingChild = dynamic_cast<VirtualModule*>(moduleToAddTo->getSubmodule(submodule->getName()));
205  assert(existingChild != nullptr);
206  submodule->findTagAndAppendToModule(
207  // true = eliminateFirstHierarchy, just add to existingChild
208  *existingChild, tag, eliminateAllHierarchies, true, negate, root, parentOfModuleToAddTo);
209  }
210  else {
211  // does not yet exist: add as new submodule to the current module
212  // (is also the case if the submodule is hidden and thus not added to the existingChild but the moduleToAddTo)
213  submodule->findTagAndAppendToModule(
214  *moduleToAddTo, tag, eliminateAllHierarchies, false, negate, root, parentOfModuleToAddTo);
215  }
216  }
217 
218  if(needToAddMyself) {
219  if(virtualMe.getAccessorList().size() > 0 || virtualMe.getSubmoduleList().size() > 0) {
220  if(_hierarchyModifier == HierarchyModifier::moveToRoot) {
221  root.addSubModule(virtualMe);
222  }
223  else if((_hierarchyModifier == HierarchyModifier::oneLevelUp) ||
224  (_hierarchyModifier == HierarchyModifier::oneUpAndHide)) {
225  if(virtualGrandparent != nullptr) { // the root does not have a parent. Could be nullptr.
226  virtualGrandparent->addSubModule(virtualMe);
227  }
228  else {
229  throw logic_error(std::string("Module ") + virtualParent.getName() +
230  ": cannot have hierarchy modifier 'oneLevelUp' or oneUpAndHide in root of the application.");
231  }
232  }
233  else {
234  virtualParent.addSubModule(virtualMe);
235  }
236  }
237  }
