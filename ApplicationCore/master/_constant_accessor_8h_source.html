58  ConstantAccessor<UserType>::ConstantAccessor(UserType value, size_t length, AccessModeFlags accessModeFlags)
59  : ChimeraTK::NDRegisterAccessor<UserType>("UnnamedConstantAccessor", accessModeFlags), _value(length, value) {
60  ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D.resize(1);
61  ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0] = _value;
62 
63  if(TransferElement::_accessModeFlags.has(AccessMode::wait_for_new_data)) {
64  // This implementation does not have a data transport queue, hence _readQueue
65  // is not set up as a continuation. We directly make it a cppext::future_queue
66  TransferElement::_readQueue = cppext::future_queue<void>(3); // minimum required length is 2
67  // Push once into the queue for the initial value.
68  TransferElement::_readQueue.push();
69  }
70  }

74  template<typename UserType>
75  void ConstantAccessor<UserType>::doPostRead(TransferType /*type*/, bool updateUserBuffer) {
76  // - updateUserBuffer is false for further calls to readLatest with wait_for_new_data.
77  // In this case the user buffer must not be touched.
78  // - updateUserBuffer is true for all calls without wait_for_new_data. The user buffer must
79  // be overwritten.
80  if(updateUserBuffer) {
81  ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0] = _value;
82  // It is OK to generate the version number just here since the read transfer is empty anyway.
83  this->_versionNumber = {};
84  this->_dataValidity = DataValidity::ok; // the constant is always valid by definiton
85  }
