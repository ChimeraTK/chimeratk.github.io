505  void ConfigParser::parseModule(const xmlpp::Element* element, std::string parent_name) {
506  auto module_name = (element->get_name() == "configuration") // root node gets special treatment
507  ?
508  "" :
509  element->get_attribute("name")->get_value() + "/";
510 
511  parent_name += module_name;
512 
513  for(const auto& child : element->get_children()) {
514  element = dynamic_cast<const xmlpp::Element*>(child);
515  if(!element) {
516  continue; // ignore if not an element (e.g. comment)
517  }
518  else if(isVariable(element)) {
519  variableList_->emplace_back(prefix(parent_name, parseVariable(element)));
520  }
521  else if(isArray(element)) {
522  arrayList_->emplace_back(prefix(parent_name, parseArray(element)));
523  }
524  else if(isModule(element)) {
525  parseModule(element, parent_name);
526  }
527  else {
528  error("Unknown tag: " + element->get_name());
529  }
530  }

619  std::map<size_t, std::string> ConfigParser::gettArrayValues(const xmlpp::Element* element) {
620  bool valueFound = false;
621  std::map<size_t, std::string> values;
622 
623  for(const auto& valueChild : element->get_children()) {
624  const xmlpp::Element* valueElement = dynamic_cast<const xmlpp::Element*>(valueChild);
625  if(!valueElement) continue; // ignore comments etc.
626  validateValueNode(valueElement);
627  valueFound = true;
628 
629  auto index = valueElement->get_attribute("i");
630  auto value = valueElement->get_attribute("v");
631 
632  // get index as number and store value as a string
633  size_t intIndex;
634  try {
635  intIndex = std::stoi(index->get_value());
636  }
637  catch(std::exception& e) {
638  error("Cannot parse string '" + std::string(index->get_value()) + "' as an index number: " + e.what());
639  }
640  values[intIndex] = value->get_value();
641  }
642  // make sure there is at least one value
643  if(!valueFound) {
644  error("Each variable must have a value, either specified as an attribute or as child tags.");
645  }
646  return values;
647  }
