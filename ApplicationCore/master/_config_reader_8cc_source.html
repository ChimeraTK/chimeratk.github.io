509  void ConfigParser::parseModule(const xmlpp::Element* element, std::string parent_name) {
510  auto module_name = (element->get_name() == "configuration") // root node gets special treatment
511  ?
512  "" :
513  element->get_attribute("name")->get_value() + "/";
514 
515  parent_name += module_name;
516 
517  for(const auto& child : element->get_children()) {
518  element = dynamic_cast<const xmlpp::Element*>(child);
519  if(!element) {
520  continue; // ignore if not an element (e.g. comment)
521  }
522  else if(isVariable(element)) {
523  variableList_->emplace_back(prefix(parent_name, parseVariable(element)));
524  }
525  else if(isArray(element)) {
526  arrayList_->emplace_back(prefix(parent_name, parseArray(element)));
527  }
528  else if(isModule(element)) {
529  parseModule(element, parent_name);
530  }
531  else {
532  error("Unknown tag: " + element->get_name());
533  }
534  }
535  }

623  std::map<size_t, std::string> ConfigParser::gettArrayValues(const xmlpp::Element* element) {
624  bool valueFound = false;
625  std::map<size_t, std::string> values;
626 
627  for(const auto& valueChild : element->get_children()) {
628  const xmlpp::Element* valueElement = dynamic_cast<const xmlpp::Element*>(valueChild);
629  if(!valueElement) continue; // ignore comments etc.
630  validateValueNode(valueElement);
631  valueFound = true;
632 
633  auto index = valueElement->get_attribute("i");
634  auto value = valueElement->get_attribute("v");
635 
636  // get index as number and store value as a string
637  size_t intIndex;
638  try {
639  intIndex = std::stoi(index->get_value());
640  }
641  catch(std::exception& e) {
642  error("Cannot parse string '" + std::string(index->get_value()) + "' as an index number: " + e.what());
643  }
644  values[intIndex] = value->get_value();
645  }
646  // make sure there is at least one value
647  if(!valueFound) {
648  error("Each variable must have a value, either specified as an attribute or as child tags.");
649  }
650  return values;
651  }
