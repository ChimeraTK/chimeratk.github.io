395  void ConfigParser::parseModule(const xmlpp::Element* element, std::string parent_name) {
396  auto module_name = (element->get_name() == "configuration") // root node gets special treatment
397  ?
398  "" :
399  element->get_attribute("name")->get_value() + "/";
400 
401  parent_name += module_name;
402 
403  for(const auto& child : element->get_children()) {
404  element = dynamic_cast<const xmlpp::Element*>(child);
405  if(!element) {
406  continue; // ignore if not an element (e.g. comment)
407  }
408  else if(isVariable(element)) {
409  variableList_->emplace_back(prefix(parent_name, parseVariable(element)));
410  }
411  else if(isArray(element)) {
412  arrayList_->emplace_back(prefix(parent_name, parseArray(element)));
413  }
414  else if(isModule(element)) {
415  parseModule(element, parent_name);
416  }
417  else {
418  error("Unknown tag: " + element->get_name());
419  }
420  }
421  }

509  std::map<size_t, std::string> ConfigParser::gettArrayValues(const xmlpp::Element* element) {
510  bool valueFound = false;
511  std::map<size_t, std::string> values;
512 
513  for(const auto& valueChild : element->get_children()) {
514  const xmlpp::Element* valueElement = dynamic_cast<const xmlpp::Element*>(valueChild);
515  if(!valueElement) continue; // ignore comments etc.
516  validateValueNode(valueElement);
517  valueFound = true;
518 
519  auto index = valueElement->get_attribute("i");
520  auto value = valueElement->get_attribute("v");
521 
522  // get index as number and store value as a string
523  size_t intIndex;
524  try {
525  intIndex = std::stoi(index->get_value());
526  }
527  catch(std::exception& e) {
528  error("Cannot parse string '" + std::string(index->get_value()) + "' as an index number: " + e.what());
529  }
530  values[intIndex] = value->get_value();
531  }
532  // make sure there is at least one value
533  if(!valueFound) {
534  error("Each variable must have a value, either specified as an attribute or as child tags.");
535  }
536  return values;
537  }
