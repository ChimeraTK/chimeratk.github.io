64  void VirtualModule::connectTo(const Module &target, VariableNetworkNode trigger) const {
65 
66  // connect all direct variables of this module to their counter-parts in the right-hand-side module
67  for(auto variable : getAccessorList()) {
68  if(variable.getDirection() == VariableDirection::feeding) {
69  variable >> target(variable.getName());
70  }
71  else {
72  // use trigger?
73  if(trigger != VariableNetworkNode() && target(variable.getName()).getMode() == UpdateMode::poll
74  && variable.getMode() == UpdateMode::push ) {
75  target(variable.getName()) [ trigger ] >> variable;
76  }
77  else {
78  target(variable.getName()) >> variable;

81  }

85  submodule->connectTo(target[submodule->getName()], trigger);
