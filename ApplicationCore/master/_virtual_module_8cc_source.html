63  void VirtualModule::connectTo(const Module& target, VariableNetworkNode trigger) const {
64  // connect all direct variables of this module to their counter-parts in the
65  // right-hand-side module
66  for(auto& variable : getAccessorList()) {
67  if(variable.getDirection().dir == VariableDirection::feeding) {
68  // use trigger?
69  if(trigger != VariableNetworkNode() && target(variable.getName()).getMode() == UpdateMode::push &&
70  variable.getMode() == UpdateMode::poll) {
71  variable[trigger] >> target(variable.getName());
72  }
73  else {
74  variable >> target(variable.getName());
75  }
76  }
77  else {
78  // use trigger?
79  if(trigger != VariableNetworkNode() && target(variable.getName()).getMode() == UpdateMode::poll &&
80  variable.getMode() == UpdateMode::push) {
81  target(variable.getName())[trigger] >> variable;
82  }
83  else {
84  target(variable.getName()) >> variable;
85  }
86  }
87  }
88 
89  // connect all sub-modules to their couter-parts in the right-hand-side module
90  for(auto submodule : getSubmoduleList()) {
91  submodule->connectTo(target[submodule->getName()], trigger);
92  }
93  }

101  void VirtualModule::addSubModule(VirtualModule module) {
102  if(!hasSubmodule(module.getName())) {
103  // Submodule doesn'st exist already: register the given module as a new submodule
104  submodules.push_back(module);
105  registerModule(&(submodules.back()));
106  submodules.back()._owner = this;
107  }
108  else {
109  // Submodule does exist already: copy content into the existing submodule
110  VirtualModule theSubmodule = dynamic_cast<VirtualModule&>(submodule(module.getName()));
111  for(auto& submod : module.getSubmoduleList()) {
112  theSubmodule.addSubModule(dynamic_cast<VirtualModule&>(*submod));
113  }
114  for(auto& acc : module.getAccessorList()) {
115  theSubmodule.addAccessor(acc);
116  }
117  }
118  }

148  VirtualModule& VirtualModule::createAndGetSubmoduleRecursive(const RegisterPath& moduleName) {
149  if(moduleName == "") return *this;
150  auto slash = std::string(moduleName).find_first_of("/", 1);
151  if(slash == std::string::npos) {
152  return createAndGetSubmodule(moduleName);
153  }
154  else {
155  auto firstSubmodule = std::string(moduleName).substr(0, slash);
156  auto remainingSubmodules = std::string(moduleName).substr(slash + 1);
157  return createAndGetSubmodule(firstSubmodule).createAndGetSubmoduleRecursive(remainingSubmodules);
158  }
159  }
