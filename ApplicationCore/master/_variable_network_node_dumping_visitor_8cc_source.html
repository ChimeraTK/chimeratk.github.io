17  void VariableNetworkNodeDumpingVisitor::dispatch(const VariableNetworkNode& t) {
18  if(t.getType() == NodeType::Application) stream() << " type = Application ('" << t.getQualifiedName() << "')";
19  if(t.getType() == NodeType::ControlSystem) stream() << " type = ControlSystem ('" << t.getPublicName() << "')";
20  if(t.getType() == NodeType::Device)
21  stream() << " type = Device (" << t.getDeviceAlias() << ": " << t.getRegisterName() << ")";
22  if(t.getType() == NodeType::TriggerReceiver) stream() << " type = TriggerReceiver";
23  if(t.getType() == NodeType::Constant) stream() << " type = Constant";
24  if(t.getType() == NodeType::invalid) stream() << " type = **invalid**";
25 
26  if(t.getMode() == UpdateMode::push) stream() << _separator << "pushing";
27  if(t.getMode() == UpdateMode::poll) stream() << _separator << "polling";
28  if(t.getDirection().withReturn) stream() << _separator << "with return";
29  stream() << _separator;
30 
31  stream() << "data type: " << boost::core::demangle(t.getValueType().name());
32  stream() << _separator;
33  stream() << "length: " << t.getNumberOfElements();
34  stream() << _separator;
35 
36  stream() << "[ptr: " << &(*(t.pdata)) << "]";
37  stream() << _separator;
38 
39  stream() << "tags: [";
40  bool first = true;
41  for(auto& tag : t.getTags()) {
42  if(!first) stream() << ",";
43  stream() << tag;
44  first = false;
45  }
46  stream() << "]";
47  stream() << _separator;
48 
49  if(t.isCircularInput()) {
