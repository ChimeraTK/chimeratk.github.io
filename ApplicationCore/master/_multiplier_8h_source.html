15  struct ConstMultiplier : public ApplicationModule {
16  ConstMultiplier(ModuleGroup* owner, const std::string& name, const std::string& description, double factor)
17  : ApplicationModule(owner, name, ""), input(this, "input", "", NELEMS, description),
18  output(this, "output", "", NELEMS, description), _factor(factor) {}
19 
20  ArrayPushInput<InputType> input;
21  ArrayOutput<OutputType> output;
22 
23  double _factor;
24 
25  void mainLoop() override {
26  while(true) {
27  // scale value (with rounding, if integral type)
28  if(!std::numeric_limits<OutputType>::is_integer) {
29  for(size_t i = 0; i < NELEMS; ++i) output[i] = input[i] * _factor;
30  }
31  else {
32  for(size_t i = 0; i < NELEMS; ++i) output[i] = std::round(input[i] * _factor);
33  }
34 
35  // write scaled value
36  output.write();
37 
38  // wait for new input value (at the end, since we want to process the
39  // initial values first)
40  input.read();
41  }
42  }
43  };
