23  ArrayPushInput<InputType> input;
24  ArrayOutput<OutputType> output;
25 
26  double _factor;
27 
28  void mainLoop() {
29  while(true) {
30  // scale value (with rounding, if integral type)
31  if(!std::numeric_limits<OutputType>::is_integer) {
32  for(size_t i = 0; i < NELEMS; ++i) output[i] = input[i] * _factor;
33  }
34  else {
35  for(size_t i = 0; i < NELEMS; ++i) output[i] = std::round(input[i] * _factor);
36  }
37 
38  // write scaled value
39  output.write();
40 
41  // wait for new input value (at the end, since we want to process the
42  // initial values first)
43  input.read();
44  }
45  }
46  };
