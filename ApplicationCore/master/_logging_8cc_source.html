78 LoggingModule::LoggingModule(ctk::EntityOwner* owner, const std::string& name, const std::string& description,
79  ctk::HierarchyModifier hierarchyModifier, const std::unordered_set<std::string>& tags)
80 : ApplicationModule(owner, name, description, hierarchyModifier, tags) {
81  for(auto tag : tags) {
82  auto virtualLogging = getOwner()->findTag(tag);
83  auto list = virtualLogging.getAccessorListRecursive();
84  for(auto it = list.begin(); it != list.end(); ++it) {
85  // do not add the module itself
86  if(it->getOwningModule() == this) continue;
87  try {
88  // virtualLogging.getQualifiedName() returns the name of the app, e.g. /test and we remove that from the module
89  // name , e.g. /test/MyModule
90  auto moduleName =
91  it->getOwningModule()->getQualifiedName().substr(virtualLogging.getQualifiedName().length() + 1);
92  auto msgSource = std::find(sources.begin(), sources.end(), moduleName);
93  if(msgSource == sources.end()) {
94  // Create new MessageSource
95  auto acc = getAccessorPair(moduleName);
96  (*it) >> acc;
97  std::cout << "Registered module " << it->getOwningModule()->getQualifiedName() << " for logging."
98  << std::endl;
99  }
100  else {
101  // Connect variables that was not connected when creating the MessageSource
102  // On creation either alias or the message could have been connected -> now connect the other one
103  if(it->getName() == "alias") {
104  (*it) >> msgSource->data.alias;
105  }
106  else if(it->getName() == "message") {
107  (*it) >> msgSource->data.msg;
108  }
109  }
110  }
111  catch(ChimeraTK::logic_error& e) {
112  std::cerr << "Failed to add logging module: " << it->getOwningModule()->getQualifiedName()
113  << " Error: " << e.what() << std::endl;
114  }
115  }
116  }
117  if(sources.empty()) std::cerr << "LoggingModule did not find any module that uses a Logger." << std::endl;
118 }

120 ctk::RegisterPath LoggingModule::prepareHierarchy(const ctk::RegisterPath& namePrefix) {
121  ctk::RegisterPath result;
122  // create variable group map for namePrefix if needed
123  if(groupMap.find(namePrefix) == groupMap.end()) {
124  // search for existing parent (if any)
125  auto parentPrefix = namePrefix;
126  while(groupMap.find(parentPrefix) == groupMap.end()) {
127  if(parentPrefix == "/") break; // no existing parent found
128  parentPrefix = std::string(parentPrefix).substr(0, std::string(parentPrefix).find_last_of("/"));
129  }
130  // create all not-yet-existing parents
131  while(true) {
132  ctk::EntityOwner* owner = this;
133  if(parentPrefix != "/") owner = &groupMap[parentPrefix];
134  auto stop = std::string(namePrefix).find_first_of("/", parentPrefix.length() + 1);
135  if(stop == std::string::npos) stop = namePrefix.length();
136  ctk::RegisterPath name = std::string(namePrefix).substr(parentPrefix.length(), stop - parentPrefix.length());
137  result = parentPrefix;
138  parentPrefix /= name;
139  if(parentPrefix == namePrefix) {
140  // do not add the last hierachy level -> will be added by the MessageSource
141  break;
142  }
143  groupMap[parentPrefix] = ctk::VariableGroup(owner, std::string(name).substr(1), "");
144  }
145  }
146  return result;
147 }

156  messageCounter--;
157  tmpLog = tmpLog.substr(tmpLog.find_first_of("\n") + 1, tmpLog.length());
158  }

178  messageCounter++;

183 void LoggingModule::mainLoop() {
184  file.reset(new std::ofstream());
185  messageCounter = 0;
186  std::stringstream greeter;
187  greeter << getName() << " " << getTime() << "There are " << sources.size()
188  << " modules registered for logging:" << std::endl;
189  broadcastMessage(greeter.str());
190  for(auto module = sources.begin(); module != sources.end(); module++) {
191  broadcastMessage(std::string("\t - ") + module->sendingModule);
192  id_list[module->data.msg.getId()] = &(*module);
193  }
194  auto group = readAnyGroup();
195  std::string msg;
196  MessageSource* currentSender;
197  LogLevel level;
198 
199  while(1) {
200  // we skip the initial value since it is empty anyway and set in Logger::prepare
201  auto id = group.readAny();
202  if(id_list.count(id) == 0) {
203  throw ChimeraTK::logic_error("Cannot find element id"
204  "when updating logging variables.");
205  }
206  try {
207  currentSender = id_list.at(id);
208  msg = (std::string)(currentSender->data.msg);
209  }
210  catch(std::out_of_range& e) {
211  throw ChimeraTK::logic_error("Cannot find element id"
212  "when updating logging variables.");
213  }
214  try {
215  level = static_cast<LogLevel>(std::strtoul(&msg.at(0), NULL, 0));
216  }
217  catch(std::out_of_range& e) {
218  throw ChimeraTK::logic_error("Cannot find message level"
219  "when updating logging variables.");
220  }
221  // if log level is INTERANEL someone called writeAll() in a module containing the Logger -> ignore
222  if(level == LogLevel::INTERNAL) {
223  continue;
224  }
225  if(targetStream == 4) continue;
226  LogLevel setLevel = static_cast<LogLevel>((uint)logLevel);
227  std::string tmpStr = msg;
228  // remove message level
229  tmpStr = tmpStr.substr(1, tmpStr.size());
230  std::stringstream ss;
231  if(((std::string)currentSender->data.alias).empty()) {
232  ss << level << getName() << ":" << currentSender->sendingModule << " " << getTime() << tmpStr;
233  }
234  else {
235  ss << level << getName() << ":" << (std::string)currentSender->data.alias << " " << getTime() << tmpStr;
236  }
237  if(targetStream == 0 || targetStream == 1) {
238  if(!((std::string)logFile).empty() && !file->is_open()) {
239  std::stringstream ss_file;
240  file->open((std::string)logFile, std::ofstream::out | std::ofstream::app);
241  if(!file->is_open() && setLevel <= LogLevel::ERROR) {
242  ss_file << LogLevel::ERROR << getName() << " " << getTime()
243  << "Failed to open log file for writing: " << (std::string)logFile << std::endl;
244  broadcastMessage(ss_file.str(), true);
245  }
246  else if(file->is_open() && setLevel <= LogLevel::INFO) {
247  ss_file << LogLevel::INFO << getName() << " " << getTime()
248  << "Opened log file for writing: " << (std::string)logFile << std::endl;
249  broadcastMessage(ss_file.str());
250  }
251  }
252  }
253  if(level >= setLevel) {
254  if(level < LogLevel::ERROR)
255  broadcastMessage(ss.str());
256  else
257  broadcastMessage(ss.str(), true);
258  }
259  }
260 }

262 ctk::VariableNetworkNode LoggingModule::getAccessorPair(const ctk::RegisterPath& namePrefix) {
263  auto it = std::find_if(sources.begin(), sources.end(),
264  boost::bind(&MessageSource::sendingModule, boost::placeholders::_1) == (std::string)namePrefix);
265  if(it == sources.end()) {
266  auto result = prepareHierarchy(namePrefix);
267  if(result == "/") {
268  sources.emplace_back(MessageSource{namePrefix, this});
269  }
270  else {
271  sources.emplace_back(MessageSource{namePrefix, &groupMap[result]});
272  }
273  }
274  else {
275  throw ChimeraTK::logic_error(
276  "Cannot add logging for module " + namePrefix + " since logging was already added for this module.");
277  }
278  return sources.back().data.msg;
279 }
