98  std::cout << "*** testTwoScalarPushAccessors<" << typeid(T).name() << ">" << std::endl;

102  app.testModule.feedingPush >> app.testModule.consumingPush;

105  app.testModule.mainLoopStarted.wait(); // make sure the module's mainLoop() is entered
106 
107  // single theaded test
108  app.testModule.consumingPush = 0;
109  app.testModule.feedingPush = 42;
110  BOOST_CHECK(app.testModule.consumingPush == 0);
111  app.testModule.feedingPush.write();
112  BOOST_CHECK(app.testModule.consumingPush == 0);
113  app.testModule.consumingPush.read();
114  BOOST_CHECK(app.testModule.consumingPush == 42);
115 
116  // launch read() on the consumer asynchronously and make sure it does not yet
117  // receive anything
118  auto futRead = std::async(std::launch::async, [&app] { app.testModule.consumingPush.read(); });

120 
121  BOOST_CHECK(app.testModule.consumingPush == 42);
122 
123  // write to the feeder
124  app.testModule.feedingPush = 120;
125  app.testModule.feedingPush.write();
126
