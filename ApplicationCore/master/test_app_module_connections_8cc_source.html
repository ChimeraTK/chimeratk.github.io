95 BOOST_AUTO_TEST_CASE_TEMPLATE(testTwoScalarPushAccessors, T, test_types) {
96  std::cout << "*** testTwoScalarPushAccessors<" << typeid(T).name() << ">" << std::endl;
97 
98  TestApplication<T> app;
99 
100  app.testModule.feedingPush >> app.testModule.consumingPush;
101  app.initialise();
102  app.run();
103  app.testModule.mainLoopStarted.wait(); // make sure the module's mainLoop() is entered
104 
105  // single theaded test
106  app.testModule.consumingPush = 0;
107  app.testModule.feedingPush = 42;
108  BOOST_CHECK(app.testModule.consumingPush == 0);
109  app.testModule.feedingPush.write();
110  BOOST_CHECK(app.testModule.consumingPush == 0);
111  app.testModule.consumingPush.read();
112  BOOST_CHECK(app.testModule.consumingPush == 42);
113 
114  // launch read() on the consumer asynchronously and make sure it does not yet
115  // receive anything
116  auto futRead = std::async(std::launch::async, [&app] { app.testModule.consumingPush.read(); });
117  BOOST_CHECK(futRead.wait_for(std::chrono::milliseconds(200)) == std::future_status::timeout);
118 
119  BOOST_CHECK(app.testModule.consumingPush == 42);
120 
121  // write to the feeder
122  app.testModule.feedingPush = 120;
123  app.testModule.feedingPush.write();
124 
125  // check that the consumer now receives the just written value
126  BOOST_CHECK(futRead.wait_for(std::chrono::milliseconds(2000)) == std::future_status::ready);
127  BOOST_CHECK(app.testModule.consumingPush == 120);
128 }
