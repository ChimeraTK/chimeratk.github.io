47  virtual void addSlave(
48  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> slave, VariableNetworkNode& /*consumer*/) {
49  if(!slave->isWriteable()) {
50  throw ChimeraTK::logic_error("FanOut::addSlave() has been called with a "
51  "receiving implementation!");
52  }
53  // check if array shape is compatible, unless the receiver is a trigger
54  // node, so no data is expected
55  if(slave->getNumberOfSamples() != 0 &&
56  (slave->getNumberOfChannels() != impl->getNumberOfChannels() ||
57  slave->getNumberOfSamples() != impl->getNumberOfSamples())) {
58  std::string what = "FanOut::addSlave(): Trying to add a slave '";
59  what += slave->getName();
60  what += "' with incompatible array shape! Name of master: ";
61  what += impl->getName();
62  what += " Length of master: " + std::to_string(impl->getNumberOfChannels()) + " x " +
63  std::to_string(impl->getNumberOfSamples());
64  what += " Length of slave: " + std::to_string(slave->getNumberOfChannels()) + " x " +
65  std::to_string(slave->getNumberOfSamples());
66  throw ChimeraTK::logic_error(what.c_str());
67  }

71  // remove a slave identified by its consuming node from the FanOut
72  void removeSlave(const boost::shared_ptr<ChimeraTK::TransferElement> &slave) override {
73  // make sure the slave is actually currently in the list, and get it by the right typ
74  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> slave_typed;
75  for(auto &s : slaves) {
76  if(s == slave) {
77  slave_typed = s;
78  break;
79  }
80  }
81  assert(slave_typed != nullptr);
82 
83  size_t nOld = slaves.size();
84  slaves.remove(slave_typed);
85  assert(slaves.size() == nOld-1);
86  }
