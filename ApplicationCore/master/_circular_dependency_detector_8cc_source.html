14  void CircularDependencyDetector::registerDependencyWait(VariableNetworkNode& node) {
15  assert(node.getType() == NodeType::Application);
16  std::unique_lock<std::mutex> lock(_mutex);
17 
18  auto* dependent = dynamic_cast<Module*>(node.getOwningModule())->findApplicationModule();
19 
20  // register the waiting node in the map (used for the detectBlockedModules() thread). This is done also for Device
21  // variables etc.
22  _awaitedNodes[dependent] = node;
23 
24  // checking of direct circular dependencies is only done for Application-type feeders
25  auto feedingModule = node.getModel().visit(Model::returnApplicationModule, Model::keepPvAccess,
26  Model::keepApplicationModules, Model::adjacentInSearch, Model::returnFirstHit(Model::ApplicationModuleProxy{}));
27  if(!feedingModule.isValid()) return;
28  Module* dependency = &feedingModule.getApplicationModule();
29 
30  // If a module depends on itself, the detector would always detect a circular dependency, even if it is resolved
31  // by writing initial values in prepare(). Hence we do not check anything in this case.
32  if(dependent == dependency) return;
33 
34  // Register dependent-dependency relation in map
35  _awaitedVariables[dependent] = node.getQualifiedName();
36  _waitMap[dependent] = dependency;
37 
38  // check for circular dependencies
39  auto* depdep = dependency;
40  while(_waitMap.find(depdep) != _waitMap.end()) {
41  auto* depdep_prev = depdep;
42  depdep = _waitMap[depdep];
43  if(depdep == dependent) {
44  // The detected circular dependency might still resolve itself, because registerDependencyWait() is called even
45  // if initial values are already present in the queues.
46  for(size_t i = 0; i < 1000; ++i) {
47  // give other thread a chance to read their initial value
48  lock.unlock();
49  usleep(10000);
50  lock.lock();
51  // if the module depending on an initial value for "us" is no longer waiting for us to send an initial value,
52  // the circular dependency is resolved.
53  if(_waitMap.find(depdep_prev) == _waitMap.end() || _waitMap[depdep] != dependent) return;
54  }
55 
56  std::cerr << "*** Circular dependency of ApplicationModules found while waiting for initial values!"
57  << std::endl;
58  std::cerr << std::endl;
59 
60  std::cerr << dependent->getQualifiedName() << " waits for " << node.getQualifiedName() << " from:" << std::endl;
61  auto* depdep2 = dependency;
62  while(_waitMap.find(depdep2) != _waitMap.end()) {
63  auto waitsFor = _awaitedVariables[depdep2];
64  std::cerr << depdep2->getQualifiedName();
65  if(depdep2 == dependent) break;
66  depdep2 = _waitMap[depdep2];
67  std::cerr << " waits for " << waitsFor << " from:" << std::endl;
68  }
69  std::cerr << "." << std::endl;
70 
71  std::cerr << std::endl;
72  std::cerr
73  << "Please provide an initial value in the prepare() function of one of the involved ApplicationModules!"
74  << std::endl;
75 
76  throw ChimeraTK::logic_error("Circular dependency of ApplicationModules while waiting for initial values");
77  }
78  else {
79  // Give other threads a chance to add to the wait map
80  lock.unlock();
81  usleep(10000);
82  lock.lock();
83  }
84  }
85  }

89  void CircularDependencyDetector::printWaiters() {
90  if(_waitMap.size() == 0) return;
91  std::cerr << "The following modules are still waiting for initial values:" << std::endl;
92  for(auto& waiters : _waitMap) {
93  std::cerr << waiters.first->getQualifiedName() << " waits for " << _awaitedVariables[waiters.first] << " from "
94  << waiters.second->getQualifiedName() << std::endl;
95  }
96  std::cerr << "(end of list)" << std::endl;
97  }

101  void CircularDependencyDetector::unregisterDependencyWait(VariableNetworkNode& node) {
102  assert(node.getType() == NodeType::Application);
103  std::lock_guard<std::mutex> lock(_mutex);
104  auto* mod = dynamic_cast<Module*>(node.getOwningModule())->findApplicationModule();
105  _waitMap.erase(mod);
106  _awaitedVariables.erase(mod);
107  _awaitedNodes.erase(mod);
108  }

118  void CircularDependencyDetector::detectBlockedModules() {
119  auto& app = Application::getInstance();
120  while(true) {
121  // wait some time to slow down this check
122  boost::this_thread::sleep_for(boost::chrono::seconds(60));
123 
124  // check for modules which did not yet reach their mainLoop
125  bool allModulesEnteredMainLoop = true;
126  for(auto* module : app.getSubmoduleListRecursive()) {
127  // Note: We are "abusing" this flag which was introduced for the testable mode. It actually just shows whether
128  // the mainLoop() has been called already (resp. will be called very soon). It is not depending on the
129  // testableMode. FIXME: Rename this flag!
130  if(module->hasReachedTestableMode()) {
131  continue;
132  }
133 
134  // Found a module which is still waiting for initial values
135  allModulesEnteredMainLoop = false;
136 
137  // require lock against concurrent accesses on the maps
138  std::lock_guard<std::mutex> lock(_mutex);
139 
140  // Check if module has registered a dependency wait. If not, situation will either resolve soon or module will
141  // register a dependency wait soon. Both cases can be found in the next interation.
142  if(_awaitedNodes.find(module) == _awaitedNodes.end()) continue;
143 
144  auto* appModule = dynamic_cast<ApplicationModule*>(module);
145  if(!appModule) {
146  // only ApplicationModule can have hasReachedTestableMode() == true, so it should not happen
147  std::cout << "CircularDependencyDetector found non-application module: " << module->getQualifiedName()
148  << std::endl;
149  continue;
150  }
151 
152  // Iteratively search for reason the module blocks
153  std::function<void(const VariableNetworkNode& node)> iterativeSearch = [&](const VariableNetworkNode& node) {
154  auto visitor = [&](auto proxy) {
155  if constexpr(Model::isApplicationModule(proxy)) {
156  // fed by other ApplicationModule: check if that one is waiting, too
157  Module* feedingAppModule = &proxy.getApplicationModule();
158  if(feedingAppModule->hasReachedTestableMode()) {
159  // the feeding module has started its mainLoop(), but not sent us an initial value
160  // FIXME: There is a race condition, if the situation has right now resolved and the feeding module just
161  // not yet sent the initial value. Ideally we should wait another iteration before printing warnings!
162  if(_modulesWeHaveWarnedAbout.find(feedingAppModule) == _modulesWeHaveWarnedAbout.end()) {
163  _modulesWeHaveWarnedAbout.insert(feedingAppModule);
164  std::cout << "Note: ApplicationModule " << appModule->getQualifiedName() << " is waiting for an "
165  << "initial value, because " << feedingAppModule->getQualifiedName()
166  << " has not yet sent one." << std::endl;
167  }
168  return;
169  }
170  if(_awaitedNodes.find(feedingAppModule) == _awaitedNodes.end()) {
171  // The other module is not right now waiting. It will either enter the mainLoop soon, or start
172  // waiting soon. Let's just wait another iteration.
173  return;
174  }
175  // The other module is right now waiting: continue iterative search
176  iterativeSearch(_awaitedNodes.at(feedingAppModule));
177  }
178  else if constexpr(Model::isDeviceModule(proxy)) {
179  // fed by device
180  const auto& deviceName = proxy.getAliasOrCdd();
181  if(_devicesWeHaveWarnedAbout.find(deviceName) == _devicesWeHaveWarnedAbout.end()) {
182  _devicesWeHaveWarnedAbout.insert(deviceName);
183  std::cout << "Note: Still waiting for device " << deviceName << " to come up";
184  auto dm = Application::getInstance()._deviceManagerMap[deviceName];
185  std::unique_lock dmLock(dm->_errorMutex);
186  if(dm->_deviceHasError) {
187  std::cout << " (" << std::string(dm->_deviceError._message) << ")";
188  }
189  std::cout << "..." << std::endl;
190  }
191  }
192  else {
193  // fed by anything else?
194  std::cout << "At least one ApplicationModule (" << appModule->getQualifiedName() << " is waiting for an "
195  << "initial value from an unexpected source." << std::endl;
196  std::cout << "This is probably a BUG in the ChimeraTK framework." << std::endl;
197  }
198  };
199 
200  node.getModel().visit(visitor, Model::keepPvAccess, Model::adjacentInSearch);
201  };
202  iterativeSearch(_awaitedNodes.at(appModule));
203  }
204 
205  // if all modules are in the mainLoop, stop this thread
206  if(allModulesEnteredMainLoop) break;
207  }
208 
209  std::cout << "All application modules are running." << std::endl;
210 
211  // free some memory
212  _waitMap.clear();
213  _awaitedVariables.clear();
214  _awaitedNodes.clear();
215  _modulesWeHaveWarnedAbout.clear();
216  _devicesWeHaveWarnedAbout.clear();
217  _otherThingsWeHaveWarnedAbout.clear();
218  }
