47  void ExceptionHandlingDecorator<UserType>::doPreWrite(TransferType type, VersionNumber versionNumber) {
48  /* For writable accessors, copy data to the recoveryAcessor before perfroming the write.
49  * Otherwise, the decorated accessor may have swapped the data out of the user buffer already.
50  * This obtains a shared lock from the DeviceModule, hence, the regular writing happeniin here
51  * can be performed in shared mode of the mutex and accessors are not blocking each other.
52  * In case of recovery, the DeviceModule thread will take an exclusive lock so that this thread can not
53  * modify the recoveryAcessor's user buffer while data is written to the device.
54  */
55  {
56  _inhibitWriteTransfer = false;
57  _hasThrownLogicError = false;
58  _dataLostInPreviousWrite = false;
59  auto recoverylock{_deviceModule->getRecoverySharedLock()};
60 
61  if(_recoveryAccessor != nullptr) {
62  if(!_recoveryHelper->wasWritten && (_recoveryHelper->writeOrder != 0)) {
63  _dataLostInPreviousWrite = true;
64  }
65 
66  // Access to _recoveryAccessor is only possible channel-wise
67  for(unsigned int ch = 0; ch < _recoveryAccessor->getNumberOfChannels(); ++ch) {
68  _recoveryAccessor->accessChannel(ch) = buffer_2D[ch];
69  }
70  _recoveryHelper->versionNumber = versionNumber;
71  _recoveryHelper->writeOrder = _deviceModule->writeOrder();
72  _recoveryHelper->wasWritten = false;
73  }
74  else {
75  _hasThrownLogicError = true;
76  throw ChimeraTK::logic_error(
77  "ChimeraTK::ExceptionhandlingDecorator: Calling write() on a non-writeable accessor is not supported ");
78  }
79 
80  boost::shared_lock<boost::shared_mutex> errorLock(_deviceModule->errorMutex);
81  if(_deviceModule->deviceHasError) {
82  _inhibitWriteTransfer = true;
83  return;
84  }
85 
86  ++_deviceModule->synchronousTransferCounter; // must be under the lock
87 
88  } // lock guard goes out of scope
89 
90  // Now delegate call to the generic decorator, which swaps the buffer, without adding our exception handling with the generic transfer
91  // preWrite and postWrite are only delegated if the transfer is allowed.
92  ChimeraTK::NDRegisterAccessorDecorator<UserType>::doPreWrite(type, versionNumber);
93  }

95  template<typename UserType>
96  void ExceptionHandlingDecorator<UserType>::doPostWrite(TransferType type, VersionNumber versionNumber) {
97  if(_hasThrownLogicError) {
98  // preWrite has not been delegated, so there is nothing to do here. Let
99  // postWrite() throw the active exception we have. Don't clear logic erros here.
100  return;
101  }
102  if(!_inhibitWriteTransfer) {
103  --_deviceModule->synchronousTransferCounter;
104  try {
105  ChimeraTK::NDRegisterAccessorDecorator<UserType>::doPostWrite(type, versionNumber);
106  {
107  auto recoverylock{_deviceModule->getRecoverySharedLock()};
108  // the transfer was successful or doPostRead did not throw and we reach this point,
109  // so we matk these data as written
110  _recoveryHelper->wasWritten = true;
111  } // end scope for recovery lock
112  }
113  catch(ChimeraTK::runtime_error& e) {
114  // Report exception to the exception backend. This would be done by the TransferElement base class only if we
115  // would let the exception through, hence we have to take care of this here.
116  this->_exceptionBackend->setException();
117  // Report exception to the DeviceModule
118  _deviceModule->reportException(std::string(e.what()) + " (seen by '" + _target->getName() + "')");
119  }
120  }
121  assert(_activeException == nullptr);
122  }

124  template<typename UserType>
125  void ExceptionHandlingDecorator<UserType>::doPostRead(TransferType type, bool hasNewData) {
126  // preRead has not been called when the transfer was not allowed. Don't call postRead in this case.
127  if(!_hasThrownToInhibitTransfer) {
128  try {
129  if(!TransferElement::_accessModeFlags.has(AccessMode::wait_for_new_data)) { // was as synchronous transfer
130  --_deviceModule->synchronousTransferCounter;
131  }
132  _target->setActiveException(this->_activeException);
133  _target->postRead(type, hasNewData);
134  if(hasNewData) {
135  // Reset the flag after a successful read.
136  // It is only reset if there was new data. A readNonBlocking on a faulty device is not different to a readNonBlocking on working device: There just
137  // is no new data. We only reset it on the next successful read with the initial value, otherwise keep the exception flag.
138  _hasReportedException = false;
139  }
140  }
141  catch(ChimeraTK::runtime_error& e) {
142  // Report exception to the exception backend. This would be done by the TransferElement base class only if we
143  // would let the exception through, hence we have to take care of this here.
144  this->_exceptionBackend->setException();
145  // Report exception to the DeviceModule
146  _deviceModule->reportException(std::string(e.what()) + " (seen by '" + _target->getName() + "')");
147  _hasReportedException = true;
148  }
149  }
150  else {
151  _activeException = nullptr;
152  }
153 
154  if(_hasReportedException || _hasThrownToInhibitTransfer) {
155  _dataValidity = DataValidity::faulty;
156  // Note: This assertion does not hold
157  // See discussion in https://github.com/ChimeraTK/DeviceAccess/pull/178
158  // assert(_deviceModule->getExceptionVersionNumber() > _versionNumber);
159  if(_deviceModule->getExceptionVersionNumber() > _versionNumber) {
160  _versionNumber = _deviceModule->getExceptionVersionNumber();
161  }
162  }
163  else {
164  _dataValidity = _target->dataValidity();
165  // Note: This assertion does not hold
166  // See discussion in https://github.com/ChimeraTK/DeviceAccess/pull/178
167  //assert(_target->getVersionNumber() >= _versionNumber);
168  if(_target->getVersionNumber() > _versionNumber) {
169  _versionNumber = _target->getVersionNumber();
170  }
171  }
172 
173  // only replace the user buffer if there really is new data
174  if(hasNewData) {
175  for(size_t i = 0; i < buffer_2D.size(); ++i)
176  buffer_2D[i].swap(this->_target->accessChannel(static_cast<unsigned int>(i)));
177  }
178  assert(_activeException == nullptr);
179  }

216  bool ExceptionHandlingDecorator<UserType>::genericWriteWrapper(Callable writeFunction) {
217  if(_inhibitWriteTransfer) {
218  return _dataLostInPreviousWrite;
219  }
220  bool transferReportsPreviousDataLost = false;
221  try {
222  transferReportsPreviousDataLost = writeFunction();
223  }
224  catch(ChimeraTK::runtime_error&) {
225  _activeException = std::current_exception();
226  }
227  return (transferReportsPreviousDataLost || _dataLostInPreviousWrite);
228  }
