47  void ExceptionHandlingDecorator<UserType>::doPreWrite(TransferType type, VersionNumber versionNumber) {
48  /* For writable accessors, copy data to the recoveryAcessor before perfroming the write.
49  * Otherwise, the decorated accessor may have swapped the data out of the user buffer already.
50  * This obtains a shared lock from the DeviceModule, hence, the regular writing happeniin here
51  * can be performed in shared mode of the mutex and accessors are not blocking each other.
52  * In case of recovery, the DeviceModule thread will take an exclusive lock so that this thread can not
53  * modify the recoveryAcessor's user buffer while data is written to the device.
54  */
55  {
56  _hasThrownToInhibitTransfer = false;
57  _hasThrownLogicError = false;
58  _dataLostInPreviousWrite = false;
59  auto recoverylock{_deviceModule->getRecoverySharedLock()};
60 
61  if(_recoveryAccessor != nullptr) {
62  if(!_recoveryHelper->wasWritten && (_recoveryHelper->writeOrder != 0)) {
63  _dataLostInPreviousWrite = true;
64  }
65 
66  // Access to _recoveryAccessor is only possible channel-wise
67  for(unsigned int ch = 0; ch < _recoveryAccessor->getNumberOfChannels(); ++ch) {
68  _recoveryAccessor->accessChannel(ch) = buffer_2D[ch];
69  }
70  _recoveryHelper->versionNumber = versionNumber;
71  _recoveryHelper->writeOrder = _deviceModule->writeOrder();
72  _recoveryHelper->wasWritten = false;
73  }
74  else {
75  _hasThrownLogicError = true;
76  throw ChimeraTK::logic_error(
77  "ChimeraTK::ExceptionhandlingDecorator: Calling write() on a non-writeable accessor is not supported ");
78  }
79 
80  boost::shared_lock<boost::shared_mutex> errorLock(_deviceModule->errorMutex);
81  if(_deviceModule->deviceHasError) {
82  _hasThrownToInhibitTransfer = true;
83  throw ChimeraTK::runtime_error(
84  "ExceptionHandlingDecorator is preventing the write transfer."); // writing will be delayed, done by the recovery accessor
85  }
86 
87  ++_deviceModule->synchronousTransferCounter; // must be under the lock
88 
89  } // lock guard goes out of scope
90 
91  // Now delegate call to the generic decorator, which swaps the buffer, without adding our exception handling with the generic transfer
92  // preWrite and postWrite are only delegated if the transfer is allowed.
93  ChimeraTK::NDRegisterAccessorDecorator<UserType>::doPreWrite(type, versionNumber);
94  }

96  template<typename UserType>
97  void ExceptionHandlingDecorator<UserType>::doPostWrite(TransferType type, VersionNumber versionNumber) {
98  if(_hasThrownLogicError) {
99  // preWrite has not been delegated, so there is nothing to do here. Let
100  // postWrite() throw the active exception we have. Don't clear logic erros here.
101  return;
102  }
103  if(!_hasThrownToInhibitTransfer) {
104  {
105  auto recoverylock{_deviceModule->getRecoverySharedLock()};
106  _recoveryHelper->wasWritten = true; // the transfer was successful, so this data was written
107  } // end scope for recovery lock
108 
109  --_deviceModule->synchronousTransferCounter;
110  try {
111  ChimeraTK::NDRegisterAccessorDecorator<UserType>::doPostWrite(type, versionNumber);
112  }
113  catch(ChimeraTK::runtime_error& e) {
114  _deviceModule->reportException(e.what());
115  }
116  }
117  else {
118  // get rid of the active exception do the calling postWrite() does not re-throw
119  _activeException = nullptr;
120  }
121  assert(_activeException == nullptr);
122  }

124  template<typename UserType>
125  void ExceptionHandlingDecorator<UserType>::doPostRead(TransferType type, bool hasNewData) {
126  // preRead has not been called when the transfer was not allowed. Don't call postRead in this case.
127  if(!_hasThrownToInhibitTransfer) {
128  try {
129  if(!TransferElement::_accessModeFlags.has(AccessMode::wait_for_new_data)) { // was as synchronous transfer
130  --_deviceModule->synchronousTransferCounter;
131  }
132  _target->setActiveException(this->_activeException);
133  _target->postRead(type, hasNewData);
134  if(hasNewData) {
135  // Reset the flag after a successful read. It might have been true in pre-read, and a read has been waiting for the device to send the next valid data
136  // It is only reset if there was new data. Otherwise calls to readNonBlocking on a recovered device would turn the data validity back to good, although
137  // nothing new has been received and the data is still the invalid stuff from the last exception.
138  _hasReportedException = false;
139  }
140  }
141  catch(ChimeraTK::runtime_error& e) {
142  _deviceModule->reportException(e.what());
143  _hasReportedException = true;
144  }
145  }
146  else {
147  _activeException = nullptr;
148  }
149 
150  if(_hasReportedException || _hasThrownToInhibitTransfer) {
151  _dataValidity = DataValidity::faulty;
152  _versionNumber = _deviceModule->getExceptionVersionNumber();
153  }
154  else {
155  _dataValidity = _target->dataValidity();
156  _versionNumber = _target->getVersionNumber();
157  }
158 
159  // only replace the user buffer if there really is new data
160  if(hasNewData) {
161  for(size_t i = 0; i < buffer_2D.size(); ++i)
162  buffer_2D[i].swap(this->_target->accessChannel(static_cast<unsigned int>(i)));
163  }
164  assert(_activeException == nullptr);
165  }
