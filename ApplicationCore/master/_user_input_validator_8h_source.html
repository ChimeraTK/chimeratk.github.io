191  template<typename... ACCESSORTYPES>
192  void UserInputValidator::add(
193  const std::string& errorMessage, const std::function<bool(void)>& isValidFunction, ACCESSORTYPES&... accessors) {
194  boost::fusion::list<ACCESSORTYPES&...> accessorList{accessors...};
195  static_assert(boost::fusion::size(accessorList) > 0, "Must specify at least one accessor!");
196  assert(isValidFunction != nullptr);
197 
198  // create validator and store in list
199  _validators.push_back(Validator(isValidFunction, errorMessage));
200 
201  // create map of accessors to validators, also add accessors/variables to list
202  boost::fusion::for_each(accessorList, [&](auto& accessor) {
203  addAccessorIfNeeded(accessor);
204  _validatorMap[accessor.getId()].push_back(&_validators.back());
205  });
206  }
