195  template<typename... ACCESSORTYPES>
196  void UserInputValidator::add(
197  const std::string& errorMessage, const std::function<bool(void)>& isValidFunction, ACCESSORTYPES&... accessors) {
198  boost::fusion::list<ACCESSORTYPES&...> accessorList{accessors...};
199  static_assert(boost::fusion::size(accessorList) > 0, "Must specify at least one accessor!");
200  assert(isValidFunction != nullptr);
201 
202  // create validator and store in list
203  _validators.push_back(Validator(isValidFunction, errorMessage));
204 
205  // create map of accessors to validators, also add accessors/variables to list
206  boost::fusion::for_each(accessorList, [&](auto& accessor) {
207  addAccessorIfNeeded(accessor);
208  _validatorMap[accessor.getId()].push_back(&_validators.back());
209  });
210  }

214  template<typename UserType, template<typename> typename Accessor>
215  void UserInputValidator::setFallback(Accessor<UserType>& accessor, UserType value) {
216  addAccessorIfNeeded(accessor);
217  auto pv = std::dynamic_pointer_cast<Variable<UserType, Accessor>>(_variableMap.at(accessor.getId()));
218  assert(pv != nullptr);
219  if(pv->_lastAcceptedValue.size() != 1) {
220  throw ChimeraTK::logic_error(
221  "UserInputValidator::setFallback() with scalar value called for array-typed accessor '" + accessor.getName() +
222  "'.");
223  }
224  pv->_lastAcceptedValue[0] = value;
