1417  template<typename VISITOR, typename... Args>
1418  auto Impl::visit(Vertex startVertex, VISITOR visitor, Args... args) {
1419  auto filteredGraph = getFilteredGraph(args...);
1420 
1421  auto vertexFilter = Model::getVertexFilter(args...);
1422 
1423  // Execute search operation based on selected type
1424  auto typeHolder = Model::getSearchType<Args...>();
1425  using type = typename decltype(typeHolder)::type;
1426  constexpr bool isAdjacent =
1427  std::is_same<AdjacentSearch, type>::value || std::is_base_of<AdjacentSearch, type>::value;
1428  constexpr bool isAdjacentIn =
1429  std::is_same<AdjacentInSearch, type>::value || std::is_base_of<AdjacentInSearch, type>::value;
1430  constexpr bool isAdjacentOut =
1431  std::is_same<AdjacentOutSearch, type>::value || std::is_base_of<AdjacentOutSearch, type>::value;
1432  constexpr bool isDFS =
1433  std::is_same<DepthFirstSearch, type>::value || std::is_base_of<DepthFirstSearch, type>::value;
1434  constexpr bool isBFS =
1435  std::is_same<BreadthFirstSearch, type>::value || std::is_base_of<BreadthFirstSearch, type>::value;
1436 
1437  constexpr bool continueDisjunct = Model::hasSearchOption<ContinueSearchDisjunctTrees, Args...>();
1438  constexpr bool returnFirst = Model::hasSearchOption<ReturnFirstHit, Args...>();
1439 
1440  // Wrapper for visitor to filter wrong objecttypes in constexpr condition, so the actual visitor does not have
1441  // to work with all proxy types, just the ones which can get past the VisitorFilter.
1442  using VisitorReturnType = detail::VisitorReturnType<VISITOR, decltype(vertexFilter)>;
1443  auto visitorWrapper = [&](auto proxy) -> VisitorReturnType {
1444  if constexpr(vertexFilter.template constevalObjecttype<decltype(proxy)>()) {
1445  return visitor(proxy);
1446  }
1447  // The VertexFilter makes sure that this function is never called with the wrong type, hence if we end up here
1448  // there is a bug in the VertexFilter. Note that this code will get instantiated, just never executed.
1449  throw ChimeraTK::logic_error("We should never end up here...");
1450  };
1451 
1452  if constexpr(isAdjacent || isAdjacentOut) {
1453  auto [start, end] = boost::out_edges(startVertex, filteredGraph);
1454  for(auto it = start; it != end; ++it) {
1455  // obtain target vertex of the current outgoing edge (i.e. vertex at the other end of the edge)
1456  auto vtx = target(*it, graph);
1457 
1458  // apply vertex filter
1459  if(!vertexFilter.evalVertexFilter(graph[vtx])) {
1460  continue;
1461  }
1462 
1463  if constexpr(returnFirst) {
1464  return filteredGraph[vtx].visitProxy(visitorWrapper, vtx, shared_from_this());
1465  }
1466  else {
1467  filteredGraph[vtx].visitProxy(visitorWrapper, vtx, shared_from_this());
1468  }
1469  }
1470  }
1471  if constexpr(isAdjacent || isAdjacentIn) {
1472  auto [start, end] = boost::in_edges(startVertex, filteredGraph);
1473  for(auto it = start; it != end; ++it) {
1474  // obtain source vertex of the current incoming edge (i.e. vertex at the other end of the edge)
1475  auto vtx = source(*it, graph);
1476 
1477  // apply vertex filter
1478  if(!vertexFilter.evalVertexFilter(graph[vtx])) {
1479  continue;
1480  }
1481 
1482  if constexpr(returnFirst) {
1483  return filteredGraph[vtx].visitProxy(visitorWrapper, vtx, shared_from_this());
1484  }
1485  else {
1486  filteredGraph[vtx].visitProxy(visitorWrapper, vtx, shared_from_this());
1487  }
1488  }
1489  }
1490  if constexpr(isDFS || isBFS) {
1491  // For DFS: By default stop the search when the start vertex is finished, unless configured otherwise
1492  // For BFS: The search anyway does not continue with disjunct parts, and stopping when the start vertex is
1493  // finished would stop too early (no recursion).
1494  Vertex stopVertex;
1495  if constexpr(continueDisjunct || isBFS) {
1496  stopVertex = std::numeric_limits<Vertex>::max();
1497  }
1498  else {
1499  stopVertex = startVertex;
1500  }
1501 
1502  // create visitor object
1503  detail::ValueHolder<detail::VisitorReturnType<decltype(visitorWrapper), decltype(vertexFilter)>> rv;
1504  auto visitorObjectFactory = [&]() {
1505  if constexpr(isDFS) {
1506  return detail::VisitorHelper<boost::dfs_visitor<>, decltype(visitorWrapper), decltype(vertexFilter),
1507  returnFirst>(visitorWrapper, shared_from_this(), vertexFilter, stopVertex, rv);
1508  }
1509  else {
1510  return detail::VisitorHelper<boost::bfs_visitor<>, decltype(visitorWrapper), decltype(vertexFilter),
1511  returnFirst>(visitorWrapper, shared_from_this(), vertexFilter, stopVertex, rv);
1512  }
1513  };
1514  auto visitorObject = visitorObjectFactory();
1515 
1516  // Somehow the named-parameter overload of depth_first_search does not work here, so we have to provide
1517  // a color map explicitly...
1518  std::vector<boost::default_color_type> colors(boost::num_vertices(filteredGraph));
1519  boost::iterator_property_map color_map(colors.begin(), boost::get(boost::vertex_index, filteredGraph));
1520 
1521  // The try-catch block is just a trick to terminate the search operation early (when stopVertex hs finished)
1522  try {
1523  if constexpr(isDFS) {
1524  boost::depth_first_search(filteredGraph, visitorObject, color_map, startVertex);
1525  }
1526  else {
1527  boost::queue<Vertex> buffer;
1528  boost::breadth_first_search(filteredGraph, startVertex, buffer, visitorObject, color_map);
1529  }
1530  }
1531  catch(typename decltype(visitorObject)::StopException&) {
1532  // Ignore this exception, as it is just to use to trick the boost::depth/breadth_first_search() to stop
1533  }
1534 
1535  if constexpr(returnFirst) {
1536  return visitorObject.getReturnValue();
1537  }
1538  }
1539 
1540  if constexpr(returnFirst) {
1541  // Nothing found but ReturnFirstHit option specified: return the not-found value
1542  return Model::getSearchOption<ReturnFirstHit, Args...>(args...).notFoundValue.get();
1543  }
1544  }

1549  bool Impl::visitByPath(std::string_view path, VISITOR visitor, PROXY startProxy) {
1550  // remove any redundant "./" at the beginning
1551  while(boost::starts_with(path, "./")) {
1552  path = path.substr(2);
1553  }
1554 
1555  // resolve reference to ourself
1556  if(path.empty() || path == ".") {
1557  visitor(startProxy);
1558  return true;
1559  }
1560 
1561  // first component is one level up
1562  if(boost::starts_with(path, "../") || path == "..") {
1563  // remove the component from the path
1564  path = (path.size() > 2) ? path.substr(3) : "";
1565 
1566  // delegate to our parent proxy
1567  bool found = false;
1568  visit(
1569  startProxy._d->vertex,
1570  [&](auto parent) {
1571  if constexpr(isDirectory(parent) || isRoot(parent)) {
1572  found = parent.visitByPath(path, visitor);
