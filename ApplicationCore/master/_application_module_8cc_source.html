43  void ApplicationModule::run() {
44  // start the module thread
45  assert(!moduleThread.joinable());
46  moduleThread = boost::thread(&ApplicationModule::mainLoopWrapper, this);
47  //Wait until the thread has launched and acquired and released the testable mode lock at least once.
48  if(Application::getInstance().isTestableModeEnabled()) {
49  while(!testableModeReached) {
50  Application::getInstance().testableModeUnlock("releaseForReachTestableMode");
51  usleep(100);
52  Application::getInstance().testableModeLock("acquireForReachTestableMode");
53  }
54  }
55  }

59  void ApplicationModule::terminate() {
60  if(moduleThread.joinable()) {
61  moduleThread.interrupt();
62  // try joining the thread
63  while(!moduleThread.try_join_for(boost::chrono::milliseconds(10))) {
64  // if thread is not yet joined, send interrupt() to all variables.
65  for(auto& var : getAccessorListRecursive()) {
66  auto el{var.getAppAccessorNoType().getHighLevelImplElement()};
67  if(el->getAccessModeFlags().has(AccessMode::wait_for_new_data)) {
68  el->interrupt();
69  }
70  }
71  // it may not suffice to send interrupt() once, as the exception might get
72  // overwritten in the queue, thus we repeat this until the thread was
73  // joined.
74  }
75  }
76  assert(!moduleThread.joinable());
77  }

85  void ApplicationModule::mainLoopWrapper() {
86  Application::registerThread("AM_" + getName());
87 
88  // Acquire testable mode lock, so from this point on we are running only one user thread concurrently
89  Application::testableModeLock("start");
90 
91  // Read all variables once to obtain the initial values from the devices and from the control system persistency
92  // layer. This is done in two steps, first for all poll-type variables and then for all push-types, because
93  // poll-type reads might trigger distribution of values to push-type variables via a ConsumingFanOut.
94  for(auto& variable : getAccessorListRecursive()) {
95  if(variable.getDirection().dir != VariableDirection::consuming) continue;
96  if(variable.getMode() == UpdateMode::poll) {
97  auto hasInitialValue = variable.hasInitialValue();
98  if(hasInitialValue != VariableNetworkNode::InitialValueMode::None) {
99  variable.getAppAccessorNoType().readLatest();
100  }
101  }
102  }
103  for(auto& variable : getAccessorListRecursive()) {
104  if(variable.getDirection().dir != VariableDirection::consuming) continue;
105  if(variable.getMode() == UpdateMode::push) {
106  auto hasInitialValue = variable.hasInitialValue();
107  if(hasInitialValue == VariableNetworkNode::InitialValueMode::Poll) {
108  variable.getAppAccessorNoType().readLatest();
109  }
110  else if(hasInitialValue == VariableNetworkNode::InitialValueMode::Push) {
111  variable.getAppAccessorNoType().read();
112  }
113  }
114  }
115 
116  // We are holding the testable mode lock, so we are sure the mechanism will work now.
117  testableModeReached = true;
118 
119  // enter the main loop
120  mainLoop();
121  Application::testableModeUnlock("terminate");
122  }
