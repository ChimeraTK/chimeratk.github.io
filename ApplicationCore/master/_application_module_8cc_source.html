56  ApplicationModule& ApplicationModule::operator=(ApplicationModule&& other) noexcept {
57  assert(!moduleThread.joinable()); // if the thread is already running, moving is no longer allowed!
58 
59  // Keep the model as is (except from updating the pointers to the C++ objects). To do so, we have to hide it from
60  // unregisterModule() which is executed in Module::operator=(), because it would destroy the model.
61  Model::ApplicationModuleProxy model = std::move(other._model);
62  other._model = {};
63 
64  VariableGroup::operator=(std::move(other));
65 
66  if(model.isValid()) model.informMove(*this);
67  _model = model;
68 
69  return *this;
70  }

82  void ApplicationModule::terminate() {
83  if(moduleThread.joinable()) {
84  moduleThread.interrupt();
85  // try joining the thread
86  while(!moduleThread.try_join_for(boost::chrono::milliseconds(10))) {
87  // if thread is not yet joined, send interrupt() to all variables.
88  for(auto& var : getAccessorListRecursive()) {
89  auto el{var.getAppAccessorNoType().getHighLevelImplElement()};
90  if(el->getAccessModeFlags().has(AccessMode::wait_for_new_data)) {
91  el->interrupt();
92  }
93  }
94  // it may not suffice to send interrupt() once, as the exception might get
95  // overwritten in the queue, thus we repeat this until the thread was
96  // joined.
97  }
98  }
99  assert(!moduleThread.joinable());
100  }

116  void ApplicationModule::mainLoopWrapper() {
117  std::string className = boost::core::demangle(typeid(*this).name());
118  Application::registerThread("AM_" + className); // + getName());
119 
120  // Acquire testable mode lock, so from this point on we are running only one user thread concurrently
121  Application::getInstance().getTestableMode().lock("start");
122 
123  // Read all variables once to obtain the initial values from the devices and from the control system persistency
124  // layer. This is done in two steps, first for all poll-type variables and then for all push-types, because
125  // poll-type reads might trigger distribution of values to push-type variables via a ConsumingFanOut.
126  for(auto& variable : getAccessorListRecursive()) {
127  if(variable.getDirection().dir != VariableDirection::consuming) continue;
128  if(variable.getMode() == UpdateMode::poll) {
129  assert(!variable.getAppAccessorNoType().getHighLevelImplElement()->getAccessModeFlags().has(
130  AccessMode::wait_for_new_data));
131  Application::getInstance().getTestableMode().unlock("Initial value read for poll-type " + variable.getName());
132  Application::getInstance().circularDependencyDetector.registerDependencyWait(variable);
133  variable.getAppAccessorNoType().read();
134  Application::getInstance().circularDependencyDetector.unregisterDependencyWait(variable);
135  if(not Application::getInstance().getTestableMode().testLock()) {
136  // The lock may have already been acquired if the above read() goes to a ConsumingFanOut, which sends out
137  // the data to a slave decorated by a TestableModeAccessorDecorator. Hence we here must acquire the lock only
138  // if we do not have it.
139  Application::getInstance().getTestableMode().lock("Initial value read for poll-type " + variable.getName());
140  }
141  }
142  }
143  for(auto& variable : getAccessorListRecursive()) {
144  if(variable.getDirection().dir != VariableDirection::consuming) continue;
145  if(variable.getMode() == UpdateMode::push) {
146  Application::getInstance().getTestableMode().unlock("Initial value read for push-type " + variable.getName());
147  Application::getInstance().circularDependencyDetector.registerDependencyWait(variable);
148  // Will internally release and lock during the read, hence surround with lock/unlock
149  Application::getInstance().getTestableMode().lock("Initial value read for push-type " + variable.getName());
150  variable.getAppAccessorNoType().read();
151  Application::getInstance().getTestableMode().unlock("Initial value read for push-type " + variable.getName());
152  Application::getInstance().circularDependencyDetector.unregisterDependencyWait(variable);
153  Application::getInstance().getTestableMode().lock("Initial value read for push-type " + variable.getName());
154  }
155  }
156 
157  // We are holding the testable mode lock, so we are sure the mechanism will work now.
158  _testableModeReached = true;
159 
160  // enter the main loop
161  mainLoop();
162  Application::getInstance().getTestableMode().unlock("terminate");
163  }

178  std::list<EntityOwner*> ApplicationModule::getInputModulesRecursively(std::list<EntityOwner*> startList) {
179  if(_recursionStopper.recursionDetected()) {
180  return startList;
181  }
182 
183  // If this module is already in the list we found a circular dependency.
184  // Remember this for the next time the recursive scan calls this function
185  if(std::count(startList.begin(), startList.end(), this)) {
186  _recursionStopper.setRecursionDetected();
187  }
188 
189  // Whether a circular dependency has been detected or not, we must loop all inputs and add this module to the list
190  // so the calling code sees the second instance and can also detect the circle. The reason why we have to scan all
191  // inputs even if a circle is detected is this:
192  // * A single input starts the scan by adding it's owning module. At this point not all inputs if that module are in
193  // the circular network.
194  // * When a circle is detected, it might only be one of multiple entangled circled. If we would break the recursion
195  // and not scan all the
196  // inputs this is sufficient to identify that the particular input is in a circle. But at this point we have to
197  // tell in which network it is and have to scan the complete network to calculate the correct hash value.
198 
199  startList.push_back(this); // first we add this module to the start list. We will call all inputs with it.
200  std::list<EntityOwner*> returnList{
201  startList}; // prepare the return list. Deltas from the inputs will be added to it.
202  for(auto& accessor : this->getAccessorListRecursive()) {
203  // not consumed from network -> not an input, just continue
204  if(accessor.getDirection().dir != VariableDirection::consuming) continue;
205 
206  // find the feeder in the network
207  auto proxy = accessor.getModel().visit(Model::returnApplicationModule, Model::keepApplicationModules,
208  Model::keepPvAccess, Model::adjacentInSearch, Model::returnFirstHit(Model::ApplicationModuleProxy{}));
209  if(!proxy.isValid()) {
210  continue;
211  }
212  auto& feedingModule = proxy.getApplicationModule();
213 
214  auto thisInputsRecursiveModuleList = feedingModule.getInputModulesRecursively(startList);
215  // only add the modules that were added by the recursive search to the output list
216  assert(startList.size() <= thisInputsRecursiveModuleList.size());
217  auto copyStartIter = thisInputsRecursiveModuleList.begin();
218  std::advance(copyStartIter, startList.size());
219 
220  returnList.insert(returnList.end(), copyStartIter, thisInputsRecursiveModuleList.end());
221  }
222  return returnList;
223  }

246  DataValidity ApplicationModule::getDataValidity() const {
247  if(dataFaultCounter == 0) return DataValidity::ok;
248  if(_circularNetworkHash != 0) {
249  // In a circular dependency network, internal inputs are ignored.
250  // If all external inputs (including the ones from this module) are OK, the
251  // data validity is set to OK.
252  if(Application::getInstance().circularNetworkInvalidityCounters[_circularNetworkHash] == 0) {
253  return DataValidity::ok;
254  }
255  }
256  // not a circular network or invalidity counter not 0 -> keep the faulty flag
257  return DataValidity::faulty;
258  }
