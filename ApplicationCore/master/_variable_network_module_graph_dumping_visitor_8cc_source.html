54  // The Devices should have the first rank. (The ControlSystem is not represented by a node in the graph...)
55  // Arrows are pointing to the last rank
56  // In the twopi mode, the first rank is in the middle
57  // First, create a symmetric connection map containing all connections (twice, because symmetric)

61  std::string first = edge.first.substr(0, arrowPos);
62  std::string second = edge.first.substr(arrowPos + 2);
63  connectionMap[first].insert(second);
64  connectionMap[second].insert(first);
65  }

67  std::list<std::pair<std::string, std::string>> swapList;
68  std::unordered_set<std::string> constraintMap;
69  for(const auto& edge : _edgeMap) {

74  recurse = [&](const std::list<std::string>& path, size_t& distance) {
75  auto nextList = connectionMap[path.back()];
76  for(const auto& next : nextList) {
77  // stop recursion if device is found
78  if(next.substr(0, 7) == "Device_") {
79  if(path.size() > distance) {
80  // update distance
81  distance = path.size();
82  // update longestPath
83  longestPath = path;
84  longestPath.push_back(next);
85  }
86  return;
87  }
88

90  if(std::find(path.begin(), path.end(), next) != path.end()) {
91  return;
92  }
93

96  nextPath.push_back(next);
97  recurse(nextPath, distance);
98  }
99

101  if(path.size() == 1) {
102  std::string prevNode;
103  for(const auto& node : longestPath) {
104  if(!prevNode.empty()) {
105  constraintMap.insert(prevNode + "->" + node);
106  constraintMap.insert(node + "->" + prevNode);
107  }
108  prevNode = node;
109  }
110  }
111  };
112 
113  // split edge name to get the two module names
114  auto arrowPos = edge.first.find("->");
115  std::string first = edge.first.substr(0, arrowPos);
116  std::string second = edge.first.substr(arrowPos + 2);
117 
118  size_t firstDistance = 0;
119  size_t secondDistance = 0;
120 
121  // fast path: if one of the modules is a Device, there is no need to compute the distances
122  if(first.substr(0, 7) == "Device_") {
123  // first is the device, do not swap
124  secondDistance = 1;
125  constraintMap.insert(first + "->" + second);
126  constraintMap.insert(second + "->" + first);
127  }
128  else if(second.substr(0, 7) == "Device_") {
129  // second is the device, swap

131  constraintMap.insert(first + "->" + second);
132  constraintMap.insert(second + "->" + first);
133  }
134  else {

136  recurse({first}, firstDistance);
137  recurse({second}, secondDistance);
138  }
139 
140  // the end with the smaller (longest) distance must come first
141  if(secondDistance < firstDistance) {
142  // swap the two ends later (we cannot alter the map in the loop)
143  swapList.emplace_back(std::make_pair(first, second));
144  }
145  }
146 
147  // perform the pre-recorded swaps
148  for(auto& swapPair : swapList) {
149  _edgeMap[swapPair.second + "->" + swapPair.first] = _edgeMap.at(swapPair.first + "->" + swapPair.second);
150  _edgeMap[swapPair.second + "->" + swapPair.first].second = true;
151  auto it = _edgeMap.find(swapPair.first + "->" + swapPair.second);
152  assert(it != _edgeMap.end());
153  _edgeMap.erase(it);
154  }
155 
156  // create edges from device to invisble center
157  stream() << "TheCenter[label=\"TheCenter\",style=\"invis\"]\n";
158  for(const auto& device : _deviceList) {
159  stream() << "TheCenter->" << device << " [style=\"invis\"]\n";
160  }
161 
162  // create collected edges
163  for(const auto& edge : _edgeMap) {
164  stream() << edge.first;
165  stream() << "[tooltip=\"" << edge.second.first << "\"";
166  if(!constraintMap.count(edge.first)) {
167  stream() << ",constraint=false";
168  }
169  if(edge.second.second) {
170  stream() << ",dir=back";
171  }
172  stream() << "]\n";
173  }

202  if(secondSlash != std::string::npos) name = name.substr(secondSlash);

218  if(feeder.getType() == NodeType::Application) {

220  try {

222  }

224  // ignore if findApplicationModule() throws (may happen for the application variables coming from DeviceModules)
225  // TODO: handle these cases properly!
226  return;
227  }
228  }
229  else if(feeder.getType() == NodeType::Device) {
230  feedingId = "Device_" + feeder.getDeviceAlias();
231  }
232  else {
233  // ignore ControlSystem and constants etc.
234  return;
235  }
236 
237  for(auto& consumer : t.getConsumingNodes()) {
238  std::string consumingId;
239  if(consumer.getType() == NodeType::Application) {
240  auto& owner = dynamic_cast<Module&>(*consumer.getOwningModule());
241  try {
242  consumingId = "Module_" + std::to_string(size_t(owner.findApplicationModule()));
243  }
244  catch(ChimeraTK::logic_error& e) {
245  // ignore if findApplicationModule() throws (may happen for the application variables coming from DeviceModules)
246  // TODO: handle these cases properly!
247  return;
248  }
249  }
250  else if(consumer.getType() == NodeType::Device) {
251  consumingId = "Device_" + consumer.getDeviceAlias();
252  }
253  else {
254  // ignore ControlSystem and constants etc.
255  continue;
256  }
257 
258  // put/extend entry in edgeMap
259  if(!_edgeMap[feedingId + "->" + consumingId].first.empty()) {
260  _edgeMap[feedingId + "->" + consumingId].first += ",";

268  if(idx != std::string::npos) {
269  name = name.substr(idx + 1);
