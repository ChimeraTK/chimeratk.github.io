118  VirtualModule& DeviceModule::virtualiseFromCatalog() const {
119  if(virtualisedModuleFromCatalog_isValid) return virtualisedModuleFromCatalog;
120 
121  virtualisedModuleFromCatalog = VirtualModule(deviceAliasOrURI, "Device module", ModuleType::Device);
122 
123  if(!deviceIsInitialized) {
124  device = Device(deviceAliasOrURI);
125  deviceIsInitialized = true;
126  }
127  // obtain register catalogue
128  auto catalog = device.getRegisterCatalogue();
129  // iterate catalogue, create VariableNetworkNode for all registers starting
130  // with the registerNamePrefix
131  size_t prefixLength = registerNamePrefix.length();
132  for(auto& reg : catalog) {
133  if(std::string(reg.getRegisterName()).substr(0, prefixLength) != std::string(registerNamePrefix)) continue;
134 
135  // ignore 2D registers
136  if(reg.getNumberOfDimensions() > 1) continue;
137 
138  // guess direction and determine update mode
139  VariableDirection direction;
140  UpdateMode updateMode;
141  if(reg.isWriteable()) {
142  direction = {VariableDirection::consuming, false};
143  updateMode = UpdateMode::push;
144  }
145  else {
146  direction = {VariableDirection::feeding, false};
147  if(reg.getSupportedAccessModes().has(AccessMode::wait_for_new_data)) {
148  updateMode = UpdateMode::push;
149  }
150  else {
151  updateMode = UpdateMode::poll;
152  }
153  }
154 
155  // guess type
156  const std::type_info* valTyp{&typeid(AnyType)};
157  auto& dd = reg.getDataDescriptor(); // numeric, string, boolean, nodata, undefined
158  if(dd.fundamentalType() == RegisterInfo::FundamentalType::numeric) {
159  if(dd.isIntegral()) {
160  if(dd.isSigned()) {
161  if(dd.nDigits() > 11) {
162  valTyp = &typeid(int64_t);
163  }
164  else if(dd.nDigits() > 6) {
165  valTyp = &typeid(int32_t);
166  }
167  else if(dd.nDigits() > 4) {
168  valTyp = &typeid(int16_t);
169  }
170  else {
171  valTyp = &typeid(int8_t);
172  }
173  }
174  else {
175  if(dd.nDigits() > 10) {
176  valTyp = &typeid(uint64_t);
177  }
178  else if(dd.nDigits() > 5) {
179  valTyp = &typeid(uint32_t);
180  }
181  else if(dd.nDigits() > 3) {
182  valTyp = &typeid(uint16_t);
183  }
184  else {
185  valTyp = &typeid(uint8_t);
186  }
187  }
188  }
189  else { // fractional
190  // Maximum number of decimal digits to display a float without loss in non-exponential display, including
191  // sign, leading 0, decimal dot and one extra digit to avoid rounding issues (hence the +4).
192  // This computation matches the one performed in the NumericAddressedBackend catalogue.
193  size_t floatMaxDigits = std::max(std::log10(std::numeric_limits<float>::max()),
194  -std::log10(std::numeric_limits<float>::denorm_min())) +
195  4;
196  if(dd.nDigits() > floatMaxDigits) {
197  valTyp = &typeid(double);
198  }
199  else {
200  valTyp = &typeid(float);
201  }
202  }
203  }
204  else if(dd.fundamentalType() == RegisterInfo::FundamentalType::boolean) {
205  valTyp = &typeid(int32_t);
206  }
207  else if(dd.fundamentalType() == RegisterInfo::FundamentalType::string) {
208  valTyp = &typeid(std::string);
209  }
210  else if(dd.fundamentalType() == RegisterInfo::FundamentalType::nodata) {
211  valTyp = &typeid(int32_t);
212  }
213 
214  auto name = std::string(reg.getRegisterName()).substr(prefixLength);
215  auto lastSlash = name.find_last_of("/");
216  std::string dirname, basename;
217  if(lastSlash != std::string::npos) {
218  dirname = name.substr(0, lastSlash);
219  basename = name.substr(lastSlash + 1);
220  }
221  else {
222  dirname = "";
223  basename = name;
224  }
225  VariableNetworkNode node(
226  basename, deviceAliasOrURI, reg.getRegisterName(), updateMode, direction, *valTyp, reg.getNumberOfElements());
227  virtualisedModuleFromCatalog.createAndGetSubmoduleRecursive(dirname).addAccessor(node);
228  }
229 
230  virtualisedModuleFromCatalog_isValid = true;
231  return virtualisedModuleFromCatalog;
232  }

236  void DeviceModule::reportException(std::string errMsg) {
237  if(owner->isTestableModeEnabled()) {
238  assert(owner->testableModeTestLock());
239  }
240 
241  // The error queue must only be modified when holding both mutexes (error mutex and testable mode mutex), because
242  // the testable mode counter must always be consistent with the content of the queue.
243  // To avoid deadlocks you must always first aquire the testable mode mutex if you need both.
244  // You can hold the error mutex without holding the testable mode mutex (for instance for checking the error predicate),
245  // but then you must not try to aquire the testable mode mutex!
246  boost::unique_lock<boost::shared_mutex> errorLock(errorMutex);
247 
248  if(!deviceHasError) { // only report new errors if the device does not have reported errors already
249  if(errorQueue.push(errMsg)) {
250  if(owner->isTestableModeEnabled()) ++owner->testableMode_counter;
251  } // else do nothing. There are plenty of errors reported already: The queue is full.
252  // set the error flag and notify the other threads
253  deviceHasError = true;
254  exceptionVersionNumber = {}; // generate a new exception version number
255  errorLock.unlock();
256  }
257  else {
258  errorLock.unlock();
259  }
260  }

271  void DeviceModule::handleException() {
272  Application::registerThread("DM_" + getName());
273  std::string error;
274  owner->testableModeLock("Startup");
275 
276  // We have the testable mode lock. The device has not been initialised yet, but from now on the testableMode_deviceInitialisationCounter will take care or it
277  testableModeReached = true;
278 
279  // flag whether the devices was opened+initialised for the first time
280  bool firstSuccess = true;
281 
282  while(true) {
283  // [Spec: 2.3.1] (Re)-open the device.
284  do {
285  owner->testableModeUnlock("Wait before open/recover device");
286  usleep(500000);
287  boost::this_thread::interruption_point();
288  owner->testableModeLock("Attempt open/recover device");
289  try {
290  device.open();
291  }
292  catch(ChimeraTK::runtime_error& e) {
293  assert(deviceError.status != StatusOutput::Status::OK); // any error must already be reported...
294  if(std::string(deviceError.message) != e.what()) {
295  std::cerr << "Device " << deviceAliasOrURI << " reports error: " << e.what() << std::endl;
296  // set proper error message in very first attempt to open the device
297  deviceError.message = e.what();
298  deviceError.setCurrentVersionNumber({});
299  deviceError.message.write();
300  }
301  continue; // should not be necessary because isFunctional() should return false. But no harm in leaving it in.
302  }
303  } while(!device.isFunctional());
304 
305  boost::unique_lock<boost::shared_mutex> errorLock(errorMutex);
306 
307  // [Spec: 2.3.3] Empty exception reporting queue.
308  while(errorQueue.pop()) {
309  if(owner->isTestableModeEnabled()) {
310  assert(owner->testableMode_counter > 0);
311  --owner->testableMode_counter;
312  }
313  }
314  errorLock.unlock(); // we don't need to hold the lock for now, but we will need it later
315 
316  for(auto& writeMe : writeRegisterPaths) {
317  auto reg = device.getOneDRegisterAccessor<std::string>(writeMe); // the user data type does not matter here.
318  if(!reg.isWriteable()) {
319  throw ChimeraTK::logic_error(std::string(writeMe) + " is not writeable!");
320  }
321  }
322 
323  for(auto& readMe : readRegisterPaths) {
324  auto reg = device.getOneDRegisterAccessor<std::string>(readMe); // the user data type does not matter here.
325  if(!reg.isReadable()) {
326  throw ChimeraTK::logic_error(std::string(readMe) + " is not readable!");
327  }
328  }
329 
330  // [Spec: 2.3.2] Run initialisation handlers
331  try {
332  for(auto& initHandler : initialisationHandlers) {
333  initHandler(this);
334  }
335  }
336  catch(ChimeraTK::runtime_error& e) {
337  assert(deviceError.status != StatusOutput::Status::OK); // any error must already be reported...
338  // update error message, since it might have been changed...
339  if(std::string(deviceError.message) != e.what()) {
340  std::cerr << "Device " << deviceAliasOrURI << " reports error: " << e.what() << std::endl;
341  deviceError.message = e.what();
342  deviceError.setCurrentVersionNumber({});
343  deviceError.message.write();
344  }
345  // Jump back to re-opening the device
346  continue;
347  }
348 
349  // Write all recovery accessors
350  // We are now entering the critical recovery section. It is protected by the recovery mutex until the deviceHasError flag has been cleared.
351  boost::unique_lock<boost::shared_mutex> recoveryLock(recoveryMutex);
352  try {
353  // sort recovery helpers according to write order
354  recoveryHelpers.sort([](boost::shared_ptr<RecoveryHelper>& a, boost::shared_ptr<RecoveryHelper>& b) {
355  return a->writeOrder < b->writeOrder;
356  });
357  for(auto& recoveryHelper : recoveryHelpers) {
358  if(recoveryHelper->versionNumber != VersionNumber{nullptr}) {
359  recoveryHelper->accessor->write(recoveryHelper->versionNumber);
360  recoveryHelper->wasWritten = true;
361  }
362  }
363  }
364  catch(ChimeraTK::runtime_error& e) {
365  // update error message, since it might have been changed...
366  if(std::string(deviceError.message) != e.what()) {
367  std::cerr << "Device " << deviceAliasOrURI << " reports error: " << e.what() << std::endl;
368  deviceError.message = e.what();
369  deviceError.setCurrentVersionNumber({});
370  deviceError.message.write();
371  }
372  // Jump back to re-opening the device
373  continue;
374  }
375 
376  errorLock.lock();
377  deviceHasError = false;
378  errorLock.unlock();
379 
380  recoveryLock.unlock();
381 
382  // send the trigger that the device is available again
383  device.activateAsyncRead();
384  if(isHoldingInitialValueLatch) {
385  isHoldingInitialValueLatch = false;
386  initialValueLatch.count_down();
387  }
388 
389  // [Spec: 2.3.5] Reset exception state and wait for the next error to be reported.
390  deviceError.status = StatusOutput::Status::OK;
391  deviceError.message = "";
392  deviceError.setCurrentVersionNumber({});
393  deviceError.writeAll();
394  deviceBecameFunctional.write();
395 
396  if(!firstSuccess) {
397  std::cerr << "Device " << deviceAliasOrURI << " error cleared." << std::endl;
398  }
399  firstSuccess = false;
400 
401  // decrement special testable mode counter, was incremented manually above to make sure initialisation completes
402  // within one "application step"
403  if(Application::getInstance().testableMode) --owner->testableMode_deviceInitialisationCounter;
404 
405  // [Spec: 2.3.8] Wait for an exception being reported by the ExceptionHandlingDecorators
406  // release the testable mode mutex for waiting for the exception.
407  owner->testableModeUnlock("Wait for exception");
408 
409  // Do not modify the queue without holding the testable mode lock, because we also consistently have to modify
410  // the counter protected by that mutex.
411  // Just call wait(), not pop_wait().
412  boost::this_thread::interruption_point();
413  errorQueue.wait();
414  boost::this_thread::interruption_point();
415 
416  owner->testableModeLock("Process exception");
417  // increment special testable mode counter to make sure the initialisation completes within one
418  // "application step"
419  if(Application::getInstance().testableMode) {
420  ++owner->testableMode_deviceInitialisationCounter; // matched above with a decrement
421  }
422 
423  errorLock.lock(); // we need both locks to modify the queue
424 
425  auto popResult = errorQueue.pop(error);
426  assert(popResult); // this if should always be true, otherwise the waiting did not work.
427  (void)popResult; // avoid warning in production build. g++5.4 does not support [[maybe_unused]] yet.
428  if(owner->isTestableModeEnabled()) {
429  assert(owner->testableMode_counter > 0);
430  --owner->testableMode_counter;
431  }
432 
433  // [ExceptionHandling Spec: C.3.3.14] report exception to the control system
434  std::cerr << "Device " << deviceAliasOrURI << " reports error: " << error << std::endl;
435  deviceError.status = StatusOutput::Status::FAULT;
436  deviceError.message = error;
437  deviceError.setCurrentVersionNumber({});
438  deviceError.writeAll();
439 
440  // We must not hold the lock while waiting for the synchronousTransferCounter to go back to 0. Only release it
441  // after deviceError has been written, so the CircularDependencyDetector can read the error message from its
442  // thread for printing.
443  errorLock.unlock();
444 
445  // [ExceptionHandling Spec: C.3.3.15] Wait for all synchronous transfers to finish before starting recovery.
446  while(synchronousTransferCounter > 0) {
447  usleep(1000);
448  }
449 
450  } // while(true)
451  }

455  void DeviceModule::prepare() {
456  if(!deviceIsInitialized) {
457  device = Device(deviceAliasOrURI);
458  deviceIsInitialized = true;

464  deviceError.setCurrentVersionNumber({});

466
