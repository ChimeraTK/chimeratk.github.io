16  void VariableNetworkDumpingVisitor::dispatch(const VariableNetwork& t) {
17  stream() << _prefix << "VariableNetwork";
18  stream() << " {" << std::endl;
19  stream() << _prefix << " value type = " << boost::core::demangle(t.getValueType().name())
20  << ", engineering unit = " << t.getUnit() << std::endl;
21  stream() << _prefix << " trigger type = ";
22  try {
23  auto tt = t.getTriggerType(false);
24  if(tt == VariableNetwork::TriggerType::feeder) stream() << "feeder";
25  if(tt == VariableNetwork::TriggerType::pollingConsumer) stream() << "pollingConsumer";
26  if(tt == VariableNetwork::TriggerType::external) stream() << "external";
27  if(tt == VariableNetwork::TriggerType::none) stream() << "none";
28  stream() << std::endl;
29  }
30  catch(ChimeraTK::logic_error& e) {
31  stream() << "**error**" << std::endl;
32  }
33  stream() << _prefix << " feeder";
34  if(t.hasFeedingNode()) {
35  t.getFeedingNode().accept(*this);
36  }
37  else {
38  stream() << " **error, no feeder found**" << std::endl;
39  }
40  stream() << _prefix << " consumers: " << t.countConsumingNodes() << std::endl;
41  size_t count = 0;
42  for(auto& consumer : t.getConsumingNodes()) {
43  if(consumer.getDirection().dir != VariableDirection::consuming) continue;
44  stream() << _prefix << " # " << ++count << ":";
45  consumer.accept(*this);
46  }
47  if(t.hasFeedingNode()) {
48  if(t.getFeedingNode().hasExternalTrigger()) {
49  stream() << _prefix << " external trigger node: ";
50  t.getFeedingNode().getExternalTrigger().accept(*this);
51  }
52  }
53  stream() << _prefix << "}" << std::endl;
54  }
