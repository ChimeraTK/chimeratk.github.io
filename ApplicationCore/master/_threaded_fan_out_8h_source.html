54  virtual void run() {
55  Application::registerThread("ThFO" + FanOut<UserType>::impl->getName());
56  Application::testableModeLock("start");
57  testableModeReached = true;
58 
59  ChimeraTK::VersionNumber version{nullptr};
60  version = readInitialValues();
61  while(true) {
62  // send out copies to slaves
63  Profiler::startMeasurement();
64  boost::this_thread::interruption_point();
65  auto validity = FanOut<UserType>::impl->dataValidity();
66  for(auto& slave : FanOut<UserType>::slaves) {
67  // do not send copy if no data is expected (e.g. trigger)
68  if(slave->getNumberOfSamples() != 0) {
69  slave->accessChannel(0) = FanOut<UserType>::impl->accessChannel(0);
70  }
71  slave->setDataValidity(validity);
72  bool dataLoss = slave->writeDestructively(version);
73  if(dataLoss) Application::incrementDataLossCounter(slave->getName());
74  }
75  // receive data
76  boost::this_thread::interruption_point();
77  Profiler::stopMeasurement();
78  FanOut<UserType>::impl->read();
79  version = FanOut<UserType>::impl->getVersionNumber();
80  }
81  }

121  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> slave, VariableNetworkNode& consumer) override {
122  // TODO Adding slaves is currently by done by the ThreadedFanOut base class.
123  // Refactor constructors and addSlaves for all FanOuts?
124  //FanOut<UserType>::addSlave(slave, consumer);
125  if(consumer.getDirection().withReturn) {
126  assert(_returnChannelSlave == nullptr);

131  void run() override {
132  Application::registerThread("ThFO" + FanOut<UserType>::impl->getName());
133  Application::testableModeLock("start");
134  testableModeReached = true;
135 
136  TransferElementID var;
137  ChimeraTK::VersionNumber version{nullptr};
138 
139  version = readInitialValues();
140 
141  ReadAnyGroup group({FanOut<UserType>::impl, _returnChannelSlave});
142  while(true) {
143  // send out copies to slaves
144  for(auto& slave : FanOut<UserType>::slaves) {
145  // do not feed back value returnChannelSlave if it was received from it
146  if(slave->getId() == var) continue;
147  // do not send copy if no data is expected (e.g. trigger)
148  if(slave->getNumberOfSamples() != 0) {
149  slave->accessChannel(0) = FanOut<UserType>::impl->accessChannel(0);
150  }
151  bool dataLoss = slave->writeDestructively(version);
152  if(dataLoss) Application::incrementDataLossCounter(slave->getName());
153  }
154  // receive data
155  boost::this_thread::interruption_point();
156  Profiler::stopMeasurement();
157  var = group.readAny();
158  Profiler::startMeasurement();
159  boost::this_thread::interruption_point();
160  // if the update came through the return channel, return it to the feeder
161  if(var == _returnChannelSlave->getId()) {
162  FanOut<UserType>::impl->accessChannel(0).swap(_returnChannelSlave->accessChannel(0));
163  if(version < _returnChannelSlave->getVersionNumber()) {
164  version = _returnChannelSlave->getVersionNumber();
165  }
166  FanOut<UserType>::impl->write(version);
167  }
168  else {
169  version = FanOut<UserType>::impl->getVersionNumber();
170  }
171  }
172  }
173 
174  protected:
176  boost::thread _thread;
177 
178  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> _returnChannelSlave;
179 
180  using ThreadedFanOut<UserType>::_network;
181  using ThreadedFanOut<UserType>::readInitialValues;
182  using EntityOwner::testableModeReached;
183  };
