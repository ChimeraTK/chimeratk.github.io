53  virtual void run() {
54  Application::registerThread("ThFO" + FanOut<UserType>::impl->getName());
55  Application::testableModeLock("start");
56  testableModeReached = true;
57 
58  ChimeraTK::VersionNumber version{nullptr};
59  version = readInitialValues();
60  while(true) {
61  // send out copies to slaves
62  Profiler::startMeasurement();
63  boost::this_thread::interruption_point();
64  auto validity = FanOut<UserType>::impl->dataValidity();
65  for(auto& slave : FanOut<UserType>::slaves) {
66  // do not send copy if no data is expected (e.g. trigger)
67  if(slave->getNumberOfSamples() != 0) {
68  slave->accessChannel(0) = FanOut<UserType>::impl->accessChannel(0);
69  }
70  slave->setDataValidity(validity);

74  // receive data
75  boost::this_thread::interruption_point();
76  Profiler::stopMeasurement();

116  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> slave, VariableNetworkNode& consumer) override {
117  // TODO Adding slaves is currently by done by the ThreadedFanOut base class.
118  // Refactor constructors and addSlaves for all FanOuts?

126  void run() override {
127  Application::registerThread("ThFO" + FanOut<UserType>::impl->getName());
128  Application::testableModeLock("start");
129  testableModeReached = true;
130 
131  TransferElementID var;
132  ChimeraTK::VersionNumber version{nullptr};
133 
134  version = readInitialValues();
135 
136  ReadAnyGroup group({FanOut<UserType>::impl, _returnChannelSlave});
137  while(true) {
138  // send out copies to slaves
139  for(auto& slave : FanOut<UserType>::slaves) {
140  // do not feed back value returnChannelSlave if it was received from it
141  if(slave->getId() == var) continue;
142  // do not send copy if no data is expected (e.g. trigger)
143  if(slave->getNumberOfSamples() != 0) {
144  slave->accessChannel(0) = FanOut<UserType>::impl->accessChannel(0);
145  }
146  bool dataLoss = slave->writeDestructively(version);
147  if(dataLoss) Application::incrementDataLossCounter();
148  }
149  // receive data
150  boost::this_thread::interruption_point();
151  Profiler::stopMeasurement();
152  var = group.readAny();
153  Profiler::startMeasurement();
154  boost::this_thread::interruption_point();
155  // if the update came through the return channel, return it to the feeder
156  if(var == _returnChannelSlave->getId()) {
157  FanOut<UserType>::impl->accessChannel(0).swap(_returnChannelSlave->accessChannel(0));
158  if(version < _returnChannelSlave->getVersionNumber()) {
159  version = _returnChannelSlave->getVersionNumber();
160  }
161  FanOut<UserType>::impl->write(version);
162  }
163  else {
164  version = FanOut<UserType>::impl->getVersionNumber();
165  }
166  }
167  }
168 
169  protected:
171  boost::thread _thread;
172 
173  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> _returnChannelSlave;
174 
175  using ThreadedFanOut<UserType>::_network;
176  using ThreadedFanOut<UserType>::readInitialValues;
177  using EntityOwner::testableModeReached;
178  };
