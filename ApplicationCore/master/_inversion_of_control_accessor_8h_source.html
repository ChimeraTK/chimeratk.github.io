108  template<typename Derived>
109  void InversionOfControlAccessor<Derived>::replace(Derived&& other) {
110  assert(static_cast<Derived*>(this)->_impl == nullptr && other._impl == nullptr);
111 
112  // remove node from model
113  if(getModel().isValid()) {
114  getModel().removeNode(node);
115  }
116 
117  // remove accessor from owning module
118  if(node.getType() == NodeType::Application) {
119  getOwner()->unregisterAccessor(node);
120  }
121 
122  // transfer the node
123  node = std::move(other.node);
124  other.node = VariableNetworkNode(); // Make sure the destructor of other sees an invalid node
125 
126  // update the app accesor pointer in the node
127  if(node.getType() == NodeType::Application) {
128  node.setAppAccessorPointer(static_cast<Derived*>(this));
129  }
130  else {
131  assert(node.getType() == NodeType::invalid);
132  }
133  // Note: the accessor is registered by the VariableNetworkNode, so we don't have to re-register.
134  }

158  auto path = Utilities::getPathName(name);

160 
162  auto addToOnwer = [&](auto& owner_casted) {
163  auto model = owner_casted.getModel();
164  if(!model.isValid()) {
165  // this happens e.g. for default-constructed owners and their sub-modules
166  return;
167  }
168  auto neighbourDir = model.visit(
169  Model::returnDirectory, Model::getNeighbourDirectory, Model::returnFirstHit(Model::DirectoryProxy{}));
170 
171  auto dir = neighbourDir.addDirectoryRecursive(Utilities::getPathName(name));
172  auto var = dir.addVariable(Utilities::getUnqualifiedName(name));
173 
174  node.setModel(var);
175 
176  model.addVariable(var, node);
177  };
178  auto* owner_am = dynamic_cast<ApplicationModule*>(owner);
179  auto* owner_vg = dynamic_cast<VariableGroup*>(owner);
180  if(owner_am) {
181  addToOnwer(*owner_am);
182  }
183  else if(owner_vg) {
184  addToOnwer(*owner_vg);
185  }
186  else {
187  throw ChimeraTK::logic_error("Hierarchy error!?");
188  }
189 
190  owner->registerAccessor(node);
191  }
