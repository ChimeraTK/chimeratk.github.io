102  template<typename Derived>
103  void InversionOfControlAccessor<Derived>::replace(Derived&& other) {
104  assert(static_cast<Derived*>(this)->_impl == nullptr && other._impl == nullptr);
105  if(getOwner() != nullptr) getOwner()->unregisterAccessor(node);
106  node = other.node; // just copies the pointer, but other will be destroyed
107  // right after this move constructor
108  other.node = VariableNetworkNode();
109  if(node.getType() == NodeType::Application) {
110  node.setAppAccessorPointer(static_cast<Derived*>(this));
111  }
112  else {
113  assert(node.getType() == NodeType::invalid);
114  }
115  // Note: the accessor is registered by the VariableNetworkNode, so we don't
116  // have to re-register.
117  }
