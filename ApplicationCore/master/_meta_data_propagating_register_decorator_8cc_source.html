13  void MetaDataPropagatingRegisterDecorator<T>::doPostRead(TransferType type, bool hasNewData) {
14  NDRegisterAccessorDecorator<T, T>::doPostRead(type, hasNewData);
15 
16  // update the version number
17  if(_target->getAccessModeFlags().has(AccessMode::wait_for_new_data) && type == TransferType::read) {
18  _owner->setCurrentVersionNumber(this->getVersionNumber());
19  }
20 
21  // Check if the data validity flag changed. If yes, propagate this information to the owning module and the application
22  if(_dataValidity != lastValidity) {
23  if(_dataValidity == DataValidity::faulty) { // data validity changes to faulty
24  _owner->incrementDataFaultCounter();
25  // external inpput in a circular dependency network
26  if(_owner->getCircularNetworkHash() && !_isCircularInput) {
27  ++(Application::getInstance().circularNetworkInvalidityCounters[_owner->getCircularNetworkHash()]);
28  }
29  }
30  else { // data validity changed to OK
31  _owner->decrementDataFaultCounter();
32  // external inpput in a circular dependency network
33  if(_owner->getCircularNetworkHash() && !_isCircularInput) {
34  --(Application::getInstance().circularNetworkInvalidityCounters[_owner->getCircularNetworkHash()]);
35  }
36  }
37  lastValidity = _dataValidity;
38  }
39  }

41  template<typename T>
42  void MetaDataPropagatingRegisterDecorator<T>::doPreWrite(TransferType type, VersionNumber versionNumber) {
43  // We cannot use NDRegisterAccessorDecorator<T> here because we need a different implementation of setting the
44  // target data validity. So we have a complete implemetation here.
45 
46  if(_owner->getCircularNetworkHash() && _dataValidity != lastValidity) {
47  // In circular dependency networks an output which actively has DataValidity::faulty set by the user logic is handled
48  // as if an external input was invalid -> increase or decrease the network's invalidity counter accordingly
49  if(_dataValidity == DataValidity::faulty) { // data validity changes to faulty
50  ++(Application::getInstance().circularNetworkInvalidityCounters[_owner->getCircularNetworkHash()]);
51  }
52  else {
53  --(Application::getInstance().circularNetworkInvalidityCounters[_owner->getCircularNetworkHash()]);
54  }
55  lastValidity = _dataValidity;
56  }
57 
58  // Now propagate the flag and the data to the target and perform the write
59  if(_dataValidity == DataValidity::faulty) { // the application has manualy set the validity to faulty
60  _target->setDataValidity(DataValidity::faulty);
61  }
62  else { // automatic propagation of the owner validity
63  _target->setDataValidity(_owner->getDataValidity());
64  }
65 
66  for(unsigned int i = 0; i < _target->getNumberOfChannels(); ++i) {
67  buffer_2D[i].swap(_target->accessChannel(i));
68  }
69  _target->preWrite(type, versionNumber);
70  }
