98 void Application::optimiseUnmappedVariables(const std::set<std::string>& names) {
99  for(const auto& pv : names) {
100  auto& node = controlSystemVariables.at(pv);
101  auto& network = node.getOwner();
102  if(network.getFeedingNode() == node) {
103  // cannot optimise if network is fed by unmapped control system variable (anyway not eating CPU ressources)
104  continue;
105  }
106  if(network.getConsumingNodes().size() == 1) {
107  if(network.getFeedingNode().getType() == NodeType::Application) {
108  callForType(network.getValueType(), [&](auto t) {
109  using UserType = decltype(t);
110 
111  // replace comsuming node with constant in the model
112  network.removeNode(network.getConsumingNodes().front());
113  auto constNode = VariableNetworkNode::makeConstant<UserType>(
114  false, UserType(), network.getFeedingNode().getNumberOfElements());
115  network.addNode(constNode);
116 
117  // change application accessor into constant
118  auto& acc = network.getFeedingNode().getAppAccessor<UserType>();
119  auto constImpl = constNode.template createConstAccessor<UserType>({});
120  acc.replace(constImpl);
121  });
122  }
123  else {
124  auto fanOut = network.getFanOut();
125  assert(fanOut);
126  // FanOut is present: disable it
127  fanOut->disable();
128  }
129  }
130  else {
131  // more than one consumer: we have a fan out -> remove control system consumer from it.
132  auto fanOut = network.getFanOut();
133  assert(fanOut != nullptr);
134  fanOut->removeSlave(_processVariableManager->getProcessVariable(pv));
135  }
136  }
137 }

167 void Application::processUnconnectedNodes() {
168  for(auto& module : getSubmoduleListRecursive()) {
169  for(auto& accessor : module->getAccessorList()) {
170  if(!accessor.hasOwner()) {
171  if(enableUnconnectedVariablesWarning) {
172  std::cerr << "*** Warning: Variable '" << accessor.getQualifiedName()
173  << "' is not connected. " // LCOV_EXCL_LINE
174  "Reading will always result in 0, writing will be ignored."
175  << std::endl; // LCOV_EXCL_LINE
176  }
177  networkList.emplace_back();
178  networkList.back().addNode(accessor);
179 
180  bool makeFeeder = !(networkList.back().hasFeedingNode());
181  size_t length = accessor.getNumberOfElements();
182  auto callable = CreateConstantForUnconnectedVar(accessor.getValueType(), makeFeeder, length);
183  boost::fusion::for_each(ChimeraTK::userTypeMap(), std::ref(callable));
184  assert(callable.done);
185  constantList.emplace_back(callable.theNode);
186  networkList.back().addNode(constantList.back());
187  }
188  }
189  }

216 void Application::run() {
217  assert(applicationName != "");
218 
219  if(testableMode) {
220  if(!testFacilityRunApplicationCalled) {
221  throw ChimeraTK::logic_error(
222  "Testable mode enabled but Application::run() called directly. Call TestFacility::runApplication() instead.");
223  }
224  }
225 
226  if(runCalled) {
227  throw ChimeraTK::logic_error("Application::run() has already been called before.");
228  }
229  runCalled = true;
230 
231  // set all initial version numbers in the modules to the same value
232  for(auto& module : getSubmoduleListRecursive()) {
233  if(module->getModuleType() != ModuleType::ApplicationModule) continue;
234  module->setCurrentVersionNumber(getStartVersion());
235  }
236 
237  // prepare the modules
238  for(auto& module : getSubmoduleListRecursive()) {
239  module->prepare();
240  }
241  for(auto& deviceModule : deviceModuleMap) {
242  deviceModule.second->prepare();
243  }
244 
245  // Switch life-cycle state to run
246  lifeCycleState = LifeCycleState::run;
247 
248  // start the necessary threads for the FanOuts etc.
249  for(auto& internalModule : internalModuleList) {
250  internalModule->activate();
251  }
252 
253  for(auto& deviceModule : deviceModuleMap) {
254  deviceModule.second->run();
255  }
256 
257  // start the threads for the modules
258  for(auto& module : getSubmoduleListRecursive()) {
259  module->run();
260  }
261 
262  // When in testable mode, wait for all modules to report that they have reched the testable mode.
263  // We have to start all module threads first because some modules might only send the initial
264  // values in their main loop, and following modules need them to enter testable mode.
265 
266  // just a small helper lambda to avoid code repetition
267  auto waitForTestableMode = [](EntityOwner* module) {
268  while(!module->hasReachedTestableMode()) {
269  Application::getInstance().testableModeUnlock("releaseForReachTestableMode");
270  usleep(100);
271  Application::getInstance().testableModeLock("acquireForReachTestableMode");
272  }
273  };
274 
275  if(Application::getInstance().isTestableModeEnabled()) {
276  for(auto& internalModule : internalModuleList) {
277  waitForTestableMode(internalModule.get());
278  }
279  for(auto& deviceModule : deviceModuleMap) {
280  waitForTestableMode(deviceModule.second);
281  }
282  for(auto& module : getSubmoduleListRecursive()) {
283  waitForTestableMode(module);
284  }
285  }
286 
287  // Launch circular dependency detector thread
288  circularDependencyDetector.startDetectBlockedModules();
289 }

357 VariableNetwork& Application::connect(VariableNetworkNode a, VariableNetworkNode b) {
358  // if one of the nodes has the value type AnyType, set it to the type of the
359  // other if both are AnyType, nothing changes.
360  if(a.getValueType() == typeid(AnyType)) {
361  a.setValueType(b.getValueType());
362  }
363  else if(b.getValueType() == typeid(AnyType)) {
364  b.setValueType(a.getValueType());
365  }
366 
367  // if one of the nodes has not yet a defined number of elements, set it to the
368  // number of elements of the other. if both are undefined, nothing changes.
369  if(a.getNumberOfElements() == 0) {
370  a.setNumberOfElements(b.getNumberOfElements());
371  }
372  else if(b.getNumberOfElements() == 0) {
373  b.setNumberOfElements(a.getNumberOfElements());
374  }
375  if(a.getNumberOfElements() != b.getNumberOfElements()) {
376  std::stringstream what;
377  what << "*** ERROR: Cannot connect array variables with difference number "
378  "of elements!"
379  << std::endl;
380  what << "Node A:" << std::endl;
381  a.dump(what);
382  what << "Node B:" << std::endl;
383  b.dump(what);
384  throw ChimeraTK::logic_error(what.str());
385  }
386 
387  // if both nodes already have an owner, we are either already done (same
388  // owners) or we need to try to merge the networks
389  if(a.hasOwner() && b.hasOwner()) {
390  if(&(a.getOwner()) != &(b.getOwner())) {
391  auto& networkToMerge = b.getOwner();
392  bool success = a.getOwner().merge(networkToMerge);
393  if(!success) {
394  std::stringstream what;
395  what << "*** ERROR: Trying to connect two nodes which are already part "
396  "of different networks, and merging these"
397  " networks is not possible (cannot have two non-control-system "
398  "or two control-system feeders)!"
399  << std::endl;
400  what << "Node A:" << std::endl;
401  a.dump(what);
402  what << "Node B:" << std::endl;
403  b.dump(what);
404  what << "Owner of node A:" << std::endl;
405  a.getOwner().dump("", what);
406  what << "Owner of node B:" << std::endl;
407  b.getOwner().dump("", what);
408  throw ChimeraTK::logic_error(what.str());
409  }
410  for(auto n = networkList.begin(); n != networkList.end(); ++n) {
411  if(&*n == &networkToMerge) {
412  networkList.erase(n);
413  break;
414  }
415  }
416  }
417  }
418  // add b to the existing network of a
419  else if(a.hasOwner()) {
420  a.getOwner().addNode(b);
421  }
422 
423  // add a to the existing network of b
424  else if(b.hasOwner()) {
425  b.getOwner().addNode(a);
426  }
427  // create new network
428  else {
429  networkList.emplace_back();
430  networkList.back().addNode(a);
431  networkList.back().addNode(b);
432  }
433  return a.getOwner();
434 }

439 boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> Application::createDeviceVariable(
440  VariableNetworkNode const& node) {
441  auto deviceAlias = node.getDeviceAlias();
442  auto registerName = node.getRegisterName();
443  auto direction = node.getDirection();
444  auto mode = node.getMode();
445  auto nElements = node.getNumberOfElements();
446 
447  // Device opens in DeviceModule
448  if(deviceMap.count(deviceAlias) == 0) {
449  deviceMap[deviceAlias] = ChimeraTK::BackendFactory::getInstance().createBackend(deviceAlias);
450  }
451 
452  // use wait_for_new_data mode if push update mode was requested
453  // Feeding to the network means reading from a device to feed it into the network.
454  ChimeraTK::AccessModeFlags flags{};
455  if(mode == UpdateMode::push && direction.dir == VariableDirection::feeding) flags = {AccessMode::wait_for_new_data};
456 
457  // obtain the register accessor from the device
458  auto accessor = deviceMap[deviceAlias]->getRegisterAccessor<UserType>(registerName, nElements, 0, flags);
459 
460  // Receiving accessors should be faulty after construction,
461  // see data validity propagation spec 2.6.1
462  if(node.getDirection().dir == VariableDirection::feeding) {
463  accessor->setDataValidity(DataValidity::faulty);
464  }
465 
466  // decorate push-type feeders with testable mode decorator, if needed
467  if(testableMode) {
468  if(mode == UpdateMode::push && direction.dir == VariableDirection::feeding) {
469  auto varId = getNextVariableId();
470  accessor = boost::make_shared<TestableModeAccessorDecorator<UserType>>(accessor, true, false, varId, varId);
471  }
472  }
473 
474  return boost::make_shared<ExceptionHandlingDecorator<UserType>>(accessor, node);
475 }

479 template<typename UserType>
480 boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> Application::createProcessVariable(
481  VariableNetworkNode const& node) {
482  // determine the SynchronizationDirection
483  SynchronizationDirection dir;
484  if(node.getDirection().withReturn) {
485  dir = SynchronizationDirection::bidirectional;
486  }
487  else if(node.getDirection().dir == VariableDirection::feeding) {
488  dir = SynchronizationDirection::controlSystemToDevice;
489  }
490  else {
491  dir = SynchronizationDirection::deviceToControlSystem;
492  }
493  AccessModeFlags flags = {};
494  if(node.getDirection().dir == VariableDirection::consuming) { // Application-to-controlsystem must be
495  // push-type
496  flags = {AccessMode::wait_for_new_data};
497  }
498  else {
499  if(node.getOwner().getConsumingNodes().size() == 1 &&
500  node.getOwner().getConsumingNodes().front().getType() == NodeType::Application) {
501  // exactly one consumer which is an ApplicationModule input: decide flag depending on input type
502  auto consumer = node.getOwner().getConsumingNodes().front();
503  if(consumer.getMode() == UpdateMode::push) flags = {AccessMode::wait_for_new_data};
504  }
505  else {
506  // multiple consumers (or a single Device/CS consumer) result in a ThreadedFanOut which requires push-type input
507  flags = {AccessMode::wait_for_new_data};
508  }
509  }
510 
511  // create the ProcessArray for the proper UserType
512  auto pvar = _processVariableManager->createProcessArray<UserType>(dir, node.getPublicName(),
513  node.getNumberOfElements(), node.getOwner().getUnit(), node.getOwner().getDescription(), {}, 3, flags);
514  assert(pvar->getName() != "");
515 
516  // create variable ID
517  auto varId = getNextVariableId();
518  pvIdMap[pvar->getUniqueId()] = varId;
519 
520  // Decorate the process variable if testable mode is enabled and this is the receiving end of the variable (feeding
521  // to the network), or a bidirectional consumer. Also don't decorate, if the mode is polling. Instead flag the
522  // variable to be polling, so the TestFacility is aware of this.
523  if(testableMode) {
524  if(node.getDirection().dir == VariableDirection::feeding) {
525  // The transfer mode of this process variable is considered to be polling,
526  // if only one consumer exists and this consumer is polling. Reason:
527  // mulitple consumers will result in the use of a FanOut, so the
528  // communication up to the FanOut will be push-type, even if all consumers
529  // are poll-type.
531  auto mode = UpdateMode::push;
532  if(node.getOwner().countConsumingNodes() == 1) {
533  if(node.getOwner().getConsumingNodes().front().getMode() == UpdateMode::poll) mode = UpdateMode::poll;
534  }
535 
536  if(mode != UpdateMode::poll) {
537  auto pvarDec = boost::make_shared<TestableModeAccessorDecorator<UserType>>(pvar, true, false, varId, varId);
538  testableMode_names[varId] = "ControlSystem:" + node.getPublicName();
539  return pvarDec;
540  }
541  else {
542  testableMode_isPollMode[varId] = true;
543  }
544  }
545  else if(node.getDirection().withReturn) {
546  // Return channels are always push. The decorator must handle only reads on the return channel, since writes into
547  // the control system do not block the testable mode.
548  auto pvarDec = boost::make_shared<TestableModeAccessorDecorator<UserType>>(pvar, true, false, varId, varId);
549  testableMode_names[varId] = "ControlSystem:" + node.getPublicName();
550  return pvarDec;
551  }
552  }
553 
554  // return the process variable
555  return pvar;
556 }

560 template<typename UserType>

562  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>>>
563  Application::createApplicationVariable(VariableNetworkNode const& node, VariableNetworkNode const& consumer) {

567  assert(name != "");
568  AccessModeFlags flags = {};
569  if(consumer.getType() != NodeType::invalid) {
570  if(consumer.getMode() == UpdateMode::push) flags = {AccessMode::wait_for_new_data};
571  }
572  else {
573  if(node.getMode() == UpdateMode::push) flags = {AccessMode::wait_for_new_data};
574  }
575 
576  // create the ProcessArray for the proper UserType
577  std::pair<boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>>,
578  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>>>
579  pvarPair;

581  assert(node.getDirection().withReturn == consumer.getDirection().withReturn);
582  if(!node.getDirection().withReturn) {
583  pvarPair =
584  createSynchronizedProcessArray<UserType>(nElements, name, node.getUnit(), node.getDescription(), {}, 3, flags);
585  }
586  else {
587  pvarPair = createBidirectionalSynchronizedProcessArray<UserType>(
588  nElements, name, node.getUnit(), node.getDescription(), {}, 3, flags);
589  }

591  assert(pvarPair.second->getName() != "");
592

594  size_t varId = getNextVariableId();
595  size_t varIdReturn;
596  if(node.getDirection().withReturn) varIdReturn = getNextVariableId();
597 
598  // decorate the process variable if testable mode is enabled and mode is push-type
599  if(testableMode && flags.has(AccessMode::wait_for_new_data)) {
600  if(!node.getDirection().withReturn) {
601  pvarPair.first =
602  boost::make_shared<TestableModeAccessorDecorator<UserType>>(pvarPair.first, false, true, varId, varId);
603  pvarPair.second =
604  boost::make_shared<TestableModeAccessorDecorator<UserType>>(pvarPair.second, true, false, varId, varId);
605  }
606  else {
607  pvarPair.first =
608  boost::make_shared<TestableModeAccessorDecorator<UserType>>(pvarPair.first, true, true, varIdReturn, varId);
609  pvarPair.second =
610  boost::make_shared<TestableModeAccessorDecorator<UserType>>(pvarPair.second, true, true, varId, varIdReturn);
611  }
612 
613  // put the decorators into the list
614  testableMode_names[varId] = "Internal:" + node.getQualifiedName();
615  if(consumer.getType() != NodeType::invalid) {
616  testableMode_names[varId] += "->" + consumer.getQualifiedName();
617  }
618  if(node.getDirection().withReturn) testableMode_names[varIdReturn] = testableMode_names[varId] + " (return)";
619  }
620 
621  // if debug mode was requested for either node, decorate both accessors
622  if(debugMode_variableList.count(node.getUniqueId()) ||
623  (consumer.getType() != NodeType::invalid && debugMode_variableList.count(consumer.getUniqueId()))) {
624  if(consumer.getType() != NodeType::invalid) {
625  assert(node.getDirection().dir == VariableDirection::feeding);
626  assert(consumer.getDirection().dir == VariableDirection::consuming);
627  pvarPair.first =
628  boost::make_shared<DebugPrintAccessorDecorator<UserType>>(pvarPair.first, node.getQualifiedName());
629  pvarPair.second =
630  boost::make_shared<DebugPrintAccessorDecorator<UserType>>(pvarPair.second, consumer.getQualifiedName());
631  }
632  else {
633  pvarPair.first =
634  boost::make_shared<DebugPrintAccessorDecorator<UserType>>(pvarPair.first, node.getQualifiedName());
635  pvarPair.second =
636  boost::make_shared<DebugPrintAccessorDecorator<UserType>>(pvarPair.second, node.getQualifiedName());
637  }
638  }
639 
640  // return the pair
641  return pvarPair;
642 }

672 void Application::findConstantNodes() {
673  for(auto& module : getSubmoduleListRecursive()) {
674  for(auto& accessor : module->getAccessorList()) {
675  if(accessor.getName().substr(0, 7) == "@CONST@") {
676  VariableNetworkNode node(accessor);
677 
678  // must be a consumer node
679  if(node.getDirection().dir != VariableDirection::consuming) {
680  std::cout << "Variable name '@CONST@' found for a feeding type node:" << std::endl;
681  node.dump();
682  if(node.hasOwner()) {
683  std::cout << "In network:" << std::endl;
684  node.getOwner().dump();
685  }
686  throw ChimeraTK::logic_error("Variable name '@CONST@' found for a feeding type node.");
687  }
688 
689  // must have no feeder, a constant feeder, or a control-system feeder
690  if(node.hasOwner() && node.getOwner().hasFeedingNode() &&
691  node.getOwner().getFeedingNode().getType() != NodeType::Constant &&
692  node.getOwner().getFeedingNode().getType() != NodeType::ControlSystem) {
693  std::cout << "Error: Variable name '@CONST@' found with wrong feeder type:" << std::endl;
694  node.getOwner().getFeedingNode().dump();
695  std::cout << "In network:" << std::endl;
696  node.getOwner().dump();
697  throw ChimeraTK::logic_error("Variable name '@CONST@' found with wrong feeder type");
698  }
699 
700  if(node.hasOwner()) {
701  // The @CONST@ variable names do not distinguish types, hence we can have differently typed consumers in our
702  // network. To mitigate this issue, we remove the node from the existing network before creating a new one
703  // below
704  auto& network = node.getOwner();
705  network.removeNode(node);
706  // if network has no consumer left, abolish it
707  if(network.countConsumingNodes() == 0) {
708  // remove feeder if existing
709  if(network.hasFeedingNode()) {
710  auto feeder = network.getFeedingNode();
711  network.removeNode(feeder);
712  }
713  // remove network from application
714  networkList.remove(network);
715  }
716  }
717 
718  // Connect node with constant (in a new VariableNetwork)
719  callForType(node.getValueType(), [&node](auto t) {
720  using UserType = decltype(t);
721  auto value = userTypeToUserType<UserType>(node.getName().substr(7));
722  makeConstant(value) >> node;
723  });
724  }
725  }
726  }

733  for(auto& network : networkList) {

739  // only if there is exactly one node with return channel we need to guess its peer
740  continue;
741  }

743  // only a feeding control system variable can be made bidirectional
744  continue;
745  }
746  network.getFeedingNode().setDirection({VariableDirection::feeding, true});

752  size_t pollingComsumers{0};
753  for(auto& consumer : network.getConsumingNodes()) {
754  if(consumer.getMode() == UpdateMode::poll) ++pollingComsumers;
755  }
756  if(pollingComsumers == 1) {
757  network.getFeedingNode().removeExternalTrigger();
758  }
759  }

765 void Application::optimiseConnections() {
766  // list of iterators of networks to be removed from the networkList after the
767  // merge operation
768  std::list<VariableNetwork*> deleteNetworks;
769 
770  // search for networks with the same feeder
771  for(auto it1 = networkList.begin(); it1 != networkList.end(); ++it1) {
772  for(auto it2 = it1; it2 != networkList.end(); ++it2) {
773  if(it1 == it2) continue;
774 
775  auto feeder1 = it1->getFeedingNode();
776  auto feeder2 = it2->getFeedingNode();
777 
778  // this optimisation is only necessary for device-type nodes, since
779  // application and control-system nodes will automatically create merged
780  // networks when having the same feeder
783  if(feeder1.getType() != NodeType::Device || feeder2.getType() != NodeType::Device) continue;
784 
785  // check if referrring to same register
786  if(feeder1.getDeviceAlias() != feeder2.getDeviceAlias()) continue;
787  if(feeder1.getRegisterName() != feeder2.getRegisterName()) continue;
788 
789  // check if directions are the same
790  if(feeder1.getDirection() != feeder2.getDirection()) continue;
791 
792  // check if value types and number of elements are compatible
793  if(feeder1.getValueType() != feeder2.getValueType()) continue;
794  if(feeder1.getNumberOfElements() != feeder2.getNumberOfElements()) continue;
795 
796  // check if transfer mode is the same
797  if(feeder1.getMode() != feeder2.getMode()) continue;
798 
799  // check if triggers are compatible, if present
800  if(feeder1.hasExternalTrigger() != feeder2.hasExternalTrigger()) continue;
801  if(feeder1.hasExternalTrigger()) {
802  if(feeder1.getExternalTrigger() != feeder2.getExternalTrigger()) continue;
803  }
804 
805  // everything should be compatible at this point: merge the networks. We
806  // will merge the network of the outer loop into the network of the inner
807  // loop, since the network of the outer loop will not be found a second
808  // time in the inner loop.
809  for(auto& consumer : it1->getConsumingNodes()) {
810  consumer.clearOwner();
811  it2->addNode(consumer);
812  }
813 
814  // if trigger present, remove corresponding trigger receiver node from the
815  // trigger network
816  if(feeder1.hasExternalTrigger()) {
817  feeder1.getExternalTrigger().getOwner().removeNodeToTrigger(it1->getFeedingNode());
818  }
819 
820  // schedule the outer loop network for deletion and stop processing it
821  deleteNetworks.push_back(&(*it1));
822  break;
823  }
824  }
825 
826  // remove networks from the network list
827  for(auto net : deleteNetworks) {
828  networkList.remove(*net);
829  }
830 }

919 template<typename UserType>
920 void Application::typedMakeConnection(VariableNetwork& network) {
921  if(enableDebugMakeConnections) {
922  std::cout << std::endl << "Executing typedMakeConnections for network:" << std::endl;
923  network.dump("", std::cout);
924  std::cout << std::endl;
925  }
926  try { // catch exceptions to add information about the failed network
927  bool connectionMade = false; // to check the logic...
928 
929  size_t nNodes = network.countConsumingNodes() + 1;
930  auto feeder = network.getFeedingNode();
931  auto consumers = network.getConsumingNodes();
932  bool useExternalTrigger = network.getTriggerType() == VariableNetwork::TriggerType::external;
933  bool useFeederTrigger = network.getTriggerType() == VariableNetwork::TriggerType::feeder;
934  bool constantFeeder = feeder.getType() == NodeType::Constant;
935 
936  // 1st case: the feeder requires a fixed implementation
937  if(feeder.hasImplementation() && !constantFeeder) {
938  if(enableDebugMakeConnections) {
939  std::cout << " Creating fixed implementation for feeder '" << feeder.getName() << "'..." << std::endl;
940  }
941 
942  // Create feeding implementation.
943  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> feedingImpl;
944  if(feeder.getType() == NodeType::Device) {
945  feedingImpl = createDeviceVariable<UserType>(feeder);
946  }
947  else if(feeder.getType() == NodeType::ControlSystem) {
948  feedingImpl = createProcessVariable<UserType>(feeder);
949  }
950  else {
951  throw ChimeraTK::logic_error("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
952  }
953 
954  // if we just have two nodes, directly connect them
955  if(nNodes == 2 && !useExternalTrigger) {
956  if(enableDebugMakeConnections) {
957  std::cout << " Setting up direct connection without external trigger." << std::endl;
958  }
959  bool needsFanOut{false};
960  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> consumingImpl;
961 
962  auto consumer = consumers.front();
963  if(consumer.getType() == NodeType::Application) {
964  consumer.setAppAccessorImplementation(feedingImpl);
965 
966  connectionMade = true;
967  }
968  else if(consumer.getType() == NodeType::Device) {
969  consumingImpl = createDeviceVariable<UserType>(consumer);
970  // connect the Device with e.g. a ControlSystem node via a
971  // ThreadedFanOut
972  needsFanOut = true;
973  }
974  else if(consumer.getType() == NodeType::ControlSystem) {
975  consumingImpl = createProcessVariable<UserType>(consumer);
976  // connect the ControlSystem with e.g. a Device node via an
977  // ThreadedFanOut
978  needsFanOut = true;
979  }
980  else if(consumer.getType() == NodeType::TriggerReceiver) {
981  consumer.getNodeToTrigger().getOwner().setExternalTriggerImpl(feedingImpl);
982  }
983  else {
984  throw ChimeraTK::logic_error("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
985  }
986 
987  if(needsFanOut) {
988  assert(consumingImpl != nullptr);
989  auto consumerImplPair = ConsumerImplementationPairs<UserType>{{consumingImpl, consumer}};
990  auto fanOut = boost::make_shared<ThreadedFanOut<UserType>>(feedingImpl, network, consumerImplPair);
991  internalModuleList.push_back(fanOut);
992  network.setFanOut(fanOut);
993  }
994 
995  connectionMade = true;
996  }
997  else { /* !(nNodes == 2 && !useExternalTrigger) */
998  if(enableDebugMakeConnections) {
999  std::cout << " Setting up triggered connection." << std::endl;
1000  }
1001 
1002  // create the right FanOut type
1003  boost::shared_ptr<FanOut<UserType>> fanOut;
1004  boost::shared_ptr<ConsumingFanOut<UserType>> consumingFanOut;
1005 
1006  // Fanouts need to know the consumers on contruction, so we collect them first
1007  auto consumerImplementationPairs = setConsumerImplementations<UserType>(feeder, consumers);
1008 
1009  if(useExternalTrigger) {
1010  if(enableDebugMakeConnections) {
1011  std::cout << " Using external trigger." << std::endl;
1012  }
1013 
1014  // if external trigger is enabled, use externally triggered threaded
1015  // FanOut. Create one per external trigger impl.
1016  void* triggerImplId = network.getExternalTriggerImpl().get();
1017  auto triggerFanOut = triggerMap[triggerImplId];
1018  if(!triggerFanOut) {
1019  assert(deviceModuleMap.find(feeder.getDeviceAlias()) != deviceModuleMap.end());
1020 
1021  // create the trigger fan out and store it in the map and the internalModuleList
1022  triggerFanOut = boost::make_shared<TriggerFanOut>(
1023  network.getExternalTriggerImpl(), *deviceModuleMap[feeder.getDeviceAlias()], network);
1024  triggerMap[triggerImplId] = triggerFanOut;
1025  internalModuleList.push_back(triggerFanOut);
1026  }
1027  fanOut = triggerFanOut->addNetwork(feedingImpl, consumerImplementationPairs);
1028  network.setFanOut(fanOut);
1029  }
1030  else if(useFeederTrigger) {
1031  if(enableDebugMakeConnections) {
1032  std::cout << " Using trigger provided by the feeder." << std::endl;
1033  }
1034 
1035  // if the trigger is provided by the pushing feeder, use the treaded
1036  // version of the FanOut to distribute new values immediately to all
1037  // consumers. Depending on whether we have a return channel or not, pick
1038  // the right implementation of the FanOut
1039  boost::shared_ptr<ThreadedFanOut<UserType>> threadedFanOut;
1040  if(!feeder.getDirection().withReturn) {
1041  threadedFanOut =
1042  boost::make_shared<ThreadedFanOut<UserType>>(feedingImpl, network, consumerImplementationPairs);
1043  }
1044  else {
1045  threadedFanOut = boost::make_shared<ThreadedFanOutWithReturn<UserType>>(
1046  feedingImpl, network, consumerImplementationPairs);
1047  }
1048  internalModuleList.push_back(threadedFanOut);
1049  fanOut = threadedFanOut;
1050  network.setFanOut(fanOut);
1051  }
1052  else {
1053  if(enableDebugMakeConnections) {
1054  std::cout << " No trigger, using consuming fanout." << std::endl;
1055  }
1056  assert(network.hasApplicationConsumer()); // checkConnections should
1057  // catch this
1058  consumingFanOut = boost::make_shared<ConsumingFanOut<UserType>>(feedingImpl, consumerImplementationPairs);
1059  fanOut = consumingFanOut;
1060  network.setFanOut(fanOut);
1061 
1062  // TODO Is this correct? we already added all consumer as slaves in the fanout constructor.
1063  // Maybe assert that we only have a single poll-type node (is there a check in checkConnections?)
1064  for(auto& consumer : consumers) {
1065  if(consumer.getMode() == UpdateMode::poll) {
1066  consumer.setAppAccessorImplementation<UserType>(consumingFanOut);
1067  break;
1068  }
1069  }
1070  }
1071  connectionMade = true;
1072  }
1073  }
1074  // 2nd case: the feeder does not require a fixed implementation
1075  else if(!constantFeeder) { /* !feeder.hasImplementation() */
1076 
1077  if(enableDebugMakeConnections) {
1078  std::cout << " Feeder '" << feeder.getName() << "' does not require a fixed implementation." << std::endl;
1079  }
1080 
1081  // we should be left with an application feeder node
1082  if(feeder.getType() != NodeType::Application) {
1083  throw ChimeraTK::logic_error("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
1084  }
1085  assert(!useExternalTrigger);
1086  // if we just have two nodes, directly connect them
1087  if(nNodes == 2) {
1088  auto consumer = consumers.front();
1089  if(consumer.getType() == NodeType::Application) {
1090  auto impls = createApplicationVariable<UserType>(feeder, consumer);
1091  feeder.setAppAccessorImplementation<UserType>(impls.first);
1092  consumer.setAppAccessorImplementation<UserType>(impls.second);
1093  }
1094  else if(consumer.getType() == NodeType::ControlSystem) {
1095  auto impl = createProcessVariable<UserType>(consumer);
1096  feeder.setAppAccessorImplementation<UserType>(impl);
1097  }
1098  else if(consumer.getType() == NodeType::Device) {
1099  auto impl = createDeviceVariable<UserType>(consumer);
1100  feeder.setAppAccessorImplementation<UserType>(impl);
1101  }
1102  else if(consumer.getType() == NodeType::TriggerReceiver) {
1103  auto impls = createApplicationVariable<UserType>(feeder, consumer);
1104  feeder.setAppAccessorImplementation<UserType>(impls.first);
1105  consumer.getNodeToTrigger().getOwner().setExternalTriggerImpl(impls.second);
1106  }
1107  else if(consumer.getType() == NodeType::Constant) {
1108  auto impl = consumer.createConstAccessor<UserType>({});
1109  feeder.setAppAccessorImplementation<UserType>(impl);
1110  }
1111  else {
1112  throw ChimeraTK::logic_error("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
1113  }
1114  connectionMade = true;
1115  }
1116  else {
1117  auto consumerImplementationPairs = setConsumerImplementations<UserType>(feeder, consumers);
1118 
1119  // create FanOut and use it as the feeder implementation
1120  auto fanOut =
1121  boost::make_shared<FeedingFanOut<UserType>>(feeder.getName(), feeder.getUnit(), feeder.getDescription(),
1122  feeder.getNumberOfElements(), feeder.getDirection().withReturn, consumerImplementationPairs);
1123  feeder.setAppAccessorImplementation<UserType>(fanOut);
1124  network.setFanOut(fanOut);
1125 
1126  connectionMade = true;
1127  }
1128  }
1129  else { /* constantFeeder */
1130 
1131  if(enableDebugMakeConnections) {
1132  std::cout << " Using constant feeder '" << feeder.getName() << "'..." << std::endl;
1133  }
1134  assert(feeder.getType() == NodeType::Constant);
1135 
1136  for(auto& consumer : consumers) {
1137  AccessModeFlags flags{};
1138  if(consumer.getMode() == UpdateMode::push) {
1139  flags = {AccessMode::wait_for_new_data};
1140  }
1141 
1142  // each consumer gets its own implementation
1143  auto feedingImpl = feeder.createConstAccessor<UserType>(flags);
1144 
1145  if(consumer.getType() == NodeType::Application) {
1146  if(testableMode && consumer.getMode() == UpdateMode::push) {
1147  auto varId = getNextVariableId();
1148  auto pvarDec =
1149  boost::make_shared<TestableModeAccessorDecorator<UserType>>(feedingImpl, true, false, varId, varId);
1150  testableMode_names[varId] = "Constant";
1151  consumer.setAppAccessorImplementation<UserType>(pvarDec);
1152  }
1153  else {
1154  consumer.setAppAccessorImplementation<UserType>(feedingImpl);
1155  }
1156  }
1157  else if(consumer.getType() == NodeType::ControlSystem) {
1158  auto impl = createProcessVariable<UserType>(consumer);
1159  impl->accessChannel(0) = feedingImpl->accessChannel(0);
1160  impl->write();
1161  }
1162  else if(consumer.getType() == NodeType::Device) {
1163  // we register the required accessor as a recovery accessor. This is just a bare RegisterAccessor without any decorations directly from the backend.
1164  if(deviceMap.count(consumer.getDeviceAlias()) == 0) {
1165  deviceMap[consumer.getDeviceAlias()] =
1166  ChimeraTK::BackendFactory::getInstance().createBackend(consumer.getDeviceAlias());
1167  }
1168  auto impl = deviceMap[consumer.getDeviceAlias()]->getRegisterAccessor<UserType>(
1169  consumer.getRegisterName(), consumer.getNumberOfElements(), 0, {});
1170  impl->accessChannel(0) = feedingImpl->accessChannel(0);
1171 
1172  assert(deviceModuleMap.find(consumer.getDeviceAlias()) != deviceModuleMap.end());
1173  DeviceModule* devmod = deviceModuleMap[consumer.getDeviceAlias()];
1174 
1175  // The accessor implementation already has its data in the user buffer. We now just have to add a valid version number
1176  // and have a recovery accessors (RecoveryHelper to be excact) which we can register at the DeviceModule.
1177  // As this is a constant we don't need to change it later and don't have to store it somewere else.
1178  devmod->addRecoveryAccessor(boost::make_shared<RecoveryHelper>(impl, VersionNumber(), devmod->writeOrder()));
1179  }
1180  else if(consumer.getType() == NodeType::TriggerReceiver) {
1181  throw ChimeraTK::logic_error("Using constants as triggers is not supported!");
1182  }
1183  else {
1184  throw ChimeraTK::logic_error("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
1185  }
1186  }
1187  connectionMade = true;
1188  }
1189 
1190  if(!connectionMade) { // LCOV_EXCL_LINE (assert-like)
1191  throw ChimeraTK::logic_error( // LCOV_EXCL_LINE (assert-like)
1192  "The variable network cannot be handled. Implementation missing!"); // LCOV_EXCL_LINE (assert-like)
1193  } // LCOV_EXCL_LINE (assert-like)
1194  }
1195  catch(ChimeraTK::logic_error& e) {
1196  std::stringstream ss;
1197  ss << "ChimeraTK::logic_error thrown in Application::typedMakeConnection():" << std::endl
1198  << " " << e.what() << std::endl
1199  << "For network:" << std::endl;
1200  network.dump("", ss);
1201  throw ChimeraTK::logic_error(ss.str());
1202  }

1456 void Application::CircularDependencyDetector::registerDependencyWait(VariableNetworkNode& node) {
1457  assert(node.getType() == NodeType::Application);
1458  std::unique_lock<std::mutex> lock(_mutex);
1459 
1460  auto* dependent = dynamic_cast<Module*>(node.getOwningModule())->findApplicationModule();
1461 
1462  // register the waiting node in the map (used for the detectBlockedModules() thread). This is done also for Device
1463  // variables etc.
1464  _awaitedNodes[dependent] = node;
1465 
1466  // checking of direct circular dependencies is only done for Application-type feeders
1467  if(node.getOwner().getFeedingNode().getType() != NodeType::Application) return;
1468  auto* dependency = dynamic_cast<Module*>(node.getOwner().getFeedingNode().getOwningModule())->findApplicationModule();
1469 
1470  // If a module depends on itself, the detector would always detect a circular dependency, even if it is resolved
1471  // by writing initial values in prepare(). Hence we do not check anything in this case.
1472  if(dependent == dependency) return;
1473 
1474  // Register dependent-dependency relation in map
1475  _awaitedVariables[dependent] = node.getQualifiedName();
1476  _waitMap[dependent] = dependency;
1477 
1478  // check for circular dependencies
1479  auto* depdep = dependency;
1480  while(_waitMap.find(depdep) != _waitMap.end()) {
1481  auto* depdep_prev = depdep;
1482  depdep = _waitMap[depdep];
1483  if(depdep == dependent) {
1484  // The detected circular dependency might still resolve itself, because registerDependencyWait() is called even
1485  // if initial values are already present in the queues.
1486  for(size_t i = 0; i < 1000; ++i) {
1487  // give other thread a chance to read their initial value
1488  lock.unlock();
1489  usleep(10000);
1490  lock.lock();
1491  // if the module depending on an initial value for "us" is no longer waiting for us to send an initial value,
1492  // the circular dependency is resolved.
1493  if(_waitMap.find(depdep_prev) == _waitMap.end() || _waitMap[depdep] != dependent) return;
1494  }
1495 
1496  std::cerr << "*** Cirular dependency of ApplicationModules found while waiting for initial values!" << std::endl;
1497  std::cerr << std::endl;
1498 
1499  std::cerr << dependent->getQualifiedName() << " waits for " << node.getQualifiedName() << " from:" << std::endl;
1500  auto* depdep2 = dependency;
1501  while(_waitMap.find(depdep2) != _waitMap.end()) {
1502  auto waitsFor = _awaitedVariables[depdep2];
1503  std::cerr << depdep2->getQualifiedName();
1504  if(depdep2 == dependent) break;
1505  depdep2 = _waitMap[depdep2];
1506  std::cerr << " waits for " << waitsFor << " from:" << std::endl;
1507  }
1508  std::cerr << "." << std::endl;
1509 
1510  std::cerr << std::endl;
1511  std::cerr
1512  << "Please provide an initial value in the prepare() function of one of the involved ApplicationModules!"
1513  << std::endl;
1514 
1515  throw ChimeraTK::logic_error("Cirular dependency of ApplicationModules while waiting for initial values");
1516  }
1517  else {
1518  // Give other threads a chance to add to the wait map
1519  lock.unlock();
1520  usleep(10000);
1521  lock.lock();
1522  }
1523  }
1524 }

1557 void Application::CircularDependencyDetector::detectBlockedModules() {
1558  auto& app = Application::getInstance();
1559  while(true) {
1560  // wait some time to slow down this check
1561  boost::this_thread::sleep_for(boost::chrono::seconds(60));
1562 
1563  // check for modules which did not yet reach their mainLoop
1564  bool allModulesEnteredMainLoop = true;
1565  for(auto* module : app.getSubmoduleListRecursive()) {
1566  // Note: We are "abusing" this flag which was introduced for the testable mode. It actually just shows whether
1567  // the mainLoop() has benn called already (resp. will be called very soon). It is not depending on the
1568  // testableMode. FIXME: Rename this flag!
1569  if(module->hasReachedTestableMode()) {
1570  continue;
1571  }
1572 
1573  // Found a module which is still waiting for initial values
1574  allModulesEnteredMainLoop = false;
1575 
1576  // require lock against concurrent accesses on the maps
1577  std::lock_guard<std::mutex> lock(_mutex);
1578 
1579  // Check if module has registered a dependency wait. If not, situation will either resolve soon or module will
1580  // register a dependency wait soon. Both cases can be found in the next interation.
1581  if(_awaitedNodes.find(module) == _awaitedNodes.end()) continue;
1582 
1583  auto* appModule = dynamic_cast<ApplicationModule*>(module);
1584  if(!appModule) {
1585  // only ApplicationModule can have hasReachedTestableMode() == true, so it should not happen
1586  std::cout << "CircularDependencyDetector found non-application module: " << module->getQualifiedName()
1587  << std::endl;
1588  continue;
1589  }
1590 
1591  // Iteratively seach for reason the module blocks
1592  std::function<void(const VariableNetworkNode& node)> iterativeSearch = [&](const VariableNetworkNode& node) {
1593  const auto& feeder = node.getOwner().getFeedingNode();
1594  if(feeder.getType() == NodeType::Application) {
1595  // fed by other ApplicationModule: check if that one is waiting, too
1596  auto* feedingAppModule = dynamic_cast<Module*>(feeder.getOwningModule())->findApplicationModule();
1597  if(feedingAppModule->hasReachedTestableMode()) {
1598  // the feeding module has started its mainLoop(), but not sent us an initial value
1599  // FIXME: There is a race condition, if the situation has right now resolved and the feeding module just
1600  // not yet sent the initial value. Ideally we should wait another iteration before printing warnings!
1601  if(_modulesWeHaveWarnedAbout.find(feedingAppModule) == _modulesWeHaveWarnedAbout.end()) {
1602  _modulesWeHaveWarnedAbout.insert(feedingAppModule);
1603  std::cout << "Note: ApplicationModule " << appModule->getQualifiedName() << " is waiting for an "
1604  << "initial value, because " << feedingAppModule->getQualifiedName() << " has not yet sent one."
1605  << std::endl;
1606  }
1607  return;
1608  }
1609  if(_awaitedNodes.find(feedingAppModule) == _awaitedNodes.end()) {
1610  // The other module is not right now waiting. It will either enter the mainLoop soon, or start waiting soon.
1611  // Let's just wait another iteration.
1612  return;
1613  }
1614  // The other module is right now waiting: continue iterative search
1615  iterativeSearch(_awaitedNodes.at(feedingAppModule));
1616  }
1617  else if(feeder.getType() == NodeType::Device) {
1618  // fed by device
1619  const auto& deviceName = feeder.getDeviceAlias();
1620  if(_devicesWeHaveWarnedAbout.find(deviceName) == _devicesWeHaveWarnedAbout.end()) {
1621  _devicesWeHaveWarnedAbout.insert(deviceName);
1622  std::cout << "Note: Still waiting for device " << deviceName << " to come up";
1623  auto* dm = Application::getInstance().deviceModuleMap[deviceName];
1624  std::unique_lock dmLock(dm->errorMutex);
1625  if(dm->deviceHasError) {
1626  std::cout << " (" << std::string(dm->deviceError.message) << ")";
1627  }
1628  std::cout << "..." << std::endl;
1629  }
1630  }
1631  else {
1632  // fed by anything else
1633  if(_otherThingsWeHaveWarnedAbout.find(feeder.getType()) == _otherThingsWeHaveWarnedAbout.end()) {
1634  _otherThingsWeHaveWarnedAbout.insert(feeder.getType());
1635  std::cout << "At least one ApplicationModule (" << appModule->getQualifiedName() << " is waiting for an "
1636  << "initial value from NodeType " << int(feeder.getType()) << "." << std::endl;
1637  std::cout << "This is probably a BUG in the ChimeraTK framework." << std::endl;
1638  std::cout << "Network:" << std::endl;
1639  feeder.getOwner().dump();
1640  }
1641  }
1642  };
1643  iterativeSearch(_awaitedNodes.at(appModule));
1644  }
1645 
1646  // if all modules are in the mainLoop, stop this thread
1647  if(allModulesEnteredMainLoop) break;
1648  }
1649

1651 
1652  // free some memory
1653  _waitMap.clear();
1654  _awaitedVariables.clear();
1655  _awaitedNodes.clear();
1656  _modulesWeHaveWarnedAbout.clear();
1657  _devicesWeHaveWarnedAbout.clear();
1658  _otherThingsWeHaveWarnedAbout.clear();
1659 }
