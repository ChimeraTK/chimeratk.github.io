117 void Application::run() {
118  assert(not applicationName.empty());
119 
120  if(testableMode.isEnabled()) {
121  if(!testFacilityRunApplicationCalled) {
122  throw ChimeraTK::logic_error(
123  "Testable mode enabled but Application::run() called directly. Call TestFacility::runApplication() instead.");
124  }
125  }
126 
127  if(runCalled) {
128  throw ChimeraTK::logic_error("Application::run() has already been called before.");
129  }
130  runCalled = true;
131 
132  // realise the PV connections
133  cm.connect();
134 
135  // set all initial version numbers in the modules to the same value
136  for(auto& module : getSubmoduleListRecursive()) {
137  if(module->getModuleType() != ModuleType::ApplicationModule) continue;
138  module->setCurrentVersionNumber(getStartVersion());
139  }
140 
141  // prepare the modules
142  for(auto& module : getSubmoduleListRecursive()) {
143  module->prepare();
144  }
145 
146  // Switch life-cycle state to run
147  lifeCycleState = LifeCycleState::run;
148 
149  // start the necessary threads for the FanOuts etc.
150  for(auto& internalModule : internalModuleList) {
151  internalModule->activate();
152  }
153 
154  // start the threads for the modules
155  for(auto& module : getSubmoduleListRecursive()) {
156  module->run();
157  }
158 
159  // When in testable mode, wait for all modules to report that they have reached the testable mode.
160  // We have to start all module threads first because some modules might only send the initial
161  // values in their main loop, and following modules need them to enter testable mode.
162 
163  // just a small helper lambda to avoid code repetition
164  auto waitForTestableMode = [](EntityOwner* module) {
165  while(!module->hasReachedTestableMode()) {
166  Application::getInstance().getTestableMode().unlock("releaseForReachTestableMode");
167  usleep(100);
168  Application::getInstance().getTestableMode().lock("acquireForReachTestableMode");
169  }
170  };
171 
172  if(Application::getInstance().getTestableMode().isEnabled()) {
173  for(auto& internalModule : internalModuleList) {
174  waitForTestableMode(internalModule.get());
175  }
176 
177  for(auto& module : getSubmoduleListRecursive()) {
178  waitForTestableMode(module);
179  }
180  }
181 
182  // Launch circular dependency detector thread
183  circularDependencyDetector.startDetectBlockedModules();
184 }

188 void Application::shutdown() {
189  // switch life-cycle state
190  lifeCycleState = LifeCycleState::shutdown;
191 
192  // first allow to run the application threads again, if we are in testable
193  // mode
194  if(testableMode.isEnabled() && testableMode.testLock()) {
195  testableMode.unlock("shutdown");
196  }
197 
198  // deactivate the FanOuts first, since they have running threads inside
199  // accessing the modules etc. (note: the modules are members of the
200  // Application implementation and thus get destroyed after this destructor)
201  for(auto& internalModule : internalModuleList) {
202  internalModule->deactivate();
203  }
204 
205  // shutdown all DeviceManagers, otherwise application modules might hang if still waiting for initial values from
206  // devices
207  for(auto& pair : _deviceManagerMap) {
208  pair.second->terminate();
209  }
210 
211  // next deactivate the modules, as they have running threads inside as well
212  for(auto& module : getSubmoduleListRecursive()) {
213  module->terminate();
214  }
215 
216  circularDependencyDetector.terminate();
217 
218  ApplicationBase::shutdown();
219 }
