106 void Application::processUnconnectedNodes() {
107  for(auto& module : getSubmoduleListRecursive()) {
108  for(auto& accessor : module->getAccessorList()) {
109  if(!accessor.hasOwner()) {
110  if(enableUnconnectedVariablesWarning) {
111  std::cerr << "*** Warning: Variable '" << accessor.getQualifiedName()

113  "Reading will always result in 0, writing will be ignored."
114  << std::endl; // LCOV_EXCL_LINE

117  networkList.back().addNode(accessor);

119  bool makeFeeder = !(networkList.back().hasFeedingNode());

121  auto callable = CreateConstantForUnconnectedVar(accessor.getValueType(), makeFeeder, length);

123  assert(callable.done);
124  constantList.emplace_back(callable.theNode);
125  networkList.back().addNode(constantList.back());
126  }

155 void Application::run() {
156  assert(applicationName != "");
157 
158  if(testableMode) {
159  if(!testFacilityRunApplicationCalled) {
160  throw ChimeraTK::logic_error(
161  "Testable mode enabled but Application::run() called directly. Call TestFacility::runApplication() instead.");
162  }
163  }
164 
165  if(runCalled) {
166  throw ChimeraTK::logic_error("Application::run() has already been called before.");
167  }
168  runCalled = true;
169 
170  // set all initial version numbers in the modules to the same value
171  for(auto& module : getSubmoduleListRecursive()) {
172  if(module->getModuleType() != ModuleType::ApplicationModule) continue;
173  module->setCurrentVersionNumber(getStartVersion());
174  }
175 
176  // prepare the modules
177  for(auto& module : getSubmoduleListRecursive()) {
178  module->prepare();
179  }
180  for(auto& deviceModule : deviceModuleMap) {
181  deviceModule.second->prepare();
182  }
183 
184  // Switch life-cycle state to run
185  lifeCycleState = LifeCycleState::run;
186 
187  // start the necessary threads for the FanOuts etc.
188  for(auto& internalModule : internalModuleList) {
189  internalModule->activate();
190  }
191 
192  for(auto& deviceModule : deviceModuleMap) {
193  deviceModule.second->run();
194  }
195 
196  // start the threads for the modules
197  for(auto& module : getSubmoduleListRecursive()) {
198  module->run();
199  }
200 
201  // When in testable mode, wait for all modules to report that they have reched the testable mode.
202  // We have to start all module threads first because some modules might only send the initial
203  // values in their main loop, and following modules need them to enter testable mode.
204 
205  // just a small helper lambda to avoid code repetition
206  auto waitForTestableMode = [](EntityOwner* module) {
207  while(!module->hasReachedTestableMode()) {
208  Application::getInstance().testableModeUnlock("releaseForReachTestableMode");
209  usleep(100);
210  Application::getInstance().testableModeLock("acquireForReachTestableMode");
211  }
212  };
213 
214  if(Application::getInstance().isTestableModeEnabled()) {
215  for(auto& internalModule : internalModuleList) {
216  waitForTestableMode(internalModule.get());
217  }
218  for(auto& deviceModule : deviceModuleMap) {
219  waitForTestableMode(deviceModule.second);
220  }
221  for(auto& module : getSubmoduleListRecursive()) {
222  waitForTestableMode(module);
223  }
224  }
225 }

284 VariableNetwork& Application::connect(VariableNetworkNode a, VariableNetworkNode b) {
285  // if one of the nodes has the value type AnyType, set it to the type of the
286  // other if both are AnyType, nothing changes.
287  if(a.getValueType() == typeid(AnyType)) {
288  a.setValueType(b.getValueType());
289  }
290  else if(b.getValueType() == typeid(AnyType)) {
291  b.setValueType(a.getValueType());
292  }
293 
294  // if one of the nodes has not yet a defined number of elements, set it to the
295  // number of elements of the other. if both are undefined, nothing changes.
296  if(a.getNumberOfElements() == 0) {
297  a.setNumberOfElements(b.getNumberOfElements());
298  }
299  else if(b.getNumberOfElements() == 0) {
300  b.setNumberOfElements(a.getNumberOfElements());
301  }
302  if(a.getNumberOfElements() != b.getNumberOfElements()) {
303  std::stringstream what;
304  what << "*** ERROR: Cannot connect array variables with difference number "
305  "of elements!"
306  << std::endl;
307  what << "Node A:" << std::endl;
308  a.dump(what);
309  what << "Node B:" << std::endl;
310  b.dump(what);
311  throw ChimeraTK::logic_error(what.str());
312  }
313 
314  // if both nodes already have an owner, we are either already done (same
315  // owners) or we need to try to merge the networks
316  if(a.hasOwner() && b.hasOwner()) {
317  if(&(a.getOwner()) != &(b.getOwner())) {
318  auto& networkToMerge = b.getOwner();
319  bool success = a.getOwner().merge(networkToMerge);
320  if(!success) {
321  std::stringstream what;
322  what << "*** ERROR: Trying to connect two nodes which are already part "
323  "of different networks, and merging these"
324  " networks is not possible (cannot have two non-control-system "
325  "or two control-system feeders)!"
326  << std::endl;
327  what << "Node A:" << std::endl;
328  a.dump(what);
329  what << "Node B:" << std::endl;
330  b.dump(what);
331  what << "Owner of node A:" << std::endl;
332  a.getOwner().dump("", what);
333  what << "Owner of node B:" << std::endl;
334  b.getOwner().dump("", what);
335  throw ChimeraTK::logic_error(what.str());
336  }
337  for(auto n = networkList.begin(); n != networkList.end(); ++n) {
338  if(&*n == &networkToMerge) {
339  networkList.erase(n);
340  break;
341  }
342  }
343  }
344  }
345  // add b to the existing network of a
346  else if(a.hasOwner()) {
347  a.getOwner().addNode(b);
348  }
349 
350  // add a to the existing network of b
351  else if(b.hasOwner()) {
352  b.getOwner().addNode(a);
353  }
354  // create new network
355  else {
356  networkList.emplace_back();
357  networkList.back().addNode(a);
358  networkList.back().addNode(b);
359  }
360  return a.getOwner();
361 }

367  VariableNetworkNode const& node) {

369  auto registerName = node.getRegisterName();
370  auto direction = node.getDirection();
371  auto mode = node.getMode();
372  auto nElements = node.getNumberOfElements();
373

376  deviceMap[deviceAlias] = ChimeraTK::BackendFactory::getInstance().createBackend(deviceAlias);
377  }

380  // Feeding to the network means reading from a device to feed it into the network.
381  ChimeraTK::AccessModeFlags flags{};

398 template<typename UserType>
399 boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> Application::createProcessVariable(
400  VariableNetworkNode const& node) {
401  // determine the SynchronizationDirection
402  SynchronizationDirection dir;
403  if(node.getDirection().withReturn) {
404  dir = SynchronizationDirection::bidirectional;
405  }
406  else if(node.getDirection().dir == VariableDirection::feeding) {
407  dir = SynchronizationDirection::controlSystemToDevice;
408  }
409  else {
410  dir = SynchronizationDirection::deviceToControlSystem;
411  }
412  AccessModeFlags flags = {};
413  if(node.getDirection().dir == VariableDirection::consuming) { // Application-to-controlsystem must be
414  // push-type
415  flags = {AccessMode::wait_for_new_data};
416  }
417  else {
418  for(auto& consumer : node.getOwner().getConsumingNodes()) {
419  if(consumer.getMode() == UpdateMode::push) flags = {AccessMode::wait_for_new_data};
420  }
421  }
422 
423  // create the ProcessArray for the proper UserType
424  auto pvar = _processVariableManager->createProcessArray<UserType>(dir, node.getPublicName(),
425  node.getNumberOfElements(), node.getOwner().getUnit(), node.getOwner().getDescription(), {}, 3, flags);
426  assert(pvar->getName() != "");
427 
428  // create variable ID
429  auto varId = getNextVariableId();
430  pvIdMap[pvar->getUniqueId()] = varId;
431 
432  // Decorate the process variable if testable mode is enabled and this is the
433  // receiving end of the variable (feeding to the network). Also don't decorate, if the mode is polling.
434  // Instead flag the variable to be polling, so the TestFacility is aware of
435  // this.
436  if(testableMode && node.getDirection().dir == VariableDirection::feeding) {
437  // The transfer mode of this process variable is considered to be polling,
438  // if only one consumer exists and this consumer is polling. Reason:
439  // mulitple consumers will result in the use of a FanOut, so the
440  // communication up to the FanOut will be push-type, even if all consumers
441  // are poll-type.
443  auto mode = UpdateMode::push;
444  if(node.getOwner().countConsumingNodes() == 1) {
445  if(node.getOwner().getConsumingNodes().front().getMode() == UpdateMode::poll) mode = UpdateMode::poll;
446  }
447 
448  if(mode != UpdateMode::poll) {
449  auto pvarDec = boost::make_shared<TestableModeAccessorDecorator<UserType>>(pvar, true, false, varId, varId);
450  testableMode_names[varId] = "ControlSystem:" + node.getPublicName();
451  return pvarDec;
452  }
453  else {
454  testableMode_isPollMode[varId] = true;
455  }
456  }
457 
458  // return the process variable
459  return pvar;
460 }

575 void Application::finaliseNetworks() {
576  // check for control system variables which should be made bidirectional
577  for(auto& network : networkList) {
578  size_t nBidir = network.getFeedingNode().getDirection().withReturn ? 1 : 0;
579  for(auto& consumer : network.getConsumingNodes()) {
580  if(consumer.getDirection().withReturn) ++nBidir;
581  }
582  if(nBidir != 1)
583  continue; // only if there is exactly one node with return channel we need
584  // to guess its peer
585  if(network.getFeedingNode().getType() != NodeType::ControlSystem) {
586  // only a feeding control system variable can be made bidirectional
587  continue;
588  }
589  network.getFeedingNode().setDirection({VariableDirection::feeding, true});
590  }
591 
592  // check for networks which have an external trigger but should be triggered by pollling consumer
593  for(auto& network : networkList) {
594  if(network.getTriggerType() == VariableNetwork::TriggerType::external) {
595  size_t pollingComsumers{0};
596  for(auto& consumer : network.getConsumingNodes()) {
597  if(consumer.getMode() == UpdateMode::poll) ++pollingComsumers;
598  }
599  if(pollingComsumers == 1) {
600  network.getFeedingNode().removeExternalTrigger();
601  }
602  }
603  }
604 }

608 void Application::optimiseConnections() {
609  // list of iterators of networks to be removed from the networkList after the
610  // merge operation
611  std::list<VariableNetwork*> deleteNetworks;
612 
613  // search for networks with the same feeder
614  for(auto it1 = networkList.begin(); it1 != networkList.end(); ++it1) {
615  for(auto it2 = it1; it2 != networkList.end(); ++it2) {
616  if(it1 == it2) continue;
617 
618  auto feeder1 = it1->getFeedingNode();
619  auto feeder2 = it2->getFeedingNode();
620 
621  // this optimisation is only necessary for device-type nodes, since
622  // application and control-system nodes will automatically create merged
623  // networks when having the same feeder
626  if(feeder1.getType() != NodeType::Device || feeder2.getType() != NodeType::Device) continue;
627 
628  // check if referrring to same register
629  if(feeder1.getDeviceAlias() != feeder2.getDeviceAlias()) continue;
630  if(feeder1.getRegisterName() != feeder2.getRegisterName()) continue;
631 
632  // check if directions are the same
633  if(feeder1.getDirection() != feeder2.getDirection()) continue;
634 
635  // check if value types and number of elements are compatible
636  if(feeder1.getValueType() != feeder2.getValueType()) continue;
637  if(feeder1.getNumberOfElements() != feeder2.getNumberOfElements()) continue;
638 
639  // check if transfer mode is the same
640  if(feeder1.getMode() != feeder2.getMode()) continue;
641 
642  // check if triggers are compatible, if present
643  if(feeder1.hasExternalTrigger() != feeder2.hasExternalTrigger()) continue;
644  if(feeder1.hasExternalTrigger()) {
645  if(feeder1.getExternalTrigger() != feeder2.getExternalTrigger()) continue;
646  }
647 
648  // everything should be compatible at this point: merge the networks. We
649  // will merge the network of the outer loop into the network of the inner
650  // loop, since the network of the outer loop will not be found a second
651  // time in the inner loop.
652  for(auto consumer : it1->getConsumingNodes()) {
653  consumer.clearOwner();
654  it2->addNode(consumer);
655  }
656 
657  // if trigger present, remove corresponding trigger receiver node from the
658  // trigger network
659  if(feeder1.hasExternalTrigger()) {

729 template<typename UserType>
730 void Application::typedMakeConnection(VariableNetwork& network) {
731  if(enableDebugMakeConnections) {
732  std::cout << std::endl << "Executing typedMakeConnections for network:" << std::endl;
733  network.dump("", std::cout);
734  std::cout << std::endl;
735  }
736  try { // catch exceptions to add information about the failed network
737  bool connectionMade = false; // to check the logic...
738 
739  size_t nNodes = network.countConsumingNodes() + 1;
740  auto feeder = network.getFeedingNode();
741  auto consumers = network.getConsumingNodes();
742  bool useExternalTrigger = network.getTriggerType() == VariableNetwork::TriggerType::external;
743  bool useFeederTrigger = network.getTriggerType() == VariableNetwork::TriggerType::feeder;
744  bool constantFeeder = feeder.getType() == NodeType::Constant;
745 
746  // 1st case: the feeder requires a fixed implementation
747  if(feeder.hasImplementation() && !constantFeeder) {
748  if(enableDebugMakeConnections) {
749  std::cout << " Creating fixed implementation for feeder '" << feeder.getName() << "'..." << std::endl;
750  }
751 
752  // Create feeding implementation.
753  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> feedingImpl;
754  if(feeder.getType() == NodeType::Device) {
755  feedingImpl = createDeviceVariable<UserType>(feeder);
756  }
757  else if(feeder.getType() == NodeType::ControlSystem) {
758  feedingImpl = createProcessVariable<UserType>(feeder);
759  }
760  else {
761  throw ChimeraTK::logic_error("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
762  }
763 
764  // if we just have two nodes, directly connect them
765  if(nNodes == 2 && !useExternalTrigger) {
766  if(enableDebugMakeConnections) {
767  std::cout << " Setting up direct connection without external trigger." << std::endl;
768  }
769  bool needsFanOut{false};
770  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> consumingImpl;
771 
772  auto consumer = consumers.front();
773  if(consumer.getType() == NodeType::Application) {
774  consumer.setAppAccessorImplementation(feedingImpl);
775 
776  connectionMade = true;
777  }
778  else if(consumer.getType() == NodeType::Device) {
779  consumingImpl = createDeviceVariable<UserType>(consumer);
780  // connect the Device with e.g. a ControlSystem node via a
781  // ThreadedFanOut
782  needsFanOut = true;
783  }
784  else if(consumer.getType() == NodeType::ControlSystem) {
785  consumingImpl = createProcessVariable<UserType>(consumer);
786  // connect the ControlSystem with e.g. a Device node via an
787  // ThreadedFanOut
788  needsFanOut = true;
789  }
790  else if(consumer.getType() == NodeType::TriggerReceiver) {
791  consumer.getNodeToTrigger().getOwner().setExternalTriggerImpl(feedingImpl);
792  }
793  else {
794  throw ChimeraTK::logic_error("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
795  }
796 
797  if(needsFanOut) {
798  assert(consumingImpl != nullptr);
799  auto consumerImplPair = ConsumerImplementationPairs<UserType>{{consumingImpl, consumer}};
800  auto fanOut = boost::make_shared<ThreadedFanOut<UserType>>(feedingImpl, network, consumerImplPair);
801  internalModuleList.push_back(fanOut);
802  }
803 
804  connectionMade = true;
805  }
806  else { /* !(nNodes == 2 && !useExternalTrigger) */
807  if(enableDebugMakeConnections) {
808  std::cout << " Setting up triggered connection." << std::endl;
809  }
810 
811  // create the right FanOut type
812  boost::shared_ptr<FanOut<UserType>> fanOut;
813  boost::shared_ptr<ConsumingFanOut<UserType>> consumingFanOut;
814 
815  // Fanouts need to know the consumers on contruction, so we collect them first
816  auto consumerImplementationPairs = setConsumerImplementations<UserType>(feeder, consumers);
817 
818  if(useExternalTrigger) {
819  if(enableDebugMakeConnections) {
820  std::cout << " Using external trigger." << std::endl;
821  }
822 
823  // if external trigger is enabled, use externally triggered threaded
824  // FanOut. Create one per external trigger impl.
825  void* triggerImplId = network.getExternalTriggerImpl().get();
826  auto triggerFanOut = triggerMap[triggerImplId];
827  if(!triggerFanOut) {
828  assert(deviceModuleMap.find(feeder.getDeviceAlias()) != deviceModuleMap.end());
829 
830  // create the trigger fan out and store it in the map and the internalModuleList
831  triggerFanOut = boost::make_shared<TriggerFanOut>(
832  network.getExternalTriggerImpl(), *deviceModuleMap[feeder.getDeviceAlias()], network);
833  triggerMap[triggerImplId] = triggerFanOut;
834  internalModuleList.push_back(triggerFanOut);
835  }
836  fanOut = triggerFanOut->addNetwork(feedingImpl, consumerImplementationPairs);
837  }
838  else if(useFeederTrigger) {
839  if(enableDebugMakeConnections) {
840  std::cout << " Using trigger provided by the feeder." << std::endl;
841  }
842 
843  // if the trigger is provided by the pushing feeder, use the treaded
844  // version of the FanOut to distribute new values immediately to all
845  // consumers. Depending on whether we have a return channel or not, pick
846  // the right implementation of the FanOut
847  boost::shared_ptr<ThreadedFanOut<UserType>> threadedFanOut;
848  if(!feeder.getDirection().withReturn) {
849  threadedFanOut =
850  boost::make_shared<ThreadedFanOut<UserType>>(feedingImpl, network, consumerImplementationPairs);
851  }
852  else {
853  threadedFanOut = boost::make_shared<ThreadedFanOutWithReturn<UserType>>(
854  feedingImpl, network, consumerImplementationPairs);
855  }
856  internalModuleList.push_back(threadedFanOut);
857  fanOut = threadedFanOut;
858  }
859  else {
860  if(enableDebugMakeConnections) {
861  std::cout << " No trigger, using consuming fanout." << std::endl;
862  }
863  assert(network.hasApplicationConsumer()); // checkConnections should
864  // catch this
865  consumingFanOut = boost::make_shared<ConsumingFanOut<UserType>>(feedingImpl, consumerImplementationPairs);
866  fanOut = consumingFanOut;
867 
868  // TODO Is this correct? we already added all consumer as slaves in the fanout constructor.
869  // Maybe assert that we only have a single poll-type node (is there a check in checkConnections?)
870  for(auto consumer : consumers) {
871  if(consumer.getMode() == UpdateMode::poll) {
872  consumer.setAppAccessorImplementation<UserType>(consumingFanOut);
873  break;
874  }
875  }
876  }
877  connectionMade = true;
878  }
879  }
880  // 2nd case: the feeder does not require a fixed implementation
881  else if(!constantFeeder) { /* !feeder.hasImplementation() */
882 
883  if(enableDebugMakeConnections) {
884  std::cout << " Feeder '" << feeder.getName() << "' does not require a fixed implementation." << std::endl;
885  }
886 
887  // we should be left with an application feeder node
888  if(feeder.getType() != NodeType::Application) {
889  throw ChimeraTK::logic_error("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
890  }
891  assert(!useExternalTrigger);
892  // if we just have two nodes, directly connect them
893  if(nNodes == 2) {
894  auto consumer = consumers.front();
895  if(consumer.getType() == NodeType::Application) {
896  auto impls = createApplicationVariable<UserType>(feeder, consumer);
897  feeder.setAppAccessorImplementation<UserType>(impls.first);
898  consumer.setAppAccessorImplementation<UserType>(impls.second);
899  }
900  else if(consumer.getType() == NodeType::ControlSystem) {
901  auto impl = createProcessVariable<UserType>(consumer);
902  feeder.setAppAccessorImplementation<UserType>(impl);
903  }
904  else if(consumer.getType() == NodeType::Device) {
905  auto impl = createDeviceVariable<UserType>(consumer);
906  feeder.setAppAccessorImplementation<UserType>(impl);
907  }
908  else if(consumer.getType() == NodeType::TriggerReceiver) {
909  auto impls = createApplicationVariable<UserType>(feeder, consumer);
910  feeder.setAppAccessorImplementation<UserType>(impls.first);
911  consumer.getNodeToTrigger().getOwner().setExternalTriggerImpl(impls.second);
912  }
913  else if(consumer.getType() == NodeType::Constant) {
914  auto impl = consumer.createConstAccessor<UserType>({});
915  feeder.setAppAccessorImplementation<UserType>(impl);
916  }
917  else {
918  throw ChimeraTK::logic_error("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
919  }
920  connectionMade = true;
921  }
922  else {
923  auto consumerImplementationPairs = setConsumerImplementations<UserType>(feeder, consumers);
924 
925  // create FanOut and use it as the feeder implementation
926  auto fanOut =
927  boost::make_shared<FeedingFanOut<UserType>>(feeder.getName(), feeder.getUnit(), feeder.getDescription(),
928  feeder.getNumberOfElements(), feeder.getDirection().withReturn, consumerImplementationPairs);
929  feeder.setAppAccessorImplementation<UserType>(fanOut);
930 
931  connectionMade = true;
932  }
933  }
934  else { /* constantFeeder */
935 
936  if(enableDebugMakeConnections) {
937  std::cout << " Using constant feeder '" << feeder.getName() << "'..." << std::endl;
938  }
939  assert(feeder.getType() == NodeType::Constant);
940 
941  for(auto& consumer : consumers) {
942  AccessModeFlags flags{};
943  if(consumer.getMode() == UpdateMode::push) {
944  flags = {AccessMode::wait_for_new_data};
945  }
946 
947  // each consumer gets its own implementation
948  auto feedingImpl = feeder.createConstAccessor<UserType>(flags);
949 
950  if(consumer.getType() == NodeType::Application) {
951  if(testableMode) {
952  auto varId = getNextVariableId();
953  auto pvarDec =
954  boost::make_shared<TestableModeAccessorDecorator<UserType>>(feedingImpl, true, false, varId, varId);
955  testableMode_names[varId] = "Constant";
956  consumer.setAppAccessorImplementation<UserType>(pvarDec);
957  }
958  else {
959  consumer.setAppAccessorImplementation<UserType>(feedingImpl);
960  }
961  }
962  else if(consumer.getType() == NodeType::ControlSystem) {
963  auto impl = createProcessVariable<UserType>(consumer);
964  impl->accessChannel(0) = feedingImpl->accessChannel(0);
965  impl->write();
966  }
967  else if(consumer.getType() == NodeType::Device) {
968  // we register the required accessor as a recovery accessor. This is just a bare RegisterAccessor without any decorations directly from the backend.
969  if(deviceMap.count(consumer.getDeviceAlias()) == 0) {
970  deviceMap[consumer.getDeviceAlias()] =
971  ChimeraTK::BackendFactory::getInstance().createBackend(consumer.getDeviceAlias());
972  }
973  auto impl = deviceMap[consumer.getDeviceAlias()]->getRegisterAccessor<UserType>(
974  consumer.getRegisterName(), consumer.getNumberOfElements(), 0, {});
975  impl->accessChannel(0) = feedingImpl->accessChannel(0);
976 
977  assert(deviceModuleMap.find(consumer.getDeviceAlias()) != deviceModuleMap.end());
978  DeviceModule* devmod = deviceModuleMap[consumer.getDeviceAlias()];
979 
980  // The accessor implementation already has its data in the user buffer. We now just have to add a valid version number
981  // and have a recovery accessors (RecoveryHelper to be excact) which we can register at the DeviceModule.
982  // As this is a constant we don't need to change it later and don't have to store it somewere else.
983  devmod->addRecoveryAccessor(boost::make_shared<RecoveryHelper>(impl, VersionNumber(), devmod->writeOrder()));
984  }
985  else if(consumer.getType() == NodeType::TriggerReceiver) {
986  throw ChimeraTK::logic_error("Using constants as triggers is not supported!");
987  }
988  else {
989  throw ChimeraTK::logic_error("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
990  }
991  }
992  connectionMade = true;
993  }
994 
995  if(!connectionMade) { // LCOV_EXCL_LINE (assert-like)
996  throw ChimeraTK::logic_error( // LCOV_EXCL_LINE (assert-like)
997  "The variable network cannot be handled. Implementation missing!"); // LCOV_EXCL_LINE (assert-like)
998  } // LCOV_EXCL_LINE (assert-like)
999  }
1000  catch(ChimeraTK::logic_error& e) {
1001  std::stringstream ss;
1002  ss << "ChimeraTK::logic_error thrown in Application::typedMakeConnection():" << std::endl
1003  << " " << e.what() << std::endl
1004  << "For network:" << std::endl;
1005  network.dump("", ss);
1006  throw ChimeraTK::logic_error(ss.str());
1007  }
1008 }

1013 std::list<std::pair<boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>>, VariableNetworkNode>> Application::
1014  setConsumerImplementations(VariableNetworkNode const& feeder, std::list<VariableNetworkNode> consumers) {

1016  consumerImplPairs;
1017

1020 
1021  for(auto& consumer : consumers) {

1023  std::pair<boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>>, VariableNetworkNode> pair{
1024  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>>(), consumer};

1027  auto impls = createApplicationVariable<UserType>(consumer);
1028  consumer.setAppAccessorImplementation<UserType>(impls.second);

1030  }

1033  pair = std::make_pair(impl, consumer);
1034  }
1035  else if(consumer.getType() == NodeType::Device) {
1036  auto impl = createDeviceVariable<UserType>(consumer);
1037  pair = std::make_pair(impl, consumer);
1038  }
1039  else if(consumer.getType() == NodeType::TriggerReceiver) {
1040  // In case we have one or more trigger receivers among our consumers, we produce one
1041  // consuming application variable for each device. Later this will create a TriggerFanOut for
1042  // each trigger consumer, i.e. one per device so one blocking device does not affect the others.
1043 
1044  std::string deviceAlias = consumer.getNodeToTrigger().getOwner().getFeedingNode().getDeviceAlias();
1045  auto triggerFanOut = triggerFanOuts[deviceAlias];
1046  if(triggerFanOut == nullptr) {
1047  // create a new process variable pair and set the sender/feeder to the fan out
1048  auto triggerConnection = createApplicationVariable<UserType>(feeder);
1049  triggerFanOut = triggerConnection.second;
1050  triggerFanOuts[deviceAlias] = triggerFanOut;
1051 
1052  assert(triggerConnection.first != nullptr);
1053  pair = std::make_pair(triggerConnection.first, consumer);
1054  }
1055  else {
1056  // We already have added a pair for this device
1057  addToConsumerImplPairs = false;
1058  }
1059  consumer.getNodeToTrigger().getOwner().setExternalTriggerImpl(triggerFanOut);
1060  }
1061  else {
1062  throw ChimeraTK::logic_error("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
1063  }
1064 
1065  if(addToConsumerImplPairs) {
1066  consumerImplPairs.push_back(pair);
1067  }
1068  }
1069 
1070  return consumerImplPairs;
1071 }

1088 void Application::stepApplication(bool waitForDeviceInitialisation) {
1089  // testableMode_counter must be non-zero, otherwise there is no input for the application to process. It is also
1090  // sufficient if testableMode_deviceInitialisationCounter is non-zero, if waitForDeviceInitialisation == true. In that

1092  if(testableMode_counter == 0 && (!waitForDeviceInitialisation || testableMode_deviceInitialisationCounter == 0)) {

1102  << std::endl; // LCOV_EXCL_LINE (only cout)
1103  oldCounter = testableMode_counter; // LCOV_EXCL_LINE (only cout)
1104  }

1107  testableModeLock("stepApplication");

1121  << " tries to obtain lock for " << name << std::endl; // LCOV_EXCL_LINE (only cout)
1122  } // LCOV_EXCL_LINE (only cout)
1123 
1124  // if last lock was obtained repeatedly by the same thread, sleep a short time
1125  // before obtaining the lock to give the other threads a chance to get the
1126  // lock first
1127  if(getInstance().testableMode_repeatingMutexOwner > 0) usleep(10000);
1128 
1129  // obtain the lock
1130  getTestableModeLockObject().lock();
1131 
1132  // check if the last owner of the mutex was this thread, which may be a hint
1133  // that no other thread is waiting for the lock
1134  if(getInstance().testableMode_lastMutexOwner == std::this_thread::get_id()) {

1140  << ". Further messages will be suppressed." << std::endl; // LCOV_EXCL_LINE (only cout)
1141  } // LCOV_EXCL_LINE (only cout)
1142

1144  getInstance().testableMode_repeatingMutexOwner++;
1145 
1146  // detect stall: if the same thread got the mutex with no other thread
1147  // obtaining it in between for one second, we assume no other thread is able
1148  // to process data at this time. The test should fail in this case
1149  if(getInstance().testableMode_repeatingMutexOwner > 100) {

1151  // currently containing unread data.
1152  std::cout << "*** Tests are stalled due to data which has been sent but "

1154  << std::endl;

1157  << std::endl;

1160  std::cout << " - " << Application::getInstance().testableMode_names[pair.first] << " ["
1161  << getInstance().testableMode_processVars[pair.first]->getId() << "]";

1164  if(getInstance().testableMode_processVars[pair.first]->readNonBlocking()) {
1165  std::cout << " (unread data in queue)";
1166  }
1167  else {
1168  std::cout << " (data loss)";
1169  }
1170  }
1171  catch(std::logic_error&) {
1172  // if we receive a logic_error in readNonBlocking() it just means
1173  // another thread is waiting on a TransferFuture of this variable,
1174  // and we actually were not allowed to read...
1175  std::cout << " (data loss)";
1176  }
1177  std::cout << std::endl;
1178  }
1179  }
1180  std::cout << "(end of list)" << std::endl;
1181  // throw a specialised exception to make sure whoever catches it really
1182  // knows what he does...
1183  throw TestsStalled();
1184  // getInstance().testableMode_counter = 0;
1185  // for(auto &pair : Application::getInstance().testableMode_perVarCounter)
1186  // pair.second = 0;
1187  }
1188  }
1189  else {
1190  // last owner of the mutex was different: reset the counter and store the
1191  // thread id
1192  getInstance().testableMode_repeatingMutexOwner = 0;
1193  getInstance().testableMode_lastMutexOwner = std::this_thread::get_id();
1194

1196  if(getInstance().enableDebugTestableMode) { // LCOV_EXCL_LINE (only cout)

1198  << " obtained lock successfully for " << name << std::endl; // LCOV_EXCL_LINE (only cout)

1217 std::string& Application::threadName() {
1218  // Note: due to a presumed bug in gcc (still present in gcc 7), the
