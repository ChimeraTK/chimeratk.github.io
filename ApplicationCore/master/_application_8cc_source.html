116 void Application::run() {
117  assert(not applicationName.empty());
118 
119  if(_testableMode.isEnabled()) {
120  if(!_testFacilityRunApplicationCalled) {
121  throw ChimeraTK::logic_error(
122  "Testable mode enabled but Application::run() called directly. Call TestFacility::runApplication() instead.");
123  }
124  }
125 
126  if(_runCalled) {
127  throw ChimeraTK::logic_error("Application::run() has already been called before.");
128  }
129  _runCalled = true;
130 
131  // realise the PV connections
132  _cm.connect();
133 
134  // set all initial version numbers in the modules to the same value
135  for(auto& module : getSubmoduleListRecursive()) {
136  if(module->getModuleType() != ModuleType::ApplicationModule) {
137  continue;
138  }
139  module->setCurrentVersionNumber(getStartVersion());
140  }
141 
142  // prepare the modules
143  for(auto& module : getSubmoduleListRecursive()) {
144  module->prepare();
145  }
146 
147  // Switch life-cycle state to run
148  _lifeCycleState = LifeCycleState::run;
149 
150  // start the necessary threads for the FanOuts etc.
151  for(auto& internalModule : _internalModuleList) {
152  internalModule->activate();
153  }
154 
155  // start the threads for the modules
156  for(auto& module : getSubmoduleListRecursive()) {
157  module->run();
158  }
159 
160  // When in testable mode, wait for all modules to report that they have reached the testable mode.
161  // We have to start all module threads first because some modules might only send the initial
162  // values in their main loop, and following modules need them to enter testable mode.
163 
164  // just a small helper lambda to avoid code repetition
165  auto waitForTestableMode = [](EntityOwner* module) {
166  while(!module->hasReachedTestableMode()) {
167  Application::getInstance().getTestableMode().unlock("releaseForReachTestableMode");
168  usleep(100);
169  Application::getInstance().getTestableMode().lock("acquireForReachTestableMode");
170  }
171  };
172 
173  if(Application::getInstance().getTestableMode().isEnabled()) {
174  for(auto& internalModule : _internalModuleList) {
175  waitForTestableMode(internalModule.get());
176  }
177 
178  for(auto& module : getSubmoduleListRecursive()) {
179  waitForTestableMode(module);
180  }
181  }
182 
183  // Launch circular dependency detector thread
184  _circularDependencyDetector.startDetectBlockedModules();
185 }

189 void Application::shutdown() {
190  // switch life-cycle state
191  _lifeCycleState = LifeCycleState::shutdown;
192 
193  // first allow to run the application threads again, if we are in testable
194  // mode
195  if(_testableMode.isEnabled() && _testableMode.testLock()) {
196  _testableMode.unlock("shutdown");
197  }
198 
199  // deactivate the FanOuts first, since they have running threads inside
200  // accessing the modules etc. (note: the modules are members of the
201  // Application implementation and thus get destroyed after this destructor)
202  for(auto& internalModule : _internalModuleList) {
203  internalModule->deactivate();
204  }
205 
206  // shutdown all DeviceManagers, otherwise application modules might hang if still waiting for initial values from
207  // devices
208  for(auto& pair : _deviceManagerMap) {
209  pair.second->terminate();
210  }
211 
212  // next deactivate the modules, as they have running threads inside as well
213  for(auto& module : getSubmoduleListRecursive()) {
214  module->terminate();
215  }
216 
217  _circularDependencyDetector.terminate();
218 
219  ApplicationBase::shutdown();
220 }
