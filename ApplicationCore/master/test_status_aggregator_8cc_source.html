171 BOOST_AUTO_TEST_CASE(testPriorities) {
172  std::cout << "testPriorities" << std::endl;
173 
174  // Define repeated check for a given priority mode
175  auto check = [&](ctk::StatusAggregator::PriorityMode mode, auto prio0, auto prio1, auto prio2, auto prio3,
176  bool warnMixed01 = false) {
177  // create app with initial values set to lowest prio value
178  TestPrioApplication app(prio0);
179 
180  app.aggregator = ctk::StatusAggregator{&app, "Aggregated/status", "aggregated status description", mode};
181 
182  ctk::TestFacility test(app);
183 
184  auto status = test.getScalar<int>("/Aggregated/status");
185 
186  test.runApplication();
187 
188  // check initial value
189  status.readNonBlocking(); // do not check return value, as it will only be written when changed
190  BOOST_CHECK_EQUAL(int(status), int(prio0));
191 
192  // Define repeated check to test all combinations of two given values with different priority.
193  // This kind of a whitebox test. We know that
194  // - the first aggregates variable has a different code path
195  // - the code path depends on the VersionNumber, so the write order matters and we have to set the version number
196  auto subcheck = [&](auto lower, auto higher, bool writeS2First, bool warnMixed = false) {
197  StatusGenerator* first;
198  StatusGenerator* second;
199  if(writeS2First) {
200  first = &app.s2;
201  second = &app.s1;
202  }
203  else {
204  first = &app.s1;
205  second = &app.s2;
206  }
207  std::cout << int(lower) << " vs. " << int(higher) << std::endl;
208  first->status = lower;
209  first->setCurrentVersionNumber({});
210  first->status.write();
211  second->status = lower;
212  second->setCurrentVersionNumber({});
213  second->status.write();
214  test.stepApplication();
215  status.readLatest();
216  BOOST_CHECK_EQUAL(int(status), int(lower));
217  first->status = lower;
218  first->setCurrentVersionNumber({});
219  first->status.write();
220  second->status = higher;
221  second->setCurrentVersionNumber({});
222  second->status.write();
223  test.stepApplication();
224  status.readLatest();
225  if(!warnMixed) {
226  BOOST_CHECK_EQUAL(int(status), int(higher));
227  }
228  else {
229  BOOST_CHECK_EQUAL(int(status), int(ctk::StatusOutput::Status::WARNING));
230  }
231  first->status = higher;
232  first->setCurrentVersionNumber({});
233  first->status.write();
234  second->status = lower;
235  second->setCurrentVersionNumber({});
236  second->status.write();
237  test.stepApplication();
238  status.readLatest();
239  if(!warnMixed) {
240  BOOST_CHECK_EQUAL(int(status), int(higher));
241  }
242  else {
243  BOOST_CHECK_EQUAL(int(status), int(ctk::StatusOutput::Status::WARNING));
244  }
245  first->status = higher;
246  first->setCurrentVersionNumber({});
247  first->status.write();
248  second->status = higher;
249  second->setCurrentVersionNumber({});
250  second->status.write();
251  test.stepApplication();
252  status.readLatest();
253  BOOST_CHECK_EQUAL(int(status), int(higher));
254  };
255 
256  // all prios against each other
257  auto runAllSubchecks = [&](bool writeS2First) {
258  subcheck(prio0, prio1, writeS2First, warnMixed01);
259  subcheck(prio0, prio2, writeS2First);
260  subcheck(prio0, prio3, writeS2First);
261  subcheck(prio1, prio2, writeS2First);
262  subcheck(prio1, prio3, writeS2First);
263  subcheck(prio2, prio3, writeS2First);
264  };
265  runAllSubchecks(false);
266  runAllSubchecks(true);
267  };
268 
269  // check all priority modes
270  std::cout << "PriorityMode::fwko" << std::endl;
271  check(ctk::StatusAggregator::PriorityMode::fwko, ctk::StatusOutput::Status::OFF, ctk::StatusOutput::Status::OK,
272  ctk::StatusOutput::Status::WARNING, ctk::StatusOutput::Status::FAULT);
273  std::cout << "PriorityMode::fwok" << std::endl;
274  check(ctk::StatusAggregator::PriorityMode::fwok, ctk::StatusOutput::Status::OK, ctk::StatusOutput::Status::OFF,
275  ctk::StatusOutput::Status::WARNING, ctk::StatusOutput::Status::FAULT);
276  std::cout << "PriorityMode::ofwk" << std::endl;
277  check(ctk::StatusAggregator::PriorityMode::ofwk, ctk::StatusOutput::Status::OK, ctk::StatusOutput::Status::WARNING,
278  ctk::StatusOutput::Status::FAULT, ctk::StatusOutput::Status::OFF);
279  std::cout << "PriorityMode::fw_warn_mixed" << std::endl;
280  check(ctk::StatusAggregator::PriorityMode::fw_warn_mixed, ctk::StatusOutput::Status::OFF,
281  ctk::StatusOutput::Status::OK, ctk::StatusOutput::Status::WARNING, ctk::StatusOutput::Status::FAULT, true);
282 }
