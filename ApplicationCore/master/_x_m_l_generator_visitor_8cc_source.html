32  void XMLGenerator::run() {
33  _rootElement->set_attribute("name", _app.getName());
34 
35  std::set<Model::ProcessVariableProxy, NetworkVisitor::ProcessVariableComperator> triggers;
36 
37  // Collect all triggers, add a TriggerReceiver placeholder for every device associated with that trigger
38  auto triggerCollector = [&](auto proxy) {
39  auto trigger = proxy.getTrigger();
40  if(not trigger.isValid()) return;
41 
42  triggers.insert(trigger);
43  proxy.addVariable(trigger, VariableNetworkNode(proxy.getAliasOrCdd(), 0));
44  };
45  _app.getModel().visit(triggerCollector, Model::depthFirstSearch, Model::keepDeviceModules);
46 
47  auto connectingVisitor = [&](auto proxy) { generateXMLNetwork(proxy); };
48 
49  // ChimeraTK::Model::keepParenthood - small optimisation for iterating the model only once
50  _app.getModel().visit(connectingVisitor, ChimeraTK::Model::depthFirstSearch, ChimeraTK::Model::keepProcessVariables,
51  ChimeraTK::Model::keepParenthood);
52  }

82  if(*type == typeid(int8_t)) {
83  dataTypeName = "int8";
84  }
85  else if(*type == typeid(uint8_t)) {
86  dataTypeName = "uint8";
87  }
88  else if(*type == typeid(int16_t)) {
89  dataTypeName = "int16";
90  }
91  else if(*type == typeid(uint16_t)) {
92  dataTypeName = "uint16";
93  }
94  else if(*type == typeid(int32_t)) {
95  dataTypeName = "int32";
96  }
97  else if(*type == typeid(uint32_t)) {
98  dataTypeName = "uint32";
99  }
100  else if(*type == typeid(int64_t)) {
101  dataTypeName = "int64";
102  }
103  else if(*type == typeid(uint64_t)) {
104  dataTypeName = "uint64";
105  }
106  else if(*type == typeid(float)) {
107  dataTypeName = "float";
108  }
109  else if(*type == typeid(double)) {
110  dataTypeName = "double";
111  }
112  else if(*type == typeid(std::string)) {
113  dataTypeName = "string";
114  }
115  else if(*type == typeid(ChimeraTK::Void)) {
116  dataTypeName = "Void";
117  }
118  else if(*type == typeid(ChimeraTK::Boolean)) {
119  dataTypeName = "Boolean";
120  }
121 
122  return dataTypeName;

127  void XMLGenerator::generateXMLForNode(NetworkInformation& net, const VariableNetworkNode& node) {
128  if(node.getType() != NodeType::ControlSystem) return;
129  // Create the directory for the path name in the XML document with all parent
130  // directories, if not yet existing: First split the publication name into
131  // components and loop over each component. For each component, try to find
132  // the directory node and create it it does not exist. After the loop, the
133  // "current" will point to the Element representing the directory.
134 
135  // strip the variable name from the path
136  ChimeraTK::RegisterPath directory(net.proxy->getFullyQualifiedPath());
137  directory--;
138 
139  // the namespace map is needed to properly refer to elements with an xpath
140  // expression in xmlpp::Element::find()
141  xmlpp::Node::PrefixNsMap nsMap{{"ac", ::detail::AC_NAMESPACE_URL}};
142 
143  // go through each directory path component
144  xmlpp::Element* current = _rootElement;
145  for(auto& pathComponent : directory.getComponents()) {
146  // find directory for this path component in the current directory
147  std::string xpath = std::string("ac:directory[@name='") + pathComponent + std::string("']");
148  auto list = current->find(xpath, nsMap);
149  if(list.empty()) { // not found: create it
150  xmlpp::Element* newChild = current->add_child("directory");
151  newChild->set_attribute("name", pathComponent);
152  current = newChild;
153  }
154  else {
155  assert(list.size() == 1);
156  current = dynamic_cast<xmlpp::Element*>(list[0]);
157  assert(current != nullptr);
158  }
159  }
160 
161  // now add the variable to the directory
162  xmlpp::Element* variable = current->add_child("variable");
163  ChimeraTK::RegisterPath pathName(node.getPublicName());
164  auto pathComponents = pathName.getComponents();
165 
166  // set the name attribute
167  variable->set_attribute("name", pathComponents[pathComponents.size() - 1]);
168 
169  auto dataTypeName = mapTypeToName(net.valueType);
170 
171  xmlpp::Element* valueTypeElement = variable->add_child("value_type");
172  valueTypeElement->set_child_text(dataTypeName);
173 
174  // add sub-element containing the data flow direction
175  std::string dataFlowName{"application_to_control_system"};
176  if(net.feeder == node) {
177  dataFlowName = "control_system_to_application";
178  if(!net.feeder.getDirection().withReturn) {
179  dataFlowName = "control_system_to_application";
180  }
181  else {
182  dataFlowName = "control_system_to_application_with_return";
183  }
184  }
185  xmlpp::Element* directionElement = variable->add_child("direction");
186  directionElement->set_child_text(dataFlowName);
187 
188  // add sub-element containing the engineering unit
189  xmlpp::Element* unitElement = variable->add_child("unit");
190  unitElement->set_child_text(net.unit);
191 
192  // add sub-element containing the description
193  xmlpp::Element* descriptionElement = variable->add_child("description");
194  descriptionElement->set_child_text(net.description);
195 
196  // add sub-element containing the description
197  xmlpp::Element* nElementsElement = variable->add_child("numberOfElements");
198  nElementsElement->set_child_text(std::to_string(net.valueLength));
199 
200  // add sub-element describing how this variable is connected
201  xmlpp::Element* connectedModules = variable->add_child("connections");
202  std::list nodeList(net.consumers.begin(), net.consumers.end());
203  nodeList.push_back(net.feeder);
204 
205  generatePeerList(connectedModules, nodeList);
206  }

210  void XMLGenerator::generatePeerList(
211  xmlpp::Element* connectedModules, const std::list<VariableNetworkNode>& nodeList) {
212  for(const auto& peerNode : nodeList) {
213  if(peerNode.getType() == NodeType::ControlSystem) continue;
214  bool feeding = peerNode == nodeList.back();
215  xmlpp::Element* peer = connectedModules->add_child("peer");
216 
217  if(peerNode.getType() == NodeType::Application) {
218  peer->set_attribute("type", "ApplicationModule");
219  // find ApplicationModule (owner might be VariableGroup deep down)
220  const auto* owningModule = peerNode.getOwningModule();
221  while(owningModule->getModuleType() == EntityOwner::ModuleType::VariableGroup) {
222  owningModule = dynamic_cast<const VariableGroup&>(*owningModule).getOwner();
223  }
224  // get name of ApplicationModule
225  auto qname = owningModule->getQualifiedName();
226  // strip leading application name
227  auto secondSlash = qname.find_first_of('/', 1);
228  if(secondSlash != std::string::npos) qname = qname.substr(secondSlash);
229  peer->set_attribute("name", qname);
230  const auto& owningModuleRef = *owningModule; // dereferencing in separate line to avoid linter warning
231  auto className = boost::core::demangle(typeid(owningModuleRef).name());
232  peer->set_attribute("class", className);
233  }
234  else if(peerNode.getType() == NodeType::Constant) {
235  peer->set_attribute("type", "Constant");
236  }
237  else if(peerNode.getType() == NodeType::Device) {
238  peer->set_attribute("type", "Device");
239  peer->set_attribute("name", peerNode.getDeviceAlias());
240  }
241  else if(peerNode.getType() == NodeType::TriggerProvider) {
242  peer->set_attribute("type", "TriggerProvider");
243  }
244  else if(peerNode.getType() == NodeType::TriggerReceiver) {
245  peer->set_attribute("type", "TriggerReceiver (" + peerNode.getDeviceAlias() + ")");
246  }
247  else {
248  peer->set_attribute("type", "Unknown (" + std::to_string(int(peerNode.getType())) + ")");
249  }
250 
251  peer->set_attribute("direction", feeding ? "feeding" : "consuming");
252  }
253  }
