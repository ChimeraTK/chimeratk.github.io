35  void StatusAggregator::populateStatusInput() {
36  scanAndPopulateFromHierarchyLevel(*getOwner(), ".");
37 
38  // Special treatment for DeviceModules, because they are formally not owned by the Application: If we are
39  // aggregating all tags at the top-level of the Application, include the status outputs of all DeviceModules.
40  if(getOwner() == &Application::getInstance() && _tagsToAggregate.empty()) {
41  for(auto& p : Application::getInstance().deviceModuleMap) {
42  scanAndPopulateFromHierarchyLevel(p.second->deviceError, ".");
43  }
44  }
45 
46  // Check if no inputs are present (nothing found to aggregate)
47  if(_inputs.empty()) {
48  throw ChimeraTK::logic_error("StatusAggregator " + VariableNetworkNode(_output._status).getQualifiedName() +
49  " has not found anything to aggregate.");
50  }
51  }

55  void StatusAggregator::scanAndPopulateFromHierarchyLevel(EntityOwner& module, const std::string& namePrefix) {
56  // a map used just for optimization of node lookup by name, which happens in inner loop below
57  std::map<std::string, VariableNetworkNode> nodesByName;
58  for(auto& node : module.getAccessorList()) {
59  nodesByName[node.getName()] = node;
60  }
61  // Search for StatusOutputs to aggregate
62  for(auto& node : module.getAccessorList()) {
63  // Filter required tags
64  // Note: findTag() cannot be used instead, since the search must be done on the original C++ hierarchy. Otherwise
65  // it is not possible to identify which StatusOutputs are aggregated by other StatusAggregators.
66  const auto& tags = node.getTags();
67  if(tags.find(StatusOutput::tagStatusOutput) == tags.end()) {
68  // StatusOutput's reserved tag not present: not a StatusOutput
69  continue;
70  }
71  bool skip = false;
72  for(const auto& tag : _tagsToAggregate) {
73  // Each tag attached to this StatusAggregator must be present at all StatusOutputs to be aggregated
74  if(tags.find(tag) == tags.end()) {
75  skip = true;
76  break;
77  }
78  }
79  if(skip) continue;
80 
81  // All variables with the StatusOutput::tagStatusOutput need to be feeding, otherwise someone has used the tag
82  // wrongly
83  if(node.getDirection().dir != VariableDirection::feeding) {
84  throw ChimeraTK::logic_error("BUG DETECTED: StatusOutput's reserved tag has been found on a consumer.");
85  }
86 
87  // Create matching input for the found StatusOutput of the other StatusAggregator
88  // Unfortunately, the qualified name of the newly-created node is useless,
89  // so we save the original one as description for indication in a message
90  _inputs.emplace_back(this, node.getName(), node.getQualifiedName(), HierarchyModifier::hideThis,
91  std::unordered_set<std::string>{tagInternalVars});
92  node >> _inputs.back()._status;
93  // look for matching status message output node
94  auto result = nodesByName.find(node.getName() + "_message");
95  if(result != nodesByName.end()) {
96  // tell the StatusWithMessageInput that it should consider the message source, and connect it
97  auto statusMsgNode = result->second;
98  _inputs.back().setMessageSource();
99  statusMsgNode >> _inputs.back()._message;
100  }
101  }
102 
103  // Search for StatusAggregators among submodules
104  const auto& ml = module.getSubmoduleList();
105  for(const auto& submodule : ml) {
106  // do nothing, if it is *this
107  if(submodule == this) continue;
108 
109  auto* aggregator = dynamic_cast<StatusAggregator*>(submodule);
110  if(aggregator != nullptr) {
111  // never aggregate on the same level
112  if(aggregator->getOwner() == getOwner()) {
113  continue;
114  }
115 
116  // if another StatusAggregator has been found, check tags
117  bool skip = false;
118  const auto& tags = aggregator->_tagsToAggregate;
119  for(const auto& tag : _tagsToAggregate) {
120  // Each tag attached to this StatusAggregator must be present at all StatusOutputs to be aggregated
121  if(tags.find(tag) == tags.end()) {
122  skip = true;
123  break;
124  }
125  }
126  if(skip) continue;
127 
128  // aggregate the StatusAggregator's result
129  auto statusNode = VariableNetworkNode(aggregator->_output._status);
130  _inputs.emplace_back(this, statusNode.getName(), "", HierarchyModifier::hideThis,
131  std::unordered_set<std::string>{tagInternalVars});
132  statusNode >> _inputs.back()._status;
133  auto msgNode = VariableNetworkNode(aggregator->_output._message);
134  _inputs.back().setMessageSource();
135  msgNode >> _inputs.back()._message;
136  return;
137  }
138  }
139 
140  // If no (matching) StatusAggregator is found, recurse into sub-modules
141  for(const auto& submodule : ml) {
142  if(dynamic_cast<StatusAggregator*>(submodule) != nullptr) continue; // StatusAggregators are already handled
143  scanAndPopulateFromHierarchyLevel(*submodule, namePrefix + "/" + submodule->getName());
144  }
145  }

149  int StatusAggregator::getPriority(StatusOutput::Status status) const {
150  using Status = StatusOutput::Status;
151 
152  // static helps against initializing over and over again
153  static const std::map<PriorityMode, std::map<Status, int32_t>> map_priorities{
154  {PriorityMode::fwko, {{Status::OK, 1}, {Status::FAULT, 3}, {Status::OFF, 0}, {Status::WARNING, 2}}},
155  {PriorityMode::fwok, {{Status::OK, 0}, {Status::FAULT, 3}, {Status::OFF, 1}, {Status::WARNING, 2}}},
156  {PriorityMode::ofwk, {{Status::OK, 0}, {Status::FAULT, 2}, {Status::OFF, 3}, {Status::WARNING, 1}}},
157  {PriorityMode::fw_warn_mixed, {{Status::OK, -1}, {Status::FAULT, 3}, {Status::OFF, -1}, {Status::WARNING, 2}}}};
158 
159  return map_priorities.at(_mode).at(status);
160  }

164  void StatusAggregator::mainLoop() {
165  // set up inputsMap which gives StatusWithMessageInput for transferelementId of members
166  std::map<TransferElementID, StatusWithMessageInput*> inputsMap;
167  for(auto& x : _inputs) {
168  inputsMap[x._status.getId()] = &x;
169  if(x.hasMessageSource) inputsMap[x._message.getId()] = &x;
170  }
171 
172  auto rag = readAnyGroup();
173  DataValidity lastStatusValidity = DataValidity::ok;
174  while(true) {
175  // find highest priority status of all inputs
176  StatusOutput::Status status;
177  StatusWithMessageInput* statusOrigin = nullptr;
178  // flag whether status has been set from an input already
179  bool statusSet = false;
180  // this stores getPriority(status) if statusSet=true
181  // Intent is to reduce evaluation frequency of getPriority
182  // the initial value provided here is only to prevent compiler warnings
183  int statusPrio = 0;
184  for(auto& inputPair : _inputs) {
185  StatusPushInput& input = inputPair._status;
186  auto prio = getPriority(input);
187  if(!statusSet || prio > statusPrio) {
188  status = input;
189  statusOrigin = &inputPair;
190  statusPrio = prio;
191  statusSet = true;
192  }
193  else if(prio == -1) { // -1 means, we need to warn about mixed values
194  if(statusPrio == -1 && input != status) {
195  status = StatusOutput::Status::WARNING;
196  statusOrigin = nullptr;
197  statusPrio = getPriority(status);
198  }
199  }
200  }
201  assert(statusSet);
202 
203  // write status only if changed, but always write initial value out
204  if(status != _output._status || _output._status.getVersionNumber() == VersionNumber{nullptr} ||
205  getDataValidity() != lastStatusValidity) {
206  if(!statusOrigin) {
207  // this can only happen if warning about mixed values
208  assert(status == StatusOutput::Status::WARNING);
209  _output.write(status, "warning - StatusAggregator inputs have mixed values");
210  }
211  else {
212  if(status != StatusOutput::Status::OK) {
213  // this either copies the message from corresponding string variable, or sets a generic message
214  auto msg = statusOrigin->getMessage();
215  _output.write(status, msg);
216  }
217  else {
218  _output.writeOk();
219  }
220  }
221 
222  lastStatusValidity = getDataValidity();
223  }
224 
225  // wait for changed inputs
226  waitForChange:
227  auto change = rag.readAny();
228  auto f = inputsMap.find(change);
229  if(f != inputsMap.end()) {
230  auto varPair = f->second;
231  if(!varPair->update(change)) goto waitForChange; // inputs not in consistent state yet
232  }
233 
234  // handle request for debug info
235  if(change == debug.value.getId()) {
236  static std::mutex debugMutex; // all aggregators trigger at the same time => lock for clean output
237  std::unique_lock<std::mutex> lk(debugMutex);
238 
239  std::cout << "StatusAggregtor " << getQualifiedName() << " debug info:" << std::endl;
240  for(auto& inputPair : _inputs) {
241  StatusPushInput& input = inputPair._status;
242  std::cout << input.getName() << " = " << input;
243  if(inputPair.hasMessageSource) {
244  std::cout << inputPair._message.getName() << " = " << (std::string)inputPair._message;
245  }
246  std::cout << std::endl;
247  }
248  std::cout << "debug info finished." << std::endl;
249  goto waitForChange;
250  }
251  }
252  }

256  void StatusAggregator::findTagAndAppendToModule(VirtualModule& virtualParent, const std::string& tag,
257  bool eliminateAllHierarchies, bool eliminateFirstHierarchy, bool negate, VirtualModule& root) const {
258  // Change behaviour to exclude the auto-generated inputs which are connected to the data sources. Otherwise those
259  // variables might get published twice to the control system, if findTag(".*") is used to connect the entire
260  // application to the control system.
261  // This is a temporary solution. In future, instead the inputs should be generated at the same place in the
262  // hierarchy as the source variable, and the connetion should not be made by the module itself. This currently would
263  // be complicated to implement, since it is difficult to find the correct virtual name for the variables.
264 
265  struct MyVirtualModule : VirtualModule {
266  using VirtualModule::VirtualModule;
267  using VirtualModule::findTagAndAppendToModule;
268  };
269 
270  MyVirtualModule tempParent("tempRoot", "", ModuleType::ApplicationModule);
271  MyVirtualModule tempRoot("tempRoot", "", ModuleType::ApplicationModule);
272  EntityOwner::findTagAndAppendToModule(
273  tempParent, tagInternalVars, eliminateAllHierarchies, eliminateFirstHierarchy, true, tempRoot);
274  tempParent.findTagAndAppendToModule(virtualParent, tag, false, true, negate, root);
275  tempRoot.findTagAndAppendToModule(root, tag, false, true, negate, root);
276  }
