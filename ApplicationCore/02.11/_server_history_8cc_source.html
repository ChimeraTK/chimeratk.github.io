36  ServerHistory::ServerHistory(EntityOwner* owner, const std::string& name, const std::string& description,
37  size_t historyLength, bool enableTimeStamps, HierarchyModifier hierarchyModifier,
38  const std::unordered_set<std::string>& tags)
39  : ApplicationModule(owner, name, description, hierarchyModifier, tags), _historyLength(historyLength),
40  _enbaleTimeStamps(enableTimeStamps) {
41  auto virtualLogging = getOwner()->findTag("history");
42  auto list = virtualLogging.getAccessorListRecursive();
43  size_t accessors = 0;
44  for(auto it = list.begin(); it != list.end(); ++it) {
45  // do not add the module itself
46  if(it->getOwningModule() == this) continue;
47  try {
48  // virtualLogging.getQualifiedName() returns the name of the app, e.g. /test and we remove that from the module
49  // name , e.g. /test/MyModule
50  auto namePrefix =
51  it->getOwningModule()->getQualifiedName().substr(virtualLogging.getQualifiedName().length() + 1);
52  prepareHierarchy(namePrefix);
53  boost::fusion::for_each(_accessorListMap.table, AccessorAttacher(*it, this, namePrefix / it->getName(), {}));
54  accessors++;
55  }
56  catch(ChimeraTK::logic_error& e) {
57  std::cerr << "Failed to add history variable: " << it->getQualifiedName() << " Error: " << e.what()
58  << std::endl;
59  }
60  }
61  if(accessors == 0) {
62  throw logic_error("No accessors for ServerHistory found. Did you use the tag 'history' for any variable?");
63  }
64  else {
65  std::cout << "Added " << accessors << " accessors to the ServerHistory Module." << std::endl;
66  }
67  }

69  void ServerHistory::prepareHierarchy(const RegisterPath& namePrefix) {
70  // create variable group map for namePrefix if needed
71  if(groupMap.find(namePrefix) == groupMap.end()) {
72  // search for existing parent (if any)
73  auto parentPrefix = namePrefix;
74  while(groupMap.find(parentPrefix) == groupMap.end()) {
75  if(parentPrefix == "/") break; // no existing parent found
76  parentPrefix = std::string(parentPrefix).substr(0, std::string(parentPrefix).find_last_of("/"));
77  }
78  // create all not-yet-existing parents
79  while(parentPrefix != namePrefix) {
80  EntityOwner* owner = this;
81  if(parentPrefix != "/") owner = &groupMap[parentPrefix];
82  auto stop = std::string(namePrefix).find_first_of("/", parentPrefix.length() + 1);
83  if(stop == std::string::npos) stop = namePrefix.length();
84  RegisterPath name = std::string(namePrefix).substr(parentPrefix.length(), stop - parentPrefix.length());
85  parentPrefix /= name;
86  groupMap[parentPrefix] = VariableGroup(owner, std::string(name).substr(1), "");
87  }
88  }
89  }

91  void ServerHistory::addSource(
92  const Module& source, const RegisterPath& namePrefix, const VariableNetworkNode& trigger) {
93  // for simplification, first create a VirtualModule containing the correct
94  // hierarchy structure (obeying eliminate hierarchy etc.)
95  auto dynamicModel = source.findTag(".*");
96 
97  prepareHierarchy(namePrefix);
98 
99  // add all accessors on this hierarchy level
100  for(auto& acc : dynamicModel.getAccessorList()) {
101  boost::fusion::for_each(_accessorListMap.table, AccessorAttacher(acc, this, namePrefix / acc.getName(), trigger));
102  }
103 
104  // recurse into submodules
105  for(auto mod : dynamicModel.getSubmoduleList()) {
106  addSource(*mod, namePrefix / mod->getName(), trigger);
107  }
108  }

124  template<typename UserType>
125  VariableNetworkNode ServerHistory::getAccessor(const std::string& variableName, const size_t& nElements) {
126  // check if variable name already registered
127  for(auto& name : _overallVariableList) {
128  if(name == variableName) {
129  throw ChimeraTK::logic_error("Cannot add '" + variableName +
130  "' to History since a variable with that "
131  "name is already registered.");
132  }
133  }
134  _overallVariableList.push_back(variableName);
135 
136  // add accessor and name to lists
137  auto& tmpList = boost::fusion::at_key<UserType>(_accessorListMap.table);
138  auto& nameList = boost::fusion::at_key<UserType>(_nameListMap.table);
139  auto dirName = variableName.substr(0, variableName.find_last_of("/"));
140  auto baseName = variableName.substr(variableName.find_last_of("/") + 1);
141  tmpList.emplace_back(std::piecewise_construct,
142  std::forward_as_tuple(
143  ArrayPushInput<UserType>{&groupMap[dirName], baseName + "_in", "", 0, "", {"_history_internal"}}),
144  std::forward_as_tuple(HistoryEntry<UserType>{_enbaleTimeStamps}));
145  for(size_t i = 0; i < nElements; i++) {
146  if(nElements == 1) {
147  // in case of a scalar history only use the variableName
148  tmpList.back().second.data.emplace_back(
149  ArrayOutput<UserType>{&groupMap[dirName], baseName, "", _historyLength, "", {getName()}});
150  if(_enbaleTimeStamps) {
151  tmpList.back().second.timeStamp.emplace_back(
152  ArrayOutput<uint64_t>{&groupMap[dirName], baseName + "_timeStamps",
153  "Time stamps for entries in the history buffer", _historyLength, "", {getName()}});
154  }
155  }
156  else {
157  // in case of an array history append the index to the variableName
158  tmpList.back().second.data.emplace_back(ArrayOutput<UserType>{
159  &groupMap[dirName], baseName + "_" + std::to_string(i), "", _historyLength, "", {getName()}});
160  if(_enbaleTimeStamps) {
161  tmpList.back().second.timeStamp.emplace_back(
162  ArrayOutput<uint64_t>{&groupMap[dirName], baseName + "_" + std::to_string(i) + "_timeStamps",
163  "Time stamps for entries in the history buffer", _historyLength, "", {getName()}});
164  }
165  }
166  }
167  nameList.push_back(variableName);
168 
169  // return the accessor
170  return tmpList.back().first;
171  }

176  template<typename PAIR>
177  void operator()(PAIR& pair) const {
178  auto& accessorList = pair.second;
179  for(auto accessor = accessorList.begin(); accessor != accessorList.end(); ++accessor) {
180  if(accessor->first.getId() == _id) {
181  for(size_t i = 0; i < accessor->first.getNElements(); i++) {
182  std::rotate(accessor->second.data.at(i).begin(), accessor->second.data.at(i).begin() + 1,
183  accessor->second.data.at(i).end());
184  *(accessor->second.data.at(i).end() - 1) = accessor->first[i];
185  accessor->second.data.at(i).write();
186  if(accessor->second.withTimeStamps) {
187  std::rotate(accessor->second.timeStamp.at(i).begin(), accessor->second.timeStamp.at(i).begin() + 1,
188  accessor->second.timeStamp.at(i).end());
189  *(accessor->second.timeStamp.at(i).end() - 1) =
190  boost::posix_time::to_time_t(boost::posix_time::second_clock::local_time());
191  accessor->second.timeStamp.at(i).write();
192  }
193  }
194  }
195  }

215  void ServerHistory::findTagAndAppendToModule(VirtualModule& virtualParent, const std::string& tag,
216  bool eliminateAllHierarchies, bool eliminateFirstHierarchy, bool negate, VirtualModule& root) const {
217  // Change behaviour to exclude the auto-generated inputs which are connected to the data sources. Otherwise those
218  // variables might get published twice to the control system, if findTag(".*") is used to connect the entire
219  // application to the control system.
220  // This is a temporary solution. In future, instead the inputs should be generated at the same place in the
221  // hierarchy as the source variable, and the connetion should not be made by the module itself. This currently would
222  // be complicated to implement, since it is difficult to find the correct virtual name for the variables.
223 
224  struct MyVirtualModule : VirtualModule {
225  using VirtualModule::VirtualModule;
226  using VirtualModule::findTagAndAppendToModule;
227  };
228 
229  MyVirtualModule tempParent("tempRoot", "", ModuleType::ApplicationModule);
230  MyVirtualModule tempRoot("tempRoot", "", ModuleType::ApplicationModule);
231  EntityOwner::findTagAndAppendToModule(
232  tempParent, "_history_internal", eliminateAllHierarchies, eliminateFirstHierarchy, true, tempRoot);
233  tempParent.findTagAndAppendToModule(virtualParent, tag, false, true, negate, root);
234  tempRoot.findTagAndAppendToModule(root, tag, false, true, negate, root);
235  }
