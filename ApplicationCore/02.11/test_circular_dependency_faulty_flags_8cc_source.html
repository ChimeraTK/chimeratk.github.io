59 struct ModuleA : TestModuleBase {
60  using TestModuleBase::TestModuleBase;
61 
62  ctk::ScalarPushInput<int> a{this, "a", "", ""};
63  ctk::ScalarPushInput<int> b{this, "b", "", ""};
64  ctk::ScalarOutput<int> circleResult{this, "circleResult", "", ""};
65 
66  void prepare() override { writeAll(); }
67 
68  void mainLoop() override {
69  // The circular inputs always are both coming as a pair, but we only want to write once.
70  // Hence we only put one of them into the ReadAnyGroup and always read the second one manually if the first one
71  // is read by the group.
72  ctk::ReadAnyGroup rag({a, b, inputGroup.circularInput1});
73 
74  while(true) {
75  auto id = rag.readAny();
76 
77  // A module with circular inputs and readAny must always actively break the circle. Otherwise for each external
78  // input and n-1 internal inputs and additional data element is inserted into the circle, which will let queues
79  // run over and re-trigger the circle all the time.
80  // This is a very typical scenario for circular connections: A module gets some input, triggers a helper module
81  // which calculates a value that is read back by the first module, and then the first module continues without
82  // re-triggering the circle.
83 
84  assert((id == a.getId() || id == b.getId()) || id == inputGroup.circularInput1.getId() ||
85  id == circularInput2.getId());
86 
87  if(id == inputGroup.circularInput1.getId()) {
88  // Read the other circular input as well. They always come in pairs.
89  circularInput2.read();
90  }
91 
92  if(id == a.getId() || id == b.getId()) {
93  circularOutput1 = static_cast<int>(inputGroup.circularInput1) + a;
94  outputGroup.circularOutput2 = static_cast<int>(circularInput2) + b;
95 
96  circularOutput1.write();
97  outputGroup.circularOutput2.write();
98  }
99  else { // new data is from the circular inputs
100  circleResult = static_cast<int>(inputGroup.circularInput1) + circularInput2;
101  circleResult.write();
102  }
103 
104  } // while(true)
105  } // mainLoop
106 }; // ModuleA
