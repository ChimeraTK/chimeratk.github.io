115  template<typename UserType>
116  void ThreadedFanOut<UserType>::run() {
117  Application::registerThread("ThFO" + FanOut<UserType>::impl->getName());
118  Application::getInstance().getTestableMode().lock("start");
119  _testableModeReached = true;
120 
121  ChimeraTK::VersionNumber version{nullptr};
122  version = readInitialValues();
123  while(true) {
124  // send out copies to slaves
125  boost::this_thread::interruption_point();
126  auto validity = FanOut<UserType>::impl->dataValidity();
127  for(auto& slave : FanOut<UserType>::slaves) {
128  // do not send copy if no data is expected (e.g. trigger)
129  if(slave->getNumberOfSamples() != 0) {
130  slave->accessChannel(0) = FanOut<UserType>::impl->accessChannel(0);
131  }
132  slave->setDataValidity(validity);
133  bool dataLoss = slave->writeDestructively(version);
134  if(dataLoss) Application::incrementDataLossCounter(slave->getName());
135  }
136  // receive data
137  boost::this_thread::interruption_point();
138  FanOut<UserType>::impl->read();
139  version = FanOut<UserType>::impl->getVersionNumber();
140  }
141  }

194  template<typename UserType>
195  void ThreadedFanOutWithReturn<UserType>::run() {
196  Application::registerThread("ThFO" + FanOut<UserType>::impl->getName());
197  Application::getInstance().getTestableMode().lock("start");
198  _testableModeReached = true;
199 
200  TransferElementID var;
201  ChimeraTK::VersionNumber version{nullptr};
202 
203  version = readInitialValues();
204 
205  ReadAnyGroup group({FanOut<UserType>::impl, _returnChannelSlave});
206  while(true) {
207  // send out copies to slaves
208  for(auto& slave : FanOut<UserType>::slaves) {
209  // do not feed back value returnChannelSlave if it was received from it
210  if(slave->getId() == var) continue;
211  // do not send copy if no data is expected (e.g. trigger)
212  if(slave->getNumberOfSamples() != 0) {
213  slave->accessChannel(0) = FanOut<UserType>::impl->accessChannel(0);
214  }
215  bool dataLoss = slave->writeDestructively(version);
216  if(dataLoss) Application::incrementDataLossCounter(slave->getName());
217  }
218  // receive data
219  boost::this_thread::interruption_point();
220  var = group.readAny();
221  boost::this_thread::interruption_point();
222  // if the update came through the return channel, return it to the feeder
223  if(var == _returnChannelSlave->getId()) {
224  FanOut<UserType>::impl->accessChannel(0).swap(_returnChannelSlave->accessChannel(0));
225  if(version < _returnChannelSlave->getVersionNumber()) {
226  version = _returnChannelSlave->getVersionNumber();
227  }
228  FanOut<UserType>::impl->write(version);
229  }
230  else {
231  version = FanOut<UserType>::impl->getVersionNumber();
232  }
233  }
234  }
