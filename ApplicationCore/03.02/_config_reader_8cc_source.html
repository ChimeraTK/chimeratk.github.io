232  template<typename T>
233  void ConfigReader::createArray(const std::string& name, const std::map<size_t, std::string>& values) {
234  std::vector<T> Tvalues;
235 
236  size_t expectedIndex = 0;
237  for(auto it = values.begin(); it != values.end(); ++it) {
238  // check index (std::map should be ordered by the index)
239  if(it->first != expectedIndex) {
240  parsingError("Array index " + std::to_string(expectedIndex) + " not found, but " + std::to_string(it->first) +
241  " was. "
242  "Sparse arrays are not supported!");
243  }
244  ++expectedIndex;
245 
246  // convert value into user type
247  T convertedValue = ChimeraTK::userTypeToUserType<T>(it->second);
248 
249  // store value in vector
250  Tvalues.push_back(convertedValue);
251  }
252 
253  auto moduleName = branch(name);
254  auto arrayName = leaf(name);
255  auto arrayOwner = _moduleTree->lookup(moduleName);
256 
257  // place the variable onto the vector
258  std::unordered_map<std::string, ConfigReader::Array<T>>& theMap = boost::fusion::at_key<T>(arrayMap.table);
259  theMap.emplace(std::make_pair(name, ConfigReader::Array<T>(arrayOwner, arrayName, Tvalues)));
260  }

283  void ConfigReader::construct(const std::string& fileName) {
284  auto fillVariableMap = [this](const Variable& var) {
285  bool processed{false};
286  boost::fusion::for_each(variableMap.table, FunctorFill(this, var.type, var.name, var.value, processed));
287  if(!processed) parsingError("Incorrect value '" + var.type + "' for attribute 'type' of the 'variable' tag.");
288  };
289 
290  auto fillArrayMap = [this](const ChimeraTK::Array& arr) {
291  // create accessor and store array value in map using functor
292  bool processed{false};
293  boost::fusion::for_each(arrayMap.table, ArrayFunctorFill(this, arr.type, arr.name, arr.values, processed));
294  if(!processed) parsingError("Incorrect value '" + arr.type + "' for attribute 'type' of the 'variable' tag.");
295  };
296 
297  auto parser = ConfigParser(fileName);
298  auto v = parser.getVariableList();
299  auto a = parser.getArrayList();
300 
301  for(const auto& var : *v) {
302  fillVariableMap(var);
303  }
304  for(const auto& arr : *a) {
305  fillArrayMap(arr);
306  }
307  }

446  void ConfigParser::parseModule(const xmlpp::Element* element, std::string parent_name) {
447  auto module_name = (element->get_name() == "configuration") // root node gets special treatment
448  ?
449  "" :
450  element->get_attribute("name")->get_value() + "/";
451 
452  parent_name += module_name;
453 
454  for(const auto& child : element->get_children()) {
455  element = dynamic_cast<const xmlpp::Element*>(child);
456  if(!element) {
457  continue; // ignore if not an element (e.g. comment)
458  }
459  else if(isVariable(element)) {
460  variableList_->emplace_back(prefix(parent_name, parseVariable(element)));
461  }
462  else if(isArray(element)) {
463  arrayList_->emplace_back(prefix(parent_name, parseArray(element)));
464  }
465  else if(isModule(element)) {
466  parseModule(element, parent_name);
467  }
468  else {
469  error("Unknown tag: " + element->get_name());
470  }
471  }
472  }

560  std::map<size_t, std::string> ConfigParser::gettArrayValues(const xmlpp::Element* element) {
561  bool valueFound = false;
562  std::map<size_t, std::string> values;
563 
564  for(const auto& valueChild : element->get_children()) {
565  const xmlpp::Element* valueElement = dynamic_cast<const xmlpp::Element*>(valueChild);
566  if(!valueElement) continue; // ignore comments etc.
567  validateValueNode(valueElement);
568  valueFound = true;
569 
570  auto index = valueElement->get_attribute("i");
571  auto value = valueElement->get_attribute("v");
572 
573  // get index as number and store value as a string
574  size_t intIndex;
575  try {
576  intIndex = std::stoi(index->get_value());
577  }
578  catch(std::exception& e) {
579  error("Cannot parse string '" + std::string(index->get_value()) + "' as an index number: " + e.what());
580  }
581  values[intIndex] = value->get_value();
582  }
583  // make sure there is at least one value
584  if(!valueFound) {
585  error("Each variable must have a value, either specified as an attribute or as child tags.");
586  }
587  return values;
588  }
