78 BOOST_AUTO_TEST_CASE_TEMPLATE(testTwoScalarPushAccessors, T, test_types) {
79  // FIXME: With the new scheme, there cannot be a 1:1 module connection any more, it will always be a network involving
80  // the ControlSystem
81  std::cout << "*** testTwoScalarPushAccessors<" << typeid(T).name() << ">" << std::endl;
82 
83  TestApplication<T> app;
84  app.testModule.feedingPush = {&app.testModule, "testTwoScalarPushAccessors", "", ""};
85  app.testModule.consumingPush = {&app.testModule, "testTwoScalarPushAccessors", "", ""};
86 
87  ctk::TestFacility tf{app, false};
88  tf.runApplication();
89  app.testModule.mainLoopStarted.wait(); // make sure the module's mainLoop() is entered
90 
91  // single threaded test
92  app.testModule.consumingPush = 0;
93  app.testModule.feedingPush = 42;
94  BOOST_CHECK(app.testModule.consumingPush == 0);
95  app.testModule.feedingPush.write();
96  BOOST_CHECK(app.testModule.consumingPush == 0);
97  app.testModule.consumingPush.read();
98  BOOST_CHECK(app.testModule.consumingPush == 42);
99 
100  // launch read() on the consumer asynchronously and make sure it does not yet
101  // receive anything
102  auto futRead = std::async(std::launch::async, [&app] { app.testModule.consumingPush.read(); });
103  BOOST_CHECK(futRead.wait_for(std::chrono::milliseconds(200)) == std::future_status::timeout);
104 
105  BOOST_CHECK(app.testModule.consumingPush == 42);
106 
107  // write to the feeder
108  app.testModule.feedingPush = 120;
109  app.testModule.feedingPush.write();
110 
111  // check that the consumer now receives the just written value
112  BOOST_CHECK(futRead.wait_for(std::chrono::milliseconds(2000)) == std::future_status::ready);
113  BOOST_CHECK(app.testModule.consumingPush == 120);
114 }

242 
243  // single threaded test

245  for(unsigned int i = 0; i < 10; ++i) app.testModule.feedingArray[i] = 99 + (T)i;
246  for(auto& val : app.testModule.consumingPushArray) BOOST_CHECK(val == 0);
247  app.testModule.feedingArray.write();
248  for(auto& val : app.testModule.consumingPushArray) BOOST_CHECK(val == 0);
249  app.testModule.consumingPushArray.read();
250  for(unsigned int i = 0; i < 10; ++i) BOOST_CHECK(app.testModule.consumingPushArray[i] == 99 + (T)i);
251 
252  // launch read() on the consumer asynchronously and make sure it does not yet
253  // receive anything
254  auto futRead = std::async(std::launch::async, [&app] { app.testModule.consumingPushArray.read(); });
255  BOOST_CHECK(futRead.wait_for(std::chrono::milliseconds(200)) == std::future_status::timeout);
256 
257  for(unsigned int i = 0; i < 10; ++i) BOOST_CHECK(app.testModule.consumingPushArray[i] == 99 + (T)i);
258 
259  // write to the feeder
