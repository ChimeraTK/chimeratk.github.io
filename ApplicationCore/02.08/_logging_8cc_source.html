130 void LoggingModule::mainLoop() {
131  file.reset(new std::ofstream());
132  messageCounter = 0;
133  std::stringstream greeter;
134  greeter << getName() << " " << getTime() << "There are " << sources.size()
135  << " modules registered for logging:" << std::endl;
136  broadcastMessage(greeter.str());
137  for(auto module = sources.begin(); module != sources.end(); module++) {
138  broadcastMessage(std::string("\t - ") + module->sendingModule);
139  id_list[module->msg.getId()] = &(*module);
140  }
141  auto group = readAnyGroup();
142  std::string msg;
143  MessageSource* currentSender;
144  LogLevel level;
145 
146  while(1) {
147  //we skip the initial value since it is empty anyway and set in Logger::prepare
148  auto id = group.readAny();
149  if(id_list.count(id) == 0) {
150  throw ChimeraTK::logic_error("Cannot find element id"
151  "when updating logging variables.");
152  }
153  try {
154  currentSender = id_list.at(id);
155  msg = (std::string)(currentSender->msg);
156  }
157  catch(std::out_of_range& e) {
158  throw ChimeraTK::logic_error("Cannot find element id"
159  "when updating logging variables.");
160  }
161  // if message is empty assume someone called writeAll() in a module containing the Logger -> ignore
162  if(msg.empty()) {
163  continue;
164  }
165  try {
166  level = static_cast<LogLevel>(std::strtoul(&msg.at(0), NULL, 0));
167  }
168  catch(std::out_of_range& e) {
169  throw ChimeraTK::logic_error("Cannot find message level"
170  "when updating logging variables.");
171  }
172  if(targetStream == 4) continue;
173  LogLevel setLevel = static_cast<LogLevel>((uint)logLevel);
174  std::string tmpStr = msg;
175  // remove message level
176  tmpStr = tmpStr.substr(1, tmpStr.size());
177  std::stringstream ss;
178  ss << level << getName() << "/" << currentSender->sendingModule << " " << getTime() << tmpStr;
179  if(targetStream == 0 || targetStream == 1) {
180  if(!((std::string)logFile).empty() && !file->is_open()) {
181  std::stringstream ss_file;
182  file->open((std::string)logFile, std::ofstream::out | std::ofstream::app);
183  if(!file->is_open() && setLevel <= LogLevel::ERROR) {
184  ss_file << LogLevel::ERROR << getName() << " " << getTime()
185  << "Failed to open log file for writing: " << (std::string)logFile << std::endl;
186  broadcastMessage(ss_file.str(), true);
187  }
188  else if(file->is_open() && setLevel <= LogLevel::INFO) {
189  ss_file << LogLevel::INFO << getName() << " " << getTime()
190  << "Opened log file for writing: " << (std::string)logFile << std::endl;
191  broadcastMessage(ss_file.str());
192  }
193  }
194  }
195  if(level >= setLevel) {
196  if(level < LogLevel::ERROR)
197  broadcastMessage(ss.str());
198  else
199  broadcastMessage(ss.str(), true);
200  }
201  }
202 }
