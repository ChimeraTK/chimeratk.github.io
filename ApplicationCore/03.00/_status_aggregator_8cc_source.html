12  StatusAggregator::StatusAggregator(ModuleGroup* owner, const std::string& name, const std::string& description,
13  PriorityMode mode, const std::unordered_set<std::string>& tagsToAggregate,
14  const std::unordered_set<std::string>& outputTags)
15  : ApplicationModule(owner, ".", description, outputTags), _output(this, name), _mode(mode),
16  _tagsToAggregate(tagsToAggregate) {
17  // check maximum size of tagsToAggregate
18  if(_tagsToAggregate.size() > 1) {
19  throw ChimeraTK::logic_error("StatusAggregator: List of tagsToAggregate must contain at most one tag.");

29  void StatusAggregator::populateStatusInput() {
30  auto model = dynamic_cast<ModuleGroup*>(_owner)->getModel();
31 
32  // set of potential inputs for this StatusAggregator instance
33  std::set<std::string> inputPathsSet;
34  // set of inputs for other, already created, StatusAggregator instances
35  std::set<std::string> anotherStatusAgregatorInputSet;
36  // map which assigns fully qualified path of StatusAggregator output to the ully qualified path StatusAggregator output message
37  std::map<std::string, std::string> statusToMessagePathsMap;
38 
39  auto scanModel = [&](auto proxy) {
40  if constexpr(ChimeraTK::Model::isApplicationModule(proxy)) {
41  StatusAggregator* staAggPtr = dynamic_cast<StatusAggregator*>(&proxy.getApplicationModule());
42  if(staAggPtr != nullptr) {
43  if(staAggPtr == this) return;
44 
45  if(_tagsToAggregate == staAggPtr->_tagsToAggregate) {
46  inputPathsSet.insert(staAggPtr->_output._status.getModel().getFullyQualifiedPath());
47 
48  statusToMessagePathsMap[staAggPtr->_output._status.getModel().getFullyQualifiedPath()] =
49  staAggPtr->_output._message.getModel().getFullyQualifiedPath();
50 
51  for(auto& anotherStatusAgregatorInput : staAggPtr->_inputs) {
52  anotherStatusAgregatorInputSet.insert(
53  anotherStatusAgregatorInput._status.getModel().getFullyQualifiedPath());
54  }
55  }
56  }
57  }
58 
59  if constexpr(ChimeraTK::Model::isVariable(proxy)) {
60  // check whether its not output of this (current) StatusAggregator. 'Current' StatusAggregator output is also
61  // visible in the scanned model and should be ignored
62  if(proxy.getFullyQualifiedPath().compare(_output._status.getModel().getFullyQualifiedPath()) == 0) {
63  return;
64  }
65 
66  auto tags = proxy.getTags();
67  // find another aggregator output - this is already covered by checking if given module is a StatusAggregator
68  if(tags.find(StatusAggregator::tagAggregatedStatus) != tags.end()) {
69  return;
70  }
71  // find status output - this is potential candidate to be aggregated
72  if(tags.find(StatusOutput::tagStatusOutput) != tags.end()) {
73  for(const auto& tagToAgregate : _tagsToAggregate) {
74  // Each tag attached to this StatusAggregator must be present at all StatusOutputs to be aggregated
75  if(tags.find(tagToAgregate) == tags.end()) {
76  return;
77  }
78  }
79  inputPathsSet.insert(proxy.getFullyQualifiedPath());
80  }
81  }
82  };
83 
84  model.visit(scanModel, ChimeraTK::Model::keepApplicationModules || ChimeraTK::Model::keepProcessVariables,
85  ChimeraTK::Model::breadthFirstSearch, ChimeraTK::Model::keepOwnership);
86 
87  for(auto& pathToBeRemoved : anotherStatusAgregatorInputSet) {
88  inputPathsSet.erase(pathToBeRemoved);
89  }
90 
91  for(auto& pathToBeAggregated : inputPathsSet) {
92  _inputs.emplace_back(
93  this, pathToBeAggregated, pathToBeAggregated, std::unordered_set<std::string>{tagInternalVars});
94  if(!statusToMessagePathsMap[pathToBeAggregated].empty())
95  _inputs.back().setMessageSource(statusToMessagePathsMap[pathToBeAggregated]);
96  }
97  }

99  int StatusAggregator::getPriority(StatusOutput::Status status) const {
100  using Status = StatusOutput::Status;
101 
102  // static helps against initializing over and over again
103  static const std::map<PriorityMode, std::map<Status, int32_t>> map_priorities{
104  {PriorityMode::fwko, {{Status::OK, 1}, {Status::FAULT, 3}, {Status::OFF, 0}, {Status::WARNING, 2}}},
105  {PriorityMode::fwok, {{Status::OK, 0}, {Status::FAULT, 3}, {Status::OFF, 1}, {Status::WARNING, 2}}},
106  {PriorityMode::ofwk, {{Status::OK, 0}, {Status::FAULT, 2}, {Status::OFF, 3}, {Status::WARNING, 1}}},
107  {PriorityMode::fw_warn_mixed, {{Status::OK, -1}, {Status::FAULT, 3}, {Status::OFF, -1}, {Status::WARNING, 2}}}};
108 
109  return map_priorities.at(_mode).at(status);
110  }

114  void StatusAggregator::mainLoop() {
115  // set up inputsMap which gives StatusWithMessageInput for transferelementId of members
116  std::map<TransferElementID, StatusWithMessageInput*> inputsMap;
117  for(auto& x : _inputs) {
118  inputsMap[x._status.getId()] = &x;
119  if(x.hasMessageSource) inputsMap[x._message.getId()] = &x;
120  }
121 
122  auto rag = readAnyGroup();
123  DataValidity lastStatusValidity = DataValidity::ok;
124  while(true) {
125  // find highest priority status of all inputs
126  StatusOutput::Status status{StatusOutput::Status::FAULT}; // initialised just to prevent warning
127  StatusWithMessageInput* statusOrigin = nullptr;
128  // flag whether status has been set from an input already
129  bool statusSet = false;
130  // this stores getPriority(status) if statusSet=true
131  // Intent is to reduce evaluation frequency of getPriority
132  // the initial value provided here is only to prevent compiler warnings
133  int statusPrio = 0;
134  for(auto& inputPair : _inputs) {
135  StatusPushInput& input = inputPair._status;
136  auto prio = getPriority(input);
137  if(!statusSet || prio > statusPrio) {
138  status = input;
139  statusOrigin = &inputPair;
140  statusPrio = prio;
141  statusSet = true;
142  }
143  else if(prio == -1) { // -1 means, we need to warn about mixed values
144  if(statusPrio == -1 && input != status) {
145  status = StatusOutput::Status::WARNING;
146  statusOrigin = nullptr;
147  statusPrio = getPriority(status);
148  }
149  }
150  }
151  assert(statusSet);
152 
153  // write status only if changed, but always write initial value out
154  if(status != _output._status || _output._status.getVersionNumber() == VersionNumber{nullptr} ||
155  getDataValidity() != lastStatusValidity) {
156  if(!statusOrigin) {
157  // this can only happen if warning about mixed values
158  assert(status == StatusOutput::Status::WARNING);
159  _output.write(status, "warning - StatusAggregator inputs have mixed values");
160  }
161  else {
162  if(status != StatusOutput::Status::OK) {
163  // this either copies the message from corresponding string variable, or sets a generic message
164  auto msg = statusOrigin->getMessage();
165  _output.write(status, msg);
166  }
167  else {
168  _output.writeOk();
169  }
170  }
171 
172  lastStatusValidity = getDataValidity();
173  }
174 
175  // wait for changed inputs
176  waitForChange:
177  auto change = rag.readAny();
178  auto f = inputsMap.find(change);
179  if(f != inputsMap.end()) {
180  auto varPair = f->second;
181  if(!varPair->update(change)) goto waitForChange; // inputs not in consistent state yet
182  }
183 
184  // handle request for debug info
185  if(change == debug.getId()) {
186  static std::mutex debugMutex; // all aggregators trigger at the same time => lock for clean output
187  std::unique_lock<std::mutex> lk(debugMutex);
188 
189  std::cout << "StatusAggregtor " << getQualifiedName() << " debug info:" << std::endl;
190  for(auto& inputPair : _inputs) {
191  StatusPushInput& input = inputPair._status;
192  std::cout << input.getName() << " = " << input;
193  if(inputPair.hasMessageSource) {
194  std::cout << inputPair._message.getName() << " = " << (std::string)inputPair._message;
195  }
196  std::cout << std::endl;
197  }
198  std::cout << "debug info finished." << std::endl;
199  goto waitForChange;
200  }
201  }
202  }
