103  template<typename UserType>
104  void FeedingFanOut<UserType>::addSlave(
105  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> slave, VariableNetworkNode&) {
106  // check if array shape is compatible, unless the receiver is a trigger
107  // node, so no data is expected
108  if(slave->getNumberOfSamples() != 0 &&
109  (slave->getNumberOfChannels() != 1 || slave->getNumberOfSamples() != this->getNumberOfSamples())) {
110  std::string what = "FeedingFanOut::addSlave(): Trying to add a slave '" + slave->getName();
111  what += "' with incompatible array shape! Name of fan out: '" + this->getName() + "'";
112  throw ChimeraTK::logic_error(what.c_str());
113  }
114 
115  // make sure slave is writeable
116  if(!slave->isWriteable()) {
117  throw ChimeraTK::logic_error("FeedingFanOut::addSlave() has been called "
118  "with a receiving implementation!");
119  }
120 
121  // handle return channels
122  if(_withReturn) {
123  if(slave->isReadable()) {
124  if(_hasReturnSlave) {
125  throw ChimeraTK::logic_error("FeedingFanOut: Cannot add multiple slaves with return channel!");
126  }
127 
128  // Assert the assumption about the return channel made in the constructor
129  assert(slave->getAccessModeFlags().has(AccessMode::wait_for_new_data));
130 
131  _hasReturnSlave = true;
132  _returnSlave = slave;
133 
134  // Set the readQeue from the return slave
135  // As this becomes the implemention of the feeding output, the flags are determined by that slave accessor
136  // If not _withReturn, the queue is not relevant because the feeding node is on output which is never read
137  this->_readQueue = _returnSlave->getReadQueue();
138  this->_accessModeFlags = _returnSlave->getAccessModeFlags();
139  }
140  }
141 
142  // add the slave
143  FanOut<UserType>::slaves.push_back(slave);
144  }

167  template<typename UserType>
168  void FeedingFanOut<UserType>::doPostRead(TransferType type, bool hasNewData) {
169  if(this->_disabled) return;
170  assert(_withReturn);
171  assert(_hasReturnSlave);
172 
173  auto _ = cppext::finally([&] {
174  if(!hasNewData) return;
175  _returnSlave->accessChannel(0).swap(ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0]);
176  // distribute return-channel update to the other slaves
177  for(auto& slave : FanOut<UserType>::slaves) { // send out copies to slaves
178  if(slave == _returnSlave) continue;
179  if(slave->getNumberOfSamples() != 0) { // do not send copy if no data is expected (e.g. trigger)
180  slave->accessChannel(0) = ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0];
181  }
182  slave->writeDestructively(_returnSlave->getVersionNumber());
183  }
184  });
185 
186  _returnSlave->postRead(type, hasNewData);
187 
188  this->_versionNumber = _returnSlave->getVersionNumber();
189  this->_dataValidity = _returnSlave->dataValidity();
190  }

194  template<typename UserType>
195  void FeedingFanOut<UserType>::doPreWrite(TransferType, VersionNumber) {
196  if(this->_disabled) return;
197  for(auto& slave : FanOut<UserType>::slaves) { // send out copies to slaves
198  if(slave->getNumberOfSamples() != 0) { // do not send copy if no data is expected (e.g. trigger)
199  if(slave == FanOut<UserType>::slaves.front()) { // in case of first slave, swap instead of copy
200  slave->accessChannel(0).swap(ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0]);
201  }
202  else { // not the first slave: copy the data from the first slave
203  slave->accessChannel(0) = FanOut<UserType>::slaves.front()->accessChannel(0);
204  }
205  }
206  slave->setDataValidity(this->dataValidity());
207  }
208 
209  // Don't call pre-write on the slaves. Each slave has to do it's own exception handling, so we call the whole
210  // operation in doWriteTansfer(). To fulfill the TransferElement specification we would have to check the
211  // pre-conditions here so no logic error is thrown in the transfer phase (logic_errors are predictable and can
212  // always pre prevented. They should be thrown here already).
213  // FIXME: At the moment we can be lazy about it. logic_errors are not treated in ApplicationCore and the only
214  // effect is that the logic_error would be delayed after postRead() and terminate the application there, and not
215  // after the transfer. Advantage about being lazy: It safes a few virtual function calls.
216  }

220  template<typename UserType>
221  bool FeedingFanOut<UserType>::doWriteTransfer(ChimeraTK::VersionNumber versionNumber) {
222  if(this->_disabled) return false;
223  bool dataLost = false;
224  bool isFirst = true;
225  for(auto& slave : FanOut<UserType>::slaves) {
226  bool ret;
227  if(isFirst) {
228  isFirst = false;
229  ret = slave->write(versionNumber);
230  }
231  else {
232  ret = slave->writeDestructively(versionNumber);
233  }
234  if(ret) dataLost = true;
235  }
236  return dataLost;
237  }
