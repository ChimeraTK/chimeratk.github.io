47  void ExceptionHandlingDecorator<UserType>::doPreWrite(TransferType type, VersionNumber versionNumber) {
48  /* For writable accessors, copy data to the recoveryAcessor before perfroming the write.
49  * Otherwise, the decorated accessor may have swapped the data out of the user buffer already.
50  * This obtains a shared lock from the DeviceModule, hence, the regular writing happeniin here
51  * can be performed in shared mode of the mutex and accessors are not blocking each other.
52  * In case of recovery, the DeviceModule thread will take an exclusive lock so that this thread can not
53  * modify the recoveryAcessor's user buffer while data is written to the device.
54  */
55  {
56  _inhibitWriteTransfer = false;
57  _hasThrownLogicError = false;
58  _dataLostInPreviousWrite = false;
59  auto recoverylock{_deviceModule->getRecoverySharedLock()};
60 
61  if(_recoveryAccessor != nullptr) {
62  if(!_recoveryHelper->wasWritten && (_recoveryHelper->writeOrder != 0)) {
63  _dataLostInPreviousWrite = true;
64  }
65 
66  // Access to _recoveryAccessor is only possible channel-wise
67  for(unsigned int ch = 0; ch < _recoveryAccessor->getNumberOfChannels(); ++ch) {
68  _recoveryAccessor->accessChannel(ch) = buffer_2D[ch];
69  }
70  _recoveryHelper->versionNumber = versionNumber;
71  _recoveryHelper->writeOrder = _deviceModule->writeOrder();
72  _recoveryHelper->wasWritten = false;
73  }
74  else {
75  _hasThrownLogicError = true;
76  throw ChimeraTK::logic_error(
77  "ChimeraTK::ExceptionhandlingDecorator: Calling write() on a non-writeable accessor is not supported ");
78  }
79 
80  boost::shared_lock<boost::shared_mutex> errorLock(_deviceModule->errorMutex);
81  if(_deviceModule->deviceHasError) {
82  _inhibitWriteTransfer = true;
83  return;
84  }
85 
86  ++_deviceModule->synchronousTransferCounter; // must be under the lock
87 
88  } // lock guard goes out of scope
89 
90  // Now delegate call to the generic decorator, which swaps the buffer, without adding our exception handling with the generic transfer
91  // preWrite and postWrite are only delegated if the transfer is allowed.
92  ChimeraTK::NDRegisterAccessorDecorator<UserType>::doPreWrite(type, versionNumber);
93  }

96  void ExceptionHandlingDecorator<UserType>::doPostWrite(TransferType type, VersionNumber versionNumber) {
97  if(_hasThrownLogicError) {
98  // preWrite has not been delegated, so there is nothing to do here. Let
99  // postWrite() throw the active exception we have. Don't clear logic erros here.
100  return;
101  }
102  if(!_inhibitWriteTransfer) {
103  --_deviceModule->synchronousTransferCounter;
104  try {
105  ChimeraTK::NDRegisterAccessorDecorator<UserType>::doPostWrite(type, versionNumber);
106  {
107  auto recoverylock{_deviceModule->getRecoverySharedLock()};
108  // the transfer was successful or doPostRead did not throw and we reach this point,
109  // so we matk these data as written
110  _recoveryHelper->wasWritten = true;
111  } // end scope for recovery lock
112  }
113  catch(ChimeraTK::runtime_error& e) {
114  _deviceModule->reportException(std::string(e.what()) + " (seen by '" + _target->getName() + "')");
115  }
116  }
117  assert(_activeException == nullptr);
118  }

121  void ExceptionHandlingDecorator<UserType>::doPostRead(TransferType type, bool hasNewData) {
122  // preRead has not been called when the transfer was not allowed. Don't call postRead in this case.
123  if(!_hasThrownToInhibitTransfer) {
124  try {
125  if(!TransferElement::_accessModeFlags.has(AccessMode::wait_for_new_data)) { // was as synchronous transfer
126  --_deviceModule->synchronousTransferCounter;
127  }
128  _target->setActiveException(this->_activeException);
129  _target->postRead(type, hasNewData);
130  if(hasNewData) {
131  // Reset the flag after a successful read.
132  // It is only reset if there was new data. A readNonBlocking on a faulty device is not different to a readNonBlocking on working device: There just
133  // is no new data. We only reset it on the next successful read with the initial value, otherwise keep the exception flag.
134  _hasReportedException = false;
135  }
136  }
137  catch(ChimeraTK::runtime_error& e) {
138  _deviceModule->reportException(std::string(e.what()) + " (seen by '" + _target->getName() + "')");
139  _hasReportedException = true;
140  }
141  }
142  else {
143  _activeException = nullptr;
144  }
145 
146  if(_hasReportedException || _hasThrownToInhibitTransfer) {
147  _dataValidity = DataValidity::faulty;
148  // Note: This assertion does not hold
149  // See discussion in https://github.com/ChimeraTK/DeviceAccess/pull/178
150  // assert(_deviceModule->getExceptionVersionNumber() > _versionNumber);
151  if(_deviceModule->getExceptionVersionNumber() > _versionNumber) {
152  _versionNumber = _deviceModule->getExceptionVersionNumber();
153  }
154  }
155  else {
156  _dataValidity = _target->dataValidity();
157  // Note: This assertion does not hold
158  // See discussion in https://github.com/ChimeraTK/DeviceAccess/pull/178
