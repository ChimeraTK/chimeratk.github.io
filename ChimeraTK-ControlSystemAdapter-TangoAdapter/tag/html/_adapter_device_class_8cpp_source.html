129  void AdapterDeviceClass::device_factory(const Tango::DevVarStringArray* devlist_ptr) {
130  // Add your own code
131 
132  auto& adapter = TangoAdapter::getInstance();
133  auto& mapper = adapter.getMapper();
134 
135  auto deviceClass = mapper.getClass(get_name());
136 
137  // Create devices and add it into the device list
138  for(unsigned int i = 0; i < devlist_ptr->length(); i++) {
139  const auto* deviceName = (*devlist_ptr)[i].in();
140 
141  if(!deviceClass->hasDevice(deviceName) &&
142  !deviceClass->hasDevice(TangoAdapter::PLAIN_IMPORT_DUMMY_DEVICE.data())) {
143  std::cerr << "Device " << deviceName << "not known in attribute mapper. Skipping." << std::endl;
144  continue;
145  }
146 
147  auto device = std::make_unique<AdapterDeviceImpl>(this, deviceName);
148  device->init_device();
149  if(!deviceClass->hasDevice(deviceName) &&
150  !deviceClass->hasDevice(TangoAdapter::PLAIN_IMPORT_DUMMY_DEVICE.data())) {
151  // See if we have the "generic" device"
152  DEV_ERROR_STREAM(device) << "Device " << deviceName << "not known in attribute mapper. Expect issues"
153  << std::endl;
154  device->set_state(Tango::FAULT);
155  device->set_status("Device was not found in mapping file, no variables could be mapped.");
156  }
157  else {
158  // Move the generic device we have to this device.
159  if(!deviceClass->hasDevice(deviceName)) {
160  auto genericDevice = deviceClass->devicesInDeviceClass[TangoAdapter::PLAIN_IMPORT_DUMMY_DEVICE.data()];
161  deviceClass->devicesInDeviceClass.erase(TangoAdapter::PLAIN_IMPORT_DUMMY_DEVICE.data());
162  genericDevice->name = deviceName;
163  deviceClass->devicesInDeviceClass[deviceName] = genericDevice;
164  }
165  device->attachToClassAttributes(deviceClass);
166  }
167 
168  // Check before if database used.
169  if(Tango::Util::_UseDb && !Tango::Util::_FileDb) {
170  export_device(device.get());
171  }
172  else {
173  export_device(device.get(), deviceName);
174  }
175 
176  // Hand over device pointer to Tango
177  device_list.push_back(device.release());
178  }
179 
180  // Manage dynamic attributes if any
181  erase_dynamic_attributes(get_class_attr()->get_attr_list());
182 
183  for(auto* dev : device_list) {
184  // Only set the device state ok ON if it is still in the default constructed
185  // value
186  if(dev->get_state() == Tango::UNKNOWN) {
187  dev->set_state(Tango::ON);
188  dev->set_status("Application is running.");
189  }
190  }
191  }

230  void AdapterDeviceClass::erase_dynamic_attributes(std::vector<Tango::Attr*>& att_list) {
231  for(auto* dev_impl : device_list) {
232  auto* dev = dynamic_cast<AdapterDeviceImpl*>(dev_impl);
233  assert(dev != nullptr);
234 
235  std::vector<Tango::Attribute*>& dev_att_list = dev->get_device_attr()->get_attribute_list();
236  std::vector<Tango::Attribute*>::iterator ite_att;
237  for(ite_att = dev_att_list.begin(); ite_att != dev_att_list.end(); ++ite_att) {
238  std::string att_name((*ite_att)->get_name_lower());
239  if((att_name == "state") || (att_name == "status")) {
240  continue;
241  }
242  auto ite_str = find(defaultAttList.begin(), defaultAttList.end(), att_name);
243  if(ite_str == defaultAttList.end()) {
244  TANGO_LOG_DEBUG << att_name << " is a UNWANTED dynamic attribute for device " << dev->name() << std::endl;
245  Tango::Attribute& att = dev->get_device_attr()->get_attr_by_name(att_name.c_str());
246  dev->remove_attribute(att_list[att.get_attr_idx()], true, false);
247  --ite_att;
248  }
249  }
250  }
