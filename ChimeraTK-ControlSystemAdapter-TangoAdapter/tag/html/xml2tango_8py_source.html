13data_map = {

29def create_attribute_list(path: str, output: typing.TextIO) -> None:
30 root = ET.parse(path)
31 application = root.xpath("/ac:application", namespaces=ns)[0]
32
33 for elem in root.xpath("//ac:variable", namespaces=ns):
34 p = elem
35 path = ""
36 while p is not application:
37 path = p.get("name") + "/" + path
38 p = p.getparent()
39 path=path[0:-1]
40
41 value_type = data_map[elem.xpath("./ac:value_type/text()", namespaces=ns)[0]]
42 elements = int(elem.xpath("./ac:numberOfElements/text()", namespaces=ns)[0])
43 if elements > 1:
44 form="SPECTRUM"
45 else:
46 form="SCALAR"
47 description = elem.xpath("./ac:description/text()", namespaces=ns)
48 if len(description) == 0:
49 description = ""
50 else:
51 description = description[0]
52 unit = elem.xpath("./ac:unit/text()", namespaces=ns)
53 if len(unit) == 0:
54 unit="unknown"
55 else:
56 unit = unit[0]
57
58 print(";".join((path.replace("/", "_"), path, form, value_type, description, unit)), file=output)
59
60
