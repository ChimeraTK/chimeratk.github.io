36 SpectrumAttribTempl<TangoType, AdapterType, TangoWriteType>::SpectrumAttribTempl(AttributeProperty& attProperty)

47
48 if(!(attProperty.unit.empty())) {
49 axis_prop.set_unit(attProperty.unit.c_str());
50 }
51
52 // Since Tango does not support int8 natively and we have to resort to SHORT,

56 axis_prop.set_max_value(std::to_string(std::numeric_limits<int8_t>::max()).c_str());
57 }
58
59 set_default_properties(axis_prop);

64 template<typename TangoType, typename AdapterType, typename TangoWriteType>
65 void SpectrumAttribTempl<TangoType, AdapterType, TangoWriteType>::read(
66 Tango::DeviceImpl* dev, Tango::Attribute& att) {
67 auto* adapterDevice = dynamic_cast<TangoAdapter::AdapterDeviceImpl*>(dev);
68 assert(adapterDevice != nullptr);
69
70 auto pv = adapterDevice->getPvForAttribute(att.get_name());
71 assert(pv.getValueType() == typeid(AdapterType));
72
73 auto processSpectrum = boost::reinterpret_pointer_cast<GenericAccessor>(pv.getHighLevelImplElement());
74
75 if(!processSpectrum) {
76 DEV_WARN_STREAM(dev) << "pv type mismatch (expected: " << boost::core::demangle(pv.getValueType().name())
77 << ", got :" << boost::core::demangle(typeid(AdapterType()).name()) << ")" << std::endl;
78
79 att.set_quality(Tango::AttrQuality::ATTR_INVALID);
80 return;
81 }
82
83 auto length = processSpectrum->getNumberOfSamples();
84 // NO, lint. Tango need a runtime-length C-Style array here, cannot use std::array.
85 // NOLINTNEXTLINE(modernize-avoid-c-arrays)
86 auto data = std::make_unique<TangoType[]>(length);
87
88 for(unsigned int i = 0; i < length; i++) {
89 if constexpr(std::is_same_v<TangoType, Tango::DevString>) {
90 data.get()[i] = Tango::string_dup(processSpectrum->accessData(i).c_str());
91 }
92 else if constexpr(std::is_same_v<AdapterType, int8_t>) {
93 // Lint: We need to use Short as transport for int8_t here as Tango does not support this type,
94 // there are no chars involved.
95 // NOLINTNEXTLINE(bugprone-signed-char-misuse)
96 data.get()[i] = Tango::DevShort(processSpectrum->accessData(i));
97 }
98 else {
99 data.get()[i] = processSpectrum->accessData(i);
100 }
101 }
102 att.set_value(data.release(), length, 0, true);
103
104 if(processSpectrum->dataValidity() != ChimeraTK::DataValidity::ok) {
105 att.set_quality(Tango::AttrQuality::ATTR_INVALID);
106 }
107 else {
108 att.set_quality(Tango::AttrQuality::ATTR_VALID);
109 }
110 }

114 template<typename TangoType, typename AdapterType, typename TangoWriteType>
115 void SpectrumAttribTempl<TangoType, AdapterType, TangoWriteType>::write(
116 Tango::DeviceImpl* dev, Tango::WAttribute& att) {
117 auto* adapterDevice = dynamic_cast<TangoAdapter::AdapterDeviceImpl*>(dev);
118 assert(adapterDevice != nullptr);
119
120 auto pv = adapterDevice->getPvForAttribute(att.get_name());
121 assert(pv.getValueType() == typeid(AdapterType));
122
123 auto processSpectrum = boost::reinterpret_pointer_cast<GenericAccessor>(pv.getHighLevelImplElement());
124
125 if(!processSpectrum) {
126 DEV_WARN_STREAM(dev) << "pv type mismatch (expected: " << boost::core::demangle(pv.getValueType().name())
127 << ", got :" << boost::core::demangle(typeid(AdapterType()).name()) << ")" << std::endl;
128
129 att.set_quality(Tango::AttrQuality::ATTR_INVALID);
130 return;
131 }
132
133 auto length = processSpectrum->getNumberOfSamples();
134
135 auto& processVector = processSpectrum->accessChannel(0);
136
137 auto arraySize = att.get_write_value_length();
138 std::string memoried_value;
139
140 if(arraySize != length) {
141 std::stringstream msg;
142 msg << "Written size " << arraySize << " does not match length " << length << "\n";
143 DEV_ERROR_STREAM(dev) << "WRITE_ERROR " << msg.str() << std::endl;
144
145 Tango::Except::throw_exception("WRITE_ERROR", msg.str(), "SpectrumAttribTempl::write()");
146 }
147
148 const TangoWriteType* data = nullptr;
149 if(att.get_user_set_write_value()) {
150 if constexpr(std::is_same_v<Tango::DevUChar, TangoType>) {
151 data = att.get_last_written_uch()->get_buffer();
152 }
153 else if constexpr(std::is_same_v<Tango::DevUShort, TangoType>) {
154 data = att.get_last_written_ush()->get_buffer();
155 }
156 else if constexpr(std::is_same_v<Tango::DevULong, TangoType>) {
157 data = att.get_last_written_ulg()->get_buffer();
158 }
159 else if constexpr(std::is_same_v<Tango::DevULong64, TangoType>) {
160 data = att.get_last_written_ulg64()->get_buffer();
161 }
162 else if constexpr(std::is_same_v<Tango::DevShort, TangoType>) {
163 data = att.get_last_written_sh()->get_buffer();
164 }
165 else if constexpr(std::is_same_v<Tango::DevLong, TangoType>) {
166 data = att.get_last_written_lg()->get_buffer();
167 }
168 else if constexpr(std::is_same_v<Tango::DevLong64, TangoType>) {
169 data = att.get_last_written_lg64()->get_buffer();
170 }
171 else if constexpr(std::is_same_v<Tango::DevFloat, TangoType>) {
172 data = att.get_last_written_fl()->get_buffer();
173 }
174 else if constexpr(std::is_same_v<Tango::DevDouble, TangoType>) {
175 data = att.get_last_written_db()->get_buffer();
176 }
177 else if constexpr(std::is_same_v<Tango::DevBoolean, TangoType>) {
178 data = att.get_last_written_boo()->get_buffer();
179 }
180 else if constexpr(std::is_same_v<Tango::DevString, TangoType>) {
181 data = att.get_last_written_str()->get_buffer();
182 }
183 else {
184 DEV_ERROR_STREAM(dev) << "Unsupported tango type " << boost::core::demangle(typeid(TangoType).name())
185 << std::endl;
186 assert(false);
187 }
188 }
189 else {
190 att.get_write_value(data);
191
192 memoried_value = arrayToString(data, std::min<size_t>(arraySize, length));
193 }
194
195 if(data != nullptr) {
196 if constexpr(std::is_same_v<TangoType, Tango::DevString>) {
197 std::transform(data, data + arraySize, processVector.begin(), [](auto v) { return std::string(v); });
198 }
199 else {
200 std::copy(data, data + arraySize, processVector.begin());
201 }
202 }
203 else {
204 DEV_WARN_STREAM(dev) << "Did not get any written data from " << att.get_name() << std::endl;
205 }
206
207 if(att.get_quality() == Tango::AttrQuality::ATTR_INVALID || data == nullptr) {
208 processSpectrum->setDataValidity(ChimeraTK::DataValidity::faulty);
209 }
210 else {
211 processSpectrum->setDataValidity(ChimeraTK::DataValidity::ok);
212 }
213
214 processSpectrum->write();
215 if(att.get_user_set_write_value()) {
216 // if user writing, set read value
217 att.set_rvalue();
218 }
219 else {
220 if(Tango::Util::_UseDb) {
221 // memory the written value if connected to database
222 TangoAdapter::setProperty<std::string>(dev, memoriedPropertyName, memoried_value);
223 }
224 }
225 }
