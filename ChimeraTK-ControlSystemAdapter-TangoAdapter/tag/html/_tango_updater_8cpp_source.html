13 void TangoUpdater::addVariable(ChimeraTK::TransferElementAbstractor variable, const std::string& attrId) {
14 TANGO_LOG_DEBUG << "TangoAdapter::Updater adding variable " << attrId << std::endl;
15
16 if(variable.isReadable()) {
17 auto id = variable.getId();
18 // device, Attribute read(device,attribute)
19 if(_descriptorMap.find(id) ==
20 _descriptorMap.end()) { // jade: push to _elementsToRead if not found in _descriptorMap
21 _elementsToRead.push_back(variable);
22 }
23 else {
24 _descriptorMap[id].additionalTransferElements.insert(variable.getHighLevelImplElement());
25 }
26 // push variable.getHighLevelImplElement()/updaterFunction/eq_fct in _descriptorMap
27 // and push TransferElementAbstractor _elementsToRead
28 _descriptorMap[id].attributeID.push_back(attrId);
29 }
30 }

34 void TangoUpdater::updateLoop() {
35 if(_elementsToRead.empty()) {
36 return;
37 }
38
39 ChimeraTK::ReadAnyGroup group(_elementsToRead.begin(), _elementsToRead.end());
40
41 // Call preRead for all TEs on additional transfer elements. waitAny() is doing this for all elements in the
42 // ReadAnyGroup. Unnecessary calls to preRead() are anyway ignored and merely pose a performance issue. For large
43 // servers, the performance impact is significant, hence we keep track of the TEs which need to be called.
44 for(auto& pair : _descriptorMap) {
45 for(const auto& elem : pair.second.additionalTransferElements) {
46 elem->preRead(ChimeraTK::TransferType::read); // TransferElement
47 }
48 }
49
50 while(true) {
51 // Wait until any variable got an update
52 auto notification = group.waitAny(); // inside has a //handlePreRead TransferElement
53 auto updatedElement = notification.getId(); // 1 ID
54 auto& descriptor = _descriptorMap[updatedElement]; // one descriptor
55
56 // Complete the read transfer of the process variable
57 notification.accept();
58
59 // Call postRead for all TEs for the updated ID
60 for(const auto& elem : descriptor.additionalTransferElements) {
61 elem->postRead(ChimeraTK::TransferType::read, true);
62 }
63
64 // FIXME: Ideally we would fill the Tango buffer for the attribute here, then attribute->read()
65 // would just send it out to CORBA
66 // FIXME: Also we would need to toggle the event here, once supported
67
68 // Call preRead for all TEs for the updated ID
69 for(const auto& elem : descriptor.additionalTransferElements) {
70 elem->preRead(ChimeraTK::TransferType::read);
71 }
72
73 // Allow shutting down this thread...
74 boost::this_thread::interruption_point();
75 }
76 }

86 void TangoUpdater::stop() {
87 try {
88 _syncThread.interrupt();
89 for(auto& var : _elementsToRead) {
90 var.getHighLevelImplElement()->interrupt();
91 }
92 _syncThread.join();
93 }
94 catch(boost::thread_interrupted&) {
95 // Ignore
96 }
97 catch(std::system_error& e) {
98 TANGO_LOG_INFO << ::TangoAdapter::AdapterDeviceClass::getClassName()
99 << ":Failed to shut down updater thread: " << e.what() << std::endl;
100 }
101 }
