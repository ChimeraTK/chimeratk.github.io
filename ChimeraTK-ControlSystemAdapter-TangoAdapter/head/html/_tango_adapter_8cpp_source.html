18  template<typename TangoType, typename AdapterType>
19  void writeInitialSpectrumValue(Tango::DeviceImpl* device, const std::string& memoriedValue,
20  Tango::WAttribute& writeAttribute, Tango::Attr* attr) {
21 
22  // StringToArray will create a std::vector<bool> which is an optimised version of std::vector
23  // But Tango needs a plain array of bool, so we have to convert it here.
24  if constexpr(std::is_same_v<Tango::DevBoolean, TangoType>) {
25  auto values = ChimeraTK::stringToArray<TangoType>(memoriedValue);
26  // No lint, we cannot use std::array here because we do not know the size at compile time
27  // NOLINTNEXTLINE(modernize-avoid-c-arrays)
28  auto data = std::make_unique<Tango::DevBoolean[]>(values.size());
29  std::copy(values.begin(), values.end(), data.get());
30 
31  // We can pass in the unique pointer here, set_write_value will copy the data
32  writeAttribute.set_write_value(data.get(), values.size());
33  } else if constexpr(std::is_same_v<Tango::DevString, TangoType>) {
34  auto values = ChimeraTK::stringToArray<std::string>(memoriedValue);
35  // No lint, we cannot use std::array here because we do not know the size at compile time
36  // NOLINTNEXTLINE(modernize-avoid-c-arrays)
37  auto data = std::make_unique<Tango::DevString[]>(values.size());
38  std::transform(values.begin(), values.end(), data.get(), [&](auto& v) {
39  //NOLINTNEXTLINE
40  return const_cast<char *>(v.c_str());
41  });
42 
43  // We can pass in the unique pointer and c_str() here, set_write_value will copy the data
44  writeAttribute.set_write_value(data.get(), values.size());
45  }
46  else {
47  auto values = ChimeraTK::stringToArray<TangoType>(memoriedValue);
48  writeAttribute.set_write_value(values.data(), values.size());
49  }
50 
51  attr->write(device, writeAttribute);
52  }

175  void TangoAdapter::write_inited_values() {
176  DEBUG_STREAM << _deviceClass <<": TangoAdapter::write_inited_values " << _write_spectrum_attr_list.size() << std::endl;
177 
178  // read spectrum values from memoried properties then write as initialised values
179  for(const auto& [attProp, index] : _write_spectrum_attr_list) {
180  DEBUG_STREAM << _deviceClass <<":name: " << attProp->name << " type:" << attProp->dataType << std::endl;
181  // get write attribute name
182  auto& write_attribute = _device->get_device_attr()->get_w_attr_by_name(attProp->name.c_str());
183  // get value of memoried property (__Memorized_<attributename>)
184  auto mem_value = getProperty<std::string>(_device, "__Memoried_" + attProp->name);
185 
186  DEBUG_STREAM << _deviceClass <<":__Memoried_" << attProp->name << " mem_value: " << mem_value << std::endl;
187 
188  if(mem_value.empty()) {
189  continue;
190  }
191 
192  Tango::Attr* base_attr = _dynamic_attribute_list[index];
193 
194  switch(attProp->dataType) {
195  case Tango::DEV_UCHAR: {
196  ::detail::writeInitialSpectrumValue<Tango::DevUChar, uint8_t>(_device, mem_value, write_attribute, base_attr);
197  break;
198  }
199 
200  case Tango::DEV_USHORT: {
201  ::detail::writeInitialSpectrumValue<Tango::DevUShort, uint16_t>(
202  _device, mem_value, write_attribute, base_attr);
203  break;
204  }
205  case Tango::DEV_ULONG: {
206  ::detail::writeInitialSpectrumValue<Tango::DevULong, uint32_t>(
207  _device, mem_value, write_attribute, base_attr);
208  break;
209  }
210  case Tango::DEV_ULONG64: {
211  ::detail::writeInitialSpectrumValue<Tango::DevULong64, uint64_t>(
212  _device, mem_value, write_attribute, base_attr);
213  break;
214  }
215 
216  case Tango::DEV_SHORT: {
217  ::detail::writeInitialSpectrumValue<Tango::DevShort, int16_t>(_device, mem_value, write_attribute, base_attr);
218  break;
219  }
220  case Tango::DEV_LONG: {
221  ::detail::writeInitialSpectrumValue<Tango::DevLong, int32_t>(_device, mem_value, write_attribute, base_attr);
222  break;
223  }
224  case Tango::DEV_LONG64: {
225  ::detail::writeInitialSpectrumValue<Tango::DevLong64, int64_t>(
226  _device, mem_value, write_attribute, base_attr);
227  break;
228  }
229  case Tango::DEV_FLOAT: {
230  ::detail::writeInitialSpectrumValue<Tango::DevFloat, float>(_device, mem_value, write_attribute, base_attr);
231  break;
232  }
233 
234  case Tango::DEV_DOUBLE: {
235  ::detail::writeInitialSpectrumValue<Tango::DevDouble, double>(_device, mem_value, write_attribute, base_attr);
236  break;
237  }
238  case Tango::DEV_BOOLEAN: {
239  ::detail::writeInitialSpectrumValue<Tango::DevBoolean, Boolean>(
240  _device, mem_value, write_attribute, base_attr);

245  _device, mem_value, write_attribute, base_attr);
246  break;
247  }
248  default:
249  ERROR_STREAM << _deviceClass <<":TangoAdapter::write_inited_values - unknown datatype: " << attProp->dataType << std::endl;
250  }
251  }

253  // once at server start.
