17 template<typename TangoType, typename AdapterType>
18 void writeInitialSpectrumValue(Tango::DeviceImpl* device, const std::string& memoriedValue,
19 Tango::WAttribute& writeAttribute, Tango::Attr& attr) {
20 // StringToArray will create a std::vector<bool> which is an optimised version of std::vector
21 // But Tango needs a plain array of bool, so we have to convert it here.
22 if constexpr(std::is_same_v<Tango::DevBoolean, TangoType>) {
23 auto values = stringToArray<TangoType>(memoriedValue);
24 // No lint, we cannot use std::array here because we do not know the size at compile time
25 // NOLINTNEXTLINE(modernize-avoid-c-arrays)
26 auto data = std::make_unique<Tango::DevBoolean[]>(values.size());
27 std::copy(values.begin(), values.end(), data.get());
28
29 // We can pass in the unique pointer here, set_write_value will copy the data
30 writeAttribute.set_write_value(data.get(), values.size());
31 }
32 else if constexpr(std::is_same_v<Tango::DevString, TangoType>) {
33 auto values = stringToArray<std::string>(memoriedValue);
34 // No lint, we cannot use std::array here because we do not know the size at compile time
35 // NOLINTNEXTLINE(modernize-avoid-c-arrays)
36 auto data = std::make_unique<Tango::DevString[]>(values.size());
37 std::transform(values.begin(), values.end(), data.get(), [&](auto& v) {
38 // NOLINTNEXTLINE
39 return const_cast<char*>(v.c_str());
40 });
41
42 // We can pass in the unique pointer and c_str() here, set_write_value will copy the data
43 // Lint: long is required by the Tango API
44 // NOLINTNEXTLINE(google-runtime-int)
45 writeAttribute.set_write_value(data.get(), static_cast<long>(values.size()));
46 }
47 else {
48 auto values = stringToArray<TangoType>(memoriedValue);
49 writeAttribute.set_write_value(values.data(), values.size());
50 }
51
52 attr.write(device, writeAttribute);
53 }
54} // namespace detail

127 void AdapterDeviceImpl::restoreMemoriedSpectra(const std::list<AttributeProperty>& writeableSpectrums) {
128 DEBUG_STREAM << AdapterDeviceClass::getClassName() << ": TangoAdapter::restoreMemoriedSpectra "
129 << writeableSpectrums.size() << std::endl;
130
131 // read spectrum values from memoried properties then write as initialised values
132 for(const auto& attProp : writeableSpectrums) {
133 DEBUG_STREAM << AdapterDeviceClass::getClassName() << ":name: " << attProp.name << " type:" << attProp.dataType
134 << std::endl;
135
136 // get write attribute name, in case it is different (READ_WITH_WRITE)
137 auto& write_attribute = get_device_attr()->get_w_attr_by_name(attProp.name.c_str());
138 std::string attName = attProp.name;
139 auto& baseAttribute = get_device_class()->get_class_attr()->get_attr(attName);
140
141 // get value of memoried property (__Memorized_<attributename>)
142 auto mem_value = getProperty<std::string>(this, "__Memoried_" + attProp.name);
143
144 DEBUG_STREAM << AdapterDeviceClass::getClassName() << ":__Memoried_" << attProp.name
145 << " mem_value: " << mem_value << std::endl;
146
147 if(mem_value.empty()) {
148 continue;
149 }
150
151 switch(attProp.dataType) {
152 case Tango::DEV_UCHAR: {
153 ::detail::writeInitialSpectrumValue<Tango::DevUChar, uint8_t>(
154 this, mem_value, write_attribute, baseAttribute);
155 break;
156 }
157
158 case Tango::DEV_ENUM: {
159 ::detail::writeInitialSpectrumValue<Tango::DevShort, int8_t>(this, mem_value, write_attribute, baseAttribute);
160 break;
161 }
162
163 case Tango::DEV_USHORT: {
164 ::detail::writeInitialSpectrumValue<Tango::DevUShort, uint16_t>(
165 this, mem_value, write_attribute, baseAttribute);
166 break;
167 }
168 case Tango::DEV_ULONG: {
169 ::detail::writeInitialSpectrumValue<Tango::DevULong, uint32_t>(
170 this, mem_value, write_attribute, baseAttribute);
171 break;
172 }
173 case Tango::DEV_ULONG64: {
174 ::detail::writeInitialSpectrumValue<Tango::DevULong64, uint64_t>(
175 this, mem_value, write_attribute, baseAttribute);
176 break;
177 }
178
179 case Tango::DEV_SHORT: {
180 ::detail::writeInitialSpectrumValue<Tango::DevShort, int16_t>(
181 this, mem_value, write_attribute, baseAttribute);
182 break;
183 }
184 case Tango::DEV_LONG: {
185 ::detail::writeInitialSpectrumValue<Tango::DevLong, int32_t>(this, mem_value, write_attribute, baseAttribute);
186 break;
187 }
188 case Tango::DEV_LONG64: {
189 ::detail::writeInitialSpectrumValue<Tango::DevLong64, int64_t>(
190 this, mem_value, write_attribute, baseAttribute);
191 break;
192 }
193 case Tango::DEV_FLOAT: {
194 ::detail::writeInitialSpectrumValue<Tango::DevFloat, float>(this, mem_value, write_attribute, baseAttribute);
195 break;
196 }
197
198 case Tango::DEV_DOUBLE: {
199 ::detail::writeInitialSpectrumValue<Tango::DevDouble, double>(
200 this, mem_value, write_attribute, baseAttribute);
201 break;
202 }
203 case Tango::DEV_BOOLEAN: {
204 ::detail::writeInitialSpectrumValue<Tango::DevBoolean, ChimeraTK::Boolean>(
205 this, mem_value, write_attribute, baseAttribute);
206 break;
207 }
208 case Tango::DEV_STRING: {
209 ::detail::writeInitialSpectrumValue<Tango::DevString, std::string>(
210 this, mem_value, write_attribute, baseAttribute);
211 break;
212 }
213 default:
214 ERROR_STREAM << AdapterDeviceClass::getClassName()
215 << ":TangoAdapter::restoreMemoriedSpectra - unknown datatype: " << attProp.dataType << std::endl;
216 }
217 }

222 void AdapterDeviceImpl::attachToClassAttributes(const std::shared_ptr<AttributeMapper::DeviceClass>& deviceClass) {
223 auto device = deviceClass->getDevice(get_name());
224 auto& adapter = TangoAdapter::getInstance();
225 auto csPvManager = adapter.getCsPvManager();
226 std::list<AttributeProperty> writeableSpectrums;
227
228 DEBUG_STREAM << "Attaching Variables to device instance " << device->name << std::endl;
229 for(auto& attr : deviceClass->attributes) {
230 auto& source = device->attributeToSource[attr.name];
231 DEBUG_STREAM << " " << source << std::endl;
232
233 auto processVariable = csPvManager->getProcessVariable(source);
234 auto pv = ChimeraTK::TransferElementAbstractor(processVariable);
235 _attributeToPvMap[attr.name] = pv;
236
237 // Properly namespace the pv in the updater so we can distinguish per device
238 adapter.getUpdater().addVariable(pv, get_name() + "/" + attr.name);
239
240 if(attr.attrDataFormat == AttrDataFormat::SPECTRUM &&
241 (attr.writeType == Tango::WRITE || attr.writeType == Tango::READ_WRITE)) {
242 writeableSpectrums.push_back(attr);
243 }
244 }
245
246 // only need to write spectrum attributes (manual writing bug in Tango)
247 // scalar attributes are memoried and initialized by Tango
248 // But only if we are running with a database. If not, there is nothing to
249 // restore anyway.
250 if(Tango::Util::_UseDb) {
251 restoreMemoriedSpectra(writeableSpectrums);
252 }
253 }
