31  void TangoUpdater::updateLoop() {
32  if(_elementsToRead.empty()) {
33  return;
34  }
35 
36  ReadAnyGroup group(_elementsToRead.begin(), _elementsToRead.end());
37 
38  // Call preRead for all TEs on additional transfer elements. waitAny() is doing this for all elements in the
39  // ReadAnyGroup. Unnecessary calls to preRead() are anyway ignored and merely pose a performance issue. For large
40  // servers, the performance impact is significant, hence we keep track of the TEs which need to be called.
41  for(auto& pair : _descriptorMap) {
42  for(const auto& elem : pair.second.additionalTransferElements) {
43  elem->preRead(ChimeraTK::TransferType::read); // TransferElement
44  }
45  }
46 
47  while(true) {
48  // Wait until any variable got an update
49  auto notification = group.waitAny(); // inside has a //handlePreRead TransferElement
50  auto updatedElement = notification.getId(); // 1 ID
51  auto& descriptor = _descriptorMap[updatedElement]; // one descriptor
52 
53  // Complete the read transfer of the process variable
54  notification.accept();
55 
56  // Call postRead for all TEs for the updated ID
57  for(const auto& elem : descriptor.additionalTransferElements) {
58  elem->postRead(ChimeraTK::TransferType::read, true);
59  }
60 
61  // FIXME: Ideally we would fill the Tango buffer for the attribute here, then attribute->read()
62  // would just send it out to CORBA
63  // FIXME: Also we would need to toggle the event here, once supported
64  DEBUG_STREAM << ::TangoAdapter::AdapterDeviceClass::getClassName()<<":TangoUpdater::updateLoop update attribute: " << descriptor.attributeID[0];
65 
66  // Call preRead for all TEs for the updated ID
67  for(const auto& elem : descriptor.additionalTransferElements) {
68  elem->preRead(ChimeraTK::TransferType::read);
69  }
70 
71  // Allow shutting down this thread...
72  boost::this_thread::interruption_point();
73  }
74  }
