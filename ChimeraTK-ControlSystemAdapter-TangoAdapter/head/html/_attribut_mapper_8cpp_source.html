13  static std::pair<std::string, std::string> splitStringAtFirstSlash(std::string& input) {
14  // find first slash
15  auto slashPosition = input.find_first_of('/');
16  if(slashPosition == 0) { // ignore leading slash
17  input = input.substr(1);
18  slashPosition = input.find_first_of('/');
19  }
20  // no slash found: return empty location name
21  if(slashPosition == std::string::npos) {
22  return std::make_pair(std::string(), input);
23  }
24  // split at first slash into location name and property name
25  auto locationName = input.substr(0, slashPosition);
26  auto propertyName = input.substr(slashPosition + 1);
27  // replace any remaining slashes in property name with dots
28  while((slashPosition = propertyName.find_first_of('/')) != std::string::npos) {
29  propertyName[slashPosition] = '.';
30  }
31  return std::make_pair(locationName, propertyName);
32  }
