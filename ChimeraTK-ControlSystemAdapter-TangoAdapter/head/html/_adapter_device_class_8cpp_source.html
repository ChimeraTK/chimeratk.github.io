30 // Apply some heuristics to derive the class name from the executable name
31 std::string AdapterDeviceClass::getClassName() {
32 std::string ourName{Tango::Util::instance()->get_ds_exec_name()};
33
34 std::regex start{"^ds_?"};
35 std::regex end{"_?ds$"};
36
37 ourName = std::regex_replace(ourName, start, "");
38 ourName = std::regex_replace(ourName, end, "");
39
40 TANGO_LOG_DEBUG << "Deriving class name from executable name: " << Tango::Util::instance()->get_ds_exec_name()
41 << " -> " << ourName << std::endl;
42
43 return ourName;
44 }

92 void AdapterDeviceClass::write_class_property() {
93 // First time, check if database used
94 if(!Tango::Util::_UseDb) {
95 return;
96 }
97
98 Tango::DbData data;
99 std::string classname = get_name();
100
101 auto ourClass = TangoAdapter::getInstance().getMapper().getClass(get_name());
102 assert(ourClass);
103
104 // Put title
105 Tango::DbDatum title("ProjectTitle");
106 std::vector<std::string> str_title{ourClass->title.value_or(ourClass->name)};
107 title << str_title;
108 data.push_back(title);
109
110 // Put Description
111 Tango::DbDatum description("Description");
112 std::vector<std::string> str_desc;
113 str_desc.emplace_back(ourClass->description.value_or("ChimeraTK-based DeviceServer"));
114 description << str_desc;
115 data.push_back(description);
116
117 // Put inheritance
118 Tango::DbDatum inher_datum("InheritedFrom");
119 std::vector<std::string> inheritance;
120 inheritance.emplace_back("TANGO_BASE_CLASS");
121 inher_datum << inheritance;
122 data.push_back(inher_datum);
123
124 // Call database and and values
125 get_db_class()->put_property(data);
126 }

130 void AdapterDeviceClass::device_factory(const Tango::DevVarStringArray* devlist_ptr) {
131 // Add your own code
132
133 auto& adapter = TangoAdapter::getInstance();
134 auto& mapper = adapter.getMapper();
135
136 auto deviceClass = mapper.getClass(get_name());
137
138 // Create devices and add it into the device list
139 for(unsigned int i = 0; i < devlist_ptr->length(); i++) {
140 const auto* deviceName = (*devlist_ptr)[i].in();
141
142 if(!deviceClass->hasDevice(deviceName) &&
143 !deviceClass->hasDevice(TangoAdapter::PLAIN_IMPORT_DUMMY_DEVICE.data())) {
144 std::cerr << "Device " << deviceName << "not known in attribute mapper. Skipping." << std::endl;
145 continue;
146 }
147
148 auto device = std::make_unique<AdapterDeviceImpl>(this, deviceName);
149 device->init_device();
150 if(!deviceClass->hasDevice(deviceName) &&
151 !deviceClass->hasDevice(TangoAdapter::PLAIN_IMPORT_DUMMY_DEVICE.data())) {
152 // See if we have the "generic" device"
153 DEV_ERROR_STREAM(device) << "Device " << deviceName << "not known in attribute mapper. Expect issues"
154 << std::endl;
155 device->set_state(Tango::FAULT);
156 device->set_status("Device was not found in mapping file, no variables could be mapped.");
157 }
158 else {
159 // Move the generic device we have to this device.
160 if(!deviceClass->hasDevice(deviceName)) {
161 auto genericDevice = deviceClass->devicesInDeviceClass[TangoAdapter::PLAIN_IMPORT_DUMMY_DEVICE.data()];
162 deviceClass->devicesInDeviceClass.erase(TangoAdapter::PLAIN_IMPORT_DUMMY_DEVICE.data());
163 genericDevice->name = deviceName;
164 deviceClass->devicesInDeviceClass[deviceName] = genericDevice;
165 }
166 device->attachToClassAttributes(deviceClass);
167 }
168
169 // Check before if database used.
170 if(Tango::Util::_UseDb && !Tango::Util::_FileDb) {
171 export_device(device.get());
172 }
173 else {
174 export_device(device.get(), deviceName);
175 }
176
177 // Hand over device pointer to Tango
178 device_list.push_back(device.release());
179 }
180
181 // Manage dynamic attributes if any
182 erase_dynamic_attributes(get_class_attr()->get_attr_list());
183
184 for(auto* dev : device_list) {
185 // Only set the device state ok ON if it is still in the default constructed
186 // value
187 if(dev->get_state() == Tango::UNKNOWN) {
188 dev->set_state(Tango::ON);
189 dev->set_status("Application is running.");
190 }
191 }
192 }

219 void AdapterDeviceClass::create_static_attribute_list(std::vector<Tango::Attr*>& att_list) {
220 for(auto* attr : att_list) {
221 auto att_name = attr->get_name();
222 std::transform(att_name.begin(), att_name.end(), att_name.begin(), ::tolower);
223 defaultAttList.push_back(att_name);
224 }
225
226 TANGO_LOG_DEBUG << defaultAttList.size() << " attributes in default list" << std::endl;
227 }

230
231 void AdapterDeviceClass::erase_dynamic_attributes(std::vector<Tango::Attr*>& att_list) {
232 for(auto* dev_impl : device_list) {
233 auto* dev = dynamic_cast<AdapterDeviceImpl*>(dev_impl);
234 assert(dev != nullptr);
235
236 std::vector<Tango::Attribute*>& dev_att_list = dev->get_device_attr()->get_attribute_list();
237 std::vector<Tango::Attribute*>::iterator ite_att;
238 for(ite_att = dev_att_list.begin(); ite_att != dev_att_list.end(); ++ite_att) {
239 std::string att_name((*ite_att)->get_name_lower());
240 if((att_name == "state") || (att_name == "status")) {
241 continue;
242 }
243 auto ite_str = find(defaultAttList.begin(), defaultAttList.end(), att_name);
244 if(ite_str == defaultAttList.end()) {
245 TANGO_LOG_DEBUG << att_name << " is a UNWANTED dynamic attribute for device " << dev->name() << std::endl;
246 Tango::Attribute& att = dev->get_device_attr()->get_attr_by_name(att_name.c_str());
247 dev->remove_attribute(att_list[att.get_attr_idx()], true, false);
248 --ite_att;
249 }
250 }
251 }
