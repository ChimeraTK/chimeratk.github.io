34void ThreadedTangoServer::start() {
35 std::mutex in_mtx;
36 std::unique_lock<std::mutex> in(in_mtx);
37 std::condition_variable cv;
38 bool threadRunning{false};
39
40 tangoServerThread = std::thread([&]() {
41 auto& adapter = TangoAdapter::TangoAdapter::getInstance();
42 argv.emplace_back(testName + "_ds");
43 argv.emplace_back("Test" + testName);
44 if(verbose || std::getenv("TANGO_TESTS_VERBOSE") != nullptr) {
45 argv.emplace_back("-v5");
46 }
47 deviceString = std::string("tango/test/") + testName;
48
49 if(offlineDatabase.empty()) {
50 argv.emplace_back("-nodb");
51 argv.emplace_back("-dlist");
52 }
53 else {
54 argv.emplace_back("-file=" + offlineDatabase);
55 }
56 argv.emplace_back(deviceString);
57 argv.emplace_back("-ORBendPoint");
58 argv.emplace_back("giop:tcp::" + port());
59
60 std::vector<const char*> args;
61 args.resize(argv.size());
62 std::transform(argv.begin(), argv.end(), args.begin(), [&](auto& s) { return s.c_str(); });
63
64 auto postInitHook = [&, this]() {
65 tg = Tango::Util::instance();
66
67 auto callback = []() -> bool {
68 auto shutdown = ThreadedTangoServer::shutdownRequested.load();
69 if(shutdown) {
70 return true;
71 }
72
73 std::this_thread::sleep_for(std::chrono::milliseconds(100));
74 return false;
75 };
76
77 tg->server_set_event_loop(callback);
78 {
79 std::lock_guard<std::mutex> lg(in_mtx);
80 threadRunning = true;
81 cv.notify_one();
82 }
83 };
84
85 // Need to pass it down to something that usually takes argc, argv directly from main
86 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-const-cast)
87 adapter.run(int(args.size()), const_cast<char**>(args.data()), std::make_optional(postInitHook));
88 });
89 cv.wait(in, [&] { return threadRunning; });
90}

101std::string ThreadedTangoServer::port() {
102 static std::string corbaPort;
103 if(corbaPort.empty()) {
104 std::random_device rd;
105 std::uniform_int_distribution<int> dist(10000, 50000);
106 corbaPort = std::to_string(dist(rd));
107 }
108
109 return corbaPort;
110}

142ThreadedTangoServer& ThreadedTangoServer::setOfflineDatabase(const std::string& basePath) {
143 offlineDatabase = basePath + ".db";
144 if(createOfflineDatabase) {
145 auto sourceTemplate = basePath + "_template.db";

147 std::filesystem::copy_file(sourceTemplate, offlineDatabase, std::filesystem::copy_options::overwrite_existing);

151 throw;
152 }
153 }
154

182void TangoTestFixtureImpl::startup() {
183 theServer.start();
184
185 using clock = std::chrono::steady_clock;
186 auto start = clock::now();
187
188 auto url = theServer.getClientUrl();
189 while(true) {
190 try {
191 std::cout << "Trying to connect to Tango server " << theServer.getClientUrl() << std::endl;
192 proxy = std::make_unique<Tango::DeviceProxy>(url);
193 proxy->read_attribute("State");
194 break;
195 }
196 catch(CORBA::Exception& e) {
197 if(auto now = clock::now(); now > start + std::chrono::seconds(10)) {
198 Tango::Except::print_exception(e);
199 assert(false);
200 }
201 }
202 }
203
204 // Cannot call any of the BOOST_ tests here, otherwise it will mark the setup as failed, regardless of the test outcome
205 assert(proxy->is_connected());
206
207 start = clock::now();
208 while(proxy->state() != Tango::ON) {
209 std::this_thread::sleep_for(std::chrono::milliseconds(250));
210 if(std::chrono::duration_cast<std::chrono::seconds>(clock::now() - start).count() > 5) {
211 std::cerr << "Timeout while waiting for Tango server to go to state \"ON\"" << std::endl;
212 break;
213 }
214 }
215
216 // Cannot call any of the BOOST_ tests here, otherwise it will mark the setup as failed, regardless of the test outcome
217 assert(proxy->state() == Tango::ON);
218
219 // Wait for the server to become ON before getting the application from the factory
220 // so that the server is definitely the one that creates the application, not the test
221 theApp = dynamic_cast<ExtendedReferenceTestApplication*>(
222 &ChimeraTK::ApplicationFactory<ExtendedReferenceTestApplication>::getApplicationInstance());
223 assert(theApp != nullptr);
224 if(manualLoopControl) {
225 theApp->initialiseManualLoopControl();
226 }
227}
