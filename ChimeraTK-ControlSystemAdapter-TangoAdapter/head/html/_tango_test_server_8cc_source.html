31 void ThreadedTangoServer::start() {
32  std::mutex in_mtx;
33  std::unique_lock<std::mutex> in(in_mtx);
34  std::condition_variable cv;
35  bool threadRunning{false};
36 
37  tangoServerThread = std::thread([&]() {
38  try {
39  argv.emplace_back(testName + "_ds");
40  argv.emplace_back("Test" + testName);
41  if(verbose || std::getenv("TANGO_TESTS_VERBOSE") != nullptr) {
42  argv.emplace_back("-v4");
43  }
44  deviceString = std::string("tango/test/") + testName;
45 
46  if(offlineDatabase.empty()) {
47  argv.emplace_back("-nodb");
48  argv.emplace_back("-dlist");
49  }
50  else {
51  argv.emplace_back("-file=" + offlineDatabase);
52  }
53  argv.emplace_back(deviceString);
54  argv.emplace_back("-ORBendPoint");
55  argv.emplace_back("giop:tcp::" + port());
56 
57  std::vector<const char*> args;
58  args.resize(argv.size());
59  std::transform(argv.begin(), argv.end(), args.begin(), [&](auto& s) { return s.c_str(); });
60  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-const-cast)
61  tg = Tango::Util::init(int(args.size()), const_cast<char**>(args.data()));
62  tg->server_init(false);
63  cv.notify_all();
64  auto callback = []() -> bool {
65  auto shutdown = ThreadedTangoServer::shutdownRequested.load();
66  if(shutdown) {
67  return true;
68  }
69 
70  std::this_thread::sleep_for(std::chrono::milliseconds(100));
71  return false;
72  };
73  tg->server_set_event_loop(callback);
74  {
75  std::lock_guard lg(in_mtx);
76  threadRunning = true;
77  cv.notify_one();
78  }
79 
80  tg->server_run();
81  }
82  catch(CORBA::Exception& e) {
83  Tango::Except::print_exception(e);
84  std::rethrow_exception(std::current_exception());
85  }
86  });
87  cv.wait(in, [&] { return threadRunning;});
88 }

179 void TangoTestFixtureImpl::startup() {
180  theServer.start();
181 
182  using clock = std::chrono::steady_clock;
183  auto start = clock::now();
184 
185  auto url = theServer.getClientUrl();
186  try {
187  std::cout << "Trying to connect to Tango server " << theServer.getClientUrl() << std::endl;
188  proxy = std::make_unique<Tango::DeviceProxy>(url);
189  }
190  catch(CORBA::Exception& e) {
191  Tango::Except::print_exception(e);
192  throw;
193  }
194 
195  while(!proxy->is_connected()) {
196  std::this_thread::sleep_for(std::chrono::milliseconds(250));
197  if(std::chrono::duration_cast<std::chrono::seconds>(clock::now() - start).count() > 5) {
198  std::cerr << "Timeout while waiting for Tango server to become available" << std::endl;
199  break;
200  }
201  proxy->connect(url);
202  }
203 
204  // Cannot call any of the BOOST_ tests here, otherwise it will mark the setup as failed, regardless of the test outcome
205  assert(proxy->is_connected());
206 
207  start = clock::now();
208  while(proxy->state() != Tango::ON) {
209  std::this_thread::sleep_for(std::chrono::milliseconds(250));
210  if(std::chrono::duration_cast<std::chrono::seconds>(clock::now() - start).count() > 5) {
211  std::cerr << "Timeout while waiting for Tango server to go to state \"ON\"" << std::endl;
212  break;
213  }
214  }
215 
216  // Cannot call any of the BOOST_ tests here, otherwise it will mark the setup as failed, regardless of the test outcome
217  assert(proxy->state() == Tango::ON);
218 
219  // Wait for the server to become ON before getting the application from the factory
220  // so that the server is definitely the one that creates the application, not the test
221  theApp = dynamic_cast<ExtendedReferenceTestApplication*>(
222  &ChimeraTK::ApplicationFactory<ExtendedReferenceTestApplication>::getApplicationInstance());
223  assert(theApp != nullptr);
224  if(manualLoopControl) {
225  theApp->initialiseManualLoopControl();
226  }
227 }
