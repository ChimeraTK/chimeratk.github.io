32 template<typename TangoType, typename AdapterType>
33 ScalarAttribTempl<TangoType, AdapterType>::ScalarAttribTempl(AttributeProperty& attProperty)
34 : Tango::Attr(attProperty.name.c_str(), attProperty.getDataType(), attProperty.writeType) {
35 // memory the written value and write at initialization
36 if(attProperty.writeType == Tango::READ_WRITE || attProperty.writeType == Tango::WRITE) {
37 set_memorized();
38 set_memorized_init(true);
39 }
40
41 Tango::UserDefaultAttrProp att_prop;
42
43 // The c_str() are fine, tango internally creates a std::string of them.
44 att_prop.set_label(attProperty.name.c_str());
45
46 att_prop.set_description(attProperty.desc.c_str());
47
48 if(!(attProperty.unit.empty())) {
49 att_prop.set_unit(attProperty.unit.c_str());
50 }
51
52 // Since Tango does not support int8 natively and we have to resort to SHORT,
53 // limit the values to the int8 limits
54 if constexpr(std::is_same_v<AdapterType, int8_t>) {
55 att_prop.set_min_value(std::to_string(std::numeric_limits<int8_t>::min()).c_str());
56 att_prop.set_max_value(std::to_string(std::numeric_limits<int8_t>::max()).c_str());
57 }
58 set_default_properties(att_prop);
59 }

63 template<typename TangoType, typename AdapterType>
64 void ScalarAttribTempl<TangoType, AdapterType>::read(Tango::DeviceImpl* dev, Tango::Attribute& att) {
65 auto* adapterDevice = dynamic_cast<TangoAdapter::AdapterDeviceImpl*>(dev);
66 assert(adapterDevice != nullptr);
67
68 auto pv = adapterDevice->getPvForAttribute(att.get_name());
69 assert(pv.getValueType() == typeid(AdapterType));
70
71 auto processScalar = boost::reinterpret_pointer_cast<GenericAccessor>(pv.getHighLevelImplElement());
72
73 if(!processScalar) {
74 DEV_WARN_STREAM(dev) << "pv type mismatch (expected: " << boost::core::demangle(pv.getValueType().name())
75 << ", got :" << boost::core::demangle(typeid(AdapterType).name()) << ")" << std::endl;
76 att.set_quality(Tango::AttrQuality::ATTR_INVALID);
77 return;
78 }
79
80 if constexpr(std::is_same_v<TangoType, Tango::DevString>) {
81 auto value = std::make_unique<Tango::DevString>();
82 *(value.get()) = Tango::string_dup(processScalar->accessData(0).c_str());
83 att.set_value(value.release(), 1, 0, true);
84 }
85 else if constexpr(std::is_same_v<TangoType, Tango::DevBoolean>) {
86 auto pvAsBool = boost::reinterpret_pointer_cast<BooleanAccessor>(processScalar);
87 assert(pvAsBool != nullptr);
88
89 auto value = std::make_unique<Tango::DevBoolean>();
90 if constexpr(std::is_same_v<AdapterType, ChimeraTK::Void>) {
91 *(value.get()) = true;
92 }
93 else {
94 *(value.get()) = pvAsBool->accessData(0);
95 }
96 att.set_value(value.release(), 1, 0, true);
97 }
98 else {
99 // Lint: Not even sure why this triggers here.
100 // NOLINTNEXTLINE(modernize-avoid-c-arrays)
101 auto value = std::make_unique<TangoType>(processScalar->accessData(0));
102 // Lint: for the signed char adapter type we need to assign it to short, since Tango does not
103 // have a signed char type. Silence the warning, we are not dealing with characters here
104 // NOLINTNEXTLINE(bugprone-signed-char-misue)
105 att.set_value(value.release(), 1, 0, true);
106 }
107
108 if(processScalar->dataValidity() != ChimeraTK::DataValidity::ok) {
109 att.set_quality(Tango::AttrQuality::ATTR_INVALID);
110 }
111 else {
112 att.set_quality(Tango::AttrQuality::ATTR_VALID);
113 }
114 }

118 template<typename TangoType, typename AdapterType>
119 void ScalarAttribTempl<TangoType, AdapterType>::write(Tango::DeviceImpl* dev, Tango::WAttribute& att) {
120 auto* adapterDevice = dynamic_cast<TangoAdapter::AdapterDeviceImpl*>(dev);
121 assert(adapterDevice != nullptr);
122
123 auto pv = adapterDevice->getPvForAttribute(att.get_name());
124 assert(pv.getValueType() == typeid(AdapterType));
125
126 auto processScalar = boost::reinterpret_pointer_cast<GenericAccessor>(pv.getHighLevelImplElement());
127
128 if(!processScalar) {
129 DEV_WARN_STREAM(dev) << "pv type mismatch (expected: " << boost::core::demangle(pv.getValueType().name())
130 << ", got :" << boost::core::demangle(typeid(AdapterType).name()) << ")" << std::endl;
131 att.set_quality(Tango::AttrQuality::ATTR_INVALID);
132 return;
133 }
134
135 if constexpr(std::is_same_v<AdapterType, std::string>) {
136 Tango::DevString st_value;
137 att.get_write_value(st_value);
138 processScalar->accessData(0) = std::string(st_value);
139 }
140 else if constexpr(std::is_same_v<AdapterType, ChimeraTK::Boolean>) {
141 auto pvAsBool = boost::reinterpret_pointer_cast<BooleanAccessor>(processScalar);
142 Tango::DevBoolean b_value;
143 att.get_write_value(b_value);
144 pvAsBool->accessData(0) = b_value;
145 }
146 else if constexpr(std::is_same_v<AdapterType, ChimeraTK::Void>) {
147 // do nothing with the data, just write the accessor
148 }
149 else {
150 TangoType value;
151 att.get_write_value(value);
152 processScalar->accessData(0) = value;
153 }
154
155 if(att.get_quality() == Tango::AttrQuality::ATTR_INVALID) {
156 processScalar->setDataValidity(ChimeraTK::DataValidity::faulty);
157 }
158 else {
159 processScalar->setDataValidity(ChimeraTK::DataValidity::ok);
160 }
161 processScalar->write();
162 }
