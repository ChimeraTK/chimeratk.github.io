171 template<typename T>
172 inline bool TangoTestFixtureImpl::checkWithTimeout(std::string readName, T referenceValues) {
173  const auto TIMEOUT = std::chrono::seconds(5);
174  using clock = std::chrono::steady_clock;
175  auto now = clock::now();
176 
177  T readValues;
178  do {
179  readValues = {};
180  std::this_thread::sleep_for(std::chrono::milliseconds(100));
181  proxy->read_attribute(readName) >> readValues;
182  } while (referenceValues != readValues && (clock::now() - now) < TIMEOUT);
183 
184  return referenceValues == readValues;
185 }

189 template<typename TargetType, typename T>
190 inline TargetType TangoTestFixtureImpl::generateDefaultValueForType() {
191  TargetType expectedValue;
192  if constexpr(std::is_same_v<std::string, TargetType>) {
193  expectedValue = std::to_string(42.0);
194  } else
195  if constexpr(std::is_same_v<Tango::DevBoolean, TargetType>) {
196  expectedValue = true;
197  } else {
198  expectedValue = TargetType(sizeof(T));
199  if constexpr(std::is_floating_point_v<TargetType>) {
200  expectedValue = TargetType(1.) / expectedValue;
201  } else {
202  expectedValue *= (std::is_signed_v<T> ? -1 : 1);
203  }
204  }
205  return expectedValue;
206 }
