187template<typename T>
188inline bool TangoTestFixtureImpl::checkWithTimeout(std::string readName, T referenceValues) {
189 const auto TIMEOUT = std::chrono::seconds(5);
190 using clock = std::chrono::steady_clock;
191 auto now = clock::now();
192
193 T readValues;
194 do {
195 readValues = {};
196 std::this_thread::sleep_for(std::chrono::milliseconds(100));
197 try {
198 proxy->read_attribute(readName) >> readValues;
199 }
200 catch(CORBA::Exception& ex) {
201 Tango::Except::print_exception(ex);
202 std::rethrow_exception(std::current_exception());
203 }
204
205 } while(referenceValues != readValues && (clock::now() - now) < TIMEOUT);
206
207 return referenceValues == readValues;
208}

212template<typename TargetType, typename T>
213inline TargetType TangoTestFixtureImpl::generateDefaultValueForType() {
214 TargetType expectedValue;
215 if constexpr(std::is_same_v<std::string, TargetType>) {
216 expectedValue = std::to_string(42.0);
217 }
218 else if constexpr(std::is_same_v<Tango::DevBoolean, TargetType>) {
219 expectedValue = true;
220 }
221 else {
222 expectedValue = TargetType(sizeof(T));
223 if constexpr(std::is_floating_point_v<TargetType>) {
224 expectedValue = TargetType(1.) / expectedValue;
225 }
226 else {
227 expectedValue *= (std::is_signed_v<T> ? -1 : 1);
228 }
229 }
230 return expectedValue;
231}
