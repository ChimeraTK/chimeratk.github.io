109 Tango::CmdArgType AttributeProperty::getDataType() const {
110 switch(dataType) {
111 case Tango::DEV_ENUM:
112 // FIXME: HACK. There is no CHAR type in Tango, we use ENUM and map to SHORT
113 return Tango::DEV_SHORT;
114 case Tango::DEV_VOID:
115 // DEV_VOID is not a type. It is usually used to signify that a Command does not have a parameter
116 // or a return value. Hence we need to map it to something else, let's choose boolean for that
117 return Tango::DEV_BOOLEAN;
118 default:
119 return dataType;
120 }
121 }
