101std::string ThreadedDoocsServer::bpn() {
102 std::lock_guard<std::mutex> lk(_mx_serverInfo);
103 if(_bpn.empty()) {
104 std::random_device rd;
105 std::uniform_int_distribution<int> dist(2000, 6500);
106 // range is 20000 to 65000 in steps of 10 (BPNs allocate a block of port numbers used for ZeroMQ)
107 // avoid the "usual" range of 6000-10000, because there might be a normally configured server running on the
108 // same machine using a BPN in that range...
109 _bpn = std::to_string(dist(rd)) + "0";
110 }
111
112 return _bpn;
113}
