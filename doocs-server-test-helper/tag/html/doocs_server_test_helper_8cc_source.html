26extern "C" int sigwait(__const sigset_t* __restrict set, int* __restrict sig) {
27 // call original-equivalent version if SIGUSR1 is not in the set
28 if(!sigismember(set, SIGUSR1)) {
29 // first wait on the specified signals
30 siginfo_t siginfo;
31 int iret = sigwaitinfo(set, &siginfo);
32 *sig = siginfo.si_signo;
33 // if (in the mean time) doNotProcessSignalsInDoocs has been set by the test
34 // code, wait on an empty set (-> forever)
35 if(DoocsServerTestHelper::doNotProcessSignalsInDoocs) {
36 sigset_t emptySet;
37 sigemptyset(&emptySet);
38 iret = sigwaitinfo(&emptySet, &siginfo);
39 *sig = siginfo.si_signo;
40 }
41 return iret;
42 }
43
44 // SIGUSR1 is in the set: wait until sigusr1 reqested via mutex
45 static thread_local std::unique_lock<std::mutex> sigusr1_lock{DoocsServerTestHelper::sigusr1_mutex};
46 do {
47 sigusr1_lock.unlock();
48 usleep(1);
49 sigusr1_lock.lock();
50 } while(!DoocsServerTestHelper::allowSigusr1);
51 DoocsServerTestHelper::allowSigusr1 = false;
52
53 // return a SIGUSR1
54 *sig = SIGUSR1;
55 return 0;
56}

145void DoocsServerTestHelper::doocsSetSpectrum(const std::string& name, const std::vector<float>& value) {
146 EqAdr ad;
147 EqData ed, res;
148 // fill spectrum data structure
149 SPECTRUM spectrum;
150 spectrum.comment.comment_len = 0;
151 spectrum.comment.comment_val = nullptr;
152 spectrum.tm = 0;
153 spectrum.s_start = 0;
154 spectrum.s_inc = 1.;
155 spectrum.status = 0;
156 spectrum.d_spect_array.d_spect_array_len = value.size();
157 spectrum.d_spect_array.d_spect_array_val = (float*)value.data(); // casts const-ness away, but will not be modified
158 // (hopefully)
159 ed.set(&spectrum);
160 // obtain location pointer
161 ad.adr(name);
162 EqFct* p = eq_get(&ad);
163 ASSERT(p != nullptr, std::string("Could not get location for property ") + name);
164 // set spectrum
165 size_t retry = 1000;
166 do {
167 p->lock();
168 p->set(&ad, &ed, &res);
169 p->unlock();
170 } while(res.error() != 0 && --retry > 0);
171 // check for error
172 ASSERT(res.error() == 0, std::string("Error writing spectrum property ") + name);
173}
