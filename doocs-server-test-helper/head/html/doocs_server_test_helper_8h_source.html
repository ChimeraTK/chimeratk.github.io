148 template<typename TYPE>
149 void DoocsServerTestHelper::doocsSet(const std::string& name, TYPE value) {
150  EqAdr ad;
151  EqData ed, res;
152  // obtain location pointer
153  ad.adr(name);
154  EqFct* p = eq_get(&ad);
155  ASSERT(p != nullptr, std::string("Could not get location for property ") + name);
156  // set value
157  ed.set(value);
158  size_t retry = 1000;
159  do {
160  p->lock();
161  p->set(&ad, &ed, &res);
162  p->unlock();
163  } while(res.error() != 0 && --retry > 0);
164  // check for error
165  ASSERT(res.error() == 0, std::string("Error writing property ") + name + ": " + res.get_string());
166 }

250 template<typename TYPE>
251 std::vector<TYPE> DoocsServerTestHelper::doocsGetArray(const std::string& name) {
252  EqAdr ad;
253  EqData ed, res;
254  // obtain location pointer
255  ad.adr(name);
256  EqFct* p = eq_get(&ad);
257  ASSERT(p != nullptr, std::string("Could not get location for property ") + name);
258  // for D_Spectrum: set IIII structure to obtain always the latest buffer
259  IIII iiii;
260  iiii.i1_data = -1;
261  iiii.i2_data = -1;
262  iiii.i3_data = -1;
263  iiii.i4_data = -1;
264  ed.set(&iiii);
265  // obtain values
266  size_t retry = 1000;
267  do {
268  p->lock();
269 
270  // Try to get the data with parameters for a spectrum
271  p->get(&ad, &ed, &res);
272  if(res.error() == eq_errors::not_implemeted) {
273  // if that fails, assume we have a plain array, and just not pass the second parameter at all
274  p->get(&ad, nullptr, &res);
275  }
276  p->unlock();
277  } while(res.error() != 0 && --retry > 0);
278  // check for errors
279  ASSERT(res.error() == 0, std::string("Error reading property ") + name);
280 
281  // copy to vector and return it
282  std::vector<TYPE> val;
283  if(std::is_integral<TYPE>()) {
284  if(res.type() != DATA_A_LONG) {
285  for(int i = 0; i < res.length(); i++) {
286  val.push_back(res.get_int(i));
287  }
288  }
289  else {
290  for(int i = 0; i < res.length(); i++) {
291  val.push_back(res.get_long(i));
292  }
293  }
294  }
295  else if(std::is_floating_point<TYPE>()) {
296  for(int i = 0; i < res.length(); i++) {
297  val.push_back(res.get_float(i));
298  }
299  }
300  else {
301  ASSERT(false, "Wrong type passed as tempalate argument.");
302  }
303  return val;
304 }
