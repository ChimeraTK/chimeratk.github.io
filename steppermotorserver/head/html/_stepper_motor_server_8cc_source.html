37StepperMotorServer::StepperMotorServer() : Application("steppermotorserver") {
38 std::cout << "****************************************************************" << std::endl;
39 std::cout << "*** ChimeraTK Stepper Motor Server version " << AppVersion::major << "." << AppVersion::minor << "."
40 << AppVersion::patch << std::endl;
41
42 // Get and validate motor configuration
43 auto nMotors = config.get<uint32_t>("Motors/nMotors");
44 auto motorType = config.get<std::vector<std::string>>("Motors/motorType");
45 auto motorDriverCardDeviceNames = config.get<std::vector<std::string>>("Motors/motorDriverDeviceName");
46 auto motorDriverCardBspNames = config.get<std::vector<std::string>>("Motors/motorDriverCardName");
47 auto motorDriverCardModuleNames = config.get<std::vector<std::string>>("Motors/motorDriverModuleName");
48 auto motorDriverCardIds = config.get<std::vector<uint32_t>>("Motors/motorDriverId");
49 auto motorDriverCardConfigFiles = config.get<std::vector<std::string>>("Motors/motorDriverConfigFile");
50 auto encoderUnitToStepsRatios = config.get<std::vector<double>>("Motors/encoderUnitToStepsRatio");
51
52 auto userUnitToStepsRatios = config.get<std::vector<float>>("Motors/userUnitToStepsRatio");
53 auto userPositionUnits = config.get<std::vector<std::string>>("Motors/userPositionUnit");
54 auto useDummyMotor = config.get<std::vector<int>>("Motors/dummy");
55
56 if(motorDriverCardDeviceNames.size() != nMotors || motorDriverCardModuleNames.size() != nMotors ||
57 motorDriverCardIds.size() != nMotors || motorDriverCardConfigFiles.size() != nMotors) {
58 std::stringstream msg;
59 msg << "A mismatch of dimensions in the MotorDriver configuration provided by ";
60 msg << SERVER_CONFIG_FILE;
61 terminateServer(msg.str());
62 }
63
64 // Set up modules for each motor
65 std::unordered_set<std::string> initializedMotorDriverHW;
66 for(size_t i = 0; i < nMotors; ++i) {
67 mtca4u::MotorDriverCardFactory::instance().setDummyMode(useDummyMotor[i] == 1);
68
69 // Motor configuration
70 ctkmot::StepperMotorParameters motorParameters;
71 motorParameters.deviceName = motorDriverCardDeviceNames[i];
72 motorParameters.moduleName = motorDriverCardModuleNames[i];
73 motorParameters.driverId = motorDriverCardIds[i];
74 motorParameters.configFileName = motorDriverCardConfigFiles[i];
75 motorParameters.motorUnitsConverter =
76 std::make_shared<ctkmot::utility::ScalingMotorStepsConverter>(userUnitToStepsRatios[i]);
77 motorParameters.encoderUnitsConverter =
78 std::make_shared<ctkmot::utility::ScalingEncoderStepsConverter>(encoderUnitToStepsRatios[i]);
79
80 // Create a motor driver according to the motor type
81 if(motorType[i] == basicLinearMotorType) {
82 motorParameters.motorType = ctkmot::StepperMotorType::BASIC;
83 }
84 else if(motorType[i] == linearMotorWithReferenceType) {
85 motorParameters.motorType = ctkmot::StepperMotorType::LINEAR;
86 }
87 else {
88 std::stringstream msg;
89 msg << "Unknown motor type \"" << motorType[i]
90 << "\" requested in MotorDriver configuration provided by: " << std::endl
91 << SERVER_CONFIG_FILE << "." << std::endl;
92
93 terminateServer(msg.str());
94 }
95
96 auto initCmd = std::string{"./initMotorDriverHW.py "} + std::string{DMAP_FILE_NAME} + " " +
97 motorDriverCardDeviceNames[i] + " " + motorDriverCardBspNames[i];
98 motorDriver.emplace_back(
99 std::make_shared<ctk::MotorDriver::StepperMotorModule>(this, "Motor" + std::to_string(i + 1),
100 "Driver of motor " + std::to_string(i + 1), motorParameters, "/Trigger/voidTick", initCmd));
101
102 std::cout << "*** Created motor driver " << motorDriverCardIds[i] << " of card " << motorDriverCardModuleNames[i]
103 << " on device " << motorDriverCardDeviceNames[i]
104 << ". Configuration file: " << motorDriverCardConfigFiles[i]
105 << " Motor is dummy: " << std::to_string(useDummyMotor[i] == 1) << std::endl;
106
107 if(useDummyMotor[i] == 1) {
108 motorDummy.emplace_back(
109 this, "Motor" + std::to_string(i + 1), "Dummy for motor " + std::to_string(i), motorParameters);
110 }
111 }
112}
