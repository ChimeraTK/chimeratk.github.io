57 static DoocsBackendRegisterCatalogue fetchCatalogue(
58  std::string serverAddress, std::string cacheFile, std::future<void> cancelFlag) {
59  auto result = CatalogueFetcher(serverAddress, std::move(cancelFlag)).fetch();
60  auto catalogue = std::move(result.first);
61  auto isCatalogueComplete = result.second;
62  bool isCacheFileNameSpecified = not cacheFile.empty();
63 
64  if(isCatalogueComplete && isCacheFileNameSpecified) {
65  Cache::saveCatalogue(catalogue, cacheFile);
66  }
67  return catalogue;

84  DoocsBackend::DoocsBackend(
85  const std::string& serverAddress, const std::string& cacheFile, const std::string& updateCache)
86  : _serverAddress(serverAddress), _cacheFile(cacheFile) {
87  if(cacheFileExists() && isCachingEnabled()) {
88  // provide catalogue immediately from cache
89  catalogue = Cache::readCatalogue(_cacheFile);
90 
91  // update cache file in the background
92  if(updateCache == "1") {
93  std::thread(fetchCatalogue, serverAddress, cacheFile, _cancelFlag.get_future()).detach();
94  }
95  }
96  else {
97  // fill catalogue in the background (and save to cache if enabled)
98  _catalogueFuture =
99  std::async(std::launch::async, fetchCatalogue, serverAddress, cacheFile, _cancelFlag.get_future());
100  }
101 
102  // Reduce ZeroMQ timeout so inconsistencies get corrected more quickly. The downside is that DOOCS will do more
103  // frequent RPC polls on rarely changing ZeroMQ variables (every 10 seconds instead of every 4 minutes), but this
104  // is acceptable as it is still slow enough.
105  doocs::zmq_set_subscription_timeout(10);
106 
107  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(getRegisterAccessor_impl);
108  }

140  boost::shared_ptr<DeviceBackend> DoocsBackend::createInstance(
141  std::string address, std::map<std::string, std::string> parameters) {
142  // if address is empty, build it from parameters (for compatibility with SDM)
143  if(address.empty()) {
144  RegisterPath serverAddress;
145  serverAddress /= parameters["facility"];
146  serverAddress /= parameters["device"];
147  serverAddress /= parameters["location"];
148  address = std::string(serverAddress).substr(1);
149  }
150  std::string cacheFile{};
151  std::string updateCache{"0"};
152  try {
153  cacheFile = parameters.at("cacheFile");
154  updateCache = parameters.at("updateCache");
155  }
156  catch(std::out_of_range&) {
157  // empty cacheFile string => no caching
158  // empty updateCache string => no cache update
159  }
160 
161  // create and return the backend
162  return boost::shared_ptr<DeviceBackend>(new DoocsBackend(address, cacheFile, updateCache));
163  }

167  void DoocsBackend::open() {
168  std::unique_lock<std::mutex> lk(_mxRecovery);
169  if(lastFailedAddress != "") {
170  // open() is called after a runtime_error: check if device is recovered.
171  EqAdr ea;
172  ea.adr(lastFailedAddress);
173  EqCall eq;
174  EqData src, dst;
175  int rc = eq.get(&ea, &src, &dst);
176  // if again error received, throw exception
177  if(rc && isCommunicationError(dst.error())) {
178  lk.unlock();
179  setException();
180  throw ChimeraTK::runtime_error(std::string("Cannot read from DOOCS property: ") + dst.get_string());
181  }
182  lastFailedAddress = "";
183  }
184  _opened = true;
185  _isFunctional = true;
186  _startVersion = {};
187  }

257  template<typename UserType>
258  boost::shared_ptr<NDRegisterAccessor<UserType>> DoocsBackend::getRegisterAccessor_impl(
259  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
260  boost::shared_ptr<NDRegisterAccessor<UserType>> p;
261  std::string path = _serverAddress + registerPathName;
262 
263  // check for additional hierarchy level, which indicates an access to a field of a complex property data type
264  bool hasExtraLevel = false;
265  if(!boost::starts_with(path, "doocs://") && !boost::starts_with(path, "epics://")) {
266  size_t nSlashes = std::count(path.begin(), path.end(), '/');
267  if(nSlashes == 4) {
268  hasExtraLevel = true;
269  }
270  else if(nSlashes < 3 || nSlashes > 4) {
271  throw ChimeraTK::logic_error(std::string("DOOCS address has an illegal format: ") + path);
272  }
273  }
274  else if(boost::starts_with(path, "doocs://")) {
275  size_t nSlashes = std::count(path.begin(), path.end(), '/');
276  // we have 3 extra slashes compared to the standard syntax without "doocs:://"
277  if(nSlashes == 4 + 3) {
278  hasExtraLevel = true;
279  }
280  else if(nSlashes < 3 + 3 || nSlashes > 4 + 3) {
281  throw ChimeraTK::logic_error(std::string("DOOCS address has an illegal format: ") + path);
282  }
283  }
284 
285  // split the path into property name and field name
286  std::string field;
287  if(hasExtraLevel) {
288  field = path.substr(path.find_last_of('/') + 1);
289  path = path.substr(0, path.find_last_of('/'));
290  }
291 
292  // if backend is open, read property once to obtain type
293  int doocsTypeId = DATA_NULL;
294  if(isOpen()) {
295  EqAdr ea;
296  EqCall eq;
297  EqData src, dst;
298  ea.adr(path);
299  int rc = eq.get(&ea, &src, &dst);
300  if(rc) {
301  if(rc == eq_errors::ill_property || rc == eq_errors::ill_location ||
302  rc == eq_errors::ill_address) { // no property by that name
303  throw ChimeraTK::logic_error("Property does not exist: " + path + "': " + dst.get_string());
304  }
305  // runtime errors are thrown later
306  }
307  else {
308  doocsTypeId = dst.type();
309  }
310  }
311 
312  // if backend is closed, or if property could not be read, use the (potentially cached) catalogue
313  if(doocsTypeId == DATA_NULL) {
314  auto reg = getBackendRegisterCatalogue().getBackendRegister(registerPathName);
315  doocsTypeId = reg.doocsTypeId;
316  }
317 
318  // check type and create matching accessor
319  bool extraLevelUsed = false;
320  auto sharedThis = boost::static_pointer_cast<DoocsBackend>(shared_from_this());
321 
322  if(field == "eventId") {
323  extraLevelUsed = true;
324  p.reset(new DoocsBackendEventIdRegisterAccessor<UserType>(sharedThis, path, registerPathName, flags));
325  }
326  else if(field == "timeStamp") {
327  extraLevelUsed = true;
328  p.reset(new DoocsBackendTimeStampRegisterAccessor<UserType>(sharedThis, path, registerPathName, flags));
329  }
330  else if(doocsTypeId == DATA_INT || doocsTypeId == DATA_A_INT || doocsTypeId == DATA_BOOL ||
331  doocsTypeId == DATA_A_BOOL || doocsTypeId == DATA_A_SHORT) {
332  p.reset(new DoocsBackendIntRegisterAccessor<UserType>(
333  sharedThis, path, registerPathName, numberOfWords, wordOffsetInRegister, flags));
334  }
335  else if(doocsTypeId == DATA_A_LONG) {
336  p.reset(new DoocsBackendLongRegisterAccessor<UserType>(
337  sharedThis, path, registerPathName, numberOfWords, wordOffsetInRegister, flags));
338  }
339  else if(doocsTypeId == DATA_FLOAT || doocsTypeId == DATA_A_FLOAT || doocsTypeId == DATA_DOUBLE ||
340  doocsTypeId == DATA_A_DOUBLE || doocsTypeId == DATA_SPECTRUM) {
341  p.reset(new DoocsBackendFloatRegisterAccessor<UserType>(
342  sharedThis, path, registerPathName, numberOfWords, wordOffsetInRegister, flags));
343  }
344  else if(doocsTypeId == DATA_TEXT || doocsTypeId == DATA_STRING) {
345  p.reset(new DoocsBackendStringRegisterAccessor<UserType>(
346  sharedThis, path, registerPathName, numberOfWords, wordOffsetInRegister, flags));
347  }
348  else if(doocsTypeId == DATA_IIII) {
349  p.reset(new DoocsBackendIIIIRegisterAccessor<UserType>(
350  sharedThis, path, registerPathName, numberOfWords, wordOffsetInRegister, flags));
351  }
352  else if(doocsTypeId == DATA_IFFF) {
353  extraLevelUsed = true;
354  p.reset(new DoocsBackendIFFFRegisterAccessor<UserType>(
355  sharedThis, path, field, registerPathName, numberOfWords, wordOffsetInRegister, flags));
356  }
357  else if(doocsTypeId == DATA_IMAGE) {
358  auto accImpl = new DoocsBackendImageRegisterAccessor(
359  sharedThis, path, registerPathName, numberOfWords, wordOffsetInRegister, flags);
360  if constexpr(std::is_same_v<UserType, std::uint8_t>) {
361  p.reset(accImpl);
362  }
363  else {
364  boost::shared_ptr<NDRegisterAccessor<std::uint8_t>> pImpl(accImpl);
365  // any UserType can hold uint8
366  boost::shared_ptr<NDRegisterAccessor<UserType>> accDecorated(
367  new TypeChangingRangeCheckingDecorator<UserType, std::uint8_t>(
368  boost::dynamic_pointer_cast<ChimeraTK::NDRegisterAccessor<std::uint8_t>>(pImpl)));
369  p = accDecorated;
370  }
371  }
372  else {
373  throw ChimeraTK::logic_error("Unsupported DOOCS data type " + std::string(EqData().type_string(doocsTypeId)) +
374  " of property '" + _serverAddress + registerPathName + "'");
375  }
376 
377  // if the field name has been specified but the data type does not use it, throw an exception
378  if(hasExtraLevel && !extraLevelUsed) {
379  throw ChimeraTK::logic_error("Specifiaction of field name is not supported for the DOOCS data type " +
380  std::string(EqData().type_string(doocsTypeId)) + ": " + _serverAddress + registerPathName);
381  }
382 
383  p->setExceptionBackend(shared_from_this());
384  return p;
385  }

388  bool DoocsBackend::isCommunicationError(int doocs_error) {
389  // logic_error-like errors are caught at a different place. If such error appears later, a runtime_error need to be
390  // generated (device does not behave as expected, as it is not expected to change)
391  switch(doocs_error) {
392  case eq_errors::unsup_domain:
393  case eq_errors::no_ens_entry:
394  case eq_errors::ill_monitor:
395  case eq_errors::faulty_chans:
396  case eq_errors::unavail_serv:
397  case eq_errors::ill_serv:
398  case eq_errors::rpc_prot_error:
399  case eq_errors::ens_fault:
400  case eq_errors::ill_protocol:
401  case eq_errors::ill_location:
402  case eq_errors::ill_property:
403  case eq_errors::no_connection:
404  case eq_errors::conn_timeout:
405  case eq_errors::alias_error:
406  case eq_errors::no_permission:
407  return true;
408  default:
409  return false;
410  }
411  }
