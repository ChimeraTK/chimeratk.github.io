96  void parseRegister(xmlpp::Element const* registerNode, DoocsBackendRegisterCatalogue& catalogue) {
97  std::string name;
98  unsigned int len{};
99  int doocsTypeId{};
100  ChimeraTK::DataDescriptor descriptor{};
101  ChimeraTK::AccessModeFlags flags{};
102 
103  for(auto& node : registerNode->get_children()) {
104  auto e = dynamic_cast<const xmlpp::Element*>(node);
105  if(e == nullptr) {
106  continue;
107  }
108  std::string nodeName = e->get_name();
109 
110  if(nodeName == "name") {
111  name = e->get_child_text()->get_content();
112  }
113  else if(nodeName == "length") {
114  len = parseLength(e);
115  }
116  else if(nodeName == "access_mode") {
117  flags = parseAccessMode(e);
118  }
119  else if(nodeName == "doocs_type_id") {
120  doocsTypeId = parseTypeId(e);
121  }
122  }
123 
124  bool is_ifff = (doocsTypeId == DATA_IFFF);
125 
126  if(is_ifff) {
127  auto pattern = detail::endsWith(name, {"/I", "/F1", "/F2", "/F3"}).second;
128  // remove pattern from name for getRegInfo to work correctly;
129  // precondition: patten is contained in name.
130  name.erase(name.end() - pattern.length(), name.end());
131  }
132 
133  // add corresponding registers (note: only registers not yet being in the catalogue will be added)
134  catalogue.addProperty(name, len, doocsTypeId, flags);
135  }
