368 MotorReferenceSwitchData MotorControlerImpl::retrieveReferenceSwitchStatus() {
369 // the bit pattern for the active flags
370 unsigned int bitMask = 0x3U << 2 * _id;
371
372 unsigned int commonReferenceSwitchWord = _controlerSPI->read(SMDA_COMMON, JDX_REFERENCE_SWITCH).getDATA();
373 unsigned int dataWord = (commonReferenceSwitchWord & bitMask) >> 2 * _id;
374 // note: the following code uses the implicit bool conversion to/from 0/1 to
375 // keep the code short.
376 MotorReferenceSwitchData motorReferenceSwitchData(dataWord);
377
378 auto configAndRampModeData = readTypedRegister<ReferenceConfigAndRampModeData>();
379 motorReferenceSwitchData.setNegativeSwitchEnabled(!configAndRampModeData.getDISABLE_STOP_L());
380 motorReferenceSwitchData.setPositiveSwitchEnabled(!configAndRampModeData.getDISABLE_STOP_R());
381
382 return motorReferenceSwitchData;
383 }

462 double MotorControlerImpl::calculateConversionFactor(MotorControlerConfig const& motorControlerConfig) {
463 // conversionFactor = fclk[Hz] / (2^pulse_div * 2048 *32)
464 double systemClockInMhz = MD_22_DEFAULT_CLOCK_FREQ_MHZ;
465
466 auto systemClkInHz = systemClockInMhz * 1000000;
467 auto pulse_div = static_cast<double>(motorControlerConfig.dividersAndMicroStepResolutionData.getPulseDivider());
468
469 auto conversionFactor = systemClkInHz / (exp2(pulse_div) * 2048 * 32);
470 return conversionFactor;
471 }
