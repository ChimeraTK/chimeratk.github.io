166BOOST_FIXTURE_TEST_CASE(testDerivedStateMachine, DerivedStateMachine) {
167 _transitionAllowed.exchange(true);
168
169 // noEvent triggers transition from idle to first state
170 BOOST_CHECK_EQUAL(getCurrentState()->getName(), "initState");
171 BOOST_CHECK_NO_THROW(setAndProcessUserEvent(initEvent));
172
173 // Wait until the async task sets the flag to false
174 while(_transitionAllowed.load()) {
175 }
176
177 // Now, we should be in first state, second state should be requested
178 // (the flags are computed in actionFirstToSecondState, before the waiting)
179 BOOST_CHECK(_isCorrectCurrentState.load());
180 BOOST_CHECK(_isCorrectRequestedState.load());
181
182 // Let async task transition to second state
183 _transitionAllowed.exchange(true);
184
185 // Wait for the async task to finish, we should then be in second state and
186 // the requested state pointer should be reset
187 while(_asyncActionActive.load() == true) {
188 }
189 BOOST_CHECK_EQUAL(getCurrentState()->getName(), "secondState");
190 BOOST_CHECK(_isCorrectRequestedState.load());
191
192 // Invoke event to move back to 1st state
193 _counter.store(0);
194 BOOST_CHECK_NO_THROW(setAndProcessUserEvent(userEvent2));
195 BOOST_CHECK_EQUAL(getCurrentState()->getName(), "firstState");
196
197 // Transition from 1st to 3rd state
198 BOOST_CHECK_NO_THROW(setAndProcessUserEvent(userEvent1));
199 BOOST_CHECK_EQUAL(getCurrentState()->getName(), "thirdState");
200 BOOST_CHECK_EQUAL(_counter.load(), 3);
201
202 // Start transition from 3rd, via 1st, to 4th state
203 // In contrast to the sequence idle->1st->2nd above, this processes
204 // the internal event without interruption
205 _counter.store(0);
206 BOOST_CHECK_NO_THROW(setAndProcessUserEvent(DerivedStateMachine::userEvent3));
207
208 while(_asyncActionActive.load() == true) {
209 }
210
211 BOOST_CHECK_EQUAL(getCurrentState()->getName(), "fourthState");
212 BOOST_CHECK_EQUAL(_counter.load(), 45);
213
214} // BOOST_FIXTURE_TEST_CASE( testDerivedStateMachine, DerivedStateMachine )
