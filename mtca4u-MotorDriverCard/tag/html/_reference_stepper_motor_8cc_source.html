51 ExitStatus ReferenceStepperMotor::setActualPositionInSteps(int actualPositionInSteps) {
52 LockGuard guard(_mutex);
53 if(motorActive()) {
54 return ExitStatus::ERR_SYSTEM_IN_ACTION;
55 }
56 setActualPositionActions(actualPositionInSteps);
57
58 // Overwrite end switch calibration and reset calibration mode
59 _calibNegativeEndSwitchInSteps.exchange(std::numeric_limits<int>::min());
60 _calibPositiveEndSwitchInSteps.exchange(std::numeric_limits<int>::max());
61
62 return ExitStatus::SUCCESS;

65 ExitStatus ReferenceStepperMotor::translateAxisInSteps(int translationInSteps) {
66 LockGuard guard(_mutex);
67 if(motorActive()) {
68 return ExitStatus::ERR_SYSTEM_IN_ACTION;
69 }
70
71 translateAxisActions(translationInSteps);
72
73 // If motor has been calibrated, translate also end switch positions
74 if(this->_calibrationMode.load() == CalibrationMode::FULL) {
75 if(utility::checkIfOverflow(_calibPositiveEndSwitchInSteps.load(), translationInSteps) ||
76 utility::checkIfOverflow(_calibNegativeEndSwitchInSteps.load(), translationInSteps)) {
77 return ExitStatus::ERR_INVALID_PARAMETER;
78 }
79
80 _calibPositiveEndSwitchInSteps.fetch_add(translationInSteps);
81 _calibNegativeEndSwitchInSteps.fetch_add(translationInSteps);
82 }
83 return ExitStatus::SUCCESS;
84 }

196 void ReferenceStepperMotor::loadEndSwitchCalibration() {
197 LockGuard guard(_mutex);
198
199 if(_motorController->getCalibrationTime() != 0) {
200 int calibPositiveEndSwitchInSteps = _motorController->getPositiveReferenceSwitchCalibration();
201 int calibNegativeEndSwitchInSteps = _motorController->getNegativeReferenceSwitchCalibration();
202 _calibPositiveEndSwitchInSteps.exchange(calibPositiveEndSwitchInSteps);
203 _calibNegativeEndSwitchInSteps.exchange(calibNegativeEndSwitchInSteps);
204
205 if(calibPositiveEndSwitchInSteps == std::numeric_limits<int>::max() &&
206 calibNegativeEndSwitchInSteps == std::numeric_limits<int>::min()) {
207 _calibrationMode.exchange(CalibrationMode::SIMPLE);
208 }
209 else {
210 _calibrationMode.exchange(CalibrationMode::FULL);
211 }
212 }

223 bool ReferenceStepperMotor::isEndSwitchActive(Sign sign) {
224 bool posActive = positiveSwitchActive();
225 bool negActive = negativeSwitchActive();
226
227 if(posActive && negActive) {
228 _errorMode.exchange(Error::BOTH_END_SWITCHES_ON);
229 _stateMachine->setAndProcessUserEvent(StateMachine::errorEvent);
230 }
231 else {
232 // The error mode was BOTH_END_SWITCHES_ON, reset the error mode. Otherwise this will stick unless the application
233 // is re-started.
234 auto expected = Error::BOTH_END_SWITCHES_ON;
235 auto replace = Error::NO_ERROR;
236 _errorMode.compare_exchange_strong(expected, replace);
237 }
238
239 if(_toleranceCalculated) {
240 if(posActive) {
241 if(std::abs(_motorController->getActualPosition() - _calibPositiveEndSwitchInSteps.load()) >
242 3 * _tolerancePositiveEndSwitch.load()) {
243 _errorMode.exchange(Error::CALIBRATION_ERROR);
244 }
245 }
246 else if(negActive) {
247 if(std::abs(_motorController->getActualPosition() - _calibNegativeEndSwitchInSteps.load()) >
248 3 * _toleranceNegativeEndSwitch.load()) {
249 _errorMode.exchange(Error::CALIBRATION_ERROR);
250 }
251 }
252 }
253
254 if(sign == Sign::POSITIVE) {
255 return posActive;
256 }
257
258 return negActive;
259 }
