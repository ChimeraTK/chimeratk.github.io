21  bool checkIfOverflow(int termA, int termB) {
22  int signTermA = (termA > 0) - (termA < 0);
23  int signTermB = (termB > 0) - (termB < 0);
24  if(signTermA * signTermB <= 0) {
25  return false;
26  }
27 
28  if(signTermA > 0) {
29  return (termB > std::numeric_limits<int>::max() - termA);
30  }
31  return (termB < std::numeric_limits<int>::min() - termA);
32  }
33  } // namespace utility

60  ExitStatus BasicStepperMotor::checkNewPosition(int newPositionInSteps) {
61  if(_calibrationMode.load() == CalibrationMode::NONE) {
62  return ExitStatus::ERR_SYSTEM_NOT_CALIBRATED;
63  }
64  if(!limitsOK(newPositionInSteps)) {
65  return ExitStatus::ERR_INVALID_PARAMETER;
66  }
67  return ExitStatus::SUCCESS;
68  }

72  bool BasicStepperMotor::limitsOK(int newPositionInSteps) {
73  if(_softwareLimitsEnabled && newPositionInSteps >= _minPositionLimitInSteps &&
74  newPositionInSteps <= _maxPositionLimitInSteps) {
75  return true;
76  }
77 
78  // Either we don't care about the software limits or the limit test above failed, so this is also
79  // correct to use this condition
80  return !_softwareLimitsEnabled;
81  }

91  ExitStatus BasicStepperMotor::moveRelativeInSteps(int delta) {
92  LockGuard guard(_mutex);
93 
94  if(motorActive()) {
95  return ExitStatus::ERR_SYSTEM_IN_ACTION;
96  }
97 
98  int newPosition = _motorControler->getActualPosition() + delta;
99  auto checkResult = checkNewPosition(newPosition);
100  if(checkResult != ExitStatus::SUCCESS) {
101  return checkResult;
102  }
103 
104  _targetPositionInSteps = newPosition;
105 
106  _stateMachine->setAndProcessUserEvent(StateMachine::moveEvent);
107  return ExitStatus::SUCCESS;
108  }

324  void BasicStepperMotor::translateLimits(int translationInSteps) {
325  if(utility::checkIfOverflow(_maxPositionLimitInSteps, translationInSteps)) {
326  _maxPositionLimitInSteps = std::numeric_limits<int>::max();
327  }
328  else {
329  _maxPositionLimitInSteps = _maxPositionLimitInSteps + translationInSteps;
330  }
331  if(utility::checkIfOverflow(_minPositionLimitInSteps, translationInSteps)) {
332  _minPositionLimitInSteps = std::numeric_limits<int>::min();
333  }
334  else {
335  _minPositionLimitInSteps = _minPositionLimitInSteps + translationInSteps;
336  }
337  }
