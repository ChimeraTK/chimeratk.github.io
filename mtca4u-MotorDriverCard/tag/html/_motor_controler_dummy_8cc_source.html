78 bool MotorControlerDummy::isStepping() {

80 if(_targetPosition == _currentPosition) {

227 void MotorControlerDummy::roundToNextFullStep(int& targetPosition) {
228 int delta = targetPosition - _currentPosition;
229 int deltaMicroStep = delta * _userMicroStepSize;
230 unsigned int actualMicroStepCount = ((_currentPosition * _userMicroStepSize) - 1) & 0x3FF;
231 unsigned int newActualMicroStepCount = (actualMicroStepCount + deltaMicroStep) & 0x3FF;
232 unsigned int distanceToPreviousFullStep = (newActualMicroStepCount + 1) % 256;
233 if(distanceToPreviousFullStep < 128) {
234 targetPosition = targetPosition - distanceToPreviousFullStep / _userMicroStepSize;
235 }
236 else {
237 targetPosition = targetPosition + (256 - distanceToPreviousFullStep) / _userMicroStepSize;
238 }
239 }

372 void MotorControlerDummy::moveTowardsTarget(
373 float fraction, bool block, bool bothEndSwitchesAlwaysOn, bool zeroPositions) {
374 lock_guard guard(_motorControllerDummyMutex);
375 if(zeroPositions) {
376 _absolutePosition = 0;
377 _targetPosition = 0;
378 _currentPosition = 0;
379 _positiveEndSwitchEnabled = true;
380 _negativeEndSwitchEnabled = true;
381 }
382
383 _bothEndSwitchesAlwaysOn = bothEndSwitchesAlwaysOn;
384
385 _blockMotor = block;
386 if(_blockMotor) {
387 return;
388 }
389
390 // check if the motor is should to be moving if it was enabled
391 // otherwise return immediately
392 if(!isStepping()) {
393 return;
394 }
395
396 // range check for the input parameter 'fraction', just force it to 0..1
397 if(fraction <= 0) {
398 return;
399 }
400 if(fraction > 1) {
401 fraction = 1.;
402 }
403
404 // calculate the new abolute target position for this move operation
405 int relativeSteps = (_targetPosition - _currentPosition) * fraction;
406 int absoluteTargetInThisMove = _absolutePosition + relativeSteps;
407 int targetInThisMove = _currentPosition + relativeSteps;
408
409 if(!_motorCurrentEnabled) {
410 // the motor is stepping to the target position, but the actual positon
411 // does not change. Thus no end switch will be hit.
412 _currentPosition = targetInThisMove;
413 return;
414 }
415
416 // ok, so the motor is really moving.
417
418 // check whether an end switch would be hit
419 if(_positiveEndSwitchEnabled && (absoluteTargetInThisMove > _positiveEndSwitchPosition)) {
420 absoluteTargetInThisMove = _positiveEndSwitchPosition;
421 // also the current positon stops if an end switch is reached
422 int stepsToEndSwitch = _positiveEndSwitchPosition - _absolutePosition;
423 targetInThisMove = _currentPosition + stepsToEndSwitch;
424 }
425 if(_negativeEndSwitchEnabled && (absoluteTargetInThisMove < _negativeEndSwitchPosition)) {
426 absoluteTargetInThisMove = _negativeEndSwitchPosition;
427 // also the current positon stops if an end switch is reached
428 int stepsToEndSwitch = _negativeEndSwitchPosition - _absolutePosition;
429 targetInThisMove = _currentPosition + stepsToEndSwitch;
430 }
431
432 // finally 'move' the motor
433 _absolutePosition = absoluteTargetInThisMove;
434 _currentPosition = targetInThisMove;
435 }
