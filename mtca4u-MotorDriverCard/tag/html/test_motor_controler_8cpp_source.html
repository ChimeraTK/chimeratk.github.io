412 void MotorControlerTest::testGetReferenceSwitchData(boost::shared_ptr<MotorDriverCardImpl> motorDriverCard) {
413 // This approach is intentinally clumsy and manual in order not to use the
414 // same algorithm as in the implementation, but still be felxible if the
415 // register content changes.
416 unsigned int referenceWord = 0; // setting to 0 to avoid compiler warning in default case.
417
418 switch(_motorControler->getID()) {
419 case 0:
420 referenceWord = motorDriverCard->getReferenceSwitchData().getDATA() & 0x3;
421 break;
422 case 1:
423 referenceWord = ((motorDriverCard->getReferenceSwitchData().getDATA() & 0xC) >> 2);
424 break;
425 case 2:
426 referenceWord = ((motorDriverCard->getReferenceSwitchData().getDATA() & 0x30) >> 4);
427 break;
428 default:
429 BOOST_FAIL("Invalid MotorID. Either there is something wrong in the code "
430 "or the test has to be adapted.");
431 }
432
433 BOOST_CHECK(_motorControler->getReferenceSwitchData().getSwitchesActiveWord() == referenceWord);
434 BOOST_CHECK(_motorControler->getReferenceSwitchData().getPositiveSwitchEnabled() ==
435 !_motorControler->getReferenceConfigAndRampModeData().getDISABLE_STOP_R());
436 BOOST_CHECK(_motorControler->getReferenceSwitchData().getNegativeSwitchEnabled() ==
437 !_motorControler->getReferenceConfigAndRampModeData().getDISABLE_STOP_L());
438 }
