85 void LinearStepperMotor::StateMachine::calibrationThreadFunction() {
86 _motor._calibrationFailed.exchange(false);
87 _stopAction.exchange(false);
88 _moveInterrupted.exchange(false);
89
90 std::cerr << "Calibration Thread enter" << std::endl;
91
92 try {
93 // Local variables for calibrated position
94 // TODO Move to else scope to remove cppcheck warning
95 int calibPositiveEndSwitchInSteps = 0;
96 int calibNegativeEndSwitchInSteps = 0;
97
98 // At least one end switch disabled -> calibration not possible
99 if(!(_motor._positiveEndSwitchEnabled.load() && _motor._negativeEndSwitchEnabled.load())) {
100 _motor._motorControler->setCalibrationTime(0);
101
102 _motor._calibrationFailed.exchange(true);
103 _motor._calibrationMode.exchange(CalibrationMode::NONE);
104 }
105 else {
106 findEndSwitch(Sign::POSITIVE);
107 calibPositiveEndSwitchInSteps = _motor.getCurrentPositionInSteps();
108 findEndSwitch(Sign::NEGATIVE);
109 calibNegativeEndSwitchInSteps = _motor.getCurrentPositionInSteps();
110
111 if(_moveInterrupted.load() || _stopAction.load()) {
112 _motor._motorControler->setCalibrationTime(0);
113
114 _motor._calibrationFailed.exchange(true);
115 _motor._errorMode.exchange(Error::CALIBRATION_ERROR);
116 _motor._calibrationMode.exchange(CalibrationMode::NONE);
117 }
118 else {
119 // Define positive axis with negative end switch as zero
120 calibPositiveEndSwitchInSteps = calibPositiveEndSwitchInSteps - calibNegativeEndSwitchInSteps;
121 calibNegativeEndSwitchInSteps = 0;
122
123 _motor._motorControler->setCalibrationTime(time(nullptr));
124 _motor._motorControler->setPositiveReferenceSwitchCalibration(calibPositiveEndSwitchInSteps);
125 _motor._motorControler->setNegativeReferenceSwitchCalibration(0);
126 _motor._calibPositiveEndSwitchInSteps.exchange(calibPositiveEndSwitchInSteps);
127 _motor._calibNegativeEndSwitchInSteps.exchange(0);
128
129 _motor._calibrationMode.exchange(CalibrationMode::FULL);
130 _motor.resetMotorControllerPositions(0);
131 }
132 }
133 }
134 catch(ChimeraTK::runtime_error&) {
135 _motor._motorControler->setCalibrationTime(0);
136
137 _motor._calibrationFailed.exchange(true);
138 _motor._errorMode.exchange(Error::CALIBRATION_ERROR);
139 _motor._calibrationMode.exchange(CalibrationMode::NONE);
140 }
141
142 _asyncActionActive.exchange(false);
143 return;
144 }

146 void LinearStepperMotor::StateMachine::findEndSwitch(Sign sign) {
147 while(!_motor.isEndSwitchActive(sign)) {
148 if(_stopAction.load() || _moveInterrupted.load()) {
149 std::cerr << "fes a: " << _stopAction.load() << " " << _moveInterrupted.load() << std::endl;
150 return;
151 }
152 else if(_motor.getTargetPositionInSteps() != _motor.getCurrentPositionInSteps() &&
153 !_motor.isPositiveReferenceActive() && !_motor.isNegativeReferenceActive()) {
154 _moveInterrupted.exchange(true);
155 std::cerr << "fes b: " << (_motor.getTargetPositionInSteps() != _motor.getCurrentPositionInSteps()) << " "
156 << (!_motor.isPositiveReferenceActive()) << " " << (!_motor.isNegativeReferenceActive()) << std::endl;
157 return;
158 }
159 moveToEndSwitch(sign);
160 }
161 std::cerr << "fes c: Found endswitch" << std::endl;
162 return;
163 }

220 double LinearStepperMotor::StateMachine::getToleranceEndSwitch(Sign sign) {
221 double meanMeasurement = 0;
222 double stdMeasurement = 0;
223 const int N_TOLERANCE_CALC_SAMPLES = 10;
224 double measurements[N_TOLERANCE_CALC_SAMPLES];
225
226 int endSwitchPosition = getPositionEndSwitch(sign);
227
228 // Get 10 samples for tolerance calculation
229 for(unsigned i = 0; i < N_TOLERANCE_CALC_SAMPLES; i++) {
230 if(_stopAction.load() || _moveInterrupted.load()) {
231 break;
232 }
233
234 // Move close to end switch
235 {
236 boost::lock_guard<boost::mutex> lck(_motor._mutex);
237 _motor._motorControler->setTargetPosition(endSwitchPosition - static_cast<int>(sign) * 1000);
238 }
239 while(_motor._motorControler->isMotorMoving()) {
240 std::this_thread::sleep_for(std::chrono::milliseconds(wakeupPeriodInMilliseconds));
241 }
242
243 // Check if in expected position
244 if(!_motor.verifyMoveAction()) {
245 _moveInterrupted.exchange(true);
246 break;
247 }
248
249 // Try to move beyond end switch
250 {
251 boost::lock_guard<boost::mutex> lck(_motor._mutex);
252 _motor._motorControler->setTargetPosition(endSwitchPosition + static_cast<int>(sign) * 1000);
253 }
254 while(_motor._motorControler->isMotorMoving()) {
255 std::this_thread::sleep_for(std::chrono::milliseconds(wakeupPeriodInMilliseconds));
256 }
257 if(!_motor.isEndSwitchActive(sign)) {
258 _moveInterrupted.exchange(true);
259 break;
260 }
261
262 // Mean calculation
263 meanMeasurement += static_cast<double>(_motor.getCurrentPositionInSteps()) / N_TOLERANCE_CALC_SAMPLES;
264 measurements[i] = _motor.getCurrentPositionInSteps();
265 } /* for (i in [0,9]) */
266
267 // Compute variance
268 if(!(_stopAction.load() || _moveInterrupted.load())) {
269 for(unsigned int i = 0; i < N_TOLERANCE_CALC_SAMPLES; i++) {
270 stdMeasurement += ((measurements[i] - meanMeasurement) / (N_TOLERANCE_CALC_SAMPLES - 1)) *
271 (measurements[i] - meanMeasurement);
272 }
273 }
274
275 return sqrt(stdMeasurement);
276 }
