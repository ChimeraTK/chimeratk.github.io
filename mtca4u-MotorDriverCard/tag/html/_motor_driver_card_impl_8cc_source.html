155 void MotorDriverCardImpl::checkFirmwareVersion() {
156 ChimeraTK::ScalarRegisterAccessor<int32_t> firmwareVersion =
157 _device->getScalarRegisterAccessor<int32_t>(_moduleName + "/" + PROJECT_VERSION_ADDRESS_STRING);
158 firmwareVersion.read();
159
160 // only allow firmware versions with the same major number
161 uint32_t maxFirmwareVersion = (MINIMAL_FIRMWARE_VERSION & 0xFF000000) | 0x00FFFFFF;
162
163 // check the version
164 if((static_cast<uint32_t>(firmwareVersion) < static_cast<uint32_t>(MINIMAL_FIRMWARE_VERSION)) ||
165 (static_cast<uint32_t>(firmwareVersion) > maxFirmwareVersion)) {
166 std::stringstream errorMessage;
167 errorMessage << "MotorDriverCardImpl detected wrong firmware version. "
168 << "Current firmware is 0x" << std::hex << firmwareVersion << ", minumum required version is 0x"
169 << MINIMAL_FIRMWARE_VERSION << ", maximum allowed version is 0x" << maxFirmwareVersion << std::dec
170 << ".";
171 throw ChimeraTK::runtime_error(errorMessage.str());
172 }
