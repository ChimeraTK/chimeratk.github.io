264 LinearStepperMotorStatusAndError LinearStepperMotor::determineMotorStatusAndError() {
265 boost::lock_guard<boost::mutex> lock_guard(_mutex);
266
267 _motorStatus = LinearStepperMotorStatusTypes::M_OK;
268 _motorError = StepperMotorErrorTypes::M_NO_ERROR;
269
270 // first check if disabled
271 if(!_motorControler->isEnabled()) {
272 _motorStatus = LinearStepperMotorStatusTypes::M_DISABLED;
273 return LinearStepperMotorStatusAndError(_motorStatus, _motorError);
274 }
275
276 // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
277 // ERROR CONDITION CHECK - it should be done first.
278 // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
279
280 // check end switches
281 bool positiveSwitchStatus =
282 _motorControler->getReferenceSwitchData().getPositiveSwitchActive() & _positiveEndSwitchEnabled;
283 bool negativeSwitchStatus =
284 _motorControler->getReferenceSwitchData().getNegativeSwitchActive() & _negativeEndSwitchEnabled;
285
286 // check if both end switches are on in the same time - this means error;
287 if(positiveSwitchStatus && negativeSwitchStatus) {
288 positiveSwitchStatus = _motorControler->getReferenceSwitchData().getPositiveSwitchActive();
289 negativeSwitchStatus = _motorControler->getReferenceSwitchData().getNegativeSwitchActive();
290
291 if(positiveSwitchStatus)
292 _logger(logging::Logger::ERROR) << "LinearStepperMotor::determineMotorStatusAndError(): Positive "
293 "end-switch on. "
294 << std::endl;
295
296 if(negativeSwitchStatus)
297 _logger(logging::Logger::ERROR) << "LinearStepperMotor::determineMotorStatusAndError(): Negative "
298 "end-switch on. "
299 << std::endl;
300
301 if(positiveSwitchStatus && negativeSwitchStatus) {
302 _motorStatus = LinearStepperMotorStatusTypes::M_ERROR;
303 _logger(logging::Logger::ERROR) << "LinearStepperMotor::determineMotorStatusAndError(): Both end "
304 "switches error detected. "
305 << std::endl;
306 _motorError = LinearStepperMotorErrorTypes::M_BOTH_END_SWITCH_ON;
307
308 return LinearStepperMotorStatusAndError(_motorStatus, _motorError);
309 }
310 }
311 // check configuration of software end switches error
312 if(_softwareLimitsEnabled && _maxPositionLimitInSteps <= _minPositionLimitInSteps) {
313 _motorStatus = LinearStepperMotorStatusTypes::M_ERROR;
314 _motorError = StepperMotorErrorTypes::M_CONFIG_ERROR_MIN_POS_GRATER_EQUAL_TO_MAX;
315
316 return LinearStepperMotorStatusAndError(_motorStatus, _motorError);
317 }
318
319 // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
320 // STATUS CHECK
321 // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
322
323 // hardware end switches active
324 if(positiveSwitchStatus) {
325 _motorStatus = LinearStepperMotorStatusTypes::M_POSITIVE_END_SWITCHED_ON;
326 return LinearStepperMotorStatusAndError(_motorStatus, _motorError);
327 }
328
329 if(negativeSwitchStatus) {
330 _motorStatus = LinearStepperMotorStatusTypes::M_NEGATIVE_END_SWITCHED_ON;
331 return LinearStepperMotorStatusAndError(_motorStatus, _motorError);
332 }
333
334 // current on -> motor is moving
335 if(_motorControler->getStatus().getStandstillIndicator() == 0) {
336 _motorStatus = LinearStepperMotorStatusTypes::M_IN_MOVE;
337 return LinearStepperMotorStatusAndError(_motorStatus, _motorError);
338 }
339
340 if(_targetPositionInSteps != _motorControler->getActualPosition()) {
341 _motorStatus = LinearStepperMotorStatusTypes::M_NOT_IN_POSITION;
342 return LinearStepperMotorStatusAndError(_motorStatus, _motorError);
343 }
344
345 // software end switches active
346 if(_softwareLimitsEnabled) {
347 if(_targetPositionInSteps >= _maxPositionLimitInSteps) {
348 _motorStatus = LinearStepperMotorStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON;
349 return LinearStepperMotorStatusAndError(_motorStatus, _motorError);
350 }
351
352 if(_targetPositionInSteps <= _minPositionLimitInSteps) {
353 _motorStatus = LinearStepperMotorStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON;
354 return LinearStepperMotorStatusAndError(_motorStatus, _motorError);
355 }
356 }
357
358 return LinearStepperMotorStatusAndError(_motorStatus, _motorError);
359 }

361 void LinearStepperMotor::moveToEndSwitch(LinearStepperMotorStatus targetStatus) {
362 LinearStepperMotorStatusAndError statusAndError;
363 bool takeFlagInAccount = false;
364
365 // condition for targetStatus equal to
366 // LinearStepperMotorStatusTypes::M_POSITIVE_END_SWITCHED_ON one want to move
367 // in positive direction
368 int dir = 1;
369 // and when negative end switch will be set when positive expected -> error
370 LinearStepperMotorStatus errortargetStatus = LinearStepperMotorStatusTypes::M_NEGATIVE_END_SWITCHED_ON;
371
372 // condition for targetStatus equal to
373 // LinearStepperMotorStatusTypes::M_NEGATIVE_END_SWITCHED_ON
374 if(targetStatus == LinearStepperMotorStatusTypes::M_NEGATIVE_END_SWITCHED_ON) {
375 dir = -1;
376 errortargetStatus = LinearStepperMotorStatusTypes::M_POSITIVE_END_SWITCHED_ON;
377 }
378
379 do {
380 // get current position and send motor to new lower position
381 int currentPosition = _motorControler->getActualPosition();
382 int newPosInSteps = currentPosition + dir * 500000;
383 _logger(logging::Logger::INFO) << "LinearStepperMotor::calibrateMotor : Current position (steps): "
384 << currentPosition << " (units): " << recalculateStepsToUnits(currentPosition)
385 << " New position (steps): " << currentPosition + dir * 500000
386 << " (units): " << recalculateStepsToUnits(currentPosition + dir * 500000)
387 << std::endl;
388 moveToPositionInSteps(newPosInSteps);
389 statusAndError = determineMotorStatusAndError();
390
391 if(_stopMotorCalibration == true) {
392 _stopMotorCalibration = false;
393 _logger(logging::Logger::WARNING)
394 << "LinearStepperMotor::calibrateMotor : Calibration stopped by user." << std::endl;
395
396 _motorCalibrationStatus = StepperMotorCalibrationStatusType::M_CALIBRATION_STOPPED_BY_USER;
397 return;
398 }
399
400 if(statusAndError.status == LinearStepperMotorStatusTypes::M_IN_MOVE ||
401 statusAndError.status == LinearStepperMotorStatusTypes::M_NOT_IN_POSITION) {
402 takeFlagInAccount = true;
403 }
404
405 if(takeFlagInAccount && statusAndError.status == errortargetStatus) {
406 _logger(logging::Logger::ERROR) << "LinearStepperMotor::calibrateMotor : "
407 << (dir == 1 ? "Negative" : "Positive") << " end switch reached when "
408 << (dir == 1 ? "positive" : "negative") << " expected. Calibration failed.\n";
409 _motorCalibrationStatus = StepperMotorCalibrationStatusType::M_CALIBRATION_FAILED;
410 return;
411 }
412 if(statusAndError.status == LinearStepperMotorStatusTypes::M_ERROR) {
413 _logger(logging::Logger::ERROR) << "LinearStepperMotor::calibrateMotor : Motor in "
414 "error state. Error is: "
415 << statusAndError.error << std::endl;
416 _motorCalibrationStatus = StepperMotorCalibrationStatusType::M_CALIBRATION_FAILED;
417 return;
418 }
419 } while(statusAndError.status != targetStatus);
420 }
