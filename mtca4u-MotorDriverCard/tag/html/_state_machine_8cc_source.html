85 /********************************************************************************************************************/
86
87 void StateMachine::performTransition(const Event& event) {
88 auto const& transitionTable = _currentState->getTransitionTable();
89 auto it = transitionTable.find(event);
90 if(it != transitionTable.end()) {
91 _requestedState = ((it->second).targetState);
92 _requestedInternalCallback = (it->second).internalCallbackAction;
93
94 // Apply new state right away, if no async action active
95 if(!_asyncActionActive.load()) {
96 moveToRequestedState();
97 }
98 (it->second).entryCallbackAction();
99 }
100 }
