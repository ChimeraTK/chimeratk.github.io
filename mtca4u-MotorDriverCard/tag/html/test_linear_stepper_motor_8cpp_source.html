666void LinearStepperMotorTest::testIsMoving() {
667 // set up the motor position to a value in a good operating range.
668 _motorControlerDummy->resetInternalStateToDefaults();
669 _stepperMotor->setEnabled(true);
670 _stepperMotor->setSoftwareLimitsEnabled(false);
671 _stepperMotor->setAutostart(true);
672
673 // The actual tests
674 // 1. Move between the +/- endwitch range
675 BOOST_CHECK(_stepperMotor->isMoving() == false);
676 _stepperMotor->setTargetPosition(100);
677 BOOST_CHECK(_stepperMotor->isMoving() == true);
678 _motorControlerDummy->moveTowardsTarget(.5);
679 BOOST_CHECK(_stepperMotor->isMoving() == true);
680 _motorControlerDummy->moveTowardsTarget(1); // motor at target position;
681 BOOST_CHECK(_stepperMotor->isMoving() == false);
682
683 // 2. Move beyond an end switch
684 _motorControlerDummy->resetInternalStateToDefaults();
685 auto negativeEndSwitchPosition = _motorControlerDummy->getNEgativeEndSwitch();
686 BOOST_CHECK(_stepperMotor->isMoving() == false);
687 _stepperMotor->setTargetPosition(negativeEndSwitchPosition - 100);
688 BOOST_CHECK(_stepperMotor->isMoving() == true);
689 _motorControlerDummy->moveTowardsTarget(1); // at negative end switch
690 BOOST_CHECK(_stepperMotor->isMoving() == false);
691 BOOST_CHECK(_stepperMotor->getStatusAndError().status == LinearStepperMotorStatusTypes::M_NEGATIVE_END_SWITCHED_ON);
692}
