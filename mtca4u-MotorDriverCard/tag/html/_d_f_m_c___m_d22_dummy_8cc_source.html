167  void DFMC_MD22Dummy::handleControlerSpiWrite() {
168  // debug functionality: cause timeouts by ignoring spi writes
169  if(checkSpiTimeoutsCounter()) {
170  return;
171  }
172  int32_t value = SPI_SYNC_ERROR;
173  // debug functionality: cause errors
174  if(_causeSpiErrors) {
175  write(_controlerSpiBar, _controlerSpiSyncAddress, &value, 4);
176  return;
177  }
178 
179  int32_t controlerSpiSyncWord;
180  read(_controlerSpiBar, _controlerSpiSyncAddress, &controlerSpiSyncWord, 4);
181 
182  if(controlerSpiSyncWord != SPI_SYNC_REQUESTED) {
183  write(_controlerSpiBar, _controlerSpiSyncAddress, &value, 4);
184  return;
185  }
186 
187  int32_t controlerSpiWriteWord;
188  read(_controlerSpiBar, _controlerSpiWriteAddress, &controlerSpiWriteWord, 4);
189 
190  TMC429InputWord inputWord(controlerSpiWriteWord);
191  uint32_t controlerSpiAddress = inputWord.getADDRESS();
192  if(inputWord.getRW() == RW_READ) {
193  writeControlerSpiContentToReadbackRegister(controlerSpiAddress);
194  // it takes some time to perform this in firmware. Implement a short delay,
195  // which proabaly will cause rescheduling, twice for write and read
196  boost::this_thread::sleep(boost::posix_time::microseconds(2 * _microsecondsControllerSpiDelay));
197  }
198  else {
199  writeContentToControlerSpiRegister(inputWord.getDATA(), controlerSpiAddress);
200  // it takes some time to perform this in firmware. Implement a short delay,
201  // which proabaly will cause rescheduling.
202  boost::this_thread::sleep(boost::posix_time::microseconds(_microsecondsControllerSpiDelay));
203  }
204 
205  triggerActionsOnControlerSpiWrite(inputWord);
206  value = SPI_SYNC_OK;
207  // SPI action done, write sync ok to finish the handshake
208  write(_controlerSpiBar, _controlerSpiSyncAddress, &value, 4);
209  }

211  void DFMC_MD22Dummy::handleDriverSpiWrite(unsigned int ID) {
212  // debug functionality: cause timeouts by ignoring spi writes
213  if(checkSpiTimeoutsCounter()) {
214  return;
215  }
216 
217  unsigned int bar = _driverSPIs[ID].bar;
218  unsigned int writeIndex = _driverSPIs[ID].pcieWriteAddress / sizeof(int32_t);
219  unsigned int syncIndex = _driverSPIs[ID].pcieSyncAddress / sizeof(int32_t);
220 
221  // debug functionality: cause errors
222  if(_causeSpiErrors) {
223  _barContents[bar].at(syncIndex) = SPI_SYNC_ERROR;
224  return;
225  }
226 
227  // check the sync register
228  if(_barContents[bar].at(syncIndex) != SPI_SYNC_REQUESTED) {
229  _barContents[bar].at(syncIndex) = SPI_SYNC_ERROR;
230  return;
231  }
232 
233  // perform the SPI transfer
234  unsigned int writtenSpiWord = _barContents[bar].at(writeIndex);
235  unsigned int spiAddress = tmc260::spiAddressFromDataWord(writtenSpiWord);
236  unsigned int payloadDataMask = tmc260::dataMaskFromSpiAddress(spiAddress);
237 
238  boost::this_thread::sleep(boost::posix_time::microseconds(_microsecondsDriverSpiDelay));
239 
240  _driverSPIs[ID].addressSpace.at(spiAddress) = writtenSpiWord & payloadDataMask;
241 
242  driverSpiActions(ID, spiAddress, writtenSpiWord & payloadDataMask);
243 
244  // SPI transfer ready, set the sync register to ok
245  _barContents[bar].at(syncIndex) = SPI_SYNC_OK;
246  }
