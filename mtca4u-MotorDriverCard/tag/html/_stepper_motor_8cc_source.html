40 StepperMotorStatusAndError StepperMotor::moveToPositionInSteps(int newPositionInSteps) {
41 StepperMotorStatusAndError statusAndError = this->determineMotorStatusAndError();
42 if(statusAndError.status == StepperMotorStatusTypes::M_ERROR ||
43 statusAndError.status == StepperMotorStatusTypes::M_DISABLED) {
44 return statusAndError;
45 }
46
47 _blockingFunctionActive = true;
48 // Ask the hardware to move the motor.
49 this->setTargetPositionInSteps(newPositionInSteps);
50 if(!this->getAutostart()) {
51 this->start();
52 }
53
54 // Block till the motor has finished moving.
55 while(StepperMotor::isMoving()) {
56 usleep(1000);
57 }
58
59 // update and return status and error once at target position.
60 _blockingFunctionActive = false;
61
62 return (this->determineMotorStatusAndError());
63 }

328 int StepperMotor::truncateMotorPosition(int newPosition) {
329 int position = newPosition;
330
331 if(!_softwareLimitsEnabled) return position;
332
333 // truncate new position according to the internal limits.
334 if(position > _maxPositionLimitInSteps) {
335 position = _maxPositionLimitInSteps;
336 }
337 else if(position < _minPositionLimitInSteps) {
338 position = _minPositionLimitInSteps;
339 }
340
341 return position;
342 }

344 StepperMotorStatusAndError StepperMotor::determineMotorStatusAndError() {
345 // InternalMutex intMutex(this);
346 boost::lock_guard<boost::mutex> guard(_mutex);
347
348 _motorStatus = StepperMotorStatusTypes::M_OK;
349 _motorError = StepperMotorErrorTypes::M_NO_ERROR;
350
351 // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
352 // ERROR CONDITION CHECK - it should be done first.
353 // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
354
355 if(_softwareLimitsEnabled && (_maxPositionLimitInSteps <= _minPositionLimitInSteps)) {
356 _motorStatus = StepperMotorStatusTypes::M_ERROR;
357 _motorError = StepperMotorErrorTypes::M_CONFIG_ERROR_MIN_POS_GRATER_EQUAL_TO_MAX;
358
359 return StepperMotorStatusAndError(_motorStatus, _motorError);
360 }
361
362 // first check if disabled
363 if(!_motorControler->isEnabled()) {
364 _motorStatus = StepperMotorStatusTypes::M_DISABLED;
365 return StepperMotorStatusAndError(_motorStatus, _motorError);
366 }
367
368 // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
369 // STATUS CHECK
370 // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
371
372 if(_motorControler->getStatus().getStandstillIndicator() == 0) {
373 _motorStatus = StepperMotorStatusTypes::M_IN_MOVE;
374 return StepperMotorStatusAndError(_motorStatus, _motorError);
375 }
376
377 if(_targetPositionInSteps != _motorControler->getActualPosition()) {
378 _motorStatus = StepperMotorStatusTypes::M_NOT_IN_POSITION;
379 return StepperMotorStatusAndError(_motorStatus, _motorError);
380 }
381
382 if(_softwareLimitsEnabled) {
383 if(_targetPositionInSteps >= _maxPositionLimitInSteps) {
384 _motorStatus = StepperMotorStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON;
385 return StepperMotorStatusAndError(_motorStatus, _motorError);
386 }
387
388 if(_targetPositionInSteps <= _minPositionLimitInSteps) {
389 _motorStatus = StepperMotorStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON;
390 return StepperMotorStatusAndError(_motorStatus, _motorError);
391 }
392 }
393
394 return StepperMotorStatusAndError(_motorStatus, _motorError);
395 }
