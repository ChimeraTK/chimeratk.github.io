57 // (also int does not cut it because static_cast<int>(-0.8)
58 // yields 0, but it should issue a warning

63 pulseDiv = minimumAllowedPulseDiv;
64 warnings.push_back("pulseDiv was too small. Check you settings, you might "
65 "have to decrease maxRPM.");
66 }

69 // "round" to the next lowest integet (as a cast to int would do).
70 // This is essential for the following calculations
71 pulseDiv = std::floor(pulseDiv);
72
73 // after rounding the value has to be <= 13
74 if(pulseDiv > maximumAllowedPulseDiv) {

77 "have to increase maxRPM.");

135 warnings.push_back(warning.str());

145 throw std::out_of_range("aMax is calculated as 0. The motor would never move. Check your input "

147 }

149 aMax = std::floor(aMax);
150
151 double p = aMax / minimumAllowedPMul / pow(2, rampDiv - pulseDiv);
152
153 double reducedP = (1. - reductionValue) * p;
154
155 // This is the formula as calculated from the data sheet. Please leave
156 // this block for reference, although a different calculation is used.
157 //
158 // Sorry for the hard coded 3. it is log2( maximumAllowedAMax / 256 )
159 // where it is not clear where the 256 comes from. Yet another divider.
160 // int pDiv =
161 // static_cast<int>(log2( maximumAllowedPMul / reducedP ) - 3);
162
163 // The formula above itself is not in the data sheet, they use the
164 // method to calculate all pairs and pick the "right" combination
165 // manually. This derrived formula leads to pmul being in the range
166 // 127..254, while it has to be in 128..255. This is fixed by exchanging
167 // maximumAllowedPMul=255 with 256, which simplifies the formula to

170 if(pDiv < minimumAllowedPDiv) {
171 throw std::out_of_range("pDiv is too small and pMul too large. Check your "

173 }
174
175 // now that we know that pDiv is not negative round
176 // to the next lowest int

178
179 if(pDiv > maximumAllowedPDiv) {
180 throw std::out_of_range("pDiv is too large and pMul too small. Check your "
181 "input parameters, expecially timeToVMax.");
182 }
183
184 int pMul = static_cast<int>(reducedP * pow(2, pDiv + 3));
185
186 double maxPossibleCurrent = std::min(iMaxTMC260, physicalParameters.iMax);
187
188 // use floor to always get the current scale which is below or
189 // equal to the maximim current, never above

191
192 // For small currents < 1/32 iMaxTMC260 the calculated current scale now
193 // is negative, which is invalid. Change it to 0 and add a warning.
194 // There is nothing we can do about it, one has to allow some current in
195 // the motor, it cannot be 0.
196 if(currentScale < minimumAllowedCurrentScale) {
197 currentScale = minimumAllowedCurrentScale;

199 std::stringstream warning;
200 warning << "Current scale at minimum."
201 << " The configured maximal current of " << iMaxTMC260 / nCurrentScaleValues * (currentScale + 1)
202 << " A is larger than the maximal coil current of " << physicalParameters.iMax << " A!";
203 warnings.push_back(warning.str());
204 }

211 // the switch statement needs an int, not a double
212 int microstepsInt = static_cast<int>(microsteps);
213 if(microstepsInt != microsteps) {
214 throw std::invalid_argument("microsteps has to an integer");
215 }
216
217 switch(microstepsInt) {

222 case 4:
223 return 6;
224 case 8:
