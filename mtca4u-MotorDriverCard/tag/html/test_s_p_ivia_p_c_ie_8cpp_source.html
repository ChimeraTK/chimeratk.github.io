92void SPIviaPCIeTest::testRead() {
93 // prepare a cover datagram word
94 mtca4u::TMC429InputWord coverDatagram;
95 coverDatagram.setSMDA(mtca4u::tmc429::SMDA_COMMON);
96 coverDatagram.setADDRESS(mtca4u::tmc429::JDX_COVER_DATAGRAM);
97
98 // set the payload content and write it
99 coverDatagram.setDATA(0xAAAAAA);
100 _readWriteSPIviaPCIe->write(coverDatagram.getDataWord());
101
102 // now try to read back. prepare the coverDatagram word for reading
103 coverDatagram.setRW(mtca4u::tmc429::RW_READ);
104 coverDatagram.setDATA(0);
105
106 unsigned int readbackInt;
107 BOOST_REQUIRE_NO_THROW(readbackInt = _readWriteSPIviaPCIe->read(coverDatagram.getDataWord()));
108 // For optimised code the require is not enough. The code might continue if an
109 // exception occured (at least the compiler says so an gives a warning) Just
110 // reexecute and let the exception through
111 readbackInt = _readWriteSPIviaPCIe->read(coverDatagram.getDataWord());
112
113 mtca4u::TMC429OutputWord readbackWord;
114 readbackWord.setDataWord(readbackInt);
115 BOOST_CHECK(readbackWord.getDATA() == 0xAAAAAA);
116
117 // test the error cases. More than 2 timeouts cause an error
118 _dummyBackend->causeSpiTimeouts(true, 3);
119 BOOST_CHECK_THROW(_readWriteSPIviaPCIe->read(coverDatagram.getDataWord()), ChimeraTK::runtime_error);
120 // up to two timeouts are caught by the backend via retry
121 _dummyBackend->causeSpiTimeouts(true, 2);
122 BOOST_CHECK_NO_THROW(_readWriteSPIviaPCIe->read(coverDatagram.getDataWord()));
123
124 _dummyBackend->causeSpiTimeouts(false);
125
126 _dummyBackend->causeSpiErrors(true);
127
128 BOOST_CHECK_THROW(_readWriteSPIviaPCIe->read(coverDatagram.getDataWord()), ChimeraTK::runtime_error);
129 _dummyBackend->causeSpiErrors(false);
130}

132void SPIviaPCIeTest::testWrite() {
133 // This test is hard coded against the dummy implementation of the TCM260
134 // driver chip
135 for(uint32_t motorID = 1; motorID < mtca4u::dfmc_md22::N_MOTORS_MAX; ++motorID) {
136 // the readDriverSpiRegister is a debug function of the dummy which bypasses
137 // the SPI interface
138 unsigned int registerContent =
139 _dummyBackend->readDriverSpiRegister(motorID, mtca4u::ChopperControlData().getAddress());
140
141 mtca4u::ChopperControlData chopperControlData(registerContent + 5);
142 BOOST_CHECK_NO_THROW(_writeSPIviaPCIe->write(chopperControlData.getDataWord()));
143
144 BOOST_CHECK(_dummyBackend->readDriverSpiRegister(motorID, mtca4u::ChopperControlData().getAddress()) ==
145 chopperControlData.getPayloadData());
146
147 // test the error cases
148 _dummyBackend->causeSpiTimeouts(true);
149 chopperControlData.setPayloadData(chopperControlData.getPayloadData() + 1);
150 BOOST_CHECK_THROW(_writeSPIviaPCIe->write(chopperControlData.getDataWord()), ChimeraTK::runtime_error);
151 BOOST_CHECK(_dummyBackend->readDriverSpiRegister(motorID, mtca4u::ChopperControlData().getAddress()) ==
152 chopperControlData.getPayloadData() - 1);
153 _dummyBackend->causeSpiTimeouts(false);
154
155 _dummyBackend->causeSpiErrors(true);
156 BOOST_CHECK_THROW(_writeSPIviaPCIe->write(chopperControlData.getDataWord()), ChimeraTK::runtime_error);
157 BOOST_CHECK(_dummyBackend->readDriverSpiRegister(motorID, mtca4u::ChopperControlData().getAddress()) ==
158 chopperControlData.getPayloadData() - 1);
159 _dummyBackend->causeSpiErrors(false);
160 }
161}
