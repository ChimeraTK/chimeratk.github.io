26 BasicStepperMotor::StateMachine::StateMachine(BasicStepperMotor& stepperMotor)
27 : _stepperMotor(stepperMotor), _motorControler(stepperMotor._motorControler) {
28 _initState.setTransition(initialEvent, &_disabled, [] {});
29 _idle.setTransition(
30 moveEvent, &_moving, [this] { actionIdleToMove(); }, [this] { waitForStandstill(); });
31 _idle.setTransition(disableEvent, &_disabled, [this] { actionDisable(); });
32 _moving.setTransition(stopEvent, &_idle, [this] { actionMovetoStop(); });
33 _moving.setTransition(errorEvent, &_error, [] {});
34 _moving.setTransition(emergencyStopEvent, &_error, [this] { actionEmergencyStop(); });
35 _moving.setTransition(disableEvent, &_disabled, [this] { actionDisable(); });
36 _disabled.setTransition(enableEvent, &_idle, [this] { actionEnable(); });
37 _error.setTransition(resetToIdleEvent, &_idle, [this] { actionResetError(); });
38 _error.setTransition(resetToDisableEvent, &_disabled, [this] { actionResetError(); });
39 }

47 void BasicStepperMotor::StateMachine::waitForStandstill() {
48 if(!_motorControler->isMotorMoving()) {
49 _asyncActionActive.exchange(false);
50
51 // Move may be stopped by
52 // user requesting state transition
53 if(hasRequestedState()) {
54 moveToRequestedState();
55 return;
56 }
57
58 Event stateExitEvnt = stopEvent;
59
60 // Motor stopped by itself
61 if(!_stepperMotor.verifyMoveAction()) {
62 _stepperMotor._errorMode.exchange(Error::MOVE_INTERRUPTED);
63
64 _motorControler->setTargetPosition(_motorControler->getActualPosition());
65 stateExitEvnt = errorEvent;
66 }
67 performTransition(stateExitEvnt);
68 }
69 }

96 void BasicStepperMotor::StateMachine::actionEmergencyStop() {
97 _motorControler->setMotorCurrentEnabled(false);
98 _motorControler->setEndSwitchPowerEnabled(false);
99 actionMovetoStop();
100 _motorControler->setCalibrationTime(0);
101 _stepperMotor._calibrationMode.exchange(CalibrationMode::NONE);
102 _stepperMotor._errorMode.exchange(Error::EMERGENCY_STOP);
103 std::cerr << "Emergency stop" << std::endl;
104 }
