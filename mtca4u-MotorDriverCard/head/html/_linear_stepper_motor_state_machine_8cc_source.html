12 void LinearStepperMotorStateMachine::performCalibration() {
13 _motor._calibrationFailed.exchange(false);
14 _stopAction.exchange(false);
15 _moveInterrupted.exchange(false);
16
17 try {
18 // Local variables for calibrated position
19 int calibPositiveEndSwitchInSteps = 0;
20 int calibNegativeEndSwitchInSteps = 0;
21
22 // At least one end switch disabled -> calibration not possible
23 if(!(_motor._positiveEndSwitchEnabled.load() && _motor._negativeEndSwitchEnabled.load())) {
24 _motor._motorController->setCalibrationTime(0);
25
26 _motor._calibrationFailed.exchange(true);
27 _motor._calibrationMode.exchange(CalibrationMode::NONE);
28 }
29 else {
30 findEndSwitch(Sign::POSITIVE);
31 calibPositiveEndSwitchInSteps = _motor.getCurrentPositionInSteps();
32 findEndSwitch(Sign::NEGATIVE);
33 calibNegativeEndSwitchInSteps = _motor.getCurrentPositionInSteps();
34
35 if(_moveInterrupted.load() || _stopAction.load()) {
36 _motor._motorController->setCalibrationTime(0);
37
38 _motor._calibrationFailed.exchange(true);
39 _motor._errorMode.exchange(Error::CALIBRATION_ERROR);
40 _motor._calibrationMode.exchange(CalibrationMode::NONE);
41 }
42 else {
43 // Define positive axis with negative end switch as zero
44 calibPositiveEndSwitchInSteps = calibPositiveEndSwitchInSteps - calibNegativeEndSwitchInSteps;
45 calibNegativeEndSwitchInSteps = 0;
46
47 _motor._motorController->setCalibrationTime(time(nullptr));
48 _motor._motorController->setPositiveReferenceSwitchCalibration(calibPositiveEndSwitchInSteps);
49 _motor._motorController->setNegativeReferenceSwitchCalibration(0);
50 _motor._calibPositiveEndSwitchInSteps.exchange(calibPositiveEndSwitchInSteps);
51 _motor._calibNegativeEndSwitchInSteps.exchange(0);
52
53 _motor._calibrationMode.exchange(CalibrationMode::FULL);
54 _motor.resetMotorControllerPositions(0);
55 }
56 }
57 }
58 catch(ChimeraTK::runtime_error&) {
59 _motor._motorController->setCalibrationTime(0);
60
61 _motor._calibrationFailed.exchange(true);
62 _motor._errorMode.exchange(Error::CALIBRATION_ERROR);
63 _motor._calibrationMode.exchange(CalibrationMode::NONE);
64 }
65
66 _asyncActionActive.exchange(false);
67 }

74 void LinearStepperMotorStateMachine::findEndSwitch(Sign sign) {
75 while(!_motor.isEndSwitchActive(sign)) {
76 if(_stopAction.load() || _moveInterrupted.load()) {
77 return;
78 }
79
80 if(_motor.getTargetPositionInSteps() != _motor.getCurrentPositionInSteps() &&
81 !_motor.isPositiveReferenceActive() && !_motor.isNegativeReferenceActive()) {
82 _moveInterrupted.exchange(true);
83 return;
84 }
85 moveToEndSwitch(sign);
86 }
87 }

92 }
93 else {

95 }
