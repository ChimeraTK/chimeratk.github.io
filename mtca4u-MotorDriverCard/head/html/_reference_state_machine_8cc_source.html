89 {

141 double ReferenceStateMachine::getToleranceEndSwitch(Sign sign) {
142 double meanMeasurement = 0;
143 double stdMeasurement = 0;
144 const int N_TOLERANCE_CALC_SAMPLES = 10;
145 double measurements[N_TOLERANCE_CALC_SAMPLES];
146
147 int endSwitchPosition = getPositionEndSwitch(sign);
148
149 // Get 10 samples for tolerance calculation
150 for(double& measurement : measurements) {
151 if(_stopAction.load() || _moveInterrupted.load()) {
152 break;
153 }
154
155 // Move close to end switch
156 {
157 boost::lock_guard<boost::mutex> lck(_motor._mutex);
158 _motor._motorController->setTargetPosition(endSwitchPosition - static_cast<int>(sign) * 1000);
159 }
160 while(_motor._motorController->isMotorMoving()) {
161 std::this_thread::sleep_for(std::chrono::milliseconds(wakeupPeriodInMilliseconds));
162 }
163
164 // Check if in expected position
165 if(!_motor.verifyMoveAction()) {
166 _moveInterrupted.exchange(true);
167 break;
168 }
169
170 // Try to move beyond end switch
171 {
172 boost::lock_guard<boost::mutex> lck(_motor._mutex);
173 _motor._motorController->setTargetPosition(endSwitchPosition + static_cast<int>(sign) * 1000);
174 }
175 while(_motor._motorController->isMotorMoving()) {
176 std::this_thread::sleep_for(std::chrono::milliseconds(wakeupPeriodInMilliseconds));
177 }
178 if(!_motor.isEndSwitchActive(sign)) {
179 _moveInterrupted.exchange(true);
180 break;
181 }
182
183 // Mean calculation
184 meanMeasurement += static_cast<double>(_motor.getCurrentPositionInSteps()) / N_TOLERANCE_CALC_SAMPLES;
185 measurement = _motor.getCurrentPositionInSteps();
186 } /* for (i in [0,9]) */
187
188 // Compute variance
189 if(!(_stopAction.load() || _moveInterrupted.load())) {
190 for(unsigned int i = 0; i < N_TOLERANCE_CALC_SAMPLES; i++) {
191 stdMeasurement += ((measurements[i] - meanMeasurement) / (N_TOLERANCE_CALC_SAMPLES - 1)) *
192 (measurements[i] - meanMeasurement);
193 }
194 }
195
196 return sqrt(stdMeasurement);
197 }
