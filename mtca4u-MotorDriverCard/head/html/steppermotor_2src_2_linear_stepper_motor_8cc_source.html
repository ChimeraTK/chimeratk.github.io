73 ExitStatus LinearStepperMotor::translateAxisInSteps(int translationInSteps) {
74 LockGuard guard(_mutex);
75 if(motorActive()) {
76 return ExitStatus::ERR_SYSTEM_IN_ACTION;
77 }
78
79 translateAxisActions(translationInSteps);
80
81 // If motor has been calibrated, translate also end switch positions
82 if(this->_calibrationMode.load() == CalibrationMode::FULL) {
83 if(utility::checkIfOverflow(_calibPositiveEndSwitchInSteps.load(), translationInSteps) ||
84 utility::checkIfOverflow(_calibNegativeEndSwitchInSteps.load(), translationInSteps)) {
85 return ExitStatus::ERR_INVALID_PARAMETER;
86 }
87
88 _calibPositiveEndSwitchInSteps.fetch_add(translationInSteps);
89 _calibNegativeEndSwitchInSteps.fetch_add(translationInSteps);
90 }

204 void LinearStepperMotor::loadEndSwitchCalibration() {
205 LockGuard guard(_mutex);
206
207 if(_motorControler->getCalibrationTime() != 0) {
208 int calibPositiveEndSwitchInSteps = _motorControler->getPositiveReferenceSwitchCalibration();
209 int calibNegativeEndSwitchInSteps = _motorControler->getNegativeReferenceSwitchCalibration();
210 _calibPositiveEndSwitchInSteps.exchange(calibPositiveEndSwitchInSteps);
211 _calibNegativeEndSwitchInSteps.exchange(calibNegativeEndSwitchInSteps);
212
213 if(calibPositiveEndSwitchInSteps == std::numeric_limits<int>::max() &&
214 calibNegativeEndSwitchInSteps == std::numeric_limits<int>::min()) {
215 _calibrationMode.exchange(CalibrationMode::SIMPLE);
216 }
217 else {
218 _calibrationMode.exchange(CalibrationMode::FULL);
219 }
220 }

231 bool LinearStepperMotor::isEndSwitchActive(Sign sign) {
232 bool posActive = _motorControler->getReferenceSwitchData().getPositiveSwitchActive() == 1U;
233 bool negActive = _motorControler->getReferenceSwitchData().getNegativeSwitchActive() == 1U;
234
235 if(posActive && negActive) {
236 _errorMode.exchange(Error::BOTH_END_SWITCHES_ON);
237 _stateMachine->setAndProcessUserEvent(StateMachine::errorEvent);
238 }
239 else {
240 // The error mode was BOTH_END_SWITCHES_ON, reset the error mode. Otherwise this will stick unless the application
241 // is re-started.
242 auto expected = Error::BOTH_END_SWITCHES_ON;
243 auto replace = Error::NO_ERROR;
244 _errorMode.compare_exchange_strong(expected, replace);
245 }
246
247 if(_toleranceCalculated) {
248 if(posActive) {
249 if(std::abs(_motorControler->getActualPosition() - _calibPositiveEndSwitchInSteps.load()) >
250 3 * _tolerancePositiveEndSwitch.load()) {
251 _errorMode.exchange(Error::CALIBRATION_ERROR);
252 }
253 }
254 else if(negActive) {
255 if(std::abs(_motorControler->getActualPosition() - _calibNegativeEndSwitchInSteps.load()) >
256 3 * _toleranceNegativeEndSwitch.load()) {
257 _errorMode.exchange(Error::CALIBRATION_ERROR);
258 }
259 }
260 }
261
262 if(sign == Sign::POSITIVE) {
263 return posActive;
264 }
265
266 return negActive;
267 }
