123bool StepperMotorChimeraTKFixture::waitForState(const std::string& stateName, unsigned timeoutInSeconds) {
124 unsigned cnt = 0;
125 bool isCorrectState = false;
126 do {
127 usleep(1000);
128
129 if(_stepperMotor->getState() == stateName) {
130 isCorrectState = true;
131 break;
132 }
133 cnt++;
