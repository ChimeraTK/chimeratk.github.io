50BOOST_FIXTURE_TEST_CASE(TestRead, SPIviaPCIeTestFixture) {
51 // prepare a cover datagram word
52 mtca4u::TMC429InputWord coverDatagram;
53 coverDatagram.setSMDA(mtca4u::tmc429::SMDA_COMMON);
54 coverDatagram.setADDRESS(mtca4u::tmc429::JDX_COVER_DATAGRAM);
55
56 // set the payload content and write it
57 coverDatagram.setDATA(0xAAAAAA);
58 _readWriteSPIviaPCIe->write(int32_t(coverDatagram.getDataWord()));
59
60 // now try to read back. prepare the coverDatagram word for reading
61 coverDatagram.setRW(mtca4u::tmc429::RW_READ);
62 coverDatagram.setDATA(0);
63
64 unsigned int readbackInt;
65 BOOST_REQUIRE_NO_THROW(readbackInt = _readWriteSPIviaPCIe->read(coverDatagram.getDataWord()));
66 // For optimised code the require is not enough. The code might continue if an
67 // exception occured (at least the compiler says so an gives a warning) Just
68 // reexecute and let the exception through
69 readbackInt = _readWriteSPIviaPCIe->read(int32_t(coverDatagram.getDataWord()));
70
71 mtca4u::TMC429OutputWord readbackWord;
72 readbackWord.setDataWord(readbackInt);
73 BOOST_CHECK(readbackWord.getDATA() == 0xAAAAAA);
74
75 // test the error cases. More than 2 timeouts cause an error
76 _dummyBackend->causeSpiTimeouts(true, 3);
77 BOOST_CHECK_THROW(_readWriteSPIviaPCIe->read(coverDatagram.getDataWord()), ChimeraTK::runtime_error);
78 // up to two timeouts are caught by the backend via retry
79 _dummyBackend->causeSpiTimeouts(true, 2);
80 BOOST_CHECK_NO_THROW(_readWriteSPIviaPCIe->read(coverDatagram.getDataWord()));
81
82 _dummyBackend->causeSpiTimeouts(false);
83
84 _dummyBackend->causeSpiErrors(true);
85
86 BOOST_CHECK_THROW(_readWriteSPIviaPCIe->read(coverDatagram.getDataWord()), ChimeraTK::runtime_error);
87 _dummyBackend->causeSpiErrors(false);
