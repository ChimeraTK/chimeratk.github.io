356  unsigned int MotorDriverCardTest::asciiToInt(std::string text) {
357  unsigned int returnValue(0);
358  int position = 0;
359  std::string::reverse_iterator characterIter = text.rbegin();
360  for(/* emtpy */; (position < 4) && (characterIter != text.rend()); ++position, ++characterIter) {

362  returnValue |= charAsInt << 4 * position;
363  }
364 
365  return returnValue;
