69 bool MotorControlerDummy::isStepping() {

71 if(_targetPosition == _currentPosition) {

220 void MotorControlerDummy::roundToNextFullStep(int& targetPosition) {
221 int delta = targetPosition - _currentPosition;
222 int deltaMicroStep = delta * _userMicroStepSize;
223 unsigned int actualMicroStepCount = ((_currentPosition * _userMicroStepSize) - 1) & 0x3FF;
224 unsigned int newActualMicroStepCount = (actualMicroStepCount + deltaMicroStep) & 0x3FF;
225 unsigned int distanceToPreviousFullStep = (newActualMicroStepCount + 1) % 256;
226 if(distanceToPreviousFullStep < 128) {
227 targetPosition = targetPosition - distanceToPreviousFullStep / _userMicroStepSize;
228 }
229 else {
230 targetPosition = targetPosition + (256 - distanceToPreviousFullStep) / _userMicroStepSize;
231 }
232 }

365 void MotorControlerDummy::moveTowardsTarget(
366 float fraction, bool block, bool bothEndSwitchesAlwaysOn, bool zeroPositions) {
367 LockGuard guard(_motorControllerDummyMutex);
368 if(zeroPositions) {
369 _hardwarePosition = 0;
370 _targetPosition = 0;
371 _currentPosition = 0;
372 _positiveEndSwitchEnabled = true;
373 _negativeEndSwitchEnabled = true;
374 }
375
376 _bothEndSwitchesAlwaysOn = bothEndSwitchesAlwaysOn;
377
378 _blockMotor = block;
379 if(_blockMotor) {
380 return;
381 }
382
383 // check if the motor is should to be moving if it was enabled
384 // otherwise return immediately
385 if(!isStepping()) {
386 return;
387 }
388
389 // range check for the input parameter 'fraction', just force it to 0..1
390 if(fraction <= 0) {
391 return;
392 }
393 if(fraction > 1) {
394 fraction = 1.;
395 }
396
397 // calculate the new abolute target position for this move operation
398 int relativeSteps = (_targetPosition - _currentPosition) * fraction;
399 int absoluteTargetInThisMove = _hardwarePosition + relativeSteps;
400 int targetInThisMove = _currentPosition + relativeSteps;
401
402 if(!_motorCurrentEnabled) {
403 // the motor is stepping to the target position, but the actual positon
404 // does not change. Thus no end switch will be hit.
405 _currentPosition = targetInThisMove;
406 return;
407 }
408
409 // ok, so the motor is really moving.
410
411 // check whether an end switch would be hit
412 if(_positiveEndSwitchEnabled && (absoluteTargetInThisMove > _positiveEndSwitchHardwarePosition)) {
413 absoluteTargetInThisMove = _positiveEndSwitchHardwarePosition;
414 // also the current positon stops if an end switch is reached
415 int stepsToEndSwitch = _positiveEndSwitchHardwarePosition - _hardwarePosition;
416 targetInThisMove = _currentPosition + stepsToEndSwitch;
417 }
418 if(_negativeEndSwitchEnabled && (absoluteTargetInThisMove < _negativeEndSwitchHardwarePosition)) {
419 absoluteTargetInThisMove = _negativeEndSwitchHardwarePosition;
420 // also the current positon stops if an end switch is reached
421 int stepsToEndSwitch = _negativeEndSwitchHardwarePosition - _hardwarePosition;
422 targetInThisMove = _currentPosition + stepsToEndSwitch;
423 }
424
425 // finally 'move' the motor
426 _hardwarePosition = absoluteTargetInThisMove;
427 _currentPosition = targetInThisMove;
428 }
