222  void MotorControlerDummy::roundToNextFullStep(int& targetPosition) {
223  int delta = targetPosition - _currentPosition;
224  int deltaMicroStep = delta * _userMicroStepSize;
225  unsigned int actualMicroStepCount = ((_currentPosition * _userMicroStepSize) - 1) & 0x3FF;
226  unsigned int newActualMicroStepCount = (actualMicroStepCount + deltaMicroStep) & 0x3FF;
227  unsigned int distanceToPreviousFullStep = (newActualMicroStepCount + 1) % 256;
228  if(distanceToPreviousFullStep < 128) {
229  targetPosition = targetPosition - distanceToPreviousFullStep / _userMicroStepSize;
230  }
231  else {
232  targetPosition = targetPosition + (256 - distanceToPreviousFullStep) / _userMicroStepSize;
233  }
234  }

367  void MotorControlerDummy::moveTowardsTarget(
368  float fraction, bool block, bool bothEndSwitchesAlwaysOn, bool zeroPositions) {
369  LockGuard guard(_motorControllerDummyMutex);
370  if(zeroPositions) {
371  _absolutePosition = 0;
372  _targetPosition = 0;
373  _currentPosition = 0;
374  _positiveEndSwitchEnabled = true;
375  _negativeEndSwitchEnabled = true;
376  }
377 
378  _bothEndSwitchesAlwaysOn = bothEndSwitchesAlwaysOn;
379 
380  _blockMotor = block;
381  if(_blockMotor) {
382  return;
383  }
384 
385  // check if the motor is should to be moving if it was enabled
386  // otherwise return immediately
387  if(!isStepping()) {
388  return;
389  }
390 
391  // range check for the input parameter 'fraction', just force it to 0..1
392  if(fraction <= 0) {
393  return;
394  }
395  if(fraction > 1) {
396  fraction = 1.;
397  }
398 
399  // calculate the new abolute target position for this move operation
400  int relativeSteps = (_targetPosition - _currentPosition) * fraction;
401  int absoluteTargetInThisMove = _absolutePosition + relativeSteps;
402  int targetInThisMove = _currentPosition + relativeSteps;
403 
404  if(!_motorCurrentEnabled) {
405  // the motor is stepping to the target position, but the actual positon
406  // does not change. Thus no end switch will be hit.
407  _currentPosition = targetInThisMove;
408  return;
409  }
410 
411  // ok, so the motor is really moving.
412 
413  // check whether an end switch would be hit
414  if(_positiveEndSwitchEnabled && (absoluteTargetInThisMove > _positiveEndSwitchPosition)) {
415  absoluteTargetInThisMove = _positiveEndSwitchPosition;
416  // also the current positon stops if an end switch is reached
417  int stepsToEndSwitch = _positiveEndSwitchPosition - _absolutePosition;
418  targetInThisMove = _currentPosition + stepsToEndSwitch;
419  }
420  if(_negativeEndSwitchEnabled && (absoluteTargetInThisMove < _negativeEndSwitchPosition)) {
421  absoluteTargetInThisMove = _negativeEndSwitchPosition;
422  // also the current positon stops if an end switch is reached
423  int stepsToEndSwitch = _negativeEndSwitchPosition - _absolutePosition;
424  targetInThisMove = _currentPosition + stepsToEndSwitch;
425  }
426 
427  // finally 'move' the motor
428  _absolutePosition = absoluteTargetInThisMove;
429  _currentPosition = targetInThisMove;
430  }
