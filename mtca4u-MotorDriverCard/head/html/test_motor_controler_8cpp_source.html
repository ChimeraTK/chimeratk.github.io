479 void MotorControlerTest::testGetReferenceSwitchData(boost::shared_ptr<MotorDriverCardImpl> motorDriverCard) {
480 // This approach is intentinally clumsy and manual in order not to use the
481 // same algorithm as in the implementation, but still be felxible if the
482 // register content changes.
483 unsigned int referenceWord = 0; // setting to 0 to avoid compiler warning in default case.
484
485 switch(_motorControler->getID()) {
486 case 0:
487 referenceWord = motorDriverCard->getReferenceSwitchData().getDATA() & 0x3;
488 break;
489 case 1:
490 referenceWord = ((motorDriverCard->getReferenceSwitchData().getDATA() & 0xC) >> 2);
491 break;
492 case 2:
493 referenceWord = ((motorDriverCard->getReferenceSwitchData().getDATA() & 0x30) >> 4);
494 break;
495 default:
496 BOOST_FAIL("Invalid MotorID. Either there is something wrong in the code "
497 "or the test has to be adapted.");
498 }
499
500 BOOST_CHECK(_motorControler->getReferenceSwitchData().getSwitchesActiveWord() == referenceWord);
501 BOOST_CHECK(_motorControler->getReferenceSwitchData().getPositiveSwitchEnabled() ==
502 !_motorControler->getReferenceConfigAndRampModeData().getDISABLE_STOP_R());
503 BOOST_CHECK(_motorControler->getReferenceSwitchData().getNegativeSwitchEnabled() ==
504 !_motorControler->getReferenceConfigAndRampModeData().getDISABLE_STOP_L());
505 }
