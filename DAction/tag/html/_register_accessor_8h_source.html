45  template<typename ConvertedDataType>
46  void read(ConvertedDataType* convertedData, size_t nWords = 1, uint32_t wordOffsetInRegister = 0) const {
47  if(nWords == 0) return;
48  // obtain the accessor through the handler
49  auto& accessorHandler = boost::fusion::at_key<ConvertedDataType>(_convertingAccessorHandlers.table);
50  accessorHandler.checkAndResize(nWords, wordOffsetInRegister, false /*not raw*/, _backend, _registerPathName);
51  // now we can be sure that the accessor is valid and has enough memory to
52  // perform the memcpy
53  accessorHandler.accessor->read();
54  // copy data to target buffer. The accessor might not start at index 0, so
55  // we have to calculate the correct offset inside the accessor
56  memcpy(convertedData,
57  accessorHandler.accessor->accessChannel(0).data() + wordOffsetInRegister - accessorHandler.beginIndex,
58  nWords * sizeof(ConvertedDataType));
59  }

156  template<class UserType>
157  class AccessorHandler {
158  public:
159  AccessorHandler() : beginIndex(0), endIndex(0) {}
160 
161  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> accessor;
162  size_t beginIndex; // the first index of the accessor;
163  size_t endIndex; // the end index (first index after the valid range, liken
164  // end iterator)
165  void checkAndResize(size_t nWords, size_t wordOffsetInRegister, bool isRaw,
166  boost::shared_ptr<ChimeraTK::DeviceBackend> const& backend, ChimeraTK::RegisterPath const& registerPathName) {
167  // no need for out of register size checking. getRegisterAccessor does the
168  // job for us and throws in case of error.
169  if((wordOffsetInRegister < beginIndex) || (wordOffsetInRegister + nWords > (endIndex) || !accessor)) {
170  // we have to (re)allocate
171  size_t newBeginIndex;
172  if(!accessor) {
173  newBeginIndex = wordOffsetInRegister;
174  }
175  else {
176  newBeginIndex = std::min(wordOffsetInRegister, beginIndex);
177  }
178  size_t newEndIndex = std::max(wordOffsetInRegister + nWords, endIndex);
179  accessor =
180  backend->getRegisterAccessor<UserType>(registerPathName, newEndIndex - newBeginIndex, newBeginIndex,
181  (isRaw ? ChimeraTK::AccessModeFlags{ChimeraTK::AccessMode::raw} :
182  ChimeraTK::AccessModeFlags{})); // 0 offset, raw
183  // we have to update the accessor with the current hardware information.
184  // It might be that a partial write is going to happen.
185  accessor->read();
186  // only if creating the new accessor succeeds we change the index
187  // bookkeeping variables
188  beginIndex = newBeginIndex;
189  endIndex = newEndIndex;
190  }
191  }
192  };
