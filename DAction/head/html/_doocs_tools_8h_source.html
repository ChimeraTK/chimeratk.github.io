67  template<class DOOCS_T>
68  bool checkEgu(DOOCS_T& property, std::string expectedEguText, int expectedLinLog = 0, float expectedStart = 0.,
69  float expectedEnd = 0.) {
70  D_hist* hist = property.get_histPointer();
71  if(!hist) {
72  return false;
73  }
74 
75  // nice doocs interface: we have to create buffers for all properties to read them into
76  int linLog;
77  float start, end;
78  // We have accomodate a null byte plus at least one extra byte in case the actual EGU is longer
79  // than the expected one, so the comparison will detect a difference
80  std::vector<char> stringbuffer;
81  stringbuffer.resize(expectedEguText.length() + 2);
82  time_t t;
83 
84  hist->plot_value(&linLog, &start, &end, &t, stringbuffer.data(), expectedEguText.length() + 2);
85  // bug in older doocs versions: there in a character '0', not a null byte at the end. This fix does not harm even
86  // with the new, correct version
87  stringbuffer[expectedEguText.length() + 1] = 0; // just to be sure there is a null byte in the buffer
88 
89  return expectedEguText.compare(stringbuffer.data()) == 0 && linLog == expectedLinLog && start == expectedStart &&
90  end == expectedEnd;
91  }
