51 void Logger::sendMessage(const std::string& msg, const logging::LogLevel& level) {
52  if(message.isInitialised()) {
53  while(!msg_buffer.empty()) {
54  message = msg_buffer.front();
55  message.write();
56  msg_buffer.pop();
57  }
58  message = std::to_string(level) + msg + "\n";
59  message.write();
60  // set emtpy message to let the LoggingModule know that someone called writeAll() without sending a message
61  message = std::to_string(logging::LogLevel::INTERNAL) + msg + "\n";
62  }
63  else {
64  // only use the buffer until ctk initialized the process variables
65  msg_buffer.push(std::to_string(level) + msg + "\n");
66  }

77 LoggingModule::LoggingModule(ctk::ModuleGroup* owner, const std::string& name, const std::string& description,
78  const std::string& loggingTag, const std::unordered_set<std::string>& tags)
79 : ctk::ApplicationModule(owner, name, description, tags), _loggingTag(loggingTag) {
80  auto model = dynamic_cast<ctk::ModuleGroup*>(_owner)->getModel();
81  auto neighbourDir = model.visit(ctk::Model::returnDirectory, ctk::Model::getNeighbourDirectory,
82  ctk::Model::returnFirstHit(ctk::Model::DirectoryProxy{}));
83  std::vector<ctk::Model::ProcessVariableProxy> pvs;
84  auto found = neighbourDir.visitByPath(".", [&](auto sourceDir) {
85  sourceDir.visit([&](auto pv) { pvs.emplace_back(pv); }, ctk::Model::breadthFirstSearch,
86  ctk::Model::keepProcessVariables && ctk::Model::keepTag(_loggingTag));
87  });
88 
89  for(auto pv : pvs) {
90  addVariableFromModel(pv);
91  }
92  if(!found) {
93  throw ChimeraTK::logic_error("Error during LoggingModule construction! Model is not valid.");
94  }
95  if(sources.empty()) {
96  throw ctk::logic_error("LoggingModule did not find any module that uses a Logger. Maybe tags used by the Logger "
97  "and the LoggingModule do not match.");
98  }
99 }

101 void LoggingModule::addVariableFromModel(const ctk::Model::ProcessVariableProxy& pv) {
102  ctk::RegisterPath name{pv.getFullyQualifiedPath()};
103  auto tag = pv.getTags();
104  // check for the logging tag
105  if(!tag.count(_loggingTag)) return;
106  ctk::RegisterPath path(name);
107  // remove variable name
108  path = path--;
109  bool found = false;
110  for(const auto& msg : sources) {
111  if(msg == path) {
112  found = true;
113  }
114  }
115  // in case PVs of the Logging module use the same tag
116  if(!found && (name.endsWith("alias") || name.endsWith("message"))) {
117  sources.emplace_back(MessageSource(path, this));
118  }
119 }

121 void LoggingModule::broadcastMessage(std::string msg, const bool& isError) {
122  if(msg.back() != '\n') {
123  msg.append("\n");
124  }
125 
126  std::string tmpLog = (std::string)logTail;
127  if(tailLength == 0 && messageCounter > 20) {
128  messageCounter--;
129  tmpLog = tmpLog.substr(tmpLog.find_first_of("\n") + 1, tmpLog.length());
130  }
131  else if(tailLength > 0) {
132  while(messageCounter >= tailLength) {
133  messageCounter--;
134  tmpLog = tmpLog.substr(tmpLog.find_first_of("\n") + 1, tmpLog.length());
135  }
136  }
137  if(targetStream == 0 || targetStream == 2) {
138  if(isError)
139  std::cerr << msg;
140  else
141  std::cout << msg;
142  }
143  if(targetStream == 0 || targetStream == 1) {
144  if(file->is_open()) {
145  (*file) << msg.c_str();
146  file->flush();
147  }
148  }
149  tmpLog.append(msg);
150  messageCounter++;
151  logTail = tmpLog;
152  logTail.write();
153 }

155 void LoggingModule::mainLoop() {
156  file.reset(new std::ofstream());
157  messageCounter = 0;
158  std::stringstream greeter;
159  greeter << getName() << " " << getTime() << "There are " << sources.size()
160  << " modules registered for logging:" << std::endl;
161  broadcastMessage(greeter.str());
162  for(auto module = sources.begin(); module != sources.end(); module++) {
163  broadcastMessage(std::string("\t - ") + module->sendingModule);
164  id_list[module->_msg.getId()] = &(*module);
165  }
166  auto group = readAnyGroup();
167  std::string msg;
168  MessageSource* currentSender;
169  LogLevel level;
170 
171  while(1) {
172  // we skip the initial value since it is empty anyway and set in Logger::prepare
173  auto id = group.readAny();
174  if(id_list.count(id) == 0) {
175  throw ctk::logic_error("Cannot find element id"
176  "when updating logging variables.");
177  }
178  try {
179  currentSender = id_list.at(id);
180  msg = (std::string)(currentSender->_msg);
181  }
182  catch(std::out_of_range& e) {
183  throw ctk::logic_error("Cannot find element id"
184  "when updating logging variables.");
185  }
186  try {
187  level = static_cast<LogLevel>(std::strtoul(&msg.at(0), NULL, 0));
188  }
189  catch(std::out_of_range& e) {
190  throw ctk::logic_error("Cannot find message level"
191  "when updating logging variables.");
192  }
193  // if log level is INTERNAL someone called writeAll() in a module containing the Logger -> ignore
194  if(level == LogLevel::INTERNAL) {
195  continue;
196  }
197  if(targetStream == 4) continue;
198  LogLevel setLevel = static_cast<LogLevel>((uint)logLevel);
199  std::string tmpStr = msg;
200  // remove message level
201  tmpStr = tmpStr.substr(1, tmpStr.size());
202  std::stringstream ss;
203  auto senderAlias = (std::string)(currentSender->_alias);
204  if(senderAlias.empty()) {
205  ss << level << getName() << ":" << currentSender->sendingModule << " " << getTime() << tmpStr;
206  }
207  else {
208  ss << level << getName() << ":" << senderAlias << " " << getTime() << tmpStr;
209  }
210  if(targetStream == 0 || targetStream == 1) {
211  if(!((std::string)logFile).empty() && !file->is_open()) {
212  std::stringstream ss_file;
213  file->open((std::string)logFile, std::ofstream::out | std::ofstream::app);
214  if(!file->is_open() && setLevel <= LogLevel::ERROR) {
215  ss_file << LogLevel::ERROR << getName() << " " << getTime()
216  << "Failed to open log file for writing: " << (std::string)logFile << std::endl;
217  broadcastMessage(ss_file.str(), true);
218  }
219  else if(file->is_open() && setLevel <= LogLevel::INFO) {
220  ss_file << LogLevel::INFO << getName() << " " << getTime()
221  << "Opened log file for writing: " << (std::string)logFile << std::endl;
222  broadcastMessage(ss_file.str());
223  }
224  }
225  }
226  if(level >= setLevel) {
227  if(level < LogLevel::ERROR)
228  broadcastMessage(ss.str());
229  else
230  broadcastMessage(ss.str(), true);
231  }
232  }
233 }
