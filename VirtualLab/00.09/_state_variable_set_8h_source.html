141  inline const STATE& getState(VirtualTime time, bool forceNoInterpolation = false) {
142 #ifndef ENABLE_EXPERIMENTAL_FEATURES
143  (void)forceNoInterpolation; // avoid warning
144 #endif
145  // check if time is the current time and return the latest element
146  if(time >= currentTime && time < currentTime + validityPeriod) {
147  return getLatestState();
148  }
149  // search in buffer: find the first element after the requested time
150  auto it = buffer.upper_bound(time);
151  if(it == buffer.begin()) {
152  std::stringstream s;
153  s << "Value request is too far into the past: ";
154  s << "requested time = " << time << ", oldest history = " << buffer.begin()->first;
155  s << ", current time = " << currentTime;
156  throw ChimeraTK::logic_error(s.str());
157  }
158  // if this is end(), a new value needs to be computed
159  if(it == buffer.end()) {
160 #ifdef ENABLE_EXPERIMENTAL_FEATURES
161  if(!forceNoInterpolation) {
162  return getValueInterpolated(time, buffer.rbegin()->first);
163  }
164  else {
165 #endif
166  return getValueFromCallback(time);
167 #ifdef ENABLE_EXPERIMENTAL_FEATURES
168  }
169 #endif
170  }
171  // decrement to get the most recent sample before the requested time
172  --it;
173  // if sample is too old, request one via callback
174  if(time >= it->first + validityPeriod) {
175 #ifdef ENABLE_EXPERIMENTAL_FEATURES
176  if(!forceNoInterpolation) {
177  return getValueInterpolated(time, it->first);
178  }
179  else {
180 #endif
181  return getValueFromCallback(time);
182 #ifdef ENABLE_EXPERIMENTAL_FEATURES
183  }
184 #endif
185  }
186  // return value from buffer
187  return it->second;
188  }

234  inline void truncateFuture(VirtualTime time) {
235  if(time < currentTime) {
236  auto firstToDelete = buffer.upper_bound(time);
237  // Note: getState() already ensures that the request does not go before

257  inline const STATE& getValueInterpolated(VirtualTime timeRequested, VirtualTime previousStep) {
258 #ifndef ENABLE_EXPERIMENTAL_FEATURES
259  class ExperimentalFeatureNotEnabled {};
260  throw ExperimentalFeatureNotEnabled();
261 #endif
262  // no interpolation or requested time is later than the interpolation period
263  // (maxGap)
264  if(!enableInterpolation || timeRequested >= previousStep + maxGap) {
265  return getValueFromCallback(timeRequested);
266  }
267 
268  // save interpolated state into the buffer and return it
269  auto state1 = getState(previousStep, true);
270  auto state2 = getState(previousStep + maxGap, true);
271  buffer[timeRequested] = interpolate(state1, state2, previousStep, previousStep + maxGap, timeRequested);
272  return buffer[timeRequested];
273  }

277  inline const STATE& getValueFromCallback(VirtualTime time) {
278  // If request goes into the past (w.r.t. currentTime), erase newer states to
279  // force recomputing them.
280  truncateFuture(time);
281 
282  // Compute the number of steps necessary to reach the requested time,
283  // obaying hugeGap. The last step at the requested time itself is not
284  // computed here, since maxGap shall be obeyed in the last gap (see below).
285  // Note: this is effectively rounding-up the integer division
286  // (time-CurrentTime)/hugeGap
287  unsigned int nHugeSteps = (time - currentTime - 1) / hugeGap + 1;
288 
289  // Loop over the time in nHugeSteps steps. The first step is potentially
290  // smaller, all consequtive steps are maxGap big. Omit the last step, so the
291  // requested time is not yet reached.
292  for(unsigned int i = 0; i < nHugeSteps - 1; i++) {
293  VirtualTime t = time - (nHugeSteps - 1) * hugeGap + i * hugeGap;
294 
295  // obtain the new state
296  STATE state = compute(t);
297 
298  // save value into buffer and update the current time
299  buffer[t] = state;
300  currentTime = t;
301  }
302 
303  // Compute the number of steps necessary to reach the requested time,
304  // obaying maxGap. Note: this is effectively rounding-up the integer
305  // division (time-CurrentTime)/maxGap
306  unsigned int nSteps = (time - currentTime - 1) / maxGap + 1;
307 
308  // Loop over the time in nSteps steps. The first step is potentially
309  // smaller, all consequtive steps are maxGap big.
310  for(unsigned int i = 0; i < nSteps; i++) {
311  VirtualTime t = time - (nSteps - 1) * maxGap + i * maxGap;
312 
313  // obtain the new state
314  STATE state = compute(t);
315 
316  // save value into buffer and update the current time
317  buffer[t] = state;
318  currentTime = t;
319  }
320 
321  // remove old history
322  truncatePast();
323 
324  // return the new state (do not use buffer[], as it will silently insert new
325  // objects if something went wrong)
326  return buffer.find(time)->second;
327  }
