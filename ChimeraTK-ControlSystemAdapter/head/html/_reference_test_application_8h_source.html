43 const std::string& typeNamePrefix, int arrayLen) {
44 toDeviceScalar = processVariableManager->createProcessArray<DataType>(
45 ChimeraTK::SynchronizationDirection::controlSystemToDevice, typeNamePrefix + "/TO_DEVICE_SCALAR", 1);
46 fromDeviceScalar = processVariableManager->createProcessArray<DataType>(
47 ChimeraTK::SynchronizationDirection::deviceToControlSystem, typeNamePrefix + "/FROM_DEVICE_SCALAR", 1);
48
49 if constexpr(!std::is_same_v<DataType, ChimeraTK::Void>) {
50 toDeviceArray = processVariableManager->createProcessArray<DataType>(
51 ChimeraTK::SynchronizationDirection::controlSystemToDevice, typeNamePrefix + "/TO_DEVICE_ARRAY", arrayLen);
52 fromDeviceArray = processVariableManager->createProcessArray<DataType>(
53 ChimeraTK::SynchronizationDirection::deviceToControlSystem, typeNamePrefix + "/FROM_DEVICE_ARRAY", arrayLen);
54 dataTypeConstant = processVariableManager->createProcessArray<DataType>(
55 ChimeraTK::SynchronizationDirection::deviceToControlSystem, typeNamePrefix + "/DATA_TYPE_CONSTANT", 1);
56 constantArray = processVariableManager->createProcessArray<DataType>(
57 ChimeraTK::SynchronizationDirection::deviceToControlSystem, typeNamePrefix + "/CONSTANT_ARRAY", arrayLen);
58 }
59 else {
60 (void)arrayLen; // suppress warning
61 }
62 double typeIdentifyingConstant = 0;
63 if(typeid(DataType) == typeid(std::string)) {
64 typeIdentifyingConstant = 42;
65 }
66 else if(typeid(DataType) == typeid(ChimeraTK::Boolean)) {
67 typeIdentifyingConstant = true;
68 }
69 else {
70 if(std::numeric_limits<DataType>::is_integer) {
71 if(std::numeric_limits<DataType>::is_signed) {
72 // signed int
73 typeIdentifyingConstant = -static_cast<int64_t>(sizeof(DataType));
74 }
75 else {
76 // unsigned int
77 typeIdentifyingConstant = sizeof(DataType);
78 }
79 }
80 else {
81 typeIdentifyingConstant = 1. / sizeof(DataType);
82 }
83 }
84
85 if constexpr(!std::is_same_v<DataType, ChimeraTK::Void>) {
86 dataTypeConstant->accessData(0) = ChimeraTK::toType<DataType>(typeIdentifyingConstant);
87
88 for(size_t i = 0; i < constantArray->accessChannel(0).size(); ++i) {
89 constantArray->accessChannel(0)[i] = ChimeraTK::toType<DataType>(typeIdentifyingConstant * i * i);

95 failedTransfers.clear();
96 if(toDeviceScalar->readLatest()) {
97 fromDeviceScalar->accessChannel(0) = toDeviceScalar->accessChannel(0);
98 fromDeviceScalar->setDataValidity(validity);

100 if(isDataLost) {
101 failedTransfers.emplace_back(toDeviceScalar->getName());

109 }
110 fromDeviceArray->setDataValidity(validity);
111 auto isDataLost = fromDeviceArray->write(version.value_or(ChimeraTK::VersionNumber()));
112 if(isDataLost) {
113 failedTransfers.emplace_back(toDeviceArray->getName());

115 }

117 if(!failedTransfers.empty()) {
118 std::cout << "WARNING: Data loss in referenceTestApplication for Process Variables: " << std::endl;

293inline bool ReferenceTestApplication::runMainLoopOnce() {
294 mainLoopExecutionRequested() = true;
295 do {
296 mainLoopMutex().unlock();
297 boost::this_thread::sleep_for(boost::chrono::microseconds(1));
298 mainLoopMutex().lock();
299 // Loop until the execution requested flag has not been reset.
300 // This is the sign that the loop actually has been performed.
301 } while(mainLoopExecutionRequested());
302
303 auto isSuccessful = [](bool initialState, auto mapEntry) {
304 bool success = mapEntry.second.failedTransfers.empty();
305 return initialState && success;
306 };
307 auto& holderMap = *dynamic_cast<ReferenceTestApplication&>(ApplicationBase::getInstance())._holderMap;
308 return boost::fusion::fold(holderMap, true, isSuccessful);
309}
