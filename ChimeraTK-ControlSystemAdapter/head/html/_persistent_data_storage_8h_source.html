168 size_t PersistentDataStorage::registerVariable(ChimeraTK::RegisterPath const& name, size_t nElements, bool fromFile) {
169 // check if already existing
170 auto position = std::find(_variableNames.begin(), _variableNames.end(), name);
171
172 size_t id = position - _variableNames.begin();
173 std::lock_guard<std::mutex> lock(_queueReadMutex);
174
175 // create new element
176 if(position == _variableNames.end()) {
177 // output information
178 if(!fromFile) {
179 std::cout << "PersistentDataStorage: registering new variable " << name << std::endl;
180 }
181
182 // store name and type
183 _variableNames.push_back(name);
184 _variableTypes.push_back(&typeid(DataType));
185
186 // set flag whether this variable has been registered from the application
187 _variableRegisteredFromApp.push_back(!fromFile);
188
189 // create value vector
190 id = _variableNames.size() - 1;
191 std::vector<DataType>& value = boost::fusion::at_key<DataType>(_dataMap.table)[id].readLatest();
192 value.resize(nElements);
193
194 // return id
195 return id;
196 }
197 // replace element (changed data type)
198 if(boost::fusion::at_key<DataType>(_dataMap.table).count(id) == 0) {
199 std::cout << "PersistentDataStorage: changing type of variable " << name << std::endl;
200 assert(_variableTypes[id] != &typeid(DataType));
201 assert(!fromFile);
202
203 // remove value vector from old type
204 boost::fusion::for_each(_dataMap.table, RegisterVariableOldTypeRemover({_variableTypes[id], id}));
205
206 // update type
207 _variableTypes[id] = &typeid(DataType);
208
209 // update flag that this variable has been registered from the application
210 _variableRegisteredFromApp[id] = true;
211
212 // create value vector
213 std::vector<DataType>& value = boost::fusion::at_key<DataType>(_dataMap.table)[id].readLatest();
214 value.resize(nElements);
215
216 // return id
217 return id;
218 }
219 // return existing id
220 assert(!fromFile);
221
222 // update flag that this variable has been registered from the application
223 _variableRegisteredFromApp[id] = true;
224
225 // check if resize required
226 std::vector<DataType>& value = boost::fusion::at_key<DataType>(_dataMap.table)[id].readLatest();
227 if(value.size() != nElements) {
228 std::cout << "PersistentDataStorage: changing size of variable " << name << " from " << value.size() << " to "
229 << nElements << std::endl;
230 value.resize(nElements);
231 }
232
233 return id;
234 }
