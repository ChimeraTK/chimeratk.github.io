15 template<class T, class U>
16 T toType(U input) {
17 if constexpr(std::is_same_v<T, std::string>) {
18 return std::to_string(input);
19 }
20 else if constexpr(std::is_same_v<T, ChimeraTK::Void>) {
21 return {};
22 }
23 else if constexpr(std::is_integral_v<T> && !std::is_integral_v<U>) {
24 // Converting floating point values into integers is undefined behaviour, if the range of the target value range
25 // is exceeded. Hence we first convert into int64_t before we convert into the target tyoe, so we get the "roll
26 // over" behaviour of the narrowing integer conversion as wanted.
27 assert(input < U(std::numeric_limits<int64_t>::max()));
28 assert(input > U(std::numeric_limits<int64_t>::lowest()));
29 return T(int64_t(input));
30 }
31 else {
32 return T(input);
33 }
34 }
