18 ApplicationBase& ApplicationFactoryBase::getApplicationInstance() {
19 // First check whether the application instance exists before going through the
20 // effort of getting the factory lock.
21 if(_applicationInstance) {
22 return *_applicationInstance;
23 }
24
25 std::lock_guard<std::recursive_mutex> lock(ApplicationBase::instanceMutex);
26 // Re-check that the application has not been created in the mean time (race condition)
27 if(_applicationInstance) {
28 return *_applicationInstance;
29 }
30
31 if(!_factoryFunction) {
32 throw ChimeraTK::logic_error("No instance of ApplicationFactory created, but "
33 "ApplicationFactoryBase::getApplicationInstance() called.");
34 }
35 _factoryIsCreating = true;
36 _factoryFunction();
37 _factoryIsCreating = false;
38
39 return *_applicationInstance;
40 }
