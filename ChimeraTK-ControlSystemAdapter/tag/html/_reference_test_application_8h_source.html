308inline bool ReferenceTestApplication::runMainLoopOnce() {
309 mainLoopExecutionRequested() = true;
310 do {
311 mainLoopMutex().unlock();
312 boost::this_thread::sleep_for(boost::chrono::microseconds(1));
313 mainLoopMutex().lock();
314 // Loop until the execution requested flag has not been reset.
315 // This is the sign that the loop actually has been performed.
316 } while(mainLoopExecutionRequested());
317
318 auto isSuccessful = [](bool initialState, auto mapEntry) {
319 bool success = mapEntry.second.failedTransfers.empty();
320 return initialState && success;
321 };
322 auto& holderMap = *dynamic_cast<ReferenceTestApplication&>(ApplicationBase::getInstance())._holderMap;
323 return boost::fusion::fold(holderMap, true, isSuccessful);
324}
