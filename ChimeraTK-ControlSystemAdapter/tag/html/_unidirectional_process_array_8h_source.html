170 struct SharedState {
171 SharedState(size_t numberOfBuffers, size_t bufferLength) : queue(numberOfBuffers) {
172 // fill the internal buffers of the queue
173 for(size_t i = 0; i < numberOfBuffers + 1; ++i) {
174 Buffer b0(bufferLength);
175 Buffer b1(bufferLength);
176 queue.push(std::move(b0));
177 queue.pop(b1); // here the buffer b1 gets swapped into the queue
178 }
179 }
180
181 // Create copy of shared state. Sice the future_queue itself already
182 // supports sharing and we have nothing else in our shared state, we do not
183 // need to store our share state as a pointer but we can "copy" it and the
184 // copies will stay linked.
185 SharedState(const SharedState& other) : queue(other.queue) {}

236 bool checkThreadSafety() {
237 // Only perform check if enableThreadSafetyCheck is enabled
238 if(!detail::processArrayEnableThreadSafetyCheck) {
239 return true;
240 }
241
242 // If threadSafetyCheckLastId has already been filled, perform the check
243 if(_threadSafetyCheckInitialised) {
244 return (_threadSafetyCheckLastId == std::hash<std::thread::id>()(std::this_thread::get_id()));
245 }
246 // ThreadSafetyCheckLastId not yet filled: fill it and set the flag.
247 _threadSafetyCheckLastId = std::hash<std::thread::id>()(std::this_thread::get_id());
248 _threadSafetyCheckInitialised = true;

353 template<class T>
354 UnidirectionalProcessArray<T>::UnidirectionalProcessArray(typename ProcessArray<T>::InstanceType instanceType,
355 const ChimeraTK::RegisterPath& name, const std::string& unit, const std::string& description,
356 const std::vector<T>& initialValue, std::size_t numberOfBuffers, const AccessModeFlags& flags)
357 : ProcessArray<T>(instanceType, name, unit, description, flags), _vectorSize(initialValue.size()),
358 _sharedState(numberOfBuffers, initialValue.size()), _localBuffer(initialValue) {
359 TransferElement::_readQueue = _sharedState.queue.template then<void>(
360 [this](Buffer& buf) { std::swap(_localBuffer, buf); }, std::launch::deferred);
361 // allocate and initialise buffer of the base class
362 ChimeraTK::NDRegisterAccessor<T>::buffer_2D.resize(1);
363 ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0] = initialValue;
364 _intermedateBuffer.resize(ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0].size());
365 // It would be better to do the validation before initializing, but this
366 // would mean that we would have to initialize twice.
367 if(!this->isReadable()) {
368 throw ChimeraTK::logic_error("This constructor may only be used for a receiver process variable.");
369 }
370 // We need at least two buffers for the queue (so four buffers in total)
371 // in order to guarantee that we never have to block.
372 if(numberOfBuffers < 2) {
373 throw ChimeraTK::logic_error("The number of buffers must be at least two.");
374 }
375 // We have to limit the number of buffers because we cannot allocate
376 // more buffers than can be represented by size_t and the total number
377 // of buffers is the specified number plus one. This extra buffer is
378 // needed internally by the future_queue.
379 if(numberOfBuffers > (std::numeric_limits<std::size_t>::max() - 1)) {
380 throw ChimeraTK::logic_error("The number of buffers is too large.");
381 }
382 }

422 void UnidirectionalProcessArray<T>::doPreWrite(ChimeraTK::TransferType, VersionNumber) {
423 if(!this->isWriteable()) {
424 throw ChimeraTK::logic_error("Send operation is only allowed for a sender process variable.");

427 // We have to check that the vector that we currently own still has the
428 // right size. Otherwise, the code using the receiver might get into
429 // trouble when it suddenly experiences a vector of the wrong size.
430 if(ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0].size() != _vectorSize) {

433 this->getName());

470 template<class T>
471 void UnidirectionalProcessArray<T>::doPostRead(ChimeraTK::TransferType, bool hasNewData) {
472 assert(checkThreadSafety());
473 if(hasNewData) {
474 // We have to check that the vector that we currently own still has the
475 // right size. Otherwise, the code using the sender might get into
476 // trouble when it suddenly experiences a vector of the wrong size.
477 assert(ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0].size() == _localBuffer.value.size());
478
479 if(this->_accessModeFlags.has(AccessMode::wait_for_new_data)) {
480 // swap data out of the local buffer into the user buffer
481 ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0].swap(_localBuffer.value);
482 }
483 else {
484 // We have to mimic synchronous mode. Here we have to copy here because there might be multiple reads, and
485 // the reading code is allowed to swap out the user buffer, and has to get the correct value on the second read.
486 ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0] = _localBuffer.value;
487 }
488 TransferElement::_versionNumber = _localBuffer.versionNumber;
489 TransferElement::_dataValidity = _localBuffer.dataValidity;
490 }

511 if(!this->isWriteable()) {
512 return;
513 }
514 bool sendInitialValue = false;
515 if(!_persistentDataStorage) {
516 sendInitialValue = true;
517 }
518 _persistentDataStorage = storage;
519 _persistentDataStorageID = _persistentDataStorage->registerVariable<T>(
520 ChimeraTK::TransferElement::getName(), ChimeraTK::NDRegisterAccessor<T>::getNumberOfSamples());
521 if(sendInitialValue) {
522 if(_persistentDataStorage->retrieveValue<T>(_persistentDataStorageID).size() ==
523 ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0].size()) {
524 ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0] =
525 _persistentDataStorage->retrieveValue<T>(_persistentDataStorageID);
526 }
527 this->write();

533 template<class T>
534 bool UnidirectionalProcessArray<T>::writeInternal(VersionNumber newVersionNumber, bool shouldCopy) {
535 // thread safety check, if enabled (only active with debug flags enabled)
536 assert(checkThreadSafety());
537
538 assert(this->isWriteable());
539
540 // First update the persistent data storage, if any was associated. This
541 // cannot be done after sending, since the value might no longer be available
542 // within this instance.
543 if(_persistentDataStorage) {
544 _persistentDataStorage->updateValue(_persistentDataStorageID, _intermedateBuffer);
545 }
546
547 // Set time stamp and version number
548 _localBuffer.versionNumber = newVersionNumber;
549 _localBuffer.dataValidity = TransferElement::dataValidity();
550
551 // set the data by copying or swapping
552 assert(_localBuffer.value.size() == _intermedateBuffer.size());
553 if(shouldCopy) {
554 _localBuffer.value = _intermedateBuffer;
555 }
556 else {
557 _localBuffer.value.swap(_intermedateBuffer);
558 }
559
560 // send the data to the queue
561 bool dataNotLost = _sharedState.queue.push_overwrite(std::move(_localBuffer));
562
563 // if receiver does not have wait_for_new_data, do not return whether data has been lost (because conceptionally it
564 // hasn't)
565 if(!_receiver->getAccessModeFlags().has(AccessMode::wait_for_new_data)) {
566 return false;
567 }
568
569 return !dataNotLost;
570 }
