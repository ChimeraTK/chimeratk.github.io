87boost::shared_ptr<NDRegisterAccessor<std::string>> ProcessArrayFactoryBackend::getRegisterAccessor_impl<std::string>(
88 const RegisterPath& path, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
89 if(numberOfWords > 1 || wordOffsetInRegister != 0) {
90 throw ChimeraTK::logic_error("Bad dimension");
91 }
92
93 std::string initialValue;
94 if(pv[path].empty()) {
95 initialValue = generateValueFromCounter();
96 }
97 else {
98 initialValue = pv[path].back()->accessData(0);
99 }
100
101 if(path == "/unidir/sender") {
102 flags.checkForUnknownFlags({}); // test expects that write-only accessors never accept wait_for_new_data...
103 auto spv = createSynchronizedProcessArray<std::string>(1, path, "", "", "", 3, {AccessMode::wait_for_new_data});
104 pv[path].push_back(spv.second);
105 return spv.first;
106 }
107 if(path == "/unidir/polledSender") {
108 flags.checkForUnknownFlags({}); // test expects that write-only accessors never accept wait_for_new_data...
109 auto spv = createSynchronizedProcessArray<std::string>(1, path, "", "", "", 3, {});
110 pv[path].push_back(spv.second);
111 return spv.first;
112 }
113 if(path == "/unidir/receiver" && flags.has(AccessMode::wait_for_new_data)) {
114 auto spv = createSynchronizedProcessArray<std::string>(1, path, "", "", "", 3, flags);
115 pv[path].push_back(spv.first);
116 // need to write initial value
117 pv[path].back()->accessData(0) = initialValue;
118 pv[path].back()->write();
119 return spv.second;
120 }
121 if(path == "/unidir/receiver" && !flags.has(AccessMode::wait_for_new_data)) {
122 auto spv = createSynchronizedProcessArray<std::string>(1, path, "", "", "", 3, flags);
123 pv[path].push_back(spv.first);
124 // need to write initial value, otherwise the first read would block
125 pv[path].back()->accessData(0) = initialValue;
126 pv[path].back()->write();
127 return spv.second;
128 }
129 if(path == "/bidir/A") {
130 flags.add(AccessMode::wait_for_new_data);
131 auto spv = createBidirectionalSynchronizedProcessArray<std::string>(1, path, "", "", "", 3, flags);
132 pv[path].push_back(spv.first);
133 // need to write initial value

135 pv[path].back()->write();
136 return spv.second;
