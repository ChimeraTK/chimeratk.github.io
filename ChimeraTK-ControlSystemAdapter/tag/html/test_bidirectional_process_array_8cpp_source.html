303  // We save the time stamp and version number so that we can reuse it in
304  // later checks.
305  auto initialVersionNumber = pv1->getVersionNumber();
306  // Nothing has been written yet, so calling read() should not have any
307  // effects.
308  BOOST_CHECK(pv1->readNonBlocking() == false);
309  BOOST_CHECK(pv2->readNonBlocking() == false);
310  BOOST_CHECK_CLOSE(pv1->accessData(0), initialValue, 0.001);
311  BOOST_CHECK(pv1->getVersionNumber() == initialVersionNumber);

313  BOOST_CHECK(pv2->getVersionNumber() == initialVersionNumber);
314  // Now we write on pv1.
315  double newValue1 = 5.0;
316  pv1->accessData(0) = newValue1;
317  pv1->write();
318  // After writing, the time stamp and version number should have increased on
319  // pv1, but pv2 should not have changed.
320  BOOST_CHECK_CLOSE(pv1->accessData(0), newValue1, 0.001);
321  BOOST_CHECK(pv1->getVersionNumber() > initialVersionNumber);
322  auto newVersionNumber1 = pv1->getVersionNumber();
323  BOOST_CHECK_CLOSE(pv2->accessData(0), initialValue, 0.001);
