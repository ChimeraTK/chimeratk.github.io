145template<typename SimpleType>
146void CsaProcessVariable<SimpleType>::constructObjectConfig(
147 std::string name, CsAccessDir controlSystemAccessDir, const CsaObjectUserConfig& csaConfig) {
148 // Add Direction suffix to name of Process Variable
149 bool addSufix = CsaAdapter::getInstance().getAddDirectionSufixToPvNames();
150 if(addSufix) {
151 if(controlSystemAccessDir == CsAccessDir::TO_CS) {
152 name += std::string(".RD");
153 }
154 else {
155 name += std::string(".WR");
156 }
157 }
158
159 this->_name = name;
160 renameCheck();
161
162 /*
163 std::string newNameBuffer("");
164 std::string renameSearch = std::string("PV_RENAME::") + name;
165 if (CsaAppConfigFileParser::getInstance().getValueAsDataString(renameSearch, newNameBuffer)) {
166 std::cout <<
167 "!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!"
168 << std::endl; std::cout << "Process variable (PV) : " << name << " will be renamed into: " << newNameBuffer <<
169 std::endl; std::cout <<
170 "!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!"
171 << std::endl; this->_name = newNameBuffer;
172 }
173 */
174
175 // set what is needed for Control System Attaching
176 this->setObjectType();
177 this->_objConfig.reset(new CsaProcessVariableConfig<SimpleType>(this->_name, this->_bindedObjType, this));
178 this->_objConfig->setAccessDir(controlSystemAccessDir);
179 this->_objConfig->setFromCS = boost::bind(&CsaProcessVariable::setFromCS, this, _1);
180 this->_objConfig->initFromCS = boost::bind(&CsaProcessVariable::initFromCS, this, _1);
181 this->_objConfig->getToCs = boost::bind(&CsaProcessVariable::getToCs, this);
182 this->_objConfig->userConfig = csaConfig;
183
184 if(!csaConfig.getUnit().get().empty()) {
185 _setUnitEnabled = true;
186 }

194template<typename SimpleType>
195void CsaProcessVariable<SimpleType>::setUnitNameFromConfigFile() {
196 std::string renameSearch = std::string("SET_EGU::") + this->_name;
197 std::string unitNameBuffer("");
198 if(CsaAppConfigFileParser::getInstance().getValueAsDataString(renameSearch, unitNameBuffer)) {
199 auto unitEnCopy = _setUnitEnabled;
200 _setUnitEnabled = true;
201 this->setUnit(unitNameBuffer);
202 // 2018.02.07 - This is on purpose. I want to trigger warning message when setting unit from without enabling it in
203 // contructor by setting default unit value
204 _setUnitEnabled = unitEnCopy;

208template<typename SimpleType>
209void CsaProcessVariable<SimpleType>::setUnit(std::string unit) {
210 if(_setUnitEnabled) {
211 // update units in user config object passed with constructor to keep data consistent
212 this->_objConfig->userConfig.set(false, P_Unit(unit));
213 if(_setUnitFunctPtr)
214 this->_setUnitFunctPtr(unit);
215 else
216 std::cout << "ERROR:: CsaProcessVariable<SimpleType>::setUnit(std::string unit): " << this->_name
217 << " : _setUnitFunctPtr not set." << std::endl;
218 }
219 else
220 std::cout << "CsaProcessVariable<SimpleType>::setUnit. Unit was not enabled in contructor. Cannot change unit for: "
221 << this->_name << std::endl;
222}

247template<typename SimpleType>
248void CsaProcessVariable<SimpleType>::synchronize() {
249 if(this->_objConfig->getAccessDir() == CsAccessDir::TO_CS) {
250 std::cout << "WARNING: Synchronization not supported for CsAccessDir::TO_CS direction of PV for " << this->_name
251 << std::endl;
252 return;
253 }
254
255 if(this->_objConfig->setToCs) {
256 // std::cout << " CsaProcessVariable<SimpleType>::synchronize(): INFO: Synchronization of CsAccessDir::FROM_CS PV
257 // triggered. It should be avoided. Try to redesign the logic." << std::endl;
258 this->_objConfig->setToCs(this->getValue());
259 }

271template<typename SimpleType>
272void CsaProcessVariable<SimpleType>::setFromCS(SimpleType value) {
273 if(this->_blockCsWriteAccess == false) {
274 // we want to execute method from inherited classes which can ex. write to hardware, or provide extended features
275 this->setValue(value);
276 }
277 else if(this->_objConfig->userConfig.getAlwaysCsWritable() == 1) {
278 std::cout << "Write from Cs is blocked but PV: " << this->_name << " is always writable." << std::endl;
279 this->setValue(value);
280 }
281 else {
282 std::cout << "PV: " << this->_name << " write from CS is blocked. Resynchronize." << std::endl;
283 this->synchronize();
284 }
285}
