148  if(addSufix) {
149  if(controlSystemAccessDir == CsAccessDir::TO_CS) {
150  name += std::string(".RD");
151  }
152  else {
153  name += std::string(".WR");
154  }
155  }
156 
157  this->_name = name;
158  renameCheck();
159 
160  /*
161  std::string newNameBuffer("");
162  std::string renameSearch = std::string("PV_RENAME::") + name;
163  if (CsaAppConfigFileParser::getInstance().getValueAsDataString(renameSearch, newNameBuffer)) {
164  std::cout << "!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!"
165  << std::endl; std::cout << "Process variable (PV) : " << name << " will be renamed into: " << newNameBuffer << std::endl;
166  std::cout << "!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!"
167  << std::endl; this->_name = newNameBuffer;
168  }
169  */
170 
171  // set what is needed for Control System Attaching
172  this->setObjectType();
173  this->_objConfig.reset(new CsaProcessVariableConfig<SimpleType>(this->_name, this->_bindedObjType, this));
174  this->_objConfig->setAccessDir(controlSystemAccessDir);
175  this->_objConfig->setFromCS = boost::bind(&CsaProcessVariable::setFromCS, this, _1);
176  this->_objConfig->initFromCS = boost::bind(&CsaProcessVariable::initFromCS, this, _1);
177  this->_objConfig->getToCs = boost::bind(&CsaProcessVariable::getToCs, this);
178  this->_objConfig->userConfig = csaConfig;
179 
180  if(!csaConfig.getUnit().get().empty()) {
181  _setUnitEnabled = true;
182  }

192  std::string renameSearch = std::string("SET_EGU::") + this->_name;
193  std::string unitNameBuffer("");
194  if(CsaAppConfigFileParser::getInstance().getValueAsDataString(renameSearch, unitNameBuffer)) {

196  _setUnitEnabled = true;

198  // 2018.02.07 - This is on purpose. I want to trigger warning message when setting unit from without enabling it in

267 template<typename SimpleType>
268 void CsaProcessVariable<SimpleType>::setFromCS(SimpleType value) {
269  if(this->_blockCsWriteAccess == false) {
270  // we want to execute method from inherited classes which can ex. write to hardware, or provide extended features
271  this->setValue(value);
272  }
273  else if(this->_objConfig->userConfig.getAlwaysCsWritable() == 1) {
274  std::cout << "Write from Cs is blocked but PV: " << this->_name << " is always writable." << std::endl;
275  this->setValue(value);
276  }
277  else {
278  std::cout << "PV: " << this->_name << " write from CS is blocked. Resynchronize." << std::endl;
279  this->synchronize();
280  }
281 }
