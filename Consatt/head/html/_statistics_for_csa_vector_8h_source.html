39 template<class T = float>
40 class DataBuffer {
41  unsigned int _lenght{0};
42  std::deque<T> _dataQueue;
43 
44  public:
45  DataBuffer() : _lenght(0){};
46  DataBuffer(unsigned int lenght) : _lenght(lenght){};
47 
48  void add(T input) {
49  if(!_lenght) return;
50 
51  // remove data when size changed or we have enough data is queue
52  while(_dataQueue.size() >= _lenght) _dataQueue.pop_front();
53  _dataQueue.push_back(input);
54  }
55 
56  void setLenght(unsigned int lenght) { _lenght = lenght; }
57 
58  unsigned int getLength() { return _lenght; }
59 
60  unsigned int getCurrentLength() { return _dataQueue.size(); }
61 
62  bool isFull() { return (_lenght == _dataQueue.size()); }
63 
64  bool isEmpty() { return (0 == _dataQueue.size()); }
65 
66  void clear() { _dataQueue.clear(); }
67 
68  std::vector<T> toStdVector() {
69  std::vector<T> v(_dataQueue.begin(), _dataQueue.end());
70  return v;
71  }
72 };

196 template<typename SimpleType>
197 void StatisticsForStdVector<SimpleType>::calculate(uint64_t timeStamp) {
198  this->_timeStamp = timeStamp;
199 
200  auto& data = *(this->_simpleDataVector);
201  auto size = this->_simpleDataVector->size();
202 
203  uint32_t startIndex = 0;
204  uint32_t stopIndex = this->_simpleDataVector->size();
205 
206  if(_indexMax > _indexMin && _indexMin < stopIndex) {
207  startIndex = _indexMin;
208  stopIndex = _indexMax;
209  }
210  _calcDataSize = stopIndex - startIndex;
211 
212  if(_stdDevEnabled) {
213  double mean = calcMeanWithMinMax(data, startIndex, stopIndex);
214  double stdDev = 0.0;
215  std::for_each(
216  data.begin() + startIndex, data.begin() + stopIndex, [&](SimpleType i) { stdDev += (i - mean) * (i - mean); });
217 
218  // std::cout << "stdDev 1: " << stdDev << std::endl;
219  stdDev /= _calcDataSize;
220  stdDev = std::sqrt(stdDev);
221  // std::cout << "stdDev 2: " << stdDev << std::endl;
222  this->_stdDev = stdDev;
223 
224  if(_meanEnabled) this->_mean = mean;
225  }
226  else if(_meanEnabled) {
227  this->_mean = calcMeanWithMinMax(data, startIndex, stopIndex);
228  }
229  else if(_minMaxEnabled) {
230  calcMinMaxWithSum(data, startIndex, stopIndex, false);
231  }
232 
233  if(_rmsEnabled) {
234  double rms = 0;
235  std::for_each(data.begin() + startIndex, data.begin() + stopIndex, [&](SimpleType i) { rms += i * i; });
236 
237  rms /= _calcDataSize;
238  this->_rms = std::sqrt(rms);
239  }
240 }

242 template<typename SimpleType>
243 inline double StatisticsForStdVector<SimpleType>::calcMeanWithMinMax(
244  std::vector<SimpleType>& data, uint32_t minIndex, uint32_t maxIndex) {
245  double mean = 0.0;
246  if(_minMaxEnabled) {
247  mean = calcMinMaxWithSum(data, minIndex, maxIndex, true);
248  }
249  else {
250  for(uint32_t i = minIndex; i < maxIndex; i++) {
251  mean += data[i];
252  }
253  }
254  mean /= (maxIndex - minIndex);
255  return mean;
256 }

258 template<typename SimpleType>
259 inline SimpleType StatisticsForStdVector<SimpleType>::calcMinMaxWithSum(
260  std::vector<SimpleType>& data, uint32_t minIndex, uint32_t maxIndex, bool calcSum) {
261  SimpleType sum = 0;
262  SimpleType min = std::numeric_limits<SimpleType>::max();
263  SimpleType max = -min;
264  if(calcSum) {
265  for(uint32_t i = minIndex; i < maxIndex; i++) {
266  sum += data[i];
267  if(data[i] > max) max = data[i];
268  if(data[i] < min) min = data[i];
269  }
270  }
271  else {
272  for(uint32_t i = minIndex; i < maxIndex; i++) {
273  if(data[i] > max) max = data[i];
274  if(data[i] < min) min = data[i];
275  }
276  }
277  this->_max = max;
278  this->_min = min;
279 
280  return sum;
281 }
