127template<typename SimpleType>
128CsaExternalProcessVariable<SimpleType>::CsaExternalProcessVariable(
129 std::string name, CsAccessDir controlSystemAccessDir) {
130 // set what is needed for Control System Attaching
131 this->setObjectType();
132 this->_objConfig.reset(new CsaExternalProcessVariableConfig<SimpleType>(name, this->_bindedObjType));
133 this->_objConfig->setAccessDir(controlSystemAccessDir);
134 this->_objConfig->isArray = false;
135 CsaAdapter::getInstance().createExternalProcessVariable(this->_objConfig);
136 _initialized = true;
137 // now object should be created and address must be set
138 if(this->_objConfig->externalObject) this->_objConfig->externalObject->setAddress(name);
139}

144template<typename SimpleType>
145void CsaExternalProcessVariable<SimpleType>::create(std::string name, CsAccessDir controlSystemAccessDir) {
146 if(_initialized) {
147 return;
148 }
149
150 this->setObjectType();
151 this->_objConfig.reset(new CsaExternalProcessVariableConfig<SimpleType>(name, this->_bindedObjType));
152 this->_objConfig->setAccessDir(controlSystemAccessDir);
153 this->_objConfig->isArray = false;
154 CsaAdapter::getInstance().createExternalProcessVariable(this->_objConfig);
155 this->_initialized = true;
156 // now object should be created and now one can try to set address, setting address may throw
157 if(this->_objConfig->externalObject) this->_objConfig->externalObject->setAddress(name);
158}

194template<typename SimpleType>
195SimpleType CsaExternalProcessVariable<SimpleType>::getValue(bool getLocalValue) {
196 if(this->_initialized == false) {
197 std::ostringstream stream;
198 stream
199 << "CsaExternalProcessVariable<SimpleType>::getValue: Object is not initialized. Cannot read from this object.";
200 throw CsaExternalProcessException(stream.str(), CsaExternalProcessException::EX_NOT_INITIALIZED);
201 }
202
203 if(this->_objConfig->getAccessDir() == CsAccessDir::TO_CS) {
204 std::ostringstream stream;
205 stream << "CsaExternalProcessVariable<SimpleType>::getValue: External variable: " << this->_objConfig->getName()
206 << " is TO_CS. Cannot read from this object.";
207 throw CsaExternalProcessException(stream.str(), CsaExternalProcessException::EX_CS_ACCESS_DIRECTION);
208 }
209
210 if(getLocalValue) {
211 if(!_localValueValid) {
212 std::ostringstream stream;
213 stream << "CsaExternalProcessVariable<SimpleType>::getValue: Local value of external variable: "
214 << this->_objConfig->getName() << " is not VALID.";
215 throw CsaExternalProcessException(stream.str(), CsaExternalProcessException::EX_LOCAL_VALUE_NOT_VALID);
216 }
217 return _localValue;
218 }
219
220 _localValue = this->_objConfig->externalObject->getValue();
221 _localValueValid = true;
222 return _localValue;
223}

225template<typename SimpleType>
226void CsaExternalProcessVariable<SimpleType>::setValue(SimpleType value, bool setLocalValue) {
227 if(this->_initialized == false) {
228 std::ostringstream stream;
229 stream
230 << "CsaExternalProcessVariable<SimpleType>::setValue: Object is not initialized. Cannot write to this object.";
231 throw CsaExternalProcessException(stream.str(), CsaExternalProcessException::EX_NOT_INITIALIZED);
232 }
233
234 if(this->_objConfig->getAccessDir() == CsAccessDir::FROM_CS) {
235 std::ostringstream stream;
236 stream << "CsaExternalProcessVariable<SimpleType>::setValue: External variable: " << this->_objConfig->getName()
237 << " is FROM_CS. Cannot write to this object.";
238 throw CsaExternalProcessException(stream.str(), CsaExternalProcessException::EX_CS_ACCESS_DIRECTION);
239 }
240
241 if(!setLocalValue) this->_objConfig->externalObject->setValue(value);
242 _localValue = value;
243 _localValueValid = true;
244}
