123bool CsaAppConfigFileParser::findConfigFileEntryWithType(
124 std::string key, EntryTypes::TypeId typeId, std::map<std::string, ConfigFileEntry>::iterator& resultIterator) {
125 auto it = _fileAsMap.find(_mainKey);
126 if(it == _fileAsMap.end()) return false;
127
128 std::map<std::string, ConfigFileEntry>::iterator it2 = it->second.find(key);
129 if(it2 == it->second.end()) return false;
130 if(it2->second.type.compare(EntryTypes::SUPPORTED[typeId]) != 0) return false;
131
132 resultIterator = it2;
133 return true;
134}

148void CsaAppConfigFileParser::parseLineInternal() {
149 static unsigned int mapFileSize = -1;
150
151 // std::cout << "CsaAppConfigFileParser::parseLineInternal: current line is: " << _fileParser.getCurrentLine() << std::endl;
152
153 std::string mainKey = _fileParser.getNextTokenDataString();
154
155 mainKey.erase(std::remove_if(mainKey.begin(), mainKey.end(), [](char x) { return std::isspace(x); }), mainKey.end());
156
157 if(mainKey.substr(0, 1).compare("#") == 0) {
158 _logger(Loger::DETAIL) << "Line: -> \"" << _fileParser.getCurrentLine() << "\" <- is commented out. " << std::endl;
159 return;
160 }
161
162 if(mainKey.compare("CONFIG_LOG_LEVEL") == 0) {
163 uint32_t logLevel = _fileParser.getNextTokenDataUInt();
164 _logger() << "Config file logging level changed to: " << logLevel << std::endl;
165 _logger.setLogLevel(logLevel);
166 return;
167 }
168
169 if(_mainKey.empty()) {
170 _mainKey = mainKey;
171 }
172
173 ConfigFileEntry entry;
174 entry.key = _fileParser.getNextTokenDataString();
175 entry.key.erase(
176 std::remove_if(entry.key.begin(), entry.key.end(), [](char x) { return std::isspace(x); }), entry.key.end());
177
178 entry.value = _fileParser.getNextTokenDataString();
179 entry.value.erase(std::remove_if(entry.value.begin(), entry.value.end(), [](char x) { return std::isspace(x); }),
180 entry.value.end());
181
182 entry.type = _fileParser.getNextTokenDataString();
183 entry.type.erase(
184 std::remove_if(entry.type.begin(), entry.type.end(), [](char x) { return std::isspace(x); }), entry.type.end());
185
186 if(!validateEntryType(entry.type)) return;
187
188 _fileAsMap[mainKey].insert(std::pair<std::string, ConfigFileEntry>(entry.key, entry));
189 if(_fileAsMap.size() != mapFileSize) {
190 _logger(Loger::INFO) << "New main key found. Name is: " << mainKey << " Main keys map size: " << _fileAsMap.size()
191 << std::endl;
192 mapFileSize = _fileAsMap.size();
193 }
194
195 _logger(Loger::INFO) << "Main key: " << mainKey << " Key: " << entry.key << " Value: " << entry.value
196 << " Type: " << entry.type << std::endl;
197}
198bool CsaAppConfigFileParser::validateEntryType(std::string typeToValidate) {
199 // validate whether type of entry is on the list of supported types
200 bool validType = false;
201 for(auto& item : EntryTypes::SUPPORTED) {
202 if(item.compare(typeToValidate) == 0) {
203 validType = true;
204 break;
205 }
206 }
207 if(!validType) {
208 _logger(Loger::ERROR) << "Type: " << typeToValidate << " in line: \"" << _fileParser.getCurrentLine()
209 << "\" is not supported. Line ignored." << std::endl;
210 }
211 return validType;
212}
