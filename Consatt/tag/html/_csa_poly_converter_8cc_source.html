42double CsaPolyConverter::convert(double x) {
43 double data = 0;
44 double d = 0;
45
46 uint32_t i1_data = _mode->getValue();
47 double f1_data = _a0->getValue();
48 double f2_data = _a1->getValue();
49 double f3_data = _a2->getValue();
50
51 switch(i1_data) {
52 case 0:
53 return x;
54
55 case 1:
56 data = f1_data + f2_data * x + f3_data * x * x;
57 return data;
58
59 case 2:
60 data = f1_data + f2_data * x + f3_data * x * x;
61 return std::exp((double)data);
62
63 case 3:
64 data = f1_data + f2_data * x;
65 data = data * data + f3_data;
66 return data;
67
68 case 4:
69 d = f1_data + f2_data * x;
70 if(d > 0.0)
71 data = log((double)d) * f3_data;
72 else if(d < 0.0)
73 data = log((double)-d) * f3_data;
74 else
75 data = 0.0;
76
77 return data;
78
79 case 5:
80 data = (f1_data + f2_data * x + f3_data * x * x) * x;
81 return data;
82
83 case 6:
84 data = f1_data + f2_data * pow(fabs((double)x), (double)f3_data);
85 return data;
86
87 case 7:
88 d = f1_data + f2_data * x;
89 return (f3_data * pow(10.0, d));
90
91 case 10:
92 // pvak pirani
93 data = -7.5188931 + 1.44424899e-2 * x + 1.84163771e-5 * x * x - 7.930881164e-8 * x * x * x +
94 6.1887331679e-11 * x * x * x * x;
95 return std::exp((double)data);
96
97 case 11:
98 // Balzers Pirani TPR 018
99 data = -6.458298635 + 1.1065126182e-2 * x + 1.724604645308e-5 * x * x - 4.470682241289e-8 * x * x * x +
100 2.8873048815568e-11 * x * x * x * x;
101 return std::exp((double)data);
102
103 case 12:
104 // Alcatel Pirani PA 101-111
105 data = -4.388932846151 + 2.196812521419e-2 * x - 3.891335175617e-5 * x * x + 2.57595961364e-8 * x * x * x;
106 return std::exp((double)data);
107
108 case 13:
109 // Alcatel Pirani PB 101-111
110 data = -9.125316008584 + 2.7479261887639e-2 * x - 3.27756486380887e-5 * x * x + 1.865511857175e-8 * x * x * x;
111 return std::exp(double(data));
112
113 case 14:
114 // Granville-Phillips Pirani with 275-Display
115 data = -5.80218101625545 + 7.21225921084386e-2 * x - 3.01506446945197e-4 * x * x +
116 5.31032962657776e-7 * x * x * x - 2.99101142006525e-10 * x * x * x * x;
117 return std::exp((double)data);
118
119 case 15:
120 // MVP pirani with JAD diode
121 data = -6.90752840773383 + 1.56914599457571e-02 * x + -3.71500742123613e-06 * x * x +
122 -3.21792780896806e-08 * x * x * x + 3.38803809282301e-11 * x * x * x * x;
123 return std::exp((double)data);
124
125 case 16:
126 // MVP penning, curve splitted into two parts
127 if(x < 860) {
128 data = -2.46742667494164e1 + 1.58559741926023e-02 * x;
129 return std::exp((double)data);

141double CsaPolyConverter::unconvert(double y) {
142 double res;
143
144 uint32_t i1_data = _mode->getValue();
145 double f1_data = _a0->getValue();
146 double f2_data = _a1->getValue();
147 double f3_data = _a2->getValue();
148
149 switch(i1_data) {
150 case 0:
151 return y;
152
153 case 1:
154 if(f3_data == 0.f && f2_data == 0.f && f1_data == 0.f) {
155 return 0.0;
156 }
157 else if(f3_data == 0.f && f2_data == 0.f) {
158 return f1_data;
159 }
160 else if(f3_data == 0.f)
161 res = (y - f1_data) / f2_data;
162 else {
163 double halfq = 0.5f * f2_data / f3_data;
164 float nearest = 0.f;
165 res = sqrt((y - f1_data) / f3_data + halfq * halfq);
166 if(halfq + nearest >= 0.f) {
167 res = res - halfq;
168 }
169 else {
170 res = -res - halfq;
171 }
172 }
173 return res;
174
175 case 2:
176 y = log(y);
177 return y;
178
179 case 6:
180 res = pow((y - f1_data) / f2_data, 1.f / f3_data);
181 return res;
182
183 default:
184 std::cout << "CsaPolyConverter::" << __func__ << ": Not supported. Unable to unconvert for mode: " << i1_data
185 << std::endl;
186 return y;
187 }
188}

256void CsaPolyConverterVector::convert(std::vector<float>& data, size_t dataSize) {
257 uint32_t i1_data = _mode->getValue();
258 if(!this->isEnabled() || i1_data != 1) return;
259
260 double f1_data = _a0->getValue();
261 double f2_data = _a1->getValue();
262 double f3_data = _a2->getValue();
263
264 if(dataSize > data.size()) dataSize = data.size();
265
266 std::transform(data.begin(), data.begin() + dataSize, data.begin(),
267 [=](float x) -> float { return f1_data + f2_data * x + f3_data * x * x; });
268}

274void CsaPolyConverterVector::unconvert(std::vector<float>& data, size_t dataSize) {
275 uint32_t i1_data = _mode->getValue();
276 if(!this->isEnabled() || i1_data != 1) return;
277
278 double f1_data = _a0->getValue();
279 double f2_data = _a1->getValue();
280 double f3_data = _a2->getValue();
281
282 if(dataSize > data.size()) dataSize = data.size();
283
284 if(f3_data == 0.f && f2_data == 0.f && f1_data == 0.f) {
285 data.assign(dataSize, 0);
286 }
287 else if(f3_data == 0.f && f2_data == 0.f) {
288 data.assign(dataSize, f1_data);
289 }
290 else if(f3_data == 0.f) {
291 std::transform(
292 data.begin(), data.begin() + dataSize, data.begin(), [=](float x) -> float { return (x - f1_data) / f2_data; });
293 }
294 else {
295 double halfq = 0.5f * f2_data / f3_data;
296 float nearest = 0.f;
297 for(size_t i = 0; i < dataSize; i++) {
298 data[i] = std::sqrt((data[i] - f1_data) / f3_data + halfq * halfq);
299 if(halfq + nearest >= 0.f) {
300 data[i] = data[i] - halfq;
301 }
302 else {
303 data[i] = -data[i] - halfq;
304 }
305 }
306 }
