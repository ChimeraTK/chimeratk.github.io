130  if(_logFileMaxSizeKb->getValue() < minSizeKbToTrim) _logFileMaxSizeKb->setValue(minSizeKbToTrim);
131 
132  if(_logFileTruncateFactor->getValue() > 0.9)
133  _logFileTruncateFactor->setValue(0.9);
134  else if(_logFileTruncateFactor->getValue() < 0.1)
135  _logFileTruncateFactor->setValue(0.1);
136

138  trimLogFileCoreThread();
139  }
140  }

143  _logFileTruncateFactor->synchronize();

148 void LogFileWorker::trimLogFileCoreThread() {
149  _fileThreadInProgress = true;
150  auto time_start = std::chrono::high_resolution_clock::now();
151 
152  std::ifstream logFileToRead(_logFilePath.c_str(), std::ifstream::binary);
153 
154  if(logFileToRead) {
155  logFileToRead.seekg(0, std::ios::end);
156  int fileSize = logFileToRead.tellg();
157  int newMarker = fileSize * (1 - _logFileTruncateFactor->getValue());
158  int charNoToRead = std::abs(fileSize - newMarker);
159  if(charNoToRead > _logFileMaxSizeKb->getValue() * 1024) charNoToRead = _logFileMaxSizeKb->getValue() * 1024;
160 
161  std::cout << ConsattGenericUtils::getCurrentDateInLoggerFormat() << " Current size: " << fileSize
162  << ", wanted size: " << charNoToRead << ", rewrite from byte: " << newMarker
163  << ", trim factor: " << _logFileTruncateFactor->getValue() << std::endl;
164 
165  logFileToRead.seekg(-charNoToRead, std::ios::end);
166 
167  std::string tempLogFilePath(_logFilePath + ".temp");
168  std::ofstream logFileToWrite(tempLogFilePath, std::ifstream::binary);
169 
170  if(logFileToWrite) {
171  auto time_temp_file_start = std::chrono::high_resolution_clock::now();
172  std::cout << ConsattGenericUtils::getCurrentDateInLoggerFormat()
173  << " LogFileWorker::trimLogFileCore: Start copying to temporary file. " << std::endl;
174 #define BUFF_SIZE 128 * 1024
175  char* buffer = new char[BUFF_SIZE];
176  int readCnt = 0;
177  while(logFileToRead.read(buffer, BUFF_SIZE)) {
178  logFileToWrite.write(buffer, BUFF_SIZE);
179  readCnt++;
180  }
181  auto dataLeft = logFileToRead.gcount();
182  logFileToRead.read(buffer, dataLeft);
183  logFileToWrite.write(buffer, dataLeft);
184 
185  delete[] buffer;
186  int newFileSize = readCnt * BUFF_SIZE + dataLeft;
187  auto time_temp_file_done = std::chrono::high_resolution_clock::now();
188 
189  logFileToWrite.close();
190  logFileToRead.close();
191 
192  auto time_copy_start = std::chrono::high_resolution_clock::now();
193  fs::path from(tempLogFilePath);
194  fs::path to(_logFilePath);
195  bool no_error = fs::copy_file(from, to, fs::copy_options::overwrite_existing);
196  auto time_copy_done = std::chrono::high_resolution_clock::now();
197  std::cout << ConsattGenericUtils::getCurrentDateInLoggerFormat()
198  << " LogFileWorker::trimLogFileCore: Temporary file creation time [ms]: "
199  << std::chrono::duration_cast<std::chrono::milliseconds>(time_temp_file_done - time_temp_file_start)
200  .count()
201  << std::endl;
202  if(no_error) {
203  std::cout << ConsattGenericUtils::getCurrentDateInLoggerFormat()
204  << " LogFileWorker::trimLogFileCore: Copy to original log file time [ms]: "
205  << std::chrono::duration_cast<std::chrono::milliseconds>(time_copy_done - time_copy_start).count()
206  << std::endl;
207  std::cout << ConsattGenericUtils::getCurrentDateInLoggerFormat()
208  << " LogFileWorker::trimLogFileCore: Old log file size[kb]: " << (fileSize / 1024)
209  << " , new file size[kb]: " << (newFileSize / 1024) << std::endl;
210  _logFileCurrentSize->setValue((newFileSize / 1024));
211  no_error = fs::remove(from);
212  if(!no_error)
213  std::cout << ConsattGenericUtils::getCurrentDateInLoggerFormat()
214  << " LogFileWorker::trimLogFileCore: Removing file " << tempLogFilePath << " failed." << std::endl;
215  }
216  else
217  std::cout << ConsattGenericUtils::getCurrentDateInLoggerFormat()
218  << " LogFileWorker::trimLogFileCore: Copy file " << tempLogFilePath << " into " << _logFilePath
219  << " failed." << std::endl;
220  }
221  else {
222  logFileToRead.close();
223  }
224  }
225  auto time_end = std::chrono::high_resolution_clock::now();
226  std::cout << ConsattGenericUtils::getCurrentDateInLoggerFormat()
227  << " LogFileWorker::trimLogFileCore: Truncation thread time [ms]: "
228  << std::chrono::duration_cast<std::chrono::milliseconds>(time_end - time_start).count() << std::endl
229  << std::endl;
230  _fileThreadInProgress = false;
231 }

257 void LogFileWorker::loadLogFileTailCore() {
258  std::fstream logFile(_logFilePath.c_str(), std::fstream::in);
259 
260  if(logFile) {
261  logFile.seekg(0, std::ios::end);
262  int fileSize = logFile.tellg();
263  int newLineCharCnt = 0;
264  int wantedLines = (_logFileTailLineNumber->getValue() + 1);
265  int charPerLine = 125;
266 
267  int index = 0;
268  bool contFlag = true; // continue flag
269  do {
270  int charNoToRead = wantedLines * charPerLine;
271  if(charNoToRead > fileSize) charNoToRead = fileSize;
272 
273  logFile.seekg(-charNoToRead, std::ios::end);
274  // std::cout << "Wanted lines no: " << wantedLines << ", assumed charPerLine " << charPerLine << ", file size: "
275  // << fileSize << ", read size: " << charNoToRead << std::endl;
276  char* buffer = new char[charNoToRead];
277  logFile.read(buffer, charNoToRead);
278  newLineCharCnt = 0;
279  for(index = charNoToRead - 1; index > 0 && newLineCharCnt < wantedLines; index--) {
280  if(buffer[index] == '\n') {
281  newLineCharCnt++;
282  }
283  }
284  contFlag = false;
285  if(newLineCharCnt < wantedLines && charNoToRead == fileSize) {
286  // std::cout << "File doesn't contain " << wantedLines << " lines. Index is: " << index << ". Print full file" << std::endl;
287  }
288  else if(newLineCharCnt < wantedLines && charNoToRead < fileSize) {
289  charPerLine = ((int)charPerLine * 1.5);
290  // std::cout << wantedLines << " lines not found in " << charNoToRead << " chars. Extend lines per line to: " <<
291  // charPerLine << std::endl;
292  contFlag = true;
293  }
294  if(!contFlag) {
295  std::string logToPrint("");
296  logToPrint.append((buffer + (index + 1) * sizeof(char)), charNoToRead - 1 - index);
297  _logFileTail->setValue(logToPrint);
298  }
299 
300  delete[] buffer;
301  } while(contFlag);
302  logFile.close();
303  }

306 void LogFileWorker::loadLogFileWithTimeFilterThread() {
307  _fileThreadInProgress = true;
308  auto time_start = std::chrono::high_resolution_clock::now();
309 
310  std::ifstream file;
311  file.open(_logFilePath.c_str(), std::ios::in);
312 
313  if(file.is_open()) {
314  time_t startTime = _logFileFilterStart->getValue();
315  time_t stopTime = _logFileFilterEnd->getValue();
316 
317  int fileLinesNoBeforeScan = fileLinesNo(_logFilePath.c_str());
318 
319  std::string currentLine("");
320  std::string logNewText("");
321  int matchedLineCnt = 0;
322  int maxLines = _logFileTailLineNumber->getValue();
323  if(maxLines == 0 || maxLines > 10000) {
324  maxLines = 10000;
325  }
326  int lineCnt1 = 0;
327  int totalLineCnt = 0;
328  std::cout << "LogFileWorker::loadLogFileWithTimeFilter: Lines in file: " << fileLinesNoBeforeScan
329  << ", startTime: " << startTime << " , endTime: " << stopTime << std::endl;
330  // from this point there cannot be any log info put to the file, because then this loop will never end.
331  // therefore, additional condition has been added -> totalLineCnt < currentLinesNo
332  while(std::getline(file, currentLine)) {
333  if(currentLine.length() > 20) {
334  time_t currentMarkerTime = getTimeFromMarker(currentLine.substr(0, 19).c_str(), "%Y-%m-%d %H:%M:%S");
335  if(currentMarkerTime >= startTime && currentMarkerTime <= stopTime) {
336  logNewText += (currentLine + "\n");
337  matchedLineCnt++;
338  }
339  lineCnt1++;
340  }
341  totalLineCnt++;
342 
343  if(matchedLineCnt >= _logFileTailLineNumber->getValue()) break;
344  if(totalLineCnt > (fileLinesNoBeforeScan + 10000)) {
345  std::cout << "LogFileWorker::loadLogFileWithTimeFilter: Emergency stop. Log file line number when parse "
346  "started exceded. Scanned lines: "
347  << totalLineCnt << std::endl;
348  break;
349  }
350  }
351  if(matchedLineCnt)
352  _logFileTail->setValue(logNewText);
353  else
354  _logFileTail->setValue("---- NO ENTRIES FOUND WHICH MATCH GIVEN DATE ----");
355  file.close();
356 
357  std::cout << "LogFileWorker::loadLogFileWithTimeFilter: Logfile line scanned: " << totalLineCnt
358  << ", longer than 20 chars: " << lineCnt1 << " , matched lines: " << matchedLineCnt << "\n";
359  auto readTimeMs =
360  std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::high_resolution_clock::now() - time_start)
361  .count();
362  std::cout << "LogFileWorker::loadLogFileWithTimeFilter: Filtering " << totalLineCnt
363  << " lines took [ms]: " << readTimeMs << std::endl;
364  }
365  _fileThreadInProgress = false;
366 }
