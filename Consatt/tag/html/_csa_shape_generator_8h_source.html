475 void CsaShapeGenerator<SimpleType>::inputParamPriorityCb() {
476  int priority = _inputParamPriority->getValue();
477 
478  uint32_t curMaxIndex = _maxSampleIndex->getValue();
479  if(curMaxIndex < 1 || curMaxIndex > _shapeMaxLength) {
480  curMaxIndex = _shapeMaxLength;
481  }
482  _maxSampleIndexInUse->setValue(curMaxIndex);
483 
484  if(priority != 1) { // this means that period number has priority
485  // number of periods cannot be greater than current max sample index and smaller than 1
486  uint32_t curPeriodsNumber = _periodsNumber->getValue();
487  if(curPeriodsNumber < 1 || curPeriodsNumber > curMaxIndex) {
488  curPeriodsNumber = 1;
489  }
490  _periodsNumberInUse->setValue(curPeriodsNumber);
491 
492  uint32_t curSamplesPerPeriod = (uint32_t)(curMaxIndex / curPeriodsNumber);
493  _samplesPerPeriodInUse->setValue(curSamplesPerPeriod);
494  }
495  else {
496  // samples per period multiplied by period amount cannot be greater than max index and smaller than 1
497  uint32_t curSamplesPerPeriod = _samplesPerPeriod->getValue();
498  if(curSamplesPerPeriod < 2 || curSamplesPerPeriod > curMaxIndex) {
499  curSamplesPerPeriod = 2;
500  }
501  _samplesPerPeriodInUse->setValue(curSamplesPerPeriod);
502 
503  int curPeriodsNumber = (int)(curMaxIndex / curSamplesPerPeriod);
504  _periodsNumberInUse->setValue(curPeriodsNumber);
505  }
506 
507  _logger(Loger::INFO) << __func__ << ": "
508  << "Max index: " << _maxSampleIndexInUse->getValue()
509  << " Periods number: " << _periodsNumberInUse->getValue()
510  << " Samples per period: " << _samplesPerPeriodInUse->getValue() << std::endl;

515 template<typename SimpleType>
516 std::vector<SimpleType> CsaShapeGenerator<SimpleType>::getSinglePeriodDataVector() {
517  ShapeType shapeType = static_cast<ShapeType>(_shapeType->getValue());
518 
519  std::vector<SimpleType> pointsVector;
520  SimpleType amplitude = _shapeAmplitude->getValue();
521 
522  if(amplitude == 0) {
523  _logger(Loger::WARNING) << __func__ << ": "
524  << "Amplitude set to 0." << std::endl;
525  }
526 
527  uint32_t samplesNoInPeriod = _samplesPerPeriodInUse->getValue();
528 
529  if(shapeType == CsaShapeGenerator::TYPE_RECTANGLE) {
530  uint32_t halfPeriod = samplesNoInPeriod / 2;
531  for(uint32_t i = 0; i < halfPeriod; i++) {
532  pointsVector.push_back(amplitude);
533  }
534 
535  for(uint32_t i = halfPeriod; i < samplesNoInPeriod; i++) pointsVector.push_back(0);
536  }
537  else if(shapeType == CsaShapeGenerator::TYPE_SAW_RISING) {
538  for(uint32_t i = 0; i < samplesNoInPeriod; i++) {
539  SimpleType val = i * amplitude / (samplesNoInPeriod - 1);
540  pointsVector.push_back(val);
541  }
542  }
543  else if(shapeType == CsaShapeGenerator::TYPE_SAW_FALLING) {
544  for(uint32_t i = 0; i < samplesNoInPeriod; i++) {
545  SimpleType val = amplitude - (i * amplitude / (samplesNoInPeriod - 1));
546  pointsVector.push_back(val);
547  }
548  }
549  else if(shapeType == CsaShapeGenerator::TYPE_CONSTANT) {
550  for(uint32_t i = 0; i < samplesNoInPeriod; i++) {
551  pointsVector.push_back(amplitude);
552  }
553  }
554  else if(shapeType == CsaShapeGenerator::TYPE_SINUS) {
555  for(uint32_t i = 0; i < samplesNoInPeriod; i++) {
556  SimpleType val = amplitude * sin(2 * M_PI * i / samplesNoInPeriod);
557  pointsVector.push_back(val);
558  }
559  }
560  else if(shapeType == CsaShapeGenerator::TYPE_COSIN) {
561  for(uint32_t i = 0; i < samplesNoInPeriod; i++) {
562  SimpleType val = amplitude * cos(2 * M_PI * i / samplesNoInPeriod);
563  pointsVector.push_back(val);
564  }
565  }
566 
567  return pointsVector;
568 }

570 template<typename SimpleType>
571 void CsaShapeGenerator<SimpleType>::generateShape() {
572  // clear the memory
573  for(unsigned int i = 0; i < _shapeMaxLength; i++) {
574  _shapeDataVector[i] = 0;
575  }
576 
577  uint32_t maxSampleIndex = _maxSampleIndexInUse->getValue();
578  uint32_t samplesInSinglePeriod = _samplesPerPeriodInUse->getValue();
579 
580  if(samplesInSinglePeriod == 0) {
581  maxSampleIndexCb();
582  maxSampleIndex = _maxSampleIndexInUse->getValue();
583  samplesInSinglePeriod = _samplesPerPeriodInUse->getValue();
584  }
585 
586  uint32_t periodsNumber = maxSampleIndex / samplesInSinglePeriod;
587  _periodsNumberInUse->setValue(periodsNumber);
588 
589  // prepare single period data (no X offset and no Y offset)
590  std::vector<SimpleType> pointsVector = getSinglePeriodDataVector();
591 
592  _logger(Loger::INFO) << __func__ << ": "
593  << "maxSampleIndex: " << maxSampleIndex << " periodsNumber: " << periodsNumber
594  << " samplesInSinglePeriod: " << samplesInSinglePeriod << " _shapeMaxLength: " << _shapeMaxLength
595  << std::endl;
596  // fill main buffer with data from single period
597  uint32_t maxIdInAllPeriods = 0;
598  for(uint32_t periodId = 0; periodId < periodsNumber; periodId++) {
599  for(uint32_t sampleId = 0; sampleId < samplesInSinglePeriod; sampleId++) {
600  //_logger(Loger::INFO) << "Period ID: " << periodId << " Sample ID: " << sampleId << " Sample in single period: "
601  //<< samplesInSinglePeriod
602  // << " Global sample ID: " << maxIdInAllPeriods << " Sample value: " << pointsVector[sampleId] << std::endl;
603  uint32_t index = periodId * samplesInSinglePeriod + sampleId;
604  if(index < _shapeMaxLength) _shapeDataVector[index] = pointsVector[sampleId];
605  maxIdInAllPeriods++;
606  }
607  }
608 
609  // fill data for last (not full) period when this behavior is enabled (by default disabled, only full periods)
610  if(_fullPeriodOff->getValue() == 1) {
611  _logger(Loger::INFO) << __func__ << ": "
612  << "Not full periods mode. maxIdInAllPeriods: " << maxIdInAllPeriods << std::endl;
613  for(uint32_t i = maxIdInAllPeriods; i < maxSampleIndex; i++) {
614  _shapeDataVector[i] = pointsVector[i - maxIdInAllPeriods];
615  }
616 
617  if(_lastFirstSampleEqualOff->getValue() == 0) _shapeDataVector[maxSampleIndex - 1] = pointsVector.front();
618  }
619  else {
620  _logger(Loger::INFO) << __func__ << ": "
621  << "Only full periods mode. maxIdInAllPeriods: " << maxIdInAllPeriods << std::endl;
622  auto data = pointsVector.front();
623  if(_lastFirstSampleEqualOff->getValue() == 1) data = pointsVector.back();
624 
625  if(maxIdInAllPeriods < maxSampleIndex) {
626  for(uint32_t i = maxIdInAllPeriods; i < maxSampleIndex; i++) _shapeDataVector[i] = data;
627  }
628  else {
629  _shapeDataVector[maxIdInAllPeriods - 1] = data;
630  }
631  }
632 
633  // Apply scaling and X/Y offsets
634  SimpleType scaleFactor = _shapeScalingFactor->getValue();
635  SimpleType offsetY = _yShapeOffset->getValue();
636 
637  if(scaleFactor == 0) {
638  _logger(Loger::WARNING) << __func__ << ": "
639  << "Scaling factor set to 0." << std::endl;
640  }
641 
642  // apply scaling and Y offset
643  for(uint32_t i = 0; i < _shapeMaxLength; i++) {
644  _shapeDataVector[i] = ((_shapeDataVector[i] + offsetY) * scaleFactor);
645  }
646 
647  // apply X offset
648  int32_t offsetX = _xShapeOffset->getValue();
649  if(offsetX) {
650  double shapeMaxLength = _shapeMaxLength;
651  // if offset is bigger than length of shape, then shape is moved out of range and only Y offset is applied
652  if(offsetX > shapeMaxLength || offsetX < -shapeMaxLength) {
653  for(uint32_t i = 0; i < _shapeMaxLength; i++) {
654  _shapeDataVector[i] = (offsetY * scaleFactor);
655  }
656  }
657  else {
658  if(offsetX > 0) {
659  for(uint32_t i = _shapeMaxLength - 1; i >= (uint32_t)offsetX; i--) {
660  _shapeDataVector[i] = _shapeDataVector[i - offsetX];
661  }
662 
663  for(int i = 0; i < offsetX; i++) {
664  _shapeDataVector[i] = (offsetY * scaleFactor);
665  }
666  }
667  else if(offsetX < 0) {
668  for(uint32_t i = 0; i < maxSampleIndex; i++) {
669  _shapeDataVector[i] = _shapeDataVector[i - offsetX];
670  }
671 
672  for(uint32_t i = -offsetX; i < _shapeMaxLength; i++) {
673  _shapeDataVector[i] = (offsetY * scaleFactor);
674  }
675  }
676  }
677  }

704 template<typename SimpleType>
705 void CsaShapeGenerator<SimpleType>::generateFromPoints() {
706  std::vector<int> t;
707  std::vector<SimpleType> val_t;
708 
709  for(uint16_t i = 0; i < _shapeTablePointsNumber; i++) {
710  t.push_back(_shapeTablePointsX[i]->getValue());
711  val_t.push_back(_shapeTablePointsY[i]->getValue());
712  _logger(Loger::MAX_DETAIL) << __func__ << ": "
713  << "Table point: " << _shapeTablePointsY[i]->getValue()
714  << " at index: " << _shapeTablePointsX[i]->getValue() << std::endl;
715  }
716 
717  int t_n = 0, dt = 0;
718 
719  for(uint16_t i = 0; i < _shapeTablePointsNumber - 1; i++) {
720  if(t[i] > t[i + 1]) {
721  _logger(Loger::ERROR) << __func__ << ": "
722  << "Times must increase, but: " << t[i] << " is bigger than: " << t[i + 1] << std::endl;
723  return;
724  }
725  }
726 
727  // clean memory
728  for(uint32_t i = 0; i < _shapeMaxLength; i++) _shapeDataVector[i] = 0;
729 
730  uint32_t cnt = 0;
731  for(uint16_t l = 0; l < _shapeTablePointsNumber - 1; l++) {
732  dt = t[l + 1] - t[l];
733 
734  if(dt) {
735  int x1 = t[l];
736  int x2 = t[l + 1];
737  SimpleType y1 = val_t[l];
738  SimpleType y2 = val_t[l + 1];
739 
740  SimpleType a = (y2 - y1) / (x2 - x1);
741  SimpleType b = (y1 * x2 - y2 * x1) / (x2 - x1);
742 
743  for(t_n = t[l]; t_n < t[l] + dt; t_n++) {
744  if(cnt >= _shapeMaxLength) {
745  _logger(Loger::WARNING) << __func__ << ": "
746  << "Max shape length exceeded. Current sample is: " << cnt
747  << " Buffer length is: " << _shapeMaxLength << " Stopping." << std::endl;
748  break;
749  }
750  SimpleType data = (a * t_n + b);
751  _shapeDataVector[cnt] = data;
752  cnt++;
753  }
754  }
755  }
