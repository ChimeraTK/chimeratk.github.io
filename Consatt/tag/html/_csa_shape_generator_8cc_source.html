272 std::vector<SimpleType> CsaShapeGenerator<SimpleType>::getSinglePeriodDataVector() {
273 
274  ShapeType shapeType = static_cast<ShapeType>(_shapeType->getValue());
275 
276  std::vector<SimpleType> pointsVector;
277  SimpleType amplitude = _shapeAmplitude->getValue();
278 
279  uint32_t samplesNoInPeriod = _samplesPerPeriodInUse->getValue();
280 
281 
282  if (shapeType == CsaShapeGenerator::TYPE_RECTANGLE) {
283  uint32_t halfPeriod = samplesNoInPeriod / 2;
284  for (uint32_t i=0; i < halfPeriod; i++) {
285  pointsVector.push_back(amplitude);
286  }
287 
288  for (uint32_t i=halfPeriod; i < samplesNoInPeriod; i++)
289  pointsVector.push_back(0);
290  }
291  else if (shapeType == CsaShapeGenerator::TYPE_SAW_RISING) {
292  for (uint32_t i=0; i < samplesNoInPeriod; i++) {
293  SimpleType val = i * amplitude/(samplesNoInPeriod-1);
294  pointsVector.push_back(val);
295  }
296  }
297  else if (shapeType == CsaShapeGenerator::TYPE_SAW_FALLING) {
298  for (uint32_t i=0; i < samplesNoInPeriod; i++) {
299  SimpleType val = amplitude - (i * amplitude/(samplesNoInPeriod-1));
300  pointsVector.push_back(val);
301  }
302  }
303  else if (shapeType == CsaShapeGenerator::TYPE_CONSTANT) {
304  for (uint32_t i=0; i < samplesNoInPeriod; i++) {
305  pointsVector.push_back(amplitude);
306  }
307  }
308  else if (shapeType == CsaShapeGenerator::TYPE_SINUS) {
309  for (uint32_t i=0; i < samplesNoInPeriod; i++) {
310  SimpleType val = amplitude * sin (2*M_PI*i/samplesNoInPeriod);
311  pointsVector.push_back(val);
312  }
313  }
314  else if (shapeType == CsaShapeGenerator::TYPE_COSIN) {
315  for (uint32_t i=0; i < samplesNoInPeriod; i++) {
316  SimpleType val = amplitude * cos (2*M_PI*i/samplesNoInPeriod);
317  pointsVector.push_back(val);
