65// very, very simple thread safe data container
66template<class T>
67class ThreadSafeData {
68 protected:
69 T _data;
70 std::mutex _mutex;
71
72 public:
73 ThreadSafeData& set(T data) {
74 std::lock_guard<std::mutex> guard(_mutex);
75 _data = data;
76 return *this;
77 }
78
79 T get() {
80 std::lock_guard<std::mutex> guard(_mutex);
81 return _data;
82 }
83};
84
85class ThreadSafeString : public ThreadSafeData<std::string> {
86 ThreadSafeString& append(std::string data) {
87 std::lock_guard<std::mutex> guard(_mutex);
88 _data.append(data);
89 return *this;
90 }
91
92 ThreadSafeString& clear() {
93 std::lock_guard<std::mutex> guard(_mutex);
94 _data.clear();
95 return *this;
96 }
97};
