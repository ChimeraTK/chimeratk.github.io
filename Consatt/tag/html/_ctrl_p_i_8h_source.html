41 void execute(double pInput, double pSetpoint, bool executeP = true, bool executeI = true) {
42 input = pInput;
43 setpoint = pSetpoint;
44 error = setpoint - input;
45 if(!saturated) {
46 // calc P part
47 if(executeP)
48 pPart = pGain * error;
49 else
50 pPart = 0;
51 // calc I part
52 if(executeI) {
53 integral += error * dt;
54 iPart = iGain * integral;
55 }
56 else
57 iPart = 0;
58
59 // integral += iGain * (error * dt);
60 // iPart = integral;
61 }
62
63 // check if controller not in saturation
64 if(outputWindow == 0 || std::fabs(pPart + iPart) < std::fabs(outputWindow)) {
65 validPPart = pPart;
66 validIPart = iPart;
67 saturated = false;
68 }
69 else if(!saturated) {
70 validPPart = pPart;
71 validIPart = iPart;
72 saturated = true;
73 }
74
75 validOutput = validPPart + validIPart;
76
77 // check whether step is big enough or we are already saturated
78 if(std::fabs(validOutput - appliedOutput) > std::fabs(minimalStep) || saturated) {
79 appliedOutput = validOutput;
80 }
81
82 // check whether calculated output is inside allowed window and if not truncate output
83 if(outputWindow > 0) {
84 if(appliedOutput > outputWindow)
85 appliedOutput = outputWindow;
86 else if(appliedOutput < -outputWindow)
87 appliedOutput = -outputWindow;
88 }
89 }
90

98 appliedOutput = 0;

101 saturated = false;
102 outputError = false;
103 }
104};
