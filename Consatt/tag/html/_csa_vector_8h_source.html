332void CsaVector<SimpleType>::setSendLengthAndResize(uint32_t length) {
333 if(_dataVector.size() > length) { // make shorter -> first limit send data amount and later limit buffer
334 this->csaSendLength(length);

336 }
337 else if(_dataVector.size() < length) { // make longer -> first extend a buffer later extend send data amount
338 _dataVector.resize(length);
339 this->csaSendLength(length);
340 }
341 else
342 this->csaSendLength(length);
