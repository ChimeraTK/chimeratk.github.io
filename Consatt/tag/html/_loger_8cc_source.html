186std::string Loger::TimeStampGen::getTime() {
187 auto now = std::chrono::high_resolution_clock::now();
188 auto in_time_t = std::chrono::system_clock::to_time_t(now);
189
190 auto millis = std::chrono::duration_cast<std::chrono::milliseconds>(now.time_since_epoch()).count();
191 auto seconds = std::chrono::duration_cast<std::chrono::seconds>(now.time_since_epoch()).count();
192 auto diff = (millis - seconds * 1000);
193
194 std::stringstream ss;
195 if(diff < 10)
196 ss << std::put_time(std::localtime(&in_time_t), "%Y-%m-%dT%H:%M:%S") << ".00" << diff;
197 else if(diff < 100)
198 ss << std::put_time(std::localtime(&in_time_t), "%Y-%m-%dT%H:%M:%S") << ".0" << diff;
199 else
200 ss << std::put_time(std::localtime(&in_time_t), "%Y-%m-%dT%H:%M:%S") << "." << diff;
201 return ss.str();
202}

281void Loger::LogStreamWrapper::endLog() {
282 std::lock_guard<std::mutex> guard(_mutex);
283
284 _message += getThreadIdAsStr();
285
286 auto logStartPos = _message.find(" -> "); // this will also include log level and preamble strings into the message
287 if(logStartPos != std::string::npos) {
288 std::string rawMessage(_message.substr(logStartPos + 3));
289 if(rawMessage.compare(_prevMessage) == 0) {
290 _repeatCnt++;
291
292 std::chrono::milliseconds millisec =
293 std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch());
294 if(_repeatCnt == 1) {
295 _loggingStream << LogStreamWrapper::getTime() << " ->" << _prevMessage << " (repeated)" << std::endl;
296 _lastPrintTime = millisec;
297 _detectionTime = millisec;
298 }
299 else if(std::chrono::duration_cast<std::chrono::milliseconds>(millisec - _lastPrintTime).count() >
300 _repeatTimegapSec * 1000) {
301 _lastPrintTime = millisec;
302
303 auto timeFromStart = std::chrono::duration_cast<std::chrono::milliseconds>(millisec - _detectionTime).count();
304 if((_repeatTimegapSec == 1800) && timeFromStart >= (24 * 60 * 60 * 1000)) { // 24h * 60 min * 60 sec * 1000 ms
305 _repeatTimegapSec = 3600; // 60 minutes
306 _loggingStream << LogStreamWrapper::getTime() << " ->" << _prevMessage << " (message repeated " << _repeatCnt
307 << " times. Change repeat period to " << int(_repeatTimegapSec / 60) << " minutes)"
308 << std::endl;
309 }
310 else if((_repeatTimegapSec == 300) && timeFromStart >= (3 * 60 * 60 * 1000)) { // 3h * 60 min * 60 sec * 1000 ms
311 _repeatTimegapSec = 1800; // 30 minutes
312 _loggingStream << LogStreamWrapper::getTime() << " ->" << _prevMessage << " (message repeated " << _repeatCnt
313 << " times. Change repeat period to " << int(_repeatTimegapSec / 60) << " minutes)"
314 << std::endl;
315 }
316 else if((_repeatTimegapSec == 60) && timeFromStart >= (1 * 60 * 60 * 1000)) { // 1h * 60 min * 60 sec * 1000 ms
317 _repeatTimegapSec = 300; // 5 minutes
318 _loggingStream << LogStreamWrapper::getTime() << " ->" << _prevMessage << " (message repeated " << _repeatCnt
319 << " times. Change repeat period to " << int(_repeatTimegapSec / 60) << " minutes)"
320 << std::endl;
321 }
322 else
323 _loggingStream << LogStreamWrapper::getTime() << " ->" << _prevMessage << " (message repeated " << _repeatCnt
324 << " times)" << std::endl;
325 }
326 }
327 else if(_repeatCnt) {
328 _loggingStream << LogStreamWrapper::getTime() << " ->" << _prevMessage << " (message repeated in total "
329 << (_repeatCnt + 1) << " times)" << std::endl;
330 _repeatCnt = 0;
331 _repeatTimegapSec = 60;
332 }
333 _prevMessage = rawMessage;
334
335 if(_repeatCnt == 0) _loggingStream << _message << std::endl;
336 }
337 else
338 _loggingStream << _message << std::endl;
339
340 _message.clear();
341}
