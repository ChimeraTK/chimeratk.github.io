147 VirtualTime getRemaining() {
148 VirtualTime remaining = std::numeric_limits<VirtualTime>::max();
149 boost::fusion::for_each(*timers, findRemaining(remaining));
150 if(remaining >= std::numeric_limits<VirtualTime>::max()) return -1;
151 return remaining;
152 }
153
158 VirtualTime getRemaining(std::string name) {
159 VirtualTime remaining = -2;
160 boost::fusion::for_each(*timers, findRemainingByName(*this, name, remaining));
161 return remaining;
162 }
163
168 bool advance(VirtualTime tval) {
169 if(tval < 0) return false;
170 current += tval;
171 bool hasFired = false;
172 do {
173 VirtualTime tstep = fmin(tval, getRemaining());
174 boost::fusion::for_each(*timers, advanceTimer(tstep, hasFired));
175 tval -= tstep;
176 } while(tval > 0);
177 return hasFired;
178 }
179
183 bool advance() { return advance(getRemaining()); }
184
190 bool advance(std::string name) { return advance(getRemaining(name)); }
191
193 VirtualTime getCurrent() { return current; }
