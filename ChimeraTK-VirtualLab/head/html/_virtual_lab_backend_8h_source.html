394 void write(uint64_t bar, uint64_t address, int32_t const* data, size_t sizeInBytes) override {
395 // save as last written data, for use inside guards of the events we may
396 // trigger now
397 {
398 std::lock_guard<std::mutex> guard(mutex);
399 lastWrittenData = data;
400 lastWrittenSize = sizeInBytes;
401 }
402
403 // perform the actual write
404 DummyBackend::write(bar, address, data, sizeInBytes);
405
406 // trigger events
407 {
408 std::lock_guard<std::mutex> guard(mutex);
409 regWriteEvents(bar, address, sizeInBytes);
410 }

425 static bool isAddressInRange(
426 const NumericAddressedRegisterInfo& registerInfo, uint8_t bar, uint32_t address, size_t length) {
427 assert(registerInfo.elementPitchBits % 8 == 0);
428 return (bar == registerInfo.bar && address >= registerInfo.address &&
429 address + length <= registerInfo.address + registerInfo.nElements * registerInfo.elementPitchBits / 8);
430 }

446 template<class timerEvent>
447 class Timer {
448 public:
449 Timer(derived* _dev) : dev(_dev), request(-1), current(0) {}
450
452 void set(VirtualTime tval) { request = tval + current; }
453
455 void clear() { request = -1; }
456
461 bool advance(VirtualTime tval) {
462 if(tval < 0) return false;
463 current += tval;
464 if(request > 0 && current >= request) {
465 request = -1;
466 try {
467 std::unique_lock<std::mutex> lk(dev->mutex);
468 dev->theStateMachine.process_event(timerEvent());
469 }
470 catch(...) {
471 std::cerr << "ERROR in VirtualLabBackend: Exception thrown while "
472 "processing timer event. The state "
473 << "machine is now in an unusable condition. Make sure to "
474 "catch all exceptions in your "
475 << "actions!" << std::endl;
476 }
477 return true;
478 }
479 return false;
480 }
481
484 VirtualTime getRemaining() {
485 if(request > 0) {
486 return request - current;
487 }
488 else {
489 return -1;
490 }
491 }
492
494 VirtualTime getCurrent() { return current; }
495
496 protected:
498 derived* dev;
499
501 VirtualTime request;
502
504 VirtualTime current;
505 };
