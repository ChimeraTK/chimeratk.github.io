159 inline const STATE& getState(VirtualTime time, bool forceNoInterpolation = false) {
160#ifndef ENABLE_EXPERIMENTAL_FEATURES
161 (void)forceNoInterpolation; // avoid warning
162#endif
163 // check if time is the current time and return the latest element
164 if(time >= currentTime && time < currentTime + validityPeriod) {
165 return getLatestState();
166 }
167 // search in buffer: find the first element after the requested time
168 auto it = buffer.upper_bound(time);
169 if(it == buffer.begin()) {
170 std::stringstream s;
171 s << "Value request is too far into the past: ";
172 s << "requested time = " << time << ", oldest history = " << buffer.begin()->first;
173 s << ", current time = " << currentTime;
174 throw ChimeraTK::logic_error(s.str());
175 }
176 // if this is end(), a new value needs to be computed
177 if(it == buffer.end()) {
178#ifdef ENABLE_EXPERIMENTAL_FEATURES
179 if(!forceNoInterpolation) {
180 return getValueInterpolated(time, buffer.rbegin()->first);
181 }
182 else {
183#endif
184 return getValueFromCallback(time);
185#ifdef ENABLE_EXPERIMENTAL_FEATURES
186 }
187#endif
188 }
189 // decrement to get the most recent sample before the requested time
190 --it;
191 // if sample is too old, request one via callback
192 if(time >= it->first + validityPeriod) {
193#ifdef ENABLE_EXPERIMENTAL_FEATURES
194 if(!forceNoInterpolation) {
195 return getValueInterpolated(time, it->first);
196 }
197 else {
198#endif
199 return getValueFromCallback(time);
200#ifdef ENABLE_EXPERIMENTAL_FEATURES
201 }
202#endif
203 }
204 // return value from buffer
205 return it->second;
206 }

256 inline void truncateFuture(VirtualTime time) {
257 if(time < currentTime) {
258 auto firstToDelete = buffer.upper_bound(time);
259 // Note: getState() already ensures that the request does not go before
260 // the first map element, so we are never deleting the entire map here.
261 buffer.erase(firstToDelete, buffer.end());
262 currentTime = buffer.rbegin()->first;
263 }
264 }

269 inline void truncatePast() {
270 // clear old values from history
271 if(buffer.size() > 1 && buffer.begin()->first < (currentTime - historyLength)) {
272 auto firstToKeep = buffer.upper_bound(currentTime - historyLength - 1);
273 buffer.erase(buffer.begin(), firstToKeep);
274 }

279 inline const STATE& getValueInterpolated(VirtualTime timeRequested, VirtualTime previousStep) {
280#ifndef ENABLE_EXPERIMENTAL_FEATURES
281 class ExperimentalFeatureNotEnabled {};
282 throw ExperimentalFeatureNotEnabled();
283#endif
284 // no interpolation or requested time is later than the interpolation period
285 // (maxGap)
286 if(!enableInterpolation || timeRequested >= previousStep + maxGap) {
287 return getValueFromCallback(timeRequested);
288 }
289
290 // save interpolated state into the buffer and return it
291 auto state1 = getState(previousStep, true);
292 auto state2 = getState(previousStep + maxGap, true);
293 buffer[timeRequested] = interpolate(state1, state2, previousStep, previousStep + maxGap, timeRequested);
294 return buffer[timeRequested];
295 }

299 inline const STATE& getValueFromCallback(VirtualTime time) {
300 // If request goes into the past (w.r.t. currentTime), erase newer states to
301 // force recomputing them.
302 truncateFuture(time);
303
304 // Compute the number of steps necessary to reach the requested time,
305 // obaying hugeGap. The last step at the requested time itself is not
306 // computed here, since maxGap shall be obeyed in the last gap (see below).
307 // Note: this is effectively rounding-up the integer division
308 // (time-CurrentTime)/hugeGap
309 unsigned int nHugeSteps = (time - currentTime - 1) / hugeGap + 1;
310
311 // Loop over the time in nHugeSteps steps. The first step is potentially
312 // smaller, all consequtive steps are maxGap big. Omit the last step, so the
313 // requested time is not yet reached.
314 for(unsigned int i = 0; i < nHugeSteps - 1; i++) {
315 VirtualTime t = time - (nHugeSteps - 1) * hugeGap + i * hugeGap;
316
317 // obtain the new state
318 STATE state = compute(t);
319
320 // save value into buffer and update the current time
321 buffer[t] = state;
322 currentTime = t;
323 }
324
325 // Compute the number of steps necessary to reach the requested time,
326 // obaying maxGap. Note: this is effectively rounding-up the integer
327 // division (time-CurrentTime)/maxGap
328 unsigned int nSteps = (time - currentTime - 1) / maxGap + 1;
329
330 // Loop over the time in nSteps steps. The first step is potentially
331 // smaller, all consequtive steps are maxGap big.
332 for(unsigned int i = 0; i < nSteps; i++) {
333 VirtualTime t = time - (nSteps - 1) * maxGap + i * maxGap;
334
335 // obtain the new state
336 STATE state = compute(t);
337
338 // save value into buffer and update the current time
339 buffer[t] = state;
340 currentTime = t;
341 }
342
343 // remove old history
344 truncatePast();
345
346 // return the new state (do not use buffer[], as it will silently insert new
347 // objects if something went wrong)
348 return buffer.find(time)->second;
349 }
