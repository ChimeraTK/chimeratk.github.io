114arguments_t parse_arguments(int argc, char* argv[]) {
115 arguments_t args;
116 const std::string raw_prefix("(");
117
118 // Declare a group of options that will be
119 // allowed only on command line
120 po::options_description generic("Generic options");
121 generic.add_options()("help,h", "produce help message")("config,c", po::value<string>(),
122 "set configuration file name containing key=value pairs (hints: use long forms like 'program' or 'device' and "
123 "set on/off flags to true)");
124
125 // Declare a group of options that will be
126 // allowed both on command line and in
127 // config file
128 po::options_description config("Configuration");
129 config.add_options()("program,p", po::bool_switch()->default_value(false), "erase and program memory")("verify,v",
130 po::bool_switch()->default_value(false), "verify memory")("dump,m", po::bool_switch()->default_value(false),
131 "dump memory")("reload,r", po::bool_switch()->default_value(false), "reload FPGA")("interface,i",
132 po::value<string>(), "memory interface (spi/jtag)")("firmware_file,f", po::value<string>(), "FPGA firmware file")(
133 "device,d", po::value<string>(), "device name")("address,a", po::value<string>(),
134 ("address and bar of boot area in the format: [address]b[bar]. Use 'default' to refer to the default address (" +
135 std::to_string(PROG_DEFAULT_ADDRESS) + "b" + std::to_string(PROG_DEFAULT_BAR) + ").")
136 .c_str())("flash_size,s", po::value<string>(),
137 "size of flash chip to dump, in bytes - example for a 256M (32MiB) chip: -s 33554432")(
138 "dmap,D", po::value<string>(), "DMAP file path")("map,M", po::value<string>(), "MAP file path")("boot_area,R",
139 po::value<string>(),
140 "boot area name in MAP file")("quiet,q", po::bool_switch()->default_value(false), "disable progress bar");
141
142 po::options_description cmdline_options;
143 cmdline_options.add(generic).add(config);
144
145 po::options_description config_file_options;
146 config_file_options.add(config);
147
148 po::variables_map vm;
149 po::store(po::parse_command_line(argc, argv, cmdline_options), vm);
150 po::notify(vm);
151
152 if(vm.count("help")) {
153 usage(argv[0]);
154 cout << cmdline_options << "\n";
155 exit(1);
156 }
157
158 if(vm.count("config")) {
159 std::string config_file_path = vm["config"].as<std::string>();
160 cout << "Configuration file path: " << config_file_path << "\n";
161
162 po::store(po::parse_config_file<char>(config_file_path.c_str(), config_file_options), vm);
163 po::notify(vm);
164 }
165
166 args.action_programming = vm["program"].as<bool>();
167 args.action_verification = vm["verify"].as<bool>();
168 args.action_dump = vm["dump"].as<bool>();
169 args.action_reload = vm["reload"].as<bool>();
170 args.quiet_mode = vm["quiet"].as<bool>();
171
172 if(vm.count("device")) {
173 args.device_name = vm["device"].as<std::string>();
174 if(args.device_name.compare(0, raw_prefix.size(), raw_prefix) == 0) args.device_name_raw = true;
175 }
176
177 if(vm.count("interface")) {
178 std::string interface = vm["interface"].as<std::string>();
179 if(interface.empty())
180 args.interface = ProgrammingInterface(ProgrammingInterface::INTERFACE_NONE);
181 else if(interface.compare("spi") == 0)
182 args.interface = ProgrammingInterface(ProgrammingInterface::INTERFACE_SPI);
183 else if(interface.compare("jtag") == 0)
184 args.interface = ProgrammingInterface(ProgrammingInterface::INTERFACE_JTAG);
185 else
186 throw std::invalid_argument("Unknown protocol, use spi or jtag\n");
187 }
188
189 if(vm.count("firmware_file")) args.firmware_file_path = vm["firmware_file"].as<std::string>();
190
191 if(vm.count("dmap")) args.dmap_file_path = vm["dmap"].as<std::string>();
192
193 if(vm.count("boot_area")) args.map_area_name = vm["boot_area"].as<std::string>();
194
195 if(vm.count("flash_size")) args.flash_size = std::stoi(vm["flash_size"].as<std::string>());
196
197 if(vm.count("address")) {
198 std::string address_input = vm["address"].as<std::string>();
199 if(address_input == "default") {
200 address_input = std::to_string(PROG_DEFAULT_ADDRESS) + "b" + std::to_string(PROG_DEFAULT_BAR);
201 }
202 std::cout << "Address input: " << address_input << std::endl;
203 uint32_t b_counter = 0;
204 std::size_t b_position = address_input.find_first_of("b");
205 while(b_position != std::string::npos) {
206 b_counter++;
207 b_position = address_input.find_first_of("b", b_position + 1);
208 }
209 if(b_counter != 1) throw std::invalid_argument("Wrong format of programmer address.\n");
210
211 b_position = address_input.find_first_of("b");
212
213 try {
214 std::string address_ch = address_input.substr(0, b_position);
215 std::string bar_ch = address_input.substr(b_position + 1, address_input.length() - b_position);
216
217 size_t idx;
218 uint32_t address = std::stoi(address_ch, &idx);
219 if(idx != address_ch.length()) throw std::invalid_argument("Wrong format of programmer address.\n");
220
221 uint8_t bar = std::stoi(bar_ch, &idx);
222 if(idx != bar_ch.length()) throw std::invalid_argument("Wrong format of programmer address.\n");
223
224 args.address = address;
225 args.bar = bar;
226 args.addressValid = true;
227 }
228 catch(...) {
229 throw std::invalid_argument("Wrong format of programmer address.\n");
230 }
231 }
232
233 return args;
234}

236void verify_arguments(arguments_t arguments) {
237 if(!arguments.action_programming && !arguments.action_reload && !arguments.action_verification &&
238 !arguments.action_dump) {
239 throw std::logic_error("Please specify action(s) to be executed");
240 }
241
242 if((arguments.action_programming || arguments.action_verification) && arguments.action_dump) {
243 throw std::logic_error("Cannot program/verify a new firmware and also dump the current firmware");
244 }
245
246 if(arguments.action_dump && arguments.flash_size == 0) {
247 throw std::logic_error("Cannot dump a flash image without knowing the size of the chip.\n"
248 "Please specify the number of bytes to dump.\n");
249 }
250
251 if(arguments.flash_size && (arguments.flash_size % 1024)) {
252 throw std::logic_error("The flash size must be a multiple of 1024.\n");
253 }
254
255 if(arguments.device_name.empty()) {
256 throw std::logic_error("Cannot make any actions without device name.\n"
257 "Please specify device for programming.\n");
258 }
259
260 if(arguments.interface.getType() == ProgrammingInterface::INTERFACE_NONE) {
261 throw std::logic_error("Programming interface (SPI/JTAG) is missing.\n"
262 "Please specify the interface appropriate for the device.\n");
263 }
264
265 if((arguments.action_programming || arguments.action_verification) && arguments.firmware_file_path.empty()) {
266 throw std::logic_error("Firmware file is missing.\n"
267 "Please specify the location of file with firmware.\n");
268 }
269 else if(arguments.action_programming || arguments.action_verification) {
270 FILE* tmp_file = fopen(arguments.firmware_file_path.c_str(), "r");
271 if(tmp_file == NULL)
272 throw std::invalid_argument("Cannot open firmware file. Please check if file exists and you have "
273 "rights to access it.\n");
274 else
275 fclose(tmp_file);
276 }
277}

279int main(int argc, char* argv[]) {
280 boost::shared_ptr<MtcaProgrammerBase> programmer;
281
282 arguments_t arguments;
283
284 try {
285 cout << "\nmtca4u_fw_programmer ver. " << VERSION << endl << endl;
286 arguments = parse_arguments(argc, argv);
287 verify_arguments(arguments);
288
289 // Quite mode disables the progress bar
290 ProgressBar::setDoNotShow(arguments.quiet_mode);
291
292 if(!arguments.dmap_file_path.empty()) // DMAP mode
293 {
294 // printf( "DMAP file is missing.\n"
295 // "Please specify the location of DMAP file.\n");
296
297 cout << "Input mode - DMAP" << endl;
298 cout << "Firmware file: " << arguments.firmware_file_path << endl;
299 cout << "DMAP file: " << arguments.dmap_file_path << endl;
300 cout << "Device name: " << arguments.device_name << endl;
301 cout << "Module name in MAP file: " << arguments.map_area_name << endl;
302
303 switch(arguments.interface.getType()) {
304 case ProgrammingInterface::INTERFACE_SPI:
305 programmer = boost::shared_ptr<MtcaProgrammerBase>(new MtcaProgrammerSPI(
306 ProgAccessDmap(arguments.device_name, arguments.dmap_file_path, arguments.map_area_name)));
307 break;
308 case ProgrammingInterface::INTERFACE_JTAG:
309 programmer = boost::shared_ptr<MtcaProgrammerBase>(new MtcaProgrammerJTAG(
310 ProgAccessDmap(arguments.device_name, arguments.dmap_file_path, arguments.map_area_name)));
311 break;
312 default:
313 throw std::invalid_argument("Unknown interface\n\n");
314 }
315 }
316 else if(arguments.device_name_raw && !arguments.addressValid) // MAP mode
317 {
318 cout << "Input mode - MAP" << endl;
319 cout << "Firmware file: " << arguments.firmware_file_path << endl;
320 cout << "Device name: " << arguments.device_name << endl;
321 cout << "Module name in MAP file: " << arguments.map_area_name << endl;
322
323 switch(arguments.interface.getType()) {
324 case ProgrammingInterface::INTERFACE_SPI:
325 programmer = boost::shared_ptr<MtcaProgrammerBase>(
326 new MtcaProgrammerSPI(ProgAccessMap(arguments.device_name, arguments.map_area_name)));
327 break;
328 case ProgrammingInterface::INTERFACE_JTAG:
329 programmer = boost::shared_ptr<MtcaProgrammerBase>(
330 new MtcaProgrammerJTAG(ProgAccessMap(arguments.device_name, arguments.map_area_name)));
331 break;
332 default:
333 throw std::invalid_argument("Unknown interface\n\n");
334 }
335 }
336 else // RAW mode
337 {
338 if(!arguments.device_name_raw) {
339 throw std::invalid_argument("Wrong device name.\nPlease specify device "
340 "name in 'sdm' format\n\n");
341 }
342
343 cout << "Input mode - Direct" << endl;
344 cout << "Firmware file: " << arguments.firmware_file_path << endl;
345 cout << "Device name: " << arguments.device_name << endl;
346 cout << "Bar: " << (uint32_t)arguments.bar << endl;
347 cout << "Address: " << arguments.address << endl;
348
349 switch(arguments.interface.getType()) {
350 case ProgrammingInterface::INTERFACE_SPI:
351 programmer = boost::shared_ptr<MtcaProgrammerBase>(
352 new MtcaProgrammerSPI(ProgAccessRaw(arguments.device_name, arguments.bar, arguments.address)));
353 break;
354 case ProgrammingInterface::INTERFACE_JTAG:
355 programmer = boost::shared_ptr<MtcaProgrammerBase>(
356 new MtcaProgrammerJTAG(ProgAccessRaw(arguments.device_name, arguments.bar, arguments.address)));
357 break;
358 default:
359 throw std::invalid_argument("Unknown interface\n\n");
360 }
361 }
362 cout << endl << endl;
363
364 if(arguments.action_programming || arguments.action_verification) {
365 if(!programmer->checkFirmwareFile(arguments.firmware_file_path)) {
366 throw std::runtime_error("Incorrect firmware file\n");
367 }
368 }
369 if(arguments.action_programming) {
370 programmer->erase();
371 programmer->program(arguments.firmware_file_path);
372 }
373 if(arguments.action_dump) {
374 if(programmer->dump(arguments.firmware_file_path, arguments.flash_size) == false) return 1;
375 }
376 if(arguments.action_verification) {
377 if(programmer->verify(arguments.firmware_file_path) == false) return 1;
378 }
379 if(arguments.action_reload) {
380 programmer->rebootFPGA();
381 }
382 }
383 catch(const std::exception& e) {
384 std::cerr << "\nError: " << e.what() << std::endl;
385 return 1;
386 }
387 catch(...) {
388 std::cerr << "An unexpected error has occured" << std::endl;
389 return 1;
390 }
391
392 return 0;
393}
