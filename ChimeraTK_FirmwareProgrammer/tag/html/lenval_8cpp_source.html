5/*****************************************************************************
6 * Function: value
7 * Description: Extract the long value from the lenval array.
8 * Parameters: plvValue - ptr to lenval.
9 * Returns: long - the extracted value.
10 *****************************************************************************/
11long value(lenVal* plvValue) {
12 long lValue; /* result to hold the accumulated result */
13 short sIndex;
14
15 lValue = 0;
16 for(sIndex = 0; sIndex < plvValue->len; ++sIndex) {
17 lValue <<= 8; /* shift the accumulated result */
18 lValue |= plvValue->val[sIndex]; /* get the last byte first */
19 }
20
21 return (lValue);
22}

37/*****************************************************************************
38 * Function: EqualLenVal
39 * Description: Compare two lenval arrays with an optional mask.
40 * Parameters: plvTdoExpected - ptr to lenval #1.
41 * plvTdoCaptured - ptr to lenval #2.
42 * plvTdoMask - optional ptr to mask (=0 if no mask).
43 * Returns: short - 0 = mismatch; 1 = equal.
44 *****************************************************************************/
45short EqualLenVal(lenVal* plvTdoExpected, lenVal* plvTdoCaptured, lenVal* plvTdoMask) {
46 short sEqual;
47 short sIndex;
48 unsigned char ucByteVal1;
49 unsigned char ucByteVal2;
50 unsigned char ucByteMask;
51
52 sEqual = 1;
53 sIndex = plvTdoExpected->len;
54
55 while(sEqual && sIndex--) {
56 ucByteVal1 = plvTdoExpected->val[sIndex];
57 ucByteVal2 = plvTdoCaptured->val[sIndex];
58 if(plvTdoMask) {
59 ucByteMask = plvTdoMask->val[sIndex];
60 ucByteVal1 &= ucByteMask;
61 ucByteVal2 &= ucByteMask;
62 }
63 if(ucByteVal1 != ucByteVal2) {
64 sEqual = 0;
65 }
66 }
67
68 return (sEqual);
69}

85/*****************************************************************************
86 * Function: SetBit
87 * Description: set the (byte, bit) of lv equal to val
88 * Example: SetBit("00000000",byte, 1) equals "01000000".
89 * Parameters: plv - ptr to lenval.
90 * iByte - the byte to get the bit from.
91 * iBit - the bit number (0=msb).
92 * sVal - the bit value to set.
93 * Returns: void.
94 *****************************************************************************/
95void SetBit(lenVal* plv, int iByte, int iBit, short sVal) {
96 unsigned char ucByteVal;
97 unsigned char ucBitMask;
98
99 ucBitMask = (unsigned char)(1 << (7 - iBit));
100 ucByteVal = (unsigned char)(plv->val[iByte] & (~ucBitMask));
101
102 if(sVal) {
103 ucByteVal |= ucBitMask;
104 }
105 plv->val[iByte] = ucByteVal;

108/*****************************************************************************
109 * Function: AddVal
110 * Description: add val1 to val2 and store in resVal;
111 * assumes val1 and val2 are of equal length.
112 * Parameters: plvResVal - ptr to result.
113 * plvVal1 - ptr of addendum.
114 * plvVal2 - ptr of addendum.
115 * Returns: void.
116 *****************************************************************************/
117void addVal(lenVal* plvResVal, lenVal* plvVal1, lenVal* plvVal2) {
118 unsigned char ucCarry;
119 unsigned short usSum;
120 unsigned short usVal1;
121 unsigned short usVal2;
122 short sIndex;
123
124 plvResVal->len = plvVal1->len; /* set up length of result */
125
126 /* start at least significant bit and add bytes */
127 ucCarry = 0;
128 sIndex = plvVal1->len;
129 while(sIndex--) {
130 usVal1 = plvVal1->val[sIndex]; /* i'th byte of val1 */
131 usVal2 = plvVal2->val[sIndex]; /* i'th byte of val2 */
132
133 /* add the two bytes plus carry from previous addition */
134 usSum = (unsigned short)(usVal1 + usVal2 + ucCarry);
135
136 /* set up carry for next byte */
137 ucCarry = (unsigned char)((usSum > 255) ? 1 : 0);
138
139 /* set the i'th byte of the result */
140 plvResVal->val[sIndex] = (unsigned char)usSum;
141 }
142}
