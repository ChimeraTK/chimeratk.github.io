23bool MtcaProgrammerJTAG::checkFirmwareFile(std::string firmwareFile) {
24 FILE* input_file;
25 unsigned char buffer[16];
26 bool ret = true;
27
28 input_file = fopen(firmwareFile.c_str(), "r");
29 if(input_file == NULL) {
30 throw std::invalid_argument("Cannot open XSVF file. Maybe the file does not exist");
31 }
32 else {
33 size_t bytes_read = fread(buffer, 1, 16, input_file);
34 if(bytes_read != 16) {
35 fclose(input_file);
36 throw std::runtime_error("Cannot read verification pattern from XSVF file");
37 }
38
39 for(int i = 0; i < 16; i++) {
40 if(buffer[i] != xsvf_pattern[i]) {
41 ret = false; // Incorrect XSVF file
42 break;
43 }
44 }
45
46 fclose(input_file);
47 }
48
49 return ret;
50}

110/* readTDOBit: Implement to return the current value of the JTAG TDO signal.*/
111/* read the TDO bit from port */
112unsigned char MtcaProgrammerJTAG::readTDOBit() {
113 // TDO checked on falling edge of CLK
114 reg_tck = 0x0;
115 reg_tck.write();
116 // make sure TCK is set low
117 while(1) {
118 reg_tck.read();
119 if(reg_tck == 0) break;
120 }
121
122 reg_tdo.read();
123 return ((unsigned char)(reg_tdo));
124}
