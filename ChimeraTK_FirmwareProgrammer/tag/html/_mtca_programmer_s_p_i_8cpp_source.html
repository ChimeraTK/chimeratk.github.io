63 bool MtcaProgrammerSPI::checkFirmwareFile(std::string firmwareFile) {
64  FILE* input_file;
65  unsigned char buffer[16];
66  bool ret = true;
67  bool bit_file = true;
68 
69  input_file = fopen(firmwareFile.c_str(), "r");
70  if(input_file == NULL) {
71  throw std::invalid_argument("Cannot open bit file. Maybe the file does not exist");
72  }
73  else {
74  // Check if it is a 'bit' file
75  size_t bytes_read = fread(buffer, 1, 14, input_file);
76  if(bytes_read != 14) {
77  fclose(input_file);
78  throw std::runtime_error("Cannot read verification pattern from bitstream file");
79  }
80 
81  for(int i = 0; i < 14; i++) {
82  if(buffer[i] != bit_pattern[i]) {
83  bit_file = false; // Incorrect bit file
84  break;
85  }
86  }
87  }
88 
89  if(!bit_file) {
90  // Check if it is a 'bin' file
91  long int offset = findDataOffset(input_file);
92  if(offset > 512 or offset < 0) ret = false;
93  }
94 
95  fclose(input_file);
96 
97  return ret;
98 }

125 bool MtcaProgrammerSPI::verify(std::string firmwareFile) {
126  reg_spi_divider = 10;
127  reg_spi_divider.write();
128 
129  uint64_t memID = getMemoryId();
130  if(!checkMemoryId(memID)) throw std::runtime_error("Unknown, not present or busy SPI prom");
131 
132  unsigned char buffer[1024];
133  uint32_t data;
134  unsigned int bread;
135  unsigned int addr = 0;
136  unsigned int file_size;
137  long int offset = 0;
138  FILE* f = fopen(firmwareFile.c_str(), "rb");
139 
140  printf("\nVerification\n");
141 
142  if(!f) throw std::invalid_argument("Cannot open firmware file");
143 
144  offset = findDataOffset(f);
145  if(offset == -1) throw std::runtime_error("Cannot find start sequence in the file");
146 
147  fseek(f, 0, SEEK_END);
148  file_size = ftell(f);
149  fseek(f, offset, SEEK_SET);
150 
151  uint64_t mem_id = getMemoryId();
152  addressing_mode_t addr_mode = known_proms.at(mem_id).addressing_mode;
153 
154  ProgressBar progress;
155  do {
156  bread = fread(buffer, 1, 1024, f);
157  if(bread == 0) {
158  printf("\nverified %d bytes\n\n", addr);
159  if(ferror(f)) {
160  fclose(f);
161  throw std::runtime_error("Error reading firmware file");
162  }
163  fclose(f);
164  return true;
165  }
166  reg_control = 0; // clear reg control status before write to buffer
167  reg_control.write();
168 
169  reg_area_write[0] = getCommand("FAST_READ", addr_mode);
170  uint32_t reg_offset = writeAddress(addr, addr_mode);
171  reg_area_write.write();
172  reg_bytes_to_write = reg_offset;
173  reg_bytes_to_write.write();
174 
175  reg_bytes_to_read = bread - 1;
176  reg_bytes_to_read.write();
177 
178  reg_control = (PCIE_V5 | SPI_PROG | SPI_R_NW | SPI_START);
179  reg_control.write();
180  waitForSpi();
181 
182  reg_area_read.read();
183  for(unsigned int i = 0; i < bread; i++) {
184  data = reg_area_read[i];
185  if(buffer[i] != (data & 0xff)) {
186  fprintf(stderr, "\nVerify error at address %u: read 0x%02x instead of 0x%02x\n", addr + i, (data & 0xff),
187  buffer[i]);
188  return false;
189  }
190  }
191  addr = addr + bread;
192  progress.update(file_size, addr);
193  } while(1);
194 }

196 bool MtcaProgrammerSPI::dump(std::string firmwareFile, uint32_t imageSize) {
197  reg_spi_divider = 10;
198  reg_spi_divider.write();
199 
200  uint64_t memID = getMemoryId();
201  if(!checkMemoryId(memID)) throw std::runtime_error("Unknown, not present or busy SPI prom");
202 
203  uint8_t data;
204  unsigned int bwritten;
205  unsigned int addr = 0;
206  FILE* f;
207 
208  uint64_t mem_id = getMemoryId();
209  addressing_mode_t addr_mode = known_proms.at(mem_id).addressing_mode;
210 
211  printf("\nDumping %d bytes from flash to %s\n\n", imageSize, firmwareFile.c_str());
212  f = fopen(firmwareFile.c_str(), "wb");
213  if(!f) throw std::invalid_argument("Cannot open firmware file");
214 
215  addr = 0;
216  ProgressBar progress;
217  do {
218  reg_control = 0; // clear reg control status before write to buffer
219  reg_control.write();
220 
221  reg_area_write[0] = getCommand("FAST_READ", addr_mode);
222  uint32_t reg_offset = writeAddress(addr, addr_mode);
223  reg_area_write.write();
224  reg_bytes_to_write = reg_offset;
225  reg_bytes_to_write.write();
226 
227  reg_bytes_to_read = 1024 - 1;
228  reg_bytes_to_read.write();
229 
230  reg_control = (PCIE_V5 | SPI_PROG | SPI_R_NW | SPI_START);
231  reg_control.write();
232  waitForSpi();
233 
234  reg_area_read.read();
235  for(unsigned int i = 0; i < 1024; i++) {
236  data = reg_area_read[i] & 0xff;
237  bwritten = fwrite(&data, 1, 1, f);
238  addr++;
239  if(bwritten != 1) {
240  throw std::runtime_error("Error writing firmware file");
241  }
242  }
243  progress.update(imageSize, addr);
244  } while(addr < imageSize);
245  fclose(f);
246  printf("\nDump finished.\n\n");
247  return true;
248 }

259 uint64_t MtcaProgrammerSPI::getMemoryId() {
260  uint64_t mem_id = 0;
261 
262  for(int i = 0; i < 2; i++) // first read returns garbage
263  {
264  reg_control = 0; // clear reg control status before write to buffer
265  reg_control.write();
266 
267  reg_area_write[0] = 0x9F;
268  reg_area_write.write();
269  reg_bytes_to_write = 0;
270  reg_bytes_to_write.write();
271  reg_bytes_to_read = 4;
272  reg_bytes_to_read.write();
273 
274  reg_control = (PCIE_V5 | SPI_PROG | SPI_R_NW | SPI_START);
275  reg_control.write();
276 
277  waitForSpi();
278  }
279 
280  // read 5 bytes
281  reg_area_read.read();
282  for(int i = 0; i < 5; i++) {
283  uint32_t id_val;
284  id_val = reg_area_read[i];
285  mem_id |= ((uint64_t)id_val & 0xFF) << (i * 8);
286  }
287 
288  printf("SPI prom ID: 0x%010lX\n\n", mem_id);
289 
290  return mem_id;
291 }

440 long int MtcaProgrammerSPI::findDataOffset(FILE* file) {
441  unsigned char buffer[512];
442  int offset = -1;
443  const unsigned char c1 = 0xaa;
444  const unsigned char c2 = 0x99;
445  const unsigned char c3 = 0x55;
446  const unsigned char c4 = 0x66;
447  const unsigned char c0 = 0xff;
448  size_t bytes_read;
449  bool bit_file = true;
450 
451  if(!file) return -1;
452 
453  fseek(file, 0, SEEK_SET);
454  bytes_read = fread(buffer, 1, 512, file);
455  if(bytes_read != 512) {
456  fclose(file);
457  throw std::runtime_error("Cannot read header data from bitstream file");
458  }
459  fseek(file, 0, SEEK_SET);
460 
461  /*check if bit file*/
462  for(int i = 0; i < 14; i++) {
463  if(buffer[i] != bit_pattern[i]) {
464  bit_file = false; // Incorrect bit file
465  break;
466  }
467  }
468  /* ok if header shorter than 512 bytes */
469  /*find SYNC sequence*/
470  for(int i = 0; i < 512; i++) {
471  if((buffer[i] == c1) && (buffer[i + 1] == c2) && (buffer[i + 2] == c3) && (buffer[i + 3] == c4)) {
472  for(int j = 1; j < i + 1; j++) {
473  if(buffer[i - j] != c0) {
474  i = i - (j - (j % 16));
475  break;
476  }
477  if(j == i) i = i - (j - (j % 16));
478  }
479  offset = i;
480  break;
481  }
482  }
483  if(offset > 0 and not bit_file) { // binary file take as is
484  offset = 0;
485  }
486  if(offset > 0 and bit_file) { // bit file, remove header
487  offset = offset - 16; // include 16bytes before SYNC WORD. Optional Bus Width Sync.
488  // include as well all NOOP DWORDS up to variable header: 0xFFFFFFFF
489  for(int i = (offset - 1); i > 4; i -= 4) {
490  if((buffer[i] == 0xFF) && (buffer[i - 1] == 0xFF) && (buffer[i - 2] == 0xFF) && (buffer[i - 3] == 0xFF)) {
491  offset = offset - 4;
492  }
493  else {
494  break;
495  }
496  }
497  }
498  return offset;
499 }
