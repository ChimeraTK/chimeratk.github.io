59void MtcaProgrammerBase::initRegisterAccessors(std::string registerPathName) {
60 if(registerPathName == "**DEFAULT**") {
61 if(mDevice.getRegisterCatalogue().hasRegister(PROG_DEFAULT_MODULE_NAME)) {
62 registerPathName = PROG_DEFAULT_MODULE_NAME;
63 }
64 else if(mDevice.getRegisterCatalogue().hasRegister(PROG_DEFAULT_MODULE_NAME2)) {
65 registerPathName = PROG_DEFAULT_MODULE_NAME2;
66 }
67 else if(mDevice.getRegisterCatalogue().hasRegister(PROG_DEFAULT_MODULE_NAME3)) {
68 registerPathName = PROG_DEFAULT_MODULE_NAME3;
69 }
70 else if(mDevice.getRegisterCatalogue().hasRegister(PROG_DEFAULT_MODULE_NAME4)) {
71 registerPathName = PROG_DEFAULT_MODULE_NAME4;
72 }
73 else if(mDevice.getRegisterCatalogue().hasRegister(PROG_DEFAULT_MODULE_NAME5)) {
74 registerPathName = PROG_DEFAULT_MODULE_NAME5;
75 }
76 else {
77 std::cout << "Neither " << PROG_DEFAULT_MODULE_NAME << " nor " << PROG_DEFAULT_MODULE_NAME2 << " nor "
78 << PROG_DEFAULT_MODULE_NAME3 << " nor " << PROG_DEFAULT_MODULE_NAME4 << " nor "
79 << PROG_DEFAULT_MODULE_NAME5 << " register has "
80 << "been found in the device. Please specify the correct name "
81 "of the AREA_BOOT register!"
82 << std::endl;
83 exit(1);
84 }
85 std::cout << "Auto-selected module name: " << registerPathName << std::endl;
86 }
87
88 if(registerPathName != PROG_DEFAULT_MODULE_NAME4 && registerPathName != PROG_DEFAULT_MODULE_NAME5) {
89 reg_area_write.replace(mDevice.getOneDRegisterAccessor<int32_t>(
90 registerPathName, AREA_WRITE_SIZE, AREA_WRITE, {ChimeraTK::AccessMode::raw}));
91 reg_area_read.replace(mDevice.getOneDRegisterAccessor<int32_t>(
92 registerPathName, AREA_READ_SIZE, AREA_READ, {ChimeraTK::AccessMode::raw}));
93 reg_spi_divider.replace(
94 mDevice.getScalarRegisterAccessor<int32_t>(registerPathName, REG_SPI_DIVIDER, {ChimeraTK::AccessMode::raw}));
95 reg_bytes_to_write.replace(
96 mDevice.getScalarRegisterAccessor<int32_t>(registerPathName, REG_BYTES_TO_WRITE, {ChimeraTK::AccessMode::raw}));
97 reg_bytes_to_read.replace(
98 mDevice.getScalarRegisterAccessor<int32_t>(registerPathName, REG_BYTES_TO_READ, {ChimeraTK::AccessMode::raw}));
99 reg_control.replace(
100 mDevice.getScalarRegisterAccessor<int32_t>(registerPathName, REG_CONTROL, {ChimeraTK::AccessMode::raw}));
101 reg_tck.replace(
102 mDevice.getScalarRegisterAccessor<int32_t>(registerPathName, REG_TCK, {ChimeraTK::AccessMode::raw}));
103 reg_tms.replace(
104 mDevice.getScalarRegisterAccessor<int32_t>(registerPathName, REG_TMS, {ChimeraTK::AccessMode::raw}));
105 reg_tdi.replace(
106 mDevice.getScalarRegisterAccessor<int32_t>(registerPathName, REG_TDI, {ChimeraTK::AccessMode::raw}));
107 reg_tdo.replace(
108 mDevice.getScalarRegisterAccessor<int32_t>(registerPathName, REG_TDO, {ChimeraTK::AccessMode::raw}));
109 reg_rev_switch.replace(
110 mDevice.getScalarRegisterAccessor<int32_t>(registerPathName, REG_REV_SWITCH, {ChimeraTK::AccessMode::raw}));
111 }
112 else if(registerPathName == PROG_DEFAULT_MODULE_NAME4) {
113 reg_area_write.replace(mDevice.getOneDRegisterAccessor<int32_t>("FCM.AREA_WRITE"));
114 reg_area_read.replace(mDevice.getOneDRegisterAccessor<int32_t>("FCM.AREA_READ"));
115 reg_spi_divider.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.WORD_SPI_DIVIDER"));
116 reg_bytes_to_write.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.WORD_BYTES_TO_WRITE"));
117 reg_bytes_to_read.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.WORD_BYTES_TO_READ"));
118 reg_control.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.WORD_CONTROL"));
119 reg_tck.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.WORD_TCK"));
120 reg_tms.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.WORD_TMS"));
121 reg_tdi.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.WORD_TDI"));
122 reg_tdo.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.WORD_TDO"));
123 reg_rev_switch.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.WORD_REV_SWITCH"));
124 }
125 else {
126 assert(registerPathName == PROG_DEFAULT_MODULE_NAME5);
127 reg_area_write.replace(mDevice.getOneDRegisterAccessor<int32_t>("FCM.WRITE_BUF"));
128 reg_area_read.replace(mDevice.getOneDRegisterAccessor<int32_t>("FCM.READ_BUF"));
129 reg_spi_divider.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.SPI_DIVIDER"));
130 reg_bytes_to_write.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.BYTES_TO_WRITE"));
131 reg_bytes_to_read.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.BYTES_TO_READ"));
132 reg_control.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.CONTROL"));
133 reg_tck.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.JTAG_TCK"));
134 reg_tms.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.JTAG_TMS"));
135 reg_tdi.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.JTAG_TDI"));
136 reg_tdo.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.JTAG_TDO"));
137 reg_rev_switch.replace(mDevice.getScalarRegisterAccessor<int32_t>("FCM.REV_SWITCH"));
138 }
