80 UA_StatusCode ua_additionalvariable::mapSelfToNamespace() {
81  UA_StatusCode retval = UA_STATUSCODE_GOOD;
82  UA_NodeId createdNodeId = UA_NODEID_NULL;
83 
84  if(UA_NodeId_equal(&this->baseNodeId, &createdNodeId) == UA_TRUE)
85  return UA_STATUSCODE_BADINVALIDARGUMENT;
86 
87  //Generate additional variable node id
88  UA_String baseNodeIdString = baseNodeId.identifier.string;
89  string baseNodeIdStringCPP;
90  UASTRING_TO_CPPSTRING(baseNodeIdString, baseNodeIdStringCPP);
91  if (!baseNodeIdStringCPP.empty()) {
92  baseNodeIdStringCPP.resize(baseNodeIdStringCPP.size() - 3);
93  }
94 
95  UA_String* opcua_node_variable_t_ns_2_variant_DataContents = UA_String_new();
96  *opcua_node_variable_t_ns_2_variant_DataContents = UA_STRING_ALLOC(this->value.c_str());
97  UA_VariableAttributes vAttr;
98  UA_VariableAttributes_init(&vAttr);
99  vAttr = UA_VariableAttributes_default;
100  vAttr.displayName = UA_LOCALIZEDTEXT_ALLOC((char*)"en_US", (char*)this->name.c_str());
101  vAttr.description = UA_LOCALIZEDTEXT_ALLOC((char*)"en_US", (char*)this->description.c_str());
102  vAttr.dataType = UA_NODEID_NUMERIC(0, UA_NS0ID_STRING);
103  UA_Variant_setScalar(&vAttr.value, opcua_node_variable_t_ns_2_variant_DataContents, &UA_TYPES[UA_TYPES_STRING]);
104 
105  UA_QualifiedName qualName = UA_QUALIFIEDNAME_ALLOC(1, this->name.c_str());
106  retval |= UA_Server_addVariableNode(this->mappedServer,
107  UA_NODEID_STRING(1, (char*)(baseNodeIdStringCPP + "/" + name + "AdditionalVariable").c_str()), this->baseNodeId,
108  UA_NODEID_NUMERIC(0, UA_NS0ID_HASCOMPONENT), qualName,
109  UA_NODEID_NUMERIC(CSA_NSID, UA_NS2ID_CTKADDITIONALVARIABLE), vAttr, (void*)this, &createdNodeId);
110  UA_NodeId_copy(&createdNodeId, &this->ownNodeId);
111  ua_mapInstantiatedNodes(
112  createdNodeId, UA_NODEID_NUMERIC(CSA_NSID, UA_NS2ID_CTKADDITIONALVARIABLE), &this->ownedNodes);
113 
114  UA_QualifiedName_clear(&qualName);
115  UA_VariableAttributes_clear(&vAttr);
116 
117  UA_VariableAttributes vAttr2;
118  UA_VariableAttributes_init(&vAttr2);
119  vAttr2 = UA_VariableAttributes_default;
120  vAttr2.displayName = UA_LOCALIZEDTEXT_ALLOC((char*)"en_US", (char*)"Description");
121  vAttr2.description = UA_LOCALIZEDTEXT_ALLOC((char*)"en_US", (char*)this->description.c_str());
122  vAttr2.dataType = UA_NODEID_NUMERIC(0, UA_NS0ID_STRING);
123  UA_String addVarDescription = UA_STRING_ALLOC(description.c_str());
124  UA_Variant_setScalarCopy(&vAttr2.value, &addVarDescription, &UA_TYPES[UA_TYPES_STRING]);
125 
126  UA_QualifiedName qualName2 = UA_QUALIFIEDNAME_ALLOC(1, this->description.c_str());
127  string parentNodeStringDescription = baseNodeIdStringCPP + "/" + name + "/Description";
128  retval |= UA_Server_addVariableNode(this->mappedServer,
129  UA_NODEID_STRING(1, (char*) parentNodeStringDescription.c_str()), this->ownNodeId,
130  UA_NODEID_NUMERIC(0, UA_NS0ID_HASCOMPONENT), qualName2, UA_NODEID_NUMERIC(0, UA_NS0ID_BASEDATAVARIABLETYPE),
131  vAttr2, NULL, NULL);
132 
133  UA_QualifiedName_clear(&qualName2);
134 
135  /* Use a datasource map to map any local getter/setter functions to opcua variables nodes */
136  UA_DataSource_Map mapDs;
137  UA_DataSource_Map_Element mapElemValue;
138  mapElemValue.typeTemplateId = UA_NODEID_NUMERIC(CSA_NSID, CSA_NSID_ADDITIONAL_VARIABLE_VALUE);
139  mapElemValue.description = vAttr2.description;
140  mapElemValue.read = ua_readproxy_ua_additionalvariable_getValue;
141  mapElemValue.write = NULL;
142  mapDs.push_back(mapElemValue);
143 
144  this->ua_mapDataSources((void*)this, &mapDs);
145 
146  UA_String_clear(&addVarDescription);
147  UA_VariableAttributes_clear(&vAttr2);
148  UA_NodeId_clear(&createdNodeId);
149 
150  return retval;
151 }
