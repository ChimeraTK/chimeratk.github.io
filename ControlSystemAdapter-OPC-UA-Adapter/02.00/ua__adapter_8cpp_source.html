103 void ua_uaadapter::readConfig() {
104  string xpath = "//config";
105  xmlXPathObjectPtr result = this->fileHandler->getNodeSet(xpath);
106  string placeHolder = "";
107  if(result) {
108  xmlNodeSetPtr nodeset = result->nodesetval;
109  if(nodeset->nodeNr > 1) {
110  throw std::runtime_error ("To many <config>-Tags in config file");
111  }
112 
113  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "rootFolder");
114  if(!placeHolder.empty()) {
115  this->serverConfig.rootFolder = placeHolder;
116  }
117 
118  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "description");
119  if(!placeHolder.empty()) {
120  this->serverConfig.descriptionFolder = placeHolder;
121  }
122  vector<xmlNodePtr> mappingExceptionsVector = this->fileHandler->getNodesByName(
123  nodeset->nodeTab[0]->children, "mapping_exceptions");
124  if(mappingExceptionsVector.empty()){
125  this->mappingExceptions = UA_FALSE;
126  } else {
127  placeHolder = this->fileHandler->getContentFromNode(mappingExceptionsVector[0]);
128  transform(placeHolder.begin(), placeHolder.end(), placeHolder.begin(), ::toupper);
129  if(placeHolder.compare("TRUE") == 0){
130  this->mappingExceptions = UA_TRUE;
131  } else {
132  this->mappingExceptions = UA_FALSE;
133  }
134  }
135 
136  xmlXPathFreeObject(result);
137  }
138  result = this->fileHandler->getNodeSet(xpath + "//login");
139  if(result) {
140  xmlNodeSetPtr nodeset = result->nodesetval;
141  if(nodeset->nodeNr > 1) {
142  throw std::runtime_error ("To many <login>-Tags in config file");
143  }
144  this->serverConfig.UsernamePasswordLogin = UA_TRUE;
145  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "password");
146  if(!placeHolder.empty()) {
147  this->serverConfig.password = placeHolder;
148  } else {
149  throw std::runtime_error ("<login>-Tag requires username");
150  }
151 
152  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "username");
153  if(!placeHolder.empty()) {
154  this->serverConfig.username = placeHolder;
155  } else {
156  throw std::runtime_error ("<login>-Tag requires password");
157  }
158 
159  xmlXPathFreeObject(result);
160  }
161  else {
162  this->serverConfig.UsernamePasswordLogin = UA_FALSE;
163  }
164  result = this->fileHandler->getNodeSet(xpath + "//server");
165  if(result) {
166  xmlNodeSetPtr nodeset = result->nodesetval;
167  if(nodeset->nodeNr > 1) {
168  throw std::runtime_error ("To many <server>-Tags in config file");
169  }
170  string opcuaPort = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "port");
171  if(!opcuaPort.empty()) {
172  this->serverConfig.opcuaPort = std::stoi(opcuaPort);
173  } else {
174  cout << "No 'port'-Attribute in config file is set. Use default Port: 16664" << endl;
175  }
176 
177  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "applicationName");
178  if(!placeHolder.empty()) {
179  this->serverConfig.applicationName = placeHolder;
180  } else {
181  string applicationName;
182  try{
183  string applicationName = ApplicationBase::getInstance().getName();
184  this->serverConfig.applicationName = applicationName.c_str();
185  } catch (ChimeraTK::logic_error){
186 
187  }
188  cout << "No 'applicationName'-Attribute is set in config file. Use default application-name." << endl;
189  }
190  //if no root folder name is set, use application name
191  if (this->serverConfig.rootFolder.empty()) {
192  this->serverConfig.rootFolder = this->serverConfig.applicationName;
193  }
194  xmlXPathFreeObject(result);
195 }
196  else {
197  cout << "No <server>-Tag in config file. Use default port 16664 and application name configuration." << endl;
198  this->serverConfig.rootFolder = this->serverConfig.applicationName;
199  }
200  result = this->fileHandler->getNodeSet(xpath+"//process_variable_hierarchy");
201  if(result){
202  xmlNodeSetPtr nodeset = result->nodesetval;
203  vector<xmlNodePtr> nodeVectorUnrollPathPV = this->fileHandler->getNodesByName(
204  nodeset->nodeTab[0]->children, "unroll");
205  for (auto nodeUnrollPath: nodeVectorUnrollPathPV) {
206  string unrollSepEnabled = this->fileHandler->getContentFromNode(nodeUnrollPath);
207  transform(unrollSepEnabled.begin(), unrollSepEnabled.end(), unrollSepEnabled.begin(), ::toupper);
208  if(unrollSepEnabled.compare("TRUE") == 0){
209  this->pvSeperator += this->fileHandler->getAttributeValueFromNode(nodeUnrollPath, "pathSep");
210  }
211  }
212 
213  xmlXPathFreeObject(result);
214 } else {
215  cout << "No <process_variable_hierarchy>-Tag in config file. Use default hierarchical mapping with '/'."
216  << endl;
217  this->pvSeperator = "/";
218  }

333 void ua_uaadapter::deepCopyHierarchicalLayer(boost::shared_ptr<ControlSystemPVManager> csManager, UA_NodeId layer, UA_NodeId target){
334  //copy pv's of current layer
335  UA_BrowseDescription bd;
336  bd.includeSubtypes = false;
337  bd.nodeId = layer;
338  bd.referenceTypeId = UA_NODEID_NUMERIC(0, UA_NS0ID_HASCOMPONENT);
339  bd.resultMask = UA_BROWSERESULTMASK_NONE;
340  bd.nodeClassMask = UA_NODECLASS_VARIABLE;
341  bd.browseDirection = UA_BROWSEDIRECTION_FORWARD;
342  UA_BrowseResult br = UA_Server_browse(this->mappedServer, UA_UINT32_MAX, &bd);
343  for (size_t j = 0; j < br.referencesSize; ++j) {
344  UA_ReferenceDescription rd = br.references[j];
345  //get unit, desc, ..
346  UA_LocalizedText foundPVName;
347  string foundPVNameCPP;
348  UA_Server_readDisplayName(this->mappedServer, rd.nodeId.nodeId, &foundPVName);
349  UASTRING_TO_CPPSTRING(foundPVName.text, foundPVNameCPP);
350 
351  string pvSourceNameid;
352  UA_String foundPVSourceName;
353  string foundPVSourceNameCPP;
354  UA_Variant value;
355  UA_STRING_TO_CPPSTRING_COPY(&rd.nodeId.nodeId.identifier.string, &pvSourceNameid);
356  pvSourceNameid = pvSourceNameid.substr(0, pvSourceNameid.length()-5);
357  UA_Server_readValue(this->mappedServer, UA_NODEID_STRING(1, (char *) (pvSourceNameid+"/name").c_str()), &value);
358  foundPVSourceName = *((UA_String *) value.data);
359  UASTRING_TO_CPPSTRING(foundPVSourceName, foundPVSourceNameCPP);
360  string varName = this->fileHandler->praseVariablePath(foundPVNameCPP, "/").back();
361  ua_processvariable *processvariable = new ua_processvariable(this->mappedServer, target, foundPVSourceNameCPP, csManager, foundPVNameCPP);
362  this->variables.push_back(processvariable);
363  }
364  UA_BrowseResult_deleteMembers(&br);
365  //copy folders of current layer
366  bd.includeSubtypes = false;
367  bd.nodeId = layer;
368  bd.referenceTypeId = UA_NODEID_NUMERIC(0, UA_NS0ID_ORGANIZES);
369  bd.resultMask = UA_BROWSERESULTMASK_NONE;
370  bd.nodeClassMask = UA_NODECLASS_OBJECT;
371  bd.browseDirection = UA_BROWSEDIRECTION_FORWARD;
372  br = UA_Server_browse(this->mappedServer, UA_UINT32_MAX, &bd);
373  for (size_t j = 0; j < br.referencesSize; ++j) {
374  UA_ReferenceDescription rd = br.references[j];
375  UA_String folderName, description;
376  UA_LocalizedText foundFolderName, foundFolderDescription;
377  UA_LocalizedText_init(&foundFolderDescription);
378  string foundFolderNameCPP;
379  UA_Server_readDescription(this->mappedServer, rd.nodeId.nodeId, &foundFolderDescription);
380  UA_StatusCode result = UA_Server_readDisplayName(this->mappedServer, rd.nodeId.nodeId, &foundFolderName);
381  UASTRING_TO_CPPSTRING(foundFolderName.text, foundFolderNameCPP);
382  if(result != UA_STATUSCODE_GOOD){
383  //error handling
384  continue;
385  }
386  UA_NodeId newRootFolder = createFolder(target, foundFolderNameCPP);
387  if(foundFolderDescription.text.length > 0)
388  UA_Server_writeDescription(this->mappedServer, newRootFolder, foundFolderDescription);
389  deepCopyHierarchicalLayer(csManager, rd.nodeId.nodeId, newRootFolder);
390  }
391  UA_BrowseResult_deleteMembers(&br);
392 }

394 void ua_uaadapter::buildFolderStructure(boost::shared_ptr<ControlSystemPVManager> csManager){
395  xmlXPathObjectPtr result = this->fileHandler->getNodeSet("//folder");
396  xmlNodeSetPtr nodeset;
397  if (result) {
398  nodeset = result->nodesetval;
399  for (int32_t i=0; i < nodeset->nodeNr; i++) {
400  UA_NodeId folderPathNodeId;
401  string destination, description, folder;
402  vector<xmlNodePtr> nodeFolderPath = this->fileHandler->getNodesByName(
403  nodeset->nodeTab[i]->children, "destination");
404  vector<xmlNodePtr> nodeFolder = this->fileHandler->getNodesByName(
405  nodeset->nodeTab[i]->children, "name");
406  vector<xmlNodePtr> nodeDescription = this->fileHandler->getNodesByName(
407  nodeset->nodeTab[i]->children, "description");
408  string sourceName = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[i], "sourceName");
409  string copy = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[i], "copy");
410  if(!nodeFolderPath.empty()){
411  destination = this->fileHandler->getContentFromNode(nodeFolderPath[0]);
412  }
413  if(!nodeDescription.empty()){
414  description = this->fileHandler->getContentFromNode(nodeDescription[0]);
415  }
416  if(!nodeFolder.empty()) {
417  folder = this->fileHandler->getContentFromNode(nodeFolder[0]);
418  }
419  if(folder.empty()){
420  if(this->mappingExceptions){
421  throw std::runtime_error ("Error! Folder creation failed. Name is missing. Mapping line number: " + to_string(nodeset->nodeTab[i]->line));
422  }
423  if(description.empty()){
424  UA_LOG_WARNING(this->server_config.logger, UA_LOGCATEGORY_USERLAND, "Warning! Skipping Folder. Folder creation failed. Name is missing. Mapping line number: %u", nodeset->nodeTab[i]->line);
425  } else {
426  UA_LOG_WARNING(this->server_config.logger, UA_LOGCATEGORY_USERLAND, "Warning! Skipping Folder. Folder creation failed. Name is missing 'description: %s'. Mapping line number: %u", description.c_str(), nodeset->nodeTab[i]->line);
427  }
428  continue;
429  }
430  //check a pv with the requested node id exists
431  UA_NodeId tmpOutput = UA_NODEID_NULL;
432  string pvNodeString;
433  if(destination.empty()){
434  pvNodeString = this->serverConfig.rootFolder + "/" + folder;
435  } else {
436  pvNodeString = this->serverConfig.rootFolder + "/" + destination + "/" + folder;
437  }
438  UA_NodeId pvNode = UA_NODEID_STRING(1, (char *) pvNodeString.c_str());
439  UA_Server_readNodeId(this->mappedServer, pvNode, &tmpOutput);
440  if(!UA_NodeId_isNull(&tmpOutput)){
441  UA_NodeId_deleteMembers(&tmpOutput);
442  //set folder description
443  if(sourceName.empty() && !description.empty()){
444  UA_Server_writeDescription(this->mappedServer, pvNode, UA_LOCALIZEDTEXT((char *) "en_US", (char *) description.c_str()));
445  continue;
446  }
447  if(this->mappingExceptions){
448  throw std::runtime_error ("Error! Folder with source mapping failed. Target folder node id exists. Mapping line number: " + to_string(nodeset->nodeTab[i]->line));
449  }
450  UA_LOG_WARNING(this->server_config.logger, UA_LOGCATEGORY_USERLAND, "Warning! Skipping folder: %s. Folder with source mapping failed. Target folder node id exists. Mapping line number: %u", sourceName.c_str(), nodeset->nodeTab[i]->line);
451  continue;
452  }
453  if(!copy.empty() && sourceName.empty()){
454  if(this->mappingExceptions){
455  throw std::runtime_error ("Error! Folder creation failed. Source folder missing. Mapping line number: " + to_string(nodeset->nodeTab[i]->line));
456  }
457  UA_LOG_WARNING(this->server_config.logger, UA_LOGCATEGORY_USERLAND, "Warning! Skipping Folder. Source 'name: %s' folder missing. Mapping line number: %u", folder.c_str(), nodeset->nodeTab[i]->line);
458  continue;
459  }
460  folderPathNodeId = enrollFolderPathFromString(destination+"/replacePart", "/");
461  if(UA_NodeId_isNull(&folderPathNodeId)){
462  folderPathNodeId = UA_NODEID_STRING(1, (char *) this->serverConfig.rootFolder.c_str());
463  }
464  //check if source name is set -> map complete hierarchical structure to the destination
465  if(!sourceName.empty()){
466  if((destination.empty() && sourceName.compare(folder) == 0) ||
467  (!destination.empty() && sourceName.compare(destination+"/"+folder) == 0)){
468  if(this->mappingExceptions){
469  throw std::runtime_error ("Error! Folder creation failed. Source and Destination equal. Mapping line number: " + to_string(nodeset->nodeTab[i]->line));
470  }
471  UA_LOG_WARNING(this->server_config.logger, UA_LOGCATEGORY_USERLAND, "Warning! Skipping Folder. Source '%s' and Destination equal. Mapping line number: %u", sourceName.c_str(), nodeset->nodeTab[i]->line);
472  continue;
473  }
474  //check if the src is a folder
475  string sourceFolder = this->serverConfig.rootFolder+"/"+sourceName;
476  bool isFolderType = false;
477  UA_NodeId sourceFolderId = UA_NODEID_STRING(1, (char *) sourceFolder.c_str());
478 
479  UA_BrowseDescription bd;
480  bd.includeSubtypes = false;
481  bd.nodeId = sourceFolderId;
482  bd.referenceTypeId = UA_NODEID_NUMERIC(0, UA_NS0ID_HASTYPEDEFINITION);
483  bd.resultMask = UA_BROWSERESULTMASK_ALL;
484  bd.nodeClassMask = UA_NODECLASS_OBJECTTYPE;
485  bd.browseDirection = UA_BROWSEDIRECTION_BOTH;
486  UA_BrowseResult br = UA_Server_browse(this->mappedServer, 1000, &bd);
487  for (size_t j = 0; j < br.referencesSize; ++j) {
488  UA_ReferenceDescription rd = br.references[j];
489  UA_NodeId folderType = UA_NODEID_NUMERIC(0, UA_NS0ID_FOLDERTYPE);
490  if(UA_NodeId_equal(&rd.nodeId.nodeId, &folderType)){
491  isFolderType = true;
492  }
493  }
494  UA_BrowseResult_deleteMembers(&br);
495  if(!isFolderType){
496  if(this->mappingExceptions){
497  throw std::runtime_error ("Error! Folder creation failed. No corresponding source folder: " + sourceName + ". Mapping line number: " + to_string(nodeset->nodeTab[i]->line));
498  }
499  UA_LOG_WARNING(this->server_config.logger, UA_LOGCATEGORY_USERLAND, "Warning! Skipping Folder. No corresponding source '%s' folder. Mapping line number: %u", sourceName.c_str(), nodeset->nodeTab[i]->line);
500  continue;
