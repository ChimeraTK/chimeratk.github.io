52 UA_StatusCode ua_callProxy_mapDataSources(UA_Server* server, nodePairList instantiatedNodesList, UA_DataSource_Map *map, void *srcClass)
53 {
54  UA_StatusCode retval = UA_STATUSCODE_GOOD;
55  if (map == nullptr || server == nullptr)
56  return retval;
57 
58  // Functions are not instantiated... they are just linked to the node.
59  for (nodePairList::iterator l = instantiatedNodesList.begin(); l != instantiatedNodesList.end(); ++l) {
60  UA_NodeId typeTemplateId = (*l)->sourceNodeId;
61  UA_NodeId instantiatedId = (*l)->targetNodeId;
62 
63  // Check if we have this node in our map
64  UA_DataSource_Map_Element *ele = nullptr;
65  // cppcheck-suppress postfixOperator REASON: List iterator cannot be prefixed
66  for (UA_DataSource_Map::iterator j=map->begin(); j != map->end(); j++) {
67  UA_DataSource_Map_Element thisEle = *(j);
68  if(UA_NodeId_equal(&typeTemplateId, (const UA_NodeId *) &thisEle.typeTemplateId) == UA_TRUE) {
69  ele = new UA_DataSource_Map_Element;
70  *ele = thisEle;
71  break;
72  }
73  }
74  if (ele == nullptr)
75  continue;
76 
77  // Found a mapping, now register the dataSource
78  UA_DataSource ds;
79  ds.read = ele->read;
80  ds.write = ele->write;
81 
82  // Set accesslevel depending on callback functions
83  UA_Byte accessLevel;
84  if(ele->write == NULL && ele->read == NULL) {
85  accessLevel = 0;
86  }
87  if(ele->write != NULL && ele->read == NULL) {
88  accessLevel = UA_ACCESSLEVELMASK_WRITE;
89  }
90  if(ele->write != NULL && ele->read != NULL) {
91  accessLevel = UA_ACCESSLEVELMASK_WRITE^UA_ACCESSLEVELMASK_READ;
92  }
93  if(ele->write == NULL && ele->read != NULL) {
94  accessLevel = UA_ACCESSLEVELMASK_READ;
95  }
96 
97  UA_Server_writeAccessLevel(server, instantiatedId, accessLevel);
98  // There is currently no high- level function to do this. (02.12.2016)
99  __UA_Server_write(server, &instantiatedId, UA_ATTRIBUTEID_USERACCESSLEVEL, &UA_TYPES[UA_TYPES_BYTE], &accessLevel);
100 
101  ds.handle = srcClass;
102 
103  UA_Server_setVariableNode_dataSource(server, instantiatedId, ds);
104 
105  UA_Server_writeDescription(server, instantiatedId, ele->description);
106  delete ele; // inhibit memleak warning during static analysis
107 
108  /* Set the right Value Datatype and ValueRank
109  * -> This is a quickfix for subjective data handling by open62541
110  * (02.12.2016)
111  */
112  UA_NodeId datatTypeNodeId = UA_NODEID_NULL;
113  retval = UA_Server_readDataType(server, instantiatedId, &datatTypeNodeId);
114  if(retval == UA_STATUSCODE_GOOD) {
115  const UA_NodeId basedatatype = UA_NODEID_NUMERIC(0, UA_NS0ID_BASEDATATYPE);
116  UA_Variant variantVal;
117  UA_Variant_init(&variantVal);
118  retval = UA_Server_readValue(server, instantiatedId, &variantVal);
119  if(UA_NodeId_equal(&datatTypeNodeId, &basedatatype) && retval == UA_STATUSCODE_GOOD) {
120 
121  // See IEC 62541-3: OPC Unified Architecture - Part 3: Address space model -> Page 75
122  UA_Int32 valueRank = -2;
123  if(variantVal.arrayDimensionsSize == 0 && UA_Variant_isScalar(&variantVal)) {
124  // Scalar
125  valueRank = -1;
126  }
127  else {
128  if(variantVal.arrayDimensionsSize > 1) {
129  // OneOrMoreDimensions
130  valueRank = 0;
131  }
132  else {
133  if(variantVal.arrayDimensionsSize == 1 || variantVal.arrayLength) {
134  // OneDimension
135  valueRank = 1;
136  }
137  }
138  }
139  UA_Server_writeValueRank(server, instantiatedId, valueRank);
140  }
141  // the right Datatype
142  retval = UA_Server_writeDataType(server, instantiatedId, variantVal.type->typeId);
143 
144  }
145  }
146 
147  return retval;
148 }
