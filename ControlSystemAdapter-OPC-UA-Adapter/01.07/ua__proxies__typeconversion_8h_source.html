25 #define UA_STRING_TO_CPPSTRING_COPY(_p_uastring, _p_cppstring) do { \
26 UA_String *ua_url = (UA_String *) _p_uastring; \
27 char *ua_str_tmp = (char *) malloc(ua_url->length+1); \
28 memset(ua_str_tmp, 0, ua_url->length+1); \
29 strncpy(ua_str_tmp, (char *) ua_url->data, ua_url->length); \
30 *(_p_cppstring) = ua_str_tmp; \
31 free(ua_str_tmp); \
32 } while(0); \

56  *
57  * Copy contents of s_cpp into s_ua
58  */
59 #define CPPSTRING_TO_UASTRING(_p_uastring, _p_cppstring) {\
60 char *s = (char *) malloc(_p_cppstring.length()+1); \
61 strncpy(s, (char*) _p_cppstring.c_str(), _p_cppstring.length()); \
62 _p_uastring.length = _p_cppstring.length(); \
63 _p_uastring.data = (UA_Byte *) malloc(_p_uastring.length); \
64 memcpy(_p_uastring.data, s, _p_uastring.length); \
65 free(s); \
66 }
