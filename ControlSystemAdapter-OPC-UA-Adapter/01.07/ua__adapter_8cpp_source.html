104 void ua_uaadapter::readConfig() {
105 
106  string xpath = "//config";
107  xmlXPathObjectPtr result = this->fileHandler->getNodeSet(xpath);
108  string placeHolder = "";
109  if(result) {
110  xmlNodeSetPtr nodeset = result->nodesetval;
111  // There should be only one <config>-Tag in config file
112  if(nodeset->nodeNr > 1) {
113  //cout << "More than one <config>-Tag was found in config file. Please provide only one Tag." << endl;
114  throw std::runtime_error ("To many <config>-Tags in config file");
115  }
116 
117  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "rootFolder");
118  if(!placeHolder.empty()) {
119  this->serverConfig.rootFolder = placeHolder;
120  }
121 
122  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "description");
123  if(!placeHolder.empty()) {
124  this->serverConfig.descriptionFolder = placeHolder;
125  }
126  }
127 
128  result = this->fileHandler->getNodeSet(xpath + "//login");
129  if(result) {
130  xmlNodeSetPtr nodeset = result->nodesetval;
131  this->serverConfig.UsernamePasswordLogin = UA_TRUE;
132  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "password");
133  if(!placeHolder.empty()) {
134  this->serverConfig.password = placeHolder;
135  }
136 
137  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "username");
138  if(!placeHolder.empty()) {
139  this->serverConfig.username = placeHolder;
140  }
141  }
142  else {
143  this->serverConfig.UsernamePasswordLogin = UA_FALSE;
144  }
145 
146  result = this->fileHandler->getNodeSet(xpath +"//serverConfig");
147  if(result) {
148  xmlNodeSetPtr nodeset = result->nodesetval;
149  string opcuaPort = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "port");
150  if(opcuaPort.compare("") != 0) {
151  this->serverConfig.opcuaPort = std::stoi(opcuaPort);
152  }
153  else {
154  cout << "No 'port'-Attribute in config file is set. Use default Port: 16664" << endl;
155  }
156 
157  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "applicationName");
158  if(placeHolder.compare("") != 0) {
159  this->serverConfig.applicationName = placeHolder;
160  }
161  else {
162  cout << "No 'applicationName'-Attribute is set in config file. Use default Applicationname." << endl;
163  }
164  }
165  else {
166  cout << "No <serverConfig>-Tag in config file. Use default port 16664 and application name configuration." << endl;
167  }

215 void ua_uaadapter::addVariable(std::string varName, boost::shared_ptr<ControlSystemPVManager> csManager) {
216 
217  ua_processvariable *processvariable = new ua_processvariable(this->mappedServer, this->variablesListId, varName, csManager);
218  this->variables.push_back(processvariable);
219 
220  xmlXPathObjectPtr result = this->fileHandler->getNodeSet("//map");
221  xmlNodeSetPtr nodeset;
222  string srcVarName = "";
223  string applicName = "";
224 
225  // TODO. What happen if application name are not unique?
226  if(result) {
227  nodeset = result->nodesetval;
228  for (int32_t i=0; i < nodeset->nodeNr; i++) {
229  srcVarName = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[i], "sourceVariableName");
230  //keyword = xmlNodeListGetString(this->fileHandler->getDoc(), nodeset->nodeTab[i]->xmlChildrenNode, 1);
231  if(varName.compare(srcVarName) == 0) {
232  // get name attribute from <application>-tag
233  applicName = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[i]->parent, "name");
234  // Check if "rename" is not empty
235  string renameVar = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[i], "rename");
236  string engineeringUnit = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[i], "engineeringUnit");
237  string description = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[i], "description");
238 
239  // Application Name have to be unique!!!
240  UA_NodeId appliFolderNodeId = this->existFolder(this->ownNodeId, applicName);
241  FolderInfo newFolder;
242  if(UA_NodeId_isNull(&appliFolderNodeId)) {
243  newFolder.folderName = applicName;
244  newFolder.folderNodeId = this->createFolder(this->ownNodeId, applicName);
245  this->folderVector.push_back(newFolder);
246  appliFolderNodeId = newFolder.folderNodeId;
247  }
248 
249  vector<string> varPathVector;
250  vector<xmlNodePtr> nodeVectorUnrollPath = this->fileHandler->getNodesByName(nodeset->nodeTab[i]->children, "unrollPath");
251  string seperator = "";
252  bool unrollPathIs = false;
253  for(auto nodeUnrollPath: nodeVectorUnrollPath) {
254  string shouldUnrollPath = this->fileHandler->getContentFromNode(nodeUnrollPath);
255  if(shouldUnrollPath.compare("True") == 0) {
256  seperator = seperator + this->fileHandler->getAttributeValueFromNode(nodeUnrollPath, "pathSep");
257  unrollPathIs = true;
258  }
259  }
260 
261  if(!seperator.empty()) {
262  vector<string> newPathVector = this->fileHandler->praseVariablePath(srcVarName, seperator);
263  varPathVector.insert(varPathVector.end(), newPathVector.begin(), newPathVector.end());
264  }
265 
266  // assumption last element is name of variable, hence no folder for name is needed
267  if(renameVar.compare("") == 0 && !unrollPathIs) {
268  renameVar = srcVarName;
269  // std::cout << "Variable '" << srcVarName << "' renamed in '" << renameVar << "' and listed in folder '" << applicName << "'." << std::endl;
270  }
271  else {
272  if(unrollPathIs && renameVar.compare("") == 0) {
273  renameVar = varPathVector.at(varPathVector.size()-1);
274  varPathVector.pop_back();
275  }
276  else {
277  if(varPathVector.size() > 0) {
278  varPathVector.pop_back();
279  }
280  }
281  // std::cout << "Variable '" << srcVarName << "' listed in folder '" << applicName << "'." << std::endl;
282  }
283 
284 
285 
286  vector<xmlNodePtr> nodeVectorFolderPath = this->fileHandler->getNodesByName(nodeset->nodeTab[i]->children, "folder");
287  vector<string> folderPathVector;
288  bool createdVar = false;
289  UA_NodeId newFolderNodeId = UA_NODEID_NULL;
290  vector<UA_NodeId> mappedVariables;
291  for(auto nodeFolderPath: nodeVectorFolderPath) {
292 
293  string folderPath = this->fileHandler->getContentFromNode(nodeFolderPath);
294  if(folderPath.empty() && unrollPathIs) {
295  break;
296  }
297 
298  folderPathVector = this->fileHandler->praseVariablePath(folderPath);
299  // Create folders
300  newFolderNodeId = appliFolderNodeId;
301  if(folderPathVector.size() > 0) {
302  newFolderNodeId = this->createFolderPath(newFolderNodeId, folderPathVector);
303  }
304 
305  if(varPathVector.size() > 0) {
306  newFolderNodeId = this->createFolderPath(newFolderNodeId, varPathVector);
307  }
308  mappedVariables.push_back(newFolderNodeId);
309  createdVar = true;
310  }
311 
312  // in case no <folder> or <unrollpath> is set
313  if(!createdVar) {
314  newFolderNodeId = appliFolderNodeId;
315 
316  if(varPathVector.size() > 0) {
317  mappedVariables.push_back(this->createFolderPath(newFolderNodeId, varPathVector));
318  }
319  else {
320  // No <folder>
321  mappedVariables.push_back(appliFolderNodeId);
322  }
323  }
324 
325  // Create all nessesary mapped ObjectVaraibles with inner variables (reference or attribute, depending attributes are set (engineeringUnit, dexcription)
326  for(auto objectNodeId:mappedVariables) {
327  UA_NodeId createdNodeId = UA_NODEID_NULL;
328 
329  // Create our new "Value" Variable
330  UA_ObjectAttributes oAttr;
331  UA_ObjectAttributes_init(&oAttr);
332  oAttr.displayName = UA_LOCALIZEDTEXT_ALLOC("en_US", renameVar.c_str());
333  oAttr.description = UA_LOCALIZEDTEXT_ALLOC("en_US", description.c_str());
334 
335  UA_INSTATIATIONCALLBACK(icb);
336  UA_Server_addObjectNode(this->mappedServer, UA_NODEID_NUMERIC(1, 0),
337  objectNodeId, UA_NODEID_NUMERIC(0, UA_NS0ID_HASCOMPONENT),
338  UA_QUALIFIEDNAME_ALLOC(1, renameVar.c_str()), UA_NODEID_NULL, oAttr, &icb, &createdNodeId);
339 
340  UA_ExpandedNodeId *targetNodeId = UA_ExpandedNodeId_new();
341  targetNodeId->nodeId = createdNodeId;
342 
343  UA_BrowseDescription bDesc;
344  UA_BrowseDescription_init(&bDesc);
345  bDesc.browseDirection = UA_BROWSEDIRECTION_FORWARD;
346  bDesc.includeSubtypes = false;
347  bDesc.nodeClassMask = UA_NODECLASS_VARIABLE;
348  bDesc.nodeId = processvariable->getOwnNodeId();
349  bDesc.resultMask = UA_BROWSERESULTMASK_ALL;
350 
351  UA_BrowseResult bRes;
352  UA_BrowseResult_init(&bRes);
353 
354  UA_VariableAttributes vAttr;
355  UA_VariableAttributes_init(&vAttr);
356  vAttr.dataType = UA_NODEID_NUMERIC(0, UA_NS0ID_STRING);
357  vAttr.accessLevel = UA_ACCESSLEVELMASK_WRITE^UA_ACCESSLEVELMASK_READ;
358  vAttr.userAccessLevel = UA_ACCESSLEVELMASK_WRITE^UA_ACCESSLEVELMASK_READ;
359  vAttr.valueRank = -1;
360 
361 
362  bRes = UA_Server_browse(this->mappedServer, 10, &bDesc);
363 
364  for(uint32_t i=0; i < bRes.referencesSize; i++) {
365  UA_NodeId newNodeId = UA_NODEID_NULL;
366 
367  UA_String varName = UA_String_fromChars("EngineeringUnit");
368  if(UA_String_equal(&bRes.references[i].browseName.name, &varName) && !engineeringUnit.empty()) {
369  vAttr.description = UA_LOCALIZEDTEXT((char*)"en_US",(char*) "EngineeringUnit");
370  vAttr.displayName = UA_LOCALIZEDTEXT((char*)"en_US",(char*) "EngineeringUnit");
371 
372  UA_String engineringUnit = UA_String_fromChars(engineeringUnit.c_str());
373  UA_Variant_setScalar(&vAttr.value, &engineringUnit, &UA_TYPES[UA_TYPES_STRING]);
374  UA_Server_addVariableNode(this->mappedServer, UA_NODEID_NUMERIC(1, 0), createdNodeId,
375  UA_NODEID_NUMERIC(0, UA_NS0ID_HASCOMPONENT), UA_QUALIFIEDNAME(1, (char*) "EngineeringUnit"),
376  UA_NODEID_NUMERIC(0, UA_NS0ID_BASEDATAVARIABLETYPE), vAttr, &icb, &newNodeId);
377  }
378 
379  varName = UA_String_fromChars("Description");
380  if(UA_String_equal(&bRes.references[i].browseName.name, &varName) && !description.empty()) {
381  vAttr.description = UA_LOCALIZEDTEXT((char*)"en_US",(char*) "Description");
382  vAttr.displayName = UA_LOCALIZEDTEXT((char*)"en_US",(char*) "Description");
383 
384  UA_String engineringUnit = UA_String_fromChars(description.c_str());
385  UA_Variant_setScalar(&vAttr.value, &engineringUnit, &UA_TYPES[UA_TYPES_STRING]);
386  UA_Server_addVariableNode(this->mappedServer, UA_NODEID_NUMERIC(1, 0), createdNodeId,
387  UA_NODEID_NUMERIC(0, UA_NS0ID_HASCOMPONENT), UA_QUALIFIEDNAME(1, (char*) "Description"),
388  UA_NODEID_NUMERIC(0, UA_NS0ID_BASEDATAVARIABLETYPE), vAttr, &icb, &newNodeId);
389  }
390 
391  if(UA_NodeId_isNull(&newNodeId)) {
392  UA_Server_addReference(this->mappedServer, bRes.references[i].nodeId.nodeId, UA_NODEID_NUMERIC(0, UA_NS0ID_HASCOMPONENT), *targetNodeId, false);
393  }
394  }
395 
396  UA_BrowseDescription_deleteMembers(&bDesc);
397  UA_BrowseResult_deleteMembers(&bRes);
398 
399  }
400  }
401  }
402  }
403 
404  xmlXPathFreeObject (result);
405 }

487 UA_NodeId ua_uaadapter::createFolderPath(UA_NodeId basenodeid, std::vector<string> folderPath) {
488 
489  if(UA_NodeId_isNull(&basenodeid)){
490  return UA_NODEID_NULL;
491  }
492 
493  // Check if path exist
494  UA_NodeId toCheckNodeId = existFolderPath(basenodeid, folderPath);
495  int32_t starter4Folder = 0;
496  UA_NodeId nextNodeId = basenodeid;
497  UA_NodeId startNodeId = basenodeid;
498  if(UA_NodeId_isNull(&toCheckNodeId)) {
499  bool setted = false;
500  // Check if path exist partly
501  for(uint32_t m=0; m < folderPath.size(); m++) {
502  for(uint32_t i=0; i < this->folderVector.size(); i++) {
503  // get correct folder NodeId from first folderPath element
504  if(!setted && (folderPath.at(m).compare(this->folderVector.at(i).folderName) == 0) && (UA_NodeId_equal(&this->folderVector.at(i).prevFolderNodeId, &nextNodeId)) && ((m+1) < folderPath.size())) {
505  // remember on witch position the folder still exist
506  setted = true;
507  starter4Folder = m+1;
508  nextNodeId = this->folderVector.at(i).folderNodeId;
509  }
510  if(setted) {
511  break;
512  }
513  }
514  setted = false;
515  }
516  }
517  else {
518  // Path exist nothing to do
519  return toCheckNodeId;
520  }
521
