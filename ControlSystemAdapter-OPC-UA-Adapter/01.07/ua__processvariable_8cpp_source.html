92 UA_RDPROXY_STRING(ua_processvariable, getType)
93 string ua_processvariable::getType() {
94  // Note: typeid().name() may return the name; may as well return the symbol's name from the binary though...
95  std::type_info const & valueType = this->csManager->getProcessVariable(this->namePV)->getValueType();
96  if (valueType == typeid(int8_t)) return "int8_t";
97  else if (valueType == typeid(uint8_t)) return "uint8_t";
98  else if (valueType == typeid(int16_t)) return "int16_t";
99  else if (valueType == typeid(uint16_t)) return "uint16_t";
100  else if (valueType == typeid(int32_t)) return "int32_t";
101  else if (valueType == typeid(uint32_t)) return "uint32_t";
102  else if (valueType == typeid(float)) return "float";
103  else if (valueType == typeid(double)) return "double";
104  else if (valueType == typeid(string)) return "string";
105  else return "Unsupported type";
106 }
