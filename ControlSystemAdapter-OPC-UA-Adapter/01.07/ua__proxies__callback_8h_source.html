31 /* Function call proxy generator
32  * UA_CALLPROXY: Generate callback for the ua_stack. Calls CLASS_F in CLASS_P (this->) is implied.
33  * UA_CALLPROXY_NAME Return the name of the call proxy
34  */
35 // Generate Call-Through functions as stack callbacks
36 #define UA_CALLPROXY_NAME(_CLASS_P, _CLASS_F) ua_callproxy_##_CLASS_P##_##_CLASS_F
37 //#define UA_CALLPROXY_TABLENAME(_CLASS_P, _CLASS_F) C_MACRO_CONCAT( UA_CALLPROXY_NAME(_CLASS_P,_CLASS_F) , _InstanceLookUpTable)
38 #define UA_CALLPROXY_TABLE(_CLASS_P, _CLASS_F) & UA_CALLPROXY_TABLENAME(_CLASS_P, _CLASS_F)
39 
40 #define UA_CALLPROXY(_CLASS_P, _CLASS_F) \
41 UA_FunctionCall_InstanceLookUpTable UA_CALLPROXY_TABLENAME(_CLASS_P, _CLASS_F); \
42 UA_StatusCode UA_CALLPROXY_NAME(_CLASS_P,_CLASS_F)(void *methodHandle, const UA_NodeId objectId, size_t inputSize, const UA_Variant *input, size_t outputSize, UA_Variant *output){ \
43  for (UA_FunctionCall_InstanceLookUpTable::iterator j=UA_CALLPROXY_TABLENAME(_CLASS_P, _CLASS_F).begin(); j != UA_CALLPROXY_TABLENAME(_CLASS_P, _CLASS_F).end(); j++) \
44  if(UA_NodeId_equal(&(*(j))->classObjectId, &objectId)) { \
45  _CLASS_P *theClass = static_cast<_CLASS_P *>( (*(j))->classInstance ); \
46  return theClass->_CLASS_F(inputSize, input, outputSize, output); \
47  } \
48  return UA_STATUSCODE_GOOD; \
49 }
