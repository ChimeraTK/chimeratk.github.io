26 /* Helpers for type converstion and such */
27 #define UA_STRING_TO_CPPSTRING_COPY(_p_uastring, _p_cppstring) \
28  do { \
29  UA_String* ua_url = (UA_String*)_p_uastring; \
30  char* ua_str_tmp = (char*)malloc(ua_url->length + 1); \
31  memset(ua_str_tmp, 0, ua_url->length + 1); \
32  strncpy(ua_str_tmp, (char*)ua_url->data, ua_url->length); \
33  *(_p_cppstring) = ua_str_tmp; \
34  free(ua_str_tmp); \
35  } while(0);

48 /* UASTRING_TO_CPPSTRING(UA_String s_ua, std::string s_cpp)
49  *
50  * Copy contents of s_ua into s_cpp
51  */
52 #define UASTRING_TO_CPPSTRING(_p_uastring, _p_cppstring) \
53  do { \
54  char* s; \
55  UASTRING_AS_CSTRING(_p_uastring, s) \
56  _p_cppstring.assign(s, _p_uastring.length); \
57  free(s); \
58  } while(0);

60 /* CPPSTRING_TO_UASTRING(UA_String s_ua, std::string s_cpp)
61  *
62  * Copy contents of s_cpp into s_ua
63  */
64 static inline UA_String CPPSTRING_TO_UASTRING(std::string& s) {
65  char* buf = (char*)malloc(s.length());
66  if(!buf) return UA_STRING_NULL;
67  strncpy(buf, (char*)s.c_str(), s.length());
68  return {s.length(), (UA_Byte*)buf};
69 }

71 #define CPPSTRING_TO_UASTRING_NEW(_p_uastring, _p_cppstring) \
72  { \
73  char* s = (char*)malloc(_p_cppstring->length() + 1); \
74  strncpy(s, (char*)_p_cppstring->c_str(), _p_cppstring->length()); \
75  _p_uastring.length = _p_cppstring->length(); \
76  _p_uastring.data = (UA_Byte*)malloc(_p_uastring.length); \
77  memcpy(_p_uastring.data, s, _p_uastring.length); \
78  free(s); \
79  }
