57 void csa_opcua_adapter::csa_opcua_adapter_InitVarMapping() {
58  // Get all ProcessVariables
59  vector<ProcessVariable::SharedPtr> allProcessVariables = this->csManager->getAllProcessVariables();
60 
61  for(ProcessVariable::SharedPtr oneProcessVariable : allProcessVariables) {
62  adapter->addVariable(oneProcessVariable->getName(), this->csManager);
63  }
64 
65  vector<string> allNotMappedVariables = adapter->getAllNotMappableVariablesNames();
66  if(allNotMappedVariables.size() > 0) {
67  cout << "The following VariableNodes cant be mapped, because they are not member in PV-Manager:" << endl;
68  for(string var:allNotMappedVariables) {
69  cout << var << endl;
70  }
71  }
72 }
