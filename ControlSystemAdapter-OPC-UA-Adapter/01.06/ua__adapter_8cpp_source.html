104 void ua_uaadapter::readConfig() {
105 
106  string xpath = "//config";
107  xmlXPathObjectPtr result = this->fileHandler->getNodeSet(xpath);
108  string placeHolder = "";
109  if(result) {
110  xmlNodeSetPtr nodeset = result->nodesetval;
111  // There should be only one <config>-Tag in config file
112  if(nodeset->nodeNr > 1) {
113  //cout << "More than one <config>-Tag was found in config file. Please provide only one Tag." << endl;
114  throw std::runtime_error ("To many <config>-Tags in config file");
115  }
116 
117  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "rootFolder");
118  if(!placeHolder.empty()) {
119  this->serverConfig.rootFolder = placeHolder;
120  }
121 
122  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "description");
123  if(!placeHolder.empty()) {
124  this->serverConfig.descriptionFolder = placeHolder;
125  }
126  }
127 
128  result = this->fileHandler->getNodeSet(xpath + "//login");
129  if(result) {
130  xmlNodeSetPtr nodeset = result->nodesetval;
131  this->serverConfig.UsernamePasswordLogin = UA_TRUE;
132  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "password");
133  if(!placeHolder.empty()) {
134  this->serverConfig.password = placeHolder;
135  }
136 
137  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "username");
138  if(!placeHolder.empty()) {
139  this->serverConfig.username = placeHolder;
140  }
141  }
142  else {
143  this->serverConfig.UsernamePasswordLogin = UA_FALSE;
144  }
145 
146  result = this->fileHandler->getNodeSet(xpath +"//serverConfig");
147  if(result) {
148  xmlNodeSetPtr nodeset = result->nodesetval;
149  string opcuaPort = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "port");
150  if(opcuaPort.compare("") != 0) {
151  this->serverConfig.opcuaPort = std::stoi(opcuaPort);
152  }
153  else {
154  cout << "No 'port'-Attribute in config file is set. Use default Port: 16664" << endl;
155  }
156 
157  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "applicationName");
158  if(placeHolder.compare("") != 0) {
159  this->serverConfig.applicationName = placeHolder;
160  }
161  else {
162  cout << "No 'applicationName'-Attribute is set in config file. Use default Applicationname." << endl;
163  }
164  }
165  else {
166  cout << "No <serverConfig>-Tag in config file. Use default port 16664 and application name configuration." << endl;
167  }
168 }

233 void ua_uaadapter::addVariable(std::string varName, boost::shared_ptr<ControlSystemPVManager> csManager) {
234 
235  ua_processvariable *processvariable = new ua_processvariable(this->mappedServer, this->variablesListId, varName, csManager);
236  this->variables.push_back(processvariable);
237 
238  xmlXPathObjectPtr result = this->fileHandler->getNodeSet("//map");
239  xmlNodeSetPtr nodeset;
240  string srcVarName = "";
241  string applicName = "";
242 
243  // TODO. What happen if application name are not unique?
244  if(result) {
245  nodeset = result->nodesetval;
246  for (int32_t i=0; i < nodeset->nodeNr; i++) {
247  srcVarName = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[i], "sourceVariableName");
248  //keyword = xmlNodeListGetString(this->fileHandler->getDoc(), nodeset->nodeTab[i]->xmlChildrenNode, 1);
249  if(varName.compare(srcVarName) == 0) {
250  // get name attribute from <application>-tag
251  applicName = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[i]->parent, "name");
252  // Check if "rename" is not empty
253  string renameVar = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[i], "rename");
254  string engineeringUnit = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[i], "engineeringUnit");
255  string description = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[i], "description");
256 
257  // Application Name have to be unique!!!
258  UA_NodeId appliFolderNodeId = this->existFolder(this->ownNodeId, applicName);
259  FolderInfo newFolder;
260  if(UA_NodeId_isNull(&appliFolderNodeId)) {
261  newFolder.folderName = applicName;
262  newFolder.folderNodeId = this->createFolder(this->ownNodeId, applicName);
263  this->folderVector.push_back(newFolder);
264  appliFolderNodeId = newFolder.folderNodeId;
265  }
266 
267  vector<string> varPathVector;
268  vector<xmlNodePtr> nodeVectorUnrollPath = this->fileHandler->getNodesByName(nodeset->nodeTab[i]->children, "unrollPath");
269  string seperator = "";
270  bool unrollPathIs = false;
271  for(auto nodeUnrollPath: nodeVectorUnrollPath) {
272  string shouldUnrollPath = this->fileHandler->getContentFromNode(nodeUnrollPath);
273  if(shouldUnrollPath.compare("True") == 0) {
274  seperator = seperator + this->fileHandler->getAttributeValueFromNode(nodeUnrollPath, "pathSep");
275  unrollPathIs = true;
276  }
277  }
278 
279  if(!seperator.empty()) {
280  vector<string> newPathVector = this->fileHandler->praseVariablePath(srcVarName, seperator);
281  varPathVector.insert(varPathVector.end(), newPathVector.begin(), newPathVector.end());
282  }
283 
284  // assumption last element is name of variable, hence no folder for name is needed
285  if(renameVar.compare("") == 0 && !unrollPathIs) {
286  renameVar = srcVarName;
287  // std::cout << "Variable '" << srcVarName << "' renamed in '" << renameVar << "' and listed in folder '" << applicName << "'." << std::endl;
288  }
289  else {
290  if(unrollPathIs && renameVar.compare("") == 0) {
291  renameVar = varPathVector.at(varPathVector.size()-1);
292  varPathVector.pop_back();
293  }
294  else {
295  if(varPathVector.size() > 0) {
296  varPathVector.pop_back();
297  }
298  }
299  // std::cout << "Variable '" << srcVarName << "' listed in folder '" << applicName << "'." << std::endl;
300  }
301 
302 
303 
304  vector<xmlNodePtr> nodeVectorFolderPath = this->fileHandler->getNodesByName(nodeset->nodeTab[i]->children, "folder");
305  vector<string> folderPathVector;
306  bool createdVar = false;
307  UA_NodeId newFolderNodeId = UA_NODEID_NULL;
308  vector<UA_NodeId> mappedVariables;
309  for(auto nodeFolderPath: nodeVectorFolderPath) {
310 
311  string folderPath = this->fileHandler->getContentFromNode(nodeFolderPath);
312  if(folderPath.empty() && unrollPathIs) {
313  break;
314  }
315 
316  folderPathVector = this->fileHandler->praseVariablePath(folderPath);
317  // Create folders
318  newFolderNodeId = appliFolderNodeId;
319  if(folderPathVector.size() > 0) {
320  newFolderNodeId = this->createFolderPath(newFolderNodeId, folderPathVector);
321  }
322 
323  if(varPathVector.size() > 0) {
324  newFolderNodeId = this->createFolderPath(newFolderNodeId, varPathVector);
325  }
326  mappedVariables.push_back(newFolderNodeId);
327  createdVar = true;
328  }
329 
330  // in case no <folder> or <unrollpath> is set
331  if(!createdVar) {
332  newFolderNodeId = appliFolderNodeId;
333 
334  if(varPathVector.size() > 0) {
335  mappedVariables.push_back(this->createFolderPath(newFolderNodeId, varPathVector));
336  }
337  else {
338  // No <folder>
339  mappedVariables.push_back(appliFolderNodeId);
340  }
341  }
342 
343  // Create all nessesary mapped ObjectVaraibles with inner variables (reference or attribute, depending attributes are set (engineeringUnit, dexcription)
344  for(auto objectNodeId:mappedVariables) {
345  UA_NodeId createdNodeId = UA_NODEID_NULL;
346 
347  // Create our new "Value" Variable
348  UA_ObjectAttributes oAttr;
349  UA_ObjectAttributes_init(&oAttr);
350  oAttr.displayName = UA_LOCALIZEDTEXT_ALLOC("en_US", renameVar.c_str());
351  oAttr.description = UA_LOCALIZEDTEXT_ALLOC("en_US", description.c_str());
352 
353  UA_INSTATIATIONCALLBACK(icb);
354  UA_Server_addObjectNode(this->mappedServer, UA_NODEID_NUMERIC(1, 0),
355  objectNodeId, UA_NODEID_NUMERIC(0, UA_NS0ID_HASCOMPONENT),
356  UA_QUALIFIEDNAME_ALLOC(1, renameVar.c_str()), UA_NODEID_NULL, oAttr, &icb, &createdNodeId);
357 
358  UA_ExpandedNodeId *targetNodeId = UA_ExpandedNodeId_new();
359  targetNodeId->nodeId = createdNodeId;
360 
361  UA_BrowseDescription bDesc;
362  UA_BrowseDescription_init(&bDesc);
363  bDesc.browseDirection = UA_BROWSEDIRECTION_FORWARD;
364  bDesc.includeSubtypes = false;
365  bDesc.nodeClassMask = UA_NODECLASS_VARIABLE;
366  bDesc.nodeId = processvariable->getOwnNodeId();
367  bDesc.resultMask = UA_BROWSERESULTMASK_ALL;
368 
369  UA_BrowseResult bRes;
370  UA_BrowseResult_init(&bRes);
371 
372  UA_VariableAttributes vAttr;
373  UA_VariableAttributes_init(&vAttr);
374  vAttr.dataType = UA_NODEID_NUMERIC(0, UA_NS0ID_STRING);
375  vAttr.accessLevel = UA_ACCESSLEVELMASK_WRITE^UA_ACCESSLEVELMASK_READ;
376  vAttr.userAccessLevel = UA_ACCESSLEVELMASK_WRITE^UA_ACCESSLEVELMASK_READ;
377  vAttr.valueRank = -1;
378 
379 
380  bRes = UA_Server_browse(this->mappedServer, 10, &bDesc);
381 
382  for(uint32_t i=0; i < bRes.referencesSize; i++) {
383  UA_NodeId newNodeId = UA_NODEID_NULL;
384 
385  UA_String varName = UA_String_fromChars("EngineeringUnit");
386  if(UA_String_equal(&bRes.references[i].browseName.name, &varName) && !engineeringUnit.empty()) {
387  vAttr.description = UA_LOCALIZEDTEXT((char*)"en_US",(char*) "EngineeringUnit");
388  vAttr.displayName = UA_LOCALIZEDTEXT((char*)"en_US",(char*) "EngineeringUnit");
389 
390  UA_String engineringUnit = UA_String_fromChars(engineeringUnit.c_str());
391  UA_Variant_setScalar(&vAttr.value, &engineringUnit, &UA_TYPES[UA_TYPES_STRING]);
392  UA_Server_addVariableNode(this->mappedServer, UA_NODEID_NUMERIC(1, 0), createdNodeId,
393  UA_NODEID_NUMERIC(0, UA_NS0ID_HASCOMPONENT), UA_QUALIFIEDNAME(1, (char*) "EngineeringUnit"),
394  UA_NODEID_NUMERIC(0, UA_NS0ID_BASEDATAVARIABLETYPE), vAttr, &icb, &newNodeId);
395  }
396 
397  varName = UA_String_fromChars("Description");
398  if(UA_String_equal(&bRes.references[i].browseName.name, &varName) && !description.empty()) {
399  vAttr.description = UA_LOCALIZEDTEXT((char*)"en_US",(char*) "Description");
400  vAttr.displayName = UA_LOCALIZEDTEXT((char*)"en_US",(char*) "Description");
401 
402  UA_String engineringUnit = UA_String_fromChars(description.c_str());
403  UA_Variant_setScalar(&vAttr.value, &engineringUnit, &UA_TYPES[UA_TYPES_STRING]);
404  UA_Server_addVariableNode(this->mappedServer, UA_NODEID_NUMERIC(1, 0), createdNodeId,
405  UA_NODEID_NUMERIC(0, UA_NS0ID_HASCOMPONENT), UA_QUALIFIEDNAME(1, (char*) "Description"),
406  UA_NODEID_NUMERIC(0, UA_NS0ID_BASEDATAVARIABLETYPE), vAttr, &icb, &newNodeId);
407  }
408 
409  if(UA_NodeId_isNull(&newNodeId)) {
410  UA_Server_addReference(this->mappedServer, bRes.references[i].nodeId.nodeId, UA_NODEID_NUMERIC(0, UA_NS0ID_HASCOMPONENT), *targetNodeId, false);
411  }
412  }
413 
414  UA_BrowseDescription_deleteMembers(&bDesc);
415  UA_BrowseResult_deleteMembers(&bRes);
416 
417  }
418  }
419  }
420  }
421 
422  xmlXPathFreeObject (result);
423 }
