196 string ua_processvariable::getType() {
197  // Note: typeid().name() may return the name; may as well return the symbol's name from the binary though...
198  std::type_info const & valueType = this->csManager->getProcessVariable(this->namePV)->getValueType();
199  if (valueType == typeid(int8_t)) return "int8_t";
200  else if (valueType == typeid(uint8_t)) return "uint8_t";
201  else if (valueType == typeid(int16_t)) return "int16_t";
202  else if (valueType == typeid(uint16_t)) return "uint16_t";
203  else if (valueType == typeid(int32_t)) return "int32_t";
204  else if (valueType == typeid(uint32_t)) return "uint32_t";
205  else if (valueType == typeid(int64_t)) return "int64_t";
206  else if (valueType == typeid(uint64_t)) return "uint64_t";
207  else if (valueType == typeid(float)) return "float";
208  else if (valueType == typeid(double)) return "double";
209  else if (valueType == typeid(string)) return "string";
210  else if (valueType == typeid(bool)) return "bool";
211  else return "Unsupported type";
212 }

215  const UA_NodeId nodeid, UA_Boolean includeSourceTimeStamp,

218

229  case UA_PV_INT16:
230  rv = thisObj->getValue_int16(&value->value);
231  break;

233  case UA_PV_UINT16:

235  break;
236 
237  case UA_PV_INT32:
238  rv = thisObj->getValue_int32(&value->value);
239  break;
240 
241  case UA_PV_UINT32:
242  rv = thisObj->getValue_uint32(&value->value);
243  break;
244 
245  case UA_PV_INT64:
246  rv = thisObj->getValue_int64(&value->value);
247  break;
248 
249  case UA_PV_UINT64:
250  rv = thisObj->getValue_uint64(&value->value);
251  break;
252 
253  case UA_PV_FLOAT:
254  rv = thisObj->getValue_float(&value->value);
255  break;
256 
257  case UA_PV_DOUBLE:
258  rv = thisObj->getValue_double(&value->value);
259  break;
260 
261  case UA_PV_STRING:
262  rv = thisObj->getValue_string(&value->value);
263  break;
264 
265  case UA_PV_BOOL:

275  value->hasValue = true;
276  if (includeSourceTimeStamp) {
277  value->sourceTimestamp = thisObj->getSourceTimeStamp();
278  value->hasSourceTimestamp = true;
279  }
280  }

284 UA_StatusCode ua_processvariable::getValue_int8(UA_Variant* v) {
285  UA_StatusCode rv = UA_STATUSCODE_BADINTERNALERROR;
286 
287  if (this->type == UA_PV_INT8) {
288  if (this->csManager->getProcessVariable(this->namePV)->isReadable()) {
289  this->csManager->getProcessArray <int8_t>(this->namePV)->readLatest();
290  }
291  if (this->csManager->getProcessArray <int8_t>(this->namePV)->accessChannel(0).size() == 1) {
292  uint8_t ival = this->csManager->getProcessArray <int8_t>(this->namePV)->accessChannel(0).at(0);
293  rv = UA_Variant_setScalarCopy(v, &ival, &UA_TYPES[UA_TYPES_SBYTE]);
294  }
295  else {
296  // Array
297  std::vector<int8_t> iarr = this->csManager->getProcessArray<int8_t>(this->namePV)->accessChannel(0);
298  UA_Variant_setArrayCopy(v, iarr.data(), iarr.size(), &UA_TYPES[UA_TYPES_SBYTE]);
299  UA_NumericRange arrayRange;
300  arrayRange.dimensionsSize = 1;
301  UA_NumericRangeDimension
302  scalarThisDimension = (UA_NumericRangeDimension)
303  {
304  .min = 0, .max = (unsigned)iarr.size()
305  };
306  arrayRange.dimensions = &scalarThisDimension;
307  rv = UA_Variant_setRangeCopy(v, (UA_SByte *)iarr.data(), iarr.size(), arrayRange);
308  }
309  }
310 
311  return rv;
312 }
