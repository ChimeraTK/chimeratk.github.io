63 static string cleanUri(string s) {
64  //replace multiple with a single space
65  std::regex r1("\\s+");
66  s = std::regex_replace(s, r1, " ");
67  s = std::regex_replace(s, r1, "_");
68  //remove all not printable basic ASCII character
69  std::regex r2("[^ -~]+");
70  s = std::regex_replace(s, r2, "");
71  return s;
72 }

130 void ua_uaadapter::readConfig() {
131  string xpath = "//config";
132  xmlXPathObjectPtr result = this->fileHandler->getNodeSet(xpath);
133  string placeHolder;
134  if(result) {
135  xmlNodeSetPtr nodeset = result->nodesetval;
136  if(nodeset->nodeNr > 1) {
137  throw std::runtime_error("To many <config>-Tags in config file");
138  }
139 
140  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "rootFolder");
141  if(!placeHolder.empty()) {
142  this->serverConfig.rootFolder = placeHolder;
143  }
144 
145  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "description");
146  if(!placeHolder.empty()) {
147  this->serverConfig.descriptionFolder = placeHolder;
148  }
149  vector<xmlNodePtr> mappingExceptionsVector =
150  this->fileHandler->getNodesByName(nodeset->nodeTab[0]->children, "mapping_exceptions");
151  if(mappingExceptionsVector.empty()) {
152  this->mappingExceptions = UA_FALSE;
153  }
154  else {
155  placeHolder = this->fileHandler->getContentFromNode(mappingExceptionsVector[0]);
156  transform(placeHolder.begin(), placeHolder.end(), placeHolder.begin(), ::toupper);
157  if(placeHolder.compare("TRUE") == 0) {
158  this->mappingExceptions = UA_TRUE;
159  }
160  else {
161  this->mappingExceptions = UA_FALSE;
162  }
163  }
164 
165  xmlXPathFreeObject(result);
166  }
167  result = this->fileHandler->getNodeSet(xpath + "//login");
168  if(result) {
169  xmlNodeSetPtr nodeset = result->nodesetval;
170  if(nodeset->nodeNr > 1) {
171  throw std::runtime_error("To many <login>-Tags in config file");
172  }
173  this->serverConfig.UsernamePasswordLogin = UA_TRUE;
174  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "password");
175  if(!placeHolder.empty()) {
176  this->serverConfig.password = placeHolder;
177  }
178  else {
179  throw std::runtime_error("<login>-Tag requires username");
180  }
181 
182  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "username");
183  if(!placeHolder.empty()) {
184  this->serverConfig.username = placeHolder;
185  }
186  else {
187  throw std::runtime_error("<login>-Tag requires password");
188  }
189 
190  xmlXPathFreeObject(result);
191  }
192  else {
193  this->serverConfig.UsernamePasswordLogin = UA_FALSE;
194  }
195  result = this->fileHandler->getNodeSet(xpath + "//server");
196  if(result) {
197  xmlNodeSetPtr nodeset = result->nodesetval;
198  if(nodeset->nodeNr > 1) {
199  throw std::runtime_error("To many <server>-Tags in config file");
200  }
201  string opcuaPort = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "port");
202  if(!opcuaPort.empty()) {
203  this->serverConfig.opcuaPort = std::stoi(opcuaPort);
204  }
205  else {
206  cout << "No 'port'-Attribute in config file is set. Use default Port: 16664" << endl;
207  }
208 
209  placeHolder = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[0], "applicationName");
210  if(!placeHolder.empty()) {
211  this->serverConfig.applicationName = placeHolder;
212  if(this->serverConfig.rootFolder.empty()) {
213  this->serverConfig.rootFolder = placeHolder;
214  }
215  }
216  else {
217  string applicationName;
218  try {
219  string applicationName = ApplicationBase::getInstance().getName();
220  this->serverConfig.applicationName = applicationName.c_str();
221  }
222  catch(ChimeraTK::logic_error) {
223  }
224  cout << "No 'applicationName'-Attribute is set in config file. Use default application-name." << endl;
225  }
226  //if no root folder name is set, use application name
227  if(this->serverConfig.rootFolder.empty()) {
228  this->serverConfig.rootFolder = this->serverConfig.applicationName;
229  }
230  xmlXPathFreeObject(result);
231  }
232  else {
233  cout << "No <server>-Tag in config file. Use default port 16664 and application name configuration." << endl;
234  this->serverConfig.rootFolder = this->serverConfig.applicationName;
235  }
236  result = this->fileHandler->getNodeSet(xpath + "//process_variable_hierarchy");
237  if(result) {
238  xmlNodeSetPtr nodeset = result->nodesetval;
239  vector<xmlNodePtr> nodeVectorUnrollPathPV =
240  this->fileHandler->getNodesByName(nodeset->nodeTab[0]->children, "unroll");
241  for(auto nodeUnrollPath : nodeVectorUnrollPathPV) {
242  string unrollSepEnabled = this->fileHandler->getContentFromNode(nodeUnrollPath);
243  transform(unrollSepEnabled.begin(), unrollSepEnabled.end(), unrollSepEnabled.begin(), ::toupper);
244  if(unrollSepEnabled.compare("TRUE") == 0) {
245  this->pvSeperator += this->fileHandler->getAttributeValueFromNode(nodeUnrollPath, "pathSep");
246  }
247  }
248  xmlXPathFreeObject(result);
249  }
250  else {
251  cout << "No <process_variable_hierarchy>-Tag in config file. Use default hierarchical mapping with '/'."
252  << endl;
253  this->pvSeperator = "/";
254  }

317 void ua_uaadapter::deepCopyHierarchicalLayer(
318  boost::shared_ptr<ControlSystemPVManager> csManager, UA_NodeId layer, UA_NodeId target) {
319  //copy pv's of current layer
320  UA_BrowseDescription bd;
321  bd.includeSubtypes = false;
322  bd.nodeId = layer;
323  bd.referenceTypeId = UA_NODEID_NUMERIC(0, UA_NS0ID_HASCOMPONENT);
324  bd.resultMask = UA_BROWSERESULTMASK_NONE;
325  bd.nodeClassMask = UA_NODECLASS_VARIABLE;
326  bd.browseDirection = UA_BROWSEDIRECTION_FORWARD;
327  UA_BrowseResult br = UA_Server_browse(this->mappedServer, UA_UINT32_MAX, &bd);
328  for(size_t j = 0; j < br.referencesSize; ++j) {
329  UA_ReferenceDescription rd = br.references[j];
330  //get unit, desc, ..
331  UA_LocalizedText foundPVName;
332  string foundPVNameCPP;
333  UA_Server_readDisplayName(this->mappedServer, rd.nodeId.nodeId, &foundPVName);
334  UASTRING_TO_CPPSTRING(foundPVName.text, foundPVNameCPP);
335 
336  string pvSourceNameid;
337  UA_String foundPVSourceName;
338  string foundPVSourceNameCPP;
339  UA_Variant value;
340  UA_STRING_TO_CPPSTRING_COPY(&rd.nodeId.nodeId.identifier.string, &pvSourceNameid);
341  UA_Server_readValue(this->mappedServer, UA_NODEID_STRING(1, (char*)(pvSourceNameid + "/Name").c_str()), &value);
342  foundPVSourceName = *((UA_String*)value.data);
343  UASTRING_TO_CPPSTRING(foundPVSourceName, foundPVSourceNameCPP);
344  string varName = this->fileHandler->praseVariablePath(foundPVNameCPP, "/").back();
345  auto* processvariable =
346  new ua_processvariable(this->mappedServer, target, foundPVSourceNameCPP, csManager, foundPVNameCPP);
347  this->variables.push_back(processvariable);
348  }
349  UA_BrowseResult_clear(&br);
350  //copy folders of current layer
351  bd.includeSubtypes = false;
352  bd.nodeId = layer;
353  bd.referenceTypeId = UA_NODEID_NUMERIC(0, UA_NS0ID_ORGANIZES);
354  bd.resultMask = UA_BROWSERESULTMASK_NONE;
355  bd.nodeClassMask = UA_NODECLASS_OBJECT;
356  bd.browseDirection = UA_BROWSEDIRECTION_FORWARD;
357  br = UA_Server_browse(this->mappedServer, UA_UINT32_MAX, &bd);
358  for(size_t j = 0; j < br.referencesSize; ++j) {
359  UA_ReferenceDescription rd = br.references[j];
360  UA_String folderName, description;
361  UA_LocalizedText foundFolderName, foundFolderDescription;
362  UA_LocalizedText_init(&foundFolderDescription);
363  string foundFolderNameCPP;
364  UA_Server_readDescription(this->mappedServer, rd.nodeId.nodeId, &foundFolderDescription);
365  UA_StatusCode result = UA_Server_readDisplayName(this->mappedServer, rd.nodeId.nodeId, &foundFolderName);
366  UASTRING_TO_CPPSTRING(foundFolderName.text, foundFolderNameCPP);
367  if(result != UA_STATUSCODE_GOOD) {
368  //error handling
369  continue;
370  }
371  UA_NodeId newRootFolder = createFolder(target, foundFolderNameCPP);
372  if(foundFolderDescription.text.length > 0)
373  UA_Server_writeDescription(this->mappedServer, newRootFolder, foundFolderDescription);
374  deepCopyHierarchicalLayer(csManager, rd.nodeId.nodeId, newRootFolder);
375  }
376  UA_BrowseResult_clear(&br);
377 }

379 void ua_uaadapter::buildFolderStructure(boost::shared_ptr<ControlSystemPVManager> csManager) {
380  xmlXPathObjectPtr result = this->fileHandler->getNodeSet("//folder");
381  xmlNodeSetPtr nodeset;
382  if(result) {
383  nodeset = result->nodesetval;
384  for(int32_t i = 0; i < nodeset->nodeNr; i++) {
385  UA_NodeId folderPathNodeId;
386  string destination, description, folder;
387  vector<xmlNodePtr> nodeFolderPath =
388  this->fileHandler->getNodesByName(nodeset->nodeTab[i]->children, "destination");
389  vector<xmlNodePtr> nodeFolder = this->fileHandler->getNodesByName(nodeset->nodeTab[i]->children, "name");
390  vector<xmlNodePtr> nodeDescription =
391  this->fileHandler->getNodesByName(nodeset->nodeTab[i]->children, "description");
392  string sourceName = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[i], "sourceName");
393  string copy = this->fileHandler->getAttributeValueFromNode(nodeset->nodeTab[i], "copy");
394  if(!nodeFolderPath.empty()) {
395  destination = this->fileHandler->getContentFromNode(nodeFolderPath[0]);
396  }
397  if(!nodeDescription.empty()) {
398  description = this->fileHandler->getContentFromNode(nodeDescription[0]);
399  }
400  if(!nodeFolder.empty()) {
401  folder = this->fileHandler->getContentFromNode(nodeFolder[0]);
402  }
403  if(folder.empty()) {
404  if(this->mappingExceptions) {
405  throw std::runtime_error("Error! Folder creation failed. Name is missing. Mapping line number: " +
406  to_string(nodeset->nodeTab[i]->line));
407  }
408  if(description.empty()) {
409  UA_LOG_WARNING(&this->server_config->logger, UA_LOGCATEGORY_USERLAND,
410  "Warning! Skipping Folder. Folder creation failed. Name is missing. Mapping line number: %u",
411  nodeset->nodeTab[i]->line);
412  }
413  else {
414  UA_LOG_WARNING(&this->server_config->logger, UA_LOGCATEGORY_USERLAND,
415  "Warning! Skipping Folder. Folder creation failed. Name is missing 'description: %s'. Mapping "
416  "line number: %u",
417  description.c_str(), nodeset->nodeTab[i]->line);
418  }
419  continue;
420  }
421  //check if a pv with the requested node id exists
422  UA_NodeId tmpOutput = UA_NODEID_NULL;
423  string pvNodeString;
424  if(destination.empty()) {
425  pvNodeString += this->serverConfig.rootFolder + "/" + folder + "Dir";
426  }
427  else {
428  pvNodeString += this->serverConfig.rootFolder + "/" + destination + "/" + folder + "Dir";
429  }
430  UA_NodeId pvNode = UA_NODEID_STRING(1, (char*)pvNodeString.c_str());
431  UA_Server_readNodeId(this->mappedServer, pvNode, &tmpOutput);
432  if(!UA_NodeId_isNull(&tmpOutput)) {
433  UA_NodeId_clear(&tmpOutput);
434  //set folder description
435  if(sourceName.empty() && !description.empty()) {
436  UA_Server_writeDescription(
437  this->mappedServer, pvNode, UA_LOCALIZEDTEXT((char*)"en_US", (char*)description.c_str()));
438  continue;
439  }
440  if(this->mappingExceptions) {
441  throw std::runtime_error("Error! Folder with source mapping failed. Target folder node id exists. "
442  "Mapping line number: " +
443  to_string(nodeset->nodeTab[i]->line));
444  }
445  UA_LOG_WARNING(&this->server_config->logger, UA_LOGCATEGORY_USERLAND,
446  "Warning! Skipping folder: %s. Folder with source mapping failed. Target folder node id exists. "
447  "Mapping line number: %u",
448  sourceName.c_str(), nodeset->nodeTab[i]->line);
449  continue;
450  }
451  if(!copy.empty() && sourceName.empty()) {
452  if(this->mappingExceptions) {
453  throw std::runtime_error(
454  "Error! Folder creation failed. Source folder missing. Mapping line number: " +
455  to_string(nodeset->nodeTab[i]->line));
456  }
457  UA_LOG_WARNING(&this->server_config->logger, UA_LOGCATEGORY_USERLAND,
458  "Warning! Skipping Folder. Source 'name: %s' folder missing. Mapping line number: %u",
459  folder.c_str(), nodeset->nodeTab[i]->line);
460  continue;
461  }
462  folderPathNodeId = enrollFolderPathFromString(destination + "/replacePart", "/");
463  if(UA_NodeId_isNull(&folderPathNodeId)) {
464  folderPathNodeId = UA_NODEID_STRING(1, (char*)(this->serverConfig.rootFolder + "Dir").c_str());
465  }
466  //check if source name is set -> map complete hierarchical structure to the destination
467  if(!sourceName.empty()) {
468  if((destination.empty() && sourceName.compare(folder) == 0) ||
469  (!destination.empty() && sourceName.compare(destination + "/" + folder) == 0)) {
470  if(this->mappingExceptions) {
471  throw std::runtime_error(
472  "Error! Folder creation failed. Source and Destination equal. Mapping line number: " +
473  to_string(nodeset->nodeTab[i]->line));
474  }
475  UA_LOG_WARNING(&this->server_config->logger, UA_LOGCATEGORY_USERLAND,
476  "Warning! Skipping Folder. Source '%s' and Destination equal. Mapping line number: %u",
477  sourceName.c_str(), nodeset->nodeTab[i]->line);
478  continue;
479  }
480  //check if the src is a folder
481  string sourceFolder = this->serverConfig.rootFolder + "/" + sourceName + "Dir";
482  bool isFolderType = false;
483  UA_NodeId sourceFolderId = UA_NODEID_STRING(1, (char*)sourceFolder.c_str());
484 
485  UA_BrowseDescription bd;
486  bd.includeSubtypes = false;
487  bd.nodeId = sourceFolderId;
488  bd.referenceTypeId = UA_NODEID_NUMERIC(0, UA_NS0ID_HASTYPEDEFINITION);
489  bd.resultMask = UA_BROWSERESULTMASK_ALL;
490  bd.nodeClassMask = UA_NODECLASS_OBJECTTYPE;
491  bd.browseDirection = UA_BROWSEDIRECTION_BOTH;
492  UA_BrowseResult br = UA_Server_browse(this->mappedServer, 1000, &bd);
493  for(size_t j = 0; j < br.referencesSize; ++j) {
494  UA_ReferenceDescription rd = br.references[j];
495  UA_NodeId folderType = UA_NODEID_NUMERIC(0, UA_NS0ID_FOLDERTYPE);
496  if(UA_NodeId_equal(&rd.nodeId.nodeId, &folderType)) {
497  isFolderType = true;
498  }
499  }
500  UA_BrowseResult_clear(&br);
501  if(!isFolderType) {
502  if(this->mappingExceptions) {
503  throw std::runtime_error("Error! Folder creation failed. No corresponding source folder: " +
504  sourceName + ". Mapping line number: " + to_string(nodeset->nodeTab[i]->line));
505  }
506  UA_LOG_WARNING(&this->server_config->logger, UA_LOGCATEGORY_USERLAND,
507  "Warning! Skipping Folder. No corresponding source '%s' folder. Mapping line number: %u",
508  sourceName.c_str(), nodeset->nodeTab[i]->line);
509  continue;
510  }
511  //enroll path destination -> copy / link the complete tree to this place
512  transform(copy.begin(), copy.end(), copy.begin(), ::toupper);
513  if(copy.compare("TRUE") == 0) {
514  bool sourceAndDestinationEqual = false;
515  if(!destination.empty()) {
516  if(sourceName.compare(destination + "/" + folder) == 0) sourceAndDestinationEqual = true;
517  }
518  else {
