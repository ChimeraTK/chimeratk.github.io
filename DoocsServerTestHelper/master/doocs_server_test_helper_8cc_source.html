25 extern "C" int sigwait(__const sigset_t* __restrict __set, int* __restrict __sig) {
26  // call original-equivalent version if SIGUSR1 is not in the set
27  if(!sigismember(__set, SIGUSR1)) {
28  // first wait on the specified signals
29  siginfo_t __siginfo;
30  int iret = sigwaitinfo(__set, &__siginfo);
31  *__sig = __siginfo.si_signo;
32  // if (in the mean time) doNotProcessSignalsInDoocs has been set by the test
33  // code, wait on an empty set (-> forever)
34  if(DoocsServerTestHelper::doNotProcessSignalsInDoocs) {
35  siginfo_t __siginfo;
36  sigset_t ___set;
37  sigemptyset(&___set);
38  iret = sigwaitinfo(&___set, &__siginfo);
39  *__sig = __siginfo.si_signo;
40  }
41  return iret;
42  }
43 
44  // SIGUSR1 is in the set: wait until sigusr1 reqested via mutex
45  static thread_local std::unique_lock<std::mutex> sigusr1_lock{DoocsServerTestHelper::sigusr1_mutex};
46  do {
47  sigusr1_lock.unlock();
48  usleep(1);
49  sigusr1_lock.lock();
50  } while(!DoocsServerTestHelper::allowSigusr1);
51  DoocsServerTestHelper::allowSigusr1 = false;
52 
53  // return a SIGUSR1
54  *__sig = SIGUSR1;
55  return 0;
56 }

87 void DoocsServerTestHelper::wait_for_update(const doocs::Server* /*server*/) {
88  if(do_shutdown) return;

90  do {

143 void DoocsServerTestHelper::doocsSetSpectrum(const std::string& name, const std::vector<float>& value) {
144  EqAdr ad;
145  EqData ed, res;
146  // fill spectrum data structure
147  SPECTRUM spectrum;
148  spectrum.comment.comment_len = 0;
149  spectrum.comment.comment_val = nullptr;
150  spectrum.tm = 0;
151  spectrum.s_start = 0;
152  spectrum.s_inc = 1.;
153  spectrum.status = 0;
154  spectrum.d_spect_array.d_spect_array_len = value.size();
155  spectrum.d_spect_array.d_spect_array_val = (float*)value.data(); // casts const-ness away, but will not be modified
156  // (hopefully)
157  ed.set(&spectrum);
158  // obtain location pointer
159  ad.adr(name.c_str());
160  EqFct* p = eq_get(&ad);
161  ASSERT(p != NULL, std::string("Could not get location for property ") + name);
162  // set spectrum
163  p->lock();
164  p->set(&ad, &ed, &res);
165  p->unlock();
166  // check for error
167  ASSERT(res.error() == 0, std::string("Error writing spectrum property ") + name);
168 }
