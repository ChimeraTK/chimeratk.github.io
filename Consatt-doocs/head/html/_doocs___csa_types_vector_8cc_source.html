115 void Doocs_CsaTypesVector::synchronizeDataToControlSystem() {
116  // std::cout << "Doocs_CsaTypesVector::synchronizeDataToControlSystem: INFO: Synchronization not implemented for
117  // spectrum: " << D_spectrum::property_name() << std::endl;
118 
119  if(getBufferToControlSystem) {
120  void* buffer = NULL;
121  uint32_t bufferSize = 0;
122  uint32_t bufferSendSize = 0;
123  bool dataHasChanged = true;
124  getBufferToControlSystem(
125  buffer, bufferSize, bufferSendSize, dataHasChanged); // this transfers array from doocs independent part
126 
127  if(this->_singleDataSendingDone && this->_sendToZmqConstData == false && dataHasChanged == false) return;
128 
129  auto ts = Doocs_CsaAdapter::getInstance().getCurrentTimeStampStructure();
130  auto dEvId = Doocs_CsaAdapter::getInstance().getDoocsEventId();
131  auto dts = Doocs_CsaAdapter::getInstance().getDoocsTimestamp();
132  auto dtsSecAndUSec = dts.get_seconds_and_microseconds_since_epoch();
133 
134  if(!this->_singleDataSendingDone && this->_sendToZmqConstData == false) {
135  if(dEvId.to_int() == 0 && _mptIsZeroCnt < 50) {
136  _mptIsZeroCnt++;
137  CsaAdapter::getInstance().globalData.getLogger()(Loger::DETAIL)
138  << __func__ << ": "
139  << "Wait to valid MPN for: " << _doocsName << " MPN is still 0. MPNisZeroCnt: " << _mptIsZeroCnt
140  << std::endl;
141  }
142  else {
143  CsaAdapter::getInstance().globalData.getLogger()(Loger::INFO)
144  << __func__ << ": "
145  << "Send once " << _doocsName << " MPN: " << dEvId.to_int() << " MPNisZeroCnt: " << _mptIsZeroCnt
146  << std::endl;
147  this->_singleDataSendingDone = true;
148  }
149  }
150 
151  _currentBufferId = 0;
152  if(_bufferAmount) {
153  _currentBufferId = dEvId.to_int() & SPECTRUM_BUFFERS_MASK;
154 
155  D_spectrum::current_buffer(_currentBufferId);
156  if(this->_dataTypeInVector == CsaPvConfigBase::BINDABLE_TYPES::INT32_T_VECTOR) {
157  this->sendFromLibraryBuffer<int32_t>(buffer, bufferSize, _currentBufferId);
158  }
159  else if(this->_dataTypeInVector == CsaPvConfigBase::BINDABLE_TYPES::UINT32_T_VECTOR) {
160  this->sendFromLibraryBuffer<uint32_t>(buffer, bufferSize, _currentBufferId);
161  }
162  else if(this->_dataTypeInVector == CsaPvConfigBase::BINDABLE_TYPES::INT16_T_VECTOR) {
163  this->sendFromLibraryBuffer<int16_t>(buffer, bufferSize, _currentBufferId);
164  }
165  else if(this->_dataTypeInVector == CsaPvConfigBase::BINDABLE_TYPES::UINT16_T_VECTOR) {
166  this->sendFromLibraryBuffer<uint16_t>(buffer, bufferSize, _currentBufferId);
167  }
168  else if(this->_dataTypeInVector == CsaPvConfigBase::BINDABLE_TYPES::FLOAT_VECTOR) {
169  this->sendFromLibraryBuffer2<float>(buffer, bufferSize, _currentBufferId);
170  }
171  else if(this->_dataTypeInVector == CsaPvConfigBase::BINDABLE_TYPES::DOUBLE_VECTOR) {
172  this->sendFromLibraryBuffer<double>(buffer, bufferSize, _currentBufferId);
173  }
174  }
175  else {
176  if(this->_dataTypeInVector == CsaPvConfigBase::BINDABLE_TYPES::INT32_T_VECTOR) {
177  this->sendFromLibraryBuffer<int32_t>(buffer, bufferSize);
178  }
179  else if(this->_dataTypeInVector == CsaPvConfigBase::BINDABLE_TYPES::UINT32_T_VECTOR) {
180  this->sendFromLibraryBuffer<uint32_t>(buffer, bufferSize);
181  }
182  else if(this->_dataTypeInVector == CsaPvConfigBase::BINDABLE_TYPES::INT16_T_VECTOR) {
183  this->sendFromLibraryBuffer2<int16_t>(buffer, bufferSize, 0);
184  }
185  else if(this->_dataTypeInVector == CsaPvConfigBase::BINDABLE_TYPES::UINT16_T_VECTOR) {
186  this->sendFromLibraryBuffer<uint16_t>(buffer, bufferSize);
187  }
188  else if(this->_dataTypeInVector == CsaPvConfigBase::BINDABLE_TYPES::FLOAT_VECTOR) {
189  this->sendFromLibraryBuffer2<float>(buffer, bufferSize, 0);
190  }
191  else if(this->_dataTypeInVector == CsaPvConfigBase::BINDABLE_TYPES::DOUBLE_VECTOR) {
192  this->sendFromLibraryBuffer<double>(buffer, bufferSize);
193  }
194  }
195 
196  D_spectrum::macro_pulse(dEvId.to_int(), _currentBufferId);
197  D_spectrum::set_mpnum(dEvId.to_int());
198  D_spectrum::set_tmstmp(dtsSecAndUSec.seconds, dtsSecAndUSec.microseconds, _currentBufferId);
199 
200  this->length(bufferSendSize);
201  if(this->_objConfApp.getExposePolicy() == P_ExposePolicy::PUSH) this->send(&ts);
202  }
203  else {
204  std::cout << "Doocs_CsaTypesVector::synchronizeDataToControlSystem: ERROR: getBufferToControlSystem not set."
205  << std::endl;
206  }
207 }
