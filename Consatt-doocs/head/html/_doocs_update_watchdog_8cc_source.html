28void DoocsUpdateWatchdog::update() {
29 std::lock_guard<std::mutex> guard(_mutex);
30 if(!_enabled.value()) {
31 if(_status.value() != DoocsUpdateWatchdog::STATUS::DISABLED) {
32 CsaAdapter::getInstance().globalData.getLogger()(Loger::ALWAYS)
33 << __func__ << ": "
34 << "DOOCS UPDATE WATCHDOG - Software interrupt watchdog set to DISABLED. Presence of software updates is "
35 "not "
36 "monitored."
37 << std::endl;
38 }
39 _status.set_value(int(DoocsUpdateWatchdog::STATUS::DISABLED));
40 return;
41 }
42
43 auto now = std::chrono::system_clock::now().time_since_epoch();
44 auto timeFromLastUpdateInSeconds =
45 std::chrono::duration_cast<std::chrono::seconds>(now - this->_lastUpdateTime).count();
46
47 if(timeFromLastUpdateInSeconds > _timeoutValueInSeconds) {
48 if(_status.value() == DoocsUpdateWatchdog::STATUS::OK || _status.value() == DoocsUpdateWatchdog::STATUS::DISABLED) {
49 CsaAdapter::getInstance().globalData.getLogger()(Loger::ALWAYS)
50 << __func__ << ": " << "DOOCS UPDATE WATCHDOG - Timeout . No software interrupt registered for last "
51 << _timeoutValueInSeconds << " seconds. Set DOOCS Update Timeout status to error." << std::endl;
52 _status.set_value(int(DoocsUpdateWatchdog::STATUS::ERROR));
53 }
54 }
55 else {
56 if(_status.value() == DoocsUpdateWatchdog::STATUS::ERROR) {
57 CsaAdapter::getInstance().globalData.getLogger()(Loger::ALWAYS)
58 << __func__ << ": "
59 << "DOOCS UPDATE WATCHDOG - Software interrupt seems to be back. Set DOOCS Update Timeout status "
60 "from error to OK."
61 << std::endl;
62 }
63 _status.set_value(int(DoocsUpdateWatchdog::STATUS::OK));
64 }
65}
