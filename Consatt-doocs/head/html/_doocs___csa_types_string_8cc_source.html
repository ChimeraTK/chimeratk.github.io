55 void Doocs_CsaTypesString::synchronizeDataToControlSystem() {
56  if(this->_csAccessDirection == CsAccessDir::TO_CS && this->getToCs) {
57  _textCopy = this->getToCs();
58 
59  unsigned int maxSize = 4 * 1024;
60  if(_textCopy.size() > maxSize) {
61  if(this->d_store()) {
62  std::cout << Loger::TimeStampGen::getTime() << " -> Property " << _doocsName << " has " << _textCopy.size()
63  << " chars and is longer than " << maxSize << ". Disable storing in config file." << std::endl;
64  this->store_disable();
65  }
66  }
67  else if(!this->d_store()) {
68  std::cout << Loger::TimeStampGen::getTime() << " -> Property " << _doocsName << " has " << _textCopy.size()
69  << " chars and is shorter than " << maxSize << " chars. Re-enable storing in config file." << std::endl;
70  this->store_enable();
71  }
72 
73  if(this->_singleDataSendingDone && this->_sendToZmqConstData == false && _textCopy.compare(this->txt_) == 0) return;
74 
75  this->txt_ = const_cast<char*>(_textCopy.c_str());
76 
77  if(this->_objConfApp.getExposePolicy() == P_ExposePolicy::PUSH) {
78  auto ts = Doocs_CsaAdapter::getInstance().getCurrentTimeStampStructure();
79  auto dEvId = Doocs_CsaAdapter::getInstance().getDoocsEventId();
80  auto dts = Doocs_CsaAdapter::getInstance().getDoocsTimestamp();
81  auto dtsSecAndUSec = dts.get_seconds_and_microseconds_since_epoch();
82 
83  if(!this->_singleDataSendingDone && this->_sendToZmqConstData == false) {
84  if(dEvId.to_int() == 0 && _mptIsZeroCnt < 50) {
85  _mptIsZeroCnt++;
86  CsaAdapter::getInstance().globalData.getLogger()(Loger::DETAIL)
87  << __func__ << ": "
88  << "Wait to valid MPN for: " << _doocsName << " Value: " << this->txt_
89  << " MPN is still 0. MPNisZeroCnt: " << _mptIsZeroCnt << std::endl;
90  }
91  else {
92  CsaAdapter::getInstance().globalData.getLogger()(Loger::INFO)
93  << __func__ << ": "
94  << "Send once " << _doocsName << " Value: " << this->txt_ << " MPN: " << dEvId.to_int()
95  << " MPNisZeroCnt: " << _mptIsZeroCnt << std::endl;
96  this->_singleDataSendingDone = true;
97  }
98  }
99 
100  D_text::set_mpnum(dEvId.to_int());
101  D_text::set_tmstmp(dtsSecAndUSec.seconds, dtsSecAndUSec.microseconds);
102  this->send(&ts);
103  }
104  }
105 
106  if(this->_csAccessDirection == CsAccessDir::FROM_CS) {
107  if(this->_singleDataSendingDone && this->_sendToZmqConstData == false && _textCopy.compare(this->txt_) == 0) return;
108 
109  auto ts = Doocs_CsaAdapter::getInstance().getCurrentTimeStampStructure();
110  auto dEvId = Doocs_CsaAdapter::getInstance().getDoocsEventId();
111  auto dts = Doocs_CsaAdapter::getInstance().getDoocsTimestamp();
112  auto dtsSecAndUSec = dts.get_seconds_and_microseconds_since_epoch();

114  D_text::set_tmstmp(dtsSecAndUSec.seconds, dtsSecAndUSec.microseconds);
115 
116  if(!this->_singleDataSendingDone && this->_sendToZmqConstData == false) {
117  if(dEvId.to_int() == 0 && _mptIsZeroCnt < 50) {
118  _mptIsZeroCnt++;
119  CsaAdapter::getInstance().globalData.getLogger()(Loger::DETAIL)
120  << __func__ << ": "
121  << "Wait to valid MPN for: " << _doocsName << " Value: " << this->txt_
122  << " MPN is still 0. MPNisZeroCnt: " << _mptIsZeroCnt << std::endl;
123  }
124  else {
125  CsaAdapter::getInstance().globalData.getLogger()(Loger::INFO)
126  << __func__ << ": "
127  << "Send once " << _doocsName << " Value: " << this->txt_ << " MPN: " << dEvId.to_int()
128  << " MPNisZeroCnt: " << _mptIsZeroCnt << std::endl;
129  this->_singleDataSendingDone = true;
130  }
