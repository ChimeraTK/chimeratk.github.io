136 void mexFunction(int nlhs, mxArray* plhs[], int nrhs, const mxArray* prhs[]) {
137  // Initalize basic stuff
138  if(isInit == false) {
139  // Setting devices.dmap as default is just for convenience. Some scripts
140  // might rely on it.
141  // DeviceAccess from 01.00 is throwing if the file is not there.
142  // We just catch the exception and continue. The user will set his own
143  // dmap file later if needed.
144  try {
145  ChimeraTK::setDMapFilePath("./devices.dmap");
146  }
147  catch(ChimeraTK::logic_error&) {
148  }
149 
150  isInit = true;
151  }
152 
153  if(nrhs == 0) {
154  mexErrMsgTxt("Not enough input arguments.");
155  return;
156  }
157  if(!mxIsChar(prhs[0])) {
158  mexErrMsgTxt("Invalid input arguments.");
159  return;
160  }
161 
162  string cmd = mxArrayToStdString(prhs[0]);
163  transform(cmd.begin(), cmd.end(), cmd.begin(), ::tolower);
164 
165  try {
166  vector<Command>::iterator it = vectorOfCommands.begin();
167 
168  // Look for the right command
169  for(; it != vectorOfCommands.end(); ++it) {
170  if(it->Name == cmd) break;
171  }
172 
173  // Check if search was successful
174  if(it == vectorOfCommands.end()) {
175  mexErrMsgTxt("Unknown command '" + cmd + "'. Use mtca4u('help') to show some help information.");
176  return;
177  }
178 
179  // Check if the method is implemented
180  else if(NULL == it->pCallback) {
181  mexErrMsgTxt("Command not implemented yet.");
182  return;
183  }
184 
185  // Ok run method
186  else
187  it->pCallback(nlhs, plhs, nrhs - 1, &prhs[1]);
188  }
189 
190  catch(Exception& e) {
191  mexErrMsgTxt(e.what());
192  }
193 }

490 void readRegister(unsigned int nlhs, mxArray* plhs[], unsigned int nrhs, const mxArray* prhs[]) {
491  static const unsigned int pp_device = 0, pp_module = 1, pp_register = 2, pp_offset = 3, pp_elements = 4;
492 
493  if(nrhs < 3) mexErrMsgTxt("Not enough input arguments.");
494  if(nrhs > 5) mexWarnMsgTxt("To many input arguments.");
495  if(nlhs > 1) mexErrMsgTxt("To many output arguments.");
496 
497  boost::shared_ptr<Device> device = getDevice(prhs[pp_device]);
498 
499  if(!mxIsChar(prhs[pp_module])) mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_module) + " input argument.");
500  if(!mxIsChar(prhs[pp_register])) mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_register) + " input argument.");
501 
502  if((nrhs > pp_offset) && (!mxIsRealScalar(prhs[pp_offset]) || (mxGetScalar(prhs[pp_offset]) < 0)))
503  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_offset) + " input argument.");
504 
505  if((nrhs > pp_elements) && !mxIsPositiveRealScalar(prhs[pp_elements]))
506  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_elements) + " input argument.");
507 
508  // offset is optional. Use 0 if not set
509  const uint32_t offset = (nrhs > pp_offset) ? mxGetScalar(prhs[pp_offset]) : 0;
510 
511  // number of elements is optional. Use 0 (=all remaining) if not set
512  const uint32_t nElements = (nrhs > pp_elements) ? mxGetScalar(prhs[pp_elements]) : 0;
513 
514  RegisterPath moduleName(mxArrayToStdString(prhs[pp_module]));
515  RegisterPath registerName(mxArrayToStdString(prhs[pp_register]));
516  // we get the register content as a std::vector<double>
517  auto registerContent = device->read<double>(moduleName / registerName, nElements, offset);
518 
519  // as both DeviceAccess and Matlab do their own memory allocation all we can
520  // do is memcpy :-(
521  plhs[0] = mxCreateDoubleMatrix(1, registerContent.size(), mxREAL);
522  double* plhsValue = mxGetPr(plhs[0]);
523  memcpy(plhsValue, registerContent.data(), registerContent.size() * sizeof(double));
524 }

594 void readDmaRaw(unsigned int nlhs, mxArray* plhs[], unsigned int nrhs, const mxArray* prhs[]) {
595  static const unsigned int pp_device = 0, pp_module = 1, pp_register = 2, pp_offset = 3, pp_elements = 4;
596  static const unsigned int pp_mode = 5, pp_signed = 6, pp_bit = 7, pp_fracbit = 8;
597 
598  if(nrhs < 3) mexErrMsgTxt("Not enough input arguments.");
599  if(nrhs > 9) mexWarnMsgTxt("Too many input arguments.");
600  if(nlhs > 1) mexErrMsgTxt("Too many output arguments.");
601 
602  boost::shared_ptr<Device> device = getDevice(prhs[pp_device]);
603 
604  if(!mxIsChar(prhs[pp_module])) mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_module) + " input argument.");
605  if(!mxIsChar(prhs[pp_register])) mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_register) + " input argument.");
606 
607  if((nrhs > pp_offset) && !mxIsRealScalar(prhs[pp_offset]))
608  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_offset) + " input argument.");
609  if((nrhs > pp_elements) && !mxIsPositiveRealScalar(prhs[pp_elements]))
610  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_elements) + " input argument.");
611 
612  if((nrhs > pp_mode) && !mxIsPositiveRealScalar(prhs[pp_mode]))
613  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_mode) + " input argument.");
614 
615  // FIXME: These don't make sense in raw mode. Shall we just skip the checks?
616  // They are not used anyway.
617  if((nrhs > pp_signed) && !mxIsRealScalar(prhs[pp_signed]))
618  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_signed) + " input argument.");
619  if((nrhs > pp_bit) && !mxIsPositiveRealScalar(prhs[pp_bit]))
620  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_bit) + " input argument.");
621  if((nrhs > pp_fracbit) && !mxIsRealScalar(prhs[pp_fracbit]))
622  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_fracbit) + " input argument.");
623 
624  // offset is optional. Use 0 if not set
625  const uint32_t offset = (nrhs > pp_offset) ? mxGetScalar(prhs[pp_offset]) : 0;
626  // number of elements is optional. Use 0 (=all remaining) if not set
627  const uint32_t nElements = (nrhs > pp_elements) ? mxGetScalar(prhs[pp_elements]) : 0;
628 
629  // The mode can be 16 or 32 bit. This does not make sense because the real raw
630  // stuff is 32 bits, it was there so we keep it in order not to break
631  // compatibility.
632  const uint32_t mode = (nrhs > pp_mode) ? mxGetScalar(prhs[pp_mode]) : 32;
633  if((mode != 32) && (mode != 16)) mexErrMsgTxt("Invalid data mode.");
634 
635  // Number of 32 bit words to read through the "bus". This currently is the
636  // only mode the raw accessor knows
637  uint32_t nWords32Bit = nElements;
638  if(mode == 16) {
639  // The number of 32 bit words is only half the number of elements if the are
640  // requested as 16 bit
641  nWords32Bit /= 2;
642  }
643 
644  // Notice: signedFlags, bits and fracBits have been removed as they were
645  // not/cannot be used anyway
646 
647  // Now that we have all parameters it's time to read the data from the device.
648  RegisterPath registerPath(mxArrayToStdString(prhs[pp_module]) + "/" + mxArrayToStdString(prhs[pp_register]));
649 
650  // we get the register content as a std::vector<double>
651  auto accessor = device->getOneDRegisterAccessor<int32_t>(registerPath, nWords32Bit, offset, {AccessMode::raw});
652  accessor.read();
653 
654  // frame matlab buffer with appropriate number of elements
655  auto bufferSize = (mode == 16) ? accessor.getNElements() * 2 : accessor.getNElements();
656  plhs[0] = mxCreateDoubleMatrix(bufferSize, 1, mxREAL);
657 
658  double* plhsValue = mxGetPr(plhs[0]);
659 
660  if(mode == 32) {
661  size_t i = 0;
662  // we can directly loop the accessor
663  for(auto value : accessor) {
664  plhsValue[i++] = value;
665  }
666  }
667  else {
668  auto values16Bit = reinterpret_cast<int16_t*>(accessor.data());
669  for(size_t i = 0; i < bufferSize; ++i) {
670  plhsValue[i] = values16Bit[i];
671  }
672  }
673 }

699  if((nlhs > 1) && (nrhs > pp_channel) && (nlhs > mxGetNumberOfElements(prhs[pp_channel])))
700  mexErrMsgTxt("Too many output arguments.");
701  if((nlhs > 1) && (nrhs > pp_channel) && (nlhs < mxGetNumberOfElements(prhs[pp_channel])))
702  mexErrMsgTxt("Not enough output arguments.");
703 
704  if((nrhs > pp_offset) && !mxIsRealScalar(prhs[pp_offset]))
705  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_offset) + " input argument.");
706  if((nrhs > pp_elements) && !mxIsPositiveRealScalar(prhs[pp_elements]))
707  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_elements) + " input argument.");
708 
709  auto twoDRegister = device->getTwoDRegisterAccessor<double>(
710  mxArrayToStdString(prhs[pp_module]) + "/" + mxArrayToStdString(prhs[pp_register]));
711  twoDRegister.read();
712 
713  const uint32_t offset = (nrhs > pp_offset) ? mxGetScalar(prhs[pp_offset]) : 0;
714  const uint32_t elements = (nrhs > pp_elements) ? mxGetScalar(prhs[pp_elements]) : (twoDRegister[0].size());
715 
716  const uint32_t totalChannels = twoDRegister.getNChannels();
717 
718  /*if (mxGetScalar(prhs[pp_channel]) > totalChannels)
719  mexErrMsgTxt("Requested Channel Index greater than available channels");*/
720 
721  const uint32_t size = elements * totalChannels;
722  std::vector<double> dmaValue(size);
723 
724  unsigned int selectedChannels = (nrhs > pp_channel) ? mxGetNumberOfElements(prhs[pp_channel]) : totalChannels;
725  // Store data in different vectors passed over lhs
726  if(nlhs == selectedChannels) {
727  for(unsigned int ic = 0; ic < nlhs; ic++) {
728  plhs[ic] = mxCreateDoubleMatrix(elements, 1, mxREAL);
729  double* plhsValue = mxGetPr(plhs[ic]);
730  unsigned int currentChannel = (nrhs > pp_channel) ? (mxGetPr(prhs[pp_channel])[ic] - 1) : ic;
731 
732  for(unsigned int is = 0; is < elements; is++) {
733  unsigned int seqIndex = (is + offset);
734  if(seqIndex >= size) mexErrMsgTxt("Data storage indexing went wrong!");
735  plhsValue[is] = twoDRegister[currentChannel][seqIndex];
736  }
737  }
738  }
739  // Store data in lhs matrix
740  else {
741  plhs[0] = mxCreateDoubleMatrix(elements, selectedChannels, mxREAL);
742  double* plhsValue = mxGetPr(plhs[0]);
743 
744  // copy and transform data
745  for(unsigned int ic = 0; ic < selectedChannels; ic++) {
746  int currentChannel = (nrhs > pp_channel) ? (mxGetPr(prhs[pp_channel])[ic] - 1) : ic;
747 
748  if(currentChannel < 0 || uint32_t(currentChannel) >= totalChannels) mexErrMsgTxt("Illegal Channel Index");
749 
750  for(unsigned int is = 0; is < elements; is++) {
751  unsigned int plhsIndex = ic * elements + is;
752  unsigned int seqIndex = (is + offset);
753 
754  if(seqIndex >= size) mexErrMsgTxt("Data storage indexing went wrong! (1)");
755  if(plhsIndex >= elements * selectedChannels) mexErrMsgTxt("Data storage indexing went wrong! (2)");
756 
757  plhsValue[plhsIndex] = twoDRegister[currentChannel][seqIndex];
758  }
759  }
760  }
