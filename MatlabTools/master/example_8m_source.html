1 %
2 % Minimalistic example for using the mtca4u library in Matlab
3 % Before running this example, you need to copy the correct map files
4 % into your current directory. They can be taken from the test
5 % subdirectory of the matlab_tools repository. You will need at least the
6 % files "dummies.dmap" and all *.map/*.mapp files referenced inside.
7 % Also, you need to add the "bin" and "matlab" subdirectories to the path
8 % of your Matlab installation (click "Set Path" in "Environment" in
9 % Matlab main window).
10 %
11 % Author: Martin Hierholzer <martin.hierholzer@desy.de>
12 %
13 %
14 % Open dummy device
15 % Use DUMMY2 to have the modern interface with modules
16 %
17 % Print all devices in map file
18 delete('print_all_devices.txt');
19 diary('print_all_devices.txt');
20 mtca4u.setDMapFilePath('dummies.dmap')
21 mtca4u.print_info();
22 diary off;
23 [status,cmdout] = system('diff print_all_devices.txt print_all_devices_reference');
24 assert(status == 0)
25 
26 dmapToSet = 'dummies.dmap'
27 mtca4u_mex('set_dmap', dmapToSet);
28 dmap = mtca4u_mex('get_dmap');
29 ret = strcmp(dmapToSet,dmap)
30 assert(ret == 1)
31 fprintf('opening dummy3\n');
32 m = mtca4u('DUMMY3');
33 
34 %
35 % Print some device info
36 %
37 delete('print_device_info.txt');
38 diary('print_device_info.txt');
39 m.print_device_info();
40 diary off;
41 [status,cmdout] = system('diff print_device_info.txt print_device_info_reference');
42 assert(status == 0)
43 
44 m.help();
45 
46 %
47 
48 
49 
50 % Read FIRMWARE and COMPILATION values and print them
51 %
52 fprintf('WORD_FIRMWARE = %d\n',m.read('BOARD', 'WORD_FIRMWARE'));
53 fprintf('WORD_COMPILATION = %d\n',m.read('BOARD', 'WORD_COMPILATION'));
54 
55 %
56 % Write 1 to WORD_ADC_ENA (so AREA_DMAABLE will be filled)
57 %
58 m.write('ADC', 'WORD_ADC_ENA',1);
59 
60 %
61 % Print some register info of AREA_DMAABLE_FIXEDPOINT16_3
62 %
63 m.print_register_info('ADC','AREA_DMAABLE_FIXEDPOINT16_3');
64 
65 %
66 % Obtain number of elements in AREA_DMAABLE_FIXEDPOINT16_3 register
67 %
68 nelements = m.get_register_size('ADC','AREA_DMAABLE_FIXEDPOINT16_3');
69 
70 %
71 % Create figure
72 %
73 fig = figure('Name','ADC data', 'NumberTitle','off', 'Position',[360,500,900,900]);
74 
75 %
76 % Read array from AREA_DMAABLE_FIXEDPOINT16_3 and plot it
77 % Interpretation as fixed point values will be performed automatically
78 %
79 adcdata = m.read('ADC', 'AREA_DMAABLE_FIXEDPOINT16_3', 0, nelements);
80 subplot(2,1,1);
81 plot(adcdata);
82 title('Normal read with fixed point data');
83 
84 %
85 % Alternatively read data via DMA and plot it
86 % The register is not configured to have fractional bits, so no
87 % conversion can be done here.
88 %
89 dmadata = m.read_dma_raw('ADC', 'AREA_DMA_VIA_DMA', 0, nelements);
90 subplot(2,1,2);
91 plot(dmadata);
92 title('DMA read with integer data');
93 
94 fig2 = figure('Name','Sequence data', 'NumberTitle','off', 'Position',[360,500,900,900]);
95 m2 = mtca4u('DUMMY2');

99 title('DMA read with seq data');
