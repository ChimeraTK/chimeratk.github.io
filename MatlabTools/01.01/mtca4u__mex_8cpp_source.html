138 void mexFunction(int nlhs, mxArray* plhs[], int nrhs, const mxArray* prhs[]) {
139  // Initalize basic stuff
140  if(isInit == false) {
141  // Setting devices.dmap as default is just for convenience. Some scripts
142  // might rely on it.
143  // DeviceAccess from 01.00 is throwing if the file is not there.
144  // We just catch the exception and continue. The user will set his own
145  // dmap file later if needed.
146  try {
147  ChimeraTK::setDMapFilePath("./devices.dmap");
148  }
149  catch(ChimeraTK::logic_error&) {
150  }
151 
152  isInit = true;
153  }
154 
155  if(nrhs == 0) {
156  mexErrMsgTxt("Not enough input arguments.");
157  return;
158  }
159  if(!mxIsChar(prhs[0])) {
160  mexErrMsgTxt("Invalid input arguments.");
161  return;
162  }
163 
164  string cmd = mxArrayToStdString(prhs[0]);
165  transform(cmd.begin(), cmd.end(), cmd.begin(), ::tolower);
166 
167  try {
168  vector<Command>::iterator it = vectorOfCommands.begin();
169 
170  // Look for the right command
171  for(; it != vectorOfCommands.end(); ++it) {
172  if(it->Name == cmd) break;
173  }
174 
175  // Check if search was successful
176  if(it == vectorOfCommands.end()) {
177  mexErrMsgTxt("Unknown command '" + cmd + "'. Use mtca4u('help') to show some help information.");
178  return;
179  }
180 
181  // Check if the method is implemented
182  else if(NULL == it->pCallback) {
183  mexErrMsgTxt("Command not implemented yet.");
184  return;
185  }
186 
187  // Ok run method
188  else
189  it->pCallback(nlhs, plhs, nrhs - 1, &prhs[1]);
190  }
191 
192  catch(ChimeraTK::runtime_error& e) {
193  mexErrMsgTxt(e.what());
194  }
195  catch(ChimeraTK::logic_error& e) {
196  mexErrMsgTxt(e.what());
197  }
198 }

499 void readRegister(unsigned int nlhs, mxArray* plhs[], unsigned int nrhs, const mxArray* prhs[]) {
500  static const unsigned int pp_device = 0, pp_module = 1, pp_register = 2, pp_offset = 3, pp_elements = 4;
501 
502  if(nrhs < 3) mexErrMsgTxt("Not enough input arguments.");
503  if(nrhs > 5) mexWarnMsgTxt("To many input arguments.");
504  if(nlhs > 1) mexErrMsgTxt("To many output arguments.");
505 
506  boost::shared_ptr<Device> device = getDevice(prhs[pp_device]);
507 
508  if(!mxIsChar(prhs[pp_module])) mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_module) + " input argument.");
509  if(!mxIsChar(prhs[pp_register])) mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_register) + " input argument.");
510 
511  if((nrhs > pp_offset) && (!mxIsRealScalar(prhs[pp_offset]) || (mxGetScalar(prhs[pp_offset]) < 0)))
512  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_offset) + " input argument.");
513 
514  if((nrhs > pp_elements) && !mxIsPositiveRealScalar(prhs[pp_elements]))
515  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_elements) + " input argument.");
516 
517  // offset is optional. Use 0 if not set
518  const uint32_t offset = (nrhs > pp_offset) ? mxGetScalar(prhs[pp_offset]) : 0;
519 
520  // number of elements is optional. Use 0 (=all remaining) if not set
521  const uint32_t nElements = (nrhs > pp_elements) ? mxGetScalar(prhs[pp_elements]) : 0;
522 
523  RegisterPath moduleName(mxArrayToStdString(prhs[pp_module]));
524  RegisterPath registerName(mxArrayToStdString(prhs[pp_register]));
525  // we get the register content as a std::vector<double>
526  auto registerContent = device->read<double>(moduleName / registerName, nElements, offset);
527 
528  // as both DeviceAccess and Matlab do their own memory allocation all we can
529  // do is memcpy :-(
530  plhs[0] = mxCreateDoubleMatrix(1, registerContent.size(), mxREAL);
531  double* plhsValue = mxGetPr(plhs[0]);
532  memcpy(plhsValue, registerContent.data(), registerContent.size() * sizeof(double));
533 }

569  writeToDevice<uint8_t>(device, registerPath, data, prhsValueElements, offset);
570  }

573  }
574  else if(arrayType == mxINT16_CLASS) {
575  writeToDevice<int16_t>(device, registerPath, data, prhsValueElements, offset);
576  }
577  else if(arrayType == mxUINT16_CLASS) {

603 void readDmaRaw(unsigned int nlhs, mxArray* plhs[], unsigned int nrhs, const mxArray* prhs[]) {
604  static const unsigned int pp_device = 0, pp_module = 1, pp_register = 2, pp_offset = 3, pp_elements = 4;
605  static const unsigned int pp_mode = 5, pp_signed = 6, pp_bit = 7, pp_fracbit = 8;
606 
607  if(nrhs < 3) mexErrMsgTxt("Not enough input arguments.");
608  if(nrhs > 9) mexWarnMsgTxt("Too many input arguments.");
609  if(nlhs > 1) mexErrMsgTxt("Too many output arguments.");
610 
611  boost::shared_ptr<Device> device = getDevice(prhs[pp_device]);
612 
613  if(!mxIsChar(prhs[pp_module])) mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_module) + " input argument.");
614  if(!mxIsChar(prhs[pp_register])) mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_register) + " input argument.");
615 
616  if((nrhs > pp_offset) && !mxIsRealScalar(prhs[pp_offset]))
617  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_offset) + " input argument.");
618  if((nrhs > pp_elements) && !mxIsPositiveRealScalar(prhs[pp_elements]))
619  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_elements) + " input argument.");
620 
621  if((nrhs > pp_mode) && !mxIsPositiveRealScalar(prhs[pp_mode]))
622  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_mode) + " input argument.");
623 
624  // FIXME: These don't make sense in raw mode. Shall we just skip the checks?
625  // They are not used anyway.
626  if((nrhs > pp_signed) && !mxIsRealScalar(prhs[pp_signed]))
627  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_signed) + " input argument.");
628  if((nrhs > pp_bit) && !mxIsPositiveRealScalar(prhs[pp_bit]))
629  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_bit) + " input argument.");
630  if((nrhs > pp_fracbit) && !mxIsRealScalar(prhs[pp_fracbit]))
631  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_fracbit) + " input argument.");
632 
633  // offset is optional. Use 0 if not set
634  const uint32_t offset = (nrhs > pp_offset) ? mxGetScalar(prhs[pp_offset]) : 0;
635  // number of elements is optional. Use 0 (=all remaining) if not set
636  const uint32_t nElements = (nrhs > pp_elements) ? mxGetScalar(prhs[pp_elements]) : 0;
637 
638  // The mode can be 16 or 32 bit. This does not make sense because the real raw
639  // stuff is 32 bits, it was there so we keep it in order not to break
640  // compatibility.
641  const uint32_t mode = (nrhs > pp_mode) ? mxGetScalar(prhs[pp_mode]) : 32;
642  if((mode != 32) && (mode != 16)) mexErrMsgTxt("Invalid data mode.");
643 
644  // Number of 32 bit words to read through the "bus". This currently is the
645  // only mode the raw accessor knows
646  uint32_t nWords32Bit = nElements;
647  if(mode == 16) {
648  // The number of 32 bit words is only half the number of elements if the are
649  // requested as 16 bit
650  nWords32Bit /= 2;
651  }
652 
653  // Notice: signedFlags, bits and fracBits have been removed as they were
654  // not/cannot be used anyway
655 
656  // Now that we have all parameters it's time to read the data from the device.
657  RegisterPath registerPath(mxArrayToStdString(prhs[pp_module]) + "/" + mxArrayToStdString(prhs[pp_register]));
658 
659  // we get the register content as a std::vector<double>
660  auto accessor = device->getOneDRegisterAccessor<int32_t>(registerPath, nWords32Bit, offset, {AccessMode::raw});
661  accessor.read();
662 
663  // frame matlab buffer with appropriate number of elements
664  auto bufferSize = (mode == 16) ? accessor.getNElements() * 2 : accessor.getNElements();
665  plhs[0] = mxCreateDoubleMatrix(bufferSize, 1, mxREAL);
666 
667  double* plhsValue = mxGetPr(plhs[0]);
668 
669  if(mode == 32) {
670  size_t i = 0;
671  // we can directly loop the accessor
672  for(auto value : accessor) {
673  plhsValue[i++] = value;
674  }
675  }
676  else {
677  auto values16Bit = reinterpret_cast<int16_t*>(accessor.data());
678  for(size_t i = 0; i < bufferSize; ++i) {
679  plhsValue[i] = values16Bit[i];
680  }
681  }
682 }

710  if((nlhs > 1) && (nrhs > pp_channel) && (nlhs < mxGetNumberOfElements(prhs[pp_channel])))

716  mexErrMsgTxt("Invalid " + getOrdinalNumerString(pp_elements) + " input argument.");
717 
718  auto twoDRegister = device->getTwoDRegisterAccessor<double>(
719  mxArrayToStdString(prhs[pp_module]) + "/" + mxArrayToStdString(prhs[pp_register]));
720  twoDRegister.read();
721 
722  const uint32_t offset = (nrhs > pp_offset) ? mxGetScalar(prhs[pp_offset]) : 0;
723  const uint32_t elements = (nrhs > pp_elements) ? mxGetScalar(prhs[pp_elements]) : (twoDRegister[0].size());
724 
725  const uint32_t totalChannels = twoDRegister.getNChannels();
726 
727  /*if (mxGetScalar(prhs[pp_channel]) > totalChannels)
728  mexErrMsgTxt("Requested Channel Index greater than available channels");*/
729 
730  const uint32_t size = elements * totalChannels;
731  std::vector<double> dmaValue(size);
732 
733  unsigned int selectedChannels = (nrhs > pp_channel) ? mxGetNumberOfElements(prhs[pp_channel]) : totalChannels;
734  // Store data in different vectors passed over lhs
735  if(nlhs == selectedChannels) {
736  for(unsigned int ic = 0; ic < nlhs; ic++) {
737  plhs[ic] = mxCreateDoubleMatrix(elements, 1, mxREAL);
738  double* plhsValue = mxGetPr(plhs[ic]);
739  unsigned int currentChannel = (nrhs > pp_channel) ? (mxGetPr(prhs[pp_channel])[ic] - 1) : ic;
740 
741  for(unsigned int is = 0; is < elements; is++) {
742  unsigned int seqIndex = (is + offset);
743  if(seqIndex >= size) mexErrMsgTxt("Data storage indexing went wrong!");
744  plhsValue[is] = twoDRegister[currentChannel][seqIndex];
745  }
746  }
747  }
748  // Store data in lhs matrix
749  else {
750  plhs[0] = mxCreateDoubleMatrix(elements, selectedChannels, mxREAL);
751  double* plhsValue = mxGetPr(plhs[0]);
752 
753  // copy and transform data
754  for(unsigned int ic = 0; ic < selectedChannels; ic++) {
755  int currentChannel = (nrhs > pp_channel) ? (mxGetPr(prhs[pp_channel])[ic] - 1) : ic;
756 
757  if(currentChannel < 0 || uint32_t(currentChannel) >= totalChannels) mexErrMsgTxt("Illegal Channel Index");
758 
759  for(unsigned int is = 0; is < elements; is++) {
760  unsigned int plhsIndex = ic * elements + is;
761  unsigned int seqIndex = (is + offset);
762 
763  if(seqIndex >= size) mexErrMsgTxt("Data storage indexing went wrong! (1)");
764  if(plhsIndex >= elements * selectedChannels) mexErrMsgTxt("Data storage indexing went wrong! (2)");
765 
766  plhsValue[plhsIndex] = twoDRegister[currentChannel][seqIndex];
767  }
768  }
769  }
