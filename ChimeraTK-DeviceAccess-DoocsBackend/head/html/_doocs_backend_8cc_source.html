56static DoocsBackendRegisterCatalogue fetchCatalogue(
57 std::string serverAddress, std::string cacheFile, std::future<void> cancelFlag) {
58 auto result = CatalogueFetcher(serverAddress, std::move(cancelFlag)).fetch();
59 auto catalogue = std::move(result.first);
60 auto isCatalogueComplete = result.second;
61 bool isCacheFileNameSpecified = not cacheFile.empty();
62
63 if(isCatalogueComplete && isCacheFileNameSpecified) {
64 Cache::saveCatalogue(catalogue, cacheFile);
65 }
66 return catalogue;
67}

83 DoocsBackend::DoocsBackend(const std::string& serverAddress, const std::string& cacheFile,
84 const std::string& updateCache, const std::string& dataConsistencyRealmName)
85 : _serverAddress(serverAddress), _cacheFile(cacheFile) {
86 if(cacheFileExists() && isCachingEnabled()) {
87 // provide catalogue immediately from cache
88 catalogue = Cache::readCatalogue(_cacheFile);
89
90 // update cache file in the background
91 if(updateCache == "1") {
92 std::thread(fetchCatalogue, serverAddress, cacheFile, _cancelFlag.get_future()).detach();
93 }
94 }
95 else {
96 // fill catalogue in the background (and save to cache if enabled)
97 _catalogueFuture =
98 std::async(std::launch::async, fetchCatalogue, serverAddress, cacheFile, _cancelFlag.get_future());
99 }
100
101 // Reduce ZeroMQ timeout so inconsistencies get corrected more quickly. The downside is that DOOCS will do more
102 // frequent RPC polls on rarely changing ZeroMQ variables (every 10 seconds instead of every 4 minutes), but this
103 // is acceptable as it is still slow enough.
104 doocs::zmq_set_subscription_timeout(10);
105
106 _dataConsistencyRealm = async::DataConsistencyRealmStore::getInstance().getRealm(dataConsistencyRealmName);
107
108 FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(getRegisterAccessor_impl);
109 }

143 boost::shared_ptr<DeviceBackend> DoocsBackend::createInstance(
144 std::string address, std::map<std::string, std::string> parameters) {
145 // if address is empty, build it from parameters (for compatibility with SDM)
146 if(address.empty()) {
147 RegisterPath serverAddress;
148 serverAddress /= parameters["facility"];
149 serverAddress /= parameters["device"];
150 serverAddress /= parameters["location"];
151 address = std::string(serverAddress).substr(1);
152 }
153 std::string cacheFile{};
154 std::string updateCache{"0"};
155 try {
156 cacheFile = parameters.at("cacheFile");
157 updateCache = parameters.at("updateCache");
158 }
159 catch(std::out_of_range&) {
160 // empty cacheFile string => no caching
161 // empty updateCache string => no cache update
162 }
163
164 std::string dataConsistencyRealmName{"doocsEventId"};
165 if(parameters.find("location") != parameters.end()) {
166 dataConsistencyRealmName = parameters.at("location");
167 }
168
169 // create and return the backend
170 return boost::shared_ptr<DeviceBackend>(
171 new DoocsBackend(address, cacheFile, updateCache, dataConsistencyRealmName));
172 }

176 void DoocsBackend::open() {
177 std::unique_lock<std::mutex> lk(_mxRecovery);
178 if(lastFailedAddress != "") {
179 // open() is called after a runtime_error: check if device is recovered.
180 doocs::EqAdr ea;
181 ea.adr(lastFailedAddress);
182 EqCall eq;
183 doocs::EqData src, dst;
184 int rc = eq.get(&ea, &src, &dst);
185 // if again error received, throw exception
186 if(rc && isCommunicationError(dst.error())) {
187 lk.unlock();
188 auto message = std::string("Cannot read from DOOCS property: ") + dst.get_string();
189 setException(message);
190 throw ChimeraTK::runtime_error(message);
191 }
192 lastFailedAddress = "";
193 }
194 _startVersion = {};
195 setOpenedAndClearException();
196 }

254 template<typename UserType>
255 boost::shared_ptr<NDRegisterAccessor<UserType>> DoocsBackend::getRegisterAccessor_impl(
256 const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
257 boost::shared_ptr<NDRegisterAccessor<UserType>> p;
258 std::string path = _serverAddress + registerPathName;
259
260 // check for additional hierarchy level, which indicates an access to a field of a complex property data type
261 bool hasExtraLevel = false;
262 if(!boost::starts_with(path, "doocs://") && !boost::starts_with(path, "epics://")) {
263 size_t nSlashes = std::count(path.begin(), path.end(), '/');
264 if(nSlashes == 4) {
265 hasExtraLevel = true;
266 }
267 else if(nSlashes < 3 || nSlashes > 4) {
268 throw ChimeraTK::logic_error(std::string("DOOCS address has an illegal format: ") + path);
269 }
270 }
271 else if(boost::starts_with(path, "doocs://")) {
272 size_t nSlashes = std::count(path.begin(), path.end(), '/');
273 // we have 3 extra slashes compared to the standard syntax without "doocs:://"
274 if(nSlashes == 4 + 3) {
275 hasExtraLevel = true;
276 }
277 else if(nSlashes < 3 + 3 || nSlashes > 4 + 3) {
278 throw ChimeraTK::logic_error(std::string("DOOCS address has an illegal format: ") + path);
279 }
280 }
281
282 // split the path into property name and field name
283 std::string field;
284 if(hasExtraLevel) {
285 field = path.substr(path.find_last_of('/') + 1);
286 path = path.substr(0, path.find_last_of('/'));
287 }
288
289 // if backend is open, read property once to obtain type
290 int doocsTypeId = DATA_NULL;
291 if(isOpen()) {
292 doocs::EqAdr ea;
293 EqCall eq;
294 doocs::EqData src, dst;
295 ea.adr(path);
296 int rc = eq.get(&ea, &src, &dst);
297 if(rc) {
298 if(rc == eq_errors::ill_property || rc == eq_errors::ill_location ||
299 rc == eq_errors::ill_address) { // no property by that name
300 throw ChimeraTK::logic_error("Property does not exist: " + path + "': " + dst.get_string());
301 }
302 // runtime errors are thrown later
303 }
304 else {
305 doocsTypeId = dst.type();
306 }
307 }
308
309 // if backend is closed, or if property could not be read, use the (potentially cached) catalogue
310 if(doocsTypeId == DATA_NULL) {
311 auto reg = getBackendRegisterCatalogue().getBackendRegister(registerPathName);
312 doocsTypeId = reg.doocsTypeId;
313 }
314
315 // check type and create matching accessor
316 bool extraLevelUsed = false;
317 auto sharedThis = boost::static_pointer_cast<DoocsBackend>(shared_from_this());
318
319 if(field == "eventId") {
320 extraLevelUsed = true;
321 p.reset(new DoocsBackendEventIdRegisterAccessor<UserType>(sharedThis, path, registerPathName, flags));
322 }
323 else if(field == "timeStamp") {
324 extraLevelUsed = true;
325 p.reset(new DoocsBackendTimeStampRegisterAccessor<UserType>(sharedThis, path, registerPathName, flags));
326 }
327 else {
328 switch(doocsTypeId) {
329 case DATA_BOOL:
330 case DATA_A_BOOL:
331 case DATA_SHORT:
332 case DATA_A_SHORT:
333 case DATA_USHORT:
334 case DATA_A_USHORT:
335 case DATA_INT:
336 case DATA_A_INT:
337 case DATA_UINT:
338 case DATA_A_UINT:
339 case DATA_LONG:
340 case DATA_A_LONG:
341 case DATA_ULONG:
342 case DATA_A_ULONG:
343 case DATA_FLOAT:
344 case DATA_SPECTRUM:
345 case DATA_GSPECTRUM:
346 case DATA_A_FLOAT:
347 case DATA_DOUBLE:
348 case DATA_A_DOUBLE:
349 p.reset(new DoocsBackendNumericRegisterAccessor<UserType>(
350 sharedThis, path, registerPathName, numberOfWords, wordOffsetInRegister, flags));
351 break;
352
353 case DATA_IIII:
354 p.reset(new DoocsBackendIIIIRegisterAccessor<UserType>(
355 sharedThis, path, registerPathName, numberOfWords, wordOffsetInRegister, flags));
356 break;
357
358 case DATA_IFFF:
359 if(!hasExtraLevel) {
360 throw ChimeraTK::logic_error("DOOCS property of IFFF type '" + _serverAddress + registerPathName +
361 "' cannot be accessed as a whole.");
362 }
363 extraLevelUsed = true;
364 p.reset(new DoocsBackendIFFFRegisterAccessor<UserType>(
365 sharedThis, path, field, registerPathName, numberOfWords, wordOffsetInRegister, flags));
366 break;
367
368 case DATA_TEXT:
369 case DATA_STRING:
370 p.reset(new DoocsBackendStringRegisterAccessor<UserType>(
371 sharedThis, path, registerPathName, numberOfWords, wordOffsetInRegister, flags));
372 break;
373
374 case DATA_IMAGE: {
375 auto accImpl = new DoocsBackendImageRegisterAccessor(
376 sharedThis, path, registerPathName, numberOfWords, wordOffsetInRegister, flags);
377 if constexpr(std::is_same_v<UserType, std::uint8_t>) {
378 p.reset(accImpl);
379 }
380 else {
381 boost::shared_ptr<NDRegisterAccessor<std::uint8_t>> pImpl(accImpl);
382 // any UserType can hold uint8
383 boost::shared_ptr<NDRegisterAccessor<UserType>> accDecorated(
384 new TypeChangingRangeCheckingDecorator<UserType, std::uint8_t>(
385 boost::dynamic_pointer_cast<ChimeraTK::NDRegisterAccessor<std::uint8_t>>(pImpl)));
386 p = accDecorated;
387 }
388 break;
389 }
390
391 default:
392 throw ChimeraTK::logic_error("Unsupported DOOCS data type " +
393 std::string(doocs::EqData().type_string(doocsTypeId)) + " of property '" + _serverAddress +
394 registerPathName + "'");
395 }
396 }
397
398 // if the field name has been specified but the data type does not use it, throw an exception
399 if(hasExtraLevel && !extraLevelUsed) {
400 throw ChimeraTK::logic_error("Specifiaction of field name is not supported for the DOOCS data type " +
401 std::string(doocs::EqData().type_string(doocsTypeId)) + ": " + _serverAddress + registerPathName);
402 }
403
404 p->setExceptionBackend(shared_from_this());
405 return p;
406 }

409 bool DoocsBackend::isCommunicationError(int doocs_error) {
410 // logic_error-like errors are caught at a different place. If such error appears later, a runtime_error need to
411 // be generated (device does not behave as expected, as it is not expected to change)
412 switch(doocs_error) {
413 case eq_errors::unsup_domain:
414 case eq_errors::no_ens_entry:
415 case eq_errors::ill_monitor:
416 case eq_errors::faulty_chans:
417 case eq_errors::unavail_serv:
418 case eq_errors::ill_serv:
419 case eq_errors::rpc_prot_error:
420 case eq_errors::ens_fault:
421 case eq_errors::ill_protocol:
422 case eq_errors::ill_location:
423 case eq_errors::ill_property:
424 case eq_errors::no_connection:
425 case eq_errors::conn_timeout:
426 case eq_errors::alias_error:
427 case eq_errors::no_permission:
428 return true;
429 default:
430 return false;
431 }
432 }
