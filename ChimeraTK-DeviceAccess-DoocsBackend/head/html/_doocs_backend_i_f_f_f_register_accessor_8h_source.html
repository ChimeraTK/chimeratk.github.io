52 template<typename UserType>
53 DoocsBackendIFFFRegisterAccessor<UserType>::DoocsBackendIFFFRegisterAccessor(boost::shared_ptr<DoocsBackend> backend,
54 const std::string& path, const std::string& fieldName, const std::string& registerPathName, size_t numberOfWords,
55 size_t wordOffsetInRegister, AccessModeFlags flags)
56 : DoocsBackendRegisterAccessor<UserType>(
57 backend, path, registerPathName, numberOfWords, wordOffsetInRegister, flags) {
58 try {
59 // number of words and offset must be at fixed values
60 if(numberOfWords > 1 || wordOffsetInRegister != 0) {
61 throw ChimeraTK::logic_error("Register '" + this->getName() + "' is scalar.");
62 }
63 this->nElements = 1; // needed since DOOCS reports a length of 4
64
65 // determine field
66 if(fieldName == "I") {
67 field = Field::I;
68 }
69 else if(fieldName == "F1") {
70 field = Field::F1;
71 }
72 else if(fieldName == "F2") {
73 field = Field::F2;
74 }
75 else if(fieldName == "F3") {
76 field = Field::F3;
77 }
78 else {
79 throw ChimeraTK::logic_error("Unknown field name '" + fieldName + "' for DOOCS IFFF data type.");
80 }
81
82 // check data type
83 if(src.type() != DATA_IFFF) {
84 throw ChimeraTK::logic_error("DOOCS data type " + std::to_string(src.type()) +
85 " not supported by DoocsBackendIFFFRegisterAccessor."); // LCOV_EXCL_LINE (already prevented in the Backend)
86 }
87 }
88 catch(...) {
89 this->shutdown();
90 throw;
91 }
92 }

103 template<typename UserType>
104 void DoocsBackendIFFFRegisterAccessor<UserType>::doPostRead(TransferType type, bool hasNewData) {
105 DoocsBackendRegisterAccessor<UserType>::doPostRead(type, hasNewData);
106 if(!hasNewData) return;
107
108 // copy data into our buffer
109 IFFF* data = dst.get_ifff();
110 switch(field) {
111 case Field::I: {
112 buffer_2D[0][0] = numericToUserType<UserType>(data->i1_data);
113 break;
114 }
115 case Field::F1: {
116 buffer_2D[0][0] = numericToUserType<UserType>(data->f1_data);
117 break;
118 }
119 case Field::F2: {
120 buffer_2D[0][0] = numericToUserType<UserType>(data->f2_data);
121 break;
122 }
123 case Field::F3: {
124 buffer_2D[0][0] = numericToUserType<UserType>(data->f3_data);
125 break;
126 }
127 default: {
128 assert(false); // LCOV_EXCL_LINE (cannot happen, see constructor)
129 }
130 }
131 }

135 template<typename UserType>
136 void DoocsBackendIFFFRegisterAccessor<UserType>::doPreWrite(TransferType type, VersionNumber version) {
137 DoocsBackendRegisterAccessor<UserType>::doPreWrite(type, version);
138
139 // read-modify-write: first read
140 DoocsBackendRegisterAccessor<UserType>::doReadTransferSynchronously();
141
142 // now modify
143 IFFF* data = dst.get_ifff();
144 switch(field) {
145 case Field::I: {
146 data->i1_data = userTypeToNumeric<int>(buffer_2D[0][0]);
147 break;
148 }
149 case Field::F1: {
150 data->f1_data = userTypeToNumeric<float>(buffer_2D[0][0]);
151 break;
152 }
153 case Field::F2: {
154 data->f2_data = userTypeToNumeric<float>(buffer_2D[0][0]);
155 break;
156 }
157 case Field::F3: {
158 data->f3_data = userTypeToNumeric<float>(buffer_2D[0][0]);
159 break;
160 }
161 default: {
162 assert(false); // LCOV_EXCL_LINE (cannot happen, see constructor)
163 }
164 }
165
166 // prepare for writing
167 src.set(data);
168 }
