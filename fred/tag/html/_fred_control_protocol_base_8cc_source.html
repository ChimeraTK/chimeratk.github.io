87 FilteredResponse filterResponseString(const std::string& unfiltered) {
88 std::stringstream responseStream(unfiltered);
89 std::string inputLine;
90 std::string response = "";
91 std::vector<std::string> info;
92 std::vector<std::string> warning;
93 std::vector<std::string> error;
94
95 while(std::getline(responseStream, inputLine)) {
96 switch(linetype(inputLine)) {
97 case LineType::RESPONSE:
98 response = response + inputLine + "\n";
99 break;
100 case LineType::INFO:
101 info.push_back(inputLine);
102 break;
103 case LineType::WARNING:
104 warning.push_back(inputLine);
105 break;
106 case LineType::ERROR:
107 error.push_back(inputLine);
108 break;
109 case LineType::EMPTY:
110 case LineType::PROMPT:
111 default:
112 break;
113 }
114 }
115 return {response, info, warning, error};
116 }

118 /******************************************************************************/
119 /*
120 * @brief finds the line type of input string.
121 *
122 * Line starting with:
123 * "\r" -> considered to be empty.
124 * "# " -> considered to be a fred prompt.
125 * alphanumeric char -> is part of a command response.
126 * % -> is an info message.
127 * %% -> is a warning message.
128 * %%% -> is an error message.
129 */
130 LineType linetype(std::string const& line) {
131 if(line == EMPTY_LINE) {
132 return LineType::EMPTY;
133 }
134 else if(line.substr(0, 2) == FUSE_RELAY_BOARD_PROMPT) {
135 return LineType::PROMPT;
136 }
137 else if(line.substr(0, 3) == "%%%") {
138 return LineType::ERROR;
139 }
140 else if(line.substr(0, 2) == "%%") {
141 return LineType::WARNING;
142 }
143 else if(line.substr(0, 1) == "%") {
144 return LineType::INFO;
145 }
146 else {
147 return LineType::RESPONSE;
148 }
149 }

152 /*
153 Expected structure of the version string:.
154 Firmware: 3.2.2 CPLD 0x36 (54)
155 UUID: 00-1E-C0-AD-EC-81

161 VersionInfo getFredVersionInfo(std::string version) {

163
164 // getAtIndex is a safe guard against fred firmware versions that claim to be

201 /******************************************************************************/
202 /*
203 * @brief function extracts contents from a string after the specified
204 * delimiter.
205 *
206 * Behavior of the function is as below:
207 * s == "ab: c", delimiter == ": " -> returns "c"
208 * s == "ab: ", delimiter == ": " -> returns ""
209 * s == "ab: c", delimiter == "*" -> returns "ab: c"
210 */
211 std::string extractAfterDelimiter(std::string const& s, std::string const& delimiter) {
212 auto pos = s.find(delimiter);
213 if(pos == std::string::npos) {
214 return s;
215 }
216 auto start = pos + delimiter.size();
217 if(start >= s.size()) {
218 return "";
219 }
220 return s.substr(start);
221 }

266 // TODO: explore boost split
267 std::vector<std::string> result;
