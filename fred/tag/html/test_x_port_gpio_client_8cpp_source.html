126void XPortGpioClientTest::testCheckResponse() {
127 TestableXPortGpioClient::ReadBuffer readBuffer;
128 unsigned char command;
129
130 readBuffer[0] = command = 0xAB;
131
132 // if only one byte it read command has to be 0xFF, so we expect an
133 // CommunicationErrorException
134 BOOST_CHECK_THROW(_gpio.checkResponse(readBuffer, command, 1), CommunicationErrorException);
135
136 // if there are five bytes the command has to match, so we expect an exception
137 BOOST_CHECK_THROW(_gpio.checkResponse(readBuffer, command + 1, 5), CommunicationErrorException);
138
139 BOOST_CHECK_NO_THROW(_gpio.checkResponse(readBuffer, command, 5));
140
141 // in case of an error response (0xFF, 1 byte) we expect an
142 // IvalidCommandException
143 readBuffer[0] = 0xFF;
144 BOOST_CHECK_THROW(_gpio.checkResponse(readBuffer, command, 1), InvalidCommandException);
145}
