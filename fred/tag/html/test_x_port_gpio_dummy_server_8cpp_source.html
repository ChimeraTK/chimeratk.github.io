91void XPortGpioDummyServerTest::testGetSetActiveLowWord() {
92 // set some output pins on to test if they turn off as they should
93 _gpioDummyServer.setPins(0xCCCCCCCC, 0x0FF00FF0);
94 BOOST_CHECK(_gpioDummyServer.getPinsActiveWord() == 0x0C00FFC0);
95
96 _gpioDummyServer.setActiveLowWord(0xAAAAAAAA, 0x0FF00FF0);

99 // to 0
100 BOOST_CHECK(_gpioDummyServer.getPinsActiveWord() == 0x0000F500);
