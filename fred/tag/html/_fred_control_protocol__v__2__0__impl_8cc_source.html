250 std::array<float, 20> FredControlProtocol_v_2_0_impl::readTmcbPinVoltages() {
251 std::stringstream responseStream;
252 responseStream << processResponse(this, "TmcbReadPinVoltages");
253
254 std::string responseLine;
255 std::string inputLine;
256 bool parsingFailed = false;
257 std::array<float, 20> pinsVoltage;
258 int pinCounter = 0;
259 try {
260 while(std::getline(responseStream, inputLine)) {
261 std::vector<std::string> strs;
262 boost::split(strs, inputLine, boost::is_any_of("- "));
263 if(strs[0] != "Pins") {
264 parsingFailed = true;
265 break;
266 }
267 int pin1 = std::stoi(strs[1]);
268 int pin2 = std::stoi(strs[2]);
269 if(pin1 != ++pinCounter || pin2 != ++pinCounter) {
270 parsingFailed = true;
271 break;
272 }
273 std::string volt = strs[5];
274 boost::trim(volt);
275 if(volt.back() != 'V') {
276 parsingFailed = true;
277 break;
278 }
279 float voltage = std::stof(volt.substr(0, volt.size() - 1));
280 pinsVoltage[pinCounter - 2] = voltage;
281 pinsVoltage[pinCounter - 1] = voltage;
282 }
283 }
284 catch(std::exception& e) {
285 parsingFailed = true;
286 }
287 if(parsingFailed) {
288 throw BadResponseException("Error reading tmcb pins voltages");
289 }
290 return pinsVoltage;
291 }

364 ChannelStatus parseChannelInfo(std::string const& channelInfo) {
365 unsigned int channel;
366 unsigned int powerState;
367 int voltage;
368 unsigned int current;
369 char name[32]; // not safe.
370 if(sscanf(channelInfo.c_str(), "%1u %2x %6d %5u %s %*s", &channel, &powerState, &voltage, &current,
371 name) == 5) { // %*s; * tells sscanf to ignore the
372 // string specifier; Only the first 5
373 // elements of the input string is
374 // read the rest gets ignored.
375
376 name[31] = '\0';
377 // fixme: check for valid states (also range of voltages and currents?)
378 return {voltage, current, powerState, name};
379 }
380 else {
381 throw BadResponseException("Reading channel status failed.");
382 }
