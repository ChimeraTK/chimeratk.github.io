122 std::unique_ptr<FredControlProtocolBase>
123 getFredControlProtocolImpl(std::string const &serverName) {
124 const double MAX_SUPPORTED_VERSION = 2.4;
125 using namespace std;
126
127 auto connection = std::make_unique<MinimalTelnetConnection>(
128 serverName, FUSE_RELAY_BOARD_PROMPT, FUSE_RELAY_BOARD_PORT);
129
130 auto clientProtocolVersion = std::stod(getSerialProtocolVersion(connection));
131
132 bool isClientMajorVersionSupported =
133 (static_cast<int>(clientProtocolVersion) ==
134 static_cast<int>(MAX_SUPPORTED_VERSION));
135
136 if (isClientMajorVersionSupported) {
137 clientProtocolVersion = (clientProtocolVersion > MAX_SUPPORTED_VERSION)
138 ? MAX_SUPPORTED_VERSION
139 : clientProtocolVersion;
140 }
141
142 if (clientProtocolVersion == 2.0) {
143 return make_unique<FredControlProtocol_v_2_0_impl>(std::move(connection));
144 } else if (clientProtocolVersion == 2.1) {
145 return make_unique<FredControlProtocol_v_2_1_impl>(std::move(connection));
146 } else if (clientProtocolVersion == 2.2) {
147 return make_unique<FredControlProtocol_v_2_2_impl>(std::move(connection));
148 } else if (clientProtocolVersion == 2.3) {
149 return make_unique<FredControlProtocol_v_2_3_impl>(std::move(connection));
150 } else if (clientProtocolVersion == 2.4) {
151 return make_unique<FredControlProtocol_v_2_4_impl>(std::move(connection));
152 } else {
153 // TODO: move to an appropriate exception
154 throw runtime_error("Unsupported Fred serial protocol version: " +
155 std::to_string(clientProtocolVersion));
156 }
157 }
