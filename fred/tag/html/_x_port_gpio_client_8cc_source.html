27 void XPortGpioClient::connectIfSocketIsClosed() {
28 if(_socket.is_open()) {
29 return;
30 }
31
32 tcp::resolver myResolver(_ioService);
33 tcp::resolver::query myQuery(_serverName, xportgpio::GPIO_PORT);
34 tcp::resolver::iterator endpointIterator = myResolver.resolve(myQuery);
35 tcp::resolver::iterator end;
36
37 while(!_socket.is_open() && (endpointIterator != end)) {
38 _deadlineTimer.expires_from_now(_connectionTimeout);
39 _deadlineTimer.async_wait(boost::bind(&XPortGpioClient::closeSocketOnTimeout, this, _1));
40
41 _socket.async_connect(*endpointIterator++, boost::bind(&XPortGpioClient::checkAsynchronousConnect, this, _1));
42
43 // run both the timer and the connection request. the run() will return when
44 // both are finished.
45 _ioService.run();
46 _ioService.reset();
47 }
48
49 if(!_socket.is_open()) {
50 throw CommunicationErrorException("Could not connect to host.");
51 }

151 boost::array<unsigned char, 9> XPortGpioClient::prepareWriteBuffer(
152 unsigned char command, unsigned int parameter1, unsigned int parameter2) {
153 boost::array<unsigned char, 9> writeBuffer;
154 writeBuffer[0] = command;
155
156 // host to little endian 32 bit. A NOOP on intel CPUs, but allows compiling on
157 // big endian CPUS.
158 unsigned int parameter1AsLittleEndian = htole32(parameter1);
159 memcpy(writeBuffer.c_array() + 1, &parameter1AsLittleEndian, sizeof(unsigned int));
160
161 unsigned int parameter2AsLittleEndian = htole32(parameter2);
162 memcpy(writeBuffer.c_array() + 5, &parameter2AsLittleEndian, sizeof(unsigned int));

167 void XPortGpioClient::checkResponse(ReadBuffer const& readBuffer, unsigned char command, size_t nBytesRead) {
168 switch(nBytesRead) {
169 case 5:
170 // The nominal case.
171 // The first byte should be the command we send
172 if(readBuffer[0] == command) {
173 break;
174 }
175 else {
176 throw(CommunicationErrorException("Response does not start with the command that was send."));
177 }
178 case 1:
179 // Error detected by the XPort, the byte should be 0xFF in this case
180 if(readBuffer[0] == 0xFF) {
181 throw InvalidCommandException("XPort reported an invalid command");
182 }
183 else {
184 // We would like to drop through to the default: unknown communication
185 // error, but newer compilers complain about implicit fallthrough while
186 // older compilers complain about the unkown attribute [[fallthrough]],
187 // which was introduced only in C++17. So we repeat the one line here :-(
188 throw(CommunicationErrorException("Unidentified communication error."));
189 }
190 default:
191 throw(CommunicationErrorException("Unidentified communication error."));
192 }
193
194 } // checkResponse
