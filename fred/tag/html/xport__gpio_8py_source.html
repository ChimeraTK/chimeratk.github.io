93 def _send_command_and_get_response(self, command, stateword, pinmask):
94 '''
95 * The command has to be a byte with one of the known commands (e.g. SET_ACTIVE_LEVELS)
96 * The state word is a bit field which pins are for instance active or inactive.
97
98 Examples:
99 send_command_and_get_response(SET_ACTIVE_STATES,5) will set pins 0 and 2 active, and pin 1 inactive
100 send_command_and_get_response(SET_DIRECTIONS, 3) will set pins 0 and 1 output, pin 2 input
101 '''
102 #The byte sequence to be send to the XPORT. Allways 9 bytes
103 xport_command = bytearray(9) #initialises 9 bytes to 0
104 xport_command[0] = command
105 #Byte 1 contains the pin mask. We allways act in all 3 GPIO pins.
106 #The state word has to be correct for all pins.
107 xport_command[1]=pinmask
108 xport_command[5]=stateword
109 #all other bytes stay 0
110
111 self.sock.send(xport_command)
112
113 #The xport answers with 5 bytes
114 response=bytearray(self.sock.recv(5))
115
116 #check that the answer is valid
117 if not response[0] == command:
118 raise IOError('Bad response from XPORT when sending command for GPIO pins.')
119
120 return response[1]
