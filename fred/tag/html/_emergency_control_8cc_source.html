20 void EmergencyControl::executeCommunicationSequence(std::vector<unsigned int> pinMasks) {
21 // The sequence is always the same
22
23 try {
24 // 1) Send the GPIO configuration. As we are just connecting to the board
25 // it might not be configured.
26
27 // Set the pins to active low. Only do this for the EMERGENCY_PINS (in
28 // future applications there might be more pins with not known purpose). Do
29 // this before switching the pins to output to avoid a falling edge when
30 // they go to low (which is interpreted as going to high by the GPLD chip
31 // which is reading the pins).
32 _gpioClient.setActiveLowWord(emergencycontrol::EMERGENCY_PINS_MASK, emergencycontrol::EMERGENCY_PINS_MASK);
33
34 // Now switch the emergency pins to output mode
35 _gpioClient.setOutputPins(emergencycontrol::EMERGENCY_PINS_MASK, emergencycontrol::EMERGENCY_PINS_MASK);
36
37 // 2) Send the sequence of stated. The CPLD only acts on states.
38 // Setting inactive in the last step is needed to leave the pins in a default
39 // state where a single voltage change does not trigger any actions.
40 _gpioClient.setPins(0, emergencycontrol::EMERGENCY_PINS_MASK);
41 for(auto& pinMask : pinMasks) {
42 _gpioClient.setPins(pinMask, pinMask & emergencycontrol::EMERGENCY_PINS_MASK);
43 }
44 _gpioClient.setPins(0, emergencycontrol::EMERGENCY_PINS_MASK);
45
46 // switch the GPIO pins back to input so it does not cause voltage
47 // fluctuations on the pin in case the XPort dies (unfortunately this
48 // happens quite frequently)
49 _gpioClient.setOutputPins(0, emergencycontrol::EMERGENCY_PINS_MASK);
50
51 // 3) disconnect so others can access the emergency control of the board
52 _gpioClient.disconnect();
53 }
54 catch(fred::Exception& e) {
55 // close the connection before rethrowing
56 _gpioClient.disconnect();
57 throw;
58 }
59 }
