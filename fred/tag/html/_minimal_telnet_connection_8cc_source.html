116void MinimalTelnetConnection::handleTelnetSequence(const size_t& bytes_transferred) {
117 _cleanedReadBuffer.prepare(_readBuffer.size());
118 std::ostream cleanedStream(&_cleanedReadBuffer);
119
120 boost::asio::buffers_iterator<boost::asio::const_buffers_1> end = boost::asio::buffers_end(_readBuffer.data());
121 for(boost::asio::buffers_iterator<boost::asio::const_buffers_1> bufferIter =
122 boost::asio::buffers_begin(_readBuffer.data());
123 bufferIter < end;
124 ++bufferIter) { // FIXME?: This also seems to work: bufferIter !=
125 // boost::asio::buffers_begin(_readBuffer.data()) +
126 // bytes_transferred.
127 if(static_cast<unsigned char>(*bufferIter) != TELNET_IAC) {
128 // noting to do, write the char to the output stream
129 cleanedStream << *bufferIter;
130 continue;
131 }
132 // ok, it's a telnet command sequence with two or three characters
133 ++bufferIter; // let's look at the next character
134 if(bufferIter == end) {
135 break;
136 }
137
138 switch(static_cast<unsigned char>(*bufferIter)) {
139 // handle the negotiation requests first. We don't need a state machine
140 // because there is only one state: All options off, we will never acctept
141 // and never send a request.
142 case TELNET_DO: // deny by answering wont
143 ++bufferIter; // let's look at the next character
144 if(bufferIter == end) {
145 break;
146 }
147 telnetReply(TELNET_WONT, *bufferIter);
148 break;
149 case TELNET_WILL: // deny by answering dont
150 ++bufferIter; // let's look at the next character
151 if(bufferIter == end) {
152 break;
153 }
154 telnetReply(TELNET_DONT, *bufferIter);
155 break;
156 case TELNET_DONT:
157 case TELNET_WONT:
158 // everyting is off on our side: just ignore donts and wonts
159 ++bufferIter; // but increase the buffer iter so the third character is
160 // skipped.
161 break;
162
163 case TELNET_IAC:
164 // the excape sequence for the 0xFF data word. write it to the output
165 // stream
166 cleanedStream << TELNET_IAC;
167 break;
168 default:; // do nothing
169 } // switch
170
171 } // for bufferIter
172
173 // empty the read buffer
174 _readBuffer.consume(bytes_transferred);
175}

177void MinimalTelnetConnection::telnetReply(unsigned char command, unsigned char option) {

192 startWriting(command);

194

282void MinimalTelnetConnection::startSynchronisation() {
283 // start with sending <CR><LF> to get a fresh prompt for the next command
284 _sendBuffer.prepare(2);

286 std::ostream sendStream(&_sendBuffer);
287 sendStream << "\r\n";
