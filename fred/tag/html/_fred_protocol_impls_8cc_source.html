332void FredSerialProtocol_2_0::setHardwarePowerStates(unsigned char inputPowerStates) {
333 _hardwarePowerStates = 0;
334 for(unsigned int channel = 0; channel < N_RELAY_BOARD_CHANNELS; ++channel) {
335 unsigned char channelMask = static_cast<unsigned char>(1 << channel);
336
337 // check for alarm pins
338 if((_pin0AlarmMask & channelMask) && (_alarmPins & 0x1)) {
339 continue;
340 }
341 if((_pin1AlarmMask & channelMask) && (_alarmPins & 0x2)) {
342 continue;
343 }
344 if((_overcurrentAlarmMask & channelMask) &&
345 (_channelStatuses[channel].current > _channelStatuses[channel].currentLimit)) {
346 continue;
347 }
348
349 // no alarms on this channel. set the hardware power as requested
350 _hardwarePowerStates |= (channelMask & inputPowerStates);
351 }
352}

354void FredSerialProtocol_2_0::updateChannelStatuses() {
355 for(unsigned int channel = 0; channel < N_RELAY_BOARD_CHANNELS; ++channel) {
356 unsigned char channelMask = static_cast<unsigned char>(1 << channel);
357
358 // fill all bits of the statusWord
359 unsigned char newStatusWord = 0;
360 if(channelMask & _hardwarePowerStates) {
361 newStatusWord |= ON_OFF_BIT;
362 }
363 if((channelMask & _hardwarePowerStates) != (channelMask & _requestedPowerStates)) {
364 newStatusWord |= POWER_DIFFERS_FROM_SW_BIT;
365 }
366 if(abs(_channelStatuses[channel].voltage) < MINIMUM_VOLTAGE_FOR_POWER_MONITOR) {
367 newStatusWord |= POWER_MONITOR_ERROR_BIT;
368 }
369 if((channelMask & _pin0AlarmMask) && (_alarmPins & 0x1)) {
370 newStatusWord |= CHANNEL_ALARM_0_BIT;
371 }
372 if((channelMask & _pin1AlarmMask) && (_alarmPins & 0x2)) {
373 newStatusWord |= CHANNEL_ALARM_1_BIT;
374 }
375 if((_overcurrentAlarmMask & channelMask) &&
376 (_channelStatuses[channel].current > _channelStatuses[channel].currentLimit)) {
377 newStatusWord |= OVERCURRENT_BIT;
378 }
379
380 _channelStatuses[channel].statusWord = newStatusWord;
381 }

940 // after 5 reboots the watchdog info is reset (in reality eg. by a fred reboot)
941 if(_xportQuery == 25) {
942 _xportQuery = 0;
943 }
944 // integer divison. Increases by 1 every 5 queries
945 _pwrCycle = _xportQuery / 5;
946 if(_pwrCycle == 0) {
947 _timeSinceLastPwrCycle = -1;
948 }
949 else {
950 _timeSinceLastPwrCycle = (_xportQuery % 5) * 10;

953 std::stringstream response;
954 response << "pwrCycles timeSinceLastPwrCycle timeout \r\n";
955 response << (unsigned int)_pwrCycle << " " << std::setw(2) << (int64_t)_timeSinceLastPwrCycle << std::setw(6)
956 << (unsigned int)_timeout;
957 return response.str();
