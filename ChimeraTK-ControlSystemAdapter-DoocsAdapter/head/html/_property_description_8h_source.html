16 struct PersistConfig {
17 enum {
18 OFF = 0,
19 ON = 1, // write long arrays to separate files below hist/, short arrays to config file
20 AUTO = 2 // default DOOCS behaviour: persist only short arrays, in config file
21 };
22 int val;
23
24 // NOLINTNEXTLINE(google-explicit-constructor)
25 PersistConfig(std::string txt) {
26 transform(txt.begin(), txt.end(), txt.begin(), ::tolower);
27 if(txt == "false" or txt == "off" or txt == "0") {
28 val = OFF;
29 }
30 else if(txt == "true" or txt == "on" or txt == "1") {
31 val = ON;
32 }
33 else if(txt == "auto") {
34 val = AUTO;
35 }
36 else {
37 throw std::invalid_argument(std::string("Error parsing xml file: invalid input for PersistConfig: ") + txt);
38 }
39 }
40 // NOLINTNEXTLINE(google-explicit-constructor)
41 PersistConfig(int enumVal) { this->val = enumVal; }
42 bool operator==(const PersistConfig& other) const { return val == other.val; }
43 };

93 struct AutoPropertyDescription : public PropertyDescription {
94 enum class DataType { Byte, Short, Int, Long, Float, Double, Bool, Void, Auto };
95 ChimeraTK::RegisterPath source;
96 explicit AutoPropertyDescription(ChimeraTK::RegisterPath const& source_ = "", std::string location_ = "",
97 std::string name_ = "", DataType dataType_ = DataType::Auto, bool hasHistory_ = true, bool isWriteable_ = true)
98 : PropertyDescription(std::move(location_), std::move(name_), PropertyAttributes(hasHistory_, isWriteable_)),
99 source(source_), dataType(dataType_) {}
100 bool operator==(PropertyDescription const& other) const override {
101 if(typeid(other) == typeid(AutoPropertyDescription)) {
102 auto casted_other = dynamic_cast<AutoPropertyDescription const&>(other);
103 return dataType == casted_other.dataType && source == casted_other.source && location == other.location &&
104 name == other.name && static_cast<const PropertyAttributes*>(this)->operator==(casted_other);
105 }
106 return false;
107 }
108 void print(std::ostream& os = std::cout) const override {
109 os << source << " -> " << location << " / " << name << std::endl;
110 }
111
112 void deriveType(std::type_info const& info) {
113 if(info == typeid(uint8_t) || info == typeid(int8_t)) {
114 dataType = DataType::Byte;
115 }
116 if(info == typeid(uint16_t) || info == typeid(int16_t)) {
117 dataType = DataType::Short;
118 }
119 if(info == typeid(uint32_t) || info == typeid(int32_t)) {
120 dataType = DataType::Int;
121 }
122 if(info == typeid(uint64_t) || info == typeid(int64_t)) {
123 dataType = DataType::Long;
124 }
125 if(info == typeid(float)) {
126 dataType = DataType::Float;
127 }
128 if(info == typeid(double)) {
129 dataType = DataType::Double;
130 }
131 if(info == typeid(ChimeraTK::Boolean)) {
132 dataType = DataType::Bool;
133 }
134 if(info == typeid(ChimeraTK::Void)) {
135 dataType = DataType::Void;
136 }
137 }
138
139 std::set<std::string> getSources() override {
140 auto baseSet = PropertyDescription::getSources();
141 baseSet.insert({source});
142 return baseSet;
143 };
144
145 DataType dataType;
146 };
