19 bool PropertyBase::updateConsistency(const TransferElementID& updatedId) {
20 // FIXME: A first implementation is checking the data consistency here. Later this should be
21 // before calling this function because calling this function through a function pointer is
22 // comparatively expensive.
23
24 // Do not check if update is coming from another DOOCS property mapped to the same variable (ID invalid), since
25 // the check would never pass. Such variables cannot use exact data matching anyway, since the update is triggered
26 // from the DOOCS write to the other property.
27 // Also do not check, if data matching turned off
28 if(!updatedId.isValid() || _consistencyGroup.getMatchingMode() == DataConsistencyGroup::MatchingMode::none) {
29 _doocsSuccessfullyUpdated = true;
30 return true;
31 }
32 assert(_outputVarForVersionNum);
33 TransferElementID compareTo = _outputVarForVersionNum->getId();
34
35 if(!_consistencyGroup.update(updatedId)) {
36 // data is not consistent (yet). Don't update the Doocs buffer.
37 // check if this will now throw away data and generate a warning
38 if(updatedId == compareTo) {
39 if(!_doocsSuccessfullyUpdated) {
40 ++_nDataLossWarnings;
41 if(DoocsAdapter::checkPrintDataLossWarning(_nDataLossWarnings)) {
42 const char* reason = (bool)(_macroPulseNumberSource) ?
43 " due to failed data matching between value and macro pulse number" :
44 " due to failed data matching between values";
45
46 std::cout << "WARNING: Data loss in property " << getEqFct()->name() << "/" << _doocsPropertyName << reason
47 << " (repeated " << _nDataLossWarnings << " times)." << std::endl;
48 }
49 }
50 // Tracking that data is modified is only done for the "compareTo" variable.
51 // If we want to detect data loss on all variables in the consistency group, we would have to track this for
52 // each variable separately.
53 _doocsSuccessfullyUpdated = false;
54 }
55 return false;
56 }
57 _doocsSuccessfullyUpdated = true;
58 return true;
59 }

67 doocs::Timestamp PropertyBase::correctDoocsTimestamp() {
68 doocs::Timestamp timestamp = getTimestamp();
69 auto* d_fct = getDfct();
70 // Make sure we never send out two absolute identical time stamps. If we would do so, the "watchdog" which
71 // corrects inconsistencies in ZeroMQ subscriptions between sender and subscriber cannot detect the inconsistency.
72 // This check also enforces that the timestamp is always increasing (never going backwards in time)
73 if(timestamp <= d_fct->get_timestamp()) {
74 timestamp = d_fct->get_timestamp() + std::chrono::microseconds(1);
75 }
76
77 // update global time stamp of DOOCS, but only if our time stamp is newer
78 if(get_global_timestamp() < timestamp) {
79 set_global_timestamp(timestamp);
80 }
81 d_fct->set_timestamp(timestamp);
82 return timestamp;
83 }

85 void PropertyBase::sendZMQ(doocs::Timestamp timestamp) {
86 auto* d_fct = getDfct();
87 // send data via ZeroMQ if enabled and if DOOCS initialisation is complete
88 if(_publishZMQ && ChimeraTK::DoocsAdapter::isInitialised) {
89 dmsg_info info{};
90 auto sinceEpoch = timestamp.get_seconds_and_microseconds_since_epoch();
91 info.sec = sinceEpoch.seconds;
92 info.usec = sinceEpoch.microseconds;
93 if(_macroPulseNumberSource != nullptr) {
94 info.ident = _macroPulseNumberSource->accessData(0);
95 }
96 else {
97 info.ident = 0;
98 }
99 dmsg_error(&info, d_fct->d_error());
100 auto ret = d_fct->send(&info);
101 if(ret) {
102 std::cout << "ZeroMQ sending failed!!!" << std::endl;
103 }
104 }

107 void PropertyBase::updateOthers(bool handleLocking) {
108 if(handleLocking) {
109 getEqFct()->unlock();
110 }
111 for(const auto& weakProp : otherPropertiesToUpdate) {
112 auto prop = weakProp.lock();
113 if(!prop) {
114 // property went away, could happen in shutdown phase
115 continue;
116 }
117 if(handleLocking) {
118 prop->getEqFct()->lock();
119 }
120 prop->updateDoocsBuffer({});
121 if(handleLocking) {
122 prop->getEqFct()->unlock();
123 }
124 }
125 if(handleLocking) {
126 getEqFct()->lock();
127 }
128 }

130 void PropertyBase::setMacroPulseNumberSource(
131 const boost::shared_ptr<ChimeraTK::NDRegisterAccessor<int64_t>>& macroPulseNumberSource) {
132 _macroPulseNumberSource = macroPulseNumberSource;
133 if(_consistencyGroup.getMatchingMode() != DataConsistencyGroup::MatchingMode::none) {
134 registerVariable(_macroPulseNumberSource);
135 }
136 else {
137 // We don't need to match up anything with it when it changes, but we have to register this at least once
138 // so the macropulse number will be included in the readAnyGroup in the updater if
139 // <data_matching> is none everywhere
140 _doocsUpdater.addVariable(
141 ChimeraTK::ScalarRegisterAccessor<int64_t>(macroPulseNumberSource), getEqFct(), []() {});
142 }
143 }
