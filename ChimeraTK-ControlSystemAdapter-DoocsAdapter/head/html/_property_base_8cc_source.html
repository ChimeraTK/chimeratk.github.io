28 bool PropertyBase::updateConsistency(const TransferElementID& updatedId) {
29 // FIXME: A first implementation is checking the data consistency here. Later this should be
30 // before calling this function because calling this function through a function pointer is
31 // comparatively expensive.
32
33 // Do not check if update is coming from another DOOCS property mapped to the same variable (ID invalid), since
34 // the check would never pass. Such variables cannot use exact data matching anyway, since the update is triggered
35 // from the DOOCS write to the other property.
36 // Also do not check, if data matching turned off
37 if(!updatedId.isValid() || _consistencyGroup.getMatchingMode() == DataConsistencyGroup::MatchingMode::none) {
38 _doocsSuccessfullyUpdated = true;
39 return true;
40 }
41 assert(_outputVarForVersionNum);
42 TransferElementID compareTo = _outputVarForVersionNum->getId();
43
44 if(!_consistencyGroup.update(updatedId)) {
45 // data is not consistent (yet). Don't update the Doocs buffer.
46 // check if this will now throw away data and generate a warning
47 if(updatedId == compareTo) {
48 if(!_doocsSuccessfullyUpdated) {
49 ++_nDataLossWarnings;
50 if(DoocsAdapter::checkPrintDataLossWarning(_nDataLossWarnings)) {
51 const char* reason = (bool)(_macroPulseNumberSource) ?
52 " due to failed data matching between value and macro pulse number" :
53 " due to failed data matching between values";
54
55 std::cout << "WARNING: Data loss in property " << getEqFct()->name() << "/" << _doocsPropertyName << reason
56 << " (repeated " << _nDataLossWarnings << " times)." << std::endl;
57 }
58 }
59 // Tracking that data is modified is only done for the "compareTo" variable.
60 // If we want to detect data loss on all variables in the consistency group, we would have to track this for
61 // each variable separately.
62 _doocsSuccessfullyUpdated = false;
63 }
64 return false;
65 }
66 _doocsSuccessfullyUpdated = true;
67 return true;
68 }

76 doocs::Timestamp PropertyBase::correctDoocsTimestamp() {
77 doocs::Timestamp timestamp = getTimestamp();
78 auto* d_fct = getDfct();
79 // Make sure we never send out two absolute identical time stamps. If we would do so, the "watchdog" which
80 // corrects inconsistencies in ZeroMQ subscriptions between sender and subscriber cannot detect the inconsistency.
81 // This check also enforces that the timestamp is always increasing (never going backwards in time)
82 if(timestamp <= d_fct->get_timestamp()) {
83 timestamp = d_fct->get_timestamp() + std::chrono::microseconds(1);
84 }
85
86 // update global time stamp of DOOCS, but only if our time stamp is newer
87 if(get_global_timestamp() < timestamp) {
88 set_global_timestamp(timestamp);
89 }
90 d_fct->set_timestamp(timestamp);
91 return timestamp;
92 }

94 void PropertyBase::sendZMQ(doocs::Timestamp timestamp) {
95 auto* d_fct = getDfct();
96 // send data via ZeroMQ if enabled and if DOOCS initialisation is complete
97 if(_publishZMQ && ChimeraTK::DoocsAdapter::isInitialised) {
98 dmsg_info info{};
99 auto sinceEpoch = timestamp.get_seconds_and_microseconds_since_epoch();
100 info.sec = sinceEpoch.seconds;
101 info.usec = sinceEpoch.microseconds;
102 if(_macroPulseNumberSource.isInitialised()) {
103 info.ident = _macroPulseNumberSource;
104 }
105 else {
106 info.ident = 0;
107 }
108 dmsg_error(&info, d_fct->d_error());
109 auto ret = d_fct->send(&info);
110 if(ret) {
111 std::cout << "ZeroMQ sending failed!!!" << std::endl;
112 }
113 }

116 void PropertyBase::updateOthers(bool handleLocking) {
117 if(handleLocking) {
118 getEqFct()->unlock();
119 }
120 for(const auto& weakProp : propertiesToUpdate()) {
121 auto prop = weakProp.lock();
122 if(!prop) {
123 // property went away, could happen in shutdown phase
124 continue;
125 }
126 if(prop == shared_from_this()) {
127 continue;
128 }
129
130 if(handleLocking) {
131 prop->getEqFct()->lock();
132 }
133 prop->updateDoocsBuffer({});
134 if(handleLocking) {
135 prop->getEqFct()->unlock();
136 }
137 }
138 if(handleLocking) {
139 getEqFct()->lock();
140 }
141 }

143 void PropertyBase::setMacroPulseNumberSource(const std::string& sourcePath) {
144 if(!sourcePath.empty()) {
145 auto mpnSource = _doocsUpdater.getMappedProcessVariable<int64_t>(sourcePath);
146 if(mpnSource->getNumberOfSamples() != 1) {
147 throw ChimeraTK::logic_error("The property '" + mpnSource->getName() +
148 "' is used as a macro pulse number source, but it has an array length of " +
149 std::to_string(mpnSource->getNumberOfSamples()) + ". Length must be exactly 1");
150 }
151 if(!mpnSource->isReadable()) {
152 throw ChimeraTK::logic_error("The property '" + mpnSource->getName() +
153 "' is used as a macro pulse number source, but it is not readable.");
154 }
155 setMacroPulseNumberSource(mpnSource);
156 }

168 CommonlyUpdatedPropertySet& PropertyBase::propertiesToUpdate() {
169 if(_propertiesToUpdate_cacheIsFinal) {
170 return _propertiesToUpdate_cache;
171 }
172
173 // no need to clear _propertiesToUpdate_cache since Properties are never removed
174 for(auto& group : doocsAdapter.writeableVariablesWithMultipleProperties) {
175 // search for group containing weak ptr to this, if found, insert whole group
176 if(group.second.contains(shared_from_this())) {
177 _propertiesToUpdate_cache.insert(group.second.begin(), group.second.end());
178 }
179 }
180 if(doocsAdapter.writeableVariablesWithMultiplePropertiesIsFinal) {
181 _propertiesToUpdate_cacheIsFinal = true;
182 }
183 return _propertiesToUpdate_cache;
184 }
