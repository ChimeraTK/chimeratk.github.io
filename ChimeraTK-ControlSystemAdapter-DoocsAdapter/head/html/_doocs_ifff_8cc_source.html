39 void DoocsIfff::checkSourceConsistency() {
40 bool areAllSourcesWritable =
41 (_i1Value->isWriteable() && _f1Value->isWriteable() && _f2Value->isWriteable() && _f3Value->isWriteable());
42 _isWriteable = areAllSourcesWritable;
43 bool areAllSourcesReadOnly =
44 (_i1Value->isReadOnly() && _f1Value->isReadOnly() && _f2Value->isReadOnly() && _f3Value->isReadOnly());
45
46 if(!areAllSourcesWritable && !areAllSourcesReadOnly) {
47 throw ChimeraTK::logic_error("Doocs Adapter IFFF configuration Error: some IFFF sources are not writable");
48 }
49 if(areAllSourcesReadOnly) {
50 this->set_ro_access();
51 }
52 }

62 void DoocsIfff::updateDoocsBuffer(const TransferElementID& transferElementId) {
63 if(!updateConsistency(transferElementId)) {
64 return;
65 }
66
67 bool storeInHistory = true;
68 auto archiverStatus = ArchiveStatus::sts_ok;
69 if(_i1Value->dataValidity() != ChimeraTK::DataValidity::ok ||
70 _f1Value->dataValidity() != ChimeraTK::DataValidity::ok ||
71 _f2Value->dataValidity() != ChimeraTK::DataValidity::ok ||
72 _f3Value->dataValidity() != ChimeraTK::DataValidity::ok) {
73 if(this->d_error()) {
74 // data are already invalid, do not store in history
75 storeInHistory = false;
76 }
77
78 archiverStatus = ArchiveStatus::sts_err;
79 this->d_error(stale_data);
80 }
81 else {
82 this->d_error(no_error);
83 }
84
85 IFFF ifff;
86 ifff.i1_data = _i1Value->accessData(0);
87 ifff.f1_data = _f1Value->accessData(0);
88 ifff.f2_data = _f2Value->accessData(0);
89 ifff.f3_data = _f3Value->accessData(0);
90
91 doocs::Timestamp timestamp = correctDoocsTimestamp();
92 if(_macroPulseNumberSource) {
93 this->set_mpnum(_macroPulseNumberSource->accessData(0));
94 }
95 // We should also checked if data should be stored (flag storeInHistory). Invalid data should NOT be stored except
96 // first invalid data point. (https://github.com/ChimeraTK/ControlSystemAdapter-DoocsAdapter/issues/40)
97 if(this->get_histPointer() && storeInHistory) {
98 /*FIXME: This set_and_archive does not support the timestamp yet (only sec and msec, and I guess m is milli?)*/
99 /*FIXME: This set_and_archive does not support eventIDs yet */
100 this->set_and_archive(&ifff, archiverStatus, 0, 0 /*msec*/);
101 }
102 else {
103 this->set_value(&ifff);
104 }
105 // Note, I guess above will in future look something like this:
106 // doocs::EventId eventid(_macroPulseNumberSource->accessData(0));
107 // this->set_value(ifff, timestamp, eventid, archiverStatus);
108 // because the feature "do not store invalid data in history" is effectily lost
109 // when set_value() now also stores to history.
110 // However, currently it does _not_ yet.
111
112 sendZMQ(timestamp);
113 }

134 void DoocsIfff::sendToApplication(bool getLock) {
135 IFFF* ifff = value();
136
137 _i1Value->accessData(0) = ifff->i1_data;
138 _f1Value->accessData(0) = ifff->f1_data;
139 _f2Value->accessData(0) = ifff->f2_data;
140 _f3Value->accessData(0) = ifff->f3_data;
141
142 // write all with the same version number
143 auto timestamp = DoocsIfff::get_timestamp().to_time_point();
144 VersionNumber v(timestamp);
145 _i1Value->write(v);
146 _f1Value->write(v);
147 _f2Value->write(v);
148 _f3Value->write(v);
149
150 updateOthers(getLock);
151 }
