60 void DoocsSpectrum::auto_init() {
61 doocsAdapter.beforeAutoInit();
62
63 // check if the macro pulse number source has been set if the spectrum is buffered
64 if(_nBuffers > 1) {
65 if(_macroPulseNumberSource == nullptr) {
66 throw ChimeraTK::logic_error(
67 "D_spectrum '" + _processArray->getName() + "' has numberOfBuffers > 1 but not macro pulse number source.");
68 }
69 }
70
71 // send the current value to the device
72 D_spectrum::read();
73 modified = false;
74 if(this->get_access() == 1 ||
75 (_processArray->isWriteable() && otherPropertiesToUpdate.empty())) { // property is writeable
76 sendToDevice(false);
77 // set DOOCS time stamp, workaround for DOOCS bug (get() always gives current time stamp if no timestamp is set,
78 // which breaks consistency check in ZeroMQ subscriptions after the 4 minutes timeout)
79 D_spectrum::set_stamp();
80 }

83 void DoocsSpectrum::write(std::ostream& s) {
84 // DOOCS is normally keeping the location lock until everything is written for that location: all D_spectrum and all
85 // other properties. This can take too long (like seconds), which leads to noticable freezes of the UI. As a
86 // work-around we release the lock here, wait some time and acquire the lock again. Since this happens in a separate
87 // thread (svr_writer), this slow-down should not be of any harm.
88 get_eqfct()->unlock();
89 usleep(1000);
90 get_eqfct()->lock();
91 if(!modified || _processArray->isReadOnly()) {
92 return;
93 }
94 modified = false;
95 D_spectrum::write(s);
96 }

109 void DoocsSpectrum::updateDoocsBuffer(const TransferElementID& transferElementId) {
110 // Note: we already own the location lock by specification of the DoocsUpdater
111
112 // There are only the processArray and the macro pulse number in the consistency
113 // group. The limits are coming asynchronously and not for every macro pulse,
114 // so we just take test latest we have.
115 if(!updateConsistency(transferElementId)) {
116 return;
117 }
118
119 doocs::Timestamp timestamp = correctDoocsTimestamp();
120
121 auto sinceEpoch = timestamp.get_seconds_and_microseconds_since_epoch();
122 auto seconds = sinceEpoch.seconds;
123 auto microseconds = sinceEpoch.microseconds;
124
125 // set macro pulse number, buffer number and time stamp
126 size_t ibuf = 0;
127 if(_macroPulseNumberSource != nullptr) {
128 ibuf = _macroPulseNumberSource->accessData(0) % _nBuffers;
129 macro_pulse(_macroPulseNumberSource->accessData(0), ibuf);
130 }
131 set_tmstmp(seconds, microseconds, ibuf);
132
133 if(_processArray->dataValidity() != ChimeraTK::DataValidity::ok) {
134 // this will set error code correctly in given buffer _and_ in main property attribute, while
135 // d_error() sets it only in latter
136 this->error(stale_data, ibuf);
137 }
138 else {
139 this->error(no_error, ibuf);
140 }
141
142 // fill the spectrum
143 std::vector<float>& processVector = _processArray->accessChannel(0);
144 if(_nBuffers == 1) {
145 // We have to fill the spectrum differently if it is unbuffered, as the internal data structures seem to be
146 // completely different.
147 memcpy(spectrum()->d_spect_array.d_spect_array_val, processVector.data(), processVector.size() * sizeof(float));
148 spectrum()->d_spect_array.d_spect_array_len = processVector.size();
149 }
150 else {
151 fill_spectrum(processVector.data(), processVector.size(), ibuf);
152 }
153
154 // mark property as modified, for (optional) persistence
155 modified = true;
156
157 sendZMQ(timestamp);
158 }

160 void DoocsSpectrum::updateParameters() {
161 // Note: we already own the location lock by specification of the DoocsUpdater
162 float start, increment;
163 if(_startAccessor) {
164 start = _startAccessor->accessData(0);
165 }
166 else {
167 start = this->spec_start();
168 }
169 if(_incrementAccessor) {
170 increment = _incrementAccessor->accessData(0);
171 }
172 else {
173 increment = this->spec_inc();
174 }
175
176 // WORKAROUND: spectrum_parameter modifies the internal timestamp of
177 // the spectrum, this confuses our code that it thinks it has already sent off the
178 // spectrum with this timestamp, bumps the timestamp to enable DOOCS to check for inconsistencies
179 // which then confuses the DAQ since it sees data for the same MP with different timestamps
180 // https://mcs-gitlab.desy.de/doocs/doocs-core-libraries/serverlib/-/issues/35
181 auto oldTimeStamp = this->get_timestamp();
182 spectrum_parameter(this->spec_time(), start, increment, this->spec_status());
183 this->set_timestamp(oldTimeStamp);
184 }
