39BOOST_AUTO_TEST_CASE_TEMPLATE(toDeviceTest, T, simple_test_types) {
40 std::pair<boost::shared_ptr<ControlSystemPVManager>, boost::shared_ptr<DevicePVManager>> pvManagers =
41 createPVManager();
42 boost::shared_ptr<ControlSystemPVManager> csManager = pvManagers.first;
43 boost::shared_ptr<DevicePVManager> devManager = pvManagers.second;
44
45 static const size_t arraySize = 8;
46 boost::shared_ptr<ChimeraTK::NDRegisterAccessor<T>> deviceVariable =
47 devManager->createProcessArray<T>(SynchronizationDirection::controlSystemToDevice, "toDeviceVariable", arraySize);
48 boost::shared_ptr<ChimeraTK::NDRegisterAccessor<T>> controlSystemVariable =
49 csManager->getProcessArray<T>("toDeviceVariable");
50
51 DoocsUpdater updater;
52
53 // Write to the doocs spectrum and send it.
54 // We use the 'testable' version which exposes sendToDevice, which otherwise
55 // is protected.
56 TestableDoocsSpectrum doocsSpectrum(nullptr, "someName", getTypeChangingDecorator<float>(controlSystemVariable),
57 updater, DataConsistencyGroup::MatchingMode::exact);
58
59 // create unique signature for each template parameter
60 // negative factor for signed values
61 T sign = (std::numeric_limits<T>::is_signed ? -1 : 1);
62 // integer size offset for integer, fractional offset for floating type
63 T offset = (std::numeric_limits<T>::is_integer ? sizeof(T) : 1. / sizeof(T));
64
65 for(size_t i = 0; i < arraySize; ++i) {
66 doocsSpectrum.fill_spectrum(i, sign * static_cast<T>(i * i) + offset);
67 }
68 doocsSpectrum.sendToDevice(false);
69
70 // receive on the device side and check that the value has arrived
71 deviceVariable->readNonBlocking();
72
73 std::vector<T>& deviceVector = deviceVariable->accessChannel(0);
74 for(size_t i = 0; i < arraySize; ++i) {
75 std::stringstream errorMessage;
76 errorMessage << "i = " << i << ", deviceVector[i] = " << deviceVector[i] << " expected "
77 << sign * static_cast<T>(i * i) + offset;
78 BOOST_CHECK_MESSAGE(deviceVector[i] == sign * static_cast<T>(i * i) + offset, errorMessage.str());
79 }
80}

82BOOST_AUTO_TEST_CASE_TEMPLATE(fromDeviceTest, T, simple_test_types) {
83 std::pair<boost::shared_ptr<ControlSystemPVManager>, boost::shared_ptr<DevicePVManager>> pvManagers =
84 createPVManager();
85 boost::shared_ptr<ControlSystemPVManager> csManager = pvManagers.first;
86 boost::shared_ptr<DevicePVManager> devManager = pvManagers.second;
87
88 static const size_t arraySize = 8;
89 typename boost::shared_ptr<ChimeraTK::NDRegisterAccessor<T>> deviceVariable = devManager->createProcessArray<T>(
90 SynchronizationDirection::deviceToControlSystem, "fromDeviceVariable", arraySize);
91 typename boost::shared_ptr<ChimeraTK::NDRegisterAccessor<T>> controlSystemVariable =
92 csManager->getProcessArray<T>("fromDeviceVariable");
93
94 DoocsUpdater updater;
95
96 // initialise the doocs spectrum
97 DoocsSpectrum doocsSpectrum(nullptr, "someName", getTypeChangingDecorator<float>(controlSystemVariable), updater,
98 DataConsistencyGroup::MatchingMode::exact, nullptr, nullptr);
99 for(size_t i = 0; i < arraySize; ++i) {
100 doocsSpectrum.fill_spectrum(i, 0);
101 }
102
103 // prepare the device side and send it
104 T sign = (std::numeric_limits<T>::is_signed ? -1 : 1);
105 T offset = (std::numeric_limits<T>::is_integer ? sizeof(T) : 1. / sizeof(T));
106
107 std::vector<T>& deviceVector = deviceVariable->accessChannel(0);
108 for(size_t i = 0; i < arraySize; ++i) {
109 deviceVector[i] = (sign * static_cast<T>(i * i) + offset);
110 }
111 deviceVariable->write();
112
113 // everything should still be 0 on the CS side
114 std::vector<T>& csVector = controlSystemVariable->accessChannel(0);
115 for(size_t i = 0; i < arraySize; ++i) {
116 BOOST_CHECK(csVector[i] == 0);
117 BOOST_CHECK(doocsSpectrum.read_spectrum(i) == 0);
118 }
119
120 updater.update();
121
122 // The actual vector buffer has changed. We have to get the new reference.
123 csVector = controlSystemVariable->accessChannel(0);
124 for(size_t i = 0; i < arraySize; ++i) {
125 BOOST_CHECK(csVector[i] == sign * static_cast<T>(i * i) + offset);
126 BOOST_CHECK_EQUAL(doocsSpectrum.read_spectrum(i), sign * static_cast<T>(i * i) + offset);
127 }
128}
