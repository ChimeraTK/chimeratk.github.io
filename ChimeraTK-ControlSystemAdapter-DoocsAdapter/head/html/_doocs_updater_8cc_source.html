12 void DoocsUpdater::addVariable(
13 TransferElementAbstractor variable, EqFct* eq_fct, const std::function<void()>& updaterFunction) {
14 // Don't add the transfer element twice into the list of elements to read (i.e. later into the ReadAnyGroup).
15 // To check if there is such an element we use the map with the lookup table
16 // which has a search function, instead of manually looking at the elements in
17 // the list and compare the ID.
18 if(_toDoocsDescriptorMap.find(variable.getId()) == _toDoocsDescriptorMap.end()) {
19 _elementsToRead.push_back(variable);
20 }
21 else {
22 _toDoocsDescriptorMap[variable.getId()].additionalTransferElements.insert(variable.getHighLevelImplElement());
23 }
24
25 _toDoocsDescriptorMap[variable.getId()].updateFunctions.push_back(updaterFunction);
26 _toDoocsDescriptorMap[variable.getId()].locations.push_back(eq_fct);
27 }

29 void DoocsUpdater::update() {
30 for(auto& transferElem : _elementsToRead) {
31 if(transferElem.readLatest()) {
32 for(auto& updaterFunction : _toDoocsDescriptorMap[transferElem.getId()].updateFunctions) {
33 updaterFunction();
34 }
35 }
36 }
37 }

39 void DoocsUpdater::updateLoop() {
40 if(_elementsToRead.empty()) {
41 return;
42 }
43
44 std::unordered_set<EqFct*> locationsToLock;
45
46 // preallocate locationsToLock to avoid breaking "realtime" behaivour
47 size_t nMaxLocationsToLock = 0;
48 for(auto& pair : _toDoocsDescriptorMap) {
49 nMaxLocationsToLock = std::max(nMaxLocationsToLock, pair.second.locations.size());
50 }
51 locationsToLock.reserve(nMaxLocationsToLock);
52
53 ReadAnyGroup group(_elementsToRead.begin(), _elementsToRead.end());
54
55 // Call preRead for all TEs on additional transfer elements. waitAny() is doing this for all elements in the
56 // ReadAnyGroup. Unnecessary calls to preRead() are anyway ignored and merely pose a performance issue. For large
57 // servers, the performance impact is significant, hence we keep track of the TEs which need to be called.
58 for(auto& pair : _toDoocsDescriptorMap) {
59 for(const auto& elem : pair.second.additionalTransferElements) {
60 elem->preRead(ChimeraTK::TransferType::read);
61 }
62 }
63
64 while(true) {
65 // Wait until any variable got an update
66 auto notification = group.waitAny();
67 auto updatedElement = notification.getId();
68 auto& descriptor = _toDoocsDescriptorMap[updatedElement];
69
70 // Gather all involved locations in a unique set
71 for(auto& location : descriptor.locations) {
72 if(locationsToLock.insert(location).second) {
73 location->lock();
74 }
75 }
76 // Complete the read transfer of the process variable
77 notification.accept();
78
79 // Call postRead for all TEs on _toDoocsAdditionalTransferElementsMap for the updated ID
80 for(const auto& elem : descriptor.additionalTransferElements) {
81 elem->postRead(ChimeraTK::TransferType::read, true);
82 }
83
84 // Call all updater functions
85 for(auto& updaterFunction : descriptor.updateFunctions) {
86 updaterFunction();
87 }
88
89 // Unlock all involved locations
90 for(const auto& location : locationsToLock) {
91 location->unlock();
92 }
93 locationsToLock.clear();
94
95 // Call preRead for all TEs on _toDoocsAdditionalTransferElementsMap for the updated ID
96 for(const auto& elem : descriptor.additionalTransferElements) {
97 elem->preRead(ChimeraTK::TransferType::read);
98 }
99
100 // Allow shutting down this thread...
101 boost::this_thread::interruption_point();
102 }
103 }
104
105 void DoocsUpdater::run() {
106 _syncThread = boost::thread([this] { updateLoop(); });
107 }
108
109 void DoocsUpdater::stop() {
110 _syncThread.interrupt();
111 for(auto& var : _elementsToRead) {
112 var.getHighLevelImplElement()->interrupt();
113 }
114 _syncThread.join();
115 }
