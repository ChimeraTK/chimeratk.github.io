23// the array must have testStartValue+i at index i.
24template<class T>
25static bool testArrayContent(std::string const& propertyName, T testStartValue, T delta) {
26 auto array = DoocsServerTestHelper::doocsGetArray<T>(propertyName);
27 bool isOK = true;
28 T currentTestValue = testStartValue;
29 size_t index = 0;
30 for(auto val : array) {
31 if(std::fabs(val - currentTestValue) > 0.001) {
32 if(isOK) {
33 std::cout << "Array " << propertyName << " does not contain expected values. First mismatching data at index "
34 << index << ": " << val << " != " << currentTestValue << std::endl;
35 }
36 isOK = false;
37 }
38 currentTestValue += delta;
39 ++index;
40 }
41 return isOK;
42}

109static void testPropertyDoesNotExist(const std::string& addressString) {
110 EqAdr ad;
111 doocs::EqData ed, res;
112 // obtain location pointer
113 ad.adr(addressString);
114 EqFct* eqFct = eq_get(&ad);
115 BOOST_REQUIRE_MESSAGE(eqFct != nullptr, "Could not get location for property.");
116 // obtain value
117 eqFct->lock();
118 eqFct->get(&ad, &ed, &res);
119 eqFct->unlock();
120 // check for errors
122 BOOST_CHECK_MESSAGE(
123 res.error() != 0, std::string("Could read from ") + addressString + ", but it should not be there");
124}
