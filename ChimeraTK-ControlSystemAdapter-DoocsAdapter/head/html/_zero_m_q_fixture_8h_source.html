72 auto* f = static_cast<ZeroMQFixture*>(fixture);

123 template<typename ValueType>
124 static ValueType generateValue(size_t seed) {
125 ValueType v;
126 if constexpr(std::is_same_v<ValueType, int32_t>) {
127 v = 42 + (int32_t)seed;
128 }
129 else if constexpr(std::is_same_v<ValueType, std::vector<int32_t>>) {
130 v = {42, 43 + (int32_t)seed, 44, 45, 46, 47, 48, 49, 50, 51};

132 else if constexpr(std::is_same_v<ValueType, std::vector<float>>) {

136 // we have a test where i1_value is mapped as generated int value!

140 BOOST_REQUIRE(false);

163 BOOST_CHECK_EQUAL(received.get_int(k), expectedValue[k]);

166 else if constexpr(std::is_same_v<ExpectedValueType, std::vector<float>>) {
167 BOOST_CHECK_CLOSE(received.get_spectrum()->s_start, 123., 0.001);

184 void checkTransport() {
185 auto appPVmanager = GlobalFixture::referenceTestApplication.getPVManager();
186 // From now on, each consistent update should be received.
187 // Make sure consistent receiving is happening whether the macro pulse number is send first or second.
188 std::array<bool, 10> sendMacroPulseFirst = {true, true, true, false, false, false, true, false, true, false};
189 std::array<bool, 10> sendError = {false, false, false, false, false, false, false, false, true, true};
190 for(size_t i = 0; i < 10; ++i) {
191 bool isErrorCase = sendError[i];
192 if(isErrorCase && not generateErrors) {
193 continue;
194 }
195
196 GlobalFixture::referenceTestApplication.versionNumber = ChimeraTK::VersionNumber();
197 ++macroPulseNumber;
198 SetValueType setValue = generateValue<SetValueType>(i);
199 ExpectedValueType expectedValue = generateValue<ExpectedValueType>(i);
200 if(sendMacroPulseFirst[i]) {
201 DoocsServerTestHelper::doocsSet<int32_t>("//UINT/TO_DEVICE_SCALAR", macroPulseNumber);
202 }
203 else { // send the value first
204 setDoocsValue(setValue);
205 }
206
207 if(configUsesDataMatching) {
208 // nothing must be received, no consistent set yet
209 BOOST_CHECK_EQUAL(dataReceived, 0);
210 GlobalFixture::referenceTestApplication.runMainLoopOnce();
211 usleep(10000);
212 BOOST_CHECK_EQUAL(dataReceived, 0);
213 }
214 else {
215 // data_matching = none in test config means we expect not-yet-consistent updates
216 GlobalFixture::referenceTestApplication.runMainLoopOnce();
217 usleep(10000);
218 if(sendMacroPulseFirst[i]) {
219 // we don't expect an update to be sent out just for mp update
220 BOOST_CHECK_EQUAL(dataReceived, 0);
221 }
222 }
223
224 // now send the variable which has not been send yet
225 if(sendMacroPulseFirst[i]) {
226 setDoocsValue(setValue);
227 }
228 else {
229 DoocsServerTestHelper::doocsSet<int32_t>("//UINT/TO_DEVICE_SCALAR", macroPulseNumber);
230 }
231 if(isErrorCase) {
232 GlobalFixture::referenceTestApplication.dataValidity = ChimeraTK::DataValidity::faulty;
233 }
234 else {
235 GlobalFixture::referenceTestApplication.dataValidity = ChimeraTK::DataValidity::ok;
236 }
237 GlobalFixture::referenceTestApplication.runMainLoopOnce();
238
239 CHECK_WITH_TIMEOUT(dataReceived > 0);
240 usleep(10000);
241 BOOST_CHECK_EQUAL(dataReceived, 1);
242 dataReceived = 0;
243 {
244 std::lock_guard<std::mutex> lock(mutex);
245 BOOST_CHECK_EQUAL(received.error() != 0, isErrorCase);
246 checkReceivedValue(expectedValue);
247
248 std::string metaDataPVPath = fromDeviceMetaDataPath.length() > 0 ? fromDeviceMetaDataPath : fromDevicePath;
249 ChimeraTK::VersionNumber v = appPVmanager->getProcessVariable(metaDataPVPath)->getVersionNumber();
250 auto time = v.getTime();
251 auto secs = std::chrono::duration_cast<std::chrono::seconds>(time.time_since_epoch()).count();
252 auto usecs =
253 std::chrono::duration_cast<std::chrono::microseconds>(time.time_since_epoch()).count() - secs * 1e6;
254 BOOST_CHECK_EQUAL(receivedInfo.sec, secs);
255 BOOST_CHECK_EQUAL(receivedInfo.usec, usecs);
256 if(configUsesDataMatching) {
257 BOOST_CHECK_EQUAL(receivedInfo.ident, macroPulseNumber);
258 }
259 BOOST_CHECK_EQUAL(receivedInfo.stat != 0, isErrorCase);
260 // note, the "status" field in spectrum metadata (of ZMQ payload) is not set as an error field and hence not checked here
261 }
262 }
