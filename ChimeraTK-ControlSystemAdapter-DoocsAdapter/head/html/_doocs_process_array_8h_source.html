55  template<typename DOOCS_T, typename DOOCS_PRIMITIVE_T>
56  DoocsProcessArray<DOOCS_T, DOOCS_PRIMITIVE_T>::DoocsProcessArray(EqFct* eqFct, const std::string& doocsPropertyName,
57  const boost::shared_ptr<ChimeraTK::NDRegisterAccessor<DOOCS_PRIMITIVE_T>>& processArray, DoocsUpdater& updater)
58  : DOOCS_T(doocsPropertyName, processArray->getNumberOfSamples(), eqFct), PropertyBase(doocsPropertyName, updater),
59  _processArray(processArray) {
60  // Check if the array length exceeds the maximum allowed length by DOOCS.
61  // DOOCS does not report this as an error and instead silently truncates the
62  // array length.
63  auto len = static_cast<size_t>(this->length());
64  if(processArray->getNumberOfSamples() != len) {
65  std::stringstream s;
66  s << "Error: The selected DOOCS data type for the variable '" << processArray->getName() << "' "
67  << "(mapped to the DOOCS name '" << _doocsPropertyName << "') seems not to support the requested length of "
68  << processArray->getNumberOfSamples() << " since the DOOCS property has a length of " << len
69  << ". Try selecting a different DOOCS type in the mappng XML file, e.g. a D_spectrum!";
70  throw ChimeraTK::logic_error(s.str());
71  }
72  setupOutputVar(processArray);
73  }

102  template<typename DOOCS_T, typename DOOCS_PRIMITIVE_T>
103  void DoocsProcessArray<DOOCS_T, DOOCS_PRIMITIVE_T>::updateDoocsBuffer(const TransferElementID& transferElementId) {
104  if(!updateConsistency(transferElementId)) {
105  return;
106  }
107 
108  // Note: we already own the location lock by specification of the
109  // DoocsUpdater
110  auto& processVector = _processArray->accessChannel(0);
111 
112  THE_DOOCS_TYPE* dataPtr;
113  if constexpr(std::is_same<THE_DOOCS_TYPE, DOOCS_PRIMITIVE_T>::value) {
114  // No cast necessary if types are identical.
115  dataPtr = processVector.data();
116  }
117  else if constexpr(std::is_same<DOOCS_PRIMITIVE_T, ChimeraTK::Boolean>::value &&
118  std::is_same<THE_DOOCS_TYPE, bool>::value) {
119  // FIXME: Is it really ok to use reinterpret_cast here?
120  static_assert(sizeof(ChimeraTK::Boolean) == sizeof(bool));
121  dataPtr = reinterpret_cast<bool*>(processVector.data());
122  }
123  else {
124  static_assert(std::is_same<THE_DOOCS_TYPE, DOOCS_PRIMITIVE_T>::value, "Bad type casting.");
125  }
126 
127  if(_processArray->dataValidity() != ChimeraTK::DataValidity::ok) {
128  this->d_error(stale_data);
129  }
130  else {
131  this->d_error(no_error);
132  }
133 
134  this->fill_array(dataPtr, processVector.size());
135  modified = true;
136 
137  doocs::Timestamp timestamp = correctDoocsTimestamp();
138 
139  if(_macroPulseNumberSource) {
140  this->set_mpnum(_macroPulseNumberSource->accessData(0));
141  }
142 
143  sendZMQ(timestamp);
144  }
