57 template<typename DOOCS_T, typename DOOCS_PRIMITIVE_T>
58 DoocsProcessArray<DOOCS_T, DOOCS_PRIMITIVE_T>::DoocsProcessArray(EqFct* eqFct, const std::string& doocsPropertyName,
59 const boost::shared_ptr<ChimeraTK::NDRegisterAccessor<DOOCS_PRIMITIVE_T>>& processArray, DoocsUpdater& updater,
60 DataConsistencyGroup::MatchingMode matchingMode)
61 : DOOCS_T(doocsPropertyName, processArray->getNumberOfSamples(), eqFct),
62 PropertyBase(doocsPropertyName, updater, matchingMode), _processArray(processArray) {
63 // Check if the array length exceeds the maximum allowed length by DOOCS.
64 // DOOCS does not report this as an error and instead silently truncates the
65 // array length.
66 auto len = static_cast<size_t>(this->length());
67 if(processArray->getNumberOfSamples() != len) {
68 std::stringstream s;
69 s << "Error: The selected DOOCS data type for the variable '" << processArray->getName() << "' "
70 << "(mapped to the DOOCS name '" << _doocsPropertyName << "') seems not to support the requested length of "
71 << processArray->getNumberOfSamples() << " since the DOOCS property has a length of " << len
72 << ". Try selecting a different DOOCS type in the mappng XML file, e.g. a D_spectrum!";
73 throw ChimeraTK::logic_error(s.str());
74 }
75 setupOutputVar(_processArray);
76 }

90 template<typename DOOCS_T, typename DOOCS_PRIMITIVE_T>
91 void DoocsProcessArray<DOOCS_T, DOOCS_PRIMITIVE_T>::auto_init() {
92 doocsAdapter.beforeAutoInit();
93
94 DOOCS_T::auto_init();
95 modified = false;
96 // send the current value to the device
97 // property is writeable OR the target accessor is writable and the only one connected to this property
98 // The second case is to have bi-directional variables that are used to persist settings into the config file
99 // and need that value back on start-up but are not supposed to be written by the control system and mapped
100 // read-only in the configuration file.
101 if(this->get_access() == 1 ||
102 (_processArray->isWriteable() && otherPropertiesToUpdate.empty())) { // property is writeable
103 sendToDevice(false);
104 // set DOOCS time stamp, workaround for DOOCS bug (get() always gives current time stamp if no timestamp is set,
105 // which breaks consistency check in ZeroMQ subscriptions after the 4 minutes timeout)
106 DOOCS_T::set_stamp();
107 }

110 template<typename DOOCS_T, typename DOOCS_PRIMITIVE_T>
111 void DoocsProcessArray<DOOCS_T, DOOCS_PRIMITIVE_T>::updateDoocsBuffer(const TransferElementID& transferElementId) {
112 if(!updateConsistency(transferElementId)) {
113 return;
114 }
115
116 // Note: we already own the location lock by specification of the
117 // DoocsUpdater
118 auto& processVector = _processArray->accessChannel(0);
119
120 THE_DOOCS_TYPE* dataPtr;
121 if constexpr(std::is_same<THE_DOOCS_TYPE, DOOCS_PRIMITIVE_T>::value) {
122 // No cast necessary if types are identical.
123 dataPtr = processVector.data();
124 }
125 else if constexpr(std::is_same<DOOCS_PRIMITIVE_T, ChimeraTK::Boolean>::value &&
126 std::is_same<THE_DOOCS_TYPE, bool>::value) {
127 // FIXME: Is it really ok to use reinterpret_cast here?
128 static_assert(sizeof(ChimeraTK::Boolean) == sizeof(bool));
129 dataPtr = reinterpret_cast<bool*>(processVector.data());
130 }
131 else {
132 static_assert(std::is_same<THE_DOOCS_TYPE, DOOCS_PRIMITIVE_T>::value, "Bad type casting.");
133 }
134
135 if(_processArray->dataValidity() != ChimeraTK::DataValidity::ok) {
136 this->d_error(stale_data);
137 }
138 else {
139 this->d_error(no_error);
140 }
141
142 this->fill_array(dataPtr, processVector.size());
143 modified = true;
144
145 doocs::Timestamp timestamp = correctDoocsTimestamp();
146
147 if(_macroPulseNumberSource) {
148 this->set_mpnum(_macroPulseNumberSource->accessData(0));
149 }
150
151 sendZMQ(timestamp);
152 }
