18 CSAdapterEqFct::CSAdapterEqFct(int code, const EqFctParameters& p)
19 : EqFct(p), _controlSystemPVManager(doocsAdapter.getControlSystemPVManager()), _updater(doocsAdapter.updater),
20 _code(code) {
21 registerProcessVariablesInDoocs();
22
23 // construct and populate the StatusHandler for this location
24 for(const ErrorReportingInfo& errorReportingInfo :
25 ChimeraTK::VariableMapper::getInstance().getErrorReportingInfos()) {
26 if(name() != errorReportingInfo.targetLocation) {
27 continue;
28 }
29
30 assert(!_statusHandler); // only single StatusHandler may be requested per location
31 auto statusCodeVariable = _controlSystemPVManager->getProcessArray<int32_t>(errorReportingInfo.statusCodeSource);
32 if(!statusCodeVariable) {
33 throw ChimeraTK::logic_error("illegal/non-existing statusCodeSource: " + errorReportingInfo.statusCodeSource);
34 }
35 // this one is optional
36 ProcessArray<std::string>::SharedPtr statusStringVariable;
37 if(_controlSystemPVManager->hasProcessVariable(errorReportingInfo.statusStringSource)) {
38 statusStringVariable =
39 _controlSystemPVManager->getProcessArray<std::string>(errorReportingInfo.statusStringSource);
40 }
41 _statusHandler.reset(new StatusHandler(this, _updater, statusCodeVariable, statusStringVariable));
42 }
43 }

53 for(auto& pair : _doocsProperties) {

66 int CSAdapterEqFct::write(std::ostream& fprt) {
67 /*
68 * iterate over all properties (i.e. instances of D_fcn),
69 * check if its a doocs::D_array()
70 * if it should be persisted and it's not by default due to the restricting length MAX_CONF_LENGTH,
71 * persist it in a separate file, but only if writable.
72 */
73 for(auto& pair : this->_doocsProperties) {
74 // try a side-cast to get property attributes
75 auto attrs = std::dynamic_pointer_cast<PropertyAttributes>(pair.first);
76 if(attrs && attrs->persist == PersistConfig::ON) {
77 D_fct* p = pair.second.get();
78
79 switch(p->data_type()) {
80 case DATA_A_BYTE:
81 saveArray<unsigned char>(p);
82 break;
83 case DATA_A_SHORT:
84 saveArray<short>(p);
85 break;
86 case DATA_A_USHORT:
87 saveArray<unsigned short>(p);
88 break;
89 case DATA_A_INT:
90 saveArray<int>(p);
91 break;
92 case DATA_A_UINT:
93 saveArray<unsigned int>(p);
94 break;
95 case DATA_A_LONG:
96 saveArray<int64_t>(p);
97 break;
98 case DATA_A_ULONG:
99 saveArray<uint64_t>(p);
100 break;
101 case DATA_A_FLOAT:
102 saveArray<float>(p);
103 break;
104 case DATA_A_DOUBLE:
105 saveArray<double>(p);
106 break;
107 // case DATA_A_TS_SPECTRUM does not need handling here, it's already persisted by base implementation.
108 }
109 }
110 }
111
112 return EqFct::write(fprt);
113 }

124 void CSAdapterEqFct::registerProcessVariablesInDoocs() {
125 // We only need the factory inside this function
126 DoocsPVFactory factory(this, *_updater, _controlSystemPVManager);
127
128 auto mappingForThisLocation = VariableMapper::getInstance().getPropertiesInLocation(name());
129
130 for(auto& propertyDescription : mappingForThisLocation) {
131 try {
132 _doocsProperties[propertyDescription] = factory.create(propertyDescription);
133
134 // if one of the PVs used by the property is among the keys of the writeableVariablesWithMultipleProperties map,
135 // add the property to the list of properties to update (value of the beforementioned map).
136 auto& theMap = doocsAdapter.writeableVariablesWithMultipleProperties;
137 for(const auto& pvNameUsedByProperty : propertyDescription->getSources()) {
138 if(theMap.find(pvNameUsedByProperty) != theMap.end()) {
139 // the PV name has been found in the map keys -> add the property to the list to update
140 auto p = boost::dynamic_pointer_cast<ChimeraTK::PropertyBase>(_doocsProperties.at(propertyDescription));
141 auto& listOfPropertiesToUpdate = theMap.at(pvNameUsedByProperty);
142 listOfPropertiesToUpdate.insert(p);
143 }
144 }
145 }
146 catch(std::invalid_argument& e) {
147 std::cerr << "**** WARNING: Could not create property for variable '" << propertyDescription->location << "/"
148 << propertyDescription->name << "': " << e.what() << ". Skipping this property." << std::endl;
149 }
150 catch(doocs::Error& e) {
151 std::cerr << "**** WARNING: Could not create property for variable '" << propertyDescription->location << "/"
152 << propertyDescription->name << "': " << e.what() << ". Skipping this property." << std::endl;
153 }
154 }
155 }
