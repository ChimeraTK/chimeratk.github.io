49 void DoocsAdapter::waitUntilInitialised() {
50 int i = 0;
51 while(true) {
52 if(isInitialised) {
53 return;
54 }
55 // just sleep a bit. Use the "cheap" usleep, we don't care about precision
56 // here
57 ++i;
58 usleep(100);
59 }

64 bool DoocsAdapter::checkPrintDataLossWarning(size_t counter) {
65 // print first time at counter == 10 to suppress the messages spamming at startup
66 if(counter < 1) {
67 return false;
68 }
69 if(counter < 100) {
70 return counter % 10 == 0;
71 }
72 if(counter < 1000) {
73 return counter % 100 == 0;
74 }
75 if(counter < 10000) {
76 return counter % 1000 == 0;
77 }
78 if(counter < 100000) {
79 return counter % 10000 == 0;
80 }
81 if(counter < 1000000) {
82 return counter % 100000 == 0;
83 }
84 return counter % 1000000 == 0; // if the rate is 10Hz, this is roughly once per day
85 }

104 std::unique_ptr<doocs::Server> DoocsAdapter::createServer() {
105 auto server = std::make_unique<doocs::Server>(ChimeraTK::ApplicationBase::getInstance().getName().c_str());
106
107 server->set_init_prolog([&] { ChimeraTK::DoocsAdapter::eqInitProlog(); });
108 server->set_post_init_epilog([&] { ChimeraTK::DoocsAdapter::postInitEpilog(); });
109 server->set_cancel_epilog([&] { ChimeraTK::DoocsAdapter::eqCancel(); });
110
111 // This is a work-around for not being able to create the same location type for any arbitrary location code
112 // found in the conf file. It works as long as the largest code is below 10000. The registration here takes
113 // around 1ms on a standard MicroTCA AMC CPU.
114 // See: https://mcs-gitlab.desy.de/doocs/doocs-core-libraries/serverlib/-/issues/79
115 for(int c = 2; c < 10000; ++c) {
116 server->register_location_class(
117 c, [=](const EqFctParameters& p) { return std::make_unique<CSAdapterEqFct>(c, p); });
118 }
119 // The DCM server uses the code 5522228 = 0x544334 = 'TC4', so we register it here as well. Need to remove this
120 // temporary hack ASAP!
121 server->register_location_class(
122 5522228, [=](const EqFctParameters& p) { return std::make_unique<CSAdapterEqFct>(5522228, p); });
123
124 // Note: We cannot go back to use eq_create(), since we have link order issues and sometimes the wrong mplementation
125 // might win (the default implementation from the serverlib, which terminates the server with an error message).
126
127 return server;
128 }

132 /* eq_init_prolog is called before the locations are created, i.e. before the
133 * first call to eq_create. We initialise the application, i.e. all process
134 * variables are created in this function. */
135 void DoocsAdapter::eqInitProlog() {
136 // set the DOOCS server name to the application name
137 // Create static instances for all applications cores. They must not have
138 // overlapping process variable names ("location/protery" must be unique).
139 ChimeraTK::ApplicationBase::getInstance().setPVManager(doocsAdapter.getDevicePVManager());
140 ChimeraTK::ApplicationBase::getInstance().initialise();
141
142 // the variable manager can only be filled after we have the CS manager
143 auto pvNames = ChimeraTK::getAllVariableNames(doocsAdapter.getControlSystemPVManager());
144
145 auto xmlFileName = ChimeraTK::ApplicationBase::getInstance().getName() + XML_CONFIG_SUFFIX;
146
147 if(std::filesystem::exists(xmlFileName)) {
148 ChimeraTK::VariableMapper::getInstance().prepareOutput(xmlFileName, pvNames);
149 }
150 else {
151 std::cerr << "WARNING: No XML file for the Doocs variable config found. Trying direct import." << std::endl;
152 ChimeraTK::VariableMapper::getInstance().directImport(pvNames);
153 }
154
155 // prepare list of unmapped read variables and pass it to the Application for optimisation
156 for(const auto& p : ChimeraTK::VariableMapper::getInstance().getUsedVariables()) {
157 auto it = pvNames.find(p);
158 if(it != pvNames.end()) {
159 pvNames.erase(it);
160 }
161 }
162 ChimeraTK::ApplicationBase::getInstance().optimiseUnmappedVariables(pvNames);
163
164 // reverse mapping will be used to find out where we need a fan-out for a source process variable
165 doocsAdapter.findReverseMapping();
166 }

170 void DoocsAdapter::findReverseMapping() {
171 for(const auto& descr : ChimeraTK::VariableMapper::getInstance().getAllProperties()) {
172 for(const auto& source : descr->getSources()) {
173 reverseMapping[source].insert(descr);
174 }
175 }
176 // filter the map to contain only PVs being used at least twice with at least one writable property
177 for(const auto& p : reverseMapping) {
178 if(!doocsAdapter.getControlSystemPVManager()->getProcessVariable(p.first)->isWriteable()) {
179 // do not add PVs to the mapping which are not writeable
180 continue;
181 }
182 if(p.second.size() < 2) {
183 continue;
184 }
185 size_t writeableCount = 0;
186 for(const auto& d : p.second) {
187 auto attr = std::dynamic_pointer_cast<ChimeraTK::PropertyAttributes>(d);
188 if(attr->isWriteable) {
189 ++writeableCount;
190 }
191 }
192 if(writeableCount == 0) {
193 continue;
194 }
195 if(writeableCount > 1) {
196 // This case is not (yet) covered. It would require some synchronisation mechanism between writeable
197 // properties, which is not trivial as inconsistencies and dead locks must be avoided.
198 std::cout << "**** WARNING: Variable '" + p.first +
199 "' mapped to more than one writeable property. Expect inconsistencies!\n";
200 }
201
202 // Add PVs which are used at least twice with at least one writable property to list
203 assert(!doocsAdapter.writeableVariablesWithMultiplePropertiesIsFinal);
204 doocsAdapter.writeableVariablesWithMultipleProperties[p.first] = {};
205 }
206
207 std::set<std::string> fanNamesFromDoocsAdapter;
208 for(const auto& el : doocsAdapter.reverseMapping) {
209 if(el.second.size() > 1) {
210 fanNamesFromDoocsAdapter.insert(el.first);
211 }
212 }
213 // we simply request fans for set_error sources without knowledge whether actually required
214 for(const auto& errInfo : ChimeraTK::VariableMapper::getInstance().getErrorReportingInfos()) {
215 fanNamesFromDoocsAdapter.insert(errInfo.statusCodeSource);
216 if(errInfo.statusStringSource.length() > 1) {
217 fanNamesFromDoocsAdapter.insert(errInfo.statusStringSource);
218 }
219 }
220 updater->setPvNamesWithFan(fanNamesFromDoocsAdapter);
221 }

223 /* post_init_epilog is called after all DOOCS properties are fully intialised,
224 * including any value intialisation from the config file. We start the
225 * application here. It will be launched in a separate thread. */
226 void DoocsAdapter::postInitEpilog() {
227 // check for locations not yet created (due to missing entries in the .conf file) and create them now
228 std::map<std::string, int> locMap = ChimeraTK::VariableMapper::getInstance().getLocationAndCode();
229 for(const auto& loc : ChimeraTK::VariableMapper::getInstance().getAllLocations()) {
230 int codeToSet = 10;
231 bool defaultUsed = true;
232 // if no code is set in xml file, 10 is default
233 if(locMap.find(loc) != locMap.end()) {
234 codeToSet = locMap.find(loc)->second;
235 defaultUsed = false;
236 }
237
238 auto* eq = find_device(loc);
239 if(eq == nullptr) {
240 add_location(codeToSet, loc);
241 }
242 else if(eq->fct_code() != codeToSet && !defaultUsed) {
243 throw ChimeraTK::logic_error("Location '" + loc + "' has already a fct_code '" +
244 std::to_string(eq->fct_code()) + "' from config file, that does not match with '" +
245 std::to_string(codeToSet) + "' from xml file");
246 }
247 }
248
249 // reverseMapping content no longer needed since variable network will not change from this point on
250 doocsAdapter.reverseMapping.clear();
251 doocsAdapter.writeableVariablesWithMultiplePropertiesIsFinal = true;
252
253 // check for variables not yet initialised - we must guarantee that all to-application variables are written exactly
254 // once at server start.
255 for(auto& pv : doocsAdapter.getControlSystemPVManager()->getAllProcessVariables()) {
256 if(!pv->isWriteable()) {
257 continue;
258 }
259 if(pv->getVersionNumber() == ChimeraTK::VersionNumber(nullptr)) {
260 // The variable has not yet been written. Do it now, even if we just send a 0.
261 pv->write();
262 }
263 }
264
265 // start the application and the updater
266 ChimeraTK::ApplicationBase::getInstance().run();
267 doocsAdapter.updater->run();
268 ChimeraTK::DoocsAdapter::isInitialised = true;
269 }
