22 void StatusHandler::updateError(TransferElementID transferElementId) {
23 if(!_varPair.update(transferElementId)) {
24 return;
25 }
26 int statusCode = _varPair._status;
27 int err_no = statusCodeMapping(statusCode);
28 std::string err_str = _varPair.getMessage();
29
30 // Note: we already own the location lock by specification of the DoocsUpdater
31 if(err_no == no_error) {
32 _eqFct->set_error(err_no);
33 }
34 else {
35 _eqFct->set_error(err_no, err_str);
36 }
37 }

39 StatusHandler::StatusHandler(EqFct* eqFct, boost::shared_ptr<DoocsUpdater> updater,
40 boost::shared_ptr<ChimeraTK::NDRegisterAccessor<int32_t>> const& statusScalar,
41 boost::shared_ptr<ChimeraTK::NDRegisterAccessor<std::string>> const& statusString)
42 : _doocsUpdater(std::move(updater)), _eqFct(eqFct),
43 _varPair(ChimeraTK::ScalarRegisterAccessor<int32_t>(statusScalar)) {
44 if(!statusScalar->isReadable()) {
45 throw ChimeraTK::logic_error(statusScalar->getName() + " is not readable!");
46 }
47
48 _doocsUpdater->addVariable(
49 _varPair._status, _eqFct, [this, capture0 = statusScalar->getId()] { updateError(capture0); });
50 if(statusString) {
51 if(!statusString->isReadable()) {
52 throw ChimeraTK::logic_error(statusString->getName() + " is not readable!");
53 }
54 _varPair.setMessageSource(ChimeraTK::ScalarRegisterAccessor<std::string>(statusString));
55
56 _doocsUpdater->addVariable(
57 _varPair._message, _eqFct, [this, capture0 = statusString->getId()] { updateError(capture0); });
58 }
59 }

61 int StatusHandler::statusCodeMapping(int x) {
62 auto err = StatusAccessorBase::Status(x);
63 // a mapping for StatusOutput values -> DOOCS error codes
64 switch(err) {
65 case StatusAccessorBase::Status::OK:
66 return no_error;
67 case StatusAccessorBase::Status::FAULT:
68 return not_available;
69 case StatusAccessorBase::Status::OFF:
70 return device_offline;
71 case StatusAccessorBase::Status::WARNING:
72 return warning;
73 default:
74 return x;
75 }
76 }
