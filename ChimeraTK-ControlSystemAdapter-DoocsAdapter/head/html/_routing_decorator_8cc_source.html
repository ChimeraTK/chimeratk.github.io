8 TransferElement::SharedPtr RoutingDecoratorDomain::add(TransferElement::SharedPtr source) {
9 TransferElement::SharedPtr ret;
10 callForType(source->getValueType(), [&](auto t) {
11 using UserType = decltype(t);
12
13 auto sourceWithType = boost::dynamic_pointer_cast<ChimeraTK::NDRegisterAccessor<UserType>>(source);
14 assert(sourceWithType);
15 TransferElementID sourceId = source->getId();
16 auto decorator = boost::make_shared<RoutingDecorator<UserType>>(sourceWithType);
17
18 if(!_sourceMasters.contains(sourceId)) {
19 decorator->setupFan();
20 _sourceMasters[sourceId] = decorator;
21 }
22
23 else {
24 auto sourceMaster = boost::dynamic_pointer_cast<RoutingDecorator<UserType>>(_sourceMasters[sourceId]);
25 assert(sourceMaster);
26 decorator->template addToFan(*sourceMaster);
27 }
28 ret = decorator;
29 });
30 return ret;
31 }

33 bool RoutingDecoratorDomain::send(TransferElementID updatedElement) {
34 auto it = _sourceMasters.find(updatedElement);
35 if(it == _sourceMasters.end()) {
36 return false;
37 }
38
39 bool ret;
40 callForType(it->second->getValueType(), [&](auto t) {
41 using UserType = decltype(t);
42
43 auto dec = boost::dynamic_pointer_cast<RoutingDecorator<UserType>>(it->second);
44 assert(dec);
45 if(!dec->isFan()) {
46 ret = false;
47 }
48
49 auto& source = dec->getSource();
50 auto vn = source->getVersionNumber();
51 assert(vn > VersionNumber{nullptr});
52
53 unsigned nCopies = dec->getCopies().size();
54 auto jt = dec->getCopies().begin();
55 for(unsigned j = 0; j < nCopies - 1; ++j, ++jt) {
56 auto dest = *jt;
57 // currently we support only 1 channel
58 dest->accessChannel(0) = source->accessChannel(0);
59 dest->write(vn);
60 }
61 // use swap for last copy
62 auto dest = *jt;
63 dest->accessChannel(0).swap(source->accessChannel(0));
64 dest->write(vn);
65 ret = true;
66 });
67 return ret;
68 }
