108 template<typename SELF, typename UserType>
109 void PropertyBase::sendArrayToDevice(SELF* dfct, OneDRegisterAccessor<UserType>& processArray) {
110 constexpr bool isSpectrum = std::is_base_of<D_spectrum, SELF>::value;
111
112 // always get a fresh reference
113 auto processVector = processArray.data();
114 size_t arraySize = processArray.getNElements();
115 auto doocsLen = static_cast<size_t>(dfct->length());
116 if(doocsLen != arraySize) {
117 std::cout << "Warning: Array length mismatch in property " << this->getEqFct()->name() << "/" << dfct->basename()
118 << ": Property has length " << doocsLen << " but " << arraySize << " expected." << std::endl;
119 arraySize = std::min(arraySize, size_t(doocsLen));
120 }
121 if constexpr(isSpectrum) {
122 // FIXME: find the efficient, memcopying function for float
123 // always get a fresh reference
124 for(size_t i = 0; i < arraySize; ++i) {
125 processVector[i] = dfct->read_spectrum((int)i);
126 }
127 }
128 else {
129 // Brute force implementation with a loop. Works for all data types.
130 for(size_t i = 0; i < arraySize; ++i) {
131 processVector[i] = dfct->value(i);
132 }
133 }
134 auto timestamp = dfct->get_timestamp().to_time_point();
135 processArray.write(VersionNumber(timestamp));
136
137 // Correct property length in case of a mismatch.
138 if(doocsLen != arraySize) {
139 if constexpr(isSpectrum) {
140 dfct->length(arraySize);
141 // FIXME - do we need to restore values like for arrays?
142 // it's more difficult with D_spectrum because of the buffered/unbuffered feature
143 }
144 else {
145 dfct->set_length(arraySize);
146 // restore value from ProcessArray, as it may have been destroyed in set_length().
147 for(size_t i = 0; i < arraySize; ++i) {
148 dfct->set_value(processVector[i], i);
149 }
150 }
151 }
152 }
