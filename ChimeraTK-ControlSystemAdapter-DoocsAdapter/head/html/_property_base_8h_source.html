135 template<typename SELF, typename UserType>
136 void PropertyBase::sendArrayToDevice(SELF* dfct, OneDRegisterAccessor<UserType>& processArray) {
137 constexpr bool isSpectrum = std::is_base_of<D_spectrum, SELF>::value;
138
139 // always get a fresh reference
140 auto processVector = processArray.data();
141 size_t arraySize = processArray.getNElements();
142 auto doocsLen = static_cast<size_t>(dfct->length());
143 if(doocsLen != arraySize) {
144 std::cout << "Warning: Array length mismatch in property " << this->getEqFct()->name() << "/" << dfct->basename()
145 << ": Property has length " << doocsLen << " but " << arraySize << " expected." << std::endl;
146 arraySize = std::min(arraySize, size_t(doocsLen));
147 }
148 if constexpr(isSpectrum) {
149 // FIXME: find the efficient, memcopying function for float
150 // always get a fresh reference
151 for(size_t i = 0; i < arraySize; ++i) {
152 processVector[i] = dfct->read_spectrum((int)i);
153 }
154 }
155 else {
156 // Brute force implementation with a loop. Works for all data types.
157 for(size_t i = 0; i < arraySize; ++i) {
158 processVector[i] = dfct->value(i);
159 }
160 }
161 auto timestamp = dfct->get_timestamp().to_time_point();
162 processArray.write(VersionNumber(timestamp));
163
164 // Correct property length in case of a mismatch.
165 if(doocsLen != arraySize) {
166 if constexpr(isSpectrum) {
167 dfct->length(arraySize);
168 // FIXME - do we need to restore values like for arrays?
169 // it's more difficult with D_spectrum because of the buffered/unbuffered feature
170 }
171 else {
172 dfct->set_length(arraySize);
173 // restore value from ProcessArray, as it may have been destroyed in set_length().
174 for(size_t i = 0; i < arraySize; ++i) {
175 dfct->set_value(processVector[i], i);
176 }
177 }
178 }
179 }
