95  template<typename SELF, typename UserType>
96  void PropertyBase::sendArrayToDevice(
97  SELF* dfct, const boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>>& processArray) {
98  constexpr bool isSpectrum = std::is_base_of<D_spectrum, SELF>::value;
99 
100  // always get a fresh reference
101  auto& processVector = processArray->accessChannel(0);
102  size_t arraySize = processVector.size();
103  auto doocsLen = static_cast<size_t>(dfct->length());
104  if(doocsLen != arraySize) {
105  std::cout << "Warning: Array length mismatch in property " << this->getEqFct()->name() << "/" << dfct->basename()
106  << ": Property has length " << doocsLen << " but " << arraySize << " expected." << std::endl;
107  arraySize = std::min(arraySize, size_t(doocsLen));
108  }
109  if constexpr(isSpectrum) {
110  // FIXME: find the efficient, memcopying function for float
111  // always get a fresh reference
112  for(size_t i = 0; i < arraySize; ++i) {
113  processVector[i] = dfct->read_spectrum((int)i);
114  }
115  }
116  else {
117  // Brute force implementation with a loop. Works for all data types.
118  for(size_t i = 0; i < arraySize; ++i) {
119  processVector[i] = dfct->value(i);
120  }
121  }
122  auto timestamp = dfct->get_timestamp().to_time_point();
123  processArray->write(VersionNumber(timestamp));
124 
125  // Correct property length in case of a mismatch.
126  if(doocsLen != processVector.size()) {
127  if constexpr(isSpectrum) {
128  dfct->length(processVector.size());
129  // FIXME - do we need to restore values like for arrays?
130  // it's more difficult with D_spectrum because of the buffered/unbuffered feature
131  }
132  else {
133  dfct->set_length(processVector.size());
134  // restore value from ProcessArray, as it may have been destroyed in set_length().
135  for(size_t i = 0; i < arraySize; ++i) {
136  dfct->set_value(processVector[i], i);
137  }
138  }
139  }
140  }
