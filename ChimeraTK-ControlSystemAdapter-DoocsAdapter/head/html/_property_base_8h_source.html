96 template<typename SELF, typename UserType>
97 void PropertyBase::sendArrayToDevice(
98 SELF* dfct, const boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>>& processArray) {
99 constexpr bool isSpectrum = std::is_base_of<D_spectrum, SELF>::value;
100
101 // always get a fresh reference
102 auto& processVector = processArray->accessChannel(0);
103 size_t arraySize = processVector.size();
104 auto doocsLen = static_cast<size_t>(dfct->length());
105 if(doocsLen != arraySize) {
106 std::cout << "Warning: Array length mismatch in property " << this->getEqFct()->name() << "/" << dfct->basename()
107 << ": Property has length " << doocsLen << " but " << arraySize << " expected." << std::endl;
108 arraySize = std::min(arraySize, size_t(doocsLen));
109 }
110 if constexpr(isSpectrum) {
111 // FIXME: find the efficient, memcopying function for float
112 // always get a fresh reference
113 for(size_t i = 0; i < arraySize; ++i) {
114 processVector[i] = dfct->read_spectrum((int)i);
115 }
116 }
117 else {
118 // Brute force implementation with a loop. Works for all data types.
119 for(size_t i = 0; i < arraySize; ++i) {
120 processVector[i] = dfct->value(i);
121 }
122 }
123 auto timestamp = dfct->get_timestamp().to_time_point();
124 processArray->write(VersionNumber(timestamp));
125
126 // Correct property length in case of a mismatch.
127 if(doocsLen != processVector.size()) {
128 if constexpr(isSpectrum) {
129 dfct->length(processVector.size());
130 // FIXME - do we need to restore values like for arrays?
131 // it's more difficult with D_spectrum because of the buffered/unbuffered feature
132 }
133 else {
134 dfct->set_length(processVector.size());
135 // restore value from ProcessArray, as it may have been destroyed in set_length().
136 for(size_t i = 0; i < arraySize; ++i) {
137 dfct->set_value(processVector[i], i);
138 }
139 }
140 }
141 }
