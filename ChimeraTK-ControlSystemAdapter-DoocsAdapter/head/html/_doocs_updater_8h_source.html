86 template<typename UserType>
87 boost::shared_ptr<NDRegisterAccessor<UserType>> DoocsUpdater::getMappedProcessVariable(
88 const RegisterPath& processVariableName, DecoratorType decoratorType) {
89 auto pv = getMappedProcessVariableUnTyped(processVariableName);
90 if(typeid(UserType) == pv->getValueType()) {
91 return boost::dynamic_pointer_cast<NDRegisterAccessor<UserType>>(pv);
92 }
93
94 if constexpr(std::is_same_v<UserType, std::string>) {
95 // FIXME: implement TypeChangigDecorator for std::string
96 throw std::invalid_argument(
97 std::string("DoocsUpdater::getMappedProcessVariable: processArray is not of string type: ") + pv->getName());
98 }
99 else {
100 return getTypeChangingDecorator<UserType>(pv, decoratorType);
101 }
102 }
