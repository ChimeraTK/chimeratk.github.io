73 template<typename T, typename DOOCS_T>
74 void DoocsProcessScalar<T, DOOCS_T>::set(EqAdr* adr, doocs::EqData* data1, doocs::EqData* data2, EqFct* eqfct) {
75 // only assign the value if the variable is writeable
76 // Otherwise the content displayed by Doocs and the value in the application
77 // are inconsistent
78 if(!_processScalar->isWriteable()) {
79 throw ChimeraTK::logic_error("Trying to write to a non-writable variable");
80 }
81 // note, current doocs implementation does not take error code, timestamp or event id from input data
82 // it takes over data only if no error, and replaces timestamp and event id by global ones
83 DOOCS_T::set(adr, data1, data2, eqfct);
84 if(_macroPulseNumberSource != nullptr) {
85 this->set_mpnum(_macroPulseNumberSource->accessData(0));
86 }
87 // let the DOOCS_T set function do all the dirty work and use the
88 // get_value function afterwards to get the already assigned value
89 _processScalar->accessData(0) = this->value();
90 auto timestamp = DOOCS_T::get_timestamp().to_time_point();
91 _processScalar->write(VersionNumber(timestamp));
92
93 updateOthers(true);
94
95 sendZMQ(getTimestamp());
96 }

98 template<typename T, typename DOOCS_T>
99 void DoocsProcessScalar<T, DOOCS_T>::auto_init() {
100 doocsAdapter.beforeAutoInit();
101
102 DOOCS_T::auto_init();
103 // send the current value to the device
104 // property is writeable OR the target accessor is writable and the only one connected to this property
105 // The second case is to have bi-directional variables that are used to persist settings into the config file
106 // and need that value back on start-up but are not supposed to be written by the control system and mapped
107 // read-only in the configuration file.
108 if(this->get_access() == 1 || (_processScalar->isWriteable() && otherPropertiesToUpdate.empty())) {
109 _processScalar->accessData(0) = DOOCS_T::value();
110 _processScalar->write();
111 // set DOOCS time stamp, workaround for DOOCS bug (get() always gives current time stamp if no timestamp is set,
112 // which breaks consistency check in ZeroMQ subscriptions after the 4 minutes timeout)
113 DOOCS_T::set_stamp();
114
115 updateOthers(false);
116 }

119 template<typename T, typename DOOCS_T>
120 void DoocsProcessScalar<T, DOOCS_T>::updateDoocsBuffer(const TransferElementID& transferElementId) {
121 if(!updateConsistency(transferElementId)) {
122 return;
123 }
124
125 // Note: we already own the location lock by specification of the
126 // DoocsUpdater
127 auto data = _processScalar->accessData(0);
128
129 auto archiverStatus = ArchiveStatus::sts_ok;
130 if(_processScalar->dataValidity() != ChimeraTK::DataValidity::ok) {
131 archiverStatus = ArchiveStatus::sts_err;
132 // set data invalid in DOOCS for current data
133 this->d_error(stale_data);
134 }
135 else {
136 this->d_error(no_error);
137 }
138
139 doocs::Timestamp timestamp = correctDoocsTimestamp();
140
141 doocs::EventId eventId;
142 if(_macroPulseNumberSource) {
143 eventId = doocs::EventId(_macroPulseNumberSource->accessData(0));
144 }
145 this->set_value(data, timestamp, eventId, archiverStatus);
146 sendZMQ(timestamp);
147 }
