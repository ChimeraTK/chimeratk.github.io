14 unsigned char* MappedDoocsImg::asDoocsImg(IMH* headerOut) {
15 auto* h = header();
16 switch(h->image_format) {
17 case ImgFormat::Unset:
18 return nullptr;
19 case ImgFormat::Gray8:
20 case ImgFormat::Gray16:
21 headerOut->image_format = TTF2_IMAGE_FORMAT_GRAY;
22 break;
23 case ImgFormat::RGB24:
24 headerOut->image_format = TTF2_IMAGE_FORMAT_RGB;
25 break;
26 case ImgFormat::RGBA32:
27 headerOut->image_format = TTF2_IMAGE_FORMAT_RGBA;
28 break;
29 default:
30 assert(false && "image format not supported!");
31 }
32 if(!((unsigned)h->options & (unsigned)ImgOptions::RowMajor)) {
33 throw logic_error("conversion to DOOCS image only possible for row-major ordering");
34 }
35
36 headerOut->width = h->width;
37 headerOut->height = h->height;
38 headerOut->bpp = h->bytesPerPixel;
39 headerOut->image_flags = TTF2_IMAGE_LOSSLESS | TTF2_IMAGE_LITTLE_ENDIAN_BYTE_ORDER;
40 headerOut->source_format = headerOut->image_format;
41 headerOut->frame = h->frame;
42 // area of interest: in our case, always the full image
43 headerOut->aoi_width = h->width;
44 headerOut->aoi_height = h->height;
45 headerOut->length = headerOut->aoi_width * headerOut->aoi_height * headerOut->bpp;
46 headerOut->x_start = h->x_start;
47 headerOut->y_start = h->y_start;
48 headerOut->ebitpp = h->effBitsPerPixel;
49 headerOut->hbin = 1;
50 headerOut->vbin = 1;
51 headerOut->event = 0; // this will be overwritten if macropulse source was set
52 // unused values are set to -1
53 headerOut->scale_x = h->scale_x;
54 headerOut->scale_y = h->scale_y;
55 headerOut->image_rotation = 0;
56 headerOut->fspare2 = -1;
57 headerOut->fspare3 = -1;
58 headerOut->fspare4 = -1;
59 headerOut->ispare2 = -1;
60 headerOut->ispare3 = -1;
61 headerOut->ispare4 = -1;
62 return data() + sizeof(ImgHeader);
63 }

80 void DoocsImage::updateDoocsBuffer(const TransferElementID& transferElementId) {
81 if(!updateConsistency(transferElementId)) {
82 return;
83 }
84
85 D_imagec* dfct = this;
86 // Note: we already own the location lock by specification of the DoocsUpdater
87
88 ChimeraTK::OneDRegisterAccessor<uint8_t> pA(_processArray);
89 MappedDoocsImg img(pA, MappedDoocsImg::InitData::No);
90 auto* dataPtr = img.asDoocsImg(&imh);
91 if(!dataPtr) {
92 throw logic_error("data provided to DoocsImage._processArray was not recognized as image");
93 }
94
95 if(_processArray->dataValidity() != ChimeraTK::DataValidity::ok) {
96 this->d_error(stale_data);
97 }
98 else {
99 this->d_error(no_error);
100 }
101
102 doocs::Timestamp timestamp = correctDoocsTimestamp();
103
104 if(_macroPulseNumberSource) {
105 this->set_mpnum(_macroPulseNumberSource->accessData(0));
106 imh.event = _macroPulseNumberSource->accessData(0);
107 }
108
109 // this copies header and data contents to DOOCS-internal buffer
110 dfct->set_value(&imh, dataPtr);
111
112 auto ts = timestamp.get_seconds_and_microseconds_since_epoch();
113 // this is needed in addition to usual dfct call, in order to set time in img meta data
114 // note, some meta data is not correctly shown by DOOCS rpc interface but is in ZMQ.
115 dfct->set_img_time(ts.seconds, ts.microseconds);
116 // dfct->set_img_status();
117
118 sendZMQ(timestamp);
119 }
