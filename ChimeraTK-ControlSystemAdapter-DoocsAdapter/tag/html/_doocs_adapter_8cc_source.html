64  bool DoocsAdapter::checkPrintDataLossWarning(size_t counter) {
65  // print first time at counter == 10 to suppress the messages spamming at startup
66  if(counter < 1) {
67  return false;
68  }
69  if(counter < 100) {
70  return counter % 10 == 0;
71  }
72  if(counter < 1000) {
73  return counter % 100 == 0;
74  }
75  if(counter < 10000) {
76  return counter % 1000 == 0;
77  }
78  if(counter < 100000) {
79  return counter % 10000 == 0;
80  }
81  if(counter < 1000000) {
82  return counter % 100000 == 0;
83  }
84  return counter % 1000000 == 0; // if the rate is 10Hz, this is roughly once per day
85  }

89  void DoocsAdapter::beforeAutoInit() {
90  // prevent concurrent execution. It is unclear whether DOOCS may call auto_init in parallel in some situations, so
91  // better implement a lock.
92  static std::mutex mx;
93  std::unique_lock<std::mutex> lk(mx);
94 
95  // execute actions only once
96  if(_before_auto_init_called) {
97  return;
98  }
99  _before_auto_init_called = true;
100 
101  // connect properties which use the same writable PV to keep the property values consistent
102  for(auto& group : doocsAdapter.writeableVariablesWithMultipleProperties) {
103  for(const auto& weakProperty : group.second) {
104  auto property = weakProperty.lock(); // cannot fail, DoocsAdapter has ownership via PV manager
105  property->otherPropertiesToUpdate = group.second;
106  property->otherPropertiesToUpdate.erase(property); // the property should not be in its own list
107  }
108  }
109  doocsAdapter.writeableVariablesWithMultipleProperties.clear(); // save memory (information no longer needed)
110  }

114  std::unique_ptr<doocs::Server> DoocsAdapter::createServer() {
115  auto server = std::make_unique<doocs::Server>(ChimeraTK::ApplicationBase::getInstance().getName().c_str());
116 
117  server->set_init_prolog([&] { ChimeraTK::DoocsAdapter::eqInitProlog(); });
118  server->set_post_init_epilog([&] { ChimeraTK::DoocsAdapter::postInitEpilog(); });
119  server->set_cancel_epilog([&] { ChimeraTK::DoocsAdapter::eqCancel(); });
120 
121  // This is a work-around for not being able to create the same location type for any arbitrary location code
122  // found in the conf file. It works as long as the largest code is below 10000. The registration here takes
123  // around 1ms on a standard MicroTCA AMC CPU.
124  // See: https://mcs-gitlab.desy.de/doocs/doocs-core-libraries/serverlib/-/issues/79
125  for(int c = 2; c < 10000; ++c) {
126  server->register_location_class(
127  c, [=](const EqFctParameters& p) { return std::make_unique<CSAdapterEqFct>(c, p); });
128  }
129  // The DCM server uses the code 5522228 = 0x544334 = 'TC4', so we register it here as well. Need to remove this
130  // temporary hack ASAP!
131  server->register_location_class(
132  5522228, [=](const EqFctParameters& p) { return std::make_unique<CSAdapterEqFct>(5522228, p); });
133 
134  // Note: We cannot go back to use eq_create(), since we have link order issues and sometimes the wrong mplementation
135  // might win (the default implementation from the serverlib, which terminates the server with an error message).
136 
137  return server;
138  }

142  /* eq_init_prolog is called before the locations are created, i.e. before the
143  * first call to eq_create. We initialise the application, i.e. all process
144  * variables are created in this function. */
145  void DoocsAdapter::eqInitProlog() {
146  // set the DOOCS server name to the application name
147  // Create static instances for all applications cores. They must not have
148  // overlapping process variable names ("location/protery" must be unique).
149  ChimeraTK::ApplicationBase::getInstance().setPVManager(doocsAdapter.getDevicePVManager());
150  ChimeraTK::ApplicationBase::getInstance().initialise();
151 
152  // the variable manager can only be filled after we have the CS manager
153  auto pvNames = ChimeraTK::getAllVariableNames(doocsAdapter.getControlSystemPVManager());
154 
155  auto xmlFileName = ChimeraTK::ApplicationBase::getInstance().getName() + XML_CONFIG_SUFFIX;
156 
157  if(std::filesystem::exists(xmlFileName)) {
158  ChimeraTK::VariableMapper::getInstance().prepareOutput(xmlFileName, pvNames);
159  }
160  else {
161  std::cerr << "WARNING: No XML file for the Doocs variable config found. Trying direct import." << std::endl;
162  ChimeraTK::VariableMapper::getInstance().directImport(pvNames);
163  }
164 
165  // prepare list of unmapped read variables and pass it to the Application for optimisation
166  for(const auto& p : ChimeraTK::VariableMapper::getInstance().getUsedVariables()) {
167  auto it = pvNames.find(p);
168  if(it != pvNames.end()) {
169  pvNames.erase(it);
170  }
171  }
172  ChimeraTK::ApplicationBase::getInstance().optimiseUnmappedVariables(pvNames);
173 
174  // prepare list of properties connected to the same writable PV
175  {
176  // create map of PV name to properties using the PV (only for writable PVs)
177  std::map<std::string, std::set<std::shared_ptr<ChimeraTK::PropertyDescription>>> reverseMapping;
178  for(const auto& descr : ChimeraTK::VariableMapper::getInstance().getAllProperties()) {
179  for(const auto& source : descr->getSources()) {
180  if(!doocsAdapter.getControlSystemPVManager()->getProcessVariable(source)->isWriteable()) {
181  // to not add PVs to the mapping which are not writeable
182  continue;
183  }
184  reverseMapping[source].insert(descr);
185  }
186  }
187  // filter the map to contain only PVs being used at least twice with at least one writable property
188  for(const auto& p : reverseMapping) {
189  if(p.second.size() < 2) {
190  continue;
191  }
192  size_t writeableCount = 0;
193  for(const auto& d : p.second) {
194  auto attr = std::dynamic_pointer_cast<ChimeraTK::PropertyAttributes>(d);
195  if(attr->isWriteable) {
196  ++writeableCount;
197  }
198  }
199  if(writeableCount == 0) {
200  continue;
201  }
202  if(writeableCount > 1) {
203  // This case is not (yet) covered. It would require some synchronisation mechanism between writeable
204  // properties, which is not trivial as inconsistencies and dead locks must be avoided.
205  std::cout << "**** WARNING: Variable '" + p.first +
206  "' mapped to more than one writeable property. Expect inconsistencies!\n";
207  }
208 
209  // Add PVs which are used at least twice with at least one writable property to list
210  doocsAdapter.writeableVariablesWithMultipleProperties[p.first] = {};
211  }
212  }
213  }

217  /* post_init_epilog is called after all DOOCS properties are fully intialised,
218  * including any value intialisation from the config file. We start the
219  * application here. It will be launched in a separate thread. */
220  void DoocsAdapter::postInitEpilog() {
221  // check for locations not yet created (due to missing entries in the .conf file) and create them now
222  std::map<std::string, int> locMap = ChimeraTK::VariableMapper::getInstance().getLocationAndCode();
223  for(const auto& loc : ChimeraTK::VariableMapper::getInstance().getAllLocations()) {
224  int codeToSet = 10;
225  bool defaultUsed = true;
226  // if no code is set in xml file, 10 is default
227  if(locMap.find(loc) != locMap.end()) {
228  codeToSet = locMap.find(loc)->second;
229  defaultUsed = false;
230  }
231 
232  auto* eq = find_device(loc);
233  if(eq == nullptr) {
234  add_location(codeToSet, loc);
235  }
236  else if(eq->fct_code() != codeToSet && !defaultUsed) {
237  throw ChimeraTK::logic_error("Location '" + loc + "' has already a fct_code '" +
238  std::to_string(eq->fct_code()) + "' from config file, that does not match with '" +
239  std::to_string(codeToSet) + "' from xml file");
240  }
241  }
242 
243  // check for variables not yet initialised - we must guarantee that all to-application variables are written exactly
244  // once at server start.
245  for(auto& pv : doocsAdapter.getControlSystemPVManager()->getAllProcessVariables()) {
246  if(!pv->isWriteable()) {
247  continue;
248  }
249  if(pv->getVersionNumber() == ChimeraTK::VersionNumber(nullptr)) {
250  // The variable has not yet been written. Do it now, even if we just send a 0.
251  pv->write();
252  }
253  }
254 
255  // start the application and the updater
256  ChimeraTK::ApplicationBase::getInstance().run();
257  doocsAdapter.updater->run();
258  ChimeraTK::DoocsAdapter::isInitialised = true;
259  }
