20 struct PersistConfig {
21 enum {
22 OFF = 0,
23 ON = 1, // write long arrays to separate files below hist/, short arrays to config file
24 AUTO = 2 // default DOOCS behaviour: persist only short arrays, in config file
25 };
26 int val;
27
28 // NOLINTNEXTLINE(google-explicit-constructor)
29 PersistConfig(std::string txt) {
30 transform(txt.begin(), txt.end(), txt.begin(), ::tolower);
31 if(txt == "false" or txt == "off" or txt == "0") {
32 val = OFF;
33 }
34 else if(txt == "true" or txt == "on" or txt == "1") {
35 val = ON;
36 }
37 else if(txt == "auto") {
38 val = AUTO;
39 }
40 else {
41 throw std::invalid_argument(std::string("Error parsing xml file: invalid input for PersistConfig: ") + txt);
42 }
43 }
44 // NOLINTNEXTLINE(google-explicit-constructor)
45 PersistConfig(int enumVal) { this->val = enumVal; }
46 bool operator==(const PersistConfig& other) const { return val == other.val; }
47 };

76 // Common for all properties, the base class to be stored.
77 // FIXME: should sort by name to put it into a set?
78 struct PropertyDescription : public PropertyAttributes {
79 std::string location;
80 std::string name;
81 explicit PropertyDescription(
82 std::string location_ = "", std::string name_ = "", const PropertyAttributes& propAttr = PropertyAttributes{})
83 : PropertyAttributes(propAttr), location(std::move(location_)), name(std::move(name_)) {}
84 virtual ~PropertyDescription() = default;
85 virtual bool operator==(PropertyDescription const& other) const {
86 return location == other.location && name == other.name;
87 }
88 virtual void print(std::ostream& os = std::cout) const { os << location << " / " << name << std::endl; }
89
90 // Return all PV names which are used (read or write) by the property.
91 std::set<std::string> getSources() {
92 auto sources = getMetadataSources();
93 auto payload = getPayloadDataSources();
94 sources.insert(payload.begin(), payload.end());
95 return sources;
96 }
97
98 std::set<std::string> getReadSources() {
99 auto sources = getMetadataSources();
100 if(!isWriteable) {
101 auto payload = getPayloadDataSources();
102 sources.insert(payload.begin(), payload.end());
103 }
104 return sources;
105 }
106
107 std::set<std::string> getWriteSources() {
108 if(isWriteable) {
109 return getPayloadDataSources();
110 }
111 return {};
112 }
113
114 virtual std::set<std::string> getMetadataSources() {
115 std::set<std::string> sources;
116 if(!macroPulseNumberSource.empty()) {
117 sources.insert(getAbsoluteSource(macroPulseNumberSource, location));
118 }
119 if(!isWriteableSource.empty()) {
120 sources.insert(getAbsoluteSource(isWriteableSource, location));
121 }
122 return sources;
123 }
124
125 virtual std::set<std::string> getPayloadDataSources() = 0;
126 };

132 struct AutoPropertyDescription : public PropertyDescription {
133 enum class DataType { Byte, Short, Int, Long, Float, Double, Bool, Void, Auto };
134 ChimeraTK::RegisterPath source;
135 explicit AutoPropertyDescription(ChimeraTK::RegisterPath const& source_ = "", std::string location_ = "",
136 std::string name_ = "", DataType dataType_ = DataType::Auto, bool hasHistory_ = true, bool isWriteable_ = true)
137 : PropertyDescription(std::move(location_), std::move(name_), PropertyAttributes(hasHistory_, isWriteable_)),
138 source(source_), dataType(dataType_) {}
139 bool operator==(PropertyDescription const& other) const override {
140 if(typeid(other) == typeid(AutoPropertyDescription)) {
141 auto casted_other = dynamic_cast<AutoPropertyDescription const&>(other);
142 return dataType == casted_other.dataType && source == casted_other.source && location == other.location &&
143 name == other.name && static_cast<const PropertyAttributes*>(this)->operator==(casted_other);
144 }
145 return false;
146 }
147 void print(std::ostream& os = std::cout) const override {
148 os << source << " -> " << location << " / " << name << std::endl;
149 }
150
151 void deriveType(std::type_info const& info) {
152 if(info == typeid(uint8_t) || info == typeid(int8_t)) {
153 dataType = DataType::Byte;
154 }
155 if(info == typeid(uint16_t) || info == typeid(int16_t)) {
156 dataType = DataType::Short;
157 }
158 if(info == typeid(uint32_t) || info == typeid(int32_t)) {
159 dataType = DataType::Int;
160 }
161 if(info == typeid(uint64_t) || info == typeid(int64_t)) {
162 dataType = DataType::Long;
163 }
164 if(info == typeid(float)) {
165 dataType = DataType::Float;
166 }
167 if(info == typeid(double)) {
168 dataType = DataType::Double;
169 }
170 if(info == typeid(ChimeraTK::Boolean)) {
171 dataType = DataType::Bool;
172 }
173 if(info == typeid(ChimeraTK::Void)) {
174 dataType = DataType::Void;
175 }
176 }
177
178 std::set<std::string> getPayloadDataSources() override { return {source}; };
179
180 DataType dataType;
181 };
