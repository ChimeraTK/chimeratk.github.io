87// the array must have testStartValue+i at index i.
88template<class T>
89static bool testArrayContent(std::vector<T>& array, std::string const& propertyName, T testStartValue, T delta) {
90 bool isOK = true;
91 T currentTestValue = testStartValue;
92 size_t index = 0;
93 for(auto val : array) {
94 if(std::fabs(val - currentTestValue) > 0.001) {
95 if(isOK) {
96 std::cout << "Array " << propertyName << " does not contain expected values. First mismatching data at index "
97 << index << ": " << val << " != " << currentTestValue << std::endl;
98 }
99 isOK = false;
100 }
101 currentTestValue += delta;
102 ++index;
103 }
104 return isOK;
105}
106template<class T>
107static bool testArrayContentFromProperty(std::string const& propertyName, T testStartValue, T delta) {
108 auto array = DoocsServerTestHelper::doocsGetArray<T>(propertyName);
109 return testArrayContent(array, propertyName, testStartValue, delta);
110}

131

134 vint[i] = 150 + i;
135 vdouble[i] = 250.3 + i;
136 }

138 DoocsServerTestHelper::doocsSet("//DOUBLE/TO_DEVICE_ARRAY", vdouble);
139
140 // run the application loop to propagate the change to the FROM_DEVICE_ARRAY properties
141 GlobalFixture::referenceTestApplication.runMainLoopOnce();
142
143 // check changed values
144 usleep(100000);
145 CHECK_WITH_TIMEOUT(testArrayContentFromProperty<float>("//INT/FROM_DEVICE_ARRAY", 150, 1) == true);
146 CHECK_WITH_TIMEOUT(testArrayContentFromProperty<float>("//DOUBLE/FROM_DEVICE_ARRAY", 250.3, 1) == true);
147
148 // change value of TO_DEVICE_ARRAY properties, this time without propagating it through
149 for(int i = 0; i < alen; i++) {
150 vint[i] = 42 + i;
151 vdouble[i] = 20.20 + i;
152 }
153 DoocsServerTestHelper::doocsSet("//INT/TO_DEVICE_ARRAY", vint);
154 DoocsServerTestHelper::doocsSet("//DOUBLE/TO_DEVICE_ARRAY", vdouble);
155
156 // trigger array storing
157 DoocsServerTestHelper::doocsSet<bool>("//ARRAY_PERSISTENCE_TEST._SVR/SVR.SAVE", true);
158
159 // check persisted file against known values. Try several times, reading from disk.
160 bool intArrayRestored = false;
161 bool doubleArrayRestored = false;
162 bool intArrayRestored2 = false;
163 bool doubleArrayRestored2 = false;
164 for(int count = 0; count < 100; ++count) {
165 usleep(100000);
166 std::vector<int> vint2(alen), vint3(alen);
167 std::vector<double> vdouble2(alen), vdouble3(alen);
168 vectorFromFile(vint2, "hist/INT-TO_DEVICE_ARRAY.intarray");
169 vectorFromFile(vint3, "hist/INT-FROM_DEVICE_ARRAY.intarray");
170 vectorFromFile(vdouble2, "hist/DOUBLE-TO_DEVICE_ARRAY.doublearray");
171 vectorFromFile(vdouble3, "hist/DOUBLE-FROM_DEVICE_ARRAY.doublearray");
172 intArrayRestored = testArrayContent<int>(vint2, "//INT/TO_DEVICE_ARRAY from file", 42, 1);
173 intArrayRestored2 = testArrayContent<int>(vint3, "//INT/FROM_DEVICE_ARRAY from file", 150, 1);
174 doubleArrayRestored = testArrayContent<double>(vdouble2, "//DOUBLE/TO_DEVICE_ARRAY from file", 20.20, 1);
175 doubleArrayRestored2 = testArrayContent<double>(vdouble3, "//DOUBLE/FROM_DEVICE_ARRAY from file", 250.3, 1);
176 if(intArrayRestored && intArrayRestored2 && doubleArrayRestored && doubleArrayRestored2) {
177 std::cout << "arrays recovered after read tries: " << count + 1 << std::endl;
178 break;
179 }
180 }
181 BOOST_CHECK(intArrayRestored);
182 BOOST_CHECK(intArrayRestored2);
183 BOOST_CHECK(doubleArrayRestored);
184 BOOST_CHECK(doubleArrayRestored2);
185}
