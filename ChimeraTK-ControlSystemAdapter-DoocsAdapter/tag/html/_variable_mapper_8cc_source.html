37  void VariableMapper::processLocationNode(xmlpp::Node const* locationNode) {
38  const xmlpp::Element* location = dynamic_cast<const xmlpp::Element*>(locationNode);
39  if(!location) {
40  throw std::invalid_argument("Error parsing xml file in location node.");
41  }
42  std::string locationName = getAttributeValue(location, "name");
43 
44  processCode(location, locationName);
45 
46  for(auto const& node : location->get_children()) {
47  if(nodeIsWhitespace(node)) continue;
48  if(dynamic_cast<xmlpp::CommentNode const*>(node)) continue;
49 
50  if(node->get_name() == "property") {
51  processNode(node, locationName);
52  }
53  else if(node->get_name() == "import") {
54  processImportNode(node, locationName);
55  }
56  else if(node->get_name() == "has_history") {
57  auto& locationInfo = _locationDefaults[locationName];
58  locationInfo.useHasHistoryDefault = true;
59  locationInfo.hasHistory = evaluateBool(getContentString(node));
60  }
61  else if(node->get_name() == "persist") {
62  auto& locationInfo = _locationDefaults[locationName];
63  locationInfo.usePersistDefault = true;
64  locationInfo.persist = PersistConfig(getContentString(node));
65  }
66  else if(node->get_name() == "is_writeable") {
67  auto& locationInfo = _locationDefaults[locationName];
68  locationInfo.useIsWriteableDefault = true;
69  locationInfo.isWriteable = evaluateBool(getContentString(node));
70  }
71  else if(node->get_name() == "macro_pulse_number_source") {
72  auto& locationInfo = _locationDefaults[locationName];
73  locationInfo.useMacroPulseNumberSourceDefault = true;
74  locationInfo.macroPulseNumberSource = getContentString(node);
75  }
76  else if(node->get_name() == "data_matching") {
77  auto& locationInfo = _locationDefaults[locationName];
78  locationInfo.useDataMatchingDefault = true;
79  locationInfo.dataMatching = evaluateDataMatching(getContentString(node));
80  }
81  else if(node->get_name() == "D_spectrum") {
82  processSpectrumNode(node, locationName);
83  }
84  else if(node->get_name() == "D_imagec") {
85  processImageNode(node, locationName);
86  }
87  else if(node->get_name() == "D_array") {
88  processNode(node, locationName);
89  }
90  else if(node->get_name() == "D_xy") {
91  processXyNode(node, locationName);
92  }
93  else if(node->get_name() == "D_ifff") {
94  processIfffNode(node, locationName);
95  }
96  else if(node->get_name() == "set_error") {
97  processSetErrorNode(node, locationName);
98  }
99  else {
100  throw std::invalid_argument(std::string("Error parsing xml file in location ") + locationName +
101  ": Unknown node '" + node->get_name() + "'");
102  }
103  }
104  }

108  std::string determineName(const xmlpp::Element* property, std::string source) {
109  const xmlpp::Attribute* nameAttribute = property->get_attribute("name");
110  // if a name is given in xml take it
111  if(nameAttribute) {
112  return nameAttribute->get_value();
113  }
114  else { // auto-determine the name from the source
116  std::string name;
117  if(source[0] == '/') {
118  name = source.substr(1);
119  }
120  else {
121  name = source;
122  }
123  // replace / with . in name
124  return std::regex_replace(name, std::regex("/"), ".");
125  }
126  }

215  void VariableMapper::processNode(xmlpp::Node const* propertyNode, std::string locationName) {
216  auto property = asXmlElement(propertyNode);
217 
218  std::string source = getAttributeValue(property, "source");
219  std::string name = determineName(property, source);
220 
221  const xmlpp::Attribute* typeAttribute = property->get_attribute("type");
222  std::map<std::string, AutoPropertyDescription::DataType> dataTypeMap(
223  {{"auto", AutoPropertyDescription::DataType::Auto}, {"byte", AutoPropertyDescription::DataType::Byte},
224  {"short", AutoPropertyDescription::DataType::Short}, {"int", AutoPropertyDescription::DataType::Int},
225  {"long", AutoPropertyDescription::DataType::Long}, {"float", AutoPropertyDescription::DataType::Float},
226  {"double", AutoPropertyDescription::DataType::Double}, {"bool", AutoPropertyDescription::DataType::Bool}});
227 
228  auto type = AutoPropertyDescription::DataType::Auto;
229 
230  if(typeAttribute) {
231  try {
232  type = dataTypeMap.at(typeAttribute->get_value());
233  }
234  catch(std::out_of_range&) {
235  throw ChimeraTK::logic_error("Unknown type attribute in line " + std::to_string(property->get_line()) + ": " +
236  typeAttribute->get_value());
237  }
238  }
239 
240  std::string absoluteSource = getAbsoluteSource(source, locationName);
241 
242  // prepare the property description
243  auto autoPropertyDescription = std::make_shared<AutoPropertyDescription>(absoluteSource, locationName, name, type);
244 
245  processHistoryAndWritableAttributes(autoPropertyDescription, property, locationName);
246 
247  addDescription(autoPropertyDescription, {absoluteSource});
248  }

252  void VariableMapper::processSpectrumNode(xmlpp::Node const* node, std::string locationName) {
253  auto spectrumXml = asXmlElement(node);
254 
255  // the "main source" of a spectum
256  std::string source = getAttributeValue(spectrumXml, "source");
257  std::string name = determineName(spectrumXml, source);
258  std::string absoluteSource = getAbsoluteSource(source, locationName);
259 
260  // prepare the property description
261  auto spectrumDescription = std::make_shared<SpectrumDescription>(absoluteSource, locationName, name);
262 
263  processHistoryAndWritableAttributes(spectrumDescription, spectrumXml, locationName);
264 
265  auto startNodes = spectrumXml->get_children("start");
266  if(!startNodes.empty()) {
267  spectrumDescription->start = std::stof(getContentString(startNodes.front()));
268  }
269  std::list<std::string> usedVariables({absoluteSource});
270  auto incrementNodes = spectrumXml->get_children("increment");
271  if(!incrementNodes.empty()) {
272  spectrumDescription->increment = std::stof(getContentString(incrementNodes.front()));
273  }
274  auto startSourceNodes = spectrumXml->get_children("startSource");
275  if(!startSourceNodes.empty()) {
276  auto startSource = getContentString(startSourceNodes.front());
277  spectrumDescription->startSource = startSource;
278  usedVariables.push_back(startSource);
279  }
280  auto incrementSourceNodes = spectrumXml->get_children("incrementSource");
281  if(!incrementSourceNodes.empty()) {
282  auto incrementSource = getContentString(incrementSourceNodes.front());
283  spectrumDescription->incrementSource = incrementSource;
284  usedVariables.push_back(incrementSource);
285  }
286  auto numberOfBuffersNodes = spectrumXml->get_children("numberOfBuffers");
287  if(!numberOfBuffersNodes.empty()) {
288  auto numberOfBuffers = getContentString(numberOfBuffersNodes.front());
289  spectrumDescription->numberOfBuffers = std::stoi(numberOfBuffers);
290  usedVariables.push_back(numberOfBuffers);
291  }
292 
293  auto descriptionNode = spectrumXml->get_first_child("description");
294  if(descriptionNode != nullptr) spectrumDescription->description = getContentString(descriptionNode);
295 
296  auto unitNodes = spectrumXml->get_children("unit");
297  for(const auto unit : unitNodes) {
298  auto unitElement = asXmlElement(unit);
299  auto axis = getAttributeValue(unitElement, "axis");
300  if(axis != "x" && axis != "y")
301  throw std::invalid_argument("Unsupported axis in D_spectrum, must be \"x\" or \"y\": " + axis);
302 
303  std::string label;
304  if(not unit->get_children().empty()) label = getContentString(unit);
305 
306  spectrumDescription->axis[axis].label = label;
307  try {
308  spectrumDescription->axis[axis].logarithmic = std::stoi(getAttributeValue(unitElement, "logarithmic"));
309  }
310  catch(std::invalid_argument&) {
311  }
312 
313  try {
314  spectrumDescription->axis[axis].start = std::stof(getAttributeValue(unitElement, "start"));
315  }
316  catch(std::invalid_argument&) {
317  }
318 
319  try {
320  spectrumDescription->axis[axis].stop = std::stof(getAttributeValue(unitElement, "stop"));
321  }
322  catch(std::invalid_argument&) {
323  }
324  }
325  addDescription(spectrumDescription, usedVariables);
326  }

465  void VariableMapper::import(std::string importSource, std::string importLocationName, std::string directory) {
466  // a slash will be added after the source, so we make the source empty for an
467  // import of everything
468  if(importSource == "/") {
469  importSource = "";
470  }
471 
472  // loop source tree, cut beginning, replace / with _ and add a property
473  for(auto const& processVariable : _inputVariables) {
474  if(_usedInputVariables.find(processVariable) != _usedInputVariables.end()) {
475  continue;
476  }
477 
478  if(processVariable.find(importSource + "/") == 0) {
479  // processVariable starts with wanted source
480  auto nameSource = processVariable.substr(importSource.size() + 1); // add the slash to be removed
481  // we use the register path because it removes duplicate separators and
482  // allows to use . as separater to replace all / with .
483  ChimeraTK::RegisterPath propertyName;
484  std::string locationName;
485  if(importLocationName.empty()) {
486  // a global import, try to get the location name from the source
487  auto locationAndPropertyName = splitStringAtFirstSlash(nameSource);
488  locationName = locationAndPropertyName.first;
489  propertyName = locationAndPropertyName.second;
490  if(locationName.empty()) {
491  throw std::logic_error(std::string("Invalid XML content in global import of ") +
492  (importSource.empty() ? "/" : importSource) + ": Cannot create location name from '" + nameSource +
493  "', one hirarchy level is missing.");
494  }
495  // convenience for the user: You get an error message is you try a
496  // global import with directory (in case you did not validate your xml
497  // against the schema).
498  if(!directory.empty()) {
499  throw std::logic_error(std::string("Invalid XML content in global import of ") +
500  (importSource.empty() ? "/" : importSource) + ": You cannot have a directory in a global import.");
501  }
502  }
503  else {
504  // import into a location, we know the location name.
505  // add the directory first, then add the name source property
506  propertyName /= directory;
507  propertyName /= nameSource;
508  locationName = importLocationName;
509  }
510 
511  // get the property name with . instead of /
512  propertyName.setAltSeparator(".");
513  auto autoPropertyDescription = std::make_shared<AutoPropertyDescription>(
514  processVariable, locationName, propertyName.getWithAltSeparator());
515 
516  // we are importing, so all properties get the intended defaults (not
517  // individual settings)
518  autoPropertyDescription->hasHistory = getHasHistoryDefault(locationName);
519  autoPropertyDescription->isWriteable = getIsWriteableDefault(locationName);
520  autoPropertyDescription->persist = getPersistDefault(locationName);
521  autoPropertyDescription->macroPulseNumberSource = getMacroPusleNumberSourceDefault(locationName);
522  autoPropertyDescription->dataMatching = getDataMatchingDefault(locationName);
523 
524  addDescription(autoPropertyDescription, {processVariable});
525  }
526  }
527  }

531  void VariableMapper::processCode(xmlpp::Element const* location, std::string locationName) {
532  // If the the code is set in the location tag of the mapping xml file,
533  // this function adds the location and fct_code to a map _inputLocationAndCode.
534  // It throws an exeption if the code is not an integer. The code must be greater than 1,
535  // because 1 is reserved for the server. And the code must be consistent.
536  auto locationCodeAttribute = location->get_attribute("code");
537  if(locationCodeAttribute) {
538  int locationCode = 0;
539  try {
540  locationCode = std::stoi(locationCodeAttribute->get_value());
541  }
542  catch(std::invalid_argument&) {
543  throw std::invalid_argument(std::string("Error parsing xml file in location ") + locationName +
544  ": Unknown code '" + locationCodeAttribute->get_value() + "'");
545  }
546  if(locationCode != 10) {
547  throw std::invalid_argument(std::string("Error parsing xml file in location ") + locationName + ": code '" +
548  locationCodeAttribute->get_value() + "' must be always equal to 10 now (or just left out), sorry!");
549  }
550 
551  auto result = _inputLocationAndCode.insert(std::pair<std::string, int>(locationName, locationCode));
552  if(result.second == false) { // test if pair is already in map
553  if(result.first->second != locationCode) { // and test if code is the same like before
554  // maybe an exection is too much and a warning is enough?
555  throw std::invalid_argument(std::string("Error parsing xml file in location ") + locationName + ": code '" +
556  std::to_string(locationCode) + "' must be consistent with location code before '" +
557  std::to_string(result.first->second) + "'");
558  }
559  }
560  }
561  }

564 
565  void VariableMapper::prepareOutput(std::string xmlFile, std::set<std::string> inputVariables) {
566  clear();
567  _inputVariables = inputVariables;
568 
569  xmlpp::DomParser parser;
570  // parser.set_validate();
571  parser.set_substitute_entities(); // We just want the text to be
572  // resolved/unescaped automatically.
573  parser.parse_file(xmlFile);
574 
575  if(parser) {
576  // Walk the tree:
577  const xmlpp::Node* rootNode = parser.get_document()->get_root_node(); // deleted by DomParser.
578 
579  for(auto const& mainNode : rootNode->get_children()) {
580  if(nodeIsWhitespace(mainNode)) continue;
581  if(dynamic_cast<xmlpp::CommentNode const*>(mainNode)) continue;
582 
583  if(mainNode->get_name() == "location") {
584  processLocationNode(mainNode);
585  }
586  else if(mainNode->get_name() == "import") {
587  processImportNode(mainNode);
588  }
589  else if(mainNode->get_name() == "has_history") {
590  _globalDefaults.hasHistory = evaluateBool(getContentString(mainNode));
591  }
592  else if(mainNode->get_name() == "is_writeable") {
593  _globalDefaults.isWriteable = evaluateBool(getContentString(mainNode));
594  }
595  else if(mainNode->get_name() == "persist") {
596  _globalDefaults.persist = PersistConfig(getContentString(mainNode));
597  }
598  else if(mainNode->get_name() == "macro_pulse_number_source") {
599  _globalDefaults.macroPulseNumberSource = getContentString(mainNode);
600  }
601  else if(mainNode->get_name() == "data_matching") {
602  _globalDefaults.dataMatching = evaluateDataMatching(getContentString(mainNode));
603  }
604  else {
605  throw std::invalid_argument(
606  std::string("Error parsing xml file ") + xmlFile + ": Unknown node '" + mainNode->get_name() + "'");
607  }
608  }
609  }
610  else {
611  throw std::invalid_argument(std::string("Error parsing xml file ") + xmlFile + ". No document produced.");
612  }
613  }

623  std::list<std::shared_ptr<PropertyDescription>> VariableMapper::getPropertiesInLocation(std::string location) const {
624  std::list<std::shared_ptr<PropertyDescription>> output;
