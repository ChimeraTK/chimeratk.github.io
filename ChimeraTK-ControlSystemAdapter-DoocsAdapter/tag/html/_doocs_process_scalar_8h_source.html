67  void DoocsProcessScalar<T, DOOCS_T>::set(EqAdr* adr, doocs::EqData* data1, doocs::EqData* data2, EqFct* eqfct) {
68  // only assign the value if the variable is writeable
69  // Otherwise the content displayed by Doocs and the value in the application
70  // are inconsistent
71  if(!_processScalar->isWriteable()) {
72  throw ChimeraTK::logic_error("Trying to write to a non-writable variable");
73  }
74  // note, current doocs implementation does not take error code, timestamp or event id from input data
75  // it takes over data only if no error, and replaces timestamp and event id by global ones
76  DOOCS_T::set(adr, data1, data2, eqfct);
77  if(_macroPulseNumberSource != nullptr) {
78  this->set_mpnum(_macroPulseNumberSource->accessData(0));
79  }
80  // let the DOOCS_T set function do all the dirty work and use the
81  // get_value function afterwards to get the already assigned value
82  _processScalar->accessData(0) = this->value();
83  auto timestamp = DOOCS_T::get_timestamp().to_time_point();
84  _processScalar->write(VersionNumber(timestamp));
85 
86  updateOthers(true);
87 
88  sendZMQ(getTimestamp());
89  }

132  this->set_value(data, timestamp, eventId, archiverStatus);
