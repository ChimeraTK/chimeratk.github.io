19 void PropertyBase::registerVariable(TransferElementAbstractor& var, bool update) {
20 if(var.isReadable()) {
21 auto id = var.getId();
22 _consistencyGroup.add(var);
23 if(update) {
24 _doocsUpdater.addVariable(var, getEqFct(), [this, id] { return updateDoocsBuffer(id); });

34 bool PropertyBase::updateConsistency(const TransferElementID& updatedId) {
35 // FIXME: A first implementation is checking the data consistency here. Later this should be
36 // before calling this function because calling this function through a function pointer is
37 // comparatively expensive.
38
39 // Do not check if update is coming from another DOOCS property mapped to the same variable (ID invalid), since
40 // the check would never pass. Such variables cannot use exact data matching anyway, since the update is triggered
41 // from the DOOCS write to the other property.
42 // Also do not check, if data matching turned off
43 if(!updatedId.isValid() || _consistencyGroup.getMatchingMode() == DataConsistencyGroup::MatchingMode::none) {
44 _doocsSuccessfullyUpdated = true;
45 return true;
46 }
47 assert(_outputVarForVersionNum);
48 TransferElementID compareTo = _outputVarForVersionNum->getId();
49
50 if(!_consistencyGroup.update(updatedId)) {
51 // data is not consistent (yet). Don't update the Doocs buffer.
52 // check if this will now throw away data and generate a warning
53 if(updatedId == compareTo) {
54 if(!_doocsSuccessfullyUpdated) {
55 ++_nDataLossWarnings;
56 if(DoocsAdapter::checkPrintDataLossWarning(_nDataLossWarnings)) {
57 const char* reason = (bool)(_macroPulseNumberSource) ?
58 " due to failed data matching between value and macro pulse number" :
59 " due to failed data matching between values";
60
61 std::cout << "WARNING: Data loss in property " << getEqFct()->name() << "/" << _doocsPropertyName << reason
62 << " (repeated " << _nDataLossWarnings << " times)." << std::endl;
63 }
64 }
65 // Tracking that data is modified is only done for the "compareTo" variable.
66 // If we want to detect data loss on all variables in the consistency group, we would have to track this for
67 // each variable separately.
68 _doocsSuccessfullyUpdated = false;
69 }
70 return false;
71 }
72 _doocsSuccessfullyUpdated = true;
73 return true;
74 }

86 doocs::Timestamp PropertyBase::correctDoocsTimestamp() {
87 doocs::Timestamp timestamp = getTimestamp();
88 auto* d_fct = getDfct();
89 // Make sure we never send out two absolute identical time stamps. If we would do so, the "watchdog" which
90 // corrects inconsistencies in ZeroMQ subscriptions between sender and subscriber cannot detect the inconsistency.
91 // This check also enforces that the timestamp is always increasing (never going backwards in time)
92 if(timestamp <= d_fct->get_timestamp()) {
93 timestamp = d_fct->get_timestamp() + std::chrono::microseconds(1);
94 }
95
96 // update global time stamp of DOOCS, but only if our time stamp is newer
97 if(get_global_timestamp() < timestamp) {
98 set_global_timestamp(timestamp);
99 }
100 d_fct->set_timestamp(timestamp);
101 return timestamp;
102 }

106 void PropertyBase::sendZMQ(doocs::Timestamp timestamp) {
107 auto* d_fct = getDfct();
108 // send data via ZeroMQ if enabled and if DOOCS initialisation is complete
109 if(_publishZMQ && ChimeraTK::DoocsAdapter::isInitialised) {
110 dmsg_info info{};
111 auto sinceEpoch = timestamp.get_seconds_and_microseconds_since_epoch();
112 info.sec = sinceEpoch.seconds;
113 info.usec = sinceEpoch.microseconds;
114 if(_macroPulseNumberSource.isInitialised()) {
115 info.ident = _macroPulseNumberSource;
116 }
117 else {
118 info.ident = 0;
119 }
120 dmsg_error(&info, d_fct->d_error());
121 auto ret = d_fct->send(&info);
122 if(ret) {
123 std::cout << "ZeroMQ sending failed!!!" << std::endl;
124 }
125 }

130 void PropertyBase::updateOthers(bool handleLocking) {
131 // Release the location lock before calling callbacks to avoid deadlocks when callbacks lock other locations
132 if(handleLocking) {
133 getEqFct()->unlock();
134 }
135 // Invoke all registered callbacks, passing this property as the caller so callbacks can identify the source
136 for(const auto& callback : callbacksOnChange()) {
137 callback(handleLocking, shared_from_this());
138 }
139 if(handleLocking) {
140 getEqFct()->lock();
141 }
142 }

146 void PropertyBase::setMacroPulseNumberSource(const std::string& sourcePath) {
147 if(!sourcePath.empty()) {
148 auto mpnSource = _doocsUpdater.getMappedProcessVariable<int64_t>(sourcePath);
149 if(mpnSource->getNumberOfSamples() != 1) {
150 throw ChimeraTK::logic_error("The property '" + mpnSource->getName() +
151 "' is used as a macro pulse number source, but it has an array length of " +
152 std::to_string(mpnSource->getNumberOfSamples()) + ". Length must be exactly 1");
153 }
154 if(!mpnSource->isReadable()) {
155 throw ChimeraTK::logic_error("The property '" + mpnSource->getName() +
156 "' is used as a macro pulse number source, but it is not readable.");
157 }
158 setMacroPulseNumberSource(mpnSource);
159 }

175 void PropertyBase::setIsWriteableSource(const std::string& sourcePath) {
176 if(!sourcePath.empty()) {
177 auto isWriteableSource = _doocsUpdater.getMappedProcessVariable<ChimeraTK::Boolean>(sourcePath);
178 if(isWriteableSource->getNumberOfSamples() != 1) {
179 throw ChimeraTK::logic_error(std::format(
180 "The property '{}' is used as a is-writeable source, but is not a scalar.", isWriteableSource->getName()));
181 }
182
183 if(isWriteableSource->isReadable()) {
184 // Readable PV (device-to-CS direction): register with DoocsUpdater to receive updates via the update loop
185 _isWriteableSource.replace(isWriteableSource);
186 _doocsUpdater.addVariable(_isWriteableSource, getEqFct(), [this] {
187 if(bool(_isWriteableSource)) {
188 this->getDfct()->set_rw_access();
189 }
190 else {
191 this->getDfct()->set_ro_access();
192 }
193 });
194 }
195 else {
196 // Write-only PV (CS-to-device direction): cannot use DoocsUpdater since that only handles readable PVs.
197 // Instead, register a callback in writeableVariablesWithMultipleProperties so it gets called when another
198 // property writes to the same PV (via updateOthers). The callback reads the current value from the caller's
199 // DOOCS property using the generic EqData interface.
200 assert(!doocsAdapter.writeableVariablesWithMultiplePropertiesIsFinal);
201 auto weakSelf = weak_from_this();
202 doocsAdapter.writeableVariablesWithMultipleProperties[sourcePath].emplace_back(
203 [weakSelf](bool handleLocking, const boost::shared_ptr<PropertyBase>& caller) {
204 auto self = weakSelf.lock();
205 if(!self) {
206 return;
207 }
208 EqData input, result;
209 caller->getDfct()->get(nullptr, &input, &result, caller->getEqFct());
210 bool value = result.get_bool();
211 if(handleLocking) {
212 self->getEqFct()->lock();
213 }
214 if(value) {
215 self->getDfct()->set_rw_access();
216 }
217 else {
218 self->getDfct()->set_ro_access();
219 }
220 if(handleLocking) {
221 self->getEqFct()->unlock();
222 }
223 });
224 }
225 }

230 void PropertyBase::subscribeToSharedPV(const std::string& pvName) {
231 // Register a callback that updates this property's DOOCS buffer when another property writes to the shared PV.
232 // Uses weak_ptr to avoid preventing destruction of this property.
233 auto weakSelf = weak_from_this();
234 doocsAdapter.writeableVariablesWithMultipleProperties[pvName].emplace_back(
235 [weakSelf](bool handleLocking, const boost::shared_ptr<PropertyBase>& caller) {
236 auto self = weakSelf.lock();
237 if(!self) {
238 return;
239 }
240 assert(caller != self);
241
242 if(handleLocking) {
243 self->getEqFct()->lock();
244 }
245 self->updateDoocsBuffer({});
246 if(handleLocking) {
247 self->getEqFct()->unlock();
248 }
249 });
250 }

254 PVChangeListeners& PropertyBase::callbacksOnChange() {
255 if(_callbacksCacheIsFinal) {
256 return _callbacksOnChange_cache;
257 }
258
259 // Collect all callbacks from PV groups this property is subscribed to. This includes both data-property
260 // callbacks (from subscribeToSharedPV) and isWriteableSource callbacks (from setIsWriteableSource).
261 _callbacksOnChange_cache.clear();
262 _hasOtherDataProperties = false;
263 for(auto& [pvName, listeners] : doocsAdapter.writeableVariablesWithMultipleProperties) {
264 if(!_sharedPVSubscriptions.contains(pvName)) {
265 continue;
266 }
267 _callbacksOnChange_cache.insert(_callbacksOnChange_cache.end(), listeners.begin(), listeners.end());
268 // More than one listener means there are other data properties besides just an isWriteableSource callback
269 if(listeners.size() > 1) {
270 _hasOtherDataProperties = true;
271 }
272 }
273 if(doocsAdapter.writeableVariablesWithMultiplePropertiesIsFinal) {
274 _callbacksCacheIsFinal = true;
275 }
276 return _callbacksOnChange_cache;
277 }
