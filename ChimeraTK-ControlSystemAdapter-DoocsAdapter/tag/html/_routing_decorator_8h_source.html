65 template<typename UserType>
66 RoutingDecorator<UserType>::RoutingDecorator(const Acc& target)
67 : NDRegisterAccessorDecorator<UserType, UserType>::NDRegisterAccessorDecorator(target) {
68 // We need unique and stable TransferElementId (differently from usual decorator behavior).
69 // This is required to distinguish between updates for source and updates for copies (from ReadAnyGroup).
70 this->_id = TransferElementID();
71 this->makeUniqueId();
72 }

74 template<typename UserType>
75 void RoutingDecorator<UserType>::setupFan() {
76 assert(!_isFan);
77 // assert that we do decoration with DataConsistencyDecorator only after setupFan.
78 // DataConsistencyDecorator implements a continuation of the readQueue, so the latter must not be
79 // exchanged later.
80 assert(!_thisIsDecorated);
81
82 std::size_t size = this->getNumberOfSamples();
83 auto pvName = this->getName() + "_fanOut_0"; // set a name just for debugging purpose
84 auto [sender, receiver] = createSynchronizedProcessArray<UserType>(size, pvName);
85 _source = _target;
86 _copies.emplace_back(sender);
87 // set receiver as our new target. this also exchanges our future_queue. But make sure to keep our id.
88 auto id = this->getId();
89 this->initFromTarget(receiver);
90 this->_id = id;
91 _isFan = true;
92 }
93
94 template<typename UserType>
95 void RoutingDecorator<UserType>::addToFan(RoutingDecorator& fan) {
96 assert(fan._isFan);
97 std::size_t size = this->getNumberOfSamples();
98 assert(fan.getNumberOfSamples() == size);
99 // a name just for debugging purpose
100 auto pvName = this->getName() + "_fanOut_" + std::to_string(fan._copies.size());
101 auto [sender, receiver] = createSynchronizedProcessArray<UserType>(size, pvName);
102 fan._copies.emplace_back(sender);
103 // set receiver as our new target. this also exchanges our future_queue. But make sure to keep our id.
104 auto id = this->getId();
105 this->initFromTarget(receiver);
106 this->_id = id;
107 }
