36  uint32_t i = 0;

39  auto settings = _excitationSignal->getXAxisSettings();

46  int32_t minValueIndex = 0;

51  }
52 
53  if(_excitationAsTime->getValue(i) < min) {
54  minValueIndex = i;
55  min = _excitationAsTime->getValue(i);
56  }
57  }
58  int32_t distance = std::abs(maxValueIndex - minValueIndex);
59  _logger(Loger::DETAIL) << __func__ << ": "
60  << "Excitation max: " << max << " on index: " << maxValueIndex << " ("
61  << (settings.start + maxValueIndex * settings.increment) << settings.unitString
62  << ") min: " << min << " on index: " << minValueIndex << " ("
63  << (settings.start + minValueIndex * settings.increment) << settings.unitString
64  << ") distance " << distance * settings.increment << settings.unitString << std::endl;
65 
66  if(maxValueIndex == minValueIndex) {
67  _logger(Loger::ERROR)
68  << __func__ << ": "
69  << "Excitation maximum index equal to minimum index. Signal seems to be constant. Cannot continue."
70  << std::endl;
71  clearTable();
72  return 0;
73  }
74 
75  // find if minValueIndex or maxValueIndex has greater value
76  uint32_t j = minValueIndex;
77  if(maxValueIndex > minValueIndex) j = maxValueIndex;
78  // find maximum index for displaying plots and limit the plot length
79  displayMaxIndex = _excitationAsTime->csaSendLength();
80  float rangePercentage = 0.05;
81  for(i = j; i < _excitationAsTime->csaSendLength(); i++) {
82  if(_excitationAsTime->getValue(i) < max * rangePercentage &&
83  _excitationAsTime->getValue(i) > min * rangePercentage) {
84  displayMaxIndex = i;
85  break;
86  }
87  }
88  displayMaxIndex *= 1.1;
89  if(displayMaxIndex > _excitationAsTime->csaSendLength()) displayMaxIndex = _excitationAsTime->csaSendLength();
90  _excitationAsTime->csaSendLength(displayMaxIndex);
91 
92  //
93  // Slope between max and min value detection
94  //
95 
96  // create vectors for calculation
97  uint32_t maxIndexForCalc = 0;
98  uint32_t minIndexForCalc = 0;
99  if(maxValueIndex > minValueIndex) {
100  minIndexForCalc = minValueIndex + ((int)distance * 0.3);
101  maxIndexForCalc = maxValueIndex - ((int)distance * 0.3);
102  startIndex = minValueIndex;
103  stopIndex = maxValueIndex;
104  }
105  else {
106  minIndexForCalc = maxValueIndex + ((int)distance * 0.3);
107  maxIndexForCalc = minValueIndex - ((int)distance * 0.3);
108  startIndex = maxValueIndex;
109  stopIndex = minValueIndex;

134 double FastCalibrationMziLaser::calculateMziSlope(int startIndex, int stopIndex, uint32_t displayMaxIndex) {
135  uint32_t detectorDataBufferLength = _detectorSignal->getBuffer().size();
136  std::vector<float>& detectorDataBuffer = *(_detectorSignalRaw.get());
137  _logger(Loger::DETAIL) << __func__ << ": "
138  << "Size of MZI data buffer: " << detectorDataBuffer.size()
139  << " send length: " << _detectorSignal->csaSendLength() << std::endl;
140 
141  uint32_t i = 0;
142  for(i = 0; i < detectorDataBufferLength && i < 16384; i++) _detectorResponce->setValue(detectorDataBuffer[i], i);
143  _detectorResponce->csaSendLength(i);
144  auto settings = _detectorSignal->getXAxisSettings();
145  _detectorResponce->setXAxisSettings(settings);
146  _detectorResponce->csaSendLength(displayMaxIndex);
147 
148  double max = -10000000;
149  double min = 10000000;
150  int maxValueIndex = 0;
151  int minValueIndex = 0;
152 
153  for(int32_t j = startIndex; (j < stopIndex) && (j < int(_detectorResponce->csaSendLength())); j++) {
154  if(_detectorResponce->getValue(j) > max) {
155  maxValueIndex = j;
156  max = _detectorResponce->getValue(j);
157  }
158 
159  if(_detectorResponce->getValue(j) < min) {
160  minValueIndex = j;
161  min = _detectorResponce->getValue(j);
162  }
163  }
164 
165  int32_t distance = std::abs(maxValueIndex - minValueIndex);
166  _logger(Loger::DETAIL) << __func__ << ": "
167  << "MZI response max: " << max << " on index: " << maxValueIndex << " ("
168  << (settings.start + maxValueIndex * settings.increment) << settings.unitString
169  << ") min: " << min << " on index: " << minValueIndex << " ("
170  << (settings.start + minValueIndex * settings.increment) << settings.unitString
171  << ") distance " << distance * settings.increment << settings.unitString << std::endl;
172 
173  if(maxValueIndex == minValueIndex) {
174  _logger(Loger::ERROR)
175  << __func__ << ": "
176  << "MZI response maximum index equal to minimum index. Signal seems to be constant. Cannot continue."
177  << std::endl;
178  clearTable();
179  return 0;
180  }
181 
182  // create vectors for calculation
183  uint32_t maxIndexForCalc = 0;
184  uint32_t minIndexForCalc = 0;
185 
186  if(maxValueIndex > minValueIndex) {
187  minIndexForCalc = minValueIndex + ((int)distance * 0.325);
188  maxIndexForCalc = maxValueIndex - ((int)distance * 0.325);
189  }
190  else {
191  minIndexForCalc = maxValueIndex + ((int)distance * 0.325);
192  maxIndexForCalc = minValueIndex - ((int)distance * 0.325);
193  }
194  _logger(Loger::DETAIL) << __func__ << ": "
195  << "Min index for calc: " << minIndexForCalc << " max index for calc: " << maxIndexForCalc
196  << std::endl;
197  // create vectors for calculations
198  std::vector<double> mziVoltageForCalc;
199  std::vector<double> timeVector;
200  for(uint32_t i = minIndexForCalc; i < maxIndexForCalc; i++) {
201  mziVoltageForCalc.push_back(_detectorResponce->getValue(i));
202  timeVector.push_back(i * settings.increment);
203  }
204 
205  for(i = 0; i < mziVoltageForCalc.size() && i < 16384; i++)
206  _detectorResponcePointsForCalc->setValue(mziVoltageForCalc[i], i);
207  _detectorResponcePointsForCalc->csaSendLength(i);
208  settings = _detectorResponce->getXAxisSettings();
209  settings.start = minIndexForCalc * settings.increment;
210  settings.dataAmount = mziVoltageForCalc.size();
211  _detectorResponcePointsForCalc->setXAxisSettings(settings);
212  _detectorResponcePointsForCalc->csaSendLength(i);
213  _logger(Loger::DETAIL) << __func__ << ": "
214  << "Size of mziVoltageForCalc: " << mziVoltageForCalc.size() << std::endl;
215 
216  // finding linear fit for a data
217  double ccFactorA = 0;
218  double ccFactorB = 0;
219  linearFit(mziVoltageForCalc.size(), timeVector.data(), mziVoltageForCalc.data(), ccFactorA, ccFactorB);
220  _logger(Loger::INFO) << __func__ << ": "
221  << "MZI lin.fit a param: " << ccFactorA << ", b param: " << ccFactorB << std::endl;
222 
223  for(i = 0; i < mziVoltageForCalc.size() && i < 16384; i++) {
224  float y = ccFactorA * (settings.start + i * settings.increment) + ccFactorB;
225  _detectorResponceCalcLinearFit->setValue(y, i);
226  }
227  _detectorResponceCalcLinearFit->setXAxisSettings(settings);
228  _detectorResponceCalcLinearFit->csaSendLength(i);
229 
230  return ccFactorA;

233 void FastCalibrationMziLaser::calibrateCore() {
234  auto time_start = std::chrono::high_resolution_clock::now();
235  auto duration =
236  std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::high_resolution_clock::now() - time_start)
237  .count();
238 
239  int startOxcSearchIndex = 0;
240  int stopOxcSearchIndex = 0;
241  uint32_t maxPlotDisplayIndex = 0;
242  auto excitationCC = calculateExcitationSlope(startOxcSearchIndex, stopOxcSearchIndex, maxPlotDisplayIndex);
243  auto mziCC = calculateMziSlope(startOxcSearchIndex, stopOxcSearchIndex, maxPlotDisplayIndex);
244  if(mziCC) {
245  _ccFsToV->setValue(std::fabs(excitationCC / mziCC));
246  _logger(Loger::INFO) << __func__ << ": "
247  << "CC value (std::fabs(mziCC/excitationCC)): " << _ccFsToV->getValue() << std::endl;
248  }
249  else
250  _logger(Loger::ERROR) << __func__ << ": "
251  << "MZI CC is 0" << std::endl;
252 
253  _logger(Loger::INFO) << __func__ << ": "
254  << "calibration core: " << duration << " milliseconds." << std::endl;
255  // restore original settings
256  _detectorSignalSequence->enableRawDataStorage(false);
257  //
258  // clearTable();
