26 RfPhaseManager::PhaseOffsetValues RfPhaseManager::calculateWrapOffsetForTiming(double timing) {
27  double rf2ChannelFreqMHz = _extData.mainLaserFreqMHz->getValue() * _extData.rf2Phase->getUnwrapFactor();
28  double rf4ChannelFreqMHz = _extData.mainLaserFreqMHz->getValue() * _extData.rf4Phase->getUnwrapFactor();
29 
30  PhaseOffsetValues offsets;
31 
32  offsets.rf2PhaseOffsetValue = (timing * 1.0e-12) * (rf2ChannelFreqMHz * 1.0e6) * _phaseRange;
33  offsets.rf4PhaseOffsetValue = (timing * 1.0e-12) * (rf4ChannelFreqMHz * 1.0e6) * _phaseRange;
34  offsets.rf2UwPhaseOffsetValue = offsets.rf2PhaseOffsetValue / _extData.rf2Phase->getUnwrapFactor();

39  offsets.rf2UwPhaseOffsetValue = wrapPhaseValue(offsets.rf2UwPhaseOffsetValue);
40  offsets.rf4UwPhaseOffsetValue = wrapPhaseValue(offsets.rf4UwPhaseOffsetValue);

42  return offsets;

45 double RfPhaseManager::wrapPhaseValue(double phase) {
46  double newPhaseOffset = phase;
47  if(phase > (_phaseRange / 2.0)) {
48  newPhaseOffset = std::fmod(phase + (_phaseRange / 2.0), _phaseRange) - (_phaseRange / 2.0);
49  }
50  else if(phase < -(_phaseRange / 2.0)) {
51  newPhaseOffset = std::fmod(phase - (_phaseRange / 2.0), _phaseRange) + (_phaseRange / 2.0);
52  }
53  return newPhaseOffset;
54 }
