164 bool CsaPidCtrlSwitcherOnStateSpace::checkAndTryEnableFastPiezoChannel() {

167  // valid ID is only 0 and 1 (channe 1 and 2 - first and second) as due to firmware data from fast ctrl are send

171  _logger(Loger::ERROR) << __func__ << ": " << "PZT4 Channel ID " << (id + 1)
172  << "(1..2) selected as fast controller channel is not functional (in "
173  "error). Cannot enable it to operate the fast controller. Check hardware and settings."
174  << std::endl;
175  return false;
176  }
177 
178  if(!_extData.bspPZT4->getChannel(id)->getEnabled()) {
179  if(_pzt4ChannelBypassModeOnLocal) {
180  _logger(Loger::INFO) << __func__ << ": " << "LLL Bypass mode ON. PZT4 Channel ID " << (id + 1)
181  << " DAC Bypass Mode will be enabled." << std::endl;
182  _extData.bspPZT4->getChannel(id)->setDacBypassModeEnable(true);
183  }
184  else {
185  _logger(Loger::INFO) << __func__ << ": " << "LLL Bypass mode OFF. PZT4 Channel ID " << (id + 1)
186  << " PZT4 DAC Bypass Mode will be disabled." << std::endl;
187  _extData.bspPZT4->getChannel(id)->setDacBypassModeEnable(false);
188  }
189 
190  _logger(Loger::INFO) << __func__ << ": " << "PZT4 Channel ID " << (id + 1)
191  << "(1..2) selected as fast controller channel is disabled. Enabling to operate the fast "
192  "controller correctly."
193  << std::endl;
194  _extData.bspPZT4->getChannel(id)->setEnable(true);
195  std::this_thread::sleep_for(std::chrono::milliseconds(50));
196  }
197  else {
198  auto dacBypassON = _extData.bspPZT4->getChannel(id)->getDacBypassModeEnabled();
199  bool changeSettings = false;
200  if(_pzt4ChannelBypassModeOnLocal && !dacBypassON) {
201  _logger(Loger::WARNING)
202  << __func__ << ": " << "PZT4 Channel ID " << (id + 1)

206  << std::endl;
207  changeSettings = true;
208  }
209 
210  if(!_pzt4ChannelBypassModeOnLocal && dacBypassON) {
211  _logger(Loger::WARNING)
212  << __func__ << ": " << "PZT4 Channel ID " << (id + 1)
213  << "(1..2) selected as fast controller channel is enabled. LLL Bypass Mode not required by "
214  "fast controller, but PZT4 DAC Bypass Mode is enabled. Wrong settings. Disable PZT4 "
215  "channel, disable PZT4 DAC Bypass Mode and enable the the PZT4 channel."
216  << std::endl;
217  changeSettings = true;
218  }
219 
220  if(changeSettings) {
221  _logger(Loger::INFO) << __func__ << ": " << "PZT4 Channel ID " << (id + 1) << "will be disabled." << std::endl;
222  _extData.bspPZT4->getChannel(id)->setEnable(false);
223  std::this_thread::sleep_for(std::chrono::milliseconds(50));
224 
225  if(_pzt4ChannelBypassModeOnLocal) {
226  _extData.bspPZT4->getChannel(id)->setDacBypassModeEnable(true);

229  _extData.bspPZT4->getChannel(id)->setDacBypassModeEnable(false);

244  }

341  }

344  _switchInputThresholds[i-1]->getValue() ) { _logger(Loger::INFO) << __func__ << ": " << "Stage ID : " << i << "

351  } // this condition is used when someone disables current stage when it is in active state, then we should relock to lower one

354  _logger(Loger::INFO) << __func__ << ": " << "Stage ID : " << i

366  }
367  }

369  auto lockValid = _extData.laserLockTimingValidator->getLockValid(_pidCtrlInputSrc[actCtrlId]->getValue());
370  // std::cout << "Check lock valid for : " << _pidCtrlInputSrc[actCtrlId]->getValue() << ", lock valid: " <<
371  // lockValid << std::endl;
372  if(ctrlActive && !_switchingInProgressFlag && !lockValid) {
373  auto actionType =
374  _extData.laserLockTimingValidator
375  ->getLockInvalidAction(); // auto actionType =
376  // _extData.ctrlInputsManager->getLockInvalidAction(_pidCtrlInputSrc[actCtrlId]->getValue());
377  switch(actionType) {
378  case LaserLockTimingValidator::LOCK_INVALID_ACTION::UNLOCK: {
379  _logger(Loger::WARNING) << __func__ << ": " << "Lock lost. Open a loop. Stage ID: " << actCtrlId << ". Info: "

381  _pidCtrlInputSrc[actCtrlId]->getValue())
382  << std::endl;
383  this->setEnable(false);
384  break;
385  }
386  case LaserLockTimingValidator::LOCK_INVALID_ACTION::INFORM: {
387  _logger(Loger::WARNING) << __func__ << ": " << "Valid lock lost. Stage ID: " << actCtrlId << ". Info: "
388  << _extData.laserLockTimingValidator->getLockValidInfo(
389  _pidCtrlInputSrc[actCtrlId]->getValue())
390  << std::endl;
391  break;
392  }
393  default:
394  break;
395  }
396  }

398  // this condition which trigger switching when given stage becomes not-lockable -> try to keep system locked 'at all
399  // costs' logic only check when controller is active and there is not switching in progress
400  if(ctrlActive && !_switchingInProgressFlag && !isStageLockable(actCtrlId)) {
401  if(_automaticSwitchingEnabled->getValue()) {
402  _logger(Loger::WARNING) << __func__ << ": " << "Stage ID : " << actCtrlId
403  << " is not lockable anymore. Start switching routine from stage 1." << std::endl;
404  std::thread t(&CsaPidCtrlSwitcherOnStateSpace::switchInputs, this, 0, false);
405  t.detach();
406  std::this_thread::sleep_for(std::chrono::milliseconds(5));
407  }
408  else {
409  _logger(Loger::WARNING) << __func__ << ": " << "Stage ID : " << actCtrlId
410  << " is not lockable anymore. Switching disabled. Open a loop." << std::endl;
411  this->setEnable(false);
412  }
413  return;
414  }
415 
416  /*
417  //this condition check if current stage is still lockable and if not loop is open - 'force lock to last enabled
418  stage' logic if (ctrlActive && !_switchingInProgressFlag && !isStageLockable(actCtrlId)) { _logger(Loger::WARNING)

421  }
422  */

651 bool CsaPidCtrlSwitcherOnStateSpace::switchInputsInternal(uint32_t startingInputId, bool manualTriggeredSwitching) {
652  uint32_t stageId = startingInputId;
653  int32_t disableMask = (~_enablePidStageMask->getValue());
654  uint32_t lockableStageCnt = 0;
655 
656  for(uint32_t i = 0; i < _pidCtrlAmount; i++)
657  if(!(disableMask & (1 << i)) && isStageLockable(i)) lockableStageCnt++;
658 
659  if(lockableStageCnt == 0) {
660  _logger(Loger::INFO) << __func__ << ": " << "All stages are disabled or not lockable." << std::endl;
661  return false;
662  }
663 
664  do {
665  if(!(disableMask & (1 << stageId))) {
666  if(_searchOxcSigWhenSwitchingEnable && _searchOxcSigWhenSwitchingEnable->getValue() == 1 &&
667  getLockType(stageId) == LOCK_TYPE::OXC) {
668  _extData.oxcSignalSearchLaser->getSearchAlgSystem()->setCtrlSwitchCheck(false);
669 
670  if(_extData.oxcSignalSearchLaser->inProgress()) {
671  _logger(Loger::INFO) << __func__ << ": " << "Start OXC signal search already in progress." << std::endl;
672  uint32_t timeoutSecCounter = 0;
673  while(_extData.oxcSignalSearchLaser->inProgress() && _switchingInProgressFlag) {
674  std::this_thread::sleep_for(std::chrono::milliseconds(1000));
675  timeoutSecCounter++;
676  if(timeoutSecCounter > 1200) {
677  _logger(Loger::INFO) << __func__ << ": "
678  << "OXC signal search took longer than 20 minutes. Timeout waiting." << std::endl;
679  break;
680  }
681  _logger(Loger::INFO) << __func__ << ": " << "OXC signal search already in progress. Wait to finish."
682  << std::endl;
683  }
684  }
685  else {
686  _logger(Loger::INFO) << __func__ << ": " << "Start OXC signal search." << std::endl;
687  _extData.oxcSignalSearchLaser->searchOxcSignal(); // use blocking method for OXC search
688  }
689 
690  _extData.oxcSignalSearchLaser->getSearchAlgSystem()->setCtrlSwitchCheck(true);
691 
692  if(!_switchingInProgressFlag) {
693  _logger(Loger::INFO) << __func__ << ": " << "Switching stopped by user during OXC search." << std::endl;
694  return false;
695  }
696  }
697 
698  if(!isStageLockable(stageId)) {
699  _logger(Loger::INFO) << __func__ << ": " << "Stage at Index: " << stageId << " is not lockable. Stop switching."
700  << std::endl;
701  return true;
702  }
703 
704  lockableStageCnt++;
705  enableStage(stageId);
706  if(!checkIfControllerActive()) return false;
707 
708  std::this_thread::sleep_for(std::chrono::milliseconds(10));
709 
710  if(_extData.ssCtrl->isActive()) {
711  waitForNewDmaData(stageId, 450, 100);
712 
713  std::string unit(" fs");
714  if(_pidCtrlInputSrc[stageId]->getValue() == 6) unit = " ps";
715 
716  if((stageId < _pidCtrlAmount - 1) &&
717  ((_automaticSwitchingEnabled->getValue() != 0) || manualTriggeredSwitching)) {
718  double jitter = _extData.ctrlInputsManager->getStd(_pidCtrlInputSrc[stageId]->getValue());
719  if(jitter > _switchInputThresholds[stageId]->getValue()) {
720  _logger(Loger::INFO) << __func__ << ": "
721  << "Jitter (std_dev) higher that allowed value for this stage. Jitter: " << jitter
722  << unit << " ,allowed value: " << _switchInputThresholds[stageId]->getValue() << unit
723  << ". Switching stopped at ID: " << stageId << std::endl;
724  break;
725  }
726  else if(jitter <= 0) {
727  _logger(Loger::INFO) << __func__ << ": " << "Jitter (std_dev) lower/equal 0. Jitter: " << jitter << unit
728  << ". Cannot switch because jitter value is invalid. Switching stopped at ID: "
729  << stageId << std::endl;
730  break;
731  }
732  else {
733  _logger(Loger::INFO) << __func__ << ": " << "Jitter (std_dev) lower that threshold for stage " << stageId
734  << ". Jitter: " << jitter << unit
735  << " ,threshold: " << _switchInputThresholds[stageId]->getValue() << unit
736  << ". Move to stage ID: " << (stageId + 1) << std::endl;
737  stageId++;
738  }
739  }
740  else if(_automaticSwitchingEnabled->getValue() == 0) {
741  _logger(Loger::INFO) << __func__ << ": " << "Auto-switching is disabled." << std::endl;
742  break;
743  }
744  else {
745  _logger(Loger::INFO) << __func__ << ": " << "Controller active. This is highest stage: " << stageId
746  << ". Cannot switch higher." << std::endl;
747  break;
748  }
749  }
750  else {
751  _logger(Loger::INFO) << __func__ << ": " << "Controller not active. Switching stopped at stage " << stageId
752  << std::endl;
753  break;
754  }
755  }
756  else {
757  _logger(Loger::INFO) << __func__ << ": " << "Stage at Index: " << stageId << " is disabled." << std::endl;
758  stageId++;
759  }
760 
761  if(!_switchingInProgressFlag) {
762  _logger(Loger::INFO) << __func__ << ": " << "Switching stopped by user." << std::endl;
763  return false;
764  }
765  } while(stageId < _pidCtrlAmount);
766 
767  return true;

770 bool CsaPidCtrlSwitcherOnStateSpace::validateLock() {
771  // first check if input jitter is above reset or safe threshold
772  bool lockStatus = true;
773  uint32_t currentStage = _activeControllerId->getValue();
774  double currentJitter = _extData.ctrlInputsManager->getStd(_pidCtrlInputSrc[currentStage]->getValue());
775  _pidCtrlInputJitter[currentStage]->setValue(currentJitter);

782  currentJitter > _resetThresholds[currentStage - 1]->getValue()) {

785  << " , reset value: " << _resetThresholds[currentStage - 1]->getValue() << std::endl;

787  }
788  else if(_isLockedThresholds[currentStage]->getValue() > 0 &&
789  currentJitter > _isLockedThresholds[currentStage]->getValue()) {
790  _logger(Loger::WARNING) << __func__ << ": " << "Lock not established. Stage ID: " << currentStage
791  << ". Jitter to high. Current value: " << currentJitter
792  << " , safety value: " << _isLockedThresholds[currentStage]->getValue() << std::endl;
793  lockStatus = false;

797  << ". Controller output out of safety ranges." << std::endl;
798  lockStatus = false;
799  }
800  else if(!_extData.laserLockTimingValidator->getLockValid(_pidCtrlInputSrc[currentStage]->getValue())) {
801  _logger(Loger::WARNING) << __func__ << ": " << "Lock not established. Stage ID: " << currentStage << ". Info: "

803  _pidCtrlInputSrc[currentStage]->getValue())
804  << std::endl;
805  lockStatus = false;
806  }
807  else
808  _logger(Loger::INFO) << __func__ << ": " << "Lock established. All validation conditions fulfilled." << std::endl;
809 
810  return lockStatus;
