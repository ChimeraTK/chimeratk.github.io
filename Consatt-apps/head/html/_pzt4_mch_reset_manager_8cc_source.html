37 bool Pzt4MchResetManager::pzt4RtmPowerCycleWithMch() {
38  bool pzt4PowerCycleFailed = true;
39  try {
40  _logger(Loger::INFO) << __func__ << ": Deactivate PZT4 RTM via MCH." << std::endl;
41  this->setActive(false);
42  bool pzt4CurrentIsZero = false;
43  int waitLoopCnt = 0;
44  int waitTimeSec = 2;
45  int waitTimeCntMax = 100;
46  do {
47  std::this_thread::sleep_for(std::chrono::seconds(waitTimeSec));
48  waitLoopCnt++;

50  if(current < 0.1) {
51  _logger(Loger::INFO) << __func__ << ": Current is: " << current << ". Board power down." << std::endl;
52  pzt4CurrentIsZero = true;
53  }
54  else {
55  _logger(Loger::INFO) << __func__ << ": Current is: " << current
56  << ". Board still powered. Total wait time: " << waitLoopCnt * waitTimeSec << " seconds."
57  << std::endl;
58  this->setActive(false);
59  }
60  } while(waitLoopCnt < waitTimeCntMax && pzt4CurrentIsZero == false);
61 
62  if(pzt4CurrentIsZero) {
63  _logger(Loger::INFO) << __func__ << ": Activate PZT4 RTM via MCH." << std::endl;
64  this->setActive(true);
65  pzt4CurrentIsZero = true;
66  waitLoopCnt = 0;
67  do {
68  std::this_thread::sleep_for(std::chrono::seconds(waitTimeSec));
69  waitLoopCnt++;
70  auto current = this->pzt4Current();
71  if(current > 0.5) {
72  _logger(Loger::INFO) << __func__ << ": Current is: " << current << ". Board power up." << std::endl;
73  pzt4CurrentIsZero = false;
74  }
75  else {
76  _logger(Loger::INFO) << __func__ << ": Current is: " << current
77  << ". Board still not powered. Total wait time: " << waitLoopCnt * waitTimeSec
78  << " seconds." << std::endl;
79  this->setActive(true);
80  }
81  } while(waitLoopCnt < waitTimeCntMax && pzt4CurrentIsZero == true);
82 
83  if(!pzt4CurrentIsZero) pzt4PowerCycleFailed = false;
84  }
85  else {
86  this->setActive(true);
87  }
88  }
89  catch(CsaExternalProcessException& ex) {
90  _logger(Loger::ERROR) << __func__ << ": "
91  << "Communication to MCH server failed: " << ex.what() << std::endl;
92  pzt4PowerCycleFailed = true;
93  }

95 }
