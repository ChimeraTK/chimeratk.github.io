114 void LOTablePhaseCorrection::initialize(boost::shared_ptr<RfPhaseManager> rfPhaseManager,
115  boost::shared_ptr<CsaMultiplexDataDaq> globalDmaDaq, boost::shared_ptr<M_RapidX_ENT_PhaseDetection> rf5,
116  boost::shared_ptr<M_RapidX_ENT_LO_Generator> loTableGenerator) {
117  _extData.globalDmaDaq = globalDmaDaq;
118  _extData.hardTimingReferenceRf = rf5;
119  _extData.loTableGenerator = loTableGenerator;
120  _extData.rfPhaseManager = rfPhaseManager;
121 
122  // initialized LO tables with previous values
123  applyLoTableBucketOffset(0, false);
124  std::this_thread::sleep_for(std::chrono::milliseconds(100));
125  _extData.globalDmaDaq->updateFromDevice(0);
126  _extData.hardTimingReferenceRf->updateFromDevice();
127 
128  // check hard reference if any correction is needed
129  int factor = calculateLoTableBucketAutoCorrection(calculateLoTablePhaseCorrection());
130  // check if manual settings are enabled - then value will be overwritten during first update
131  if(_loTablePhaseBucketCorrectionManualEnable->getValue() == 0 && factor) {
132  _logger(Loger::WARNING) << __func__ << ": "
133  << "LO Table correction required. Correction factor is: " << factor << std::endl;
134  applyLoTableBucketOffset(factor);
135  }
136 
137  double refFreqMhz = 54.16666666666666;
138  CsaAppConfigFileParser::getInstance().getValueAsDataDouble("X2_TIMER_REFERENCE_SIG_FREQ_MHZ", refFreqMhz);
139  _referenceSignalFrequencyMhz->setValue(refFreqMhz);
140 
141  _extData.globalDmaDaq->getSequence(_hardTimingReferenceChannel)->enableRawDataStorage(true);
142 }

149 void LOTablePhaseCorrection::feedbackExecute() {
150  if(_recalibrationInProgress) return;
151 
152  float error = _feedbackSetpoint->getValue() - _loTablePhaseCorrectionAuto->getValue();
153  _feedbackError->setValue(error);
154  if(_enableFeedback->getValue()) {
155  if(std::fabs(error) > _correctionPhaseStdDev / 3) {
156  float ctrlPropontional = error * _feedbackPCtrlParam->getValue();
157  //_ctrlIntegral = (_ctrlIntegral + error) * _feedbackICtrlParam->getValue();
158  float newPhase = _loTablePhaseCorrectionApplied->getValue() + (ctrlPropontional + _ctrlIntegral);
159  _loTablePhaseCorrectionApplied->setValue(newPhase);
160  applyLoTableBucketOffset(0, false);
161  }
162  }
163  else
164  _ctrlIntegral = 0;
165 }

198 double LOTablePhaseCorrection::calculateLoTablePhaseCorrection() {
199  auto seqPtr = _extData.globalDmaDaq->getSequence(_hardTimingReferenceChannel);
200  if(seqPtr->isRawDataStorageEnabled() == false) {
201  seqPtr->enableRawDataStorage(true);
202  return 0; // here exception should be thrown
203  }
204 
205  _correctionPhaseMean = 0;
206  _correctionPhaseStdDev = 0;
207  if(_extData.hardTimingReferenceRf->signalIsValid()) {
208  _statisticCalculator.setEnableStdDev(true);
209  _statisticCalculator.setEnableMean(true);
210 
211  boost::shared_ptr<std::vector<float>> hardTimeRegUnitData = seqPtr->getRawDataVector();
212  _statisticCalculator.setVector(*hardTimeRegUnitData);
213  _statisticCalculator.calculate();
214 
215  _correctionPhaseMean = _statisticCalculator.getMean();
216  _correctionPhaseStdDev = _statisticCalculator.getStdDev();
217  }
218  _loTablePhaseCorrectionAuto->setValue(_correctionPhaseMean);
219  _loTablePhaseCorrectionAutoPs->setValue(timingShiftFromPhase(_correctionPhaseMean));
220 
221  if(!_pauseCalcOfLoTablePhaseCorrectionAutoWithLsuSwitchCorrPs) {
222  _loTablePhaseCorrectionAutoWithLsuSwitchCorrPs->setValue(
223  _loTablePhaseCorrectionAutoPs->getValue() - _loTablePhaseCorrectionAutoLsuSwitchCorrOffset->getValue());
224  }
225 
226  return _correctionPhaseMean;
227 }

229 void LOTablePhaseCorrection::applyLoTableBucketOffset(int newBucketOffset, bool updateTimeStamp) {
230  int loTableSize = _extData.loTableGenerator->getLoTableSize();
231  int correction = _extData.rfPhaseManager->wrapCorrection(
232  _loTablePhaseBucketCorrectionApplied->getValue() + newBucketOffset, loTableSize);
233  double bucketCorrPhase = (_extData.rfPhaseManager->getPhaseRange() / loTableSize * correction);
234 
235  _loTablePhaseBucketCorrectionApplied->setValue(correction);
236  _loTablePhaseBucketCorrectionPhase->setValue(_extData.rfPhaseManager->wrapPhaseValue(bucketCorrPhase));
237 
238  double phase = _loTablePhaseCorrectionApplied->getValue() + _loTablePhaseBucketCorrectionPhase->getValue();
239  _loTablePhaseCorrectionAppliedPs->setValue(timingShiftFromPhase(phase));
240 
241  double degree = _extData.rfPhaseManager->wrapPhaseValue(phase) / _extData.rfPhaseManager->getPhaseRange() * 360.0;
242  _extData.loTableGenerator->setLoPhaseOffsetInDeg(-degree);
243  if(updateTimeStamp) _loTablePhaseCorrectionAppliedTimestamp->setValue(time(0));
244 }

246 void LOTablePhaseCorrection::recalibrate() {
247  if(!_extData.hardTimingReferenceRf->signalIsValid()) {
248  _logger(Loger::INFO) << "Re-calibrate hard timing reference not possible - RF signal not valid (low amplitude)."
249  << std::endl;
250  return;
251  }
252 
253  _recalibrationInProgress = true;
254 
255  if(_loTablePhaseBucketCorrectionManualEnable->getValue() == 0) {
256  _ctrlIntegral = 0;
257 
258  _logger(Loger::INFO) << __func__ << ": "
259  << "Re-calibrate hard timing reference - LO Table." << std::endl;
260 
261  // first zero all applied coefficients
262  _loTablePhaseCorrectionApplied->setValue(0);
263  applyLoTableBucketOffset(-_loTablePhaseBucketCorrectionApplied->getValue());
264  // give time to refresh all data with new settings - assume at least 1Hz of operation
265  std::this_thread::sleep_for(std::chrono::milliseconds(1500));
266 
267  // disable feature for test - 02.10.2019 - TSK
268  /*
269  // calculate and apply buckets offset
270  applyLoTableBucketOffset(calculateLoTableBucketAutoCorrection(calculateLoTablePhaseCorrection()));
271  std::this_thread::sleep_for(std::chrono::milliseconds(1100));
272  */
273 
274  // calculate rest which left after moving with buckets
275  calculateLoTablePhaseCorrection();
276  _loTablePhaseCorrectionApplied->setValue(_loTablePhaseCorrectionAuto->getValue());
277  applyLoTableBucketOffset(0);
278  _loTablePhaseCorrectionAutoLsuSwitchCorrOffset->setValue(0);
279  // set time of calibration
280  }
281  else
282  _logger(Loger::ERROR) << __func__ << ": "
283  << "Re-calibration of hard timing reference not possible. Disable Manual offset."
284  << std::endl;
285 
286  _recalibrationInProgress = false;
287 }

293 void LOTablePhaseCorrection::lsuSwitchCorrectionCalculate() {
294  auto cv = _loTablePhaseCorrectionAutoPs->getValue();
295  auto diff = cv - _phaseCorrsectionBeforeLsuSwitch;
296 
297  _logger(Loger::INFO) << __func__ << ": "
298  << "Value of LO correction before LSU switch: " << _phaseCorrsectionBeforeLsuSwitch
299  << " ps. Value after: " << cv << " ps. Difference: " << diff << " ps." << std::endl;
300 
301  if(std::fabs(diff) < 50) {
302  _logger(Loger::INFO)
303  << __func__ << ": "
304  << "Difference before and after LSU pulse switch smaller than 50 ps. LSU OXC pulse switch was unsuccessful."
305  << std::endl;
306  _pauseCalcOfLoTablePhaseCorrectionAutoWithLsuSwitchCorrPs = false;
307  return;
308  }
309  auto newDiff = _loTablePhaseCorrectionAutoLsuSwitchCorrOffset->getValue() + diff;
310  _logger(Loger::INFO) << __func__ << ": "
311  << "Old RF offset correction for LO calibration: "
312  << _loTablePhaseCorrectionAutoLsuSwitchCorrOffset->getValue()
313  << " ps. New RF offset correction for LO calibration: " << newDiff << " ps." << std::endl;
314 
315  _loTablePhaseCorrectionAutoLsuSwitchCorrOffset->setValue(newDiff);
316  _pauseCalcOfLoTablePhaseCorrectionAutoWithLsuSwitchCorrPs = false;
317 }

323 void LOTablePhaseCorrection::triggerCommand(int command) {
324  switch(command) {
325  case CMD::CMD_LO_TABLE_CALIBRATION_MANUAL_APPLY: {
326  _loTablePhaseCorrectionApplied->setValue(_loTablePhaseCorrectionManual->getValue());
327  _loTablePhaseCorrectionAppliedTimestamp->setValue(time(0));
328  applyLoTableBucketOffset(0);
329  break;
330  }
331 
332  case CMD::CMD_LO_TABLE_CALIBRATION_AUTO_APPLY: {
333  calculateLoTablePhaseCorrection();
334  _loTablePhaseCorrectionApplied->setValue(_loTablePhaseCorrectionAuto->getValue());
335  _loTablePhaseCorrectionAppliedTimestamp->setValue(time(0));
336  applyLoTableBucketOffset(0);
337  break;
338  }
339 
340  default:
341  break;
342  }
343 }

360  double phaseJumpMax = _extData.rfPhaseManager->getPhaseRange() / _extData.loTableGenerator->getLoTableSize();
361  double threshold = phaseJumpMax * (0.8);
362  int autoFactor = 0;
363  if(fabs(phaseCorrection) > threshold) {
364  autoFactor = std::round(phaseCorrection / phaseJumpMax);
365  }
366 
367  // disable feature for test - 02.10.2019 - TSK
368  if(_loTablePhaseBucketCorrectionApplied
369  ->getValue()) // if there was some value from history or from manual, we will zero it by this
370  autoFactor = -_loTablePhaseBucketCorrectionApplied->getValue();
371  else
372  autoFactor = 0;
373 
374  _loTablePhaseBucketCorrectionAuto->setValue(autoFactor);
375  return autoFactor; //_loTablePhaseBucketCorrectionAuto->getValue();
