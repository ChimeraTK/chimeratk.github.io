66 bool LaserLockTimingValidator::getLockValid(uint32_t channelId) {
67  // system locked to RF fundamental, then bucket is always valid
68  if(channelId == 6) return true;
69 
70  // if bucket not OK and action set to UNLOCK when bucket invalid
71  if(_rfBucketIncorrectFlag->getValue() == 1 && _rfBucketIncorrectAction->getValue() == TIMING_STATUS_VALUE::ERROR)
72  return false;
73 
74  // return true, even when bucket itself is invalid, but incorrect action is set to 0 (inform)
75  return true;
76 }

90 void LaserLockTimingValidator::update() {
91  if(_rfBucketCorrectThreshold->getValue() <= 0) {
92  _rfBucketCorrectThreshold->setValue(100);
93  _rfBucketCorrectThreshold->synchronize();
94  }
95 
96  auto disableMask = _laserTimingDisableValidationMask->getValue();
97  if(disableMask & 0x1) {
98  _rfBucketIncorrectFlag->setValue(TIMING_STATUS_VALUE::DISABLED);
99  _rf2rf4TimingIncorrectFlag->setValue(TIMING_STATUS_VALUE::DISABLED);
100  _loPhaseIncorrectFlag->setValue(TIMING_STATUS_VALUE::DISABLED);
101  _laserTimingIncorrectFlag->setValue(TIMING_STATUS_VALUE::DISABLED);
102  return;
103  }
104 
105  // evaluate warnings
106  auto rf2Valid = _extData.rf2Phase->signalIsValid();
107  auto rf4Valid = _extData.rf4Phase->signalIsValid();
108  auto rfLoValid = _extData.rf5Phase->signalIsValid();
109  auto unlocedOnPurpose = _extData.lockStatusValidator->getUnlockByUser();
110  auto ctrlActive = _extData.mainCtrl->isActive();
111  auto threshold = _rfBucketCorrectThreshold->getValue();
112  //
113  // evaluate _rfBucketIncorrectFlag - rf4
114  if(disableMask & 0x2) {
115  _rfBucketIncorrectFlag->setValue(TIMING_STATUS_VALUE::DISABLED);
116  }
117  else if(!rf4Valid) {
118  _rfBucketIncorrectFlag->setValue(TIMING_STATUS_VALUE::WARNING);
119  }
120  else if(unlocedOnPurpose || !ctrlActive) {
121  _rfBucketIncorrectFlag->setValue(TIMING_STATUS_VALUE::DISABLED);
122  }
123  else if(std::fabs(_extData.laserDaqStats->_rf4PhaseStats->getMean()) > threshold) {
124  _rfBucketIncorrectFlag->setValue(TIMING_STATUS_VALUE::ERROR);
125  }
126  else {
127  _rfBucketIncorrectFlag->setValue(TIMING_STATUS_VALUE::OK);
128  }
129 
130  // evaluate _rf2rf4TimingIncorrectFlag
131  if(disableMask & 0x4) {
132  _rf2rf4TimingIncorrectFlag->setValue(TIMING_STATUS_VALUE::DISABLED);
133  }
134  else if(!rf4Valid || !rf2Valid) {
135  _rf2rf4TimingIncorrectFlag->setValue(TIMING_STATUS_VALUE::WARNING);
136  }
137  else if(unlocedOnPurpose || !ctrlActive) {
138  _rf2rf4TimingIncorrectFlag->setValue(TIMING_STATUS_VALUE::DISABLED);
139  }
140  else if(std::fabs(_extData.rf2rf4Correction->getRf2Rf4PhaseCorrectionInPs()) > threshold) {
141  _rf2rf4TimingIncorrectFlag->setValue(TIMING_STATUS_VALUE::ERROR);
142  }
143  else {
144  _rf2rf4TimingIncorrectFlag->setValue(TIMING_STATUS_VALUE::OK);
145  }
146 
147  // evaluate _loPhaseIncorrectFlag
148  if(disableMask & 0x8) {
149  _loPhaseIncorrectFlag->setValue(TIMING_STATUS_VALUE::DISABLED);
150  }
151  else if(!rfLoValid) {
152  _loPhaseIncorrectFlag->setValue(TIMING_STATUS_VALUE::WARNING);
153  }
154  else if(std::fabs(_extData.loTableCorrection->getLoTablePhaseCorrectionPsWithLsuSwitchCorrection()) > threshold) {
155  _loPhaseIncorrectFlag->setValue(TIMING_STATUS_VALUE::ERROR);
156  }
157  else
158  _loPhaseIncorrectFlag->setValue(TIMING_STATUS_VALUE::OK);
159 
160  // evaluate global flag - _laserTimingIncorrectFlag
161  if(disableMask & 0x10) {
162  _laserTimingIncorrectFlag->setValue(TIMING_STATUS_VALUE::DISABLED);
163  }
164  else if(_rfBucketIncorrectFlag->getValue() == TIMING_STATUS_VALUE::ERROR ||
165  _rf2rf4TimingIncorrectFlag->getValue() == TIMING_STATUS_VALUE::ERROR ||
166  _loPhaseIncorrectFlag->getValue() == TIMING_STATUS_VALUE::ERROR) {
167  _laserTimingIncorrectFlag->setValue(TIMING_STATUS_VALUE::ERROR);
168  }
169  else if(_rfBucketIncorrectFlag->getValue() == TIMING_STATUS_VALUE::WARNING ||
170  _rf2rf4TimingIncorrectFlag->getValue() == TIMING_STATUS_VALUE::WARNING ||
171  _loPhaseIncorrectFlag->getValue() == TIMING_STATUS_VALUE::WARNING) {
172  _laserTimingIncorrectFlag->setValue(TIMING_STATUS_VALUE::WARNING);
173  }
174  else if(_rfBucketIncorrectFlag->getValue() == TIMING_STATUS_VALUE::DISABLED &&
175  _rf2rf4TimingIncorrectFlag->getValue() == TIMING_STATUS_VALUE::DISABLED &&
176  _loPhaseIncorrectFlag->getValue() == TIMING_STATUS_VALUE::DISABLED) {
177  _laserTimingIncorrectFlag->setValue(TIMING_STATUS_VALUE::DISABLED);
178  }
179  else {
180  _laserTimingIncorrectFlag->setValue(TIMING_STATUS_VALUE::OK);
181  }
182 }
