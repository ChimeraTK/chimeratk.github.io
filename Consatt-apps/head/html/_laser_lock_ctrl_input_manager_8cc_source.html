57 bool LaserLockCtrlInputManager::isInputLockable(uint32_t inputId) {
58  if(getLockingType(inputId) == LOCK_TYPE::MZI_PHASE) {
59  if(!_extData.auxiliaryCtrlManager) {
60  _logger(Loger::WARNING) << __func__ << ": "
61  << "Input " << inputId
62  << " is MZI Phase. Auxiliary controller status not known. Cannot lock." << std::endl;
63  return false;
64  }
65  else if(!_extData.auxiliaryCtrlManager->isCtrlActive() || _extData.auxiliaryCtrlManager->isCtrlSafetyHoldOn() ||
66  !_extData.auxiliaryCtrlManager->isLockable()) {
67  _logger(Loger::WARNING) << __func__ << ": "
68  << "Input " << inputId << " is MZI Phase. MZM BIAS controller not locked." << std::endl;
69  return false;
70  }
71 
72  if(!_extData.rf3Phase) {
73  _logger(Loger::WARNING) << __func__ << ": "
74  << "Input " << inputId << " is MZI Phase. RF3 status not known. Cannot lock."
75  << std::endl;
76  return false;
77  }
78  else if(_extData.rf3Phase->getErrorType() == M_RapidX_ENT_PhaseDetection::ERROR_TYPE::SIGNAL) {
79  _logger(Loger::WARNING) << __func__ << ": "
80  << "Input " << inputId << " is MZI Phase. MZI signal not valid (too low power)."
81  << std::endl;
82  return false;
83  }
84 
85  //_logger(Loger::INFO) << __func__ << ": " << " Input " << inputId << " is MZI Phase. Signal is valid and BIAS
86  // controller locked. Main controller can be locked." << std::endl;
87  }
88  else if(getLockingType(inputId) == LOCK_TYPE::RF_FUND) {
89  if(_extData.rf4Phase->getErrorType() == M_RapidX_ENT_PhaseDetection::ERROR_TYPE::SIGNAL) {
90  _logger(Loger::WARNING) << __func__ << ": "
91  << "Input " << inputId
92  << " is RF fundamental. RF signal not valid (too low power or too high amplitude noise)."
93  << std::endl;
94  return false;
95  }
96  //_logger(Loger::INFO) << __func__ << ": " << " Input " << inputId << " is RF fundamental. Signal is valid. Main
97  // controller can be locked." << std::endl;
98  }
99  else if(getLockingType(inputId) == LOCK_TYPE::RF_HARM) {
100  if(_extData.rf2Phase->getErrorType() == M_RapidX_ENT_PhaseDetection::ERROR_TYPE::SIGNAL) {
101  _logger(Loger::WARNING) << __func__ << ": "
102  << "Input " << inputId
103  << " is RF harmonic. RF signal not valid (too low power or too high amplitude noise)."
104  << std::endl;
105  return false;
106  }
107  //_logger(Loger::INFO) << __func__ << ": " << " Input " << inputId << " is RF harmonic. Signal is valid. Main
108  // controller can be locked." << std::endl;
109  }
110  else if(getLockingType(inputId) == LOCK_TYPE::OXC) {
111  if(!_extData.oxcSignalValidator) {
112  _logger(Loger::WARNING) << __func__ << ": "
113  << "Input " << inputId << " is OXC. Setup do not support OXC locking." << std::endl;
114  return false;
115  }
116 
117  if(!_extData.oxcSignalValidator->validate()) {
118  _logger(Loger::WARNING) << __func__ << ": "
119  << "Input " << inputId << " is OXC. OXC signal not valid." << std::endl;
120  return false;
121  }
122  //_logger(Loger::INFO) << __func__ << ": " << " Input " << inputId << " is OXC. Signal is valid. Main controller can
123  // be locked." << std::endl;
124  }
125  else if(getLockingType(inputId) == LOCK_TYPE::UNKNOWN) {
126  _logger(Loger::WARNING) << __func__ << ": "
127  << "Input " << inputId
128  << " is UNKNOWN type. Still server will consider it lockable, but check your settings !!!."
129  << std::endl;
130  }
131 
132  return true;
133 }

135 LOCK_TYPE LaserLockCtrlInputManager::getLockingType(uint32_t channelId) {
136  switch(channelId) {
137  case 1:
138  return LOCK_TYPE::OXC;
139  case 4:
140  return LOCK_TYPE::RF_HARM;
141  case 5:
142  return LOCK_TYPE::RF_HARM;
143  case 6:
144  return LOCK_TYPE::RF_FUND;
145  case 7:
146  if(_firmwareVer == 2)
147  return LOCK_TYPE::UNKNOWN;
148  else
149  return LOCK_TYPE::RF_FUND;
150  case 8:
151  if(_firmwareVer == 2)
152  return LOCK_TYPE::MZI_PHASE;
153  else
154  return LOCK_TYPE::MZI_BIAS;
155  case 9:
156  return LOCK_TYPE::MZI_PHASE;
157  default:
158  return LOCK_TYPE::UNKNOWN;
159  }

179 double LaserLockCtrlInputManager::getStatValue(uint32_t channelId, STAT_TYPE statsType) {
180  switch(channelId) {
181  case 1:
182  return checkAndReturn(_extData.laserDaqStats->_oxcCorrStats, statsType, "OXC corrected");
183  case 2:
184  return checkAndReturn(_extData.laserDaqStats->_oxcPCorrStats, statsType, "OXC P corrected");
185  case 3:
186  return checkAndReturn(_extData.laserDaqStats->_oxcNCorrStats, statsType, "OXC N corrected");
187  case 4:
188  return checkAndReturn(_extData.laserDaqStats->_rf2PhaseStats, statsType, "Phase RF2-Harmonic");
189  case 5:
190  return checkAndReturn(_extData.laserDaqStats->_rf2PhaseUwStats, statsType, "Phase RF2-Harmonic Unwrapped");
191  case 6:
192  return checkAndReturn(_extData.laserDaqStats->_rf4PhaseStats, statsType, "Phase RF4-Fundamental");
193  case 7:
194  if(_firmwareVer == 2)
195  return checkAndReturn(_extData.laserDaqStats->_dc7, statsType, "DC7");
196  else
197  return checkAndReturn(_extData.laserDaqStats->_rf4PhaseUwStats, statsType, "Phase RF4-Fundamental Unwrapped");
198  case 8:
199  if(_firmwareVer == 2)
200  return checkAndReturn(_extData.laserDaqStats->_mziQStats, statsType, "MZM Q");
201  else
202  return checkAndReturn(_extData.laserDaqStats->_mziIStats, statsType, "MZM I");
203  case 9:
204  return checkAndReturn(_extData.laserDaqStats->_mziQStats, statsType, "MZM Q");
205  default:
206  return 0;
207  }
208  return 0.0;
209 }
