470void CsaLaserLockHed::checkAppCurrentClockFreq() {
471 if(_checkAppCurrentClockFreqDisable->getValue()) return;
472
473 int currentClockFreq = _bspSIS8300L2->clk()->getApplicationClockFreq();
474 uint32_t clockDifference = std::abs(static_cast<int>(_appCurrentClockFreq - currentClockFreq));
475
476 if(clockDifference > 0.02 * _appCurrentClockFreq)
477 _appCurrentClockFreqChangedCnt++;
478 else
479 _appCurrentClockFreqChangedCnt = 0;
480
481 //_logger(Loger::INFO) << __func__ << ": " << "current clock: " << _bspSIS8300L2->clk()->getApplicationClockFreq() <<
482 //", stored clock: " << _appCurrentClockFreq << ", diff: " << clockDifference << ", cnt: " <<
483 //_appCurrentClockFreqChangedCnt << std::endl;
484
485 if(_appCurrentClockFreqChangedCnt > 5) {
486 _logger(Loger::WARNING) << __func__ << ": "
487 << "Change of FPGA application clock frequency detected. Previous clock: "
488 << _appCurrentClockFreq << ", current clock: " << currentClockFreq
489 << ". Reinitialize clock dependant values." << std::endl;
490 _appCurrentClockFreqChangedCnt = 0;
491 _appCurrentClockFreq = currentClockFreq;
492
493 _rapidXCommon->setAppClockFreq(_appCurrentClockFreq);
494 _rapidXCommon->updateFromDevice(); // in order to update Algorithm Frequency in this object
495 auto appAlgFrequency = _rapidXCommon->getAlgorithmFrequency();
496 _logger(Loger::INFO) << __func__ << ": " << "Current FPGA application clock: " << _appCurrentClockFreq
497 << ", application algorithm clock: " << appAlgFrequency << std::endl;
498
499 _daq->setXAxisTimingParameters(
500 _appCurrentClockFreq, _bspSIS8300L2->getDaqDmaBusWidth(), _rapidXCommon->getSamplingPeriod());
501 _daq->initXAxisTimingParameters();
502
503 _c2PhaseDetection->setAlgorithFrequency(appAlgFrequency);
504 _c4PhaseDetection->setAlgorithFrequency(appAlgFrequency);
505 _c5PhaseDetection->setAlgorithFrequency(appAlgFrequency);

509void CsaLaserLockHed::executeAlgorithm() {
510 if(_systemResetManager) _systemResetManager->update();
511
512 auto jitter = _ctrl0AdvManager->getCurrentJitterInFs();
513 _logger(Loger::MAX_DETAIL) << __func__ << ": Current jitter: " << jitter << std::endl;
514 _currentJitterFs->setValue(jitter);
515
516 _laserTimingValidator->update();
517
518 try {
519 for(auto& item : _coarseTuneActuators) item->updateFromDevice();
520
521 _mutliActCoarseTuneAlgorithm->update();
522 _ctrl0AdvManager->update();
523 }
524 catch(StepperMotorActuatorException& ex) {
525 _logger(Loger::ERROR)
526 << __func__
527 << ": StepperMotorActuatorException exception detected. Cannot execute coarse tuning. Exception: " << ex.what()
528 << std::endl;
529 }
530
531 _algorithms->update();
532
533 if(_genericDriftLessDetCalib) {
534 _genericDriftLessDetCalib->update();
535 _genericDriftLessDetCalib->setLogLevel(_logger.getLogLevel());
536
537 // this part will un-pause safety jitter validation in controller whenever error table is not longer load by OXC/MZI
538 // calibration algorithm
539 if(_genericDriftLessDetCalib->getTableLoaded())
540 _driftlessDetTableNotLoadCnt = 0;
541 else if(_ctrl0AdvManager->getSafetyCheckPaused()) {
542 if(_driftlessDetTableNotLoadCnt > 3) _ctrl0AdvManager->setSafetyCheckPaused(false);
543 _driftlessDetTableNotLoadCnt++;
544 }
545 }
546
547 if(_oxcSignalSearch) _oxcSignalSearch->update();
548 if(_slowFeedbackReference) _slowFeedbackReference->update();
549
550 lockStatusValidationEnableCheck();
551 if(_validationEnableCheckCnt ==
552 0) { // it means that we are not in delay between finishing coarse tune/ switching and re-enabling status validation
553 auto currLockStatus = _lockStatusValidator->validate();
554
555 if(_mutliActCoarseTuneAlgorithm->getTuningInProgres() || _ctrl0AdvManager->isCtrlSwitchingInProgress())
556 _relockScriptActionManager->setPauseScriptExecution(true);
557 else
558 _relockScriptActionManager->setPauseScriptExecution(false);
559 _relockScriptActionManager->update(currLockStatus);
560 }
561
562 checkAppCurrentClockFreq();
563
564 // call in case address has been changed in _lockStatusValidator, requires access to external servers, so run in
565 // separate thread not to block calling thread
566 _lsuController->reinitialize(
567 _lockStatusValidator->getExternalStatusValidator()
568 ->getPrimaryAddress()); // too update in case address has been changed in _lockStatusValidator
569
570 auto triggerSelect = _timing->getMainTriggerSelect();
571 if(_mainTriggerSourceMuxCurrValue != triggerSelect) {
572 _mainTriggerSourceMux->setValue(triggerSelect);
573 _mainTriggerSourceMuxCurrValue = triggerSelect;
574 }
