62bool Pzt4ResetManager::pzt4FullPowerReset() {
63 if(!_pzt4MchManager->isInitialized()) {
64 _logger(Loger::ERROR) << __func__
65 << ": PZT4 MCH manager object not initialized. Check MCH address and slot number settings."
66 << std::endl;
67 return false;
68 }
69 if(_pzt4FullPowerResetInProgress) {
70 _logger(Loger::WARNING) << __func__ << ": PZT4 full power-cycle with resets already in progress." << std::endl;
71 return false;
72 }
73
74 _pzt4FullPowerResetInProgress = true;
75 _resetPzt4NotInProgress->setValue(0);
76
77 _logger(Loger::INFO) << __func__ << ": Store original LP filter settings and set filter to 300 Hz." << std::endl;
78 for(int i = 0; i < 4; i++) {
79 _pzt4Settings.pzt4FilterMode[i] = _bspPZT4->getChannel(i)->getFilterMode();
80
81 _bspPZT4->getChannel(i)->setFilterMode(M_ENT_PZT4Channel::FILTER_MODE::HZ_300);
82 std::this_thread::sleep_for(std::chrono::milliseconds(100));
83 _bspPZT4->getChannel(i)->synchronize();
84 }
85 _bspPZT4->updateFromDevice();
86
87 _logger(Loger::INFO) << __func__ << ": Enable reset of PZT4 DACs, ADC and sending logic." << std::endl;
88 this->enablePzt4Resets();
89 _bspPZT4->updateFromDevice();
90
91 _logger(Loger::INFO) << __func__ << ": Trigger deactivation of PZT4 RTM via MCH." << std::endl;
92 bool pzt4PowerCycleFailed = _pzt4MchManager->pzt4RtmPowerCycleWithMch();
93 if(pzt4PowerCycleFailed)
94 _logger(Loger::ERROR) << __func__ << ": Restart of PZT4 RTM via MCH seems to fail." << std::endl;
95 else
96 _logger(Loger::INFO) << __func__ << ": Deactivation and activation of PZT4 RTM via MCH successful." << std::endl;
97
98 std::this_thread::sleep_for(std::chrono::milliseconds(100));
99 _logger(Loger::INFO) << __func__ << ": Disable reset of PZT4 DACs, ADC and sending logic." << std::endl;
100 this->disablePzt4Resets();
101
102 _logger(Loger::INFO) << __func__ << ": Update PZT4 logic to get new status." << std::endl;
103 // 10 updates here is not by chance ... PZT4 logic requires at least 5 updates to detect if all is fine and restarts channels
104 for(int i = 0; i < 10; i++) {
105 _bspPZT4->updateFromDevice();
106 std::this_thread::sleep_for(std::chrono::milliseconds(100));
107 }
108 std::this_thread::sleep_for(std::chrono::milliseconds(1000));
109
110 auto pzt4Status = _bspPZT4->getStatus();
111 if(pzt4Status == M_BSP_PZT4::STATUS::ST_ADC_FROZEN || pzt4Status == M_BSP_PZT4::STATUS::ST_PWR_SUPPLY_ERROR) {
112 pzt4PowerCycleFailed = true;
113 _logger(Loger::INFO) << __func__ << ": PZT4 still in error. Power-cycle failed." << std::endl;
114 }
115 else if(pzt4PowerCycleFailed) {
116 _logger(Loger::INFO) << __func__ << ": PZT4 reports no error. MCH reported wrong state. Power-cycle successful."
117 << std::endl;
118 pzt4PowerCycleFailed = false;
119 }
120 else {
121 _logger(Loger::INFO) << __func__ << ": PZT4 reports no error. Power-cycle successful." << std::endl;
122 }
123
124 if(pzt4PowerCycleFailed) {
125 _pzt4PowerResetFailCnt->setValue((_pzt4PowerResetFailCnt->getValue() + 1));
126 }
127 else {
128 _pzt4PowerResetFailCnt->setValue(0);
129 }
130
131 _logger(Loger::INFO) << __func__ << ": Restore LP filter setting." << std::endl;
132 for(int i = 0; i < 4; i++) {
133 _bspPZT4->getChannel(i)->setFilterMode(_pzt4Settings.pzt4FilterMode[i]);
134 std::this_thread::sleep_for(std::chrono::milliseconds(100));
135 _bspPZT4->getChannel(i)->synchronize();
136 }
137
138 _pzt4FullPowerResetInProgress = false;
139 _resetPzt4NotInProgress->setValue(2);
140
141 return (!pzt4PowerCycleFailed);
142}

148bool Pzt4ResetManager::pzt4StatusValidate() {
149 int pzt4MaxError = 30;
150 if(_enablePzt4Autoreset->getValue() == 0 || _pzt4FullPowerResetInProgress) return false;
151
152 auto pzt4Status = _bspPZT4->getStatus();
153 if(pzt4Status != M_BSP_PZT4::STATUS::ST_ADC_FROZEN && pzt4Status != M_BSP_PZT4::STATUS::ST_PWR_SUPPLY_ERROR) {
154 if(_pzt4StatusErrorCnt != 0 /*&& getPzt4PowerResetsFailCnt() >= _maxAutoResetNo->getValue()*/) {
155 _logger(Loger::WARNING) << __func__ << ": PZT4 status OK. Validate for next " << --_pzt4StatusErrorCnt
156 << " updates." << std::endl;
157 if(_pzt4StatusErrorCnt == 0) _pzt4PowerResetFailCnt->setValue(0);
158 }
159 return false;
160 }
161
162 if(_pzt4StatusErrorCnt <= pzt4MaxError) {
163 if(pzt4Status == M_BSP_PZT4::STATUS::ST_ADC_FROZEN)
164 _logger(Loger::WARNING) << __func__ << ": PZT4 status: ST_ADC_FROZEN. Validate for next "
165 << (pzt4MaxError - _pzt4StatusErrorCnt) << " updates." << std::endl;
166 else if(pzt4Status == M_BSP_PZT4::STATUS::ST_PWR_SUPPLY_ERROR)
167 _logger(Loger::WARNING) << __func__ << ": PZT4 status: ST_PWR_SUPPLY_ERROR. Validate for next "
168 << (pzt4MaxError - _pzt4StatusErrorCnt) << " updates." << std::endl;
169 else
170 _logger(Loger::WARNING) << __func__ << ": PZT4 status error. Validate for next "
171 << (pzt4MaxError - _pzt4StatusErrorCnt) << " updates." << std::endl;
172
173 _pzt4StatusErrorCnt++;
174 return false;
175 }
176
177 if(_maxAutoResetNo->getValue() <= 0) {
178 _maxAutoResetNo->setValue(1);
179 _maxAutoResetNo->synchronize();
180 }
181 if(_minPeriodSystemAutoreset->getValue() <= 0) {
182 _minPeriodSystemAutoreset->setValue(3);
183 _minPeriodSystemAutoreset->synchronize();
184 }
185 else if(_minPeriodSystemAutoreset->getValue() > 60) {
186 _minPeriodSystemAutoreset->setValue(60);
187 _minPeriodSystemAutoreset->synchronize();
188 }
189
190 if(getPzt4PowerResetsFailCnt() >= _maxAutoResetNo->getValue()) {
191 _logger(Loger::DETAIL) << __func__
192 << ": PZT4 state incorrect, but number of failed restarts reached. Failed power-cycles no.: "
193 << getPzt4PowerResetsFailCnt() << std::endl;
194 return false;
195 }
196
197 auto duration =
198 std::chrono::duration_cast<std::chrono::seconds>(std::chrono::high_resolution_clock::now() - _lastAutoPzt4Reset);
199 if(duration.count() > _minPeriodSystemAutoreset->getValue() * 60) {
200 _lastAutoPzt4Reset = std::chrono::high_resolution_clock::now();
201 _logger(Loger::INFO) << __func__ << ": Status incorrect. PZT4 power-cycle reset required." << std::endl;
202 _pzt4StatusErrorCnt = 0;
203 return true;
204 }
205 else {
206 _logger(Loger::INFO) << __func__ << ": Wait to trigger PZT4 power-cycle. Minimum time between resets is "
207 << _minPeriodSystemAutoreset->getValue() << " minute/s. Last try was no longer than "
208 << std::chrono::duration_cast<std::chrono::minutes>(duration).count() << " minute/s ago."
209 << std::endl;
210 return false;
211 }

266void Pzt4ResetManager::enablePzt4Resets() {
267 std::this_thread::sleep_for(std::chrono::milliseconds(10));
268 _bspPZT4->setDacReset(true, 0);
269 _bspPZT4->setDacReset(true, 1);
