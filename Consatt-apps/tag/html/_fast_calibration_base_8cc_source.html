54 void FastCalibrationBase::loadTable() {
55  auto time_start = std::chrono::high_resolution_clock::now();
56  _calibStartTime = time_start;
57  // setting up the shape generator to generate correct excitation shape
58  auto shapeGen = _ctrlSpCorrectionMem->getShapeGenerator();
59  auto maxLength = shapeGen->getMaxLength();
60 
61  int startPoint = (1 - _fillFactor) / 2 * maxLength;
62  int endPoint = (1 - (1 - _fillFactor) / 2) * maxLength;
63  _logger(Loger::DETAIL) << __func__ << ": "
64  << "Start time point: " << startPoint << ", end time point: " << endPoint
65  << ", amplitude is: " << _amplitude << std::endl;
66  auto no = shapeGen->getTableParamNo();
67 
68  for(uint32_t i = 0; i < no; i++) shapeGen->setTableParam(i, 0, 0);
69 
70  if(no > 3) {
71  shapeGen->setTableParam(no - 1, maxLength, 0);
72  shapeGen->setTableParam(no - 2, endPoint, -_amplitude);
73  shapeGen->setTableParam(no - 3, startPoint, _amplitude);
74  }
75  shapeGen->setShapeType(CsaShapeGenerator<float>::ShapeType::TYPE_TABLE);
76  shapeGen->synchronize();
77 
78  _ctrlSpCorrectionMem->shapeGeneratorAction(CsaRegisteredMemoryAccesor<float>::COMMAND_ID::GENERATE_WRITE);
79  auto duration =
80  std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::high_resolution_clock::now() - time_start)
81  .count();
82  _tableLoaded = true;
83  _logger(Loger::DETAIL) << __func__ << ": "
84  << "Calibration error table load took: " << duration << " milliseconds." << std::endl;
85 }

118 void FastCalibrationBase::update() {
119  uint32_t wait = _updateCntWaitForData->getValue();
120  if(wait <= 0 || wait > 20) wait = 2;
121 
122  if(_calibratePhase == CC_PHASE::IDLE && _tableLoaded && _continuousCcCalculationEnable->getValue()) {
123  if(_ccCoreDelayCnt >= wait) {
124  calibrateCore();
125  _ccCoreDelayCnt = 0;
126  }
127  else
128  _ccCoreDelayCnt++;
129  return;
130  }
131 
132  if(_calibratePhase == CC_PHASE::IDLE) {
133  _ccCoreDelayCnt = 0;
134  return;
135  }
136 
137  if(_calibratePhase == CC_PHASE::LOAD_TABLE_END) {
138  _ccCoreDelayCnt++;
139  if(_ccCoreDelayCnt >= wait) {
140  calibrateCore();
141  clearTable();
142  _calibratePhase = CC_PHASE::
143  CLEAR_TABLE_END; // one could set here already the CC_PHASE::IDLE, but 1 extra delay cycle introduced
144  _ccCoreDelayCnt = 0;
145  }
146  }
147 
148  if(_calibratePhase == CC_PHASE::CLEAR_TABLE_END) _calibratePhase = CC_PHASE::IDLE;
149 }

170 bool FastCalibrationBase::linearFit(int size, double* xData, double* yData, double& slope, double& y_intercept) {
171  int i;
172  double SUMx = 0, SUMy = 0, SUMxy = 0, SUMxx = 0;
173 
174  for(i = 0; i < size; i++) {
175  SUMx = SUMx + xData[i];
176  SUMy = SUMy + yData[i];
177 
178  SUMxy = SUMxy + xData[i] * yData[i];
179  SUMxx = SUMxx + xData[i] * xData[i];
180  }
181 
182  if(size != 0 || (SUMx * SUMx != size * SUMxx)) {
183  slope = (SUMx * SUMy - size * SUMxy) / (SUMx * SUMx - size * SUMxx);
184  y_intercept = (SUMy - slope * SUMx) / size;
185  return true;
186  }
187  else
188  return false;
189 }
