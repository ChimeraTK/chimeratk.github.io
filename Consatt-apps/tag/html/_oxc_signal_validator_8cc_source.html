70bool OxcSignalValidator::validate(bool use_AND_condition) {
71 if(!_extData.oxcCorrPSignalStats || !_extData.oxcCorrPSignalStats) {
72 std::cout << "OxcSignalValidator not initialized. _extData.oxcCorrPSignalStats or _extData.oxcCorrPSignalStats are "
73 "empty!!! EXIT "
74 << std::endl;
75 exit(1);
76 }
77
78 if(!_pauseValidation) {
79 bool sigValid = false;
80 auto factor = _scaleFactor->getValue();
81 if(factor <= 0.01) factor = 0.1;
82 if(use_AND_condition) {
83 if(std::fabs(_extData.oxcCorrPSignalStats->getMean()) > std::fabs(_oxcPAbsMaxInUse->getValue() * factor) &&
84 std::fabs(_extData.oxcCorrNSignalStats->getMean()) > std::fabs(_oxcNAbsMaxInUse->getValue() * factor))
85 sigValid = true;
86 }
87 else {
88 if(std::fabs(_extData.oxcCorrPSignalStats->getMean()) > std::fabs(_oxcPAbsMaxInUse->getValue() * factor) ||
89 std::fabs(_extData.oxcCorrNSignalStats->getMean()) > std::fabs(_oxcNAbsMaxInUse->getValue() * factor))
90 sigValid = true;
91 }
92
93 if(_oxcValidForcer) {
94 if(_oxcValidForcer->getValue() == 1) {
95 sigValid = true;
96 }
97 else if(_oxcValidForcer->getValue() == 2) {
98 sigValid = false;
99 }
100 }
101
102 if(sigValid)
103 _oxcNotValid->setValue(0);
104 else
105 _oxcNotValid->setValue(1);
106 }
107
108 return getValid();
109}
