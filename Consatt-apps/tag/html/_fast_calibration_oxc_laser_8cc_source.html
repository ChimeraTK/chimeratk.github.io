135 double FastCalibrationOxcLaser::calculateOxcSlope(int startIndex, int stopIndex, uint32_t displayMaxIndex) {
136  _logger(Loger::DETAIL) << __func__ << ": "
137  << "Minimal negative OXC signal value: " << _minNegOxcAmp
138  << ", minimal positive OXC signal value: " << _minPosOxcAmp << std::endl;
139 
140  // auto settings = _detectorSignal->getXAxisSettings();
141 
142  // uint32_t oxcDataBufferLength = _detectorSignal->getBuffer().size();
143  // std::vector<float>& oxcDataBuffer = *(_oxcSignalRaw.get());
144  //_logger(Loger::DETAIL) << __func__ << ": " << "Size of oxc data buffer: " << oxcDataBuffer.size() << " send length:
145  //" << _detectorSignal->csaSendLength() << std::endl;
146 
147  auto xAxisSettings = _extData.oxcSignalCorrectionAlg->getCorrectedPlotXAxisSettings();
148  const std::vector<float>& oxcDataBuffer = _extData.oxcSignalCorrectionAlg->getCorrectedRawData();
149 
150  // std::cout << __func__ << ": " << "!!!!!!!!!!!!!!!!!!!!! size of vector: " << oxcDataBuffer.size() << "
151  // xAxisSettingSize: " << xAxisSettings.dataAmount << std::endl;
152  uint32_t i = 0;
153  for(i = 0; i < xAxisSettings.dataAmount && i < 16384; i++) _detectorResponce->setValue(oxcDataBuffer[i], i);
154  _detectorResponce->setXAxisSettings(xAxisSettings);
155  _detectorResponce->csaSendLength(displayMaxIndex);
156 
157  double max = -10000000;
158  double min = 10000000;
159  int maxValueIndex = 0;
160  int minValueIndex = 0;
161  bool minPosOxcCrossed = false;
162  bool minNegOxcCrossed = false;
163 
164  int inSignalStartIndex = -1;
165  int inSignalEndIndex = -1;
166 
167  int outOfSignalCnt = 0;
168 
169  for(int32_t j = startIndex; (j < stopIndex) && (j < int(_detectorResponce->csaSendLength())); j++) {
170  if(_detectorResponce->getValue(j) > _minPosOxcAmp) minPosOxcCrossed = true;
171 
172  if(_detectorResponce->getValue(j) < -_minNegOxcAmp) minNegOxcCrossed = true;
173 
174  if(_detectorResponce->getValue(j) > max) {
175  maxValueIndex = j;
176  max = _detectorResponce->getValue(j);
177  }
178 
179  if(_detectorResponce->getValue(j) < min) {
180  minValueIndex = j;
181  min = _detectorResponce->getValue(j);
182  }
183 
184  if((minPosOxcCrossed || minNegOxcCrossed) && inSignalStartIndex == -1) inSignalStartIndex = j;
185 
186  if(minPosOxcCrossed && minNegOxcCrossed) {
187  if((_detectorResponce->getValue(j) < _minPosOxcAmp) && (_detectorResponce->getValue(j) > -_minNegOxcAmp)) {
188  inSignalEndIndex = j;
189  break;
190  }
191  }
192 
193  if(_detectorResponce->getValue(j) == 0)
194  outOfSignalCnt++;
195  else
196  outOfSignalCnt = 0;
197 
198  if(outOfSignalCnt == 15 && inSignalStartIndex != -1) {
199  _logger(Loger::WARNING) << __func__ << ": "
200  << "We are out the OXC signal. Reset searching." << std::endl;
201  max = -10000000;
202  min = 10000000;
203  maxValueIndex = 0;
204  minValueIndex = 0;
205  minPosOxcCrossed = false;
206  minNegOxcCrossed = false;
207  inSignalStartIndex = -1;
208  inSignalEndIndex = -1;
209  }
210  }
211 
212  if(!minPosOxcCrossed) {
213  _logger(Loger::ERROR) << __func__ << ": "
214  << "Positive minimal OXC signal level not crossed. Positive arm seems be be missing "
215  << std::endl;
216  return 0;
217  }
218  if(!minNegOxcCrossed) {
219  _logger(Loger::ERROR) << __func__ << ": "
220  << "Negative minimal OXC signal level not crossed. Negative arm seems be be missing "
221  << std::endl;
222  return 0;
223  }
224 
225  int32_t distance = std::abs(maxValueIndex - minValueIndex);
226  _logger(Loger::DETAIL) << __func__ << ": "
227  << "OXC response signal starts at index: " << inSignalStartIndex
228  << " , ends on index: " << inSignalEndIndex << std::endl;
229  _logger(Loger::DETAIL) << __func__ << ": "
230  << "OXC response max: " << max << " on index: " << maxValueIndex << " min: " << min
231  << " on index: " << minValueIndex << " distance: " << distance << std::endl;
232 
233  if(maxValueIndex == minValueIndex) {
234  _logger(Loger::ERROR)
235  << __func__ << ": "
236  << "OXC response maximum index equal to minimum index. Signal seems to be constant. Cannot continue."
237  << std::endl;
238  clearTable();
239  return 0;
240  }
241 
242  // create vectors for calculation
243  uint32_t maxIndexForCalc = 0;
244  uint32_t minIndexForCalc = 0;
245 
246  if(maxValueIndex > minValueIndex) {
247  minIndexForCalc = minValueIndex + ((int)distance * 0.325);
248  maxIndexForCalc = maxValueIndex - ((int)distance * 0.325);
249  }
250  else {
251  minIndexForCalc = maxValueIndex + ((int)distance * 0.325);
252  maxIndexForCalc = minValueIndex - ((int)distance * 0.325);
253  }
254  _logger(Loger::DETAIL) << __func__ << ": "
255  << "Min index for calc: " << minIndexForCalc << " max index for calc: " << maxIndexForCalc
256  << std::endl;
257  // create vectors for calculations
258  std::vector<double> oxcVoltageForCalc;
259  std::vector<double> timeVector;
260  for(uint32_t i = minIndexForCalc; i < maxIndexForCalc; i++) {
261  oxcVoltageForCalc.push_back(_detectorResponce->getValue(i));
262  timeVector.push_back(i * xAxisSettings.increment);
263  }
264 
265  for(i = 0; i < oxcVoltageForCalc.size() && i < 16384; i++)
266  _detectorResponcePointsForCalc->setValue(oxcVoltageForCalc[i], i);
267  _detectorResponcePointsForCalc->csaSendLength(i);
268  auto settings = _detectorResponce->getXAxisSettings();
269  settings.start = minIndexForCalc * settings.increment;
270  settings.dataAmount = oxcVoltageForCalc.size();
271  _detectorResponcePointsForCalc->setXAxisSettings(settings);
272  _detectorResponcePointsForCalc->csaSendLength(i);
273  _logger(Loger::DETAIL) << __func__ << ": "
274  << "Size of oxcVoltageForCalc: " << oxcVoltageForCalc.size() << std::endl;
275 
276  // finding linear fit for a data
277  double ccFactorA = 0;
278  double ccFactorB = 0;
279  linearFit(oxcVoltageForCalc.size(), timeVector.data(), oxcVoltageForCalc.data(), ccFactorA, ccFactorB);
280  _logger(Loger::INFO) << __func__ << ": "
281  << "OXC lin.fit 'a' param: " << ccFactorA << ", 'b' param: " << ccFactorB << std::endl;
282 
283  for(i = 0; i < oxcVoltageForCalc.size() && i < 16384; i++) {
284  float y = ccFactorA * (settings.start + i * settings.increment) + ccFactorB;
285  _detectorResponceCalcLinearFit->setValue(y, i);
286  }
287  _detectorResponceCalcLinearFit->setXAxisSettings(settings);
288  _detectorResponceCalcLinearFit->csaSendLength(i);
289 
290  return ccFactorA;
291 }

313  auto time_start = std::chrono::high_resolution_clock::now();
314  auto duration =

317 
318  int startOxcSearchIndex = 0;

324  auto oxcCC = calculateOxcSlope(startOxcSearchIndex, stopOxcSearchIndex, maxPlotDisplayIndex);

326  //_ccFsToV->setValue(std::fabs(oxcCC / excitationCC));
327  _ccFsToV->setValue(std::fabs(excitationCC / oxcCC));
328  _logger(Loger::INFO) << __func__ << ": "

330  }
331  else
332  _logger(Loger::ERROR) << __func__ << ": "
333  << "Excitation CC is 0" << std::endl;
334 
335  _logger(Loger::INFO) << __func__ << ": "
336  << "calibration core: " << duration << " milliseconds." << std::endl;
337 
338  // restore original settings to OXC auto-lock feature in firmware
339  _logger(Loger::INFO) << __func__ << ": "
340  << "Restore parameters to \"Firmware OXC auto-lock\" feature." << std::endl;
