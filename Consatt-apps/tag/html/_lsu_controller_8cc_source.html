167 void LsuController::initializeWrapper(std::string lsuAddress) {
168  // std::cout << Loger::TimeStampGen::getTime() << " !!!!!!!!!!!!!!!!!!!! LsuController::initializeWrapper(std::string
169  // lsuAddress) START " << std::endl;
170  if(_initMutex.try_lock()) {
171  // if new address which was not tested is set
172  if(_initFailedAddress.compare(lsuAddress)) {
173  _initFailedCnt = 0;
174  _initFailedTimestampt = std::chrono::seconds::zero();
175  _initFailedAddress = lsuAddress;
176  }
177 
178  auto timeFromLastFail = std::chrono::duration_cast<std::chrono::seconds>(
179  std::chrono::system_clock::now().time_since_epoch() - this->_initFailedTimestampt)
180  .count();
181  if(timeFromLastFail > 10) { // try every 10 seconds
182  // std::cout << Loger::TimeStampGen::getTime() << " !!!!!!!!!!!!!!!!!!!!
183  // LsuController::initializeWrapper(std::string lsuAddress) LOCKED : " << _init << std::endl;
184  initialize(lsuAddress);
185 
186  if(!_init) {
187  if(_initFailedCnt > 5) // first try 5 times
188  _initFailedTimestampt =
189  std::chrono::duration_cast<std::chrono::seconds>(std::chrono::system_clock::now().time_since_epoch());
190  else
191  _initFailedCnt++;
192  }
193  else {
194  _initFailedCnt = 0;
195  _initFailedTimestampt = std::chrono::seconds::zero();
196  }
197  }
198  _initMutex.unlock();
199  // std::cout << Loger::TimeStampGen::getTime() << " !!!!!!!!!!!!!!!!!!!!
200  // LsuController::initializeWrapper(std::string lsuAddress) UNLOCKED : " << _init << std::endl;
201  }
202  // std::cout << Loger::TimeStampGen::getTime() << " !!!!!!!!!!!!!!!!!!!! LsuController::initializeWrapper(std::string
203  // lsuAddress) END " << std::endl;
204 }

229  }
230 
231  std::string statusPV("LOCK_STATUS.VALUE.RD");

234  externalAddress.replace(externalAddress.find(statusPV), statusPV.length(), "");

236  << "External address after removing status PV name: " << externalAddress << std::endl;

250  _logger(Loger::DETAIL) << __func__ << ": "
251  << "External LOCAL LOCK STATUS address: " << lsuLockStatusLocalAdrress << std::endl;

253  _appliedLsuLocationAddress + std::string("COARSE_TUNE.ALG.LAST_LOCK_POSITION.RD");

255  << "External LSU last lock position address: " << lsuLastLockPosAdrress << std::endl;

257  _appliedLsuLocationAddress + std::string("COARSE_TUNE.ALG.OXC_MAX_DISTANCE.WR");
258  _logger(Loger::DETAIL) << __func__ << ": "
259  << "External LSU max OXC pulses distance address: " << lsuOxcPulsesDistanceAdrress
260  << std::endl;
261 
262  try {
263  if(_lsuCommand->isInitialized())
264  _lsuCommand->changeAddress(cmdAddress);
265  else
266  _lsuCommand->init(cmdAddress, CsAccessDir::TO_CS);
267

278  if(_lsuLockStatusLocal->isInitialized())

280  else
281  _lsuLockStatusLocal->init(lsuLockStatusLocalAdrress, CsAccessDir::FROM_CS);
282 
283  if(_lsuLastLockPosition->isInitialized())
284  _lsuLastLockPosition->changeAddress(lsuLastLockPosAdrress);
285  else
286  _lsuLastLockPosition->init(lsuLastLockPosAdrress, CsAccessDir::FROM_CS);
287 
288  if(_lsuOxcPuslesMaxDistance->isInitialized())
289  _lsuOxcPuslesMaxDistance->changeAddress(lsuOxcPulsesDistanceAdrress);
290  else
291  _lsuOxcPuslesMaxDistance->init(lsuOxcPulsesDistanceAdrress, CsAccessDir::FROM_CS);
292 
293  if(!setMd22RemoteProperties(true)) return;
294  }

368  _logger(Loger::WARNING) << __func__ << ": "
