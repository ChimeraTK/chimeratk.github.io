155void LsuController::reinitialize(std::string lsuAddress) {
156 // check if object is not initialized with the same value
157 if(_init && _appliedLsuAddress.compare(lsuAddress) == 0) {
158 return;
159 }
160 // if not initialized and address in empty, it doesn't make sense to do a init, it will not work
161 if(!_init && lsuAddress.empty()) return;
162
163 std::thread t1(&LsuController::initializeWrapper, this, lsuAddress);
164 t1.detach();
165}

167void LsuController::initializeWrapper(std::string lsuAddress) {
168 // std::cout << Loger::TimeStampGen::getTime() << " !!!!!!!!!!!!!!!!!!!! LsuController::initializeWrapper(std::string
169 // lsuAddress) START " << std::endl;
170 if(_initMutex.try_lock()) {
171 // if new address which was not tested is set
172 if(_initFailedAddress.compare(lsuAddress)) {
173 _initFailedCnt = 0;
174 _initFailedTimestampt = std::chrono::seconds::zero();
175 _initFailedAddress = lsuAddress;
176 }
177
178 auto timeFromLastFail = std::chrono::duration_cast<std::chrono::seconds>(
179 std::chrono::system_clock::now().time_since_epoch() - this->_initFailedTimestampt)
180 .count();
181 if(timeFromLastFail > 10) { // try every 10 seconds
182 // std::cout << Loger::TimeStampGen::getTime() << " !!!!!!!!!!!!!!!!!!!!
183 // LsuController::initializeWrapper(std::string lsuAddress) LOCKED : " << _init << std::endl;
184 initialize(lsuAddress);
185
186 if(!_init) {
187 if(_initFailedCnt > 5) // first try 5 times
188 _initFailedTimestampt =
189 std::chrono::duration_cast<std::chrono::seconds>(std::chrono::system_clock::now().time_since_epoch());
190 else
191 _initFailedCnt++;
192 }
193 else {
194 _initFailedCnt = 0;
195 _initFailedTimestampt = std::chrono::seconds::zero();
196 }
197 }
198 _initMutex.unlock();
199 // std::cout << Loger::TimeStampGen::getTime() << " !!!!!!!!!!!!!!!!!!!!
200 // LsuController::initializeWrapper(std::string lsuAddress) UNLOCKED : " << _init << std::endl;
201 }
202 // std::cout << Loger::TimeStampGen::getTime() << " !!!!!!!!!!!!!!!!!!!! LsuController::initializeWrapper(std::string
203 // lsuAddress) END " << std::endl;
204}

206void LsuController::initialize(std::string lsuAddress) {
207 // std::cout << Loger::TimeStampGen::getTime() << " !!!!!!!!!!!!!!!!!!!! LsuController::initializeCore(std::string
208 // lsuAddress) START : " << lsuAddress << std::endl;
209
210 _init = false;
211
212 if(lsuAddress.empty()) {
213 _logger(Loger::MAX_DETAIL) << __func__ << ": "
214 << "External address is empty." << std::endl;
215 return;
216 }
217
218 std::string externalAddress = lsuAddress;
219 _logger(Loger::DETAIL) << __func__ << ": "
220 << "External address: " << externalAddress << std::endl;
221
222 std::size_t found = externalAddress.find("LSU.");
223 if(found == std::string::npos) {
224 _logger(Loger::WARNING)
225 << __func__ << ": "
226 << "External address doesn't contain 'LSU.' substring. It doesn't fit location naming. External address: "
227 << externalAddress << std::endl;
228 return;
229 }
230
231 std::string statusPV("LOCK_STATUS.VALUE.RD");
232 found = externalAddress.find(statusPV);
233 if(found != std::string::npos) {
234 externalAddress.replace(externalAddress.find(statusPV), statusPV.length(), "");
235 _logger(Loger::DETAIL) << __func__ << ": "
236 << "External address after removing status PV name: " << externalAddress << std::endl;
237 _appliedLsuLocationAddress = externalAddress;
238
239 std::string cmdAddress = _appliedLsuLocationAddress + std::string("COARSE_TUNE.ALG.COMMAND.WR");
240 _logger(Loger::DETAIL) << __func__ << ": "
241 << "External CMD address: " << cmdAddress << std::endl;
242 std::string switchLsuPulseStatusAddress =
243 _appliedLsuLocationAddress + std::string("COARSE_TUNE.ALG.SWITCH_OXC_PULSE_STATUS.RD");
244 _logger(Loger::DETAIL) << __func__ << ": "
245 << "External RELOCK STATUS address: " << switchLsuPulseStatusAddress << std::endl;
246 std::string lsuLockStatusAdrress = _appliedLsuLocationAddress + std::string("LOCK_STATUS.VALUE.RD");
247 _logger(Loger::DETAIL) << __func__ << ": "
248 << "External LOCK STATUS address: " << lsuLockStatusAdrress << std::endl;
249 std::string lsuLockStatusLocalAdrress = _appliedLsuLocationAddress + std::string("LOCK_STATUS.LOCAL.VALUE.RD");
250 _logger(Loger::DETAIL) << __func__ << ": "
251 << "External LOCAL LOCK STATUS address: " << lsuLockStatusLocalAdrress << std::endl;
252 std::string lsuLastLockPosAdrress =
253 _appliedLsuLocationAddress + std::string("COARSE_TUNE.ALG.LAST_LOCK_POSITION.RD");
254 _logger(Loger::DETAIL) << __func__ << ": "
255 << "External LSU last lock position address: " << lsuLastLockPosAdrress << std::endl;
256 std::string lsuOxcPulsesDistanceAdrress =
257 _appliedLsuLocationAddress + std::string("COARSE_TUNE.ALG.OXC_MAX_DISTANCE.WR");
258 _logger(Loger::DETAIL) << __func__ << ": "
259 << "External LSU max OXC pulses distance address: " << lsuOxcPulsesDistanceAdrress
260 << std::endl;
261
262 try {
263 if(_lsuCommand->isInitialized())
264 _lsuCommand->changeAddress(cmdAddress);
265 else
266 _lsuCommand->init(cmdAddress, CsAccessDir::TO_CS);
267
268 if(_lsuRelockToAnotherPulseStatus->isInitialized())
269 _lsuRelockToAnotherPulseStatus->changeAddress(switchLsuPulseStatusAddress);
270 else
271 _lsuRelockToAnotherPulseStatus->init(switchLsuPulseStatusAddress, CsAccessDir::FROM_CS);
272
273 if(_lsuLockStatus->isInitialized())
274 _lsuLockStatus->changeAddress(lsuLockStatusAdrress);
275 else
276 _lsuLockStatus->init(lsuLockStatusAdrress, CsAccessDir::FROM_CS);
277
278 if(_lsuLockStatusLocal->isInitialized())
279 _lsuLockStatusLocal->changeAddress(lsuLockStatusLocalAdrress);
280 else
281 _lsuLockStatusLocal->init(lsuLockStatusLocalAdrress, CsAccessDir::FROM_CS);
282
283 if(_lsuLastLockPosition->isInitialized())
284 _lsuLastLockPosition->changeAddress(lsuLastLockPosAdrress);
285 else
286 _lsuLastLockPosition->init(lsuLastLockPosAdrress, CsAccessDir::FROM_CS);
287
288 if(_lsuOxcPuslesMaxDistance->isInitialized())
289 _lsuOxcPuslesMaxDistance->changeAddress(lsuOxcPulsesDistanceAdrress);
290 else
291 _lsuOxcPuslesMaxDistance->init(lsuOxcPulsesDistanceAdrress, CsAccessDir::FROM_CS);
292
293 if(!setMd22RemoteProperties(true)) return;
294 }
295 catch(CsaExternalProcessException& ex) {
296 _logger(Loger::WARNING) << __func__ << ": "
297 << "" << ex.what() << std::endl;
298 return;
299 }
300
301 _appliedLsuAddress = lsuAddress;
302 _init = true;
303 return;
304 }
305 else {
306 _logger(Loger::WARNING) << __func__ << ": "
307 << "External address doesn't contain 'LOCK_STATUS.VALUE.RD' substring. It doesn't fit "
308 "location naming. Address is: "
309 << externalAddress << std::endl;
310 }

348bool LsuController::executeLsuCommand(int command) {
349 if(!_init) return false;
350
351 if(command == CMD::CMD_CLEAN_LSU_COMMAND_SEND_FLAG) {
352 _lsuCommandSend->setValue(0);
353 return true;
354 }
355
356 if(!checkCmd(command)) return false;
357
358 try {
359 _logger(Loger::INFO) << __func__ << ": "
360 << "Send command ID: " << command << " to location: " << _appliedLsuLocationAddress
361 << std::endl;
362 _lsuCommand->setValue(command);
363
364 int cnt = _lsuCommandSend->getValue() + 1;
365 _lsuCommandSend->setValue(cnt);
366 }
367 catch(CsaExternalProcessException& ex) {
368 _logger(Loger::WARNING) << __func__ << ": "
369 << "" << ex.what() << std::endl;
370 return false;
371 }
372 return true;
373}

375bool LsuController::checkCmd(int cmd) {
376 if(cmd == CMD::CMD_LSU_START_TUNE || cmd == CMD::CMD_LSU_STOP_TUNE || cmd == CMD::CMD_LSU_STEP_UP ||
377 cmd == CMD::CMD_LSU_STEP_DOWN || cmd == CMD::CMD_LSU_OPEN_FAST_LOOP || cmd == CMD::CMD_LSU_CLOSE_FAST_LOOP ||
378
379 cmd == CMD::CMD_LSU_OXC_SIGNAL_SEARCH || cmd == CMD::CMD_LSU_OXC_SIGNAL_SEARCH_STOP ||
380 cmd == CMD::CMD_LSU_START_TUNE_AND_LOCK || cmd == CMD::CMD_LSU_OXC_SIGNAL_SEARCH_AND_LOCK ||
381
382 cmd == CMD::CMD_LSU_OXC_LOCK_TO_NEXT_PULSE || cmd == CMD::CMD_LSU_OXC_LOCK_TO_PREVIOUS_PULSE ||
383 cmd == CMD::CMD_LSU_OXC_LOCK_TO_ANOTHER_PULSE ||
384 cmd == CMD::CMD_LSU_OXC_LOCK_TO_ANOTHER_PULSE_DETERMINE_DIRECTION)
385 return true;
386
387 _logger(Loger::WARNING) << __func__ << ": "
388 << "Command ID: " << cmd << " not recognized." << std::endl;
389 return false;
390}
