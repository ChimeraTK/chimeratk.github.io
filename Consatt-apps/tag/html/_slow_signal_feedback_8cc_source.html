59 void RefSlowFeedbackCtrl::executePICtrl() {
60  _ctrl.error = _setpoint - _input;
61  if(!_ctrl.saturated) {
62  // calc P part
63  _ctrl.pPart = _ctrlPGain->getValue() * _ctrl.error;
64  // calc I part
65  _ctrl.integral += _ctrl.error * _ctrl.dt;
66  _ctrl.iPart = _ctrlIGain->getValue() * _ctrl.integral;
67 
68  //_ctrl.integral += _ctrlIGain->getValue() * (_ctrl.error * _ctrl.dt);
69  //_ctrl.iPart = _ctrl.integral;
70  }
71 
72  // check if controller not in saturation
73  if(std::fabs(_ctrl.pPart + _ctrl.iPart) < std::fabs(_window->getValue())) {
74  _ctrl.validPPart = _ctrl.pPart;
75  _ctrl.validIPart = _ctrl.iPart;
76  _ctrl.saturated = false;
77  }
78  else if(!_ctrl.saturated) {
79  _ctrl.validPPart = _ctrl.pPart;
80  _ctrl.validIPart = _ctrl.iPart;
81  _ctrl.saturated = true;
82  }
83 
84  _ctrl.validOutput = _ctrl.validPPart + _ctrl.validIPart;
85 
86  // check whether step is big enough or we are already saturated
87  if(std::fabs(_ctrl.validOutput - _ctrl.appliedOutput) > std::fabs(_minStep->getValue()) || _ctrl.saturated) {
88  _ctrl.appliedOutput = _ctrl.validOutput;
89  }
90 
91  // check whether calculated output is inside allowed window and if not trunkate output
92  if(_ctrl.appliedOutput > _window->getValue())
93  _ctrl.appliedOutput = _window->getValue();
94  else if(_ctrl.appliedOutput < -_window->getValue())
95  _ctrl.appliedOutput = -_window->getValue();
96 }

259 void SlowSignalFeedback::feedbackThread() {
260  _feedbackThreadActive = true;
261  _logger(Loger::DETAIL) << __func__ << ": "
262  << "Thread started. Active: " << _feedbackThreadActive << std::endl;
263 
264  enableFeedback();
265  resetCtrl();
266  do {

268  std::this_thread::sleep_for(std::chrono::milliseconds(200));

278 void SlowSignalFeedback::feedbackThreadCore() {
279  evaluateStatus();
280 
281  if(_statusInternal == STATUS::OFF) return;
282 
283  if(_ctrl) {
284  float masterActPosition = 0;
285  if(_reference ==
286  REFERENCE::
287  OXC) // OXC as reference -> rf shifter used as actuator, OXC ODL is master actuator still available for users
288  masterActPosition = _extData.odlMotor->getCurrentPosition();
289  else if(_reference ==
290  REFERENCE::
291  RF) // RF as reference -> OXC ODL used as actuator, RF shifter is master actuator still available for users
292  masterActPosition = _extData.rfTimingShifter->getTimingShiftUser();
293  // set master actuator position
294  _ctrl->setMasterPosition(masterActPosition);
295 
296  if((_statusCode->getValue() & STATUS_FLAG::OXC_SEARCH_ACTIVE)) {
297  _ctrl->reset(masterActPosition);
298  sendCtrlStateToCs();
299  return;
300  }
301 
302  if(_statusInternal == STATUS::ACTIVE || _statusInternal == STATUS::WARNING) {
303  // introduce delay between changing state to active and starting feedback by 20 cycles -> in 10 Hz rate it is 2 seconds.
304  if(_activeStateCnt < 20) {
305  _activeStateCnt++;
306  return;
307  }
308 
309  // check lock type -> lock type defines input for controller
310  float ctrlInput = 0;
311  auto lockType = getLockType();
312  if(lockType == LOCK_TYPE::RF_HARM) {
313  ctrlInput = _extData.oxcCorrDiffSignalStats->getMean() /
314  1000; // this is scaled in fs, so we want to convert it to ps as actuators uses ps
315  }
316  else if(lockType == LOCK_TYPE::OXC)
317  ctrlInput = _extData.rf2Phase->getMean() /
318  1000; // this is scaled in fs, so we want to convert it to ps as actuators uses ps
319 
320  if(_resetControllerWhenStatusActive) {
321  _ctrl->reset(masterActPosition);
322  _resetControllerWhenStatusActive = false;
323  _logger(Loger::INFO) << __func__ << ": "
324  << " Reset controller after recover." << std::endl;
325  }
326  // set input to the selected controller
327  _ctrl->setInput(ctrlInput);
328 
329  if(getSetting(SETTINGS_FLAG::PID_SUSPENDED) == false) _ctrl->executePICtrl();
330  }
331  else {
332  _activeStateCnt = 0;
333  }
334 
335  if(getSetting(SETTINGS_FLAG::ACT_POS_FOLLOWER_SUSPENDED) == false) {
336  if(_statusInternal == STATUS::ACTIVE || _statusInternal == STATUS::WARNING) {
337  _ctrl->executePositionFollowerCtrl();
338  }
339  else if( // equivalent to 'if (_statusInternal == STATUS::ERROR || _statusInternal == STATUS::ALTERNATING)'
340  (_statusCode->getValue() &
341  STATUS_FLAG::
342  CTRL_SATURATED) || // controller is saturated -> we can still track the position of both actuators
343  (_statusCode->getValue() &
344  STATUS_FLAG::OXC_NOT_VALID) || // OXC is invalid -> we can still track the position of both actuators
345  (_statusCode->getValue() &
346  STATUS_FLAG::RF_NOT_VALID) || // RF is invalid -> we can still track the position of both actuators
347  (_statusCode->getValue() &
348  STATUS_FLAG::LOCKED_NOT_TO_HARM_OR_OXC) // locked to fundamental - tracking can be done
349  ) {
350  _ctrl->executePositionFollowerCtrl();
351  }
352  // else { // 13.10.2020 - just for tests - follower will work always -> even if laser not locked.
353  // _ctrl->executePositionFollowerCtrl();
354  // }
355  }
356 
357  _ctrl->applyOutput();
358  sendCtrlStateToCs();
359  }
360 }

394 SFB_REF_ST SlowSignalFeedback::evaluateStatusInternal() {
395  // clear status from previous evaluation
396  _statusCode->setValue(0);
397  // assume that all is fine
398  SFB_REF_ST statusOutput = SFB_REF_ST::ACTIVE;
399  int status = 0;
400 
401  if(isFeedbackEnabled() == false) {
402  if(getSetting(SETTINGS_FLAG::FEEDBACK_DISABLED_WHEN_RF_LOCKED) &&
403  getSetting(SETTINGS_FLAG::FEEDBACK_DISABLED_WHEN_OXC_LOCKED))
404  return SFB_REF_ST::ERROR;
405 
406  if(_enable->getValue() == 0)
407  return SFB_REF_ST::OFF;
408  else if(_fbPaused) {
409  status |= STATUS_FLAG::FB_INTERAL_PAUSED;
410  _statusCode->setValue(status);
411  return SFB_REF_ST::WARNING;
412  }
413  return SFB_REF_ST::OFF;
414  }
415 
416  // check fatal condition for the algorithm - cannot recover from them. Done only here, because its after checking if
417  // fb is enabled by user
418  try {
419  _extData.odlMotor->getCurrentPosition();
420  }
421  catch(StepperMotorActuatorException& ex) {
422  _logger(Loger::ERROR) << __func__ << ": "
423  << "ODL exception detected: " << ex.what() << std::endl;
424  _statusCode->setValue(STATUS_FLAG::ODL_ACTUATOR_ERROR);
425  return SFB_REF_ST::ERROR;
426  }
427  //
428 
429  // if system is disabled on purpose also return OFF -> enum STATUS_VALUE {OK=0, ERROR=1, DISABLED=2, WARNING=3};
430  if(_extData.laserLockStatusValidator->getStatus() == LOCK_STATUS_VALUE::DISABLED) {
431  if(!_resetControllerWhenStatusActive) {
432  _logger(Loger::INFO) << __func__ << ": "
433  << "Global status disabled. Controller will be reset after resume." << std::endl;
434  _resetControllerWhenStatusActive = true;
435  }
436  _statusCode->setValue(STATUS_FLAG::SYSTEM_STATUS_NOT_OK);
437  return SFB_REF_ST::OFF;
438  }
439 
440  if(_extData.laserLockStatusValidator->getStatus() == LOCK_STATUS_VALUE::ERROR) {
441  if(!_resetControllerWhenStatusActive) {
442  _logger(Loger::INFO) << __func__ << ": "
443  << "Global status error. Controller will be reset after resume." << std::endl;
444  _resetControllerWhenStatusActive = true;
445  }
446  _statusCode->setValue(STATUS_FLAG::SYSTEM_STATUS_NOT_OK);
447  return SFB_REF_ST::ERROR;
448  }
449 
450  if(!_extData.laserLockStatusValidator->getExternalStatusValidator()->addressEmpty()) {
451  auto extStatus = _extData.laserLockStatusValidator->getExternalStatusValidator()->getStatus();
452  if(extStatus == LOCK_STATUS_VALUE::ERROR || extStatus == LOCK_STATUS_VALUE::DISABLED) {
453  if(!_resetControllerWhenStatusActive) {
454  _logger(Loger::INFO) << __func__ << ": "
455  << "External system status error or disabled. Controller will be reset after resume."
456  << std::endl;
457  _resetControllerWhenStatusActive = true;
458  }
459  _statusCode->setValue(STATUS_FLAG::EXT_SYSTEM_STATUS_NOT_OK);
460  return SFB_REF_ST::ERROR;
461  }
462  }
463 
464  // evaluate warning
465  if(_extData.laserLockStatusValidator->getStatus() == LOCK_STATUS_VALUE::WARNING) {
466  statusOutput = SFB_REF_ST::WARNING;
467  }
468 
469  if(getSetting(SETTINGS_FLAG::ACT_POS_FOLLOWER_SUSPENDED) == true) {
470  //_statusCode->setValue(STATUS_FLAG::ACT_POS_FOLLOWER_SUSPENDED_BY_USER);
471  status |= STATUS_FLAG::ACT_POS_FOLLOWER_SUSPENDED_BY_USER;
472  statusOutput = SFB_REF_ST::WARNING;
473  }
474 
475  if(getSetting(SETTINGS_FLAG::PID_SUSPENDED) == true) {
476  //_statusCode->setValue(STATUS_FLAG::PID_SUSPENDED_BY_USER);
477  status |= STATUS_FLAG::PID_SUSPENDED_BY_USER;
478  statusOutput = SFB_REF_ST::WARNING;
479  }
480 
481  if(_extData.laserLockStatusValidator->getStatus() == LOCK_STATUS_VALUE::ALTERNATING) {
482  /*
483  if (!_resetControllerWhenStatusActive) {
484  _logger(Loger::INFO) << __func__ << ": " << " Global status Alternate. Controller will be reset after resume."
485  << std::endl; _resetControllerWhenStatusActive = true;
486  }
487  */
488  status |= STATUS_FLAG::SYSTEM_STATUS_NOT_OK;
489  statusOutput = SFB_REF_ST::ALTERNATING;
490  }
491 
492  if(_extData.ctrlManager->isCtrlSwitchingInProgress()) {
493  status |= STATUS_FLAG::SYSTEM_SWITCHING;
494  statusOutput = SFB_REF_ST::ALTERNATING;
495  _resetControllerWhenStatusActive = true;
496  }
497 
498  if(_extData.oxcSearchAlg->inProgress()) {
499  status |= STATUS_FLAG::OXC_SEARCH_ACTIVE;
500  statusOutput = SFB_REF_ST::ALTERNATING;
501  if(!_resetControllerWhenStatusActive) {
502  _logger(Loger::INFO) << __func__ << ": "
503  << " OXC search active detected. Controller will be reset after OXC search done."
504  << std::endl;
505  _resetControllerWhenStatusActive = true;
506  }
507  }
508  else {
509  auto valid = _extData.oxcSigValidator->getValid();
510  if(valid)
511  _oxcValidAvg.add(1);
512  else
513  _oxcValidAvg.add(0);
514  if(_oxcValidAvg() < 0.5) {
515  status |= STATUS_FLAG::OXC_NOT_VALID;
516  statusOutput = SFB_REF_ST::ERROR;
517  }
518  }
519 
520  // evaluate errors
521  if(getSetting(SETTINGS_FLAG::PID_SUSPENDED) && getSetting(SETTINGS_FLAG::ACT_POS_FOLLOWER_SUSPENDED))
522  statusOutput = SFB_REF_ST::ERROR;
523 
524  if(_ctrl && !_ctrl->isSettingValid()) {
525  if(_reference == REFERENCE::OXC)
526  status |= STATUS_FLAG::RF_CTRL_SETTING_ERROR;
527  else if(_reference == REFERENCE::RF)
528  status |= STATUS_FLAG::ODL_CTRL_SETTING_ERROR;
529  statusOutput = SFB_REF_ST::ERROR;
530  }
531 
532  if(_ctrl && _ctrl->getState().outputError) {
533  if(_reference == REFERENCE::OXC)
534  status |= STATUS_FLAG::RF_SHIFTER_ACTUATOR_ERROR;
535  else if(_reference == REFERENCE::RF)
536  status |= STATUS_FLAG::ODL_ACTUATOR_ERROR;
537  statusOutput = SFB_REF_ST::ERROR;
538  }
539 
540  if(_ctrl && _ctrl->getState().saturated) {
541  status |= STATUS_FLAG::CTRL_SATURATED;
542  statusOutput = SFB_REF_ST::ERROR;
543  }
544 
545  if(_extData.rf2PhaseDetector->getErrorType() == M_RapidX_ENT_PhaseDetection::ERROR_TYPE::SIGNAL) {
546  status |= STATUS_FLAG::RF_NOT_VALID;
547  statusOutput = SFB_REF_ST::ERROR;
548  }
549 
550  auto lockType = getLockType();
551  if(!(lockType == LOCK_TYPE::RF_HARM || lockType == LOCK_TYPE::OXC)) {
552  status |= STATUS_FLAG::LOCKED_NOT_TO_HARM_OR_OXC;
553  if(!_resetControllerWhenStatusActive) {
554  _logger(Loger::INFO) << __func__ << ": "
555  << " System locked not to RF HARM nor OXC. Controller will be reset relocing to OXC/RF HARM."
556  << std::endl;
557  _resetControllerWhenStatusActive = true;
558  }
559  statusOutput = SFB_REF_ST::ERROR;
560  }
561 
562  _statusCode->setValue(status);
563  return statusOutput;
564 }

566 LOCK_TYPE SlowSignalFeedback::getLockType() {
567  if(_testForceLockType && _testForceLockType->getValue()) {
568  int type = _testForceLockType->getValue();
569 
570  switch(type) {
571  case 1:
