17void CsaLsuOxcAdvAlg::triggerCommand(int cmd, bool printLog) {
18 if(printLog) _logger(Loger::DETAIL) << __func__ << ": " << _logIdString << ": Command id: " << cmd << std::endl;
19
20 switch(cmd) {
21 case CsaLsuOxcAdvAlg::CMD::OXC_LOCK_TO_NEXT_PULSE: {
22 std::thread t1(&CsaLsuOxcAdvAlg::switchPulsesWrapper, this, 1);
23 t1.detach();
24 break;
25 }
26 case CsaLsuOxcAdvAlg::CMD::OXC_LOCK_TO_PREVIOUS_PULSE: {
27 std::thread t1(&CsaLsuOxcAdvAlg::switchPulsesWrapper, this, -1);
28 t1.detach();
29 break;
30 }
31 case CsaLsuOxcAdvAlg::CMD::OXC_LOCK_TO_ANOTHER_PULSE: {
32 std::thread t1(&CsaLsuOxcAdvAlg::switchPulsesWithAutoDirSelection, this);
33 t1.detach();
34 break;
35 }
36
37 case CsaLsuOxcAdvAlg::CMD::OXC_LOCK_TO_ANOTHER_PULSE_DETERMINE_DIRECTION: {
38 if(determinePulsesSwitchDirectionSelection() == 1)
39 _switchPulseStatus->setValue(SWITCH_LSU_PULSE_STATUS::ST_SWITCH_IN_POSITIVE);
40 else
41 _switchPulseStatus->setValue(SWITCH_LSU_PULSE_STATUS::ST_SWITCH_IN_NEGATIVE);
42 break;
43 }
44
45 case CsaLsuOxcAdvAlg::CMD::OXC_CANCEL_PULSE_SWITCH: {
46 _switchToAnotherPulseCanceled = true;
47 _actuator->stop();
48 // CsaLsuOxcTuneSearchAlg::triggerCommand(CsaLsuOxcTuneSearchAlg::CMD::STOP_TUNE, false);
49 break;
50 }
51
52 default:
53 CsaLsuOxcTuneSearchAlg::triggerCommand(cmd, false);
54 break;
55 }
56}

58int CsaLsuOxcAdvAlg::determinePulsesSwitchDirectionSelection() {
59 auto currentPos = _actuator->getCurrentPositionInSteps();
60 auto pulsesDistance = _maxOxcSignalDistance->getValue();
61 int dir = -1;
62 if(((_actuator->getMaxPositionLimitInSteps() - 0.02 * pulsesDistance) - currentPos) >
63 (currentPos - (_actuator->getMinPositionLimitInSteps() + 0.02 * pulsesDistance)))
64 dir = 1;
65 return dir;
66}

73void CsaLsuOxcAdvAlg::switchPulsesWrapper(int dir) {
74 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Switch direction: " << dir << std::endl;
75
76 if(!_controller->isActive()) {
77 _messageString.set("Controller not active. Lock link first.");
78 return;
79 }
80
81 if(_stepInProgress) {
82 _messageString.set("Step in progress. Cannot switch pulses.");
83 return;
84 }
85 if(_tuneAlg.getTuningInProgres()) {
86 _messageString.set("Tune in progress. Cannot switch pulses.");
87 return;
88 }
89
90 if(_oxcSignalSearchThreadInProgress) {
91 _messageString.set("OXC signal search in progress. Cannot switch pulses.");
92 return;
93 }
94 _messageString.set("Start OXC pulse switch.");
95
96 _closeLoopWhenOxcFound = true; // we always want to try to close a loop in new position
97 _switchToAnotherPulseCanceled = false; // clear 'cancel switch' flag before starting switch
98 _oxcSignalSearchThreadInProgress = true;
99
100 _switchPulseStatus->setValue(SWITCH_LSU_PULSE_STATUS::ST_IN_PROGRESS);
101 if(switchPulsesCheckPosLimits(dir)) {
102 switchPulsesWrapperCore(dir);
103 if(!_oxcSignalSearchThreadInProgress) _switchPulseStatus->setValue(SWITCH_LSU_PULSE_STATUS::ST_STOPPED_BY_USER);
104 }
105 else
106 _switchPulseStatus->setValue(SWITCH_LSU_PULSE_STATUS::ST_WRONG_PARAMETERS);
107
108 _closeLoopWhenOxcFound = false;
109 _oxcSignalSearchThreadInProgress = false;
110}

112bool CsaLsuOxcAdvAlg::switchPulsesCheckPosLimits(int dir) {
113 // check if we can fit - if we have enough space on motor
114 auto currentPos = _actuator->getCurrentPositionInSteps();
115 auto pulsesDistance = _maxOxcSignalDistance->getValue();
116 if(dir == 1) {
117 auto limitPosition = _actuator->getMaxPositionLimitInSteps();
118 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": POS DIR. Cur.pos: " << currentPos
119 << ", limitPosition: " << limitPosition
120 << ", limit with safety: " << (limitPosition - (0.05 * pulsesDistance))
121 << ", end move position: " << (currentPos + pulsesDistance) << std::endl;
122 if((limitPosition - (0.05 * pulsesDistance)) < (currentPos + pulsesDistance)) {
123 _messageString.set("Cannot switch in positive direction. Too close to pos. end-switch.");
124 std::cout << "Out of range. Stop." << std::endl;
125 return false;
126 }
127 }
128 else if(dir == -1) {
129 auto limitPosition = _actuator->getMinPositionLimitInSteps();
130 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": NEG DIR. Cur.pos: " << currentPos
131 << ", limitPosition: " << limitPosition
132 << ", limit with safety: " << (limitPosition + (0.05 * pulsesDistance))
133 << ", end move position: " << (currentPos - pulsesDistance) << std::endl;
134 if((limitPosition + (0.05 * pulsesDistance)) > (currentPos - pulsesDistance)) {
135 _messageString.set("Cannot switch in negative direction. Too close to neg. end-switch.");
136 std::cout << "Out of range. Stop." << std::endl;
137 return false;
138 }
139 }
140 else if(dir == 0)
141 return false;
142
143 return true;
144}

146void CsaLsuOxcAdvAlg::switchPulsesWrapperCore(int dir) {
147 if(!coarseTuneSystem()) return;
148
149 // open fast loop
150 this->_ctrlManager->setEnable(false);
151 // move to new position
152 //_tuneAlg.prepareActuatorToMove(true);
153 auto currentPos = _actuator->getCurrentPositionInSteps();
154
155 auto newPos = (currentPos + dir * _maxOxcSignalDistance->getValue());
156 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": In of range. Continue. Move to position: " << newPos
157 << std::endl;
158 _stream.str("");
159 if(dir == 1)
160 _stream << "OXC pulse switch in positive direction. Move to position: " << newPos;
161 else
162 _stream << "OXC pulse switch in negative direction. Move to position: " << newPos;
163 _messageString.set(_stream.str());
164
165 auto csStatus = lockInPosition(newPos);
166 switch(csStatus) {
167 case CS_RESULT::OXC_FOUND: {
168 if(dir == 1)
169 _switchPulseStatus->setValue(SWITCH_LSU_PULSE_STATUS::ST_SUCCESS_IN_POSITIVE);
170 else if(dir == -1)
171 _switchPulseStatus->setValue(SWITCH_LSU_PULSE_STATUS::ST_SUCCESS_IN_NEGATIVE);
172 _logger(Loger::INFO) << __func__ << ": " << _logIdString << "OXC pulse switch done. Loop closed" << std::endl;
173 _messageString.set("OXC pulse switch done. Loop closed");
174 return;
175 break;
176 }
177
178 case CS_RESULT::STOP_BY_USER: {
179 if(_switchToAnotherPulseCanceled) {
180 //_oxcSignalSearchThreadInProgress = true;
181 _switchToAnotherPulseCanceled = false;
182 _logger(Loger::INFO) << __func__ << ": " << _logIdString
183 << "OXC pulse switch canceled by user. Link will be re-lock in original position: "
184 << currentPos << std::endl;
185 _messageString.set("OXC pulse switch stopped by user. Re-lock in original position.");
186 }
187 else {
188 _logger(Loger::INFO) << __func__ << ": " << _logIdString << "OXC pulse switch stopped by user." << std::endl;
189 _messageString.set("OXC pulse switch stopped by user.");
190 return;
191 }
192 break;
193 }
194 case CS_RESULT::CONTINUE: {
195 _logger(Loger::INFO) << __func__ << ": " << _logIdString << "OXC pulse switch failed. Pulse only party found."
196 << std::endl;
197 _messageString.set("OXC pulse switch failed. Pulse only party found.");
198 break;
199 }
200 case CS_RESULT::OXC_NOT_FOUND: {
201 _logger(Loger::INFO) << __func__ << ": " << _logIdString << "OXC pulse switch failed. Pulse not found."
202 << std::endl;
203 _messageString.set("OXC pulse switch failed. Pulse not found.");
204 break;
205 }
206 case CS_RESULT::ERROR: {
207 _logger(Loger::INFO) << __func__ << ": " << _logIdString << "OXC pulse switch failed. Motor error." << std::endl;
208 _messageString.set("OXC pulse switch failed. Motor error. Check logfile.");
209 printMotorErrorLog();
210 break;
211 }
212 default:
213 break;
214 }
215
216 // here we know that switching failed. we will try to lock in old (original) position
217 csStatus = lockInPosition(currentPos);
218 switch(csStatus) {
219 case CS_RESULT::OXC_FOUND: {
220 _switchPulseStatus->setValue(SWITCH_LSU_PULSE_STATUS::ST_FAILED_AND_RELOCKED);
221 _logger(Loger::INFO) << __func__ << ": " << _logIdString << "Re-lock done. Loop closed" << std::endl;
222 _messageString.set("Re-lock done. Loop closed");
223 return;
224 break;
225 }
226
227 case CS_RESULT::STOP_BY_USER: {
228 _logger(Loger::INFO) << __func__ << ": " << _logIdString << "OXC re-lock stopped by user." << std::endl;
229 _messageString.set("Re-lock stopped by user.");
230 return;
231 break;
232 }
233 case CS_RESULT::CONTINUE: {
234 _logger(Loger::INFO) << __func__ << ": " << _logIdString << "Re-lock failed. Pulse only party found."
235 << std::endl;
236 _messageString.set("Re-lock failed. Pulse only party found.");
237 break;
238 }
239 case CS_RESULT::OXC_NOT_FOUND: {
240 _logger(Loger::INFO) << __func__ << ": " << _logIdString << "Re-lock failed. Pulse not found." << std::endl;
241 _messageString.set("Re-lock failed. Pulse not found.");
242 break;
243 }
244 case CS_RESULT::ERROR: {
245 _logger(Loger::INFO) << __func__ << ": " << _logIdString << "Re-lock failed. Motor error." << std::endl;
246 _messageString.set("Re-lock failed. Motor error. Check logfile.");
247 printMotorErrorLog();
248 break;
249 }
250 default:
251 break;
252 }
253 _switchPulseStatus->setValue(SWITCH_LSU_PULSE_STATUS::ST_FAILED);
254}

256CsaLsuOxcTuneSearchAlg::CS_RESULT CsaLsuOxcAdvAlg::lockInPosition(int newPosition) {
257 this->_actuatorHelper.enableActuatorWithNoAutostart();
258 this->_actuatorHelper.storeAndchangeActuatorSpeedSettings(ACT_SPEED_SETTING::FAST); // Change speed settings to FAST
259 if(_actuator->moveToPositionInSteps(newPosition) != SMAStatusTypes::M_OK) {
260 _logger(Loger::ERROR) << __func__ << ": " << _logIdString
261 << ": Cannot move to position. Actuator status: " << _actuator->getStatus()
262 << ", actuator error: " << _actuator->getError()
263 << ", position: " << _actuator->getCurrentPositionInSteps() << std::endl;
264 _messageString.set("Cannot move to position. Actuator error.");
265 this->_actuatorHelper.restoreSpeedSettings();
266 return CS_RESULT::ERROR;
267 }
268 this->_actuatorHelper.restoreSpeedSettings();
269
270 if(!_oxcSignalSearchThreadInProgress) return CS_RESULT::STOP_BY_USER;
271
272 if(_switchToAnotherPulseCanceled) return CS_RESULT::STOP_BY_USER;
273
274 // try to search signal only by scanning with piezo near range
275 // if successful we can end
276 if(searchForOxcCtrlOutScan()) return CS_RESULT::OXC_FOUND;
277
278 auto pulsesDistance = _maxOxcSignalDistance->getValue();
279 // we will do first scan with 1/4 of max range. Usually signal should not drift more, we will use current position as a start point
280 auto csStatus = coarseSearchForOxcSignal(std::fabs(pulsesDistance / 4), true);
281 // in case signal was partially found we do one more coarse scan with more limited range - 1/8
282 if(csStatus == CS_RESULT::CONTINUE) {
283 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Pulse party found. Rescanning with 1/8 range."
284 << std::endl;
285 _messageString.set("Pulse party found. Rescanning.");
286 csStatus = coarseSearchForOxcSignal(std::fabs(pulsesDistance / 8), true);
287 };
288
289 return csStatus;
290}

292void CsaLsuOxcAdvAlg::printMotorErrorLog() {
293 auto motorStatus = _actuator->getStatus();
294 if((motorStatus == SMAStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON ||
295 motorStatus == SMAStatusTypes::M_POSITIVE_END_SWITCHED_ON)) {
296 _logger(Loger::INFO) << __func__ << ": " << _logIdString
297 << ": Motor hit positive end-switch. Check calibration. Position: "
298 << _actuator->getCurrentPositionInSteps() << std::endl;
299 }
300 else if((motorStatus == SMAStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON ||
301 motorStatus == SMAStatusTypes::M_NEGATIVE_END_SWITCHED_ON)) {
302 _logger(Loger::WARNING) << __func__ << ": " << _logIdString
303 << ": Motor hit negative end-switch. Check calibration. Position: "
304 << _actuator->getCurrentPositionInSteps() << std::endl;
305 }
306 else if(motorStatus == SMAStatusTypes::M_ERROR) {
307 _logger(Loger::ERROR) << __func__ << ": " << _logIdString << ": Motor is in error: " << _actuator->getError()
308 << std::endl;
309 }
