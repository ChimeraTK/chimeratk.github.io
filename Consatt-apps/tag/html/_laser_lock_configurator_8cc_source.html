32 void LaserLockConfigurator::initDaqRfCalibCoeff() {
33  double phaseRange = 2.0;
34 
35  // calibration constant for y-axis in ps for channel 4 AND unwrapped channel 2 in [ps]
36  double coeffPs = 1.0 / (_laserFreqMHz * 1e6) / phaseRange * 1e12;
37  _daq->getDAQ()->getSequence(5)->getUnitConverter()->setParameters(1, 0, coeffPs, 0);
38  _daq->getDAQ()->getSequence(6)->getUnitConverter()->setParameters(1, 0, coeffPs, 0);
39  // this channel we want in [fs]
40  _daq->getDAQ()->getSequence(3)->getUnitConverter()->setParameters(1, 0, (coeffPs * 1000), 0);
41 
42  std::cout << "Calibration constant in [ps] for channel 4 (DAQ:5 DAQ:6) AND unwrapped channel 2 y-axis (DAQ:3): "
43  << coeffPs << std::endl;
44 
45  // calibration constant fo y-axis in fs for channel 2
46  double coeffFs = 1.0 / (_rfFreqMHz * 1e6 + _laserFreqMHz * 1e6) / phaseRange * 1e15;
47  _daq->getDAQ()->getSequence(2)->getUnitConverter()->setParameters(1, 0, coeffFs, 0);
48 
49  std::cout << "Calibration constant in [fs] for channel 2 (DAQ:2) y-axis: " << coeffFs << std::endl;
50 
51  double refFreqMhz = 54.16666666666666;
52  CsaAppConfigFileParser::getInstance().getValueAsDataDouble("X2_TIMER_REFERENCE_SIG_FREQ_MHZ", refFreqMhz);
53  double coeffPsX2Timer = 1.0 / (refFreqMhz * 1e6) / phaseRange * 1e12;
54  _daq->getDAQ()->getSequence(19)->getUnitConverter()->setParameters(1, 0, coeffPsX2Timer, 0);
55 
56  std::cout << "Calibration constant in [ps] for x2timer (DAQ:19) y-axis: " << coeffPsX2Timer << std::endl;
57 
58  _daq->getDAQ()->getSequence(2)->getUnitConverter()->synchronize();
59  _daq->getDAQ()->getSequence(3)->getUnitConverter()->synchronize();
60  _daq->getDAQ()->getSequence(5)->getUnitConverter()->synchronize();
61  _daq->getDAQ()->getSequence(6)->getUnitConverter()->synchronize();
62  _daq->getDAQ()->getSequence(19)->getUnitConverter()->synchronize();
63 }
