19 bool OxcSignalSearchLaserSys::checkSystemStateCore() {
20  if(_extData.ctrlManager->isCtrlActive() == false) {
21  // if (_systemState) // this is to print message only once - when returned status goes from true to false.
22  _logger(Loger::INFO) << __func__ << ": "
23  << "Controller not active. Report LASER_UNLOCKED event." << std::endl;
24  setStatusAndError(OXC_SR_ST_ERR::LASER_UNLOCKED, true);

26  }
27 
28  if(_checkCtrlSwitch && _extData.ctrlManager->isCtrlSwitchingInProgress()) {
29  // if (_systemState)
30  _logger(Loger::INFO) << __func__ << ": "
31  << "Controller switching is in progress. Report LASER_UNLOCKED event." << std::endl;
32  setStatusAndError(OXC_SR_ST_ERR::LASER_UNLOCKED, true);
33  return false;
34  }
35 
36  if(!(_extData.ctrlManager->getCurrentLockType() == LOCK_TYPE::RF_HARM ||
37  _extData.ctrlManager->getCurrentLockType() == LOCK_TYPE::RF_FUND)) {
38  // if (_systemState)
39  _logger(Loger::INFO) << __func__ << ": "
40  << "Controller is not RF locked. Report LASER_UNLOCKED event." << std::endl;
41  setStatusAndError(OXC_SR_ST_ERR::LASER_UNLOCKED, true);
42  return false;
43  }
44 
45  auto globalStatus = _extData.laserLockStatusValidator->getStatus();
46  if(globalStatus == LOCK_STATUS_VALUE::ALTERNATING) {
47  if(_checkCtrlSwitch) {
48  if(_systemState)
49  _logger(Loger::INFO) << __func__ << ": "
50  << "Laser status is alternating and status was not triggered by automatic OXC search. "
51  "Report LASER_UNLOCKED event."
52  << std::endl;
53  setStatusAndError(OXC_SR_ST_ERR::LASER_UNLOCKED, true);
54  return false;
55  }
56  }
57  else if(globalStatus == LOCK_STATUS_VALUE::WARNING &&
58  _extData.laserLockStatusValidator->getStatus() == LOCK_STATUS_VALUE::WARNING &&
59  _extData.ctrlManager->checkIfLockedToSelectedStage() == false) {
60  _logger(Loger::DETAIL)
61  << __func__ << ": "
62  << "Laser status is warning - reason: \"not locked to desired input stage\" . OXC search can be done."
63  << std::endl;
64  }
65  else if(globalStatus != LOCK_STATUS_VALUE::OK) {
66  if(_systemState)
67  _logger(Loger::INFO) << __func__ << ": "
68  << "Laser status is 'not locked' (not OK). Report LASER_UNLOCKED event." << std::endl;
69  setStatusAndError(OXC_SR_ST_ERR::LASER_UNLOCKED, true);
70  return false;
71  }
72 
73  return true;
74 }
