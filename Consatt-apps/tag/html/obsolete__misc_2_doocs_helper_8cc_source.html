31void DoocsHelper::update() {
32 auto cmd = _cmd.value();
33 _cmd.set_value(0);
34
35 if(cmd == 1) {
36 loadLogFileTail();
37 _autoupdateCnt = 0;
38 }
39
40 if(cmd == 2) {
41 loadLogFileWithTimeFilter();
42 _autoupdateCnt = 0;
43 }
44
45 if(_logFileTailLineNumberAutoUpdate.value()) {
46 _autoupdateCnt++;
47 if(_autoupdateCnt == 20) {
48 loadLogFileTail();
49 _autoupdateCnt = 0;
50 }
51 }
52
53 if(_updateCnt % 50 == 0) {
54 trimLogFile();
55 }
56
57 _updateCnt++;
58}

93void DoocsHelper::trimLogFile() {
94 // set default
95 if(_logFileMaxSizeKb.value() == 0) _logFileMaxSizeKb.set_value(102400); // 100MB (100*1024kB)
96 if(_logFileTruncateFactor.value() == 0) _logFileTruncateFactor.set_value(0.7);
97
98 // check currnet file size and number of lines
99 std::string logFileName = std::string(__progname) + ".log";
100 _logFileCurrentSize.set_value(fileSize(logFileName));
101
102 // if max size is 0 or lower trimming is OFF
103 if(_logFileMaxSizeKb.value() > 0) {
104 // only file biger that 1024 (1MB) can be trimmed
105 if(_logFileMaxSizeKb.value() < 1024) _logFileMaxSizeKb.set_value(1024);
106
107 if(_logFileTruncateFactor.value() > 0.9)
108 _logFileTruncateFactor.set_value(0.9);
109 else if(_logFileTruncateFactor.value() < 0.1)
110 _logFileTruncateFactor.set_value(0.1);
111
112 if(_logFileCurrentSize.value() > _logFileMaxSizeKb.value()) {
113 std::thread t1(std::thread(&DoocsHelper::trimLogFileCore, this));
114 t1.detach();
115 }
116 }

119void DoocsHelper::trimLogFileCore() {
120 auto time_start = std::chrono::high_resolution_clock::now();
121
122 std::string logFileName = std::string(__progname) + ".log";
123 std::string tempFileName(logFileName + std::string(".tmp"));
124 std::fstream logFile(logFileName.c_str(), std::fstream::in);
125 std::fstream logFileTemp(tempFileName.c_str(), std::fstream::out | std::fstream::app);
126
127 int oldSizeKb = _logFileCurrentSize.value();
128 int oldLinesNo = fileLinesNo(logFileName);
129 std::cout << "DoocsHelper::trimLogFileCore: Current size [kB]: " << _logFileCurrentSize.value()
130 << " Max size [kB]: " << _logFileMaxSizeKb.value() << ", trim factor: " << _logFileTruncateFactor.value()
131 << std::endl;
132
133 int newLinesNo = oldLinesNo * _logFileTruncateFactor.value();
134 int newLinesNoCnt = 0;
135
136 std::string line;
137 if(logFile.is_open() && logFileTemp.is_open()) {
138 for(int i = 0; std::getline(logFile, line); ++i) {
139 if(i > (oldLinesNo - newLinesNo)) {
140 newLinesNoCnt++;
141 logFileTemp << line << std::endl;
142 }
143 }
144
145 // close both files
146 logFile.close();
147 logFileTemp.close();
148 // reopen files in different mode
149 std::fstream logFile(logFileName.c_str(), std::fstream::out | std::fstream::trunc);
150 std::fstream logFileTemp(tempFileName.c_str(), std::fstream::in);
151
152 // rewrite temporary file to log file
153 if(logFile.is_open() && logFileTemp.is_open())
154 while(std::getline(logFileTemp, line)) logFile << line << std::endl;
155 // close them again
156 logFile.close();
157 logFileTemp.close();
158 // remove temp file
159 if(remove(tempFileName.c_str()) != 0)
160 std::cout << "DoocsHelper::trimLogFileCore: Error deleting temporary file: " << tempFileName << std::endl;
161
162 std::cout << "DoocsHelper::trimLogFileCore: Old log file size[kb]: " << oldSizeKb
163 << " , new file size[kb]: " << fileSize(logFileName) << std::endl;
164 std::cout << "DoocsHelper::trimLogFileCore: Old log file lines: " << oldLinesNo
165 << " , new file lines: " << newLinesNoCnt << std::endl;
166 std::cout << "DoocsHelper::trimLogFileCore: Truncation took[ms]: "
167 << std::chrono::duration_cast<std::chrono::milliseconds>(
168 std::chrono::high_resolution_clock::now() - time_start)
169 .count()
170 << std::endl
171 << std::endl;
172 }

175void DoocsHelper::loadLogFileTail() {
176 auto time_start = std::chrono::high_resolution_clock::now();
177 std::ostringstream ss;
178 if(_logFileTailLineNumber.value() > 4000)
179 _logFileTailLineNumber.set_value(4000);
180 else if(_logFileTailLineNumber.value() < 50)
181 _logFileTailLineNumber.set_value(50);
182
183 ss << _logFileTailLineNumber.value();
184
185 std::string command = "tail -" + ss.str() + " " + std::string(__progname) + ".log > temp.log.txt";
186 auto result = std::system(command.c_str());
187 if(result == 0) {
188 // std::cout << "Command is: " << command << ", command result: " << result << std::endl;
189 ss.str("");
190 ss << std::ifstream("temp.log.txt").rdbuf();
191 _logFileTail.set_value(ss.str());
192 std::system("rm -f ./temp.log.txt");
193 }
194 else {
195 std::cout << "Command: " << command << " failed, command result: " << result << std::endl;
196 }
197
198 auto readTimeMs =
199 std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::high_resolution_clock::now() - time_start)
200 .count();
201 if(readTimeMs > 20)
202 std::cout << "DoocsHelper::loadLogFileTail: Warning. Loading " << _logFileTailLineNumber.value()
203 << " lines took[ms]: " << readTimeMs << std::endl;
204}

206void DoocsHelper::loadLogFileWithTimeFilter() {
207 std::ifstream file;
208 std::string filePath = std::string(__progname) + ".log";
209 file.open(filePath.c_str(), std::ios::in);
210
211 if(file.is_open()) {
212 time_t startTime = _logFileFilterStart.value();
213 time_t stopTime = _logFileFilterEnd.value();
214
215 std::string currentLine("");
216 std::string logNewText("");
217 int matchedLineCnt = 0;
218 int maxLines = _logFileTailLineNumber.value();
219 if(maxLines == 0 || maxLines > 5000) {
220 maxLines = 5000;
221 }
222 int lineCnt = 0;
223 int lineCnt1 = 0;
224 std::cout << "startTime: " << startTime << " , endTime: " << stopTime << std::endl;
225
226 while(matchedLineCnt < _logFileTailLineNumber.value() && std::getline(file, currentLine)) {
227 if(currentLine.length() > 20) {
228 time_t currentMarkerTime = getTimeFromMarker(currentLine.substr(0, 19).c_str());
229 if(currentMarkerTime >= startTime && currentMarkerTime <= stopTime) {
230 logNewText += (currentLine + "\n");
231 matchedLineCnt++;
232 }
233 lineCnt1++;
234 }
235 lineCnt++;
236 }
237 _logFileTail.set_value(logNewText);
238 file.close();
239
240 std::cout << "Logfile line cnt: " << lineCnt << " , longer than 20 chars: " << lineCnt1 << " , matched lines"
241 << matchedLineCnt << "\n";
242 }
243}
