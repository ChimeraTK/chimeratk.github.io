187void LaserMenhir1550Controller::update() {
188 if(_protocolFwVer == PROTOCOL_FW_VER::VER_3X)
189 _firmwareVersionFromConfig->setValue("3.x");
190 else if(_protocolFwVer == PROTOCOL_FW_VER::VER_2X)
191 _firmwareVersionFromConfig->setValue("2.x");
192
193 applyUpdatePeriod();
194 // TESTS FOR LOGGER
195 //_menhirCommunicationLogger->logPeriodically("TESTLOG ARW", _command.arw);
196 //_menhirCommunicationLogger->logPeriodically("TESTLOG SEN", _command.sen);
197 //_menhirCommunicationLogger->logPeriodically("TESTLOG VER", _command.ver);
198 //_menhirCommunicationLogger->logPeriodically("TESTLOG LD0", _command.ld0PumpCurrent);
199 //_menhirCommunicationLogger->logPeriodically("TESTLOG LD1", _command.ld1PumpCurrent);
200 // END TESTS FOR LOGGER
201
202 if(_communicationThreadActive) {
203 if(ld0PumpCurrentApplyFlag || ld1PumpCurrentApplyFlag) {
204 std::lock_guard<std::mutex> guard(_commandQueueMutex);
205 if(ld0PumpCurrentApplyFlag) {
206 ld0PumpCurrentApplyFlag = false;
207 _commandQueue.emplace(buildCmdForLDPumpCurrent(0));
208 }
209
210 if(ld1PumpCurrentApplyFlag) {
211 ld1PumpCurrentApplyFlag = false;
212 _commandQueue.emplace(buildCmdForLDPumpCurrent(1));
213 }
214 }
215
216 if(this->isExecuteAutoUpdate()) {
217 // this part of code should be protected by mutex lock_guard, while it access _commandQueue
218 {
219 std::lock_guard<std::mutex> guard(_commandQueueMutex);
220 if(_checkFirmwareVersion || !_firmwareVersionReceivedFromHardware) {
221 _commandQueue.emplace(_command.ver);
222 _checkFirmwareVersion = false;
223 }
224 else {
225 _commandQueue.emplace(_command.arw);
226 _commandQueue.emplace(_command.sen);
227 }
228 }
229 evaluateCommStatus();
230 }
231 }
232 else {
233 evaluateCommStatus();
234 while(!_commandQueue.empty()) _commandQueue.pop();
235 }
236 _communicationErrorsCounter->setValue(_commErrorCnt);
237}

239void LaserMenhir1550Controller::evaluateCommStatus() {
240 if(!_communicationThreadActive) {
241 _communicationStatus->setValue(STATUS_INDICATOR::DISABLED);
242 _communicationLostCnt = (_communicationLostCntMax +
243 1); //_communicationLostCnt value is consider not valid when set to (_communicationLostCntMax + 1)
244 return;
245 }
246 // _communicationLostCnt is set to (_communicationLostCntMax + 1) when communication is OFF, because
247 // status should stay DISABLED until first successful/not successful transfer is done. When transfer is done it sets
248 // _communicationLostCnt to 0 ( success) or 1 (failed) and can be correctly evaluated by below logic
249 if(_communicationLostCnt == (_communicationLostCntMax + 1))
250 _communicationStatus->setValue(STATUS_INDICATOR::DISABLED);
251 else if(_communicationLostCnt == _communicationLostCntMax)
252 _communicationStatus->setValue(STATUS_INDICATOR::ERROR);
253 else if(_communicationLostCnt)
254 _communicationStatus->setValue(STATUS_INDICATOR::WARNING);
255
256 if(_communicationLostCnt) {
257 _checkFirmwareVersion = true;
258 return;
259 }
260
261 // in case status seems to be OK, check the _commErrorCnt
262 //_commErrorCnt must be constant for _commErrorCntIsConstantCntMax update cycles to be sure that no new communication
263 // errors are present
264 if(_commErrorCnt != _commErrorCntPrev)
265 _commErrorCntIsConstantCnt = 0;
266 else if(_commErrorCntIsConstantCnt < _commErrorCntIsConstantCntMax)
267 _commErrorCntIsConstantCnt++;
268
269 if(_commErrorCntIsConstantCnt == _commErrorCntIsConstantCntMax)
270 _communicationStatus->setValue(STATUS_INDICATOR::OK);
271 else
272 _communicationStatus->setValue(STATUS_INDICATOR::WARNING);
273 _commErrorCntPrev = _commErrorCnt;
274}

276void LaserMenhir1550Controller::communicationThread() {
277 _communicationThreadActive = true;
278 _logger(Loger::INFO) << __func__ << ": "
279 << "Start communication thread " << std::endl;
280 while(_stopCommunicationThread == false) {
281 std::this_thread::sleep_for(std::chrono::milliseconds(20));
282 bool cmdQueueEmpty = true;
283 do {
284 std::string currentCommand("");
285 // this part of code should be protected by mutex lock_guard, while it access _commandQueue
286 {
287 std::lock_guard<std::mutex> guard(_commandQueueMutex);
288 if(!_commandQueue.empty()) {
289 currentCommand = _commandQueue.front();
290 _commandQueue.pop();
291 }
292 cmdQueueEmpty = _commandQueue.empty();
293 }
294 if(currentCommand.length()) {
295 _logger(Loger::MAX_DETAIL) << __func__ << ": "
296 << "Processing command " << currentCommand << " , Queue empty: " << cmdQueueEmpty
297 << " , Queue size: " << _commandQueue.size() << std::endl;
298 processCommnand(currentCommand);
299 std::this_thread::sleep_for(std::chrono::milliseconds(5)); // give Menhir 5ms to internally apply all commands
300 }
301 } while(!cmdQueueEmpty && !_stopCommunicationThread);
302 }
303 _logger(Loger::INFO) << __func__ << ": "
304 << "Stop communication thread " << std::endl;
305 _communicationThreadActive = false;
306}

308void LaserMenhir1550Controller::processCommnand(std::string command) {
309 // send command with new line character attached
310 std::string commnadWithNewLine = command;
311 commnadWithNewLine.append("\n");
312 _extData.uart->sendString(commnadWithNewLine);
313 // wait up to defined timeout ms for the response, observe if there are any characters in the RX FIFO
314 uint32_t timeoutCnt = 0;
315 uint32_t timeoutMs = 600;
316 while(_extData.uart->getRxFifoCounter() == 0 && timeoutCnt < timeoutMs) {
317 std::this_thread::sleep_for(std::chrono::milliseconds(1));
318 timeoutCnt++;
319 }
320 auto rxFifoCnt = _extData.uart->getRxFifoCounter();
321 if(rxFifoCnt == 0) {
322 _logger(Loger::ERROR) << __func__ << ": "
323 << "No answer on command: " << command << " . Timeout after " << timeoutMs << " ms. "
324 << std::endl;
325 if(_communicationLostCnt == (_communicationLostCntMax + 1))
326 _communicationLostCnt = 1;
327 else if(_communicationLostCnt < _communicationLostCntMax)
328 _communicationLostCnt++;
329 return;
330 }
331
332 _communicationLostCnt = 0;
333 _logger(Loger::MAX_DETAIL) << __func__ << ": "
334 << "Answer on command: " << command << " received in " << timeoutCnt * 5 << " ms."
335 << std::endl;
336
337 // check whether transmission is over, transmission is over when there is no new characters for 3 ms.
338 // default speed is 38400 baud, so around 4800 bytes/s (char/s), it gives around 4.8 char/ms
339 uint32_t transmissionFinishedCnt = 0;
340 do {
341 std::this_thread::sleep_for(std::chrono::milliseconds(1));
342 auto rxFifoCntNew = _extData.uart->getRxFifoCounter();
343 if(rxFifoCntNew == rxFifoCnt)
344 transmissionFinishedCnt++;
345 else
346 transmissionFinishedCnt = 0;
347 rxFifoCnt = rxFifoCntNew;
348 } while(transmissionFinishedCnt < 3);
349
350 std::string response("");
351 bool responseReceived = _extData.uart->receiveStringNoBlocking(response);
352 if(responseReceived) {
353 _menhirCommunicationLogger->logPeriodically(response, command);
354 processResponse(response, command);
355 }
356 else {
357 _commErrorCnt++;
358 _logger(Loger::ERROR) << __func__ << ": "
359 << "Response on command " << command << " not received. " << std::endl;
360 }
361}

363void LaserMenhir1550Controller::processResponse(std::string response, std::string command) {
364 if(command.compare(_command.ver) == 0) {
365 processResponseForVerCommand(response);
366 }
367 else if(command.compare(_command.arw) == 0) {
368 processResponseForArwCommand(response);
369 }
370 else if(command.compare(_command.sen) == 0) {
371 processResponseForSenCommand(response);
372 }
373 else if(command.substr(0, _command.ld0PumpCurrent.length()).compare(_command.ld0PumpCurrent) == 0) {
374 processResponseForSetLDCurrentCommand(response, command, 0);
375 }
376 else if(command.substr(0, _command.ld1PumpCurrent.length()).compare(_command.ld1PumpCurrent) == 0) {
377 processResponseForSetLDCurrentCommand(response, command, 1);
378 }
379 else {
380 _logger(Loger::ERROR) << __func__ << ": "
381 << "Unknown command. Command is " << command << std::endl;
382 }

385void LaserMenhir1550Controller::processResponseForSetLDCurrentCommand(
386 std::string response, std::string command, int ldId) {
387 _logger(Loger::INFO) << __func__ << ": "
388 << "Answer on command '" << command << "' is: " << response << std::endl;
389 if(response.find("Attempt to set the current outside the valid range.") != std::string::npos) {
390 _logger(Loger::WARNING) << __func__ << ": "
391 << "Attempt to set the current for LD" << ldId << " outside the valid range. " << std::endl;
392 if(ldId == 0) _ld0PumpCurrentOutOfRange->setValue(1);
393 if(ldId == 1) _ld1PumpCurrentOutOfRange->setValue(1);
394 return;
395 }
396 // Current of LD0 was set to: 200.00 mA //Current of LD1 was set to: 200.00 mA
397 response.erase(std::remove_if(response.begin(), response.end(), ::isspace), response.end());
398 std::vector<std::string> result;
399 boost::split(result, response, boost::is_any_of(":"));
400 if(result.size() == (expectedSize = 2)) {
401 if(ldId == 0) {
402 convertAndFill<float>(result[1], _ld0PumpCurrentRb);
403 _ld0PumpCurrentOutOfRange->setValue(0);
404 }
405 else if(ldId == 1) {
406 convertAndFill<float>(result[1], _ld1PumpCurrentRb);
407 _ld1PumpCurrentOutOfRange->setValue(0);
408 }
409 }
410 else {
411 _commErrorCnt++;
412 _logger(Loger::ERROR) << __func__ << ": "
413 << "Invalid format. Response for 'id" << ldId << "' is: : " << response
414 << " . Delimiter: ':'. Expected size of spitted array: " << expectedSize
415 << ", received: " << result.size() << std::endl;
416 }
