70void OxcSignalCorrectionLaser::findRangeIndexesForData() {
71 _oxcCorrRangeMaxIndex = _oxcCorrRangeMinIndex = 0;
72
73 auto xAxisSetting = _extData.oxcInSequence->getCsaDataVector()->getXAxisSettings();
74 float scaleFactorToMs = 1.0;
75
76 if(xAxisSetting.unitString.compare("ms")) {
77 if(xAxisSetting.unitString.compare("sec") == 0)
78 scaleFactorToMs = 0.001;
79 else if(xAxisSetting.unitString.compare("us") == 0)
80 scaleFactorToMs = 1000;
81 else if(xAxisSetting.unitString.compare("ns") == 0)
82 scaleFactorToMs = 1000 * 1000;
83 else if(xAxisSetting.unitString.compare("ps") == 0)
84 scaleFactorToMs = 1000 * 1000 * 1000;
85 }
86 if(xAxisSetting.increment) {
87 _oxcCorrRangeMaxIndex = _oxcCorrRangeMaxMs->getValue() * scaleFactorToMs / xAxisSetting.increment;
88 _oxcCorrRangeMinIndex = _oxcCorrRangeMinMs->getValue() * scaleFactorToMs / xAxisSetting.increment;
89 }
90 else {
91 _limitRangeStatus = STATUS::ERROR;
92 return;
93 }
94 // std::cout << "unit: " << xAxisSetting.unitString << " Max time: " << _oxcCorrRangeMaxMs->getValue() << " , max
95 // index: " << _oxcCorrRangeMaxIndex << " , min time: " << _oxcCorrRangeMinMs->getValue() << " , min index " <<
96 // _oxcCorrRangeMinIndex << std::endl;
97
98 if(_oxcCorrRangeMaxIndex > _oxcInDataCorrectedRaw.size()) _oxcCorrRangeMaxIndex = _oxcInDataCorrectedRaw.size();
99
100 if(_oxcCorrRangeMinIndex > _oxcInDataCorrectedRaw.size()) _oxcCorrRangeMinIndex = _oxcInDataCorrectedRaw.size();
101
102 if(_oxcCorrRangeMaxIndex == _oxcCorrRangeMinIndex)
103 _limitRangeStatus = STATUS::DISABLED;
104 else if(_oxcCorrRangeMaxIndex < _oxcCorrRangeMinIndex)
105 _limitRangeStatus = STATUS::ERROR;
106}

108void OxcSignalCorrectionLaser::limitDataToDefinedTimeRange() {
109 _limitRangeStatus = STATUS::OK;
110 if(_enableRangeLimits->getValue() == 0) {
111 _limitRangeStatus = STATUS::DISABLED;
112 return;
113 }
114
115 if(_oxcInDataCorrectedRaw.size() != _oxcInPDataCorrectedRaw.size() ||
116 _oxcInDataCorrectedRaw.size() != _oxcInNDataCorrectedRaw.size()) {
117 _logger(Loger::ERROR) << __func__ << ": "
118 << "Size mismatch of raw vectors. OXC: " << _oxcInDataCorrectedRaw.size()
119 << " OXC P: " << _oxcInPDataCorrectedRaw.size()
120 << " OXC N: " << _oxcInNDataCorrectedRaw.size() << std::endl;
121 _limitRangeStatus = STATUS::ERROR;
122 return;
123 }
124 _limitedDataAmountCounter = _oxcInDataCorrectedRaw.size(); // must be set here, its used later for creating the plots
125
126 findRangeIndexesForData();
127 if(_limitRangeStatus != STATUS::OK) return;
128
129 _limitedDataAmountCounter = 0;
130 for(uint32_t i = _oxcCorrRangeMinIndex; i < _oxcCorrRangeMaxIndex; i++, _limitedDataAmountCounter++) {
131 _oxcInPDataCorrectedRaw[_limitedDataAmountCounter] = _oxcInPDataCorrectedRaw[i];
132 _oxcInNDataCorrectedRaw[_limitedDataAmountCounter] = _oxcInNDataCorrectedRaw[i];
133 _oxcInDataCorrectedRaw[_limitedDataAmountCounter] = _oxcInDataCorrectedRaw[i];
134 }
135 _oxcInPDataCorrectedRaw.resize(_limitedDataAmountCounter);
136 _oxcInNDataCorrectedRaw.resize(_limitedDataAmountCounter);
137 _oxcInDataCorrectedRaw.resize(_limitedDataAmountCounter);
138}

140void OxcSignalCorrectionLaser::correctRawData() {
141 if(_disableCorrection->getValue() == 1) {
142 //_logger(Loger::DETAIL) << __func__ << ": " << "Convertion disabled by user." << std::endl;
143 _correctionStatus = STATUS::DISABLED;
144 return;
145 }
146 else if(_oxcCorrN_m->getValue() == 0) {
147 //_logger(Loger::DETAIL) << __func__ << ": " << "Convertion disabled. PAR_P_M is 0.0 ." << std::endl;
148 _correctionStatus = STATUS::DISABLED;
149 return;
150 }
151
152 double n_ctm = _oxcCorrN_ctm->getValue();
153 double n_m = _oxcCorrN_m->getValue();
154 double p_ctm = _oxcCorrP_ctm->getValue();
155 double p_m = _oxcCorrP_m->getValue();
156 // auto calParamStart= std::chrono::high_resolution_clock::now();
157 double p_cc = (n_ctm / n_m);
158 double n_cc_1 = (p_m - ((n_ctm / n_m) * p_ctm)) / n_m;
159 double n_cc_2 = p_ctm / n_m;
160 // auto calParamDuration = std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::high_resolution_clock::now()
161 // - calParamStart).count(); _logger(Loger::INFO) << __func__ << ": " << "p_cc [(n_ctm / n_m)]: " << p_cc << " n_cc_1
162 //[(p_m - ((n_ctm / n_m) * p_ctm)) / n_m]: " << n_cc_1 << " n_cc_2 [p_ctm / n_m]: " << n_cc_2 << std::endl;
163
164 if(p_cc == 0 && n_cc_1 == 1 && n_cc_2 == 0) {
165 //_logger(Loger::DETAIL) << __func__ << ": " << "Convertion disabled. Parameters set in a way that output will be
166 // equal to input data." << std::endl;
167 _correctionStatus = STATUS::DISABLED;
168 return;
169 }
170
171 if(_oxcInPDataCorrectedRaw.size() != _oxcInNDataCorrectedRaw.size() ||
172 _oxcInPDataCorrectedRaw.size() != _oxcInDataCorrectedRaw.size()) {
173 _logger(Loger::ERROR) << __func__ << ": "
174 << "Size mismatch of raw vectors. OXC: " << _oxcInDataCorrectedRaw.size()
175 << " OXC P: " << _oxcInPDataCorrectedRaw.size()
176 << " OXC N: " << _oxcInNDataCorrectedRaw.size() << std::endl;
177 _correctionStatus = STATUS::ERROR;
178 return;
179 }
180 // auto calCorrectionStart= std::chrono::high_resolution_clock::now();
181 for(uint32_t i = 0; i < _oxcInDataCorrectedRaw.size(); i++) {
182 double newInP = _oxcInPDataCorrectedRaw[i] - (p_cc * _oxcInNDataCorrectedRaw[i]);
183 double newInN = (_oxcInNDataCorrectedRaw[i] * n_cc_1) - (_oxcInPDataCorrectedRaw[i] * n_cc_2);
184 double newIn = newInP - newInN;
185
186 _oxcInPDataCorrectedRaw[i] = newInP;
187 _oxcInNDataCorrectedRaw[i] = newInN;
188 _oxcInDataCorrectedRaw[i] = newIn;
189 }
190 _correctionStatus = STATUS::OK;
191 // auto calCorrectionDuration =
192 // std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::high_resolution_clock::now()
193 // - calCorrectionStart).count();
194}

218void OxcSignalCorrectionLaser::fillCorrectedSpectrum(bool useOriginalData,
219 boost::shared_ptr<CsaMultiplexDataDaqSequence> daqSeqShP, boost::shared_ptr<CsaPlot<float>> plotShP,
220 std::vector<float>& data) {
221 auto unitConverter = daqSeqShP->getUnitConverter();
222 auto& plotDataBuff = plotShP->getBuffer();
223
224 uint32_t length = 0;
225 if(useOriginalData) {
226 //_logger(Loger::MAX_DETAIL) << __func__ << ": " << "Using orignal data. Status is not OK." << std::endl;
227 auto ptrToOrigBuffer = daqSeqShP->getCsaDataVector()->getBuffer();
228 length = daqSeqShP->getDaqReadLength();
229 for(uint32_t i = 0; i < length; i++) plotDataBuff[i] = ptrToOrigBuffer[i];
230 }
231 else {
232 length = data.size();
233 if(unitConverter->isEnabled()) {
234 if(unitConverter->getMode() == 1 && unitConverter->getA2() == 0 && unitConverter->getA0() == 0) {
235 //_logger(Loger::MAX_DETAIL) << __func__ << ": " << "Units converter is enabled and in linear mode." << std::endl;
236 auto cc = unitConverter->getA1();
237 for(uint32_t i = 0; i < length; i++) plotDataBuff[i] = data[i] * cc;
238 }
239 else {
240 //_logger(Loger::MAX_DETAIL) << __func__ << ": " << "Units converter is enabled and in not linear mode." << std::endl;
241 for(uint32_t i = 0; i < length; i++) plotDataBuff[i] = unitConverter->convert(data[i]);
242 }
243 }
244 else {
245 //_logger(Loger::MAX_DETAIL) << __func__ << ": " << "Units converter is disabled." << std::endl;
246 plotDataBuff = plotShP->getBuffer();
247 for(uint32_t i = 0; i < length; i++) plotDataBuff[i] = data[i];
248 }
249 }
250 plotShP->setSendLengthAndResize(length);
251 // plotShP->csaSendLength(length);
252 auto xSettings = daqSeqShP->getCsaDataVector()->getXAxisSettings();
253 if(_limitRangeStatus == STATUS::OK) {
254 xSettings.start = _oxcCorrRangeMinIndex * xSettings.increment;
255 }
256 plotShP->setXAxisSettings(xSettings);
257}
