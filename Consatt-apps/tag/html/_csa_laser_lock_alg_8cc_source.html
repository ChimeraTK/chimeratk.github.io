201void CsaLaserLockAlg::update() {
202 commandInternal();
203
204 auto loTableACAllowed = loTableAutoRecalibAllowed();
205 auto rf2rf4ACAllowed = rf2rf4AutoRecalibAllowed();
206
207 _autoRecalibOff->setValue(0);
208 if(!loTableACAllowed) _autoRecalibOff->setValue(_autoRecalibOff->getValue() | (1 << 0));
209 if(!rf2rf4ACAllowed) _autoRecalibOff->setValue(_autoRecalibOff->getValue() | (1 << 1));
210
211 // std::cout << "loTableACAllowed " << loTableACAllowed << " , rf2rf4ACAllowed " << rf2rf4ACAllowed << std::endl;
212
213 _autoRecalibActive->setValue(0);
214 if(_autoRecalibrateActive) {
215 _autoRecalibActive->setValue(1);
216 return;
217 }
218
219 // LO Table part !!!!!
220
221 // hard timing reference calculation - update to calculate current error and apply feedback if enabled
222 _loTableCorrection->update();
223 // check if full re-calibration is needed.
224 if(loTableACAllowed &&

226 _logger(Loger::WARNING) << __func__ << ": "

228 autorecalibrateThread(CORRECTION_TYPE::FULL);
229 return;
230 }
231
232 int loFactor = _loTableCorrection->getBucketCorrection();
233 if(_loTableCorrection->getManualBucketCorrectionEnabled() == 1) {
234 loFactor = _loTableCorrection->getManualBucketCorrection() - _loTableCorrection->getBucketCorrectionApplied();
235 }
236 if(loFactor && loFactor != _prevLoTablePhaseBucketCorrectionAuto && autoRecalibAllowedExternal()) {
237 _loTableCorrection->applyLoTableBucketOffset(loFactor);
238 _prevLoTablePhaseBucketCorrectionAuto = loFactor;
239 }
240
241 // RF/RF4 part !!!!!
242
243 // calculates RF2/RF4 correction
244 _rf2rf4Correction->update();

247 _logger(Loger::WARNING) << __func__ << ": "

249 autorecalibrateThread(CORRECTION_TYPE::RF2RF4);

251 }
252
253 int rf2rf4Factor = _rf2rf4Correction->getBucketCorrection();
254 if(_rf2rf4Correction->getManualBucketCorrectionEnabled() == 1) {
255 rf2rf4Factor = _rf2rf4Correction->getManualBucketCorrection() - _rf2rf4Correction->getBucketCorrectionApplied();
256 }
257 if(rf2rf4Factor && rf2rf4Factor != _prevRf2Rf4BucketCorrection && autoRecalibAllowedExternal()) {
258 _rf2rf4Correction->applyR2Rf4BucketOffset(rf2rf4Factor);
259 _prevRf2Rf4BucketCorrection = rf2rf4Factor;
