49void UnlockEventDataFileWriter::storeUnlockEvent(std::string message) {
50 std::ostringstream oss;
51 oss << ConsattGenericUtils::getCurrentDateWithMiliseconds("%Y-%m-%d_%H-%M-%S") << "_" << _fileNamesPrefix << "txt";
52 std::string fullPath = this->_folderName + "/" + oss.str();
53 // std::cout << "FullPath: " << fullPath << std::endl;
54 std::ofstream file;
55 file.open(fullPath.c_str()); // create regular file
56 if(file.is_open()) {
57 file << "EVENT START " << std::endl;
58 file << "MESSAGE: " << message << std::endl;
59
60 this->storeData(file);
61
62 file << "EVENT END" << std::endl;
63 file.close();
64 }
65 else {
66 std::cout << "File is not opened. " << std::endl;
67 }
68}
