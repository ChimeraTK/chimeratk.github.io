29void FastCalibrationOxcLsu::calibrateCore() {
30 auto time_start = std::chrono::high_resolution_clock::now();
31 auto duration =
32 std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::high_resolution_clock::now() - time_start)
33 .count();
34 auto oxcDataBuffer = _detectorSignal->getBuffer();
35 _logger(Loger::DETAIL) << __func__ << ": "
36 << "Size of OXC data buffer: " << oxcDataBuffer.size()
37 << " send length: " << _detectorSignal->csaSendLength() << std::endl;
38
39 uint32_t i = 0;
40 for(i = 0; i < oxcDataBuffer.size() && i < 16384; i++) _detectorResponce->setValue(oxcDataBuffer[i], i);
41 _detectorResponce->setXAxisSettings(_detectorSignal->getXAxisSettings());
42 _detectorResponce->csaSendLength(i);
43
44 auto excitationDataBuffer = _excitationSignal->getBuffer();
45 _logger(Loger::DETAIL) << __func__ << ": "
46 << "Size of excitation data buffer: " << excitationDataBuffer.size()
47 << " send length: " << _excitationSignal->csaSendLength() << std::endl;
48 _logger(Loger::DETAIL) << __func__ << ": "
49 << "Excitation to Fs factor: " << _excitationToFs << std::endl;
50 // auto shapeGenData = _ctrlSpCorrectionMem->getShapeGenerator()->getShapeDataVector();
51 for(i = 0; i < excitationDataBuffer.size() && i < 16384; i++)
52 _excitationAsTime->setValue(excitationDataBuffer[i] * _excitationToFs, i);
53 _excitationAsTime->setXAxisSettings(_excitationSignal->getXAxisSettings());
54 _excitationAsTime->csaSendLength(i);
55
56 // find max and min indexes of excitation
57 double max = -10000000;
58 double min = 10000000;
59 int32_t maxIndex = 0;
60 int32_t minIndex = 0;
61 for(i = 0; i < _excitationAsTime->csaSendLength(); i++) {
62 if(_excitationAsTime->getValue(i) > max) {
63 maxIndex = i;
64 max = _excitationAsTime->getValue(i);
65 }
66
67 if(_excitationAsTime->getValue(i) < min) {
68 minIndex = i;
69 min = _excitationAsTime->getValue(i);
70 }
71 }
72 int32_t distance = std::abs(maxIndex - minIndex);
73 _logger(Loger::DETAIL) << __func__ << ": "
74 << "Excitation max: " << max << " on index: " << maxIndex << " min: " << min
75 << " on index: " << minIndex << " distance: " << distance << std::endl;
76
77 max = -10000000;
78 min = 10000000;
79 maxIndex = 0;
80 minIndex = 0;
81 for(i = 0; i < _detectorResponce->csaSendLength(); i++) {
82 if(_detectorResponce->getValue(i) > max) {
83 maxIndex = i;
84 max = _detectorResponce->getValue(i);
85 }
86
87 if(_detectorResponce->getValue(i) < min) {
88 minIndex = i;
89 min = _detectorResponce->getValue(i);
90 }
91 }
92 distance = std::abs(maxIndex - minIndex);
93 _logger(Loger::DETAIL) << __func__ << ": "
94 << "OXC response max: " << max << " on index: " << maxIndex << " min: " << min
95 << " on index: " << minIndex << " distance: " << distance << std::endl;
96
97 if(maxIndex == minIndex) {
98 _logger(Loger::ERROR)
99 << __func__ << ": "
100 << "OXC response maximum index equal to minimum index. Signal seems to be constant. Cannot continue."
101 << std::endl;
102 clearTable();
103 return;
104 }
105
106 // create vectors for calculation
107 uint32_t maxIndexForCalc = 0;
108 uint32_t minIndexForCalc = 0;
109
110 if(maxIndex > minIndex) {
111 minIndexForCalc = minIndex + ((int)distance * 0.3);
112 maxIndexForCalc = maxIndex - ((int)distance * 0.3);
113 }
114 else {
115 minIndexForCalc = maxIndex + ((int)distance * 0.3);
116 maxIndexForCalc = minIndex - ((int)distance * 0.3);
117 }
118 _logger(Loger::DETAIL) << __func__ << ": "
119 << "Min index for calc: " << minIndexForCalc << " max index for calc: " << maxIndexForCalc
120 << std::endl;
121 // create vectors for calculations
122
123 auto xAxisSettings = _detectorResponce->getXAxisSettings();
124 xAxisSettings.start = minIndexForCalc * xAxisSettings.increment;
125
126 std::vector<double> exticationTimeForCalc;
127 std::vector<double> oxcVoltageForCalc;
128 std::vector<double> oxcVoltageForCalcXAxis;
129 for(i = minIndexForCalc; i < maxIndexForCalc; i++) {
130 exticationTimeForCalc.push_back(_excitationAsTime->getValue(i));
131 oxcVoltageForCalc.push_back(_detectorResponce->getValue(i));
132 oxcVoltageForCalcXAxis.push_back(i * xAxisSettings.increment);
133 }
134
135 _logger(Loger::DETAIL) << __func__ << ": "
136 << "Size of exticationTimeForCalc: " << exticationTimeForCalc.size()
137 << " , size of exticationTimeForCalc: " << oxcVoltageForCalc.size() << std::endl;
138 double ccFactorA = 0;
139 double ccFactorB = 0;
140 linearFit(exticationTimeForCalc.size(), exticationTimeForCalc.data(), oxcVoltageForCalc.data(), ccFactorA, ccFactorB);
141 _logger(Loger::INFO) << __func__ << ": "
142 << "CC value (a param): " << ccFactorA << ", b param: " << ccFactorB << std::endl;
143 if(ccFactorA) _ccFsToV->setValue(std::fabs(1 / ccFactorA));
144
145 double linFitCcFactorA = 0;
146 double linFitCcFactorB = 0;
147 linearFit(exticationTimeForCalc.size(), oxcVoltageForCalcXAxis.data(), oxcVoltageForCalc.data(), linFitCcFactorA,
148 linFitCcFactorB);
149 for(i = 0; i < oxcVoltageForCalc.size() && i < 16384; i++) {
150 _detectorResponcePointsForCalc->setValue(oxcVoltageForCalc[i], i);
151
152 float y = linFitCcFactorA * (xAxisSettings.start + i * xAxisSettings.increment) + linFitCcFactorB;
153 _detectorResponceCalcLinearFit->setValue(y, i);
154 }
155
156 _detectorResponcePointsForCalc->setXAxisSettings(xAxisSettings);
157 _detectorResponcePointsForCalc->csaSendLength(i);
158 _detectorResponceCalcLinearFit->setXAxisSettings(xAxisSettings);
159 _detectorResponceCalcLinearFit->csaSendLength(i);
160
161 _logger(Loger::INFO) << __func__ << ": "
162 << "calibration core: " << duration << " milliseconds." << std::endl;
163 //_calibratePhase = CALIBRATE_CORE_END;
164 // clearTable();
165}
