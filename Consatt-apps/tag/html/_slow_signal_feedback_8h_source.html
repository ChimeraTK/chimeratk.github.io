31template<class T = float>
32class MovingAverage {
33 unsigned int _avgLenght{0};
34 std::deque<T> _dataQueue;
35 T _average{0};
36
37 public:
38 MovingAverage(){};
39 MovingAverage(unsigned int lenght) : _avgLenght(_avgLenght){};
40
41 void add(T input) {
42 // remove data when size changed or we have enough data is queue
43 while(_dataQueue.size() >= _avgLenght) _dataQueue.pop_front();
44 _dataQueue.push_back(input);
45 }
46
47 T average() {
48 _average = 0;
49 for(auto& element : _dataQueue) _average += element;
50
51 if(_dataQueue.size()) _average /= _dataQueue.size();
52 // std::cout << "OXC valid: " << *(_dataQueue.end()-1) << " , avg: " << _average << " , queue size: " <<
53 // _dataQueue.size() << std::endl;
54 return _average;
55 }
56
57 T average(T input) {
58 add(input);
59 return average();
60 }
61
62 T operator()(T input) { return average(input); }
63
64 T operator()() { return average(); }
65
66 void setLenght(unsigned int lenght) { _avgLenght = lenght; }
67
68 unsigned int getLenght(unsigned int lenght) { return _avgLenght; }
69
70 T getAverage() { return _average; }
71
72 void clear() { _dataQueue.clear(); }
73};
