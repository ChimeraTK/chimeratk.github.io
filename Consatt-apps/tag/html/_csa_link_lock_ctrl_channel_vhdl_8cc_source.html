119void CsaLinkLockCtrlChannelVhdl::calcAvgPhotoDiodeSygnal() {
120 if(_mathPlotSelector->getValue() == 0) {
121 auto& photoDiodePositive = *(_globalDmaDaq->getDAQ()
122 ->getSequence(_controllerChannelId * 2 + _photoDiodePositiveDaqOffset)
123 ->getCsaDataVector());
124 auto& photoDiodeNegative = *(_globalDmaDaq->getDAQ()
125 ->getSequence(_controllerChannelId * 2 + _photoDiodeNegativeDaqOffset)
126 ->getCsaDataVector());
127
128 if(photoDiodePositive.size() == photoDiodeNegative.size()) {
129 for(uint32_t i = 0; i < photoDiodePositive.size(); i++) {
130 float average = (photoDiodeNegative[i] + photoDiodePositive[i]) / 2.0;
131 _mathPlot->setValue(average, i);
132 }
133 _mathPlot->csaSendLength(photoDiodePositive.size());
134
135 auto xAxisSettings = _globalDmaDaq->getDAQ()
136 ->getSequence(_controllerChannelId * 2 + _photoDiodePositiveDaqOffset)
137 ->getCsaDataVector()
138 ->getXAxisSettings();
139 _mathPlot->setXAxisSettings(xAxisSettings);
140 }
141 else
142 std::cout << "Size mismatch. positive diode: " << photoDiodePositive.size()
143 << " , negative diode: " << photoDiodeNegative.size() << std::endl;
144 }
145 else if(_mathPlotSelector->getValue() == 2 || _mathPlotSelector->getValue() == 1) {
146 auto& ctrlIn =
147 *(_globalDmaDaq->getDAQ()->getSequence(_controllerChannelId * 1 + _ctrlInDaqOffset)->getCsaDataVector());
148 auto alpha = _mathHpfParameter->getValue();
149 if(_mathPlotSelector->getValue() == 2) {
150 //_mathPlot->setValue(ctrlIn[0], 0);
151 _mathPlot->setValue(0, 0);
152 for(uint32_t i = 0; i < ctrlIn.size() - 1; i++)
153 _mathPlot->setValue((alpha * _mathPlot->getValue(i) + alpha * (ctrlIn[i + 1] - ctrlIn[i])), i + 1);
154 }
155 else {
156 _mathPlot->setValue(alpha * ctrlIn[0], 0);
157 for(uint32_t i = 0; i < ctrlIn.size() - 1; i++)
158 _mathPlot->setValue((alpha * ctrlIn[i + 1] + (1 - alpha) * _mathPlot->getValue(i)), i + 1);
159 }
160
161 auto gain = _mathHpfAmpParameter->getValue();
162 for(uint32_t i = 0; i < ctrlIn.size(); i++) _mathPlot->setValue(gain * _mathPlot->getValue(i), i);
163
164 _mathPlot->csaSendLength(ctrlIn.size());
165
166 auto xAxisSettings = _globalDmaDaq->getDAQ()
167 ->getSequence(_controllerChannelId * 1 + _ctrlInDaqOffset)
168 ->getCsaDataVector()
169 ->getXAxisSettings();
170 _mathPlot->setXAxisSettings(xAxisSettings);
171 }

174void CsaLinkLockCtrlChannelVhdl::calculateDsForExcitationMemories() {
175 auto userDs = _ctrlMemoriesUserDs->getValue();
176 if(userDs >= 0) {
177 // 10 is constant factor coming from fact that ADC data stream is 5 MHz and FPGA clock is 50 MHz
178 uint32_t dsForR = (10 * (userDs + 1) * (_ctrlInputStageAlias->getDecimation() + 1));
179 uint32_t dsForU = dsForR * (_controllerAlias->getDecimation() + 1);
180 _ctrlMemories->setRMemReadDs(dsForR - 1);
181 _ctrlMemories->setUMemReadDs(dsForU - 1);
182 _ctrlMemories->synchronize();
183 }
