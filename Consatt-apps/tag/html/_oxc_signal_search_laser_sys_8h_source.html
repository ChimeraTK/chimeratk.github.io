35 enum ST_ERR_FLAG {

49 // errors
50 ACTUATOR_ERROR =
51 (1 << 16), // actuator error detected - ODL or RF shifter movement failed (check log file for details)

53 DATA_TIMEOUT_ERROR = (1 << 18), // no new data for calculation - DAQ frozen
54 SETTINGS_ERROR = (1 << 19), // setting error - probably actuator limits
55 LASER_UNLOCKED = (1 << 20), // laser lost lock or input switching triggered during OXC signal search
56 SYSTEM_TIMING_CORRECTION_ERROR = (1 << 21), //
57 SEARCH_START_FORBID_ERROR = (1 << 22), // search cannot be done because starting is forbid
58
59 // warning
60 ACTUATOR_UNEXPECTED_SOFT_END_SW_ERROR = (1

62 ACTUATOR_UNEXPECTED_HARD_END_SW_ERROR = (1
63 << 24), // search alg. assumes that end-switch is never triggered, because there is additional margin for the scans
64 // error
65 FINE_SEARCH_ERROR = (1 << 25), // Fine search should always finish with finding the signal, if not report error
66 //
67 OXC_ZERO_CROSSING_SEARCH_ERROR = (1 << 26) // even when fine search is OK, still zero-crossing search may fail
68 };
