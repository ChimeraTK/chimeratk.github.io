16 struct NewtonPoint {
17  private:
18  double _x0 = 0.0, _x1 = 0.0, _y0 = 0.0, _y1 = 0.0;
19  double _delX = 0;
20  double _delY = 0;
21  double _targetX = 0;
22  double _step = 0;
23 
24  public:
25  NewtonPoint(double x0, double x1, double y0, double y1, double factor = 0.5) {
26  _x0 = x0;
27  _x1 = x1;
28  _y0 = y0;
29  _y1 = y1;
30  _delX = _x1 - _x0;
31  _delY = _y1 - _y0;
32  _targetX = (_delY * _x0 - _delX * _y0) / _delY;
33  _step = (_targetX - _x1) * factor;
34  _targetX = _targetX + _step;
35  }
36 
37  double delX() { return _delX; }
38  double delY() { return _delY; }
39  double x0() { return _x0; }
40  double x1() { return _x1; }
41  double y0() { return _y0; }
42  double y1() { return _y1; }
43  double targetX() { return _targetX; }
44  double step() { return _step; }
45 };

336  double phaseStep = 5;
337  double ffStep = 0.02;

351  _logger(Loger::INFO) << __func__ << ": "

356  << "Finish. Current bias mean: " << _mziBiasIStats->getMean()
