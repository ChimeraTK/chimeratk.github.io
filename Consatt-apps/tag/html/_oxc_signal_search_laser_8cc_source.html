219// blocking method - public
220bool OxcSignalSearchLaser::searchOxcSignal(
221 ACT_TYPE searchActuatorType, bool switchLsuPulseFlag, bool correctTimingWihtRFShifter, bool searchOxcZeroCrossing) {
222 if(_searchInProgress) {
223 _logger(Loger::WARNING) << __func__ << ": "
224 << "OXC search thread already in progress. Cannot start this search !!!!!!!!!!!!!!!!!!!!!!!"
225 << std::endl;
226 return false;
227 }
228
229 _logger(Loger::INFO) << __func__ << ": OXC search algorithm. Selected actuator: " << searchActuatorType
230 << ", switch LSU pulse: " << switchLsuPulseFlag
231 << ", correct timing with RF: " << correctTimingWihtRFShifter << ", adjust RF shifter limits: 1"
232 << ", search OXC zero-crossing RF shifter limits: " << searchOxcZeroCrossing
233 << ", RF correction: " << _rfTimingCorrection->getValue() << " ps." << std::endl;
234
235 if(_startingPaused) {
236 _logger(Loger::WARNING) << __func__
237 << ": OXC signal search starting is forbidden (internally paused). Search cannot be done. "
238 << std::endl;
239 _searchAlgSystem->setStatusAndError(0);
240 _searchAlgSystem->setStatusAndError(OXC_SR_ST_ERR::OXC_SIG_NOT_FOUND, true);
241 _searchAlgSystem->setStatusAndError(OXC_SR_ST_ERR::SEARCH_START_FORBID_ERROR, true);
242 return false;
243 }
244
245 _searchInProgress = true;
246 _currentActuatorType = searchActuatorType;
247 _searchAlgSystem->setStatusAndError(0); // OXC search will start - zero status from previous search
248 _searchAlgSystem->setStatusAndError(OXC_SR_ST_ERR::ACTIVE, true);
249
250 actuatorCsBlock();
251 /*
252 // tests start
253 int switchDirection = 0;
254 int predictedLsuSwitchDirection = 0;
255 if(!checkLsuSwitchDirection(predictedLsuSwitchDirection)) {
256 _logger(Loger::INFO) << __func__ << ": "
257 << "LSU switch direction check failed. Cannot continue." << std::endl;
258 return false;
259 }
260
261 bool status =
262 switchLsuPulseAndDoRfCorrection(switchDirection, predictedLsuSwitchDirection, correctTimingWihtRFShifter);
263 if(!_searchInProgress) {
264 _logger(Loger::INFO) << __func__ << ": "
265 << "Search stopped by user." << std::endl;
266 return false;
267 }
268 _searchInProgress = false;
269 return status;
270 // tests stop
271 */
272 bool result = checkSignalValidAndInRange(searchActuatorType);
273 if(!result)
274 result =
275 searchOxcSignalCore(searchActuatorType, switchLsuPulseFlag, correctTimingWihtRFShifter, searchOxcZeroCrossing);
276
277 if(!result) {
278 _logger(Loger::INFO) << __func__ << ": "
279 << "OXC search thread done. Signal not found." << std::endl;
280 _searchAlgSystem->setStatusAndError(OXC_SR_ST_ERR::OXC_SIG_NOT_FOUND, true);
281 }
282 else {
283 _logger(Loger::INFO) << __func__ << ": "
284 << "OXC search thread done. Signal found. Success!" << std::endl;
285 _searchAlgSystem->setStatusAndError(OXC_SR_ST_ERR::OXC_SIG_FOUND, true);
286 }
287
288 if(!_searchInProgress) {
289 _searchAlgSystem->setStatusAndError(OXC_SR_ST_ERR::STOPPED_BY_USER, true);
290 _logger(Loger::INFO) << __func__ << ": "
291 << "Search stopped by user." << std::endl;
292 }
293
294 actuatorCsRestore();
295
296 _searchAlgSystem->setStatusAndError(OXC_SR_ST_ERR::ACTIVE, false);
297 _searchInProgress = false;
298 _lastSearchResult = result;
299 return result;
300}

324bool OxcSignalSearchLaser::searchOxcSignalCore(
325 ACT_TYPE searchActuatorType, bool switchLsuPulseFlag, bool correctTimingWihtRFShifter, bool searchOxcZeroCrossing) {
326 _logger(Loger::INFO) << __func__ << ": "
327 << "Start core of OXC signal search." << std::endl;
328 if(!_searchAlgSystem->checkSystemState()) {
329 _logger(Loger::INFO) << __func__ << ": "
330 << "System is not OK." << std::endl;
331 return false;
332 }
333 _logger(Loger::INFO) << __func__ << ": "
334 << "System is OK." << std::endl;
335 // float rfTimingCorr = _rfTimingCorrection->getValue();
336
337 bool result = false;
338
339 OXC_SEARCH_TYPE searchType = OXC_SEARCH_TYPE::SINGLE;
340 if(switchLsuPulseFlag) searchType = OXC_SEARCH_TYPE::BEFORE_LSU_SWITCH;
341 if(searchActuatorType == ACT_TYPE::ODL)
342 result = _searchAlgWithOdl->search(searchType, searchOxcZeroCrossing);
343 else
344 result = _searchAlgWithRf->search(searchType, searchOxcZeroCrossing);
345
346 if(!_searchInProgress) return false;
347
348 // case fine search failed with error do not switch pulse and stay in current position
349 if(_searchAlgSystem->getStatusAndError(OXC_SR_ST_ERR::FINE_SEARCH_ERROR)) {
350 _logger(Loger::INFO) << __func__ << ": "
351 << "Signal was found in coarse search, but fine search reported an error. Finish OXC search."
352 << std::endl;
353 return false;
354 }
355
356 if(result == false && switchLsuPulseFlag == true && _searchAlgSystem->checkSystemState()) {
357 if(_searchAlgSystem->isAnyErrorActive()) {
358 _logger(Loger::INFO) << __func__ << ": "
359 << "Error was reported during first search. LSU switch cannot be done." << std::endl;
360 return false;
361 }
362
363 // check if one must do timing correction (for XFEL)
364 // if yes one must in which direction link will switch
365 int predictedLsuSwitchDirection = 0;
366 float offsetBeforeRfCorrection = 0;
367 if(correctTimingWihtRFShifter) {
368 if(!checkLsuSwitchDirection(predictedLsuSwitchDirection)) {
369 _logger(Loger::INFO) << __func__ << ": "
370 << "LSU switch direction check failed. Cannot continue." << std::endl;
371 return false;
372 }
373 offsetBeforeRfCorrection = _extData.rfTimingShifter->getTimingShiftOffset();
374 _logger(Loger::INFO) << __func__ << ": "
375 << "RF system timing offset before LSU switch is " << offsetBeforeRfCorrection << " ps."
376 << std::endl;
377 }
378
379 int switchDirection = 0;
380 bool switchStatus = false;
381 if(correctTimingWihtRFShifter) { // one knows direction of LSU switch and want to do RF timing correction
382 _logger(Loger::INFO) << __func__ << ": "
383 << "Switch LSU pulse (timeout: " << getLsuSwitchTimeout() << " sec.) in direction "
384 << predictedLsuSwitchDirection << " and correct timing." << std::endl;
385
386 switchStatus = switchLsuPulseAndDoRfCorrection(switchDirection, predictedLsuSwitchDirection, true);
387 }
388 else {
389 _logger(Loger::INFO) << __func__ << ": "
390 << "Switch LSU pulse (timeout: " << getLsuSwitchTimeout()
391 << " sec.) and do NOT correct timing." << std::endl;
392 switchStatus = switchLsuPulseAndDoRfCorrection(switchDirection);
393 }
394
395 if(!_searchInProgress) return false;
396
397 if(switchStatus) {
398 if(correctTimingWihtRFShifter && predictedLsuSwitchDirection != switchDirection) {
399 _logger(Loger::ERROR) << __func__ << ": "
400 << "Switch LSU pulse failed. Switched in wrong direction. Expected dir: "
401 << predictedLsuSwitchDirection << ", resulted " << switchDirection
402 << " . Cannot continue." << std::endl;
403 _searchAlgSystem->setStatusAndError(OXC_SR_ST_ERR::LSU_SWITCH_ERROR, true);
404 return false;
405 }
406
407 result = searchOxcSignalAfterLsuPulseSwitch(searchActuatorType, searchOxcZeroCrossing);
408
409 if(!_searchInProgress || !_searchAlgSystem->checkSystemState()) return false;
410
411 if(_searchAlgSystem->getStatusAndError(OXC_SR_ST_ERR::FINE_SEARCH_ERROR)) {
412 _logger(Loger::INFO) << __func__ << ": "
413 << "Signal was found after LSU pulse switch in coarse search, but fine search reported an "
414 "error. LSU pulse and RF timing will not be restored as system should be close to the "
415 "OXC signal. Finish this OXC search."
416 << std::endl;
417 return false;
418 }
419
420 // signal not found or RF correction failed - restore original link timing,
421 if(!result) {
422 int newSwitchDirection = switchDirection * -1;
423 _logger(Loger::INFO) << __func__ << ": "
424 << "OXC signal not found after LSU switch. Restore original timing. Switch in direction: "
425 << newSwitchDirection << std::endl;
426
427 if(switchLsuPulseAndDoRfCorrection(switchDirection, newSwitchDirection, correctTimingWihtRFShifter)) {
428 _logger(Loger::INFO) << __func__ << ": "
429 << "Restore original LSU position and system timing done." << std::endl;
430 }
431 else {
432 _logger(Loger::INFO) << __func__ << ": "
433 << "Restore original LSU position and system timing failed." << std::endl;
434 }
435 }
436 }
437
438 if(_searchInProgress && correctTimingWihtRFShifter && !result &&
439 (offsetBeforeRfCorrection != _extData.rfTimingShifter->getTimingShiftOffset())) {
440 _logger(Loger::INFO) << __func__ << ": "
441 << "Restore RF system timing offset to " << offsetBeforeRfCorrection << " ps." << std::endl;
442 _extData.rfTimingShifter->moveOffsetTimingBlocking(offsetBeforeRfCorrection);
443 }
444 }
445 return result;
446}

448bool OxcSignalSearchLaser::checkLsuSwitchDirection(int& lsuSwitchDirection) {
449 // check in which direction link will switch
450 bool continueFlag = _extData.lsuController->executeLsuCommand(
451 LsuController::CMD::CMD_LSU_OXC_LOCK_TO_ANOTHER_PULSE_DETERMINE_DIRECTION);
452 if(!continueFlag) {
453 _logger(Loger::INFO) << __func__ << ": "
454 << "Sending LSU 'check switch direction' command failed." << std::endl;
455 return false;
456 }
457 std::this_thread::sleep_for(std::chrono::seconds(3));
458
459 int status = 0;
460 continueFlag = _extData.lsuController->getLinkRelockStatus(status);
461 if(!continueFlag) {
462 _logger(Loger::INFO) << __func__ << ": "
463 << "Check LSU 'check switch direction' status failed." << std::endl;
464 return false;
465 }
466
467 lsuSwitchDirection = 0;
468 switch(status) {
469 case CsaLsuOxcAdvAlg::SWITCH_LSU_PULSE_STATUS::ST_SWITCH_IN_NEGATIVE: {
470 _logger(Loger::INFO) << __func__ << ": "
471 << "LSU will switch in negative direction." << std::endl;
472 lsuSwitchDirection = -1;
473 break;
474 }
475 case CsaLsuOxcAdvAlg::SWITCH_LSU_PULSE_STATUS::ST_SWITCH_IN_POSITIVE: {
476 _logger(Loger::INFO) << __func__ << ": "
477 << "LSU will switch in positive direction." << std::endl;
478 lsuSwitchDirection = 1;
479 break;
480 }
481 default:
482 _logger(Loger::INFO) << __func__ << ": "
483 << "Unexpected LSU \"check switch direction\" status : " << status << std::endl;
484 return false;
485 }
486 return true;
487}

489bool OxcSignalSearchLaser::switchLsuPulseAndDoRfCorrection(
490 int& switchDirection, int wantedLsuSwitchDirection, bool doTimingCorrection) {
491 bool rfTimingCorrectionResult = true;
492 if(doTimingCorrection && wantedLsuSwitchDirection == 0) {
493 _logger() << __func__ << ": "
494 << "Settings error. doTimingCorrection set to true and wantedLsuSwitchDirection set to 0. Overwrite "
495 "doTimingCorrection to false."
496 << std::endl;
497 doTimingCorrection = false;
498 }
499 if(doTimingCorrection) _extData.loTableCorrection->lsuSwitchCorrectionCalculatePrepare();
500
501 _lsuSwitchInProgress = true;
502 _extData.rfTimingShifter->setIgnoreLaserLockStatus(true);
503 _searchAlgSystem->setStatusAndError(OXC_SR_ST_ERR::LSU_PULSE_SWITCH, true);
504 bool lsuSwitchResult = switchLsuPulse(switchDirection, wantedLsuSwitchDirection, doTimingCorrection);
505 _searchAlgSystem->setStatusAndError(OXC_SR_ST_ERR::LSU_PULSE_SWITCH, true);
506 _extData.rfTimingShifter->setIgnoreLaserLockStatus(false);
507 _lsuSwitchInProgress = false;
508
509 if(!lsuSwitchResult) _searchAlgSystem->setStatusAndError(OXC_SR_ST_ERR::LSU_SWITCH_ERROR, true);
510
511 if(doTimingCorrection) {
512 rfTimingCorrectionResult = checkTimingCorrection();
513 if(!rfTimingCorrectionResult)
514 _searchAlgSystem->setStatusAndError(OXC_SR_ST_ERR::SYSTEM_TIMING_CORRECTION_ERROR, true);
515 _extData.rfTimingShifter->setSpeed(_rfTimingSpeed);
516 _searchAlgSystem->setStatusAndError(OXC_SR_ST_ERR::SYSTEM_TIMING_CORRECTION, true);
517 }
518
519 if(_searchAlgSystem->getStatusAndError(OXC_SR_ST_ERR::STOPPED_BY_USER_REQUESTED)) this->searchStop();
520
521 if(doTimingCorrection) _extData.loTableCorrection->lsuSwitchCorrectionCalculate();
522
523 return (lsuSwitchResult && rfTimingCorrectionResult);
524}

526bool OxcSignalSearchLaser::startTimingCorrection(int wantedLsuSwitchDirection) {
527 float lsuMd22Speed = 0;
528 _rfTimingCurrentOffset = 0;
529 _rfTimingNewOffset = 0;
530
531 double maxSpeed = 0;
532 if(!_extData.lsuController->getMD22MaxSpeed(maxSpeed)) {
533 _logger(Loger::ERROR) << __func__ << ": "
534 << "Check LSU MD22 max speed failed. Wait 200ms and retry." << std::endl;
535 std::this_thread::sleep_for(std::chrono::milliseconds(200));
536 _extData.lsuController->getMD22MaxSpeed(maxSpeed);
537 }
538
539 double fastSpeedSettingPercentage = 0;
540 if(!_extData.lsuController->getMD22FastSettingSpeedInPercentage(fastSpeedSettingPercentage)) {
541 _logger(Loger::ERROR) << __func__ << ": "
542 << "Check LSU MD22 fast speed setting failed. Wait 200ms and retry." << std::endl;
543 std::this_thread::sleep_for(std::chrono::milliseconds(200));
544 _extData.lsuController->getMD22FastSettingSpeedInPercentage(fastSpeedSettingPercentage);
545 }
546
547 if(fastSpeedSettingPercentage == 0) {
548 fastSpeedSettingPercentage = 80.0;
549 _logger(Loger::ERROR) << __func__ << ": "
550 << "External settings of \'LSU MD22 fast setting speed in percetage\' not loaded correctly "
551 "or is set to 0 %. Using default: "
552 << fastSpeedSettingPercentage << " %." << std::endl;
553 }
554
555 if(maxSpeed == 0) {
556 maxSpeed = 26657.1;
557 _logger(Loger::ERROR) << __func__ << ": "
558 << "External settings of \'LSU MD22 maximum speed\' not loaded correctly or is set "
559 "to 0 fs/sec. Using default: "
560 << maxSpeed << " fs/sec." << std::endl;
561 }
562
563 _rfTimingCurrentOffset = _extData.rfTimingShifter->getTimingShiftOffset();
564 _rfTimingNewOffset = _rfTimingCurrentOffset + wantedLsuSwitchDirection * _rfTimingCorrection->getValue();
565 _logger(Loger::INFO) << __func__ << ": "
566 << "Correction size: " << _rfTimingCorrection->getValue()
567 << " , current timing offset: " << _rfTimingCurrentOffset
568 << " , new timing offset: " << _rfTimingNewOffset << std::endl;
569
570 // as MD22 always applies speed which is a litte bit smaller than wanted c.a. 0.02 % smaller -> do a correction of the
571 // number, but only when result will be positive
572 if(fastSpeedSettingPercentage > 0.03) fastSpeedSettingPercentage -= 0.02;
573 float fastSpeed = (1.0 * fastSpeedSettingPercentage / 100.0) * maxSpeed;
574 _logger(Loger::INFO)
575 << __func__ << ": "
576 << "LSU MD22 fast setting set to " << fastSpeedSettingPercentage
577 << " % (with applied -0.02 % correction). Fast speed will be used during LSU pulse switch. Fast speed is: "
578 << fastSpeed << " fs/sec." << std::endl;
579 lsuMd22Speed = fastSpeed;
580
581 // lsuMd22Speed is expressed in fs/sec . RF timing shifter speed is ps/sec
582 // one needs to multiply LSU MD22 by 2, because LSU end sees double speed, due to LSU construction
583 double newRfSpeed = (lsuMd22Speed * 2) / 1000;
584 _logger(Loger::INFO) << __func__ << ": "
585 << "LSU MD22 speed: " << lsuMd22Speed << " fs/sec . RF shifter speed " << newRfSpeed
586 << " ps/sec." << std::endl;
587
588 return startTimingShifter(_rfTimingNewOffset, newRfSpeed);
589}

591bool OxcSignalSearchLaser::startTimingShifter(float newOffsetTiming, double newRfSpeed) {
592 _rfTimingSpeed = _extData.rfTimingShifter->getSpeed();
593 _extData.rfTimingShifter->setSpeed(newRfSpeed);
594 if(!_extData.rfTimingShifter->moveOffsetTiming(newOffsetTiming)) {
595 _logger(Loger::INFO) << __func__ << ": "
596 << "Starting RF shifter failed. Restore original RF shifter speed: " << _rfTimingSpeed
597 << std::endl;
598 _extData.rfTimingShifter->setSpeed(_rfTimingSpeed);
599 return false;
600 }
601 return true;
602}

604bool OxcSignalSearchLaser::checkTimingCorrection() {
605 int timeoutCnt = 0;
606 int timeoutMaxSec = getLsuSwitchTimeout();
607 while(_extData.rfTimingShifter->shiftingInProgress()) {
608 if(!_searchInProgress) {
609 _extData.rfTimingShifter->stopMoving();
610 return false;
611 }
612
613 _logger(Loger::INFO) << __func__ << ": "
614 << "LSU pulse switched, but RF correction still in progress. Wait for "
615 << int(timeoutMaxSec / 5 - timeoutCnt) << " seconds." << std::endl;
616 timeoutCnt++;
617 std::this_thread::sleep_for(std::chrono::seconds(1));
618 if(timeoutCnt > int(timeoutMaxSec / 5)) {
619 _logger(Loger::INFO) << __func__ << ": "
620 << "RF timing correction timeout. Stop RF shifter." << std::endl;
621 _extData.rfTimingShifter->stopMoving();
622 return false;
623 }
624 }
625 if(std::fabs(float(_extData.rfTimingShifter->getTimingShiftOffset() - _rfTimingNewOffset)) > 1) {
626 _logger(Loger::INFO) << __func__ << ": "
627 << "RF timing correction failed. Target and current RF offset mismatch more than 1 ps."
628 << std::endl;
629 return false;
630 }
631
632 _logger(Loger::INFO) << __func__ << ": "
633 << "RF timing correction successful." << std::endl;
634 return true;

637bool OxcSignalSearchLaser::switchLsuPulse(int& switchDirection, int wantedLsuSwitchDirection, bool doTimingCorrection) {
638 switchDirection = 0;
639 int lsuLockStatus = 1;
640 bool continueFlag = _extData.lsuController->getLockStatus(lsuLockStatus);
641
642 if(!continueFlag) {
643 _logger(Loger::INFO) << __func__ << ": "
644 << "Check LSU lock status failed." << std::endl;
645 return false;
646 }
647 if(lsuLockStatus != 0) {
648 _logger(Loger::INFO) << __func__ << ": "
649 << "LSU not locked. Cannot continue LSU pulse switching." << std::endl;
650 return false;
651 }
652
653 double currentLockPos = 0;
654 continueFlag = _extData.lsuController->getLastLockPositon(currentLockPos);
655 if(!continueFlag) {
656 _logger(Loger::INFO) << __func__ << ": "
657 << "Check LSU last lock position failed." << std::endl;
658 return false;
659 }
660 _logger(Loger::INFO) << __func__ << ": "
661 << "LSU locked in ODL position: " << currentLockPos << " ." << std::endl;
662
663 // return false;
664
665 int relockStatusOnStart = 0;
666 continueFlag = _extData.lsuController->getLinkRelockStatus(relockStatusOnStart);
667 if(!continueFlag) {
668 _logger(Loger::INFO) << __func__ << ": "
669 << "Check LSU lock relock status failed." << std::endl;
670 return false;
671 }
672 std::this_thread::sleep_for(std::chrono::milliseconds(100));
673
674 //_logger(Loger::INFO) << __func__ << ": " << "Send LSU relock to another pulse command. Command ID: " <<
675 // LsuController::CMD::CMD_LSU_OXC_LOCK_TO_ANOTHER_PULSE << std::endl;
676 if(wantedLsuSwitchDirection == 0) {
677 continueFlag = _extData.lsuController->executeLsuCommand(LsuController::CMD::CMD_LSU_OXC_LOCK_TO_ANOTHER_PULSE);
678 _logger(Loger::INFO) << __func__ << ": "
679 << "Send LSU relock to another pulse command." << std::endl;
680 }
681 else if(wantedLsuSwitchDirection == -1) {
682 continueFlag = _extData.lsuController->executeLsuCommand(LsuController::CMD::CMD_LSU_OXC_LOCK_TO_PREVIOUS_PULSE);
683 _logger(Loger::INFO) << __func__ << ": "
684 << "Send LSU relock to previous pulse command." << std::endl;
685 }
686 else if(wantedLsuSwitchDirection == 1) {
687 continueFlag = _extData.lsuController->executeLsuCommand(LsuController::CMD::CMD_LSU_OXC_LOCK_TO_NEXT_PULSE);
688 _logger(Loger::INFO) << __func__ << ": "
689 << "Send LSU relock to next pulse command." << std::endl;
690 }
691 else {
692 _logger(Loger::INFO) << __func__ << ": "
693 << "Sending LSU relock command failed. Not supported direction : " << wantedLsuSwitchDirection
694 << std::endl;
695 return false;
696 }
697
698 if(!continueFlag) {
699 _logger(Loger::INFO) << __func__ << ": "
700 << "Sending LSU relock command failed." << std::endl;
701 return false;
702 }
703
704 do {
705 std::this_thread::sleep_for(std::chrono::milliseconds(100));
706
707 int relockStatus = 0;
708 int timeoutCnt = 0;
709 continueFlag = _extData.lsuController->getLinkRelockStatus(relockStatus);
710 if(!continueFlag) {
711 _logger(Loger::INFO) << __func__ << ": "
712 << "Check LSU lock relock status failed." << std::endl;
713 return false;
714 }
715 if(relockStatus != relockStatusOnStart) break;
716
717 if(timeoutCnt++ > 100) {
718 _logger(Loger::INFO) << __func__ << ": "
719 << "Send LSU command not accepted. LSU status didn't changed for 10 seconds." << std::endl;
720 return false;
721 }
722 } while(1);
723
724 // start timing correction - we know that LSU pulse switch command is accepted
725 // this should limit delay between starting a shift of RF and physical movement of the LSU-ODL
726 if(doTimingCorrection) {
727 _logger(Loger::INFO) << __func__ << ": "
728 << "RF timing correction requested. Start RF timing shifter." << std::endl;
729 _searchAlgSystem->setStatusAndError(OXC_SR_ST_ERR::SYSTEM_TIMING_CORRECTION, true);
730 if(!startTimingCorrection(wantedLsuSwitchDirection)) {
731 _searchAlgSystem->setStatusAndError(OXC_SR_ST_ERR::SYSTEM_TIMING_CORRECTION_ERROR, true);
732 }
733 }
734
735 int timeoutCnt = 0;
736 int timeoutMaxSec = getLsuSwitchTimeout();
737 bool timingShiftingInProgress = _extData.rfTimingShifter->shiftingInProgress();
738
739 do {
740 // toggle to make LSU status blink
741 _searchAlgSystem->setStatusAndError(
742 OXC_SR_ST_ERR::LSU_PULSE_SWITCH, !_searchAlgSystem->getStatusAndError(OXC_SR_ST_ERR::LSU_PULSE_SWITCH));
743 // toggle to make RF timing correction status blink
744 if(doTimingCorrection && _extData.rfTimingShifter->shiftingInProgress()) {
745 _searchAlgSystem->setStatusAndError(OXC_SR_ST_ERR::SYSTEM_TIMING_CORRECTION,
746 !_searchAlgSystem->getStatusAndError(OXC_SR_ST_ERR::SYSTEM_TIMING_CORRECTION));
747 }
748
749 std::this_thread::sleep_for(std::chrono::seconds(1));
750 timeoutCnt++;
751 int relockStatus = 0;
752 continueFlag = _extData.lsuController->getLinkRelockStatus(relockStatus);
753 if(!continueFlag) {
754 _logger(Loger::INFO) << __func__ << ": "
755 << "Check LSU lock relock status failed. Wait 200 ms and retry." << std::endl;
756 std::this_thread::sleep_for(std::chrono::milliseconds(200));
757 continueFlag = _extData.lsuController->getLinkRelockStatus(relockStatus);
758 if(!continueFlag) {
759 _logger(Loger::INFO) << __func__ << ": "
760 << "Check LSU lock relock status failed again. Retry." << std::endl;
761 return false;
762 }
763 }
764
765 switch(relockStatus) {
766 case CsaLsuOxcAdvAlg::SWITCH_LSU_PULSE_STATUS::ST_UNKNOWN: {
767 _logger(Loger::INFO) << __func__ << ": "
768 << "LSU pulse switch status error: SWITCH_LSU_PULSE_STATUS::ST_UNKNOWN" << std::endl;
769 return false;
770 }
771 case CsaLsuOxcAdvAlg::SWITCH_LSU_PULSE_STATUS::ST_WRONG_PARAMETERS: {
772 _logger(Loger::INFO) << __func__ << ": "
773 << "LSU pulse switch status error: SWITCH_LSU_PULSE_STATUS::ST_WRONG_PARAMETERS"
774 << std::endl;
775 return false;
776 }
777 case CsaLsuOxcAdvAlg::SWITCH_LSU_PULSE_STATUS::ST_IN_PROGRESS: {
778 if(timeoutCnt % 5 == 0) {
779 if(doTimingCorrection) {
780 if(_extData.rfTimingShifter->shiftingInProgress())
781 _logger(Loger::INFO) << __func__ << ": "
782 << "LSU pulse switch in progress. Wait. Switch time " << timeoutCnt
783 << " seconds. RF timing correction in progress." << std::endl;
784 else
785 _logger(Loger::INFO) << __func__ << ": "
786 << "LSU pulse switch in progress. Wait. Switch time " << timeoutCnt
787 << " seconds. RF timing correction done." << std::endl;
788 }
789 else
790 _logger(Loger::INFO) << __func__ << ": "
791 << "LSU pulse switch in progress. Wait. Switch time " << timeoutCnt << " seconds. "
792 << std::endl;
793 }
794 break;
795 }
796 case CsaLsuOxcAdvAlg::SWITCH_LSU_PULSE_STATUS::ST_SUCCESS_IN_POSITIVE: {
797 _logger(Loger::INFO) << __func__ << ": "
798 << "LSU pulse switch done in positive direction -> timing changed by -"
799 << _rfTimingCorrection->getValue() << " ps. " << std::endl;
800 switchDirection = 1;
801 break;
802 }
803 case CsaLsuOxcAdvAlg::SWITCH_LSU_PULSE_STATUS::ST_SUCCESS_IN_NEGATIVE: {
804 _logger(Loger::INFO) << __func__ << ": "
805 << "LSU pulse switch done in negative direction -> timing changed by +"
806 << _rfTimingCorrection->getValue() << " ps. " << std::endl;
807 switchDirection = -1;
808 break;
809 }
810 case CsaLsuOxcAdvAlg::SWITCH_LSU_PULSE_STATUS::ST_FAILED_AND_RELOCKED: {
811 _logger(Loger::INFO)
812 << __func__ << ": "
813 << "LSU pulse switch failed with status: ST_FAILED_AND_RELOCKED . Link lock to original point."
814 << std::endl;
815 return false;
816 }
817 case CsaLsuOxcAdvAlg::SWITCH_LSU_PULSE_STATUS::ST_FAILED: {
818 _logger(Loger::INFO) << __func__ << ": "
819 << "LSU pulse switch failed with status: ST_FAILED . Link lost lock." << std::endl;
820 return false;
821 }
822 case CsaLsuOxcAdvAlg::SWITCH_LSU_PULSE_STATUS::ST_STOPPED_BY_USER: {
823 _logger(Loger::INFO) << __func__ << ": "
824 << "LSU pulse switch failed with status: ST_STOPPED_BY_USER . " << std::endl;
825 return false;
826 }
827 }
828
829 if(doTimingCorrection && timingShiftingInProgress && !_extData.rfTimingShifter->shiftingInProgress()) {
830 //_searchAlgSystem->checkSystemState(); //just to update LASER UNLOCK error flag
831 timingShiftingInProgress = false;
832 _logger(Loger::INFO) << __func__ << ": "
833 << "RF timing shifter done. " << std::endl;
834 }
835 //_logger(Loger::INFO) << __func__ << ": " << "RF shift in progress flag: " <<
836 //_extData.rfTimingShifter->shiftingInProgress()
837 //<< " timingShiftingInProgress:" << timingShiftingInProgress << std::endl;
838 if(timeoutCnt > timeoutMaxSec) {
839 _logger(Loger::INFO) << __func__ << ": "
840 << "LSU pulse switch timeout (takes more than " << timeoutMaxSec << " seconds)" << std::endl;
841 return false;
842 }
843
844 if(!_searchInProgress) {
845 _logger(Loger::INFO) << __func__ << ": "
846 << "Search stopped by user. Stopping LSU pulse switch routine. " << std::endl;
847 _extData.lsuController->executeLsuCommand(LsuController::CMD::CMD_LSU_STOP_TUNE);
848 std::this_thread::sleep_for(std::chrono::milliseconds(100));
849 _extData.lsuController->executeLsuCommand(LsuController::CMD::CMD_LSU_OXC_SIGNAL_SEARCH_STOP);
850 return false;
851 }
852
853 } while(switchDirection == 0);
854
855 _logger(Loger::INFO) << __func__ << ": "
856 << "LSU pulse switch routine done. Wait 3 seconds to check LSU status." << std::endl;
857 // wait 3 second and check LSU status - if locked and if in correct position
858 std::this_thread::sleep_for(std::chrono::seconds(3));
859
860 double newLockPos = 0;
861 continueFlag = _extData.lsuController->getLastLockPositon(newLockPos);
862 if(!continueFlag) {
863 _logger(Loger::INFO) << __func__ << ": "
864 << "Check LSU lock position failed." << std::endl;
865 return false;
866 }
867
868 float delta = std::fabs(newLockPos - currentLockPos);
869 if(delta < 100) {
870 _logger(Loger::INFO) << __func__ << ": "
871 << "LSU locked in ODL position: " << newLockPos << ". Previously locked in: " << currentLockPos
872 << " . Delta: " << delta << " MD22 steps. Positions are too close. Something went wrong."
873 << std::endl;
874 return false;
875 }
876 else
877 _logger(Loger::INFO) << __func__ << ": "
878 << "LSU locked in ODL position: " << newLockPos << ". Previously locked in: " << currentLockPos
879 << " . Delta: " << delta << " MD22 steps." << std::endl;
880
881 lsuLockStatus = 1;
882 continueFlag = _extData.lsuController->getLockStatus(lsuLockStatus);
883 if(!continueFlag) {
884 _logger(Loger::INFO) << __func__ << ": "
885 << "Check LSU lock status failed." << std::endl;
886 return false;
887 }
888 if(lsuLockStatus != 0) {
889 _logger(Loger::INFO) << __func__ << ": "
890 << "LSU not locked after switch. Status is: " << lsuLockStatus << " . LSU switching failed."
891 << std::endl;
892 return false;
893 }
894 _logger(Loger::INFO) << __func__ << ": "
895 << "LSU locked after switch. Status is: " << lsuLockStatus << " . LSU switching successful."
896 << std::endl;
897 return true;
898}
