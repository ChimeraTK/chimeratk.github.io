133void CsaCoarseTuningMultiDetMultiActAlgorithm::settingsChangedCb() {
134 std::vector<double> unlockedStepSizes;
135 std::vector<double> lockedStepSizes;
136 for(uint32_t index = 0; index < _actuatorNumber; index++) {
137 unlockedStepSizes.push_back(_stepSizeUnlocked[index]->getValue());
138 lockedStepSizes.push_back(_stepSizeLocked[index]->getValue());
139 }
140 _tuneAlg.setActuatorsStepSizes(unlockedStepSizes, lockedStepSizes);
141
142 std::vector<double> startPostions;
143 for(uint32_t index = 0; index < _actuatorNumber - 1; index++)
144 startPostions.push_back(_startActuatorPositionToCoarseTune[index]->getValue());
145 _tuneAlg.setActuatorStartPositionForTune(startPostions);
146
147 _tuneAlg.setSafetyOpenLoopDisabled(_openLoopWhenSafetyDisabled->getValue());
148
149 _tuneAlg.setAutolockMaxAttempts(_autolockMaxAttempts->getValue());
150 (_autolockEnable->getValue()) ? _tuneAlg.setAutolockEnable(true) : _tuneAlg.setAutolockEnable(false);
151
152 (_allowDisablingAllActuatorOrDetector->getValue()) ? _tuneAlg.setAllowDisabledAllActuatorOrDetectors(true) :
153 _tuneAlg.setAllowDisabledAllActuatorOrDetectors(false);

200void CsaCoarseTuningMultiDetMultiActAlgorithm::triggerCommand(int cmd) {
201 bool tuneInProgress = _tuneAlg.getTuningInProgres();
202
203 uint32_t actId = 0;
204 if(cmd >= CMD::CMD_STEP_UP_MIN_ACT_ID && cmd <= CMD::CMD_STEP_UP_MAX_ACT_ID) {
205 actId = cmd - CMD::CMD_STEP_UP_MIN_ACT_ID;
206 cmd = CMD::CMD_STEP_UP;
207 }
208 else if(cmd >= CMD::CMD_STEP_DOWN_MIN_ACT_ID && cmd <= CMD::CMD_STEP_DOWN_MAX_ACT_ID) {
209 actId = cmd - CMD::CMD_STEP_DOWN_MIN_ACT_ID;
210 cmd = CMD::CMD_STEP_DOWN;
211 }
212 if(actId > _actuatorNumber) {
213 return;
214 }
215
216 bool lockWhenTuned = false;
217
218 switch(cmd) {
219 case CMD::CMD_START_TUNE_AND_LOCK:
220 if(!_tuneAlg.getEnable()) return;
221
222 _tuneAlg.setAutolockPause(false);
223 lockWhenTuned = true;
224 // add 'fall through' comment to suppress warning: this statement may fall through [-Wimplicit-fallthrough=] in
225 // GCC 7 (Ubuntu18) fall through
226 case CMD::CMD_START_TUNE: {
227 if(!_tuneAlg.getEnable()) return;
228
229 _extData.ctrlManager->checkAndTryEnableFastPiezoChannel();
230
231 //_tuneAlg.setAutolockPause(false);
232 if(tuneInProgress == false) {
233 _logger(Loger::INFO) << __func__ << ": "
234 << "Start tuning." << std::endl;
235 _tuneAlg.triggerTuning(lockWhenTuned);
236 }
237 else {
238 _logger(Loger::WARNING) << __func__ << ": "
239 << "Tuning or step in progress. Tune command ignored." << std::endl;
240 }
241 break;
242 }
243 case CMD::CMD_STOP_TUNE: {
244 //_tuneAlg.setAutolockPause(true);
245 if(_extData.ctrlManager->isCtrlSwitchingInProgress()) _extData.ctrlManager->stopSwitching();
246 _tuneAlg.stopTuning();
247
248 for(uint32_t i = 0; i < _actuatorNumber; i++)
249 if(_stepInProgress[i]) _stepInProgress[i] = false;
250 break;
251 }
252
253 case CMD::CMD_STEP_UP:
254 case CMD::CMD_STEP_DOWN: {
255 if(_stepInProgress[actId] == false && tuneInProgress == false) {
256 if(cmd == CMD::CMD_STEP_UP)
257 // step(1);
258 _stepRequest[actId] = 1;
259 else
260 // step(-1);
261 _stepRequest[actId] = -1;
262 }
263 else if(_stepInProgress[actId]) {
264 _logger(Loger::WARNING) << __func__ << ": "
265 << "Actuator " << actId << " step in progress. Step command ignored." << std::endl;
266 }
267 else {
268 _logger(Loger::WARNING) << __func__ << ": "
269 << "Tuning step in progress. Step command ignored." << std::endl;
270 }
271 break;
272 }
273 case CMD::CMD_DISABLE_FAST_LOOP: {
274 _tuneAlg.setAutolockPause(true);
275 _extData.ctrlManager->setEnable(false);
276 break;
277 }
278 case CMD::CMD_ENABLE_FAST_LOOP: {
279 _tuneAlg.setAutolockPause(false);
280 _extData.ctrlManager->setEnable(true);
281 break;
282 }
283 case CMD::CMD_TRIGGER_FAST_LOOP_SWITCHING: {
284 _extData.ctrlManager->triggerSwitching();
285 break;
286 }
287 case CMD::CMD_UNPAUSE_AUTOLOCK_FEATURE: {
288 _tuneAlg.setAutolockPause(false);
289 break;
290 }
291
292 case CMD::CMD_DISABLE_TEMPORARY_SAFETY_ON_60SEC: {
293 _tuneAlg.setSafetyOpenLoopTemporaryDisabled(true, 60);
294 break;
295 }
296 case CMD::CMD_DISABLE_TEMPORARY_SAFETY_ON_120SEC: {
297 _tuneAlg.setSafetyOpenLoopTemporaryDisabled(true, 120);
298 break;
299 }
300 case CMD::CMD_DISABLE_TEMPORARY_SAFETY_ON_300SEC: {
301 _tuneAlg.setSafetyOpenLoopTemporaryDisabled(true, 300);
302 break;
303 }
304 case CMD::CMD_DISABLE_TEMPORARY_SAFETY_ON_600SEC: {
305 _tuneAlg.setSafetyOpenLoopTemporaryDisabled(true, 600);
306 break;
307 }
308 case CMD::CMD_DISABLE_TEMPORARY_SAFETY_OFF: {
309 _tuneAlg.setSafetyOpenLoopTemporaryDisabled(false);
310 break;
311 }
312 case CMD::CMD_CLEAR_AUTOLOCK_CNT: {
313 // std::cout << "LOL Clear autolock cnt" << std::endl;
314 _tuneAlg.setAutolockPause(false);
315 _tuneAlg.clearAutolockCounter();
316 break;
317 }
318
319 default:
320 break;
321 }
322}

332bool CsaCoarseTuningMultiDetMultiActAlgorithm::setDetectors(
333 std::vector<boost::shared_ptr<CsaCoarseTuningDetector>> detectors, CsaCoarseTuningDetector::DETECTOR_TYPE type) {
334 if(type == CsaCoarseTuningDetector::DETECTOR_TYPE::NO_LOCKED) {
335 if(detectors.size() != 1) {
336 _logger(Loger::ERROR) << __func__ << ": "
337 << "Wrong detector number for Not-Locked case. Allowed value: 1, proviced: "
338 << detectors.size() << std::endl;
339 return false;
340 }
341
342 for(auto& item : detectors) item->initialize();
343 _tuneAlg.setDetectorsUnlocked(detectors);
344 }
345 else if(type == CsaCoarseTuningDetector::DETECTOR_TYPE::LOCKED) {
346 if(detectors.size() != _actuatorNumberInConstuctor) {
347 _logger(Loger::ERROR) << __func__ << ": "
348 << "Wrong detector number for Locked case. Allowed value: " << _actuatorNumberInConstuctor
349 << ", provided: " << detectors.size() << std::endl;
350 return false;
351 }
352
353 for(auto& item : detectors) item->initialize();
354 _tuneAlg.setDetectorsLocked(detectors);
355 }
356 _detectors.insert(_detectors.end(), detectors.begin(), detectors.end());
357
358 return true;
359}

374void CsaCoarseTuningMultiDetMultiActAlgorithm::update() {
375 terminateProgramIfCritical();
376
377 _status->setValue(static_cast<int32_t>(_tuneAlg.evaluateStatus()));
378 _statusInfo->setValue(_tuneAlg.getStatusInfo());
379
380 _autolockAttemptsCnt->setValue(_tuneAlg.getAutolockCounter());
381
382 auto status = _extData.laserLockStatusValidator->getExternalStatus();
383 bool referenceLocked = true;
384 if(status == LOCK_STATUS_VALUE::ERROR || status == LOCK_STATUS_VALUE::DISABLED ||
385 status == LOCK_STATUS_VALUE::ALTERNATING)
386 referenceLocked = false;
387 _tuneAlg.setRefenceLockedTuneAllowanceAndStatus(_tuneOnlyWhenRefOk->getValue(), referenceLocked);
388
389 // check which detector should be used
390 auto ctrlActive = _extData.ctrlManager->isCtrlActive();
391 if(ctrlActive) { // when enabled-locked - active detector type is 1
392 _activeDetectorId->setValue(1);
393 }
394 else { // when disabled - unlocked - active detector type is 0
395 _activeDetectorId->setValue(0);
396 }
397 _prevCtrlActive = ctrlActive;
398
399 for(uint32_t i = 0; i < _actuatorNumber; i++)
400 if(_stepInProgress[i]) {
401 // std::cout << "Steps is in progress" << std::endl;
402 if(_actuators[i]->motorIsBusy() == 0) {
403 _stepInProgress[i] = false;
404 }
405 }
406
407 bool tuneInProgress = false;
408 _tuneAlg.checkTuning();
409 tuneInProgress = _tuneAlg.getTuningInProgres();
410
411 for(uint32_t i = 0; i < _actuatorNumber; i++) {
412 if(!tuneInProgress && !_stepInProgress[i]) {
413 if(_stepRequest[i]) {
414 step(_stepRequest[i], i);
415 _stepRequest[i] = 0;
416 }
417 _algorithmActive->setValue(0);
418 }
419 else {
420 _stepRequest[i] = 0;
421 _algorithmActive->setValue(1);
422 }
423 }
424
425 // set current actuator positions
426 for(uint32_t i = 0; i < _actuatorNumber; i++) {
427 _actPositions[i]->setValue(_actuators[i]->getCurrentPosition());
428 _actPositionsInSteps[i]->setValue(_actuators[i]->getCurrentPositionInSteps());
429 }
430
431 if(_updateDividerCnt == 5) {
432 setDetectorActuatorsNameAndIds();
433 _updateDividerCnt = 0;
434 for(uint32_t index = 0; index < _actuatorNumber; index++)
435 if(_actuators[index]->getStatus() == SMAStatusTypes::M_ERROR &&
436 _actuators[index]->getError() == SMAErrorTypes::M_COMMUNICATION_LOST)
437 _actuators[index]->resetMotorFlags();
438 }
439 _updateDividerCnt++;
440
441 _csaOppStatus->set(_tuneAlg.getOppStatus());
442
443 _autolockPaused->setValue(_tuneAlg.getAutolockPause());
444}

446void CsaCoarseTuningMultiDetMultiActAlgorithm::step(int32_t direction, uint32_t actuatorId) {
447 if(actuatorId >= _actuators.size()) {
448 _logger() << __func__ << ": "
449 << "Id out of range: " << direction << std::endl;
450 return;
451 }
452
453 _logger(Loger::INFO) << __func__ << ": "
454 << "Start step in direction: " << direction << std::endl;
455 _actuators[actuatorId]->getStatus();
456 if(_actuators[actuatorId]->motorIsBusy() == 0) {
457 _stepInProgress[actuatorId] = true;
458 auto currentPos = _actuators[actuatorId]->getCurrentPosition();
459
460 double step = 0;
461 if(_activeDetectorId->getValue())
462 step = _stepSizeLocked[actuatorId]->getValue();
463 else
464 step = _stepSizeUnlocked[actuatorId]->getValue();
465
466 if(step == 0) {
467 _logger(Loger::WARNING) << __func__ << ": "
468 << "Step size is 0. Check you settings." << std::endl;
469 }
470
471 if(direction > 0)
472 _actuators[actuatorId]->setTargetPosition(currentPos + step);
473 else
474 _actuators[actuatorId]->setTargetPosition(currentPos - step);
475
476 if(!_actuators[actuatorId]->getAutostart()) _actuators[actuatorId]->start();
477 _algorithmActive->setValue(1);
478 }
479 else {
480 _logger(Loger::WARNING) << __func__ << ": "
481 << "Motor " << actuatorId << "is busy. Step ignored. " << direction << std::endl;
482 }
483}

534void CsaCoarseTuningMultiDetMultiActAlgorithm::coarseTuningEnabledCb() {
535 try {
536 if(_coarseTuningEnabled->getValue()) {
537 if(_tuneAlg.getEnable()) {
538 _logger(Loger::DETAIL) << __func__ << ": "
539 << "Coarse tuning already enabled." << std::endl;
540 return;
541 }
542
543 _extData.ctrlManager->checkAndTryEnableFastPiezoChannel();
544
545 bool ctrlEnabled = _extData.ctrlManager->isCtrlActive();
546 this->storeActuatorSetting();
547 for(uint32_t index = 0; index < _actuatorNumber; index++) {
548 _actuators[index]->setCsControlAllowed(false);
549 if(!_actuators[index]->isEnabled()) {
550 _logger(Loger::INFO) << __func__ << ": "
