217 CsaLsuOxcTuneSearchAlg::OXC_SIGNAL_STATUS CsaLsuOxcTuneSearchAlg::checkOxcSignal() {
218  float floorLevel = _floorLevel->getValue();
219  auto maxOXC = _oxcDetector->getRawMax(true);
220  auto minOXC = _oxcDetector->getRawMin(true);
221  float minSignalLevel = _oxcMinimalSignalLevel->getValue();
222 
223  _maxOXCValue = maxOXC;
224  _minOXCValue = minOXC;
225 
226  if(maxOXC > (floorLevel + minSignalLevel) && minOXC < (floorLevel - minSignalLevel)) {
227  _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": OXC signal detected. Positive amp.: " << maxOXC
228  << " Negative amp.: " << minOXC
229  << " . Floor level not included. Motor position: " << _actuator->getCurrentPositionInSteps()
230  << std::endl;
231  return OXC_SIGNAL_STATUS::DOWN_UP_CROSSED;
232  }
233  else if(maxOXC > (floorLevel + minSignalLevel)) {
234  _logger(Loger::INFO) << __func__ << ": " << _logIdString
235  << ": OXC positive slope detected. Amplitude (floor not included): " << maxOXC
236  << ", Motor position: " << _actuator->getCurrentPositionInSteps() << std::endl;
237  return OXC_SIGNAL_STATUS::UP_CROSSED;
238  }
239  else if(minOXC < (floorLevel - minSignalLevel)) {
240  _logger(Loger::INFO) << __func__ << ": " << _logIdString
241  << ": OXC negative slope detected. Amplitude (floor not included): " << minOXC
242  << ", Motor position: " << _actuator->getCurrentPositionInSteps() << std::endl;
243  return OXC_SIGNAL_STATUS::DOWN_CROSSED;
244  }
245  else
246  return OXC_SIGNAL_STATUS::NO_SIGNAL;
247 }

263  }
264

288  _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Search routine took: " << duration << " seconds."
289  << std::endl;

292 void CsaLsuOxcTuneSearchAlg::searchOxcSignal() {
293  try {
294  _oxcSignalSearchThreadInProgress = true;
295 
296  // check whether controller need ramping and ramp it if needed
297  if(_controller->isSafetyHold()) {
298  _logger(Loger::INFO)
299  << __func__ << ": " << _logIdString
300  << ": Safety hold enabled. Need to ramp controller and reset safety flag before OXC search. Start ramping."
301  << std::endl;
302  _controller->rampAndResetSafetyHold();
303  _logger(Loger::DETAIL) << __func__ << ": " << _logIdString << ": Wait for ramping thread to start." << std::endl;
304  while(_controller->isRampingInProgress() == false) {
305  _logger(Loger::DETAIL) << __func__ << ": " << _logIdString << ": Wait for ramping thread to start."
306  << std::endl;
307  }
308  while(_controller->isRampingInProgress()) {
309  _logger(Loger::INFO) << __func__ << ": " << _logIdString
310  << ": Controller ramping in progress. Search algorithm paused." << std::endl;
311  std::this_thread::sleep_for(std::chrono::milliseconds(15));
312  }
313  }
314 
315  // try to search signal only by scanning with piezo near range
316  // if successful we can end
317  if(searchForOxcCtrlOutScan()) return;
318 
319  // prepare actuator for fast movements
320  this->_actuatorHelper.enableActuatorWithNoAutostart();
321 
322  // move to last lock position if "use current position" option is disabled
323  if(_lastLockPositionTime->getValue() != 0 && _startPositionAsCurrent->getValue() == 0) {
324  _stream.str("");
325  _stream << "Search start. Move to last lock position: " << _lastLockPosition->getValue();
326  _messageString.set(_stream.str());
327  if(_actuator->moveToPositionInSteps(_lastLockPosition->getValue()) != SMAStatusTypes::M_OK) {
328  _logger(Loger::ERROR) << __func__ << ": " << _logIdString << ": Cannot search signal. Actuator error."
329  << std::endl;
330  _messageString.set("Actuator error. Cannot search OXC signal.");
331  return;
332  }
333  // try to scan close to last lock position
334  if(searchForOxcCtrlOutScan()) return;
335  }
336  // check if during moving to start position search routine was not stopped by user
337  if(_oxcSignalSearchThreadInProgress == false) {
338  _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Search routine stopped by user. Return."
339  << std::endl;
340  _messageString.set("OXC search stopped by user.");
341  return;
342  }
343 
344  // OXC floor level search (if 'use manual floor level value' is off)
345  float floorLevel = 0;
346  //_searchStatus->setValue(OXC_SEARCH_STATUS_FLOOR_LEVEL_IN_PROGRESS);
347  if(_manualFloorLevelEnabled->getValue()) {
348  floorLevel = _manualFloorLevel->getValue();
349  }
350  else if(!oxcSignalFloorDetection(floorLevel)) {
351  if(_oxcSignalSearchThreadInProgress) {
352  _logger(Loger::ERROR) << __func__ << ": " << _logIdString
353  << ": Floor detection failed. OXC signal failed. Return." << std::endl;
354  _messageString.set("Floor level detection failed.");
355  }
356  return;
357  }
358  _floorLevel->setValue(floorLevel);
359  _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Detected OXC signal floor level is: " << floorLevel
360  << std::endl;
361  // OXC floor level search end
362 
363  // we will do first scan with 1/4 of max range. Usually signal should not drift more
364  auto csStatus =
365  coarseSearchForOxcSignal(std::fabs(_maxOxcSignalDistance->getValue() / 4), _startPositionAsCurrent->getValue());
366  if(csStatus == CS_RESULT::STOP_BY_USER || csStatus == CS_RESULT::OXC_FOUND || csStatus == CS_RESULT::ERROR) return;
367 
368  // OXC coarse scan start here for full range between OXC pulses
369  csStatus = coarseSearchForOxcSignal(std::fabs(_maxOxcSignalDistance->getValue()), true);
370  if(csStatus == CS_RESULT::STOP_BY_USER || csStatus == CS_RESULT::OXC_FOUND || csStatus == CS_RESULT::ERROR) {
371  return;
372  }
373  else if(csStatus == CS_RESULT::CONTINUE) {
374  this->_actuatorHelper.changeActuatorSpeedSettings(ACT_SPEED_SETTING::SLOW); // Change speed settings to SLOW
375  // OXC fine scan start
376  bool fineOxcFind = false;
377  fineOxcFind = fineSearchForOxcSignal(_lastMoveDir, _oxcSignalFineSearchStep->getValue() * 5000);
378  if(fineOxcFind && !_oxcSignalSearchThreadInProgress) {
379  _messageString.set("Fine search stopped by user.");
380  return;
381  }
382  else if(!fineOxcFind) {
383  _messageString.set("Signal not found in fine search.");
384  return;
385  }
386  // OXC fine scan end
387  // store last lock position - we do it here not in fineSearchForOxcSignal routine, because fineSearchForOxcSignal is recursive
388  _lastLockPosition->setValue(_actuator->getCurrentPositionInSteps());
389  _lastLockPositionTime->setValue(((time_t)time(0)));
390  }
391  }
392  catch(CsaExternalProcessException& ex) {
393  _logger(Loger::INFO) << __func__ << ": " << _logIdString
394  << ": OXC search failed to communication exception: " << ex.what() << std::endl;
395  }
396 }

406 CsaLsuOxcTuneSearchAlg::CS_RESULT CsaLsuOxcTuneSearchAlg::coarseSearchForOxcSignalInternal(
407  float maxOxcSignalDistance, bool useCurrentPostion) {
408  _maxOXCValue = 0;
409  _minOXCValue = 0;
410  _fineTuneStopPosition = 0.0;
411  _lastMoveDir = 1;
412 
413  std::string messagePrefix("Limited range scan. ");
414  if(maxOxcSignalDistance == std::fabs(_maxOxcSignalDistance->getValue())) messagePrefix = "Full range scan. ";
415 
416  float prevPosition, currentPosition = 0.0;
417  int motorErrorCnt = 0;
418 
419  auto startPosition = _lastLockPosition->getValue();
420  if(_lastLockPositionTime->getValue() == 0 || useCurrentPostion)
421  startPosition = _actuator->getCurrentPositionInSteps();
422 
423  double correction = 0.05;
424  int newPosition = startPosition - std::fabs(_maxOxcSignalDistance->getValue() * correction);
425 
426  _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Start pos. for coarse search: " << startPosition
427  << " , new pos with " << (correction * 100) << "% move back: " << newPosition
428  << " , last lock pos: " << _lastLockPosition->getValue()
429  << " , current actuator pos: " << _actuator->getCurrentPositionInSteps() << std::endl;
430 
431  bool positiveEndSwitchHit = false;
432  int32_t positiveEndSwitchPosition = std::numeric_limits<int32_t>::min();
433  bool negativeEndSwitchHit = false;
434  int32_t negativeEndSwitchPosition = std::numeric_limits<int32_t>::max();
435  float difference = 0;
436  auto step = _oxcSignalFineSearchStep->getValue() * 5000;
437  bool endTheMove = false;
438 
439  // move motor 3% of maximum pulse range back, because we will scan first in positive direction
440  _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Prepare to scan. Try to send motor to: " << newPosition
441  << std::endl;
442  if(moveToPositionWithEndSwitchCorrection(newPosition) != SMAStatusTypes::M_OK) {
443  _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Prepare to scan motor moving failed. " << std::endl;
444  return CS_RESULT::ERROR;
445  }
446  _logger(Loger::DETAIL) << __func__ << ": " << _logIdString << ": Motor in position: " << newPosition << std::endl;
447 
448  // reset OXC detector (input stage)
449  this->resetOxcDetector();
450 
451  while(_oxcSignalSearchThreadInProgress) {
452  currentPosition = newPosition;
453  newPosition = newPosition + (_lastMoveDir * step);
454  _logger(Loger::DETAIL) << __func__ << ": " << _logIdString << ": Send motor to: " << newPosition << std::endl;
455  _actuator->startInSteps(newPosition);
456  std::this_thread::sleep_for(std::chrono::milliseconds(150));
457  endTheMove = false;
458  uint32_t postionNotChangingCounter = 0;
459  uint32_t motorStateCheckCnt = 0;
460  do {
461  if(_oxcSignalSearchThreadInProgress == false) {
462  _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Search routine stopped by user. Return."
463  << std::endl;
464  _messageString.set("Coarse search stopped by user.");
465  return CS_RESULT::STOP_BY_USER;
466  }
467 
468  std::this_thread::sleep_for(std::chrono::milliseconds(10));
469  motorStateCheckCnt++;
470 
471  _stream.str("");
472  _stream << messagePrefix << "Search. Position: " << currentPosition;
473  _messageString.set(_stream.str());
474  auto checkSignal = this->checkOxcSignal();
475  if(checkSignal == OXC_SIGNAL_STATUS::DOWN_UP_CROSSED) {
476  // signal found - we crossed both positive and negative slopes - stop as fast as possible
477  _actuator->stop();
478  std::this_thread::sleep_for(std::chrono::milliseconds(300));
479  // just to update _maxOXCValue and _minOXCValue
480  this->checkOxcSignal();
481  // store max and min OXC signal values in PV
482  _oxcSignalMax->setValue(_maxOXCValue);
483  _oxcSignalMin->setValue(_minOXCValue);
484 
485  currentPosition = _actuator->getCurrentPositionInSteps();
486  difference = std::fabs(currentPosition - prevPosition);
487 
488  _fineTuneStopPosition = prevPosition - 5 * _lastMoveDir * difference;
489 
490  _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": OXC signal detected. Min value: " << _minOXCValue
491  << " Max value: " << _maxOXCValue << ", Curr. pos: " << currentPosition
492  << ", prev. pos: " << prevPosition << ", diff: " << difference << std::endl;
493  _logger(Loger::INFO) << __func__ << ": " << _logIdString
494  << ": OXC signal detected. Curr. dir.: " << _lastMoveDir << " Step change to: " << step
495  << ". Fine scan position range from: " << (currentPosition + _lastMoveDir * 750)
496  << " to: " << _fineTuneStopPosition << std::endl;
497 
498  std::this_thread::sleep_for(std::chrono::milliseconds(150));
499

805 bool CsaLsuOxcTuneSearchAlg::searchForOxcCtrlOutScanCore() {
806  float floorLevel = _floorLevel->getValue();
807  float minSignalLevel = _oxcMinimalSignalLevel->getValue();
808  //
809  enum OxcSignalSearchState { UNKNOWN = 0, SLOPE_FOUND, SIGNAL_FOUND, SIGNAL_NOT_FOUND };
810  OxcSignalSearchState searchState = OxcSignalSearchState::UNKNOWN;
811 
812  const double range = std::fabs(_controller->getMaxValue() * 0.8);
813  double step = range / 50.0;
814  double dir = 1;
815  double slopeSign = 0;
816  int loopsDelayMs = 3;
817  auto ffValue = _controller->getFeedforwardValue();
818 
819  _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Start. Range: " << range << ", step: " << step
820  << std::endl;
821  do {
822  auto oxcSignalVal = _oxcDetector->getRaw(true);
823  if(searchState == OxcSignalSearchState::UNKNOWN) {
824  if(oxcSignalVal < (floorLevel - minSignalLevel) || oxcSignalVal > (floorLevel + minSignalLevel)) {
825  step = 0.01;
826  if(oxcSignalVal < (floorLevel - minSignalLevel))
827  _logger(Loger::INFO) << __func__ << ": " << _logIdString
828  << ": OXC Signal negative slope detected. Continue in this direction. FF: " << ffValue
829  << ", OXC value: " << oxcSignalVal << std::endl;
830  else
831  _logger(Loger::INFO) << __func__ << ": " << _logIdString
832  << ": OXC Signal positive slope detected. Continue in this direction. FF: " << ffValue
833  << ", OXC value: " << oxcSignalVal << std::endl;
834 
835  searchState = OxcSignalSearchState::SLOPE_FOUND;
836  slopeSign = oxcSignalVal;
837  }
838  else {
839  _logger(Loger::DETAIL) << __func__ << ": " << _logIdString << ": OXC Signal NOT detected. FF: " << ffValue
840  << ", OXC value: " << oxcSignalVal << std::endl;
841  }
842  if(ffValue > range)
843  dir = -1;
844  else if(dir == -1 && ffValue < -range)
845  searchState = OxcSignalSearchState::SIGNAL_NOT_FOUND;
846  }
847  else if(searchState == OxcSignalSearchState::SLOPE_FOUND) {
848  if(slopeSign * oxcSignalVal < 0) {
849  searchState = OxcSignalSearchState::SIGNAL_FOUND;
850  _logger(Loger::INFO) << __func__ << ": " << _logIdString
851  << ": OXC Signal found. Zero crossing detected. FF value: " << ffValue
852  << ", OXC value: " << oxcSignalVal << std::endl;
853  }
854  else if(fabs(ffValue) > range) {
855  _logger(Loger::INFO) << __func__ << ": " << _logIdString
856  << ": OXC Signal slope detected, but not enough range to go into signal. FF value:"
857  << ffValue << std::endl;
858  searchState = OxcSignalSearchState::SIGNAL_NOT_FOUND;
859  }
860  else
861  _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": OXC Signal slope detected. FF value: " << ffValue
862  << ", OXC value: " << oxcSignalVal << std::endl;
863  }
864 
865  if(searchState == OxcSignalSearchState::UNKNOWN || searchState == OxcSignalSearchState::SLOPE_FOUND) {
866  ffValue = ffValue + dir * step;
867  _controller->setFeedforwardValue(ffValue);
868  std::this_thread::sleep_for(std::chrono::milliseconds(loopsDelayMs));
869  }
870  } while(searchState == OxcSignalSearchState::UNKNOWN || searchState == OxcSignalSearchState::SLOPE_FOUND);
871  _logger(Loger::DETAIL) << __func__ << ": " << _logIdString << ": OXC Signal search state: " << searchState
872  << std::endl;
873 
874  if(searchState != OxcSignalSearchState::SIGNAL_FOUND) return false;
875  return true;
876 }
