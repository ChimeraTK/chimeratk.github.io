125bool MultiActuatorCoarseTuningAlgorithm::isAutoRelockExecutable() {
126 if(!isAutoRelockEnabled()) return false;
127
128 if(_autolockPaused || (_doNotAllowForTuningWhenReferenceUnlocked && !_referenceLocked)) return false;
129
130 // trick for a panel logic -> here condition could be >= , but then on last attempt of lock when _autolockCnt ==
131 // _autolockMax, the 'autolocking failed' info would be displayed on the panel.
132 if(_autolockCnt > _autolockMax) return false;
133
134 if(_tuningInProgress || _controllerManager->isCtrlSwitchingInProgress()) return false;
135
136 // trick for a panel logic -> 'autolocking failed' info is displayed only when _autolockCnt > _autolockMax
137 // this is checked when _tuningInProgress is false, so last attempt of relock has finished.
138 if(_autolockCnt == _autolockMax) {
139 _autolockCnt++;
140 return false;
141 }
142
143 if(_status == CoarseTuningAlgorithmBase::STATUS::ERROR) return false;
144 return true;

174void MultiActuatorCoarseTuningAlgorithm::checkTuning() {
175 if(_detectorsUnlocked.size() == 0 || _detectorsLocked.size() == 0) {
176 _logger() << __func__ << ": " << _logIdString << ": _detectors not set. exit(1)." << std::endl;
177 exit(1);
178 return;
179 }
180 if(_actuators.size() == 0) {
181 _logger() << __func__ << ": " << _logIdString << ": _actuators not set. exit(1)." << std::endl;
182 exit(1);
183 return;
184 }
185
186 // some debug to be removed later - TODO, TO_DO, FIXME, FIX_ME
187 /*
188 if (_enabledPrev != _enabled)
189 _logger(Loger::DETAIL) << __func__ << ": " << _logIdString << ": _enabled flag changed to: " << _enabled <<
190 std::endl; if (_enabled) _enabledPrev = true; else _enabledPrev = false;
191
192 if (_safetyLimitsCrossedFlagPrev != _safetyLimitsCrossedFlag)
193 _logger(Loger::DETAIL) << __func__ << ": " << _logIdString << ": _safetyLimitsCrossedFlag flag changed to: " <<
194 _safetyLimitsCrossedFlag << std::endl; if (_safetyLimitsCrossedFlag) _safetyLimitsCrossedFlagPrev = true; else
195 _safetyLimitsCrossedFlagPrev = false;
196
197 if (_tuningInProgressPrev != _tuningInProgress)
198 _logger(Loger::DETAIL) << __func__ << ": " << _logIdString << ": _tuningInProgress flag changed to: " <<
199 _tuningInProgress << std::endl; if (_tuningInProgress) _tuningInProgressPrev = true; else _tuningInProgressPrev =
200 false;
201 //
202 */
203 auto period = std::chrono::duration_cast<std::chrono::seconds>(
204 std::chrono::high_resolution_clock::now() - _safetyTempDisTimeStart)
205 .count();
206 if(_safetyOpenLoopTemporaryDisabled && period >= _safetyTempDisTimePeriodSec) {
207 _logger(Loger::WARNING) << __func__ << ": " << _logIdString
208 << ": Temporal 'safety OFF' done. Safety is enabled again." << std::endl;
209 _safetyOpenLoopTemporaryDisabled = false;
210 }
211
212 // update all detectors
213 for(auto& item : _detectorsLocked) item->getCurrentValue();
214 for(auto& item : _detectorsUnlocked) item->getCurrentValue();
215
216 // coarse tuning module is disabled - stop all actuators if tuning in progress, stop tuning
217 if(!_enabled) {
218 if(_tuningInProgress) {
219 _tuningInProgress = false;
220 for(auto& item : _actuators) item->stop();
221 }
222 _logger(Loger::MAX_DETAIL) << __func__ << ": " << _logIdString << ": Coarse tuning disabled." << std::endl;
223 _safetyLimitsCrossedFlag = false;
224 return;
225 }
226
227 bool controllerActive = _controllerManager->isCtrlActive();
228
229 // this part is validating safety limits
230 bool resetSafetyPossible = true;
231 if(controllerActive == false) {
232 if(isDetSafetyOn(_detectorsUnlocked[0], false)) {
233 //_logger(Loger::ERROR) << __func__ << ": " << _logIdString << ": Safety limits crossed for beat frequency
234 // detector." << std::endl;
235 if(_tuningInProgress) {
236 _logger(Loger::ERROR) << __func__ << ": " << _logIdString
237 << ": Stop current tuning process when controller not active due to safety reasons."
238 << std::endl;
239 _tuningInProgress = false;
240 }
241 resetSafetyPossible = false;
242 _safetyAutoresetCnt = 0;
243 _safetyLimitsCrossedFlag = true;
244 }
245 }
246 else {
247 for(auto& item : _detectorsLocked) {
248 if(isDetSafetyOn(item, false)) {
249 if(_safetyOpenLoopDisabled == false && _safetyOpenLoopTemporaryDisabled == false) {
250 if(_controllerManager->isCtrlSwitchingInProgress() == false) {
251 if(_tuningInProgress) {
252 _logger(Loger::ERROR)
253 << __func__ << ": " << _logIdString
254 << ": Stop current tuning process when controller active due to safety reasons. Detector "
255 << item->getActiveDetectorName() << " reported safety limits crossed." << std::endl;
256 _tuningInProgress = false;
257 std::this_thread::sleep_for(std::chrono::milliseconds(2));
258 }
259
260 _logger(Loger::INFO) << __func__ << ": " << _logIdString
261 << ": Open loop due to safety. Detector: " << item->getActiveDetectorName()
262 << " reported safety threshold crossed." << std::endl;
263 _controllerManager->setEnable(false);
264 }
265 else {
266 _logger(Loger::WARNING)
267 << __func__ << ": " << _logIdString << ": Detector: " << item->getActiveDetectorName()
268 << " reported safety threshold crossed, but ctrl input switching is in progress. Loop will not be open."
269 << std::endl;
270 }
271 }
272 else
273 _logger(Loger::WARNING) << __func__ << ": " << _logIdString << ": Detector: " << item->getActiveDetectorName()
274 << " reported safety threshold crossed, but safety is disabled or temporary "
275 "disabled. Loop will not be opened. ."
276 << std::endl;
277
278 resetSafetyPossible = false;
279 _safetyAutoresetCnt = 0;
280 _safetyLimitsCrossedFlag = true;
281
282 break;
283 }
284 }
285 }
286
287 if(_tuningInProgress && _doNotAllowForTuningWhenReferenceUnlocked && !_referenceLocked) {
288 _logger(Loger::ERROR) << __func__ << ": " << _logIdString
289 << ": Stop current tuning process, because System Reference is NOT locked." << std::endl;
290 _tuningInProgress = false;
291 }
292
293 // check if we can start new tuning after detector malfunction or after safety was crossed
294 if(_safetyLimitsCrossedFlag) {
295 if(resetSafetyPossible) {
296 _logger(Loger::DETAIL) << __func__ << ": " << _logIdString
297 << ": Safety flag auto-reset in: " << (5 - _safetyAutoresetCnt) << std::endl;
298 if(_safetyAutoresetCnt > 5) {
299 _safetyLimitsCrossedFlag = false;
300 }
301 _safetyAutoresetCnt++;
302 }
303 return;
304 }
305
306 if(!_pauseFlag && !_tuningInProgress) {
307 if(!controllerActive) { // unlocked
308 checkAndStartTuningForUnlocked();
309 checkAutolock();
310 }
311 else {
312 for(uint32_t i = 0; i < _detectorsLocked.size(); i++)
313 if(checkAndStartTuningForLocked(i)) break;
314
315 checkAndResetAutolockCounter();
316 }
317 }
318 else if(_pauseFlag) {
319 _logger(Loger::MAX_DETAIL) << __func__ << ": " << _logIdString << ": Tuning is paused. " << std::endl;
320 }
321 else if(_tuningInProgress) {
322 _logger(Loger::MAX_DETAIL) << __func__ << ": " << _logIdString << ": Tuning is already in progress. " << std::endl;
323 }
324}

359void MultiActuatorCoarseTuningAlgorithm::tuneLockedCore(uint32_t idOfDetector) {
360 if((_actuactor4LockedDisableMask->getValue() & (1 << idOfDetector))) {
361 if(idOfDetector > 0) {
362 --idOfDetector;
363 _logger(Loger::WARNING) << __func__ << ": " << _logIdString << ": Actuator for detector ID: " << idOfDetector + 1
364 << " is disabled. Tune with detector ID:" << idOfDetector << std::endl;
365 tuneLockedCore(idOfDetector);
366 }
367 else if(!_allowDisabledAllActuatorOrDetectors)
368 _logger(Loger::WARNING) << __func__ << ": " << _logIdString << ": Actuator for detector ID: " << idOfDetector
369 << " is disabled. Cannot continue." << std::endl;
370 return;
371 }
372
373 TUNE_FLAG tuningNotDone = TUNE_FLAG::CONTINUE;
374 bool actuatorError = false;
375 do {
376 if(_tuningInProgress == false) {
377 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Tuning (locked) stopped." << std::endl;
378 _oppStatus.set("Tuning stopped.", OppStatus::TYPE::INFO);
379 return;
380 }
381
382 // check stop condition
383 auto currentStopCondition = _detectorsLocked[idOfDetector]->checkStopCondition();
384 // check safety first
385 if(isDetSafetyOn(_detectorsLocked[idOfDetector])) { //_detectorsLocked[idOfDetector]->checkSafetyCondition() !=
386 // CsaCoarseTuningDetector::LIMIT::IN_RANGE) {
387 _logger(Loger::ERROR) << __func__ << ": " << _logIdString
388 << ": Stopping tuning (locked) process due to safety reasons. Detector ID: " << idOfDetector
389 << " safety triggered." << std::endl;
390 _oppStatus.set("Stop tuning for safety. Check log file.", OppStatus::TYPE::ERROR);
391 tuningNotDone = TUNE_FLAG::ERROR;
392 _safetyAutoresetCnt = 0;
393 _safetyLimitsCrossedFlag = true;
394 }
395 else if(currentStopCondition == CsaCoarseTuningDetector::LIMIT::IN_RANGE) {
396 tuningNotDone = TUNE_FLAG::SUCCESSFUL;
397 _logger(Loger::INFO) << __func__ << ": " << _logIdString
398 << ": Tune (locked) done. Signal in range for detector ID: " << idOfDetector << std::endl;
399 _oppStatus.set("Tuning successful.", OppStatus::TYPE::OK);
400 }
401 else {
402 auto currentDetVal = _detectorsLocked[idOfDetector]->getCurrentValue();
403 auto currentPos = _actuators[idOfDetector]->getCurrentPosition();
404 auto newPosition = currentPos;
405 if(currentStopCondition == CsaCoarseTuningDetector::LIMIT::MAX_CROSSED) {
406 newPosition = currentPos - _stepScaling * _actuatorsStepsSize4Locked[idOfDetector];
407 _logger(Loger::DETAIL) << __func__ << ": " << _logIdString
408 << ": Detector 'stop max limit' still crossed. Doing step down. Current detector value: "
409 << currentDetVal << std::endl;
410 }
411 else if(currentStopCondition == CsaCoarseTuningDetector::LIMIT::MIN_CROSSED) {
412 newPosition = currentPos + _stepScaling * _actuatorsStepsSize4Locked[idOfDetector];
413 _logger(Loger::DETAIL) << __func__ << ": " << _logIdString
414 << ": Detector 'stop min limit' still crossed. Doing step up. Current detector value: "
415 << currentDetVal << std::endl;
416 }
417 _logger(Loger::DETAIL) << __func__ << ": " << _logIdString << ": Send actuator from : " << currentPos
418 << "(hwd read: " << _actuators[idOfDetector]->getCurrentPositionInSteps()
419 << ") to: " << newPosition << std::endl;
420
421 auto status =
422 moveAndstopActuatorWhenInRange(newPosition, _actuators[idOfDetector], _detectorsLocked[idOfDetector]);
423 if(_tuningInProgress) {
424 if(status != SMAStatusTypes::M_OK && !_actStoppedWhenInRange) {
425 _logger(Loger::ERROR) << __func__ << ": " << _logIdString << ": Send to position: " << newPosition
426 << " failed with status: " << status.asString() << ". Stop tuning. " << std::endl;
427 _oppStatus.set("Actuator error detected.", OppStatus::TYPE::WARNING);
428 tuningNotDone = TUNE_FLAG::ERROR;
429 actuatorError = true;
430 _actMovementErrorCnt++;
431 }
432 else
433 _actMovementErrorCnt = 0;
434
435 if(_actMovementErrorCnt > 30) {
436 _actMovementErrorCnt = 0;
437 _logger(Loger::ERROR)
438 << __func__ << ": " << _logIdString
439 << ": Actuators failed to move for more than 30 tuning cycles. Disable fast loop due to safety."
440 << std::endl;
441 _controllerManager->setEnable(false);
442 _oppStatus.set("Actuator permanent error. Open loop for safety.", OppStatus::TYPE::ERROR);
443 return;
444 }
445
446 if(idOfDetector > 0 && !actuatorError) {
447 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Step with detector ID: " << idOfDetector
448 << " done. Now moving to detector ID: " << idOfDetector - 1 << std::endl;
449 std::this_thread::sleep_for(std::chrono::milliseconds(500));
450 tuneLockedCore(idOfDetector - 1);
451 }
452 else if(_actStoppedWhenInRange) {
453 _actStoppedWhenInRange = TUNE_FLAG::SUCCESSFUL;
454 _oppStatus.set("Tuning successful.", OppStatus::TYPE::OK);
455 return;
456 }
457
458 if(_controllerManager->isCtrlSafetyHoldOn()) {
459 _logger(Loger::ERROR) << __func__ << ": " << _logIdString
460 << ": Fast controller safety triggered. Tuning (locked) failed. Finish thread."
461 << std::endl;
462 tuningNotDone = TUNE_FLAG::ERROR;
463 _oppStatus.set("Controller safety triggered. Tuning failed.", OppStatus::TYPE::WARNING);
464 }
465 else if(!_controllerManager->isCtrlActive()) {
466 _logger(Loger::ERROR)
467 << __func__ << ": " << _logIdString
468 << ": Fast controller is not active. Tuning (locked) cannot be continued. Finish thread." << std::endl;
469 tuningNotDone = TUNE_FLAG::ERROR;
470 _oppStatus.set("Controller not active. Tuning terminated. Check log file.", OppStatus::TYPE::ERROR);
471 }
472 }
473 else {
474 tuningNotDone = TUNE_FLAG::ERROR;
475 _logger(Loger::INFO) << __func__ << ": " << _logIdString
476 << ": Tuning (locked) stopped after/during stepper motor movement." << std::endl;
477 }
478 }
479 std::this_thread::sleep_for(std::chrono::milliseconds(50));
480 } while(tuningNotDone == TUNE_FLAG::CONTINUE);
481}

490void MultiActuatorCoarseTuningAlgorithm::tuneUnlockedInternal() {
491 _currentTuneType = UNLOCKED;
492
493 _logger(Loger::INFO) << __func__ << ": " << _logIdString
494 << ": Start tuning for unlocked. Beat frequency: " << _detectorsUnlocked[0]->getCurrentValue()
495 << std::endl;
496
497 if(!tuneCommon()) {
498 _tuningInProgress = false;
499 return;
500 }
501
502 if(isDetSafetyOn(_detectorsUnlocked[0])) {
503 _logger(Loger::ERROR) << __func__ << ": " << _logIdString
504 << ": Unlocked detector error: " << _detectorsUnlocked[0]->getErrorMsg() << ". Cannot tune."
505 << std::endl;
506 _oppStatus.set("Beat freq. detector safety error. Stop tuning. " + _detectorsUnlocked[0]->getErrorMsg(),
507 OppStatus::TYPE::ERROR);
508 _tuningInProgress = false;
509 return;
510 }
511
512 if(_controllerManager->isCtrlActive()) {
513 _logger(Loger::ERROR) << __func__ << ": " << _logIdString
514 << ": Fast controller is active. Tuning (unlocked) cannot be continued. Finish thread."
515 << std::endl;
516 _oppStatus.set("Controller active. Tuning terminated. Check log file.", OppStatus::TYPE::ERROR);
517 _tuningInProgress = false;
518 return;
519 }
520
521 TUNE_FLAG tuningNotDone = TUNE_FLAG::CONTINUE;
522 bool detectorError = _detectorsUnlocked[0]->getError();
523
524 if(detectorError || _detectorsUnlocked[0]->checkStopCondition() == CsaCoarseTuningDetector::LIMIT::IN_RANGE) {
525 if(!detectorError) {
526 _logger(Loger::INFO) << __func__ << ": " << _logIdString
527 << ": Tune done. Frequency already in range. No stepping needed. " << std::endl;
528 if(_lockWhenTuned) {
529 _controllerManager->setEnable(true);
530 _oppStatus.set("Tuning successful. Closing loop.", OppStatus::TYPE::OK);
531 }
532 else
533 _oppStatus.set("Tuning successful.", OppStatus::TYPE::OK);
534 }
535 else {
536 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Not locked detector error. Cannot tune."
537 << std::endl;
538 _oppStatus.set("Cannot tune. Error:" + _detectorsUnlocked[0]->getErrorMsg(), OppStatus::TYPE::ERROR);
539 }
540 _lockWhenTuned = false;
541 _tuningInProgress = false;
542 return;
543 }
544
545 uint32_t highestIndex = 0;
546 for(uint32_t i = 0; i < _actuators.size(); i++)
547 if(((_actuactor4UnlockedDisableMask->getValue() & (1 << i)) == false) && (highestIndex < i)) highestIndex = i;
548
549 // tune higher order actuators (more precise) to middle of their operation ranges - most precise actuator must have
550 // index 0 in the _actuators vector
551 if(highestIndex > 0) {
552 _oppStatus.set("Set actuators in tune start position.", OppStatus::TYPE::INFO);
553 for(uint32_t i = 0; i < highestIndex; i++)
554 if(!(_actuactor4UnlockedDisableMask->getValue() & (1 << i))) {
555 double newPos = 0.0;
556 if(_actuatorsStartPositionForTune.size() > i) newPos = _actuatorsStartPositionForTune[i];
557
558 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Move actuator id: " << i
559 << " to operation point: " << newPos
560 << ". Current detector value: " << _detectorsUnlocked[0]->getCurrentValue() << std::endl;
561 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Actuator id: " << i
562 << " current position is: " << _actuators[i]->getCurrentPosition() << ". Moving."
563 << std::endl;
564 _actuators[i]->moveToPosition(newPos);
565 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Detector value after actuator id: " << i
566 << " move is: " << _detectorsUnlocked[0]->getCurrentValue()
567 << ". Current actuator position" << _actuators[i]->getCurrentPosition() << std::endl;
568 }
569 }
570 std::this_thread::sleep_for(std::chrono::milliseconds(500));
571
572 uint32_t activeActuatorId = _actuators.size() - 1;
573 CsaCoarseTuningDetector::LIMIT stopConditionStatusPrev = _detectorsUnlocked[0]->checkStopCondition();
574
575 auto currBeatFreq = _detectorsUnlocked[0]->getCurrentValue(); // just to update detector
576 _logger(Loger::INFO) << __func__ << ": " << _logIdString
577 << ": Start tuning stepping loop. Beat frequency: " << currBeatFreq << std::endl;
578 do {
579 if(_tuningInProgress == false) {
580 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Tuning stopped." << std::endl;
581 _oppStatus.set("Tuning stopped.", OppStatus::TYPE::INFO);
582 return;
583 }
584
585 if((_actuactor4UnlockedDisableMask->getValue() & (1 << activeActuatorId))) {
586 _logger(Loger::WARNING) << __func__ << ": " << _logIdString << ": Actuator: " << activeActuatorId
587 << " is disabled for tuning when controller is not active." << std::endl;
588 if(activeActuatorId > 0)
589 activeActuatorId--;
590 else {
591 _logger(Loger::WARNING) << __func__ << ": " << _logIdString << ": Most precise actuator disabled. Cannot tune."
592 << std::endl;
593 tuningNotDone = TUNE_FLAG::ERROR;
594 }
595 }
596 else {
597 // check stop condition
598 CsaCoarseTuningDetector::LIMIT currentStopCondition = _detectorsUnlocked[0]->checkStopCondition();
599 // first check safety
600 if(_detectorsUnlocked[0]->checkSafetyCondition() != CsaCoarseTuningDetector::LIMIT::IN_RANGE) {
601 _logger(Loger::ERROR) << __func__ << ": " << _logIdString
602 << ": Stopping current tuning process due to safety reasons." << std::endl;
603 _oppStatus.set("Beat freq. detector safety error. Stop tuning. " + _detectorsUnlocked[0]->getErrorMsg(),
604 OppStatus::TYPE::ERROR);
605 tuningNotDone = TUNE_FLAG::ERROR;
606 _safetyAutoresetCnt = 0;
607 _safetyLimitsCrossedFlag = true;
608 }
609 else if(currentStopCondition == CsaCoarseTuningDetector::LIMIT::IN_RANGE) {
610 tuningNotDone = TUNE_FLAG::SUCCESSFUL;
611 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Tune done. Signal in range" << std::endl;
612 _oppStatus.set("Tuning successful.", OppStatus::TYPE::OK);
613 }
614 else if(currentStopCondition != stopConditionStatusPrev) {
615 activeActuatorId--;
616 if(activeActuatorId > _actuators.size() - 1) { // we are using fact that activeActuatorId is unsigned, when it
617 // is turn it is jumping to +4billions and condition is not met
618 _logger(Loger::ERROR) << __func__ << ": " << _logIdString
619 << ": Tuning failed. Check step sizes and directions." << std::endl;
620 tuningNotDone = TUNE_FLAG::ERROR;
621 }
622 else {
623 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Tuning with actuator id: " << activeActuatorId
624 << " cannot be done. Change actuator to more precise." << std::endl;
625 stopConditionStatusPrev = currentStopCondition;
626 }
627 }
628 else {
629 auto currentDetVal = _detectorsUnlocked[0]->getCurrentValue();
630 auto currentPos = _actuators[activeActuatorId]->getCurrentPosition();
631 auto newPosition = currentPos;
632 if(currentStopCondition == CsaCoarseTuningDetector::LIMIT::MAX_CROSSED) {
633 newPosition = currentPos - _stepScaling * _actuatorsStepsSize4Unlocked[activeActuatorId];
634 _logger(Loger::DETAIL) << __func__ << ": " << _logIdString
635 << ": Detector 'stop max limit' still crossed. Doing step ( "
636 << _actuatorsStepsSize4Unlocked[activeActuatorId]
637 << " ) down. Current detector value: " << currentDetVal << std::endl;
638 }
639 else if(currentStopCondition == CsaCoarseTuningDetector::LIMIT::MIN_CROSSED) {
640 newPosition = currentPos + _stepScaling * _actuatorsStepsSize4Unlocked[activeActuatorId];
641 _logger(Loger::DETAIL) << __func__ << ": " << _logIdString
642 << ": Detector 'stop min limit' still crossed. Doing step ( "
643 << _actuatorsStepsSize4Unlocked[activeActuatorId]
644 << " ) up. Current detector value: " << currentDetVal << std::endl;
645 }
646 _logger(Loger::DETAIL) << __func__ << ": " << _logIdString << ": Send actuator (id:" << activeActuatorId
647 << ") from : " << currentPos
648 << "(hwd read: " << _actuators[activeActuatorId]->getCurrentPositionInSteps()
649 << ") to: " << newPosition << std::endl;
650 // auto status = _actuators[activeActuatorId]->moveToPosition(newPosition);
651 auto status = moveAndstopActuatorWhenInRange(newPosition, _actuators[activeActuatorId], _detectorsUnlocked[0]);
652 if(_actStoppedWhenInRange) {
653 tuningNotDone = TUNE_FLAG::SUCCESSFUL;
654 _actStoppedWhenInRange = false;
655 _oppStatus.set("Tuning successful.", OppStatus::TYPE::OK);
656 }
657 else if(_tuningInProgress) {
658 if(status != SMAStatusTypes::M_OK && !_actStoppedWhenInRange) {
659 _logger(Loger::ERROR) << __func__ << ": " << _logIdString << ": Send to position: " << newPosition
660 << " failed with status: " << status.asString()
661 << " Error is: " << _actuators[activeActuatorId]->getError().asString() << std::endl;
662 activeActuatorId--;
663 if(activeActuatorId >
664 _actuators.size() - 1) { // we are using fact that activeActuatorId is unsigned, when it is turn it is
665 // jumping to +4billions and condition is not met
666 _logger(Loger::ERROR) << __func__ << ": " << _logIdString
667 << ": Tuning (unlocked) failed. Most precise actuator error." << std::endl;
668 tuningNotDone = TUNE_FLAG::ERROR;
669 _oppStatus.set("Tuning failed. Actuator error.", OppStatus::TYPE::ERROR);
670 }
671 else {
672 _logger(Loger::INFO) << __func__ << ": " << _logIdString
673 << ": Tuning with actuator id: " << activeActuatorId + 1
674 << " cannot be done due to actuator error. Change actuator to more precise."
675 << std::endl;
676 stopConditionStatusPrev = currentStopCondition;
677 }
678 }
679
680 if(_controllerManager->isCtrlSafetyHoldOn()) {
681 _logger(Loger::ERROR) << __func__ << ": " << _logIdString
682 << ": Controller safety triggered. Tuning (unlocked) failed. Finish thread."
683 << std::endl;
684 _oppStatus.set("Controller safety triggered. Tuning failed.", OppStatus::TYPE::ERROR);
685 tuningNotDone = TUNE_FLAG::ERROR;
686 }
687 else if(_controllerManager->isCtrlActive()) {
688 _logger(Loger::ERROR)
689 << __func__ << ": " << _logIdString
690 << ": Fast controller is active. Tuning (unlocked) cannot be continued. Finish thread." << std::endl;
691 tuningNotDone = TUNE_FLAG::ERROR;
692 _oppStatus.set("Controller active. Tuning terminated. Check log file.", OppStatus::TYPE::ERROR);
693 }
694 }
695 else {
696 _logger(Loger::INFO) << __func__ << ": " << _logIdString
697 << ": Tuning (unlocked) stopped after/during stepper motor movement." << std::endl;
698 }
699 }
700 }
701 std::this_thread::sleep_for(std::chrono::milliseconds(50));
702 } while(tuningNotDone == TUNE_FLAG::CONTINUE);
703
704 if(tuningNotDone == TUNE_FLAG::SUCCESSFUL && _lockWhenTuned) {
705 _oppStatus.set("Tuning successful. Closing loop.", OppStatus::TYPE::OK);
706 _controllerManager->setEnable(true, false);
707 }
708
709 _lockWhenTuned = false;
710 _tuningInProgress = false;
711}

713bool MultiActuatorCoarseTuningAlgorithm::tuneCommon() {
714 if(_actuatorsStepsSize4Locked.size() != _actuators.size() ||
715 _actuatorsStepsSize4Unlocked.size() != _actuators.size()) {
716 _logger() << __func__ << ": " << _logIdString
717 << ": Tuning step size vector mismatch. Actuator vector size: " << _actuators.size()
718 << ". Vector for active controller: " << _actuatorsStepsSize4Locked.size()
719 << " not active: " << _actuatorsStepsSize4Unlocked.size() << std::endl;
720 exit(1);
721 }
722
723 _oppStatus.set("Tuning in progress.", OppStatus::TYPE::INFO);
724
725 if(_tuningInProgress) {
726 _logger(Loger::WARNING) << __func__ << ": " << _logIdString << ": Tuning thread already active. Ignore command."
727 << std::endl;
728 _oppStatus.set("Tuning thread already active. Ignore command.", OppStatus::TYPE::WARNING);
729 return false;
730 }
731
732 if(_controllerManager->isCtrlSwitchingInProgress()) {
733 _logger(Loger::WARNING) << __func__ << ": " << _logIdString << ": Controller switching in progress. Cannot tune."
734 << std::endl;
735 _oppStatus.set("Controller switching in progress. Cannot tune.", OppStatus::TYPE::WARNING);
736 return false;
737 }
738
739 if(_doNotAllowForTuningWhenReferenceUnlocked && !_referenceLocked) {
740 _logger(Loger::ERROR) << __func__ << ": " << _logIdString << ": System Reference NOT locked. Cannot tune."
741 << std::endl;
742 _oppStatus.set("System Reference NOT locked. Cannot tune.", OppStatus::TYPE::ERROR);
743 return false;
744 }
745
746 _tuningInProgress = true;
747 for(auto& item : _actuators) item->setAutostart(false);
748
749 if(!cleanSafetyAndRampController()) {
750 _oppStatus.set("Ramping controller failed. Cannot tune.", OppStatus::TYPE::ERROR);
751 return false;
752 }
753
754 return true;
755}

768void MultiActuatorCoarseTuningAlgorithm::stopActuatorWhenInRangeThread(
769 boost::shared_ptr<StepperActuator> actuator, boost::shared_ptr<CsaCoarseTuningDetector> detector) {
770 std::this_thread::sleep_for(std::chrono::milliseconds(5));
771 _actStoppedWhenInRange = false;
772 while(_terminateThread == false && actuator->getStatus() != SMAStatusTypes::M_OK &&
773 actuator->getError() == SMAErrorTypes::M_NO_ERROR) {
774 if(!_tuningInProgress) {
775 _logger(Loger::ERROR) << __func__ << ": " << _logIdString << ": Tuning stopped, stop actuator." << std::endl;
776 _terminateThread = true;
777 }
778 else if(isDetSafetyOn(
779 detector)) { // detector->checkSafetyCondition() != CsaCoarseTuningDetector::LIMIT::IN_RANGE) {
780 _logger(Loger::ERROR) << __func__ << ": " << _logIdString << ": Stop actuator due to safety." << std::endl;
781 _terminateThread = true;
782 }
783 else if(detector->checkStopCondition() == CsaCoarseTuningDetector::LIMIT::IN_RANGE) {
784 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Stop actuator - detector in range." << std::endl;
785 _actStoppedWhenInRange = true;
786 _terminateThread = true;
787 }
788 // wait for 200 ms, but with control of _terminateThread flag
789 for(int i = 0; i < 100; i++) {
790 if(_terminateThread) break;
791 std::this_thread::sleep_for(std::chrono::milliseconds(2));
792 }
793 if(_terminateThread) {
794 actuator->stop();
795 return;
796 }
797 }

800bool MultiActuatorCoarseTuningAlgorithm::cleanSafetyAndRampController() {
801 int32_t timeoutCnt = 0;
802 // check whether controller need ramping and ramp it if needed
803 if(_controllerManager->isCtrlSafetyHoldOn()) {
804 _logger(Loger::INFO) << __func__ << ": " << _logIdString
805 << ": Safety hold enabled. Need to ramp controller and reset safety flag. Start ramping."
806 << std::endl;
807 _oppStatus.set("Safety hold enabled. Ramp controller.", OppStatus::TYPE::INFO);
808 _controllerManager->rampAndResetCtrl();
809 while(_controllerManager->isCtrlRampingInProgress() == false) {
810 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Wait for ramping thread to start." << std::endl;
811 std::this_thread::sleep_for(std::chrono::milliseconds(1));
812 timeoutCnt++;
813 if(!_tuningInProgress || timeoutCnt > 10) // stopped by user or timeout bigger than 10 milliseconds
814 return false;
815 }
816 timeoutCnt = 0;
817 while(_controllerManager->isCtrlRampingInProgress()) {
818 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Controller ramping in progress." << std::endl;
819 std::this_thread::sleep_for(std::chrono::milliseconds(15));
820 timeoutCnt++;
821 if(!_tuningInProgress || timeoutCnt > 60) // stopped by user or timeout bigger than 15*60=900 milliseconds
822 return false;
823 }
824 // wait extra second to see the influence of ramping on the physical system
825 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Controller ramping done. Wait extra 200 millisecond."
826 << std::endl;
827 std::this_thread::sleep_for(std::chrono::milliseconds(200));
828 }
829 return true;
830}

1006CoarseTuningAlgorithmBase::STATUS MultiActuatorCoarseTuningAlgorithm::evaluateStatus() {
1007 // if disabled then status is just Disabled
1008 if(!_enabled) {
1009 _status = CoarseTuningAlgorithmBase::STATUS::DISABLED;
1010 _statusInfo = "Coarse tuning disabled.";
1011 return _status;
1012 }
1013
1014 // assume that status will be OK, store value in local variable
1015 CoarseTuningAlgorithmBase::STATUS status = CoarseTuningAlgorithmBase::STATUS::OK;
1016 std::string statusInfo("");
1017
1018 // count how many detector for locked and unlocked is in the error state
1019 uint32_t detUnlockedInErrorCnt = 0;
1020 for(auto& detector : _detectorsUnlocked)
1021 if(detector->getError()) detUnlockedInErrorCnt++;
1022
1023 uint32_t detLockedInErrorCnt = 0;
1024 for(auto& detector : _detectorsLocked)
1025 if(detector->getError()) detLockedInErrorCnt++;
1026
1027 // count how many actuators is in error
1028 uint32_t actInErrorCnt = 0;
1029 for(auto& actuator : _actuators)
1030 if(actuator->getStatus() == SMAStatusTypes::M_ERROR) actInErrorCnt++;
1031
1032 // count how many detector for locked and unlocked has a wrong settings
1033 uint32_t detLockedSettingErrorCnt = 0;
1034 for(auto detector : _detectorsLocked)
1035 if(!detector->checkSettingsValidity()) detLockedSettingErrorCnt++;
1036
1037 uint32_t detUnlockedSettingErrorCnt = 0;
1038 for(auto detector : _detectorsUnlocked)
1039 if(!detector->checkSettingsValidity()) detUnlockedSettingErrorCnt++;
1040
1041 // count how many disabled actuators and actuators with step size equal 0 for locked case
1042 uint32_t zeroStepSizeLockedCnt = 0;
1043 uint32_t disabledActuatorsLockedCnt = 0;
1044 for(uint32_t i = 0; i < _actuators.size() && i < _actuatorsStepsSize4Locked.size(); i++) {
1045 if(_actuactor4LockedDisableMask->getValue() & (1 << i)) disabledActuatorsLockedCnt++;
1046 // actuator is not disabled and size step is 0
1047 if(!(_actuactor4LockedDisableMask->getValue() & (1 << i)) && _actuatorsStepsSize4Locked[i] == 0)
1048 zeroStepSizeLockedCnt++;
1049 }
1050 // count how many disabled actuators and actuators with step size equal 0 for unlocked case
1051 uint32_t zeroStepSizeUnlockedCnt = 0;
1052 uint32_t disabledActuatorsUnlockedCnt = 0;
1053 for(uint32_t i = 0; i < _actuators.size() && i < _actuatorsStepsSize4Unlocked.size(); i++) {
1054 if(_actuactor4UnlockedDisableMask->getValue() & (1 << i)) disabledActuatorsUnlockedCnt++;
1055 // actuator is not disabled and size step is 0
1056 if(!(_actuactor4UnlockedDisableMask->getValue() & (1 << i)) && _actuatorsStepsSize4Unlocked[i] == 0)
1057 zeroStepSizeUnlockedCnt++;
1058 }
1059
1060 _allActuatorsDisabledForLocked = (disabledActuatorsLockedCnt == _actuators.size()) ? true : false;
1061 _allActuatorsDisabledForUnlocked = (disabledActuatorsUnlockedCnt == _actuators.size()) ? true : false;
1062
1063 //
1064 // evaluate errors
1065 //
1066
1067 if(_doNotAllowForTuningWhenReferenceUnlocked && !_referenceLocked) {
1068 status = CoarseTuningAlgorithmBase::STATUS::ERROR;
1069 statusInfo.append("Reference NOT locked - cannot tune. ");
1070 }
1071
1072 // if all actuators are in error, it is also error
1073 if(actInErrorCnt == _actuators.size()) {
1074 status = CoarseTuningAlgorithmBase::STATUS::ERROR;
1075 if(_actuators.size() > 1)
1076 statusInfo.append("All actuators in ERROR. ");
1077 else
1078 statusInfo.append("Actuator in ERROR. ");
1079 }
1080
1081 if(_controllerManager->isCtrlActive()) {
1082 // if locked, and all detectors for locked are in error, it means error
1083 if(detLockedInErrorCnt == _detectorsLocked.size()) {
1084 status = CoarseTuningAlgorithmBase::STATUS::ERROR;
1085 statusInfo.append("Detector for locked case (ctrl/PZT/motors) in ERROR. ");
1086 }
1087
1088 // if locked, and all detectors for locked have wrong settings, it means error
1089 if(detLockedSettingErrorCnt == _detectorsLocked.size()) {
1090 status = CoarseTuningAlgorithmBase::STATUS::ERROR;
1091 statusInfo.append("Setting error for locked case. ");
1092 }
1093 }
1094 else {
1095 // if unlocked, and all detectors for unlocked are in error, it means error
1096 if(detUnlockedInErrorCnt == _detectorsUnlocked.size()) {
1097 status = CoarseTuningAlgorithmBase::STATUS::ERROR;
1098 statusInfo.append("Phase detector (beat frequency) in ERROR. ");
1099 }
1100 // if unlocked, and all detectors for unlocked have wrong settings, it means error
1101 if(detUnlockedSettingErrorCnt == _detectorsUnlocked.size()) {
1102 status = CoarseTuningAlgorithmBase::STATUS::ERROR;
1103 statusInfo.append("Setting error for unlocked case. ");
1104 }
1105 }
1106
1107 if(_controllerManager->isCtrlActive()) {
1108 if(_allActuatorsDisabledForLocked) {
