43 void LaserMenhir1550CommLogger::enableLoggingCb() {
44  if(!_loggingEnabled && _enableLogging->getValue()) {
45  _logFileNameApplied = _logFileName->getValue();
46 
47  _logger(Loger::INFO) << __func__ << ": "
48  << "Enabling Menhir communication logging. Opening in append mode the file: "
49  << _logFileNameApplied << std::endl;
50  {
51  std::lock_guard<std::mutex> guard(_logFileMutex);
52  _logFile.exceptions(std::ofstream::failbit | std::ofstream::badbit);
53 
54  try {
55  if(_logFile.is_open()) _logFile.close();
56  _logFile.open(_logFileNameApplied, std::ofstream::out | std::ofstream::app);
57 
58  _loggingEnabled = true;
59  }
60  catch(std::ofstream::failure& e) {
61  _enableLogging->setValue(0, false);
62  _enableLogging->synchronize();
63  _logger(Loger::ERROR) << __func__ << ": "
64  << "Menhir logfile error when opening. Disabling logging Exception: " << e.what()
65  << std::endl;
66  }
67  }
68  }
69 
70  if(_loggingEnabled && (_enableLogging->getValue() == 0)) {
71  _logger(Loger::INFO) << __func__ << ": "
72  << "Disabling Menhir communication logging. Closing file: " << _logFileNameApplied
73  << std::endl;
74  _loggingEnabled = false;
75  {
76  std::lock_guard<std::mutex> guard(_logFileMutex);
77  try {
78  _logFile.flush();

80  }

146 void LaserMenhir1550CommLogger::logPeriodically(std::string response, std::string command) {
147  if(!_loggingEnabled) return;
148 
149  if(command == _commands.arw) {
150  if(arwCmdUpdate.isExecuteAutoUpdate()) logAlways(response, command);
151  }
152  else if(command == _commands.sen) {
153  if(senCmdUpdate.isExecuteAutoUpdate()) {
154  auto charPos = response.find_first_of("\r");
155  if(charPos != std::string::npos) {
156  response.erase(charPos, 1);
157  }
158  charPos = response.find_first_of("\n");
159  if(charPos != std::string::npos) {
160  response.replace(charPos, 1, std::string(", "));
161  }
162  logAlways(response, command);
163  }
164  }
165  else {
166  logAlways(response, command);
167  }
