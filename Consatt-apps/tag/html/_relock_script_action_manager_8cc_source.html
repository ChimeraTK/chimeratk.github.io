30void RelockScriptActionManager::update(LOCK_STATUS_VALUE currentLockStatus) {
31 if(_relockActionScriptEnable->getValue()) {
32 if((_lockStatusPrevious != LOCK_STATUS_VALUE::OK && currentLockStatus == LOCK_STATUS_VALUE::OK) ||
33 _testFakeRelockEvent) {
34 _logger(Loger::INFO) << __func__ << ": Relock event detected. Arm script execution. Wait delay time."
35 << std::endl;
36 _relockActionRelockTime = std::chrono::high_resolution_clock::now();
37 _relockActionArmed = true;
38 _testFakeRelockEventRegistered = true;
39 }
40 if(_relockActionArmed && currentLockStatus != LOCK_STATUS_VALUE::OK && _testFakeRelockEventRegistered == false) {
41 _relockActionArmed = false;
42 _logger(Loger::INFO) << __func__ << ": Unlocked event detected. Disarm script execution." << std::endl;
43 }
44 _testFakeRelockEvent = false;
45
46 if(_relockActionArmed) {
47 auto delay = _relockActionScriptDelaySec->getValue();
48 if(delay < 1) delay = 1;
49
50 auto duration = std::chrono::duration_cast<std::chrono::seconds>(
51 std::chrono::high_resolution_clock::now() - _relockActionRelockTime)
52 .count();
53 if(duration >= delay) {
54 if(_pauseRelockScript) {
55 _logger(Loger::INFO) << __func__ << ": Relock action execution paused until tuning and switching done."
56 << std::endl;
57 return;
58 }
59 executeRelockEvent();
60 _relockActionArmed = false;
61 _testFakeRelockEventRegistered = false;
62 }
63 }
64 }
65 else {
66 _relockActionArmed = false;
67 }
68
69 _lockStatusPrevious = currentLockStatus;
70}

85void RelockScriptActionManager::relockActionScriptThreadCore() {
86 auto addr = _relockActionSendDataToExternalAddress->getValue();
87 if(!addr.empty()) {
88 auto val = _relockActionSendDataToExternalValue->getValue();
89 _logger(Loger::INFO) << __func__ << ": Relock action: Try send: " << val << " to address: " << addr << " ."
90 << std::endl;
91 try {
92 _relockActionSendDataToExternalAccesor.setValue(val);
93 }
94 catch(CsaExternalProcessException& ex) {
95 _logger(Loger::INFO) << __func__ << ": External property not available - check address(current address: '" << addr
96 << "') or status of external server. Exception is: " << ex.what() << std::endl;
97 }
98 return;
99 }
100
101 std::string path = _relockActionScriptPath->getValue();
102 if(path.empty()) {
103 _logger(Loger::INFO) << __func__ << ": Relock action script path is empty." << std::endl;
104 return;
105 }
106 _logger(Loger::INFO) << __func__ << ": Start relock action script. Script path: " << path << std::endl;
107 int result = std::system(path.c_str());
108 if(result) _logger(Loger::INFO) << __func__ << ": Result code of system call: " << result << std::endl;
109}
