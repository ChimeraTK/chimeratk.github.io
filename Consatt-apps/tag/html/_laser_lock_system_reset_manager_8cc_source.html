33void LaserLockSystemResetManager::update() {
34 _pzt4Manager->update();
35
36 if(!_resetSystemInProgress) _resetClkNotInProgress->setValue(2);
37
38 if(_resetSystemInProgress || _pzt4PowerResetInProgress) {
39 _resetNotInProgress->setValue(0);
40 return;
41 }
42 else
43 _resetNotInProgress->setValue(2);
44
45 if(_resetSystem || _autoResetSystem) {
46 std::thread t1(std::thread(&LaserLockSystemResetManager::resetSystem, this));
47 _resetNotInProgress->setValue(0);
48 _resetClkNotInProgress->setValue(0);
49 t1.detach();
50 _resetSystem = false;
51 _autoResetSystem = false;
52 return;
53 }
54
55 if(_pzt4PowerReset || _autoPzt4PowerReset) {
56 std::thread t1(std::thread(&LaserLockSystemResetManager::pzt4PowerReset, this));
57 _resetNotInProgress->setValue(0);
58 t1.detach();
59 _pzt4PowerReset = false;
60 _autoPzt4PowerReset = false;
61 return;
62 }
63
64 sis8300ClkValidate();
65 if(!_pzt4Manager->getPzt4FullPowerResetsInProgress()) _autoPzt4PowerReset = _pzt4Manager->pzt4StatusValidate();
66}

68void LaserLockSystemResetManager::sis8300ClkValidate() {
69 if(_enableSysResetOnClkLost->getValue() == 0 || _resetSystemInProgress) {
70 _sis8300lClockErrorCnt = 0;
71 return;
72 }
73
74 auto autoResetMinGapMinutes = _pzt4Manager->getMinPeriodBetweenResets();
75 if(autoResetMinGapMinutes <= 0) autoResetMinGapMinutes = 3;
76
77 bool resetAllowed = false;
78 auto duration =
79 std::chrono::duration_cast<std::chrono::seconds>(std::chrono::high_resolution_clock::now() - _lastAutoSystemReset)
80 .count();
81 if(duration > autoResetMinGapMinutes * 60) resetAllowed = true;
82
83 int clockMaxError = 20;
84 bool clockValid = _laserLockSvr->_bspSIS8300L2->clk()->validateCurrentClockSettings();
85 int32_t currentClk = _laserLockSvr->_bspSIS8300L2->clk()->getApplicationClockFreq();
86 if(!clockValid && _sis8300lClockErrorCnt < clockMaxError) {
87 _logger(Loger::WARNING) << __func__ << ": SIS8300L Clock is NOT OK. Validate for next "
88 << (clockMaxError - _sis8300lClockErrorCnt) << " updates." << std::endl;
89 _sis8300lClockErrorCnt++;
90 }
91 else if(!clockValid && _sis8300lClockErrorCnt >= clockMaxError) {
92 int32_t maxConstatCnt = 50;
93 if(_sis8300lClockIsConstantCntInErrorState >= maxConstatCnt)
94 _autoResetSystem = true;
95 else {
96 if(std::abs(_sis8300lClockAppValueInErrorState - currentClk) < 5000 && currentClk > 1000000) {
97 _sis8300lClockIsConstantCntInErrorState++;
98 _logger(Loger::INFO) << __func__ << ": SIS8300L Clock is NOT OK, but stable now. Reset in: "
99 << (maxConstatCnt - _sis8300lClockIsConstantCntInErrorState) << " updates." << std::endl;
100 }
101 else {
102 _sis8300lClockIsConstantCntInErrorState = 0;
103 _logger(Loger::INFO) << __func__ << ": SIS8300L Clock is NOT OK and not stable." << std::endl;
104 }
105 }
106 }
107 else if(clockValid && _sis8300lClockErrorCnt >= clockMaxError) {
108 _logger(Loger::WARNING) << __func__ << ": SIS8300L Clock is returned to be OK. Triggering the system reset."
109 << std::endl;
110 _autoResetSystem = true;
111 }
112
113 if(!clockValid) _sis8300lClockAppValueInErrorState = currentClk;
114
115 if(_autoResetSystem) {
116 if(resetAllowed) {
117 _lastAutoSystemReset = std::chrono::high_resolution_clock::now();
118 _sis8300lClockIsConstantCntInErrorState = 0;
119 _sis8300lClockErrorCnt = 0;
120 _logger(Loger::INFO) << __func__ << ": Triggering system reset." << std::endl;
121 }
122 else {
123 _logger(Loger::INFO) << __func__ << ": Wait to trigger system reset. Minimum time between auto resets is "
124 << autoResetMinGapMinutes << " minute/s." << std::endl;
125 _autoResetSystem = false;
126 }
127 }

130void LaserLockSystemResetManager::resetSystem() {
131 if(_pzt4PowerResetInProgress) {
132 _logger(Loger::ERROR) << __func__ << ": PZT4 RTM reset already in progress." << std::endl;
133 return;
134 }
135 if(_resetSystemInProgress) {
136 _logger(Loger::ERROR) << __func__ << ": System reset already in progress." << std::endl;
137 return;
138 }
139
140 _resetSystemInProgress = true;
141 auto time_start = std::chrono::high_resolution_clock::now();
142
143 // wait 1 sec for main loop to be frozen
144 std::this_thread::sleep_for(std::chrono::milliseconds(1000));
145
146 auto coarseTuneEnabled = _laserLockSvr->_mutliActCoarseTuneAlgorithm->getCoarseTuningEnabled();
147 if(coarseTuneEnabled) {
148 _logger(Loger::INFO) << __func__ << ": Disable coarse tuning." << std::endl;
149 _laserLockSvr->_mutliActCoarseTuneAlgorithm->triggerCommand(
150 CsaCoarseTuningMultiDetMultiActAlgorithm::CMD::CMD_STOP_TUNE);
151 _laserLockSvr->_mutliActCoarseTuneAlgorithm->setCoarseTuningEnabled(false);
152 std::this_thread::sleep_for(std::chrono::milliseconds(500));
153 }
154
155 _logger(Loger::INFO) << __func__ << ": Disable fast loop for both controller. First main controller." << std::endl;
156 auto initUnlockByUser = _laserLockSvr->_lockStatusValidator->getUnlockByUser();
157 _laserLockSvr->_lockStatusValidator->setUnlockByUser(true);
158 _laserLockSvr->_mutliActCoarseTuneAlgorithm->triggerCommand(
159 CsaCoarseTuningMultiDetMultiActAlgorithm::CMD::CMD_DISABLE_FAST_LOOP);
160 if(_laserLockSvr->_ctrl1Manager) {
161 _logger(Loger::INFO) << __func__ << ": Disable auxiliary controller." << std::endl;
162 _laserLockSvr->_ctrl1Manager->setEnable(false);
163 }
164 std::this_thread::sleep_for(std::chrono::milliseconds(100));
165
166 std::vector<int> actToReenableIndexes;
167 for(uint32_t i = 0; i < _laserLockSvr->_coarseTuneActuators.size(); i++) {
168 // auto type = _laserLockSvr->_coarseTuneActuators[i]->getMotorType();
169 if(_laserLockSvr->_coarseTuneActuators[i]->isEnabled()) {
170 _logger(Loger::INFO) << __func__ << ": Actuator enabled at index: " << i << ". Disabling." << std::endl;
171 _laserLockSvr->_coarseTuneActuators[i]->enable(false);
172 actToReenableIndexes.push_back(i);
173 }
174 }
175
176 _logger(Loger::INFO) << __func__ << ": Prepare to reset PZT4 board." << std::endl;
177
178 _logger(Loger::INFO) << __func__ << ": Store settings and deactivate all PZT4 channels." << std::endl;
179 _pzt4Manager->storePzt4State();
180 _pzt4Manager->shutdownPzt4Channels();
181 std::this_thread::sleep_for(std::chrono::milliseconds(500));
182 _logger(Loger::INFO) << __func__ << ": Enable reset of PZT4 DACs, ADC and sending logic." << std::endl;
183 _pzt4Manager->enablePzt4Resets();
184 _logger(Loger::INFO) << __func__ << ": Reset FMC20 board." << std::endl;
185 _laserLockSvr->_bspFMC20->bspCommon()->setMainReset(true);
186 std::this_thread::sleep_for(std::chrono::milliseconds(10));
187 _logger(Loger::INFO) << __func__ << ": Disable PZT4 board reset." << std::endl;
188 _laserLockSvr->_bspFMC20->bspCommon()->setMainReset(false);
189 std::this_thread::sleep_for(std::chrono::milliseconds(10));
190 _logger(Loger::INFO) << __func__ << ": Disable reset of PZT4 DACs, ADC and sending logic." << std::endl;
191 _pzt4Manager->disablePzt4Resets();
192 _logger(Loger::INFO) << __func__ << ": Restore settings and reactivate previously enabled PZT4 channels."
193 << std::endl;
194 _pzt4Manager->restartPzt4Channels();
195
196 _logger(Loger::INFO) << __func__ << ": Reset SIS8300L board." << std::endl;
197 _laserLockSvr->_bspSIS8300L2->bspCommon()->setMainReset(true);
198 _laserLockSvr->_bspSIS8300L2->adc()->setEnable(false);
199 std::this_thread::sleep_for(std::chrono::milliseconds(10));
200 _laserLockSvr->_bspSIS8300L2->bspCommon()->setMainReset(false);
201 _laserLockSvr->_bspSIS8300L2->adc()->setEnable(true);
202
203 _logger(Loger::INFO) << __func__ << ": Initialize SIS8300L clocks." << std::endl;
204 _laserLockSvr->_bspSIS8300L2->clk()->setClockSource(0);
205 _laserLockSvr->_bspSIS8300L2->clk()->initizalizeClock();
206 _logger(Loger::INFO) << __func__ << ": Toggle SIS8300L ADC enable." << std::endl;
207 _laserLockSvr->_bspSIS8300L2->adc()->setEnable(false);
208 std::this_thread::sleep_for(std::chrono::milliseconds(1));
209 _laserLockSvr->_bspSIS8300L2->adc()->setEnable(true);
210
211 // re-enabling actuators
212 for(auto& index : actToReenableIndexes) _laserLockSvr->_coarseTuneActuators[index]->enable(true);
213 std::this_thread::sleep_for(std::chrono::milliseconds(500));
214
215 if(coarseTuneEnabled) {
216 _laserLockSvr->_mutliActCoarseTuneAlgorithm->setCoarseTuningEnabled(true);
217 std::this_thread::sleep_for(std::chrono::milliseconds(50));
218 }
219
220 if(!initUnlockByUser) {
221 _laserLockSvr->laserServerCommands._validationEnableCheckTrigger = true;
222 _laserLockSvr->_mutliActCoarseTuneAlgorithm->triggerCommand(
223 CsaCoarseTuningMultiDetMultiActAlgorithm::CMD::CMD_UNPAUSE_AUTOLOCK_FEATURE);
224 }
225
226 _logger(Loger::INFO) << __func__ << ": Reset took[ms]: "
227 << std::chrono::duration_cast<std::chrono::milliseconds>(
228 std::chrono::high_resolution_clock::now() - time_start)
229 .count()
230 << std::endl;
231 _resetSystemInProgress = false;
232}

234void LaserLockSystemResetManager::shutdownSystem() {
235 if(_sysSetPzt4Reset.coarseTuneEnabled) {
236 _logger(Loger::INFO) << __func__ << ": Disable coarse tuning." << std::endl;
237 _laserLockSvr->_mutliActCoarseTuneAlgorithm->triggerCommand(
238 CsaCoarseTuningMultiDetMultiActAlgorithm::CMD::CMD_STOP_TUNE);
239 _laserLockSvr->_mutliActCoarseTuneAlgorithm->setCoarseTuningEnabled(false);
240 std::this_thread::sleep_for(std::chrono::milliseconds(500));
241 }
242
243 _logger(Loger::INFO) << __func__ << ": Disable fast loop for both controller. Main controller." << std::endl;
244 _laserLockSvr->_lockStatusValidator->setUnlockByUser(true);
245 _laserLockSvr->_mutliActCoarseTuneAlgorithm->triggerCommand(
246 CsaCoarseTuningMultiDetMultiActAlgorithm::CMD::CMD_DISABLE_FAST_LOOP);
247 if(_laserLockSvr->_ctrl1Manager) {
248 _logger(Loger::INFO) << __func__ << ": Disable auxiliary controller." << std::endl;
249 _laserLockSvr->_ctrl1Manager->setEnable(false);
250 }
251 std::this_thread::sleep_for(std::chrono::milliseconds(100));
252
253 for(uint32_t i = 0; i < _laserLockSvr->_coarseTuneActuators.size(); i++) {
254 if(_laserLockSvr->_coarseTuneActuators[i]->isEnabled())
255 _logger(Loger::INFO) << __func__ << ": Disable actuator at index: " << i << std::endl;
256 _laserLockSvr->_coarseTuneActuators[i]->enable(false);
257 }
258
259 _logger(Loger::INFO) << __func__ << ": Shutdown all PZT4 enabled channels with ramping." << std::endl;
260 _pzt4Manager->shutdownPzt4Channels();
261}

269 for(uint32_t i = 0; i < _laserLockSvr->_coarseTuneActuators.size(); i++) {

271 }

274 _pzt4Manager->storePzt4State();

277void LaserLockSystemResetManager::restartSystem() {
278 _logger(Loger::INFO) << __func__ << ": Restart the system." << std::endl;
279
280 _logger(Loger::INFO) << __func__ << ": Re-enable PZT4 channels." << std::endl;
281 _pzt4Manager->restartPzt4Channels();
282
283 _logger(Loger::INFO) << __func__ << ": Re-enable slow actuators." << std::endl;
284 for(auto& index : _sysSetPzt4Reset.actuatorActiveVec) {
285 _laserLockSvr->_coarseTuneActuators[index]->enable(true);
286 std::this_thread::sleep_for(std::chrono::milliseconds(500));
287 }
288
289 if(_laserLockSvr->_ctrl1Manager && _sysSetPzt4Reset.ctrlAuxActive) {
290 _laserLockSvr->_ctrl1Manager->setEnable(true);
291 }
292
293 if(_sysSetPzt4Reset.coarseTuneEnabled) {
294 _logger(Loger::INFO) << __func__ << ": Re-enable coarse tuning." << std::endl;
295 _laserLockSvr->_mutliActCoarseTuneAlgorithm->setCoarseTuningEnabled(true);
296 std::this_thread::sleep_for(std::chrono::milliseconds(500));
297 }
298
299 if(!_sysSetPzt4Reset.unlockByUser) {
300 _laserLockSvr->laserServerCommands._validationEnableCheckTrigger = true;
301 _laserLockSvr->_mutliActCoarseTuneAlgorithm->triggerCommand(
302 CsaCoarseTuningMultiDetMultiActAlgorithm::CMD::CMD_UNPAUSE_AUTOLOCK_FEATURE);
303 }

306void LaserLockSystemResetManager::pzt4PowerReset() {
307 if(_resetSystemInProgress) {
308 _logger(Loger::INFO) << __func__ << ": System reset already in progress." << std::endl;
309 return;
310 }
311
312 if(_pzt4PowerResetInProgress) {
313 _logger(Loger::INFO) << __func__ << ": PZT4 RTM reset already in progress." << std::endl;
314 return;
315 }
316
317 _pzt4PowerResetInProgress = true;
318 // wait 1 sec for main loop to be frozen
319 std::this_thread::sleep_for(std::chrono::milliseconds(1000));
320
321 auto time_start = std::chrono::high_resolution_clock::now();
322 if(_pzt4Manager->getPzt4PowerResetsFailCnt() == 0) {
323 storeSystemState();
324 }
325 shutdownSystem();
326 // wait 1 sec for main loop to apply all shutdown commands
327 std::this_thread::sleep_for(std::chrono::milliseconds(1000));
328 _laserLockSvr->_bspPZT4->updateFromDevice();
329
330 if(_pzt4Manager->pzt4FullPowerReset()) restartSystem();
331
332 _logger(Loger::INFO) << __func__ << ": Reset took [ms]: "
333 << std::chrono::duration_cast<std::chrono::milliseconds>(
334 std::chrono::high_resolution_clock::now() - time_start)
335 .count()
336 << std::endl;
337 _pzt4PowerResetInProgress = false;
338}
