150void CsaLinkLockCtrlChannel::commandCb() {
151 auto cmd = _command->getValue();
152 switch(cmd) {
153 case CsaLsuOxcTuneSearchAlg::CMD::OXC_SIGNAL_SEARCH_AND_LOCK:
154 _validationEnableCheckTrigger = true;
155 _coarseTuneAlg->triggerCommand(cmd);
156 break;
157
158 case CsaLsuOxcTuneSearchAlg::CMD::OXC_SIGNAL_SEARCH:
159 _coarseTuneAlg->triggerCommand(cmd);
160 break;
161
162 case CsaLsuOxcTuneSearchAlg::CMD::OXC_SIGNAL_SEARCH_STOP:
163 _coarseTuneAlg->triggerCommand(cmd);
164 break;
165
166 case CsaLsuOxcTuneSearchAlg::CMD::START_TUNE_AND_LOCK:
167 _validationEnableCheckTrigger = true;
168 _coarseTuneAlg->triggerCommand(cmd);
169 break;
170
171 case CsaLsuOxcTuneSearchAlg::CMD::START_TUNE:
172 _coarseTuneAlg->triggerCommand(cmd);
173 break;
174
175 case CsaLsuOxcTuneSearchAlg::CMD::STOP_TUNE:
176 _coarseTuneAlg->triggerCommand(cmd);
177 break;
178
179 case CsaCoarseTuningSingleDetectorAlgorithm::CMD::OPEN_FAST_LOOP:
180 _lockStatusValidator->setUnlockByUser(true);
181 _coarseTuneAlg->triggerCommand(cmd);
182 break;
183
184 case CsaCoarseTuningSingleDetectorAlgorithm::CMD::CLOSE_FAST_LOOP:
185 _validationEnableCheckTrigger = true;
186 _coarseTuneAlg->triggerCommand(cmd);
187 break;
188
189 case CsaLsuOxcAdvAlg::CMD::OXC_LOCK_TO_ANOTHER_PULSE:
190 // add 'fall through' comment to suppress warning: this statement may fall through [-Wimplicit-fallthrough=] in
191 // GCC 7 (Ubuntu18) fall through
192 case CsaLsuOxcAdvAlg::CMD::OXC_LOCK_TO_PREVIOUS_PULSE:
193 // fall through
194 case CsaLsuOxcAdvAlg::CMD::OXC_LOCK_TO_NEXT_PULSE:
195 _lockStatusValidator->setUnlockByUser(true);
196 _validationEnableCheckTrigger = true;
197 _coarseTuneAlg->triggerCommand(cmd);
198 break;
199
200 default:
201 _coarseTuneAlg->triggerCommand(cmd);
202 break;
203 }

426void CsaLinkLockCtrlChannel::_calculateLinkDriftAndJitterThread() {
427 _calculateLinkDriftThreadActive = true;
428 if(!_calibAlgorithms->getFastActCalibInProgress() && !_calibAlgorithms->getOxcDetCalibInProgress() &&
429 !_calibAlgorithms->getOxcFastCalibrationInProgress()) {
430 double timeDriftOfCtrlOut = (_controller->getOutputValue() * _calibAlgorithms->getFs2CtrlOutCalibration()) /
431 1000; // this should be scaled in ps
432 double timeDriftMotor = (_remoteStepperMotor->getCurrentPositionInSteps() * _calibAlgorithms->getOdlFsToStepCc()) /
433 1000; // this should be scaled in ps
434 double timeDrift = timeDriftMotor - timeDriftOfCtrlOut;
435 _linkTimingDrift->setValue(timeDrift);
436 _linkTimingJitter->setValue(
437 std::fabs(_lsuCtrlInputStats->getStdDev() * _calibAlgorithms->getOxcFsToVCalibration()));
438
439 _lockStatusValidator->setCurrentLockJitter(_linkTimingJitter->getValue());
440
441 _logger(Loger::DETAIL) << __func__ << ": "
442 << "Timing drift[ps]: " << timeDrift << " , ctrl part: " << timeDriftOfCtrlOut
443 << " , motor part: " << timeDriftMotor
444 << ", timing jitter[fs]: " << _linkTimingJitter->getValue() << std::endl;
445 }
446 _calculateLinkDriftThreadActive = false;
447}
