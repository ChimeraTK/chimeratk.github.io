19 void ProcessInfoModule::mainLoop() {
20  info.processPID = getpid();
21  info.processPID.write();
22 
23  auto group = readAnyGroup();
24  while(true) {
25  try {
26  FillProcInfo(proc_util::getInfo(info.processPID));
27  logger->sendMessage(
28  std::string("Process is running (PID: ") + std::to_string(info.processPID) + ")", logging::LogLevel::DEBUG);
29  }
30  catch(std::runtime_error& e) {
31  logger->sendMessage(
32  std::string("Failed to read process information for process with PID: ") + std::to_string(info.processPID),
33  logging::LogLevel::ERROR);
34  }
35  writeAll();
36  group.readUntil(trigger.getId());
37  }

40 void ProcessInfoModule::FillProcInfo(const std::shared_ptr<proc_t>& infoPtr) {
41  if(infoPtr != nullptr) {
42  auto now = boost::posix_time::microsec_clock::local_time();
43  int old_time = 0;
44  try {
45  //\FixMe: Exception is thrown on server start -> find out why.
46  old_time = std::stoi(std::to_string(statistics.utime + statistics.stime + statistics.cutime + statistics.cstime));
47  statistics.utime = std::stoi(std::to_string(infoPtr->utime));
48  statistics.stime = std::stoi(std::to_string(infoPtr->stime));
49  statistics.cutime = std::stoi(std::to_string(infoPtr->cutime));
50  statistics.cstime = std::stoi(std::to_string(infoPtr->cstime));
51 
52  // info->start_time reads s since system was started
53  int relativeStartTime = 1. * std::stoi(std::to_string(infoPtr->start_time)) / system.info.ticksPerSecond;
54  statistics.startTime = system.status.sysStartTime + relativeStartTime;
55  statistics.startTimeStr =
56  boost::posix_time::to_simple_string(boost::posix_time::from_time_t(statistics.startTime));
57  statistics.priority = std::stoi(std::to_string(infoPtr->priority));
58  statistics.nice = std::stoi(std::to_string(infoPtr->nice));
59  statistics.rss = std::stoi(std::to_string(infoPtr->rss));
60  statistics.mem = std::stoi(std::to_string(infoPtr->vm_rss));
61 
62  statistics.memoryUsage = 1. * statistics.mem / system.status.maxMem * 100.;
63 
64  statistics.runtime = std::stoi(std::to_string(
65  system.status.sysUpTime - std::stoi(std::to_string(infoPtr->start_time)) * 1. / system.info.ticksPerSecond));
66  }
67  catch(std::exception& e) {
68  logger->sendMessage(std::string("FillProcInfo::Conversion failed: ") + e.what(), logging::LogLevel::ERROR);
69  }
70  // check if it is the first call after process is started (time_stamp == not_a_date_time)
71  if(!time_stamp.is_special()) {
72  boost::posix_time::time_duration diff = now - time_stamp;
73  statistics.pcpu = 1. * (statistics.utime + statistics.stime + statistics.cutime + statistics.cstime - old_time) /
74  system.info.ticksPerSecond / (1. * diff.total_milliseconds() / 1000) * 100;
75  statistics.avgcpu = 1. * (statistics.utime + statistics.stime + statistics.cutime + statistics.cstime) /
76  system.info.ticksPerSecond / statistics.runtime * 100;
77  }
78  time_stamp = now;
79  }
80  else {
81  time_stamp = boost::posix_time::not_a_date_time;
82  statistics.utime = 0;
83  statistics.stime = 0;
84  statistics.cutime = 0;
85  statistics.cstime = 0;
86  statistics.startTime = 0;
87  statistics.startTimeStr = "";
88  statistics.priority = 0;
89  statistics.nice = 0;
90  statistics.rss = 0;
91  statistics.pcpu = 0;
92  statistics.avgcpu = 0;
93  statistics.runtime = 0;
94  statistics.mem = 0;
95  statistics.memoryUsage = 0.;
96  }
97 }

99 void ProcessControlModule::mainLoop() {
100  std::stringstream handlerMessage;
101  logger->sendMessage(std::string("New ProcessModule started!"), logging::LogLevel::INFO);
102  SetOffline();
103  status.nRestarts = 0;
104 
105  try {
106  process.reset(new ProcessHandler(getName(), false, info.processPID, handlerMessage));
107  evaluateMessage(handlerMessage);
108  if(info.processPID > 0) {
109  logger->sendMessage(
110  std::string("Found process that is still running. PID is: ") + std::to_string(info.processPID),
111  logging::LogLevel::INFO);
112  SetOnline(info.processPID);
113  }
114  else {
115  if(config.bootDelay > 0) {
116  logger->sendMessage(std::string("Process sleeping before starting main loop. Delay: ") +
117  std::to_string(config.bootDelay) + "s.",
118  logging::LogLevel::INFO);
119  sleep(config.bootDelay);
120  }
121  }
122  }
123  catch(std::runtime_error& e) {
124  logger->sendMessage(
125  std::string("Failed to check for existing processes. Message:\n") + e.what(), logging::LogLevel::ERROR);
126  }
127  auto group = readAnyGroup();
128  status.nFailed = 0;
129  status.nRestarts = 0;
130  while(true) {
131  // reset number of failed tries and restarts in case the process is set offline
132  if(!enableProcess) {
133  status.nFailed = 0;
134  status.nRestarts = 0;
135  _stop = false;
136  _restartRequired = false;
137  }
138 
139  /*
140  * don't do anything in case failed more than the user set limit
141  * or if the server was restarted more than the user set limit
142  * -> to reset turn off/on the process
143  */
144  if(_stop) {
145  logger->sendMessage(std::string("Process sleeping. Fails: ") + std::to_string(status.nFailed) + "/" +
146  std::to_string(config.maxFails) + ", Restarts: " + std::to_string(status.nRestarts) + "/" +
147  std::to_string(config.maxRestarts),
148  logging::LogLevel::DEBUG);
149  if(_historyOn) FillProcInfo(nullptr);
150  writeAll();
151  group.readUntil(trigger.getId());
152  continue;
153  }
154 
158  if(info.processPID > 0 && enableProcess) {
159  CheckIsOnline(info.processPID);
160  if(!status.isRunning) {
161  if(config.maxRestarts == 0) {
162  _stop = true;
163  logger->sendMessage(std::string("Maximum number of restarts is 0. Process will not be started again."),
164  logging::LogLevel::ERROR);
165  _restartRequired = false;
166  }
167  else {
168  _restartRequired = true;
169  }
170  }
171  }
172 
176  if(config.maxRestarts == 0 && _restartRequired) {
177  _restartRequired = false;
178  _stop = true;
179  }
180 
184  if(config.maxRestarts != 0 && status.nRestarts == config.maxRestarts) {
185  logger->sendMessage(std::string("Maximum number of restarts reached. Restarts: ") +
186  std::to_string(status.nRestarts) + "/" + std::to_string(config.maxRestarts),
187  logging::LogLevel::DEBUG);
188  // Only stop if the process terminated. This ensures that the process status is updated.
189  if(status.isRunning == 0) {
190  _stop = true;
191  logger->sendMessage(std::string("Process terminated after maximum number of restarts reached. Restarts: ") +
192  std::to_string(status.nRestarts) + "/" + std::to_string(config.maxRestarts),
193  logging::LogLevel::ERROR);
194  resetProcessHandler(&handlerMessage);
195  }
196  else {
197  _restartRequired = false;
198  }
199  }
200 
201  if(enableProcess) {
202  // process should run
203  if(info.processPID < 0 && !_stop && (status.nRestarts < config.maxRestarts || config.maxRestarts == 0)) {
204  // process should run and is not running
205  if(_restartRequired) {
206  status.nRestarts += 1;
207  _restartRequired = false;
208  }
209  // fill 0 since the process is started here and not running yet
210  if(_historyOn) FillProcInfo(nullptr);
211  try {
212  logger->sendMessage(
213  std::string("Trying to start a new process: ") + (std::string)config.path + "/" + (std::string)config.cmd,
214  logging::LogLevel::INFO);
215  // log level of the process handler is DEBUG per default. So all messages will end up here
216  process.reset(new ProcessHandler(getName(), false, handlerMessage, this->getName()));
217 
218  SetOnline(process->startProcess((std::string)config.path, (std::string)config.cmd,
219  (std::string)config.externalLogfile, (std::string)config.env, config.overwriteEnv));
220  evaluateMessage(handlerMessage);
221  status.nChilds = proc_util::getNChilds(info.processPID, handlerMessage);
222  logger->sendMessage(handlerMessage.str(), logging::LogLevel::DEBUG);
223  }
224  catch(std::runtime_error& e) {
225  logger->sendMessage(e.what(), logging::LogLevel::ERROR);
226  Failed();
227  SetOffline();
228  }
229  }
230  else if(info.processPID > 0) {
231  // process should run and is running
232  logger->sendMessage(std::string("Process is running...") + std::to_string(status.isRunning) +
233  " PID: " + std::to_string(info.processPID),
234  logging::LogLevel::DEBUG);
235  try {
236  FillProcInfo(proc_util::getInfo(info.processPID + config.pidOffset));
237  }
238  catch(std::runtime_error& e) {
239  logger->sendMessage(std::string("Failed to read information for process ") +
240  std::to_string(info.processPID + config.pidOffset) + ". Check if pidOffset is set correctly!",
241  logging::LogLevel::ERROR);
242  }
243  }
244  }
245  else {
246  // process should not run
247  if(info.processPID < 0) {
248  // process should not run and is not running
249  status.isRunning = 0;
250  logger->sendMessage(
251  std::string("Process Running: ") + std::to_string(status.isRunning) + ". Process is not running...OK",
252  logging::LogLevel::DEBUG);
253  if(_historyOn) FillProcInfo(nullptr);
254  }
255  else {
256  // process should not run and is running
257  logger->sendMessage(std::string("Trying to kill the process... PID: ") + std::to_string(info.processPID),
258  logging::LogLevel::INFO);
259  // Here the process is stopped in case enableProcess is set to 0. If it is already reset due to restart stop
260  // don't do anything here
261  if(process.get() != nullptr) {
262  resetProcessHandler(&handlerMessage);
263  }
264  SetOffline();
265  }
266  }
267  writeAll();
268  group.readUntil(trigger.getId());
269  }
270 }

272 void ProcessControlModule::SetOnline(const int& pid) {
273  usleep(100000);
274  // set external log file in order to read the log file even if starting the process failed
275  status.externalLogfile = (std::string)config.externalLogfile;
276  CheckIsOnline(pid);
277  if(status.isRunning == 1) {
278  info.processPID = pid;
279  status.path = (std::string)config.path;
280  status.cmd = (std::string)config.cmd;
281  status.env = (std::string)config.env;
282  logger->sendMessage(std::string("Ok process is started successfully with PID: ") + std::to_string(info.processPID),
283  logging::LogLevel::INFO);
284  }
285  else {
286  SetOffline();
287  logger->sendMessage(
288  std::string("Failed to start process ") + config.path + "/" + config.cmd, logging::LogLevel::ERROR);
289  Failed();
290  }
291 }

307 void ProcessControlModule::Failed() {
308  status.nFailed = status.nFailed + 1;
309  if(!_stop && status.nFailed == config.maxFails) {
310  _stop = true;
311  _restartRequired = false;
312  logger->sendMessage(std::string("Failed to start the process ") + config.path + "/" + (std::string)config.cmd +
313  " " + std::to_string(config.maxFails) +
314  " times. It will not be started again until you reset the process by switching it off and on again.",
315  logging::LogLevel::ERROR);
316  }
317  else {
318  _restartRequired = true;
319  }
