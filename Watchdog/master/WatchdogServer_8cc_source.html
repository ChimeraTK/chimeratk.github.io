19 /*
20  * Select all mount points that mount a physical hard drive on the local machine.
21  * The boot partition is not selected.
22  * Only devices that include /dev/sd are considered.
23  */
24 std::set<std::pair<std::string, std::string>> findMountPoints() {
25  std::set<std::pair<std::string, std::string>> out;
26  std::ifstream in("/proc/mounts");
27  std::string strin[4];
28  int iin[2];
29  while(in.good()) {
30  in >> strin[0] >> strin[1] >> strin[2] >> strin[3] >> iin[0] >> iin[1];
31  bfs::path p(strin[0]);
32  if(strin[0].substr(0, 7).compare("/dev/sd") == 0) {
33  if(strin[1].substr(0, 6).compare("/boot/") != 0) {
34  out.insert(std::make_pair(p.filename().string(), strin[1]));
35  }
36  }
37  }
38  in.close();
39  return out;
40 }

42 /*
43  * Select all non virtual network adapter.
44  */
45 std::set<std::string> findNetworkDevices() {
46  std::set<std::string> out;
47  bfs::path p("/sys/class/net");
48  try {
49  if(exists(p)) { // does p actually exist?
50  if(is_directory(p)) { // is p a directory?
51  for(auto i = bfs::directory_iterator(p); i != bfs::directory_iterator(); i++) {
52  if(bfs::canonical(i->path()).string().find("virtual") == std::string::npos)
53  out.insert(i->path().filename().string());
54  }
55  }
56  }
57  }
58  catch(const bfs::filesystem_error& ex) {
59  std::cout << ex.what() << '\n';
60  }
61  return out;
62 }
