51 void SystemInfoModule::mainLoop() {
52  // Set variables that are read by other modules here
53  double uptime_secs;
54  double idle_secs;
55  // read the system start time (uncertainty in the order of seconds...)
56  uptime(&uptime_secs, &idle_secs);
57  auto now = boost::posix_time::second_clock::local_time();
58  status.startTime = boost::posix_time::to_time_t(now) - std::stoi(std::to_string(uptime_secs));
59  status.startTimeStr = boost::posix_time::to_simple_string(boost::posix_time::from_time_t(status.startTime));
60  status.startTime.write();
61  status.startTimeStr.write();
62  status.uptime_secTotal = std::stoi(std::to_string(uptime_secs));
63  status.uptime_secTotal.write();
64 
65  meminfo();
66  try {
67  status.maxMem = std::stoi(std::to_string(kb_main_total));
68  status.maxMem.write();
69  } catch(std::exception &e) {
70  (*logStream) << getTime(this) << "meminfo conversion failed: " << e.what() << std::endl;
71 #ifdef ENABLE_LOGGING
72  sendMessage(logging::LogLevel::ERROR);
73 #endif
74  }
75 
76  info.ticksPerSecond = sysconf(_SC_CLK_TCK);
77  info.ticksPerSecond.write();
78 #ifdef ENABLE_LOGGING
79  logging.message.write();
80  logging.messageLevel.write();
81 #endif
82  for(auto it = sysInfo.ibegin(); it != sysInfo.iend(); it++) {
83  info.strInfos.at(it->first) = it->second;
84  }
85  info.nCPU = sysInfo.getNCpu();
86  info.writeAll();
87 
88  if(lastInfo.size() != (unsigned) (info.nCPU + 1)) {
89  (*logStream) << getName() << "Size of lastInfo: " << lastInfo.size()
90  << "\t nCPU: " << info.nCPU << std::endl;
91 #ifdef ENABLE_LOGGING
92  sendMessage(logging::LogLevel::ERROR);
93 #endif
94  throw std::runtime_error(
95  "Vector size mismatch in SystemInfoModule::mainLoop.");
96  }
97 
98  std::ifstream file("/proc/stat");
99  if(!file.is_open()) {
100  (*logStream) << getTime(this) << "Failed to open system file /proc/stat"
101  << std::endl;
102 #ifdef ENABLE_LOGGING
103  sendMessage(logging::LogLevel::ERROR);
104 #endif
105  file.close();
106  return;
107  }
108 
109  readCPUInfo(lastInfo);
110 
111  // findTag takes some time and should not be called with every loop
112  auto toWrite = findTag("SYS");
113  // ToDo: Remove once the bug in the VitualModule is fixed (issue #30)
114  toWrite.setOwner(this);
115 
116  while(true) {
117  meminfo();
118  try {
119  status.maxMem = std::stoi(std::to_string(kb_main_total));
120  status.freeMem = std::stoi(std::to_string(kb_main_free));
121  status.cachedMem = std::stoi(std::to_string(kb_main_cached));
122  status.usedMem = std::stoi(std::to_string(status.maxMem - status.freeMem - status.cachedMem));
123  status.maxSwap = std::stoi(std::to_string(kb_swap_total));
124  status.freeSwap = std::stoi(std::to_string(kb_swap_free));
125  status.usedSwap = std::stoi(std::to_string(status.maxSwap - status.freeSwap));
126  } catch(std::exception &e) {
127  (*logStream) << getTime(this) << "meminfo conversion failed: " << e.what() << std::endl;
128 #ifdef ENABLE_LOGGING
129  sendMessage(logging::LogLevel::ERROR);
130 #endif
131  }
132  status.memoryUsage = 1.*status.usedMem/status.maxMem*100.;
133  status.swapUsage = 1.*status.usedSwap/status.maxSwap*100.;
134 
135  // get system uptime
136  try {
137  uptime(&uptime_secs, &idle_secs);
138  //\ToDo: Use stoul once unsigned long is available
139  status.uptime_secTotal = std::stoi(std::to_string(uptime_secs));
140  status.uptime_day = std::stoi(std::to_string(status.uptime_secTotal / 86400));
141  status.uptime_hour = std::stoi(
142  std::to_string((status.uptime_secTotal - (status.uptime_day * 86400)) / 3600));
143  status.uptime_min = std::stoi(std::to_string(
144  (status.uptime_secTotal - (status.uptime_day * 86400) - (status.uptime_hour * 3600))
145  / 60));
146  status.uptime_sec = std::stoi(std::to_string(
147  status.uptime_secTotal - (status.uptime_day * 86400) - (status.uptime_hour * 3600) - (status.uptime_min*60)));
148  } catch(std::exception &e) {
149  (*logStream) << getTime(this) << "uptime conversion failed: " << e.what() << std::endl;
150 #ifdef ENABLE_LOGGING
151  sendMessage(logging::LogLevel::ERROR);
152 #endif
153  }
154 
155  std::vector<double> v_tmp(3);
156 
157  loadavg(&v_tmp[0], &v_tmp[1], &v_tmp[2]);
158  status.loadAvg = v_tmp;
159 
160  calculatePCPU();
161 
162  toWrite.writeAll();
163 #ifdef ENABLE_LOGGING
164  (*logStream) << getTime(this) << "System data updated" << std::endl;
165  sendMessage(logging::LogLevel::DEBUG);
166 #endif
167  triggerGroup.trigger.read();
168  }

171 void SystemInfoModule::calculatePCPU() {
172  unsigned long long total;
173  double tmp;
174  std::vector<double> usage_tmp(info.nCPU + 1);
175  std::vector<std::string> strs;
176  std::vector<cpu> vcpu(info.nCPU + 1);
177  readCPUInfo(vcpu);
178  auto lastcpu = lastInfo.begin();
179  auto newcpu = vcpu.begin();
180  for(size_t iCPU = 0; iCPU < (info.nCPU + 1); iCPU++) {
181  if(newcpu->totalUser < lastcpu->totalUser
182  || newcpu->totalUserLow < lastcpu->totalUserLow
183  || newcpu->totalSys < lastcpu->totalSys
184  || newcpu->totalIdle < lastcpu->totalIdle) {
185  //Overflow detection. Just skip this value.
186  usage_tmp.at(iCPU) =(-1.0);
187  } else {
188  total = (newcpu->totalUser - lastcpu->totalUser)
189  + (newcpu->totalUserLow - lastcpu->totalUserLow)
190  + (newcpu->totalSys - lastcpu->totalSys);
191  tmp = total;
192  total += (newcpu->totalIdle - lastcpu->totalIdle);
193  tmp /= total;
194  tmp *= 100.;
195  usage_tmp.at(iCPU) = tmp;
196 
197  }
198  *lastcpu = *newcpu;
199  lastcpu++;
200  newcpu++;
201  }
202  status.cpu_useTotal = usage_tmp.back();
203  usage_tmp.pop_back();
204  status.cpu_use->operator =(usage_tmp);
205 }

207 void SystemInfoModule::readCPUInfo(std::vector<cpu> &vcpu) {
208  std::ifstream file("/proc/stat");
209  if(!file.is_open()) {
210  (*logStream) << getTime(this) << "Failed to open system file /proc/stat"
211  << std::endl;
212 #ifdef ENABLE_LOGGING
213  sendMessage(logging::LogLevel::ERROR);
214 #endif
215  file.close();
216  return;
217  }
218  std::string::size_type sz = 0;
219  std::string line;
220  for(auto it = vcpu.begin(); it != vcpu.end(); it++) {
221  if(!std::getline(file, line)) {
222  (*logStream) << getTime(this) << "Could not find enough lines in /proc/stat"
223  << std::endl;
224 #ifdef ENABLE_LOGGING
225  sendMessage(logging::LogLevel::ERROR);
226 #endif
227  file.close();
228  return;
229  }
230  auto strs = split_arguments(line);
231  if(strs.size() < 5) {
232  (*logStream) << getTime(this) << "Line seems to contain not enough data. Line: "
233  << line << std::endl;
234  }
235 
236  try {
237  it->totalUser = std::stoull(strs.at(1), &sz, 0);
238  it->totalUserLow = std::stoull(strs.at(2), &sz, 0);
239  it->totalSys = std::stoull(strs.at(3), &sz, 0);
240  it->totalIdle = std::stoull(strs.at(4), &sz, 0);
241  } catch(std::exception &e) {
242  (*logStream) << getTime(this) << "Caught an exception: " << e.what()
243  << std::endl;
244 #ifdef ENABLE_LOGGING
245  sendMessage(logging::LogLevel::ERROR);
246 #endif
247  for(auto s : strs) {
248  (*logStream) << getTime(this) << "String token: " << s << std::endl;
249 #ifdef ENABLE_LOGGING
250  sendMessage(logging::LogLevel::ERROR);
251 #endif
252  }
253  }
254  }

409 bool NetworkModule::read(){
410  raw tmp((std::string)deviceName);
411  for(size_t i = 0; i < tmp.files.size(); i++){
412  std::ifstream in(tmp.files.at(i));
413  if(!in.is_open()){
414  (*logStream) << getTime(this) << "Failed to open: " << tmp.files.at(i) << std::endl;
415 #ifdef ENABLE_LOGGING
416  sendMessage(logging::LogLevel::ERROR);
417 #endif
418  return false;
419  }
420  tmp.time.at(i) = boost::posix_time::microsec_clock::local_time();
421  in >> tmp.data.at(i);
422  // check if this is the first reading and no data is stored yet in previousData
423  if(previousData.files.size() != 0){
424  boost::posix_time::time_duration diff = tmp.time.at(i) - previousData.time.at(i);
425  status.data.at(i) = 1.*(tmp.data.at(i) - previousData.data.at(i))/1024/1024/(diff.total_nanoseconds()/1e9);
426  }
427  in.close();
428  }
429  previousData = raw(tmp);
430  return true;
431 }
