15 void ProcessInfoModule::mainLoop() {
16  info.processPID = getpid();
17  info.processPID.write();
18 #ifdef ENABLE_LOGGING
19  logStream = new std::stringstream();
20 #else
21  logStream = &std::cerr;
22 #endif
23 
24  auto group = readAnyGroup();
25  while(true) {
26  try {
27  FillProcInfo(proc_util::getInfo(info.processPID));

32  }

35 #ifdef ENABLE_LOGGING

39  writeAll();
40  group.readUntil(triggerGroup.trigger.getId());

44 void ProcessInfoModule::FillProcInfo(const std::shared_ptr<proc_t>& infoPtr) {
45  if(infoPtr != nullptr) {
46  auto now = boost::posix_time::microsec_clock::local_time();
47  int old_time = 0;
48  try {
49  //\FixMe: Exception is thrown on server start -> find out why.
50  old_time = std::stoi(std::to_string(statistics.utime + statistics.stime + statistics.cutime + statistics.cstime));
51  statistics.utime = std::stoi(std::to_string(infoPtr->utime));
52  statistics.stime = std::stoi(std::to_string(infoPtr->stime));
53  statistics.cutime = std::stoi(std::to_string(infoPtr->cutime));
54  statistics.cstime = std::stoi(std::to_string(infoPtr->cstime));
55 
56  // info->start_time reads s since system was started
57  int relativeStartTime = 1. * std::stoi(std::to_string(infoPtr->start_time)) / input.ticksPerSecond;
58  statistics.startTime = input.sysStartTime + relativeStartTime;
59  statistics.startTimeStr =
60  boost::posix_time::to_simple_string(boost::posix_time::from_time_t(statistics.startTime));
61  statistics.priority = std::stoi(std::to_string(infoPtr->priority));
62  statistics.nice = std::stoi(std::to_string(infoPtr->nice));
63  statistics.rss = std::stoi(std::to_string(infoPtr->rss));
64  statistics.mem = std::stoi(std::to_string(infoPtr->vm_rss));
65 
66  statistics.memoryUsage = 1. * statistics.mem / input.maxMem * 100.;
67 
68  statistics.runtime = std::stoi(
69  std::to_string(input.sysUpTime - std::stoi(std::to_string(infoPtr->start_time)) * 1. / input.ticksPerSecond));
70  }
71  catch(std::exception& e) {
72  (*logStream) << getTime() << "FillProcInfo::Conversion failed: " << e.what() << std::endl;
73 #ifdef ENABLE_LOGGING
74  sendMessage(logging::LogLevel::ERROR);
75 #endif
76  }
77  // check if it is the first call after process is started (time_stamp == not_a_date_time)
78  if(!time_stamp.is_special()) {
79  boost::posix_time::time_duration diff = now - time_stamp;
80  statistics.pcpu = 1. * (statistics.utime + statistics.stime + statistics.cutime + statistics.cstime - old_time) /
81  input.ticksPerSecond / (1. * diff.total_milliseconds() / 1000) * 100;
82  statistics.avgcpu = 1. * (statistics.utime + statistics.stime + statistics.cutime + statistics.cstime) /
83  input.ticksPerSecond / statistics.runtime * 100;
84  }
85  time_stamp = now;
86  }
87  else {
88  time_stamp = boost::posix_time::not_a_date_time;
89  statistics.utime = 0;
90  statistics.stime = 0;
91  statistics.cutime = 0;
92  statistics.cstime = 0;
93  statistics.startTime = 0;
94  statistics.startTimeStr = "";
95  statistics.priority = 0;
96  statistics.nice = 0;
97  statistics.rss = 0;
98  statistics.pcpu = 0;
99  statistics.avgcpu = 0;
100  statistics.runtime = 0;
101  statistics.mem = 0;
102  statistics.memoryUsage = 0.;
103  }
104 }

116 void ProcessControlModule::mainLoop() {
117 #ifdef ENABLE_LOGGING
118  logStream = new std::stringstream();
119  std::stringstream handlerMessage;
120 #else
121  logStream = &std::cerr;
122 #endif
123  (*logStream) << getTime() << "New ProcessModule started!" << std::endl;
124 #ifdef ENABLE_LOGGING
125  sendMessage(logging::LogLevel::INFO);
126 #endif
127  SetOffline();
128  status.nRestarts = 0;
129 
130  try {
131 #ifdef ENABLE_LOGGING
132  process.reset(new ProcessHandler(getName(), false, info.processPID, handlerMessage));
133  evaluateMessage(handlerMessage);
134 #else
135  process.reset(new ProcessHandler(getName(), false, info.processPID, std::cout));
136 #endif
137  if(info.processPID > 0) {
138  (*logStream) << getTime() << "Found process that is still running. PID is: " << info.processPID << std::endl;
139 #ifdef ENABLE_LOGGING
140  sendMessage(logging::LogLevel::INFO);
141 #endif
142  SetOnline(info.processPID);
143  }
144  else {
145  if(config.bootDelay > 0) {
146  (*logStream) << getTime() << "Process sleeping before starting main loop. Delay: " << config.bootDelay << "s."
147  << std::endl;
148 #ifdef ENABLE_LOGGING
149  sendMessage(logging::LogLevel::INFO);
150 #endif
151  sleep(config.bootDelay);
152  }
153  }
154  }
155  catch(std::runtime_error& e) {
156  (*logStream) << getTime() << " Failed to check for existing processes. Message:\n" << e.what() << std::endl;
157 #ifdef ENABLE_LOGGING
158  sendMessage(logging::LogLevel::ERROR);
159 #endif
160  }
161  auto group = readAnyGroup();
162  status.nFailed = 0;
163  status.nRestarts = 0;
164  while(true) {
165  // reset number of failed tries and restarts in case the process is set offline
166  if(!enableProcess) {
167  status.nFailed = 0;
168  status.nRestarts = 0;
169  _stop = false;
170  _restartRequired = false;
171  }
172 
173  /*
174  * don't do anything in case failed more than the user set limit
175  * or if the server was restarted more than the user set limit
176  * -> to reset turn off/on the process
177  */
178  if(_stop) {
179  (*logStream) << getTime() << "Process sleeping. Fails: " << status.nFailed << "/" << config.maxFails
180  << ", Restarts: " << status.nRestarts << "/" << config.maxRestarts << std::endl;
181 #ifdef ENABLE_LOGGING
182  sendMessage(logging::LogLevel::DEBUG);
183 #endif
184  if(_historyOn) FillProcInfo(nullptr);
185  writeAll();
186  group.readUntil(triggerGroup.trigger.getId());
187  continue;
188  }
189 
193  if(info.processPID > 0 && enableProcess) {
194  CheckIsOnline(info.processPID);
195  if(!status.isRunning) {
196  if(config.maxRestarts == 0) {
197  _stop = true;
198  (*logStream) << getTime() << "Maximum number of restarts is 0. Process will not be started again."
199  << std::endl;
200 #ifdef ENABLE_LOGGING
201  sendMessage(logging::LogLevel::WARNING);
202 #endif
203  _restartRequired = false;
204  }
205  else {
206  _restartRequired = true;
207  }
208  }
209  }
210 
214  if(config.maxRestarts == 0 && _restartRequired) {
215  _restartRequired = false;
216  _stop = true;
217  }
218 
222  if(config.maxRestarts != 0 && status.nRestarts == config.maxRestarts) {
223  (*logStream) << getTime() << "Maximum number of restarts reached. Restarts: " << status.nRestarts << "/"
224  << config.maxRestarts << std::endl;
225 #ifdef ENABLE_LOGGING
226  sendMessage(logging::LogLevel::DEBUG);
227 #endif
228  // Only stop if the process terminated. This ensures that the process status is updated.
229  if(status.isRunning == 0) {
230  _stop = true;
231  (*logStream) << getTime()
232  << "Process terminated after maximum number of restarts reached. Restarts: " << status.nRestarts
233  << "/" << config.maxRestarts << std::endl;
234 #ifdef ENABLE_LOGGING
235  sendMessage(logging::LogLevel::ERROR);
236  resetProcessHandler(&handlerMessage);
237 #else
238  resetProcessHandler(nullptr);
239 #endif
240  }
241  else {
242  _restartRequired = false;
243  }
244  }
245 
246  if(enableProcess) {
247  // process should run
248  if(info.processPID < 0 && !_stop && (status.nRestarts < config.maxRestarts || config.maxRestarts == 0)) {
249  // process should run and is not running
250  if(_restartRequired) {
251  status.nRestarts += 1;
252  _restartRequired = false;
253  }
254  // fill 0 since the process is started here and not running yet
255  if(_historyOn) FillProcInfo(nullptr);
256  try {
257 #ifdef ENABLE_LOGGING
258  (*logStream) << getTime() << "Trying to start a new process: " << (std::string)config.path << "/"
259  << (std::string)config.cmd << std::endl;
260  sendMessage(logging::LogLevel::INFO);
261  // log level of the process handler is DEBUG per default. So all messages will end up here
262  process.reset(new ProcessHandler(getName(), false, handlerMessage, this->getName()));
263 #else
264  process.reset(new ProcessHandler(getName(), false, std::cout, this->getName()));
265 #endif
266 
267 #ifdef ENABLE_LOGGING
268  SetOnline(process->startProcess((std::string)config.path, (std::string)config.cmd,
269  (std::string)config.externalLogfile, (std::string)config.env, config.overwriteEnv));
270  evaluateMessage(handlerMessage);
271  status.nChilds = proc_util::getNChilds(info.processPID, handlerMessage);
272  sendMessage(logging::LogLevel::DEBUG);
273 #else
274  SetOnline(process->startProcess((std::string)config.path, (std::string)config.cmd, std::string(""),
275  (std::string)config.env, config.overwriteEnv));
276  status.nChilds = proc_util::getNChilds(info.processPID);
277 #endif
278  }
279  catch(std::runtime_error& e) {
280  (*logStream) << getTime() << e.what() << std::endl;
281 #ifdef ENABLE_LOGGING
282  sendMessage(logging::LogLevel::ERROR);
283 #endif
284  Failed();
285  SetOffline();
286  }
287  }
288  else if(info.processPID > 0) {
289  // process should run and is running
290 #ifdef ENABLE_LOGGING
291  (*logStream) << getTime() << "Process is running..." << status.isRunning << " PID: " << info.processPID
292  << std::endl;
293  sendMessage(logging::LogLevel::DEBUG);
294 #endif
295  try {
296  FillProcInfo(proc_util::getInfo(info.processPID + config.pidOffset));
297  }
298  catch(std::runtime_error& e) {
299  (*logStream) << getTime() << "Failed to read information for process " << (info.processPID + config.pidOffset)
300  << ". Check if pidOffset is set correctly!" << std::endl;
301 #ifdef ENABLE_LOGGING
302  sendMessage(logging::LogLevel::ERROR);
303 #endif
304  }
305  }
306  }
307  else {
308  // process should not run
309  if(info.processPID < 0) {
310  // process should not run and is not running
311  status.isRunning = 0;
312 #ifdef ENABLE_LOGGING
313  (*logStream) << getTime() << "Process Running: " << status.isRunning << ". Process is not running...OK"
314  << std::endl;
315  sendMessage(logging::LogLevel::DEBUG);
316 #endif
317  if(_historyOn) FillProcInfo(nullptr);
318  }
319  else {
320  // process should not run and is running
321 #ifdef ENABLE_LOGGING
322  (*logStream) << getTime() << "Trying to kill the process..."
323  << " PID: " << info.processPID << std::endl;
324  sendMessage(logging::LogLevel::INFO);
325 #endif
326  // Here the process is stopped in case enableProcess is set to 0. If it is already reset due to restart stop don't do anything here
327  if(process.get() != nullptr) {
328 #ifdef ENABLE_LOGGING
329  resetProcessHandler(&handlerMessage);
330 #else
331  resetProcessHandler(nullptr);
332 #endif
333  }
334  SetOffline();
335  }
336  }
337  writeAll();
338  group.readUntil(triggerGroup.trigger.getId());
339  }
340 }

342 void ProcessControlModule::SetOnline(const int& pid) {
343  usleep(100000);
344 #ifdef ENABLE_LOGGING
345  // set external log file in order to read the log file even if starting the process failed
346  status.externalLogfile = (std::string)config.externalLogfile;
347 #endif
348  CheckIsOnline(pid);
349  if(status.isRunning == 1) {
350  info.processPID = pid;
351  status.path = (std::string)config.path;
352  status.cmd = (std::string)config.cmd;
353  status.env = (std::string)config.env;
354 #ifdef ENABLE_LOGGING
355  (*logStream) << getTime() << "Ok process is started successfully with PID: " << info.processPID << std::endl;
356  sendMessage(logging::LogLevel::INFO);
357 #endif
358  }
359  else {
360  SetOffline();
361  (*logStream) << getTime() << "Failed to start process " << (std::string)config.path << "/"
362  << (std::string)config.cmd << std::endl;
363 #ifdef ENABLE_LOGGING
364  sendMessage(logging::LogLevel::ERROR);
365 #endif
366  Failed();
367  }
