39  void formatLogTail(std::istream& data, std::ostream& os, size_t numberOfLines) {

48

54  line++;

65  std::vector<Message> stripMessages(std::stringstream& msg, const size_t maxCharacters) {
66  std::vector<Message> messages;
67  char* s = new char[maxCharacters];
68  while(msg.good()) {
69  Message singleMsg;
70  msg.getline(s, maxCharacters);
71  if(std::string(s).empty()) continue;
72  if(msg.eof()) {
73  break;
74  }
75  if(msg.fail()) {
76  singleMsg.message << s << "|\n The above line was cut by logger!" << std::endl;
77  }
78  else {
79  singleMsg.message << s;
80  }
81 
82  std::vector<LogLevel> levelNames = {LogLevel::INFO, LogLevel::WARNING, LogLevel::ERROR, LogLevel::DEBUG};
83  for(auto it = levelNames.begin(); it != levelNames.end(); ++it) {
84  std::stringstream ss;
85  ss << (*it);
86  size_t pos = singleMsg.message.str().find(ss.str());
87  if(pos != std::string::npos) {
88  singleMsg.logLevel = (*it);
89  singleMsg.message.str(
90  singleMsg.message.str().substr(pos + ss.str().length() - 1, singleMsg.message.str().length()));
91  break;
92  }
93  }
94 
95  messages.push_back(std::move(singleMsg));
96  }
97  delete[] s;
98  return messages;
99  }
