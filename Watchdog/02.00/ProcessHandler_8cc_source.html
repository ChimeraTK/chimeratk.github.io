36 void ProcessHandler::setupHandler() {
37  // see http://www.microhowto.info/howto/reap_zombie_processes_using_a_sigchld_handler.html
38  struct sigaction sa;
39  sa.sa_handler = &handle_sigchld;
40  sigemptyset(&sa.sa_mask);

70 void ProcessHandler::cleanup() {
71  if(pid > 0 && proc_util::isProcessRunning(pid)) {
72  if(log == logging::LogLevel::DEBUG) {
73  os << logging::LogLevel::DEBUG << name << logging::getTime() << "Going to kill (" << signum
74  << ") process in the destructor of ProcessHandler for process: " << pid << std::endl;
75  }
76  kill(-pid, signum);
77  // allow 1s for terminating the process, else it will be killed
78  // \ToDo: Verify that 1s is ok...
79  bool running = true;
80  if(killTimeout < 1) killTimeout = 1;
81  if(log == logging::LogLevel::DEBUG) {
82  os << logging::LogLevel::DEBUG << name << logging::getTime() << "Waiting for the process to exit (no longer than "
83  << killTimeout << "s)." << std::endl;
84  }
85  for(size_t i = 0; i < killTimeout; i++) {
86  sleep(1);
87  if(!proc_util::isProcessRunning(pid)) {
88  running = false;
89  break;
90  }
91  }
92 
93  if(running) {
94  if(log == logging::LogLevel::DEBUG) {
95  os << logging::LogLevel::DEBUG << name << logging::getTime()
96  << "Going to kill (SIGKILL) process in the destructor of ProcessHandler for process: " << pid << std::endl;
97  }
98  kill(-pid, SIGKILL);
99  usleep(200000);
100  if(proc_util::isProcessRunning(pid)) {
101  os << logging::LogLevel::ERROR << name << logging::getTime()
102  << "When cleaning up the ProcessHandler the process " << pid
103  << " could not be stopped. Even using signal SIGKILL!" << std::endl;
104  }
105  else {
106  os << logging::LogLevel::INFO << name << logging::getTime() << "Ok process was terminated." << std::endl;
107  }
108  }
109  else {
110  if(log == logging::LogLevel::DEBUG) {
111  os << logging::LogLevel::DEBUG << name << logging::getTime() << "Process exited normally." << std::endl;
112  }
113  }
114  }
115  else if(pid > 0 && log == logging::LogLevel::DEBUG) {
116  os << logging::LogLevel::DEBUG << name << logging::getTime()
117  << "Destructor called for the handler. Seems like process with PID: " << pid << " died (no attemp to kill it)."
118  << std::endl;
119  }
120  if(!deletePIDFile) remove(pidFile.c_str());
121 }

132 size_t ProcessHandler::startProcess(const std::string& path, const std::string& cmd, const std::string& logfile,
133  const std::string& environment, const bool& overwriteENV) {
134  if(path.empty() || cmd.empty()) {
135  throw std::runtime_error("Path or command not set before starting a process!");
136  }
137  // process could be stopped even if it was present when the ProcessHandler was constructed.
138  if(pid > 0 && proc_util::isProcessRunning(pid)) {
139  if(log <= logging::LogLevel::ERROR) {
140  os << logging::LogLevel::ERROR << name << logging::getTime()
141  << "There is still a process running that was not cleaned up! I will do a cleanup now." << std::endl;
142  }
143  cleanup();
144  }
145 
146  if(!isPIDFolderWritable()) {
147  std::stringstream ss;
148  ss << "Failed to change to the directory where to create the PID file (/tmp).";
149  throw std::runtime_error(ss.str());
150  }
151  // empty streams before forking to have empty copies in the child.
152  std::cout.clear();
153  std::cerr.clear();
154  std::cout.flush();
155  std::cerr.flush();
156 
157  pid_t p = fork();
158  if(p == 0) {
159  if(logfile.empty()) {
160  if(log <= logging::LogLevel::WARNING)
161  std::cout << logging::LogLevel::WARNING << name << logging::getTime()
162  << "No log file name is set. Process output is dumped to stout/stderr." << std::endl;
163  }
164  else {
165  // open the logfile
166  int fd =
167  open(logfile.c_str(), O_RDWR | O_CREAT | O_APPEND, S_IRUSR | S_IWUSR | S_IRGRP | S_IWGRP | S_IROTH | S_IWOTH);
168  if(fd == -1 && log <= logging::LogLevel::ERROR) {
169  std::cerr << logging::LogLevel::ERROR << name << logging::getTime()
170  << "Failed to open log file. No logfile will be written." << std::endl;
171  }
172  else {
173  dup2(fd, 1); // make stdout go to file
174  dup2(fd, 2); // make stderr go to file
175  close(fd);
176  }
177  }
178  std::cout << logging::LogLevel::INFO << name << logging::getTime() << "Going to start a new process." << std::endl;
179  // Don't throw in the child since the parent will not catch it
180  pid_t child = (int)getpid();
181  if(setpgid(0, child) && log <= logging::LogLevel::ERROR) {
182  std::cerr << logging::LogLevel::ERROR << name << logging::getTime() << "Failed to reset GPID." << std::endl;
183  }
184  if(log == logging::LogLevel::DEBUG) {
185  std::cout << logging::LogLevel::DEBUG << name << logging::getTime() << "Child running and its PID is: " << child
186  << std::endl;
187  }
188  std::ofstream file;
189  file.open(pidFile);
190  if(!file.is_open()) {
191  file.close();
192  if(log <= logging::LogLevel::ERROR) {
193  std::cerr << logging::LogLevel::ERROR << name << logging::getTime() << "Failed to create PID file: " << pidFile
194  << std::endl;
195  }
196  _exit(0);
197  }
198  else {
199  file << child;
200  file.close();
201  }
202  std::string path_copy = path;
203  if(path.back() != '/') path_copy.append(std::string("/").c_str());
204  if(chdir(path.c_str())) {
205  if(log <= logging::LogLevel::ERROR) {
206  std::cerr << logging::LogLevel::ERROR << name << logging::getTime() << "Failed to change to directory: " << path
207  << std::endl;
208  }
209  _exit(0);
210  }
211 
212  // prepare arguments
213  const auto args = split_arguments(cmd);
214  char* exec_args[args.size()];
215  int arg_count = 0;
216  if(log == logging::LogLevel::DEBUG) {
217  std::cout << logging::LogLevel::DEBUG << name << logging::getTime()
218  << "Going to call: execve with command:" << (path_copy + args.at(0)).c_str() << std::endl;
219  std::cout << logging::LogLevel::DEBUG << name << logging::getTime() << "Adding arguments: ";
220  }
221 
222  for(size_t x = 0; x < args.size(); x++) {
223  exec_args[arg_count++] = strdup(args[x].c_str());
224  if(log == logging::LogLevel::DEBUG) std::cout << exec_args[x] << ", ";
225  }
226  exec_args[arg_count++] = 0; // tell it when to stop!
227  if(log == logging::LogLevel::DEBUG) std::cout << "NULL" << std::endl;
228 
229  // prepare environment
230  auto env_args = split_arguments(environment, ",");
231  if(log == logging::LogLevel::DEBUG)
232  std::cout << logging::LogLevel::DEBUG << name << logging::getTime() << "Adding " << env_args.size()
233  << " environment variables." << std::endl;
234  for(auto& env_arg : env_args) {
235  std::size_t sep = env_arg.find_first_of("=");
236  if(sep != std::string::npos) {
237  setenv(env_arg.substr(0, sep).c_str(), env_arg.substr(sep + 1, env_arg.length()).c_str(), overwriteENV);
238  if(log == logging::LogLevel::DEBUG)
239  std::cout << logging::LogLevel::DEBUG << name << logging::getTime() << "Setting environment variable "
240  << env_arg.substr(0, sep) << ": " << env_arg.substr(sep + 1, env_arg.length()) << std::endl;
241  }
242  else if(log == logging::LogLevel::ERROR) {
243  std::cerr << logging::LogLevel::ERROR << name << logging::getTime()
244  << "Failed to interpret environment string: " << env_arg << std::endl;
245  }
246  }
247  // environ is a variable declared in unistd.h
248  extern char** environ;
249 
250  // close file handles when calling execv -> release the OPC UA port
251  setAllFHCloseOnExec();
252  execve((path_copy + args.at(0)).c_str(), exec_args, environ);
253  _exit(0);
254  }
255  else {
256  sleep(1);
257  if(readTempPID(pid)) {
258  if(log == logging::LogLevel::DEBUG)
259  os << logging::LogLevel::DEBUG << name << logging::getTime() << "PID was read:" << pid << std::endl;
260  if(deletePIDFile) remove(pidFile.c_str());
261  }
262  else {
263  throw std::runtime_error("Process is not started!");
264  }
265  }
266 
267  return pid;
268 }

286 void ProcessHandler::setAllFHCloseOnExec() {
287  struct rlimit rlim;
288  long max;
289  int fd;
290 
291  /* Resource limit? */
292 #if defined(RLIMIT_NOFILE)
293  if(getrlimit(RLIMIT_NOFILE, &rlim) != 0) rlim.rlim_max = 0;
294 #elif defined(RLIMIT_OFILE)
295  if(getrlimit(RLIMIT_OFILE, &rlim) != 0) rlim.rlim_max = 0;
296 #else
297  // POSIX: 8 message queues, 20 files, 8 streams
298  rlim.rlim_max = 36;
299 #endif
300 
301  // Configured limit?
302 #if defined(_SC_OPEN_MAX)
303  max = sysconf(_SC_OPEN_MAX);
304 #else
305  max = 36L;
306 #endif
307 
308  // Use the bigger of the two.
309  if((int)max > (int)rlim.rlim_max)
310  fd = max;
311  else
312  fd = rlim.rlim_max;
313 
314  while(fd-- > 0)
315  if(fd != STDIN_FILENO && fd != STDOUT_FILENO && fd != STDERR_FILENO) fcntl(fd, F_SETFD, FD_CLOEXEC);
316 }
