44 void SystemInfoModule::mainLoop() {
45  // Set variables that are read by other modules here
46  double uptime_secs;
47  double idle_secs;
48  // read the system start time (uncertainty in the order of seconds...)
49  uptime(&uptime_secs, &idle_secs);
50  auto now = boost::posix_time::second_clock::local_time();
51  status.startTime = boost::posix_time::to_time_t(now) - std::stoi(std::to_string(uptime_secs));
52  status.startTimeStr = boost::posix_time::to_simple_string(boost::posix_time::from_time_t(status.startTime));
53  status.startTime.write();
54  status.startTimeStr.write();
55  status.uptime_secTotal = std::stoi(std::to_string(uptime_secs));
56  status.uptime_secTotal.write();
57 
58  meminfo();
59  try {
60  status.maxMem = std::stoi(std::to_string(kb_main_total));
61  status.maxMem.write();
62  }
63  catch(std::exception& e) {
64  logger->sendMessage(std::string("meminfo conversion failed: ") + e.what(), logging::LogLevel::ERROR);
65  }
66 
67  info.ticksPerSecond = sysconf(_SC_CLK_TCK);
68  for(auto it = sysInfo.ibegin(); it != sysInfo.iend(); it++) {
69  info.strInfos.at(it->first) = it->second;
70  }
71  info.nCPU = sysInfo.getNCpu();
72  info.writeAll();
73 
74  if(lastInfo.size() != (unsigned)(info.nCPU + 1)) {
75  logger->sendMessage(std::string("Failed to open system file /proc/stat") + std::to_string(lastInfo.size()) +
76  "\t nCPU" + std::to_string(info.nCPU),
77  logging::LogLevel::ERROR);
78  throw std::runtime_error("Vector size mismatch in SystemInfoModule::mainLoop.");
79  }
80 
81  std::ifstream file("/proc/stat");
82  if(!file.is_open()) {
83  logger->sendMessage("Failed to open system file /proc/stat", logging::LogLevel::ERROR);
84  file.close();
85  return;
86  }
87 
88  readCPUInfo(lastInfo);
89 
90  while(true) {
91  meminfo();
92  try {
93  status.maxMem = std::stoi(std::to_string(kb_main_total));
94  status.freeMem = std::stoi(std::to_string(kb_main_free));
95  status.cachedMem = std::stoi(std::to_string(kb_main_cached));
96  status.usedMem = std::stoi(std::to_string(status.maxMem - status.freeMem - status.cachedMem));
97  status.maxSwap = std::stoi(std::to_string(kb_swap_total));
98  status.freeSwap = std::stoi(std::to_string(kb_swap_free));
99  status.usedSwap = std::stoi(std::to_string(status.maxSwap - status.freeSwap));
100  }
101  catch(std::exception& e) {
102  logger->sendMessage(std::string("meminfo conversion failed: ") + e.what(), logging::LogLevel::ERROR);
103  }
104  status.memoryUsage = 1. * status.usedMem / status.maxMem * 100.;
105  status.swapUsage = 1. * status.usedSwap / status.maxSwap * 100.;
106 
107  // get system uptime
108  try {
109  uptime(&uptime_secs, &idle_secs);
110  //\ToDo: Use stoul once unsigned long is available
111  status.uptime_secTotal = std::stoi(std::to_string(uptime_secs));
112  status.uptime_day = std::stoi(std::to_string(status.uptime_secTotal / 86400));
113  status.uptime_hour = std::stoi(std::to_string((status.uptime_secTotal - (status.uptime_day * 86400)) / 3600));
114  status.uptime_min = std::stoi(
115  std::to_string((status.uptime_secTotal - (status.uptime_day * 86400) - (status.uptime_hour * 3600)) / 60));
116  status.uptime_sec = std::stoi(std::to_string(status.uptime_secTotal - (status.uptime_day * 86400) -
117  (status.uptime_hour * 3600) - (status.uptime_min * 60)));
118  }
119  catch(std::exception& e) {
120  logger->sendMessage(std::string("uptime conversion failed: ") + e.what(), logging::LogLevel::ERROR);
121  }
122 
123  std::vector<double> v_tmp(3);
124 
125  loadavg(&v_tmp[0], &v_tmp[1], &v_tmp[2]);
126  status.loadAvg = v_tmp;
127 
128  calculatePCPU();
129 
130  status.writeAll();
131  logger->sendMessage("System data updated", logging::LogLevel::DEBUG);
132 
133  triggerGroup.trigger.read();
134  }

137 void SystemInfoModule::calculatePCPU() {
138  unsigned long long total;
139  double tmp;
140  std::vector<double> usage_tmp(info.nCPU + 1);
141  std::vector<std::string> strs;
142  std::vector<cpu> vcpu(info.nCPU + 1);
143  readCPUInfo(vcpu);
144  auto lastcpu = lastInfo.begin();
145  auto newcpu = vcpu.begin();
146  for(size_t iCPU = 0; iCPU < (info.nCPU + 1); iCPU++) {
147  if(newcpu->totalUser < lastcpu->totalUser || newcpu->totalUserLow < lastcpu->totalUserLow ||
148  newcpu->totalSys < lastcpu->totalSys || newcpu->totalIdle < lastcpu->totalIdle) {
149  // Overflow detection. Just skip this value.
150  usage_tmp.at(iCPU) = (-1.0);
151  }
152  else {
153  total = (newcpu->totalUser - lastcpu->totalUser) + (newcpu->totalUserLow - lastcpu->totalUserLow) +
154  (newcpu->totalSys - lastcpu->totalSys);
155  tmp = total;
156  total += (newcpu->totalIdle - lastcpu->totalIdle);
157  tmp /= total;
158  tmp *= 100.;
159  usage_tmp.at(iCPU) = tmp;
160  }
161  *lastcpu = *newcpu;
162  lastcpu++;
163  newcpu++;
164  }
165  status.cpu_useTotal = usage_tmp.back();
166  usage_tmp.pop_back();
167  status.cpu_use->operator=(usage_tmp);
168 }

170 void SystemInfoModule::readCPUInfo(std::vector<cpu>& vcpu) {
171  std::ifstream file("/proc/stat");
172  if(!file.is_open()) {
173  logger->sendMessage("Failed to open system file /proc/stat", logging::LogLevel::ERROR);
174  file.close();
175  return;
176  }
177  std::string::size_type sz = 0;
178  std::string line;
179  for(auto it = vcpu.begin(); it != vcpu.end(); it++) {
180  if(!std::getline(file, line)) {
181  logger->sendMessage("Could not find enough lines in /proc/stat", logging::LogLevel::ERROR);
182  file.close();
183  return;
184  }
185  auto strs = split_arguments(line);
186  if(strs.size() < 5) {
187  logger->sendMessage(
188  std::string("Line seems to contain not enough data. Line: ") + line, logging::LogLevel::ERROR);
189  }
190 
191  try {
192  it->totalUser = std::stoull(strs.at(1), &sz, 0);
193  it->totalUserLow = std::stoull(strs.at(2), &sz, 0);
194  it->totalSys = std::stoull(strs.at(3), &sz, 0);
195  it->totalIdle = std::stoull(strs.at(4), &sz, 0);
196  }
197  catch(std::exception& e) {
198  std::stringstream ss;
199  ss << "Caught an exception: " << e.what() << "\n";
200 
201  for(auto s : strs) {
202  ss << "String token: " << s << "\n";
203  }
204  logger->sendMessage(ss.str(), logging::LogLevel::ERROR);
205  }
206  }
207 }

228 bool FileSystemModule::read() {
229  std::lock_guard<std::mutex> lock(fs_mutex);
230  struct statfs fs;
231  if((statfs(((std::string)status.mountPoint).c_str(), &fs)) < 0) {
232  logger->sendMessage(std::string("Failed to stat ") + (std::string)status.mountPoint + ": " + std::to_string(errno),

234  return false;
235  }
236  else {
237  status.disk_size = 1. * (fs.f_blocks * fs.f_frsize) * GiB;
238  status.disk_free = 1. * (fs.f_bfree * fs.f_frsize) * GiB;
239  status.disk_user = 1. * (fs.f_bavail * fs.f_frsize) * GiB;
240  return true;
241  }

244 void FileSystemModule::mainLoop() {
245  deviceName = tmp[0];
246  status.mountPoint = tmp[1];
247  deviceName.write();
248  status.mountPoint.write();
249 
250  auto group = readAnyGroup();
251  while(1) {
252  if(read()) {
253  status.disk_usage = (status.disk_size - status.disk_user) / status.disk_size * 100.;
254  if((config.errorLevel > 0 && status.disk_usage > config.errorLevel) ||
255  (config.errorLevel == 0 && status.disk_usage > 95)) {
256  logger->sendMessage(std::string("More than ") + std::to_string(config.errorLevel) + "% (" +
257  std::to_string(status.disk_usage) + "%) of " + (std::string)deviceName + " mounted at " +
258  (std::string)status.mountPoint + " are used!",
259  logging::LogLevel::WARNING);
260  status.disk_status = 2;
261  }
262  else if((config.warningLevel > 0 && status.disk_usage > config.warningLevel) ||
263  (config.warningLevel == 0 && status.disk_usage > 90)) {
264  logger->sendMessage(std::string("More than ") + std::to_string(config.warningLevel) + "% (" +
265  std::to_string(status.disk_usage) + "%) of " + (std::string)deviceName + " mounted at " +
266  (std::string)status.mountPoint + " are used!",
267  logging::LogLevel::INFO);
268  status.disk_status = 1;
269  }
270  else {
271  status.disk_status = 0;
272  logger->sendMessage(std::string("Disc usage: ") + std::to_string(status.disk_usage), logging::LogLevel::DEBUG);
273  }
274  status.writeAll();
275  }
276  group.readUntil(triggerGroup.trigger.getId());
277  }

297 void NetworkModule::read() {
298  raw tmp((std::string)deviceName);
299  for(size_t i = 0; i < tmp.files.size(); i++) {
300  std::ifstream in(tmp.files.at(i));
301  if(!in.is_open()) {
302  logger->sendMessage(std::string("Failed to open: ") + tmp.files.at(i), logging::LogLevel::ERROR);
303  return;
304  }
305  tmp.time.at(i) = boost::posix_time::microsec_clock::local_time();
306  in >> tmp.data.at(i);
307  // check if this is the first reading and no data is stored yet in previousData
308  if(previousData.files.size() != 0) {
309  boost::posix_time::time_duration diff = tmp.time.at(i) - previousData.time.at(i);
310  status.data.at(i) =
311  1. * (tmp.data.at(i) - previousData.data.at(i)) / 1024 / 1024 / (diff.total_nanoseconds() / 1e9);
312  }
313  in.close();
314  }
315  previousData = raw(tmp);
316  status.writeAll();
317 }
