18  template<>
19  std::string LogicalNameMapParser::getValueFromXmlSubnode<std::string>(
20  const xmlpp::Node* node, const std::string& subnodeName, bool hasDefault, std::string defaultValue) {
21  auto list = node->find(subnodeName);
22  if(list.size() < 1 && hasDefault) return defaultValue;
23  if(list.size() != 1) {
24  parsingError(node,
25  "Expected exactly one subnode of the type '" + subnodeName + "' below node '" + node->get_name() + "'.");
26  }
27  auto childList = list[0]->get_children();
28 
29  std::string value;
30  for(auto& child : childList) {
31  // check for plain text
32  const xmlpp::TextNode* textNode = dynamic_cast<xmlpp::TextNode*>(child);
33  if(textNode) {
34  // put to stream buffer
35  value += textNode->get_content();
36  continue;
37  }
38 
39  // check for reference
40  if(child->get_name() == "ref") {
41  auto childChildList = child->get_children();
42  const xmlpp::TextNode* refNameNode = dynamic_cast<xmlpp::TextNode*>(childChildList.front());
43  if(refNameNode && childChildList.size() == 1) {
44  std::string regName = refNameNode->get_content();
45  if(!_catalogue.hasRegister(regName)) {
46  parsingError(child, "Reference to constant '" + regName + "' could not be resolved.");
47  }
48  auto reg = _catalogue.getRegister(regName);
49  auto reg_casted = boost::dynamic_pointer_cast<LNMBackendRegisterInfo>(reg);
50  assert(reg_casted != nullptr); // this is our own catalogue
51  // fetch the value of the target constant
52  if(reg_casted->targetType == LNMBackendRegisterInfo::TargetType::CONSTANT) {
53  if(reg_casted->plugins.size() > 0) {
54  parsingError(childList.front(), "'" + regName + "' uses plugins which is not supported for <ref>");
55  }
56  // put to stream buffer
57  callForType(reg_casted->valueType, [&](auto arg) {
58  std::stringstream buf;
59  buf << boost::fusion::at_key<decltype(arg)>(reg_casted->valueTable.table).latestValue[0];
60  value = buf.str();
61  });
62  continue;
63  }
64  else {
65  parsingError(child, "Reference to '" + regName + "' does not refer to a constant.");
66  }
67  }
68  else {
69  parsingError(child, "The <ref> node must contain only text.");
70  }
71  }
72 
73  // check for parameter
74  if(child->get_name() == "par") {
75  auto childChildList = child->get_children();
76  const xmlpp::TextNode* parNameNode = dynamic_cast<xmlpp::TextNode*>(childChildList.front());
77  if(parNameNode && childChildList.size() == 1) {
78  std::string parName = parNameNode->get_content();
79  if(_parameters.find(parName) == _parameters.end()) {
80  parsingError(child, "Parameter '" + parName + "' could not be resolved.");
81  }
82  // put to stream buffer
83  value += _parameters[parName];
84  continue;
85  }
86  else {
87  parsingError(child, "The <par> node must contain only text.");
88  }
89  }
90 
91  // neither found: throw error
92  parsingError(node,
93  "Node '" + subnodeName + "' should contain only text, references or parameters. Instead child '" +
94  child->get_name() + "' was found.");
95  }
96  return value;
97  }

116  template<typename T>
117  std::vector<T> LogicalNameMapParser::getValueVectorFromXmlSubnode(
118  const xmlpp::Node* node, const std::string& subnodeName) {
119  auto list = node->find(subnodeName);
120  if(list.size() < 1) {
121  parsingError(node,
122  "Expected at least one subnode of the type '" + subnodeName + "' below node '" + node->get_name() + "'.");
123  }
124 
125  std::vector<T> valueVector;
126 
127  for(auto& child : list) {
128  const xmlpp::Element* childElement = dynamic_cast<const xmlpp::Element*>(child);
129  if(!childElement) continue; // ignore comments etc.
130 
131  // obtain index and resize valueVector if necessary
132  auto indexAttr = childElement->get_attribute("index");
133  size_t index = 0;
134  if(indexAttr) {
135  index = std::stoi(indexAttr->get_value());
136  }
137  if(index >= valueVector.size()) valueVector.resize(index + 1);
138 
139  // get content
140  auto childList = childElement->get_children();
141  if(childList.size() != 1) {
142  parsingError(childElement,
143  "Node '" + subnodeName +
144  "' should contain only text or exactly one reference, "
145  "instead multiple childs "
146  "were found.");
147  }
148 
149  // check for plain text
150  const xmlpp::TextNode* textNode = dynamic_cast<xmlpp::TextNode*>(childList.front());
151  if(textNode) {
152  std::stringstream buf(textNode->get_content());
153  buf >> valueVector[index];
154  continue;
155  }
156 
157  // check for reference
158  if(childList.front()->get_name() == "ref") {
159  auto childChildList = childList.front()->get_children();
160  const xmlpp::TextNode* refNameNode = dynamic_cast<xmlpp::TextNode*>(childChildList.front());
161  if(refNameNode && childChildList.size() == 1) {
162  std::string regName = refNameNode->get_content();
163  if(!_catalogue.hasRegister(regName)) {
164  parsingError(childList.front(), "Reference to constant '" + regName + "' could not be resolved.");
165  }
166  auto reg = _catalogue.getRegister(regName);
167  auto reg_casted = boost::dynamic_pointer_cast<LNMBackendRegisterInfo>(reg);
168  assert(reg_casted != nullptr); // this is our own catalogue
169  // fetch the value of the target constant
170  if(reg_casted->targetType == LNMBackendRegisterInfo::TargetType::CONSTANT) {
171  if(reg_casted->plugins.size() > 0) {
172  parsingError(childList.front(), "'" + regName + "' uses plugins which is not supported for <ref>");
173  }
174  // convert via string
175  std::stringstream buf;
176  callForType(reg_casted->valueType, [&](auto arg) {
177  buf << boost::fusion::at_key<decltype(arg)>(reg_casted->valueTable.table).latestValue[0];
178  });
179  buf >> valueVector[index];
180  continue;
181  }
182  else {
183  parsingError(childList.front(), "Reference to '" + regName + "' does not refer to a constant.");
184  }
185  }
186  else {
187  parsingError(childList.front(), "The <ref> node must contain only text.");
188  }
189  }
190 
191  if(childList.front()->get_name() == "par") {
192  auto childChildList = childList.front()->get_children();
193  const xmlpp::TextNode* parNameNode = dynamic_cast<xmlpp::TextNode*>(childChildList.front());
194  if(parNameNode && childChildList.size() == 1) {
195  std::string parName = parNameNode->get_content();
196  if(_parameters.find(parName) == _parameters.end()) {
197  parsingError(childList.front(), "Parameter '" + parName + "' could not be resolved.");
198  }
199  // put to stream buffer
200  std::stringstream buf;
201  buf << _parameters[parName];
202  buf >> valueVector[index];
203  continue;
204  }
205  else {
206  parsingError(childList.front(), "The <par> node must contain only text.");
207  }
208  }
209 
210  // neither found: throw error
211  parsingError(node,
212  "Node '" + subnodeName + "' should contain only text or exactly one reference, instead child '" +
213  childList.front()->get_name() + "' was found.");
214  }
215 
216  return valueVector;
217  }

221  void LogicalNameMapParser::parseFile(const std::string& fileName) {
222  _fileName = fileName;
223 
224  // parse the file into a DOM structure
225  xmlpp::DomParser parser;
226  try {
227  parser.parse_file(fileName);
228  }
229  catch(xmlpp::exception& e) {
230  throw ChimeraTK::logic_error("Error opening the xlmap file '" + fileName + "': " + e.what());
231  }
232 
233  // get root element
234  const auto root = parser.get_document()->get_root_node();
235  if(root->get_name() != "logicalNameMap") {
236  parsingError(root, "Expected 'logicalNameMap' tag instead of: " + root->get_name());
237  }
238 
239  // parsing loop
240  for(const auto& child : root->get_children()) {
241  // cast into element, ignore if not an element (e.g. comment)
242  const xmlpp::Element* element = dynamic_cast<const xmlpp::Element*>(child);
243  if(!element) continue;
244 
245  // parse the element
246  parseElement(RegisterPath(), element);
247  }

250  void LogicalNameMapParser::parseElement(RegisterPath currentPath, const xmlpp::Element* element) {
251  // module tag found: look for registers and sub-modules in module
252  if(element->get_name() == "module") {
253  // obtain name of the module
254  auto nameAttr = element->get_attribute("name");
255  if(!nameAttr) {
256  parsingError(element, "Missing name attribute of 'module' tag.");
257  }
258  std::string moduleName = nameAttr->get_value();
259 
260  // iterate over childs in module
261  for(const auto& child : element->get_children()) {
262  // cast into element, ignore if not an element (e.g. comment)
263  const xmlpp::Element* childElement = dynamic_cast<const xmlpp::Element*>(child);
264  if(!childElement) continue;
265 
266  // parse the element
267  parseElement(currentPath / moduleName, childElement);
268  }
269  }
270 
271  // register tag found: create new entry in map
272  else {
273  // obtain the type
274  std::string type = element->get_name();
275 
276  // create new RegisterInfo object
277  auto info = boost::shared_ptr<LNMBackendRegisterInfo>(new LNMBackendRegisterInfo());
278 
279  if(type == "redirectedRegister") {
280  info->targetType = LNMBackendRegisterInfo::TargetType::REGISTER;
281  info->deviceName = getValueFromXmlSubnode<std::string>(element, "targetDevice");
282  info->registerName = getValueFromXmlSubnode<std::string>(element, "targetRegister");
283  info->firstIndex = getValueFromXmlSubnode<unsigned int>(element, "targetStartIndex", true, 0);
284  info->length = getValueFromXmlSubnode<unsigned int>(element, "numberOfElements", true, 0);
285  info->nDimensions = 0;
286  info->nChannels = 0;
287  }
288  else if(type == "redirectedChannel") {
289  info->targetType = LNMBackendRegisterInfo::TargetType::CHANNEL;
290  info->deviceName = getValueFromXmlSubnode<std::string>(element, "targetDevice");
291  info->registerName = getValueFromXmlSubnode<std::string>(element, "targetRegister");
292  info->channel = getValueFromXmlSubnode<unsigned int>(element, "targetChannel");
293  info->firstIndex = getValueFromXmlSubnode<unsigned int>(element, "targetStartIndex", true, 0);
294  info->length = getValueFromXmlSubnode<unsigned int>(element, "numberOfElements", true, 0);
295  info->nDimensions = 1;
296  info->nChannels = 1;
297  }
298  else if(type == "redirectedBit") {
299  info->targetType = LNMBackendRegisterInfo::TargetType::BIT;
300  info->deviceName = getValueFromXmlSubnode<std::string>(element, "targetDevice");
301  info->registerName = getValueFromXmlSubnode<std::string>(element, "targetRegister");
302  info->bit = getValueFromXmlSubnode<unsigned int>(element, "targetBit");
303  info->firstIndex = 0;
304  info->length = 0;
305  info->nDimensions = 1;
306  info->nChannels = 1;
307  }
308  else if(type == "constant") {
309  std::string constantType = getValueFromXmlSubnode<std::string>(element, "type");
310  if(constantType == "integer") constantType = "int32";
311  info->targetType = LNMBackendRegisterInfo::TargetType::CONSTANT;
312  info->valueType = DataType(constantType);
313  callForType(info->valueType, [&](auto arg) {
314  boost::fusion::at_key<decltype(arg)>(info->valueTable.table).latestValue =
315  this->getValueVectorFromXmlSubnode<decltype(arg)>(element, "value");
316  });
317  info->firstIndex = 0;
318  info->length = getValueFromXmlSubnode<unsigned int>(element, "numberOfElements", true, 1);
319  info->nDimensions = info->length > 1 ? 1 : 0;
320  info->nChannels = 1;
321  info->writeable = false;
322  info->readable = true;
323  info->_dataDescriptor = ChimeraTK::RegisterInfo::DataDescriptor(info->valueType);
324  }
325  else if(type == "variable") {
326  std::string constantType = getValueFromXmlSubnode<std::string>(element, "type");
327  if(constantType == "integer") constantType = "int32";
328  info->targetType = LNMBackendRegisterInfo::TargetType::VARIABLE;
329  info->valueType = DataType(constantType);
330  callForType(info->valueType, [&](auto arg) {
331  boost::fusion::at_key<decltype(arg)>(info->valueTable.table).latestValue =
332  this->getValueVectorFromXmlSubnode<decltype(arg)>(element, "value");
333  });
334  info->firstIndex = 0;
335  info->length = getValueFromXmlSubnode<unsigned int>(element, "numberOfElements", true, 1);
336  info->nDimensions = info->length > 1 ? 1 : 0;
337  info->nChannels = 1;
338  info->writeable = true;
339  info->readable = true;
340  info->_dataDescriptor = ChimeraTK::RegisterInfo::DataDescriptor(info->valueType);
341  info->supportedFlags = {AccessMode::wait_for_new_data};
342  }
343  else {
344  parsingError(element, "Wrong logical register type: " + type);
345  }
346 
347  // obtain name of logical register
348  auto nameAttr = element->get_attribute("name");
349  if(!nameAttr) {
350  parsingError(element, "Missing name attribute of '" + type + "' tag.");
351  }
352  RegisterPath registerName = currentPath / std::string(nameAttr->get_value());
353  info->name = registerName;
354 
355  // iterate over childs of the register to find plugins
356  for(const auto& child : element->get_children()) {
357  // cast into element, ignore if not an element (e.g. comment)
358  const xmlpp::Element* childElement = dynamic_cast<const xmlpp::Element*>(child);
359  if(!childElement) continue;
360  if(childElement->get_name() != "plugin") continue; // look only for plugins
361 
362  // get name of plugin
363  auto pluginNameAttr = childElement->get_attribute("name");
364  if(!pluginNameAttr) {
365  parsingError(childElement, "Missing name attribute of 'plugin' tag.");
366  }
367  std::string pluginName = pluginNameAttr->get_value();
368 
369  // collect parameters
370  std::map<std::string, std::string> parameters;
371  for(const auto& paramchild : childElement->get_children()) {
372  // cast into element, ignore if not an element (e.g. comment)
373  const xmlpp::Element* paramElement = dynamic_cast<const xmlpp::Element*>(paramchild);
374  if(!paramElement) continue;
375  if(paramElement->get_name() != "parameter") {
376  parsingError(paramElement, "Unexpected element '" + paramElement->get_name() + "' inside plugin tag.");
377  }
378 
379  // get name of parameter
380  auto parameterNameAttr = paramElement->get_attribute("name");
381  if(!pluginNameAttr) {
382  parsingError(paramElement, "Missing name attribute of 'parameter' tag.");
383  }
384  std::string parameterName = parameterNameAttr->get_value();
385 
386  // get value of parameter and store in map
387  parameters[parameterName] =
388  getValueFromXmlSubnode<std::string>(childElement, "parameter[@name='" + parameterName + "']");
389  }
390 
391  // create instance of plugin and add to the list in the register info
392  info->plugins.push_back(LNMBackend::makePlugin(info, pluginName, parameters));
393  }
394 
395  // add register to catalogue
396  _catalogue.addRegister(info);
397  }
398  }

402  std::unordered_set<std::string> LogicalNameMapParser::getTargetDevices() const {
403  std::unordered_set<std::string> ret;
404  for(auto it = _catalogue.begin(); it != _catalogue.end(); ++it) {
405  auto info = boost::static_pointer_cast<const LNMBackendRegisterInfo>(it.get());
