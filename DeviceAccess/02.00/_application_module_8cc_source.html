51  void ApplicationModule::terminate() {
52  if(moduleThread.joinable()) {
53  moduleThread.interrupt();
54  // try joining the thread
55  while(!moduleThread.try_join_for(boost::chrono::milliseconds(10))) {
56  // if thread is not yet joined, send interrupt() to all variables.
57  for(auto& var : getAccessorListRecursive()) {
58  auto el{var.getAppAccessorNoType().getHighLevelImplElement()};
59  if(el->getAccessModeFlags().has(AccessMode::wait_for_new_data)) {
60  el->interrupt();
61  }
62  }
63  // it may not suffice to send interrupt() once, as the exception might get
64  // overwritten in the queue, thus we repeat this until the thread was
65  // joined.
66  }
67  }
68  assert(!moduleThread.joinable());
69  }

77  void ApplicationModule::mainLoopWrapper() {
78  Application::registerThread("AM_" + getName());
79 
80  // Acquire testable mode lock, so from this point on we are running only one user thread concurrently
81  Application::testableModeLock("start");
82 
83  // Read all variables once to obtain the initial values from the devices and from the control system persistency
84  // layer. This is done in two steps, first for all poll-type variables and then for all push-types, because
85  // poll-type reads might trigger distribution of values to push-type variables via a ConsumingFanOut.
86  for(auto& variable : getAccessorListRecursive()) {
87  if(variable.getDirection().dir != VariableDirection::consuming) continue;
88  if(variable.getMode() == UpdateMode::poll) {
89  assert(!variable.getAppAccessorNoType().getHighLevelImplElement()->getAccessModeFlags().has(
90  AccessMode::wait_for_new_data));
91  Application::testableModeUnlock("Initial value read for poll-type " + variable.getName());
92  Application::getInstance().circularDependencyDetector.registerDependencyWait(variable);
93  variable.getAppAccessorNoType().read();
94  Application::getInstance().circularDependencyDetector.unregisterDependencyWait(variable);
95  if(!Application::testableModeTestLock()) {
96  // The lock may have already been acquired if the above read() goes to a ConsumingFanOut, which sends out
97  // the data to a slave decorated by a TestableModeAccessorDecorator. Hence we heer must acquire the lock only
98  // if we do not have it.
99  Application::testableModeLock("Initial value read for poll-type " + variable.getName());
100  }
101  }
102  }
103  for(auto& variable : getAccessorListRecursive()) {
104  if(variable.getDirection().dir != VariableDirection::consuming) continue;
105  if(variable.getMode() == UpdateMode::push) {
106  Application::testableModeUnlock("Initial value read for push-type " + variable.getName());
107  Application::getInstance().circularDependencyDetector.registerDependencyWait(variable);
108  Application::testableModeLock("Initial value read for push-type " + variable.getName());
109  variable.getAppAccessorNoType().read();
110  Application::testableModeUnlock("Initial value read for push-type " + variable.getName());
111  Application::getInstance().circularDependencyDetector.unregisterDependencyWait(variable);
112  Application::testableModeLock("Initial value read for push-type " + variable.getName());
113  }
114  }
115 
116  // We are holding the testable mode lock, so we are sure the mechanism will work now.
117  testableModeReached = true;
118 
119  // enter the main loop
120  mainLoop();
121  Application::testableModeUnlock("terminate");
122  }
