277  inline bool ReadAnyGroup::Notification::accept() {
278  if(!this->valid) {
279  throw std::logic_error("This notification object is invalid.");
280  }
281  if(this->accepted) {
282  throw std::logic_error("This notification has already been accepted.");
283  }
284  this->accepted = true;
285  try {
286  auto tf = this->transferElement.readAsync();
287  detail::getFutureQueueFromTransferFuture(tf).pop_wait();
288  }
289  catch(detail::DiscardValueException&) {
290  return false;
291  }
292  this->transferElement.getHighLevelImplElement()->postRead();
293  return true;
294  }

360  inline void ReadAnyGroup::add(TransferElementAbstractor element) {
361  if(isFinalised) {
362  throw std::logic_error("ReadAnyGroup has already been finalised, calling "
363  "add() is no longer allowed.");
364  }
365  if(!element.isReadable()) {
366  throw std::logic_error(
367  "Cannot add non-readable accessor for register " + element.getName() + " to ReadAnyGroup.");
368  }
369  if(element.getAccessModeFlags().has(AccessMode::wait_for_new_data)) {
370  push_elements.push_back(element);
371  }
372  else {
373  poll_elements.push_back(element);
374  }
375  }

383  inline void ReadAnyGroup::finalise() {
384  if(isFinalised)
385  throw std::logic_error("ReadAnyGroup has already been finalised, calling "
386  "finalise() is no longer allowed.");
387  std::vector<cppext::future_queue<void>> queueList;
388  bool groupEmpty = true;
389  for(auto& e : push_elements) {
390  auto tf = e.readAsync();
391  queueList.push_back(detail::getFutureQueueFromTransferFuture(tf));
392  groupEmpty = false;
393  }
394  if(groupEmpty) {
395  throw std::logic_error("ReadAnyGroup has no element with AccessMode::wait_for_new_data.");
396  }
397  notification_queue = cppext::when_any(queueList.begin(), queueList.end());
398  isFinalised = true;
399  }

416  inline ReadAnyGroup::Notification ReadAnyGroup::waitAny() {
417  // We might block here until we receive an update, so call the
418  // transferFutureWaitCallback. Note this is a slightly ugly approximation
419  // here, as we call it for the first element in the group. It is used in
420  // ApplicationCore testable mode, where it doesn't matter which callback
421  // within the same group is called.
422  push_elements[0].transferFutureWaitCallback();
423 
424  // Wait for notification
425  std::size_t index;
426  notification_queue.pop_wait(index);
427  return Notification(push_elements[index], index);
428  }

454  inline void ReadAnyGroup::readUntilAll(const std::vector<TransferElementID>& ids) {
455  std::map<TransferElementID, bool> found;
456  for(auto& id : ids)
457  found[id] = false; // initialise map so we can tell from the map if a
458  // variable is in the vector
459  size_t leftToFind = ids.size();
460  while(true) {
461  auto read = readAny();
462  if(found.count(read) == 0) continue; // variable is not in the vector ids
463  if(found[read] == true) continue; // variable has been read already
464  found[read] = true;
465  --leftToFind;
466  if(leftToFind == 0) return;
467  }
468  }
