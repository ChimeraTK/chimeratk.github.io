41  template<typename UserType>
42  boost::shared_ptr<NDRegisterAccessor<UserType>> getRegisterAccessor_impl(const RegisterPath& registerPathName,
43  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
44  auto path = registerPathName;
45  path.setAltSeparator(".");
46  auto pathComponents = path.getComponents();
47  bool pushRead = false;
48  if(pathComponents[pathComponents.size() - 1] == "PUSH_READ") {
49  if(flags.has(AccessMode::wait_for_new_data)) {
50  pushRead = true;
51  flags.remove(AccessMode::wait_for_new_data);
52  }
53  path--; // remove last component
54  }
55 
56  auto acc = DummyBackendBase::getRegisterAccessor_impl<UserType>(path, numberOfWords, wordOffsetInRegister, flags);
57  if(pushRead) {
58  std::unique_lock<std::mutex> lk(_pushDecoratorsMutex);
59 
60  auto decorator = boost::make_shared<ExceptionDummyPushDecorator<UserType>>(
61  acc, boost::dynamic_pointer_cast<ExceptionDummy>(this->shared_from_this()));
62 
63  _pushDecorators[registerPathName].push_back(decorator);
64 
65  if(_activateNewPushAccessors) {
66  decorator->_isActive = true;
67  decorator->trigger(); // initial value
68  }
69 
70  acc = decorator;
71  }
72  else {
73  // decorate all poll-type variable so returned validity of the data can be controlled
74  auto decorator = boost::make_shared<ExceptionDummyPollDecorator<UserType>>(
75  acc, boost::dynamic_pointer_cast<ExceptionDummy>(this->shared_from_this()));
76  acc = decorator;
77  }
78 
79  // create entry in _writeOrderMap and _writeCounterMap if necessary
80  if(pathComponents[pathComponents.size() - 1] != "DUMMY_WRITEABLE" &&
81  (pathComponents[0].find("DUMMY_INTERRUPT_") != 0)) {
82  auto info = getRegisterInfo(path);
83  auto adrPair = std::make_pair(info.bar, info.address);
84  if(_writeOrderMap.find(adrPair) == _writeOrderMap.end()) {
85  _writeOrderMap[adrPair] = 0;
86  _writeCounterMap[adrPair] = 0;
87  }
88  }
89 
90  acc->setExceptionBackend(shared_from_this());
91 
92  return acc;
93  }

167  template<typename UserType>
168  struct ExceptionDummyPushDecorator : NDRegisterAccessorDecorator<UserType>, ExceptionDummyPushDecoratorBase {
169  ExceptionDummyPushDecorator(const boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>>& target,
170  boost::shared_ptr<ExceptionDummy> backend)
171  : NDRegisterAccessorDecorator<UserType>(target), _backend(std::move(backend)) {
172  assert(_target->isReadable());
173 
174  this->_accessModeFlags = _target->getAccessModeFlags();
175  this->_accessModeFlags.add(AccessMode::wait_for_new_data);
176 
177  _readQueue =
178  _myReadQueue.template then<void>([&](Buffer current) { this->_current = current; }, std::launch::deferred);
179 
180  _path = _target->getName();
181  _path.setAltSeparator(".");
182  _path /= "PUSH_READ";
183  }
184 
185  ~ExceptionDummyPushDecorator() override {
186  std::unique_lock<std::mutex> lk(_backend->_pushDecoratorsMutex);
187  try {
188  auto& list = _backend->_pushDecorators.at(_path);
189  for(auto it = list.begin(); it != list.end(); ++it) {
190  if(it->lock().get() == nullptr) { // weak_ptr is already not lockable any more
191  list.erase(it);
192  return;
193  }
194  }
195  }
196  catch(std::exception& e) {
197  // The exception that could show up here is coming from >_pushDecorators.at().
198  // It might be std::out_of_range if the _path is not in, or an exception from the string comparison inside of
199  // std::map::at(). Both should not be possible be due to the code logic. The code will run into an assertion in
200  // debug mode. Let's hope the what() string gives us valuable information.
201  std::cout << "~ExceptionDummyPushDecorator(): Unexpected exception: " << e.what() << std::endl;
202  }
203  std::cout << "~ExceptionDummyPushDecorator(): Could not unlist instance!" << std::endl;
204  assert(false);
205  }
206 
207  void interrupt() override { this->interrupt_impl(this->_myReadQueue); }
208 
209  void setExceptionBackend(boost::shared_ptr<DeviceBackend> exceptionBackend) override {
210  // do not set it for the target, since we read from the target in trigger(), but that is the wrong place to
211  // call setException(). So we don't call it on our base class NDRegisterAccessorDecorator but on the
212  // TransferElement itself.
213  // Turn off the linter warning. This is intentional.
214  // NOLINTNEXTLINE(bugprone-parent-virtual-call)
215  TransferElement::setExceptionBackend(exceptionBackend);
216  }

227  void trigger() override {
228  try {
229  _hasException = false;
230  _target->read();
231  Buffer b;
232  b._data = _target->accessChannels();
233  b._version = _backend->_pushVersions[_path];
234  b._validity = _target->dataValidity();
235  _myReadQueue.push_overwrite(b);
236  }
237  // fixme: clang tidy says something else is escaping, don't know what
238  catch(ChimeraTK::runtime_error&) {
239  _isActive = false;
240  if(!_hasException) _myReadQueue.push_overwrite_exception(std::current_exception());
241  _hasException = true;
242  }
243  }

258  _dataValidity = _current._validity;

260  }
