41  void BackendFactory::registerBackendType(std::string backendType,
42  boost::shared_ptr<DeviceBackend> (*creatorFunction)(
43  std::string address, std::map<std::string, std::string> parameters),
44  std::vector<std::string> sdmParameterNames, std::string version) {
45 #ifdef _DEBUG
46  std::cout << "adding:" << backendType << std::endl << std::flush;
47 #endif
48  called_registerBackendType = true;
49 
50  if(creatorMap.find(backendType) != creatorMap.end()) {
51  throw ChimeraTK::logic_error("A backend with the type name '" + backendType + "' has already been registered.");
52  }
53 
54  if(version != CHIMERATK_DEVICEACCESS_VERSION) {
55  // Register a function that throws an exception with the message when trying
56  // to create a backend. We do not throw here because this would throw an
57  // uncatchable exception inside the static instance of a backend registerer,
58  // which would violate the policy that a dmap file with broken backends
59  // should be usable if the particular backend is not used.
60  std::stringstream errorMessage;
61  errorMessage << "Backend plugin '" << backendType << "' compiled with wrong DeviceAccess version " << version
62  << ". Please recompile with version " << CHIMERATK_DEVICEACCESS_VERSION;
63  creatorMap_compat[make_pair(backendType, "")] =
64  boost::bind(BackendFactory::failedRegistrationThrowerFunction, boost::placeholders::_1,
65  boost::placeholders::_2, boost::placeholders::_3, boost::placeholders::_4, errorMessage.str());
66  std::string errorMessageString = errorMessage.str();
67  creatorMap[backendType] = [errorMessageString](std::string,
68  std::map<std::string, std::string>) -> boost::shared_ptr<ChimeraTK::DeviceBackend> {
69  throw ChimeraTK::logic_error(errorMessageString);
70  };
71  return;
72  }
73  creatorMap[backendType] = creatorFunction;
74  creatorMap_compat[make_pair(backendType, "")] = [creatorFunction, sdmParameterNames](std::string,
75  std::string instance, std::list<std::string> parameters,
76  std::string mapFileName) {
77  std::map<std::string, std::string> pars;
78  size_t i = 0;
79  for(auto& p : parameters) {
80  if(i >= sdmParameterNames.size()) break;
81  pars[sdmParameterNames[i]] = p;
82  ++i;
83  }
84  if(!mapFileName.empty()) {
85  if(pars["map"].empty()) {
86  pars["map"] = mapFileName;
87  }
88  else {
89  std::cout << "WARNING: You have specified the map file name twice, in "
90  "the parameter list and in the 3rd "
91  "column of the DMAP file."
92  << std::endl;
93  std::cout << "Please only specify the map file name in the parameter list!" << std::endl;
94  }
95  }
96  return creatorFunction(instance, pars);
97  };
98  }

179  boost::shared_ptr<DeviceBackend> BackendFactory::createBackend(std::string aliasOrUri) {
180  std::lock_guard<std::mutex> lock(_mutex);
181 
182  if(Utilities::isDeviceDescriptor(aliasOrUri) || Utilities::isSdm(aliasOrUri)) {
183  // it is a URI, directly create a deviceinfo and call the internal creator
184  // function
185  DeviceInfoMap::DeviceInfo deviceInfo;
186  deviceInfo.uri = aliasOrUri;
187  return createBackendInternal(deviceInfo);
188  }
189 
190  // it's not a URI. Try finding the alias in the dmap file.
191  if(_dMapFile.empty()) {
192  throw ChimeraTK::logic_error("DMap file not set.");
193  }
194  auto deviceInfo = Utilities::aliasLookUp(aliasOrUri, _dMapFile);
195  return createBackendInternal(deviceInfo);
196  }

200  boost::shared_ptr<DeviceBackend> BackendFactory::createBackendInternal(const DeviceInfoMap::DeviceInfo& deviceInfo) {
201 #ifdef _DEBUG
202  std::cout << "uri to parse" << deviceInfo.uri << std::endl;
203  std::cout << "Entries" << creatorMap.size() << std::endl << std::flush;
204 #endif
205 
206  // Check if backend already exists
207  auto iterator = _existingBackends.find(deviceInfo.uri);
208  if(iterator != _existingBackends.end()) {
209  auto strongPtr = iterator->second.lock();
210  if(strongPtr) {
211  return strongPtr;
212  }
213  }
214 
215  // Check if descriptor string is a ChimeraTK device descriptor
216  if(Utilities::isDeviceDescriptor(deviceInfo.uri)) {
217  auto cdd = Utilities::parseDeviceDesciptor(deviceInfo.uri);
218  try {
219  auto backend = (creatorMap.at(cdd.backendType))(cdd.address, cdd.parameters);
220  _existingBackends[deviceInfo.uri] = backend;
221  return backend;
222  }
223  catch(std::out_of_range&) {
224  throw ChimeraTK::logic_error("Unknown backend: \"" + cdd.backendType + "\" at " + deviceInfo.dmapFileName +
225  ":" + std::to_string(deviceInfo.dmapFileLineNumber) + " for " + deviceInfo.uri);
226  }
227  }
228 
229  // Check for deprecated SDM or device node
230  // Note: Deprecation message was added 2022-07-28. Remove functionality past end of 2023.
231  Sdm sdm;
232  if(Utilities::isSdm(deviceInfo.uri)) {
233  sdm = Utilities::parseSdm(deviceInfo.uri);
234  std::cout << "Using the SDM descriptor is deprecated. Please change to CDD (ChimeraTK device descriptor)."
235  << std::endl;
236  }
237  else {
238  sdm = Utilities::parseDeviceString(deviceInfo.uri);
239  std::cout
240  << "Using the device node in a dmap file is deprecated. Please change to CDD (ChimeraTK device descriptor)."
241  << std::endl;
242  }
243 #ifdef _DEBUG
244  std::cout << "sdm._Host:" << sdm._Host << std::endl;
245  std::cout << "sdm.Interface:" << sdm._Interface << std::endl;
246  std::cout << "sdm.Instance:" << sdm._Instance << std::endl;
247  std::cout << "sdm._Protocol:" << sdm._Protocol << std::endl;
248  std::cout << "sdm.parameter:" << sdm._Parameters.size() << std::endl;
249  for(std::list<std::string>::iterator it = sdm._Parameters.begin(); it != sdm._Parameters.end(); ++it) {
250  std::cout << *it << std::endl;
251  }
252 #endif
253  for(auto iter = creatorMap_compat.begin(); iter != creatorMap_compat.end(); ++iter) {
254 #ifdef _DEBUG
255  std::cout << "Pair:" << iter->first.first << "+" << iter->first.second << std::endl;
256 #endif
257  if((iter->first.first == sdm._Interface)) {
258  auto backend = (iter->second)(sdm._Host, sdm._Instance, sdm._Parameters, deviceInfo.mapFileName);
259  boost::weak_ptr<DeviceBackend> weakBackend = backend;
260  _existingBackends[deviceInfo.uri] = weakBackend;
261  // return the shared pointer, not the weak pointer
262  return backend;
263  }
264  }
265 
266  throw ChimeraTK::logic_error("Unregistered device: Interface = " + sdm._Interface + " Protocol = " + sdm._Protocol);
267  return boost::shared_ptr<DeviceBackend>(); // won't execute
268  }

272  void BackendFactory::loadPluginLibrary(std::string soFile) {
273  // reset flag to check if the registerBackedType() function was called
274  called_registerBackendType = false;
275 
276  // open the plugin library. RTLD_LAZY is enough since all symbols are loaded
277  // when needed
278  void* hndl = dlopen(soFile.c_str(), RTLD_LAZY);
279  if(hndl == nullptr) {
280  throw ChimeraTK::logic_error(dlerror());
281  }
282 
283  // if no backend was registered, close the library and throw an exception
284  if(!called_registerBackendType) {
285  dlclose(hndl);
286  throw ChimeraTK::logic_error(
287  "'" + soFile + "' is not a valid DeviceAccess plugin, it does not register any backends!");
288  }
289  }

293  void BackendFactory::loadAllPluginsFromDMapFile() {
294  if(_dMapFile.empty()) {
295  return;
296  }
297 
298  auto dmap = DMapFileParser().parse(_dMapFile);
299 
300  for(auto lib : dmap->getPluginLibraries()) {
301  try {
302  loadPluginLibrary(lib);
303  }
304  catch(ChimeraTK::logic_error& e) {
305  // Ignore library loading errors when doing this automatically for all
306  // plugins in the dmap file to keep dmap files usable if the broken
307  // backends are not used. Print a warning that the loading failed.
308  std::cerr << "Error: Caught exception loading plugin '" << lib << "' specified in dmap file: " << e.what()
309  << std::endl;
310  std::cerr << "Some backends will not be available!" << std::endl;
311  }
312  }
313  }
