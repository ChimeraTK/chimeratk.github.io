105 template<typename Derived>
106 struct OneDRegisterDescriptorBase : RegisterDescriptorBase<Derived> {
107  using RegisterDescriptorBase<Derived>::derived;
108 
109  size_t nChannels() { return 1; }
110 
111  size_t myOffset() { return 0; }
112 
113  // T is always minimumUserType, but C++ doesn't allow to use Derived::minimumUserType here (circular dependency)
114  template<typename T, typename Traw>
115  T convertRawToCooked(Traw value) {
116  return static_cast<T>(value);
117  }
118 
119  template<typename UserType>
120  void generateValueHook(std::vector<UserType>&) {} // override in derived if needed
121 
122  // type can be user type or raw type
123  template<typename Type>
124  std::vector<std::vector<Type>> generateValue(bool getRaw = false) {
125  std::vector<Type> v;
126  typedef typename Derived::rawUserType Traw;
127  typedef typename Derived::minimumUserType T;
128  auto cv = derived->template getRemoteValue<Traw>(true)[0];
129  for(size_t i = 0; i < derived->nElementsPerChannel(); ++i) {
130  Traw e = cv[i] + derived->increment * (static_cast<Traw>(i) + 1);
131  if(!getRaw) {
132  v.push_back(derived->template convertRawToCooked<T, Traw>(e));
133  }
134  else {
135  v.push_back(static_cast<T>(e));
136  }
137  }
138  derived->generateValueHook(v);
139  return {v};
140  }
141 
142  template<typename UserType>
143  std::vector<std::vector<UserType>> getRemoteValue(bool getRaw = false) {
144  typedef typename Derived::rawUserType Traw;
145 
146  std::vector<UserType> cookedValues(derived->nElementsPerChannel());
147  std::vector<Traw> rawValues(derived->nElementsPerChannel());
148 
149  // Keep the scope of the dummy buffer lock as limited as possible (see #12332).
150  // The rawToCooked conversion will acquire a lock via the math pluging decorator,
151  // which will cause lock order inversion if you hold the dummy buffer lock at that point.
152  {
153  auto bufferLock = derived->acc.getBufferLock();
154 
155  for(size_t i = 0; i < derived->nElementsPerChannel(); ++i) {
156  rawValues[i] = derived->acc[i + derived->myOffset()];
157  }
158  } // end of bufferLock scope
159 
160  for(size_t i = 0; i < derived->nElementsPerChannel(); ++i) {
161  if(!getRaw) {
162  cookedValues[i] = derived->template convertRawToCooked<UserType, Traw>(rawValues[i]);
163  }
164  else {
165  cookedValues[i] = static_cast<UserType>(rawValues[i]); // you can only use raw if user type and raw type are
166  // the same, so the static cast is a no-op
167  }
168  }
169  return {cookedValues};
170  }
171 
172  void setRemoteValue() {
173  auto v = generateValue<typename Derived::rawUserType>(true)[0];
174  { // scope for the buffer lock
175  auto bufferLock = derived->acc.getBufferLock();
176  for(size_t i = 0; i < derived->nElementsPerChannel(); ++i) {
177  derived->acc[i + derived->myOffset()] = v[i];
178  }
179  } // release the buffer lock before triggering another thread
180  if(derived->isPush()) {
181  dynamic_cast<ExceptionDummy&>(derived->acc.getBackend()).triggerInterrupt(6);
182  }
183  }
184 };

225 struct VariableRegisterDescriptorBase : OneDRegisterDescriptorBase<Derived> {
226  using RegisterDescriptorBase<Derived>::derived;
227  static constexpr auto capabilities = RegisterDescriptorBase<Derived>::capabilities.disableTestRawTransfer();
228 
229  size_t nChannels() { return 1; }
230  ChimeraTK::AccessModeFlags supportedFlags() { return {ChimeraTK::AccessMode::wait_for_new_data}; }
231 
232  size_t nRuntimeErrorCases() { return 0; }
233 
234  template<typename UserType>
235  std::vector<std::vector<UserType>> getRemoteValue(bool = false) {
236  // For Variables we don't have a backdoor. We have to use the normal read and write
237  // functions which are good enough. It seems like a self consistency test, but all
238  // functionality the variable has to provide is that I can write something, and
239  // read it back, which is tested with it.
240 
241  // We might have to open/recover the backend to perform the operation. We have to remember
242  // that we did so and close/set-exception it again it we did. Some tests require the backend to be closed.
243  bool backendWasOpened = lmapBackend->isOpen();
244  bool backendWasFunctional = lmapBackend->isFunctional();
245  if(!backendWasOpened || !backendWasFunctional) {
246  lmapBackend->open();
247  }
248  auto acc = lmapBackend->getRegisterAccessor<typename Derived::minimumUserType>(derived->path(), 0, 0, {});
249  acc->read();
250  if(!backendWasOpened) {
251  lmapBackend->close();
252  }
253  else if(!backendWasFunctional) {
254  lmapBackend->setException();
255  }
256  std::vector<UserType> v;
257  for(size_t k = 0; k < derived->nElementsPerChannel(); ++k) {
258  v.push_back(acc->accessData(k));
259  }
260  return {v};
261  }
262 
263  void setRemoteValue() {
264  auto acc = lmapBackend->getRegisterAccessor<typename Derived::minimumUserType>(derived->path(), 0, 0, {});
265  auto v = derived->template generateValue<typename Derived::minimumUserType>()[0];
266  for(size_t k = 0; k < derived->nElementsPerChannel(); ++k) {
267  acc->accessData(k) = v[k];
268  }
269  bool backendWasOpened = lmapBackend->isOpen();
270  if(!backendWasOpened) {
271  lmapBackend->open();
272  }
273  try {
274  acc->write();
275  }
276  catch(...) {
277  // ignore any exceptions: if the device is in an exception state, this write must not take place but the exception
278  // must not get through.
279  }
280  if(!backendWasOpened) {
281  lmapBackend->close();
282  }
283  }
284 
285  void setForceRuntimeError(bool, size_t) { assert(false); }
