27  template<typename DATA_TYPE>
28  struct pitched_iterator {
29  // standard iterator traits
30  using iterator_category = std::random_access_iterator_tag;
31  using value_type = DATA_TYPE;
32  using difference_type = std::ptrdiff_t;
33  using pointer = DATA_TYPE*;
34  using reference = DATA_TYPE&;
35 
36  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
37  pitched_iterator(void* begin, size_t pitch) : _ptr(reinterpret_cast<std::byte*>(begin)), _pitch(pitch) {}
38 
39  template<typename OTHER_DATA_TYPE>
40  explicit pitched_iterator(pitched_iterator<OTHER_DATA_TYPE>& other) : _ptr(other._ptr), _pitch(other._pitch) {}
41 
42  pitched_iterator& operator++() {
43  _ptr += _pitch;
44  return *this;
45  }
46  pitched_iterator operator++(int) {
47  pitched_iterator retval = *this;
48  ++(*this);
49  return retval;
50  }
51  pitched_iterator operator+(size_t n) { return pitched_iterator(_ptr + n * _pitch, _pitch); }
52  bool operator==(pitched_iterator other) const { return _ptr == other._ptr; }
53  bool operator!=(pitched_iterator other) const { return !(*this == other); }
54  size_t operator-(pitched_iterator other) const { return _ptr - other._ptr; }
55  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
56  DATA_TYPE& operator*() const { return *reinterpret_cast<DATA_TYPE*>(_ptr); }
57 
58  private:
59  std::byte* _ptr;
60  const size_t _pitch;
61 
62  template<typename OTHER_DATA_TYPE>
63  friend struct pitched_iterator;
64  };

135  template<class UserType, class ConverterType>
136  NumericAddressedBackendMuxedRegisterAccessor<UserType, ConverterType>::NumericAddressedBackendMuxedRegisterAccessor(
137  const RegisterPath& registerPathName, size_t numberOfElements, size_t elementsOffset,
138  const boost::shared_ptr<DeviceBackend>& _backend)
139  : NDRegisterAccessor<UserType>(registerPathName, {}),
140  _ioDevice(boost::dynamic_pointer_cast<NumericAddressedBackend>(_backend)) {
141  // Obtain information about the area
142  _registerInfo = _ioDevice->_registerMap.getBackendRegister(registerPathName);
143  assert(!_registerInfo.channels.empty());
144 
145  // Create a fixed point converter for each channel
146  for(size_t i = 0; i < _registerInfo.getNumberOfChannels(); ++i) {
147  if(_registerInfo.channels[i].bitOffset % 8 != 0) {
148  throw ChimeraTK::logic_error("NumericAddressedBackendMuxedRegisterAccessor: elements must be byte aligned.");
149  }
150  _converters.emplace_back(detail::createDataConverter<ConverterType>(_registerInfo, i));
151  }
152  // check information
153  if(_registerInfo.elementPitchBits % 8 != 0) {
154  throw ChimeraTK::logic_error("NumericAddressedBackendMuxedRegisterAccessor: blocks must be byte aligned.");
155  }
156 
157  // compute effective numberOfElements
158  if(numberOfElements == 0) {
159  numberOfElements = _registerInfo.nElements;
160  }
161 
162  // check number of words
163  if(numberOfElements + elementsOffset > _registerInfo.nElements) {
164  throw ChimeraTK::logic_error("Requested number of elements exceeds the size of the register! Requested end: " +
165  std::to_string(numberOfElements + elementsOffset) +
166  ", register length: " + std::to_string(_registerInfo.nElements));
167  }
168 
169  // update register info
170  _registerInfo.nElements = numberOfElements;
171  assert(_registerInfo.elementPitchBits % 8 == 0);
172  _registerInfo.address += elementsOffset * _registerInfo.elementPitchBits / 8;
173 
174  // allocate the buffer for the converted data
175  NDRegisterAccessor<UserType>::buffer_2D.resize(_converters.size());
176  for(size_t i = 0; i < _converters.size(); ++i) {
177  NDRegisterAccessor<UserType>::buffer_2D[i].resize(_registerInfo.nElements);
178  }
179 
180  // allocate the raw io buffer. Make it one element larger to make sure we can access the last byte via int32_t*
181  _ioBuffer.resize(
182  static_cast<size_t>(_registerInfo.elementPitchBits) / 8 * _registerInfo.nElements / sizeof(int32_t) + 1);
183 
184  // compute pitched iterators for accessing the channels
185  // Silence the linter: Yes, we are doing a reinterpet cast. There is nothing we can do about it when we're
186  // bit-fiddling
187 
188  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
189  auto* ioBuffer = reinterpret_cast<uint8_t*>(&_ioBuffer[0]);
190  for(auto& c : _registerInfo.channels) {
191  assert(c.bitOffset % 8 == 0);
192  _startIterators.emplace_back(ioBuffer + c.bitOffset / 8, _registerInfo.elementPitchBits / 8);
193  _endIterators.push_back(_startIterators.back() + _registerInfo.nElements);
194  }
195  }

199  template<class UserType, class ConverterType>
200  void NumericAddressedBackendMuxedRegisterAccessor<UserType, ConverterType>::doReadTransferSynchronously() {
201  assert(_registerInfo.elementPitchBits % 8 == 0);
202 
203  auto nbt = _registerInfo.elementPitchBits / 8 * _registerInfo.nElements;
204  nbt = ((nbt - 1) / 4 + 1) * 4; // round up to multiple of 4 bytes
205 
206  _ioDevice->read(_registerInfo.bar, _registerInfo.address, _ioBuffer.data(), nbt);
207  }

211  template<class UserType, class ConverterType>
212  void NumericAddressedBackendMuxedRegisterAccessor<UserType, ConverterType>::doPostRead(
213  TransferType, bool hasNewData) {
214  if(hasNewData) {
215  for(size_t i = 0; i < _converters.size(); ++i) {
216  _converters[i].template vectorToCooked<UserType>(_startIterators[i], _endIterators[i], buffer_2D[i].begin());
217  }
218  // it is acceptable to create the version number in post read because this accessor does not have
219  // wait_for_new_data. It is basically synchronous.
220  this->_versionNumber = {};
221 
222  // we just read good data. Set validity back to ok if someone marked it faulty for writing.
223  this->_dataValidity = DataValidity::ok;
224  }

229  template<class UserType, class ConverterType>
230  bool NumericAddressedBackendMuxedRegisterAccessor<UserType, ConverterType>::doWriteTransfer(VersionNumber) {
231  assert(_registerInfo.elementPitchBits % 8 == 0);
232 
233  auto nbt = _registerInfo.elementPitchBits / 8 * _registerInfo.nElements;
234  nbt = ((nbt - 1) / 4 + 1) * 4; // round up to multiple of 4 bytes
235 
236  _ioDevice->write(_registerInfo.bar, _registerInfo.address, &(_ioBuffer[0]), nbt);
237  return false;

242  template<class UserType, class ConverterType>
243  void NumericAddressedBackendMuxedRegisterAccessor<UserType, ConverterType>::doPreWrite(TransferType, VersionNumber) {
244  if(!_ioDevice->isOpen()) throw ChimeraTK::logic_error("Device not opened.");
245 
246  assert(_registerInfo.channels.size() == _converters.size());
247 
248  // Convert channel by channel
249  for(size_t i = 0; i < _registerInfo.channels.size(); ++i) {
250  callForRawType(_registerInfo.channels[i].getRawType(), [&](auto x) {
251  using RawType = decltype(x);
252 
253  // Call FixedPointConverter::toRaw() for each value in the channel. The result (C++ type int32) is written to
254  // the target buffer through the pitched iterators after converting it into the RawType matching the actual
255  // bit width of the channel. This is important to avoid overwriting data of other channels.
256  std::transform(buffer_2D[i].begin(), buffer_2D[i].end(), detail::pitched_iterator<RawType>(_startIterators[i]),
257  [&](UserType cookedValue) { return _converters[i].toRaw(cookedValue); });
258  });
259  }
260  }
