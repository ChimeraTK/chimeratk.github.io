32  _transferType = type;
33  if(this->_accessModeFlags.has(AccessMode::wait_for_new_data) && type == TransferType::readNonBlocking) {
34  // if the access mode has wait_for_new_data then readNonBlocking can be called multiple times by readLatest,

37  BOOST_CHECK(_preRead_counter <= (this->_readQueue.size()));

135  void doPostRead(TransferType type, bool updateDataBuffer) override {
136  // doPreRead and doPortRead must always be called in pairs.
137  // This can happen multiple times in readLatest. The absolute counting is done in doPreRead()
138  BOOST_CHECK_EQUAL(_preRead_counter, _postRead_counter + 1);
139  BOOST_CHECK(_preWrite_counter == 0);
140  if(!_throwLogicErr && !_throwRuntimeErrInPre && !_throwThreadInterruptedInPre) {
141  if(!this->_accessModeFlags.has(AccessMode::wait_for_new_data)) {
142  BOOST_CHECK_EQUAL(_readTransfer_counter, 1);
143  }
144  else {
145  BOOST_CHECK_EQUAL(_readTransfer_counter, 0);
146  }
147  }
148  else {
149  /* Here B.6.1 is tested for read operations */
150  BOOST_CHECK_EQUAL(_readTransfer_counter, 0);
151  }
152  if(_throwLogicErr || _throwRuntimeErrInPre || _throwThreadInterruptedInPre || _throwRuntimeErrInTransfer ||
153  _throwThreadInterruptedInTransfer) {
154  BOOST_CHECK(this->_activeException != nullptr);
155  }
156  /* Check B.7.4 */
157  if(this->_activeException) {
158  BOOST_CHECK(updateDataBuffer == false);

160  BOOST_CHECK(_writeTransfer_counter == 0);
161  BOOST_CHECK(_postWrite_counter == 0);
162  BOOST_CHECK(_transferType == type);

165  if(_throwNumericCast) throw boost::numeric::bad_numeric_cast();
166  if(_throwThreadInterruptedInPost) throw boost::thread_interrupted();
