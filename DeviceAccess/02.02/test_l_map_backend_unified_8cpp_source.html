211  // For Variables we don't have a backdoor. We have to use the normal read and write
212  // functions which are good enough. It seems like a self consistency test, but all
213  // functionality the variable has to provide is that I can write something, and
214  // read it back, which is tested with it.
215 
216  // We might have to open/recover the backend to perform the operation. We have to remember
217  // that we did so and close/set-exception it again it we did. Some tests require the backend to be closed.
218  bool backendWasOpened = lmapBackend->isOpen();
219  bool backendWasFunctional = lmapBackend->isFunctional();
220  if(!backendWasOpened || !backendWasFunctional) {
221  lmapBackend->open();
222  }
223  auto acc = lmapBackend->getRegisterAccessor<typename Derived::minimumUserType>(derived->path(), 0, 0, {});
224  acc->read();
225  if(!backendWasOpened) {
226  lmapBackend->close();
227  }
228  else if(!backendWasFunctional) {
229  lmapBackend->setException();
230  }
231  std::vector<UserType> v;
232  for(size_t k = 0; k < derived->nElementsPerChannel(); ++k) {
233  v.push_back(acc->accessData(k));
234  }
235  return {v};

239  auto acc = lmapBackend->getRegisterAccessor<typename Derived::minimumUserType>(derived->path(), 0, 0, {});
240  auto v = derived->template generateValue<typename Derived::minimumUserType>()[0];
241  for(size_t k = 0; k < derived->nElementsPerChannel(); ++k) {
242  acc->accessData(k) = v[k];

245  if(!backendWasOpened) {
