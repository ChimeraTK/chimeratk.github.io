411  void ConfigParser::parseModule(const xmlpp::Element* element, std::string parent_name) {
412  auto module_name = (element->get_name() == "configuration") // root node gets special treatment
413  ?
414  "" :
415  element->get_attribute("name")->get_value() + "/";
416 
417  parent_name += module_name;
418 
419  for(const auto& child : element->get_children()) {
420  element = dynamic_cast<const xmlpp::Element*>(child);
421  if(!element) {
422  continue; // ignore if not an element (e.g. comment)
423  }
424  else if(isVariable(element)) {
425  variableList_->emplace_back(prefix(parent_name, parseVariable(element)));
426  }
427  else if(isArray(element)) {
428  arrayList_->emplace_back(prefix(parent_name, parseArray(element)));
429  }
430  else if(isModule(element)) {
431  parseModule(element, parent_name);
432  }
433  else {
434  error("Unknown tag: " + element->get_name());
435  }
436  }
437  }

525  std::map<size_t, std::string> ConfigParser::gettArrayValues(const xmlpp::Element* element) {
526  bool valueFound = false;
527  std::map<size_t, std::string> values;
528 
529  for(const auto& valueChild : element->get_children()) {
530  const xmlpp::Element* valueElement = dynamic_cast<const xmlpp::Element*>(valueChild);
531  if(!valueElement) continue; // ignore comments etc.
532  validateValueNode(valueElement);
533  valueFound = true;
534 
535  auto index = valueElement->get_attribute("i");
536  auto value = valueElement->get_attribute("v");
537 
538  // get index as number and store value as a string
539  size_t intIndex;
540  try {
541  intIndex = std::stoi(index->get_value());
542  }
543  catch(std::exception& e) {
544  error("Cannot parse string '" + std::string(index->get_value()) + "' as an index number: " + e.what());
545  }
546  values[intIndex] = value->get_value();
547  }
548  // make sure there is at least one value
549  if(!valueFound) {
550  error("Each variable must have a value, either specified as an attribute or as child tags.");
551  }
552  return values;
553  }
