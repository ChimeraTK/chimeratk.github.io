113  /*** Each implementation must provide a function to create specific AsyncVariables.
114  * When the variable is returned, async accessor is not set yet. This would leave the whole creation
115  * of the async accessor to the backend, would mean a lot of code duplication. It also cannot be
116  * retrieved from the AsyncVariable as this only contains a weak pointer.
117  * If the isActive flag is set, the _sendBuffer must already contain the initial value. The variable itself
118  * is not activated yet as the async accessor is still not set.

121  std::unique_ptr<AsyncVariable>(

123 
124  // This mutex protects the _asyncVariables container and all its contents, and the _isActive flag. You must not
125  // touch those variables without holding the mutex. It serves two purposes:
126  // 1. Variables can be added or removed from the container at any time. It is not safe to handle an element without
127  // holding the lock.
128  // 2. The elements in the container are not thread-safe as well. We use the same lock as it is needed for 1. anyway.

173  template<typename UserType>
174  void AsyncVariableImpl<UserType>::activateAndSend() {
175  // The initial value must have been set before calling this function. We can just send it.
176  auto subscriber = _asyncAccessor.lock();
177  if(subscriber.get() != nullptr) { // Possible race condition: The subscriber is being destructed.
178  subscriber->activate(_sendBuffer);
179  _isActive = true;
180  }
181  }
182 
183  //*********************************************************************************************************************/
184  template<typename UserType>
185  void AsyncVariableImpl<UserType>::sendException(std::exception_ptr e) {
186  _isActive = false;
187  auto subscriber = _asyncAccessor.lock();
188  if(subscriber.get() != nullptr) { // Possible race condition: The subscriber is being destructed.
189  subscriber->sendException(e);
190  }
191  }

194  template<typename UserType>
195  AsyncVariableImpl<UserType>::AsyncVariableImpl(size_t nChannels, size_t nElements)
196  : _sendBuffer(nChannels, nElements) {
197  // _isActive is false. You have to activate it after creation when necessary. Reason: The AsyncNDRegisterAccessor is
198  // not set, and the send buffer needs to be filled before so we can send the initial value. As the latter is backend
199  // specific it cannot happen here, because this code is executed first.
200  }

202  //*********************************************************************************************************************/
203  template<typename UserType>
204  void AsyncVariableImpl<UserType>::deactivate() {
205  auto subscriber = _asyncAccessor.lock();
206  if(subscriber.get() != nullptr) { // Possible race condition: The subscriber is being destructed.
207  subscriber->deactivate();
208  }
209  _isActive = false;

214  void AsyncVariableImpl<UserType>::send() {
215  auto subscriber = _asyncAccessor.lock();
216  if(subscriber.get() != nullptr) { // Possible race condition: The subscriber is being destructed.
217  subscriber->sendDestructively(_sendBuffer);

221  //*********************************************************************************************************************/
222  template<typename UserType>
223  boost::shared_ptr<AsyncNDRegisterAccessor<UserType>> AsyncAccessorManager::subscribe(
224  boost::shared_ptr<DeviceBackend> backend, RegisterPath name, size_t numberOfWords, size_t wordOffsetInRegister,
225  AccessModeFlags flags) {
226  std::lock_guard<std::recursive_mutex> variablesLock(_variablesMutex);
227 
228  AccessorInstanceDescriptor descriptor(name, typeid(UserType), numberOfWords, wordOffsetInRegister, flags);
229  auto untypedAsyncVariable =
230  CALL_VIRTUAL_FUNCTION_TEMPLATE(createAsyncVariable, UserType, backend, descriptor, _isActive);
231 
232  auto asyncVariable = dynamic_cast<AsyncVariableImpl<UserType>*>(untypedAsyncVariable.get());
233  // we take all the information we need for the async accessor from AsyncVariable because we cannot use the catalogue
234  // here
235  auto newSubscriber = boost::make_shared<AsyncNDRegisterAccessor<UserType>>(backend, shared_from_this(), name,
236  asyncVariable->getNumberOfChannels(), asyncVariable->getNumberOfSamples(), flags, asyncVariable->getUnit(),
237  asyncVariable->getDescription());
238  // Set the exception backend here. It might be that the accessor is already activated during subscription, and the
239  // backend should be set at that point
240  newSubscriber->setExceptionBackend(backend);
241 
242  if(asyncVariable->isWriteable()) {
243  // for writeable variables we add another synchronous accessors (which knows how to access the data) to the
244  // asyncAccessor (which is generic)
245  auto synchronousFlags = flags;
246  synchronousFlags.remove(AccessMode::wait_for_new_data);
247  }
248 
249  asyncVariable->_asyncAccessor = newSubscriber;
250  // Now that the AsyncVariable is complete we can finally activate it.
251  if(_isActive) {
252  asyncVariable->activateAndSend();
253  }
254  _asyncVariables[newSubscriber->getId()] = std::move(untypedAsyncVariable);
255  asyncVariableMapChanged();
256  return newSubscriber;
257  }
