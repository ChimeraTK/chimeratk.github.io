210  template<typename UserType>
211  boost::shared_ptr<AsyncNDRegisterAccessor<UserType>> AsyncAccessorManager::subscribe(
212  boost::shared_ptr<DeviceBackend> backend, RegisterPath name, size_t numberOfWords, size_t wordOffsetInRegister,
213  AccessModeFlags flags) {
214  std::lock_guard<std::recursive_mutex> variablesLock(_variablesMutex);
215 
216  AccessorInstanceDescriptor descriptor(name, typeid(UserType), numberOfWords, wordOffsetInRegister, flags);
217  auto untypedAsyncVariable =
218  CALL_VIRTUAL_FUNCTION_TEMPLATE(createAsyncVariable, UserType, backend, descriptor, _isActive);
219 
220  auto asyncVariable = dynamic_cast<AsyncVariableImpl<UserType>*>(untypedAsyncVariable.get());
221  // we take all the information we need for the async accessor from AsyncVariable because we cannot use the catalogue here
222  auto newSubscriber = boost::make_shared<AsyncNDRegisterAccessor<UserType>>(backend, shared_from_this(), name,
223  asyncVariable->getNumberOfChannels(), asyncVariable->getNumberOfSamples(), flags, asyncVariable->getUnit(),
224  asyncVariable->getDescription());
225  // Set the exception backend here. It might be that the accessor is already activated during subscription, and the backend should be set at that point
226  newSubscriber->setExceptionBackend(backend);
227 
228  if(asyncVariable->isWriteable()) {
229  // for writeable variables we add another synchronous accessors (which knows how to access the data) to the asyncAccessor (which is generic)
230  auto synchronousFlags = flags;
231  synchronousFlags.remove(AccessMode::wait_for_new_data);
232 
233  // Don't call backend->getSyncRegisterAccessor() here. It might skip the overriding of a backend.
234  newSubscriber->setWriteAccessor(
235  backend->getRegisterAccessor<UserType>(name, numberOfWords, wordOffsetInRegister, synchronousFlags));
236  }
237 
238  asyncVariable->_asyncAccessor = newSubscriber;
239  // Now that the AsyncVariable is complete we can finally activate it.
240  if(_isActive) {
241  asyncVariable->activateAndSend();
242  }
243  _asyncVariables[newSubscriber->getId()] = std::move(untypedAsyncVariable);
244  return newSubscriber;
245  }
