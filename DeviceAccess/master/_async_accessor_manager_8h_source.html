183  template<typename UserType>
184  boost::shared_ptr<AsyncNDRegisterAccessor<UserType>> AsyncAccessorManager::subscribe(
185  RegisterPath name, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
186  AccessorInstanceDescriptor descriptor{name, typeid(UserType), numberOfWords, wordOffsetInRegister, flags};
187  auto untypedAsyncVariable = CALL_VIRTUAL_FUNCTION_TEMPLATE(createAsyncVariable, UserType, descriptor);
188 
189  auto asyncVariable = dynamic_cast<AsyncVariableImpl<UserType>*>(untypedAsyncVariable.get());
190  // we take all the information we need for the async accessor from AsyncVariable because we cannot use the catalogue
191  // here
192  auto newSubscriber = boost::make_shared<AsyncNDRegisterAccessor<UserType>>(_backend, shared_from_this(),
193  _asyncDomain, name, asyncVariable->getNumberOfChannels(), asyncVariable->getNumberOfSamples(), flags,
194  asyncVariable->getUnit(), asyncVariable->getDescription());
195  // Set the exception backend here. It might be that the accessor is already activated during subscription, and the
196  // backend should be set at that point
197  newSubscriber->setExceptionBackend(_backend);
198 
199  asyncVariable->_asyncAccessor = newSubscriber;
200  // Now that the AsyncVariable is complete we can finally activate it.
201  if(_asyncDomain->unsafeGetIsActive()) {
202  asyncVariable->fillSendBuffer();
203  asyncVariable->send();
204  }
205 
206  _asyncVariables[newSubscriber->getId()] = std::move(untypedAsyncVariable);
207 
208  asyncVariableMapChanged();
209  return newSubscriber;
210  }
