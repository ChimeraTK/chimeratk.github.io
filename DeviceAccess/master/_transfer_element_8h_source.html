116  void read() {
117  if(TransferElement::_isInTransferGroup) {
118  throw ChimeraTK::logic_error("Calling read() or write() on the TransferElement '" + _name +
119  "' which is part of a TransferGroup is not allowed.");
120  }
121  this->readTransactionInProgress = false;
122 
123  preReadAndHandleExceptions(TransferType::read);
124  if(!_activeException) {
125  handleTransferException([&] { readTransfer(); });
126  }
127 
128  postReadAndHandleExceptions(TransferType::read, !_activeException);
129  }

142  bool readNonBlocking() {
143  if(TransferElement::_isInTransferGroup) {
144  throw ChimeraTK::logic_error("Calling read() or write() on the TransferElement '" + _name +
145  "' which is part of a TransferGroup is not allowed.");
146  }
147  this->readTransactionInProgress = false;
148  preReadAndHandleExceptions(TransferType::readNonBlocking);
149  bool updateDataBuffer = false;
150  if(!_activeException) {
151  handleTransferException([&] { updateDataBuffer = readTransferNonBlocking(); });
152  }
153 
154  bool retVal = updateDataBuffer;
155  if(_activeException) {
156  auto previousVersionNumber = _versionNumber;
157  auto previousDataValidity = _dataValidity;
158  // always call postRead with updateDataBuffer = false in case of an exception
159  postReadAndHandleExceptions(TransferType::readNonBlocking, false);
160  // Usually we do not reach this point because postRead() is re-throwing the _activeException.
161  // If we reach this point the exception has been suppressed. We have to calculate a
162  // new return value because the dataBuffer has not changed, but the meta data
163  // could have, in which case we have to return true.
164  retVal = (previousVersionNumber != _versionNumber) || (previousDataValidity != _dataValidity);
165  }
166  else {
167  // call postRead with updateDataBuffer as returned by readTransferNonBlocking
168  postReadAndHandleExceptions(TransferType::readNonBlocking, updateDataBuffer);
169  }
170  return retVal;
171  }
172 
177  bool readLatest() {
178  if(_accessModeFlags.has(AccessMode::wait_for_new_data)) {
179  bool updateDataBuffer = false;
180  // Call readNonBlocking until there is no new data to be read any more
181  while(readNonBlocking()) {
182  // remember whether we have new data
183  updateDataBuffer = true;
184  }
185  return updateDataBuffer;
186  }
187  // Without wait_for_new_data readNonBlocking always returns true, and the while loop above would never end.
188  // Hence we just call the (synchronous) read and return true;
189  read();
190  return true;
191  }

196  bool write(ChimeraTK::VersionNumber versionNumber = {}) {
197  if(TransferElement::_isInTransferGroup) {
198  throw ChimeraTK::logic_error("Calling read() or write() on the TransferElement '" + _name +
199  "' which is part of a TransferGroup is not allowed.");
200  }
201  this->writeTransactionInProgress = false;
202  bool previousDataLost = true; // the value here does not matter. If there was an exception, it will be re-thrown
203  // in postWrite, so it is never returned
204 
205  preWriteAndHandleExceptions(TransferType::write, versionNumber);
206  if(!_activeException) {
207  handleTransferException([&] { previousDataLost = writeTransfer(versionNumber); });
208  }
209 
210  postWriteAndHandleExceptions(TransferType::write, versionNumber);
211  return previousDataLost;
212  }

464  void postRead(TransferType type, bool updateDataBuffer) {
465  // only delegate to doPostRead() the first time postRead() is called in a row.
466  if(readTransactionInProgress) {
467  readTransactionInProgress = false;
468  doPostRead(type, updateDataBuffer);
469  }
470 
471  // Throw on each call of postRead(). All high-level elements for a shared low-level transfer element must see the
472  // exception. Note: doPostRead can throw an exception, but in that case _activeException must be false (we can
473  // only have one exception at a time). In case other code is added here later which needs to be executed after
474  // doPostRead() always, a try-catch block may be necessary.
475  if(_activeException) {
476  // don't clear the active connection. This is done in preRead().
477  std::rethrow_exception(_activeException);
478  }
