114  void read() {
115  if(TransferElement::_isInTransferGroup) {
116  throw ChimeraTK::logic_error("Calling read() or write() on the TransferElement '" + _name +
117  "' which is part of a TransferGroup is not allowed.");
118  }
119  this->readTransactionInProgress = false;
120 
121  preReadAndHandleExceptions(TransferType::read);
122  if(!_activeException) {
123  handleTransferException([&] { readTransfer(); });
124  }
125 
126  postReadAndHandleExceptions(TransferType::read, !_activeException);
127  }
128 
140  bool readNonBlocking() {
141  if(TransferElement::_isInTransferGroup) {
142  throw ChimeraTK::logic_error("Calling read() or write() on the TransferElement '" + _name +
143  "' which is part of a TransferGroup is not allowed.");
144  }
145  this->readTransactionInProgress = false;
146  preReadAndHandleExceptions(TransferType::readNonBlocking);
147  bool updateDataBuffer = false;
148  if(!_activeException) {
149  handleTransferException([&] { updateDataBuffer = readTransferNonBlocking(); });
150  }
151 
152  bool retVal = updateDataBuffer;
153  if(_activeException) {
154  auto previousVersionNumber = _versionNumber;
155  auto previousDataValidity = _dataValidity;
156  // always call postRead with updateDataBuffer = false in case of an exception
157  postReadAndHandleExceptions(TransferType::readNonBlocking, false);
158  // Usually we do not reach this point because postRead() is re-throwing the _activeException.
159  // If we reach this point the exception has been suppressed. We have to calculate a
160  // new return value because the dataBuffer has not changed, but the meta data
161  // could have, in which case we have to return true.
162  retVal = (previousVersionNumber != _versionNumber) || (previousDataValidity != _dataValidity);
163  }
164  else {
165  // call postRead with updateDataBuffer as returned by readTransferNonBlocking
166  postReadAndHandleExceptions(TransferType::readNonBlocking, updateDataBuffer);
167  }
168  return retVal;
169  }
170 
175  bool readLatest() {
176  if(_accessModeFlags.has(AccessMode::wait_for_new_data)) {
177  bool updateDataBuffer = false;
178  // Call readNonBlocking until there is no new data to be read any more
179  while(readNonBlocking()) {
180  // remember whether we have new data
181  updateDataBuffer = true;
182  }
183  return updateDataBuffer;
184  }
185  // Without wait_for_new_data readNonBlocking always returns true, and the while loop above would never end.
186  // Hence we just call the (synchronous) read and return true;
187  read();
188  return true;
189  }

194  bool write(ChimeraTK::VersionNumber versionNumber = {}) {
195  if(TransferElement::_isInTransferGroup) {
196  throw ChimeraTK::logic_error("Calling read() or write() on the TransferElement '" + _name +
197  "' which is part of a TransferGroup is not allowed.");
198  }
199  this->writeTransactionInProgress = false;
200  bool previousDataLost = true; // the value here does not matter. If there was an exception, it will be re-thrown
201  // in postWrite, so it is never returned
202 
203  preWriteAndHandleExceptions(TransferType::write, versionNumber);
204  if(!_activeException) {
205  handleTransferException([&] { previousDataLost = writeTransfer(versionNumber); });
206  }
207 
208  postWriteAndHandleExceptions(TransferType::write, versionNumber);
209  return previousDataLost;
210  }

462  void postRead(TransferType type, bool updateDataBuffer) {
463  // only delegate to doPostRead() the first time postRead() is called in a row.
464  if(readTransactionInProgress) {
465  readTransactionInProgress = false;
466  doPostRead(type, updateDataBuffer);
467  }
468 
469  // Throw on each call of postRead(). All high-level elements for a shared low-level transfer element must see the
470  // exception. Note: doPostRead can throw an exception, but in that case _activeException must be false (we can
471  // only have one exception at a time). In case other code is added here later which needs to be executed after
472  // doPostRead() always, a try-catch block may be necessary.
473  if(_activeException) {
474  // don't clear the active connection. This is done in preRead().
475  std::rethrow_exception(_activeException);
476  }
