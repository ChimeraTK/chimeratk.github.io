117  void read() {
118  if(TransferElement::_isInTransferGroup) {
119  throw ChimeraTK::logic_error("Calling read() or write() on the TransferElement '" + _name +
120  "' which is part of a TransferGroup is not allowed.");
121  }
122  this->readTransactionInProgress = false;
123

125  if(!_activeException) {
126  handleTransferException([&] { readTransfer(); });
127  }
128 
129  postReadAndHandleExceptions(TransferType::read, !_activeException);

143  bool readNonBlocking() {
144  if(TransferElement::_isInTransferGroup) {
145  throw ChimeraTK::logic_error("Calling read() or write() on the TransferElement '" + _name +
146  "' which is part of a TransferGroup is not allowed.");
147  }
148  this->readTransactionInProgress = false;
149  preReadAndHandleExceptions(TransferType::readNonBlocking);
150  bool updateDataBuffer = false;
151  if(!_activeException) {
152  handleTransferException([&] { updateDataBuffer = readTransferNonBlocking(); });
153  }
154 
155  bool retVal = updateDataBuffer;
156  if(_activeException) {
157  auto previousVersionNumber = _versionNumber;
158  auto previousDataValidity = _dataValidity;
159  // always call postRead with updateDataBuffer = false in case of an exception
160  postReadAndHandleExceptions(TransferType::readNonBlocking, false);
161  // Usually we do not reach this point because postRead() is re-throwing the _activeException.
162  // If we reach this point the exception has been suppressed. We have to calculate a
163  // new return value because the dataBuffer has not changed, but the meta data
164  // could have, in which case we have to return true.
165  if((previousVersionNumber != _versionNumber) || (previousDataValidity != _dataValidity)) {
166  retVal = true;
167  }
168  else {
169  retVal = false;
170  }
171  }
172  else {
173  // call postRead with updateDataBuffer as returned by readTransferNonBlocking
174  postReadAndHandleExceptions(TransferType::readNonBlocking, updateDataBuffer);
175  }
176  return retVal;
177  }
178 
183  bool readLatest() {
184  if(_accessModeFlags.has(AccessMode::wait_for_new_data)) {
185  bool updateDataBuffer = false;
186  // Call readNonBlocking until there is no new data to be read any more
187  while(readNonBlocking()) {
188  // remember whether we have new data
189  updateDataBuffer = true;
190  }
191  return updateDataBuffer;
192  }
193  else {
194  // Without wait_for_new_data readNonBlocking always returns true, and the while loop above would never end.
195  // Hence we just call the (synchronous) read and return true;
196  read();
197  return true;
198  }
199  }

207  "' which is part of a TransferGroup is not allowed.");
208  }
209  this->writeTransactionInProgress = false;
210  bool previousDataLost = true; // the value here does not matter. If there was an exception, it will be re-thrown
211  // in postWrite, so it is never returned
212 
213  preWriteAndHandleExceptions(TransferType::write, versionNumber);
214  if(!_activeException) {
215  handleTransferException([&] { previousDataLost = writeTransfer(versionNumber); });

475  void postRead(TransferType type, bool updateDataBuffer) {
476  // only delegate to doPostRead() the first time postRead() is called in a row.
477  if(readTransactionInProgress) {
478  readTransactionInProgress = false;
479  doPostRead(type, updateDataBuffer);
480  }

482  // Throw on each call of postRead(). All high-level elements for a shared low-level transfer element must see the
483  // exception. Note: doPostRead can throw an exception, but in that case _activeException must be false (we can
484  // only have one exception at a time). In case other code is added here later which needs to be executed after
485  // doPostRead() always, a try-catch block may be necessary.
486  if(_activeException) {
487  // don't clear the active connection. This is done in preRead().
488  std::rethrow_exception(_activeException);
