82  template<typename UserType>
83  LNMBackendVariableAccessor<UserType>::LNMBackendVariableAccessor(boost::shared_ptr<DeviceBackend> dev,
84  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
85  : NDRegisterAccessor<UserType>(registerPathName, flags), _registerPathName(registerPathName),
86  _wordOffsetInRegister(wordOffsetInRegister), _flags(flags) {
87  // cast device
88  _dev = boost::dynamic_pointer_cast<LogicalNameMappingBackend>(dev);
89 
90  // obtain the register info
91  _info = _dev->_catalogue_mutable.getBackendRegister(_registerPathName);
92  // boost::static_pointer_cast<BackendRegisterCatalogue<LNMBackendRegisterInfo>>(_dev->getRegisterCatalogue().getRegister(_registerPathName));
93 
94  // check for unknown flags
95  if(_info.targetType == LNMBackendRegisterInfo::TargetType::VARIABLE) {
96  flags.checkForUnknownFlags({AccessMode::wait_for_new_data});
97  }
98  else {
99  // no flags are supported for constants
100  flags.checkForUnknownFlags({});
101  }
102 
103  // numberOfWords default to full register length
104  if(numberOfWords == 0) numberOfWords = _info.length;
105 
106  // check for illegal parameter combinations
107  if(wordOffsetInRegister + numberOfWords > _info.length) {
108  throw ChimeraTK::logic_error(
109  "Requested number of words and/or offset exceeds length of register '" + registerPathName + "'.");
110  }
111 
112  // check for incorrect usage of this accessor
113  if(_info.targetType != LNMBackendRegisterInfo::TargetType::CONSTANT &&
114  _info.targetType != LNMBackendRegisterInfo::TargetType::VARIABLE) {
115  throw ChimeraTK::logic_error("LNMBackendVariableAccessor used for wrong register type."); // LCOV_EXCL_LINE
116  // (impossible to
117  // test...)
118  }
119 
120  // if wait_for_new_data is specified, make subscription
121  if(flags.has(AccessMode::wait_for_new_data)) {
122  // allocate _queueValue buffer
123  this->_queueValue.value.resize(numberOfWords);
124 
125  auto& lnmVariable = _dev->_variables[_info.name];
126  std::lock_guard<std::mutex> lock(lnmVariable.valueTable_mutex);
127 
128  callForType(_info.valueType, [&, this](auto arg) {
129  typedef decltype(arg) T;
130  auto& vtEntry = boost::fusion::at_key<T>(lnmVariable.valueTable.table);
131  // create subscription queue
132  cppext::future_queue<typename LNMVariable::ValueTable<T>::QueuedValue> queue(3);
133  // place queue in map
134  vtEntry.subscriptions[this->getId()] = queue;
135  // make void-typed continuation of subscription queue, which stores the received value into the _queueValue
136  this->_readQueue = queue.template then<void>(
137  [this](const typename LNMVariable::ValueTable<T>::QueuedValue& queueValue) {
138  this->_queueValue.validity = queueValue.validity;
139  this->_queueValue.version = queueValue.version;
140  for(size_t i = 0; i < queueValue.value.size(); ++i) {
141  this->_queueValue.value[i] =
142  userTypeToUserType<UserType>(queueValue.value[i + this->_wordOffsetInRegister]);
143  }
144  },
145  std::launch::deferred);
146  // put initial value into queue, if async reads activated
147  if(_dev->_asyncReadActive) {
148  queue.push({vtEntry.latestValue, vtEntry.latestValidity, vtEntry.latestVersion});
149  }
150  });
151  }
152 
153  // allocate application buffer
154  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
155  NDRegisterAccessor<UserType>::buffer_2D[0].resize(numberOfWords);
156  }

160  template<typename UserType>
161  LNMBackendVariableAccessor<UserType>::~LNMBackendVariableAccessor() {
162  if(_flags.has(AccessMode::wait_for_new_data)) {
163  // unsubscribe the update queue
164  auto& lnmVariable = _dev->_variables[_info.name];
165  std::lock_guard<std::mutex> lock(lnmVariable.valueTable_mutex);
166  callForType(_info.valueType, [&, this](auto arg) {
167  typedef decltype(arg) T;
168  auto& vtEntry = boost::fusion::at_key<T>(lnmVariable.valueTable.table);
169  vtEntry.subscriptions.erase(this->getId());
170  });
171  }
172  }

187  std::ignore = type;

199  template<typename UserType>
200  bool LNMBackendVariableAccessor<UserType>::doWriteTransfer(ChimeraTK::VersionNumber v) {
201  if(_dev->_hasException) {
202  throw ChimeraTK::runtime_error("previous, unrecovered fault");
203  }
204  auto& lnmVariable = _dev->_variables[_info.name];
205  std::lock_guard<std::mutex> lock(lnmVariable.valueTable_mutex);
206 
207  callForType(_info.valueType, [&, this](auto arg) {
208  auto& vtEntry = boost::fusion::at_key<decltype(arg)>(lnmVariable.valueTable.table);
209 
210  // store new value as latest value
211  for(size_t i = 0; i < this->buffer_2D[0].size(); ++i) {
212  vtEntry.latestValue[i + _wordOffsetInRegister] = userTypeToUserType<decltype(arg)>(this->buffer_2D[0][i]);
213  }
214  vtEntry.latestValidity = this->dataValidity();
215  vtEntry.latestVersion = v;
216 
217  // push new value to subscription queues, if async read is activated
218  if(_dev->_asyncReadActive) {
219  for(auto& sub : vtEntry.subscriptions) {
220  sub.second.push_overwrite({vtEntry.latestValue, this->dataValidity(), v});
221  }
222  }
223  });
224  return false;
225  }

266  template<typename UserType>
267  void LNMBackendVariableAccessor<UserType>::doPostRead(TransferType, bool hasNewData) {
268  if(!hasNewData) return;
269 
270  if(!_flags.has(AccessMode::wait_for_new_data)) {
271  // poll-type read transfer: fetch latest value from ValueTable
272  auto& lnmVariable = _dev->_variables[_info.name];
273  std::lock_guard<std::mutex> lock(lnmVariable.valueTable_mutex);
274 
275  callForType(_info.valueType, [&, this](auto arg) {
276  auto& vtEntry = boost::fusion::at_key<decltype(arg)>(lnmVariable.valueTable.table);
277  for(size_t i = 0; i < this->buffer_2D[0].size(); ++i) {
278  this->buffer_2D[0][i] = userTypeToUserType<UserType>(vtEntry.latestValue[i + _wordOffsetInRegister]);
279  this->_dataValidity = vtEntry.latestValidity;
280  }
281  });
282  this->_versionNumber = {};
283  }
284  else {
285  // push-type read transfer: received value is in _queueValue (cf. readQueue continuation in constructor)
286  this->buffer_2D[0].swap(_queueValue.value);
287  this->_versionNumber = _queueValue.version;
288  this->_dataValidity = _queueValue.validity;
289  }
290  }
