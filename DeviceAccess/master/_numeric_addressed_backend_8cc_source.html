21  NumericAddressedBackend::NumericAddressedBackend(
22  std::string mapFileName, std::unique_ptr<NumericAddressedRegisterCatalogue> registerMapPointer)
23  : _registerMapPointer(std::move(registerMapPointer)), _registerMap(*_registerMapPointer) {
24  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(getRegisterAccessor_impl);
25  if(mapFileName != "") {
26  MapFileParser parser;
27  std::tie(_registerMap, _metadataCatalogue) = parser.parse(mapFileName);
28 
29  // create all the interrupt dispatchers that are described in the map file
30  for(auto& interruptController : _registerMap.getListOfInterrupts()) {
31  // interruptController is a pair<int, set<int>>, containing the controller number and a set of associated interrupts
32  for(auto interruptNumber : interruptController.second) {
33  _interruptDispatchers[{interruptController.first, interruptNumber}] =
34  boost::make_shared<NumericAddressedInterruptDispatcher>();
35  }
36  }
37  }
38  }

42  NumericAddressedRegisterInfo NumericAddressedBackend::getRegisterInfo(const RegisterPath& registerPathName) {
43  if(!registerPathName.startsWith(numeric_address::BAR)) {
44  return _registerMap.getBackendRegister(registerPathName);
45  }
46  else {
48 
49  auto components = registerPathName.getComponents();
50  if(components.size() != 3) {
51  throw ChimeraTK::logic_error("Illegal numeric address: '" + (registerPathName) + "'");
52  }
53  auto bar = std::stoi(components[1]);
54  size_t pos = components[2].find_first_of("*");
55  auto address = std::stoi(components[2].substr(0, pos));
56  size_t nBytes;
57  if(pos != std::string::npos) {
58  nBytes = std::stoi(components[2].substr(pos + 1));
59  }
60  else {
61  nBytes = sizeof(int32_t);
62  }
63  auto nElements = nBytes / sizeof(int32_t);
64  if(nBytes == 0 || nBytes % sizeof(int32_t) != 0) {
65  throw ChimeraTK::logic_error("Illegal numeric address: '" + (registerPathName) + "'");
66  }
67  return NumericAddressedRegisterInfo(registerPathName, nElements, address, nBytes, bar);
68  }
69  }

106  template<typename UserType>
107  boost::shared_ptr<NDRegisterAccessor<UserType>> NumericAddressedBackend::getRegisterAccessor_impl(
108  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
109  if(flags.has(AccessMode::wait_for_new_data)) {
110  // get the interrupt information from the map file
111  auto registerInfo = _registerMap.getBackendRegister(registerPathName);
112  if(!registerInfo.getSupportedAccessModes().has(AccessMode::wait_for_new_data)) {
113  throw ChimeraTK::logic_error(
114  "Register " + registerPathName + " does not support AccessMode::wait_for_new_data.");
115  }
116 
117  auto interruptDispatcher =
118  _interruptDispatchers[{registerInfo.interruptCtrlNumber, registerInfo.interruptNumber}];
119  assert(interruptDispatcher);
120  auto newSubscriber = interruptDispatcher->subscribe<UserType>(
121  boost::dynamic_pointer_cast<NumericAddressedBackend>(shared_from_this()), registerPathName, numberOfWords,
122  wordOffsetInRegister, flags);
123  // The new subscriber might already be activated. Hence the exception backend is already set by the interrupt dispatcher.
124  startInterruptHandlingThread(registerInfo.interruptCtrlNumber, registerInfo.interruptNumber);
125  return newSubscriber;
126  }
127  else {
128  return getSyncRegisterAccessor<UserType>(registerPathName, numberOfWords, wordOffsetInRegister, flags);
129  }
130  }

134  template<typename UserType>
135  boost::shared_ptr<NDRegisterAccessor<UserType>> NumericAddressedBackend::getSyncRegisterAccessor(
136  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
137  boost::shared_ptr<NDRegisterAccessor<UserType>> accessor;
138  // obtain register info
139  auto registerInfo = getRegisterInfo(registerPathName);
140 
141  // 1D or scalar register
142  if(registerInfo.getNumberOfDimensions() <= 1) {
143  if((registerInfo.channels.front().dataType == NumericAddressedRegisterInfo::Type::FIXED_POINT) ||
144  (registerInfo.channels.front().dataType == NumericAddressedRegisterInfo::Type::VOID)) {
145  if(flags.has(AccessMode::raw)) {
146  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
147  new NumericAddressedBackendRegisterAccessor<UserType, FixedPointConverter, true>(
148  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
149  }
150  else {
151  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
152  new NumericAddressedBackendRegisterAccessor<UserType, FixedPointConverter, false>(
153  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
154  }
155  }
156  else if(registerInfo.channels.front().dataType == NumericAddressedRegisterInfo::Type::IEEE754) {
157  if(flags.has(AccessMode::raw)) {
158  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
159  new NumericAddressedBackendRegisterAccessor<UserType, IEEE754_SingleConverter, true>(
160  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
161  }
162  else {
163  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
164  new NumericAddressedBackendRegisterAccessor<UserType, IEEE754_SingleConverter, false>(
165  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
166  }
167  }
168  else {
169  throw ChimeraTK::logic_error("NumericAddressedBackend:: trying to get "
170  "accessor for unsupported data type");
171  }
172  }
173  // 2D multiplexed register
174  else {
175  flags.checkForUnknownFlags({});
176  if(registerInfo.channels.front().dataType == NumericAddressedRegisterInfo::Type::IEEE754) {
177  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
178  new NumericAddressedBackendMuxedRegisterAccessor<UserType, IEEE754_SingleConverter>(
179  registerPathName, numberOfWords, wordOffsetInRegister, shared_from_this()));
180  }
181  else {
182  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
183  new NumericAddressedBackendMuxedRegisterAccessor<UserType, FixedPointConverter>(
184  registerPathName, numberOfWords, wordOffsetInRegister, shared_from_this()));
185  }
186  }
187 
188  accessor->setExceptionBackend(shared_from_this());
189  return accessor;
190  }
