19  NumericAddressedBackend::NumericAddressedBackend(std::string mapFileName) {
20  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(getRegisterAccessor_impl);
21  if(mapFileName != "") {
22  MapFileParser parser;
23  _registerMap = parser.parse(mapFileName);
24  _catalogue = _registerMap->getRegisterCatalogue();
25 
26  // create all the interrupt dispatchers that are described in the map file
27  for(auto& interruptController : _registerMap->getListOfInterrupts()) {
28  // interruptController is a pair<int, set<int>>, containing the controller number and a set of associated interrupts
29  for(auto interruptNumber : interruptController.second) {
30  _interruptDispatchers[{interruptController.first, interruptNumber}] =
31  boost::make_shared<NumericAddressedInterruptDispatcher>();
32  }
33  }
34  }
35  else {
36  _registerMap = boost::shared_ptr<RegisterInfoMap>();
37  }
38  }

42  boost::shared_ptr<RegisterInfoMap::RegisterInfo> NumericAddressedBackend::getRegisterInfo(
43  const RegisterPath& registerPathName) {
44  if(!registerPathName.startsWith(numeric_address::BAR)) {
45  boost::shared_ptr<RegisterInfo> info = _catalogue.getRegister(registerPathName);
46  return boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(info);
47  }
48  else {
49  auto components = registerPathName.getComponents();
50  if(components.size() != 3) {
51  throw ChimeraTK::logic_error("Illegal numeric address: '" + (registerPathName) + "'");
52  }
53  auto bar = std::stoi(components[1]);
54  size_t pos = components[2].find_first_of("*");
55  auto address = std::stoi(components[2].substr(0, pos));
56  size_t nBytes;
57  if(pos != std::string::npos) {
58  nBytes = std::stoi(components[2].substr(pos + 1));
59  }
60  else {
61  nBytes = sizeof(int32_t);
62  }
63  auto nElements = nBytes / sizeof(int32_t);
64  if(nBytes == 0 || nBytes % sizeof(int32_t) != 0) {
65  throw ChimeraTK::logic_error("Illegal numeric address: '" + (registerPathName) + "'");
66  }
67  return boost::make_shared<RegisterInfoMap::RegisterInfo>(registerPathName, nElements, address, nBytes, bar);
68  }
69  }

100  void NumericAddressedBackend::checkRegister(const std::string& regName, const std::string& regModule, size_t dataSize,
101  uint32_t addRegOffset, uint32_t& retDataSize, uint32_t& retRegOff, uint8_t& retRegBar) const {
102  RegisterInfoMap::RegisterInfo registerInfo;
103  _registerMap->getRegisterInfo(regName, registerInfo, regModule);
104  if(addRegOffset % 4) {
105  throw ChimeraTK::logic_error("Register offset must be divisible by 4");
106  }
107  if(dataSize) {
108  if(dataSize % 4) {
109  throw ChimeraTK::logic_error("Data size must be divisible by 4");
110  }
111  if(dataSize > registerInfo.nBytes - addRegOffset) {
112  throw ChimeraTK::logic_error("Data size exceed register size");
113  }
114  retDataSize = dataSize;
115  }
116  else {
117  retDataSize = registerInfo.nBytes;
118  }
119  retRegBar = registerInfo.bar;
120  retRegOff = registerInfo.address + addRegOffset;
121  }

125  template<typename UserType>
126  boost::shared_ptr<NDRegisterAccessor<UserType>> NumericAddressedBackend::getRegisterAccessor_impl(
127  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
128  if(flags.has(AccessMode::wait_for_new_data)) {
129  // get the interrupt information from the map file
130  boost::shared_ptr<RegisterInfo> info = getRegisterInfo(registerPathName);
131  auto registerInfo = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(info);
132  if(!registerInfo->getSupportedAccessModes().has(AccessMode::wait_for_new_data)) {
133  throw ChimeraTK::logic_error(
134  "Register " + registerPathName + " does not support AccessMode::wait_for_new_data.");
135  }
136 
137  auto interruptDispatcher =
138  _interruptDispatchers[{registerInfo->interruptCtrlNumber, registerInfo->interruptNumber}];
139  assert(interruptDispatcher);
140  auto newSubscriber = interruptDispatcher->subscribe<UserType>(
141  boost::dynamic_pointer_cast<NumericAddressedBackend>(shared_from_this()), registerPathName, numberOfWords,
142  wordOffsetInRegister, flags);
143  // The new subsciber might already be activated. Hence the exception backend is already set by the interrupt dispatcher.
144  startInterruptHandlingThread(registerInfo->interruptCtrlNumber, registerInfo->interruptNumber);
145  return newSubscriber;
146  }
147  else {
148  return getSyncRegisterAccessor<UserType>(registerPathName, numberOfWords, wordOffsetInRegister, flags);
149  }
150  }

154  template<typename UserType>
155  boost::shared_ptr<NDRegisterAccessor<UserType>> NumericAddressedBackend::getSyncRegisterAccessor(
156  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
157  boost::shared_ptr<NDRegisterAccessor<UserType>> accessor;
158  // obtain register info
159  boost::shared_ptr<RegisterInfo> info = getRegisterInfo(registerPathName);
160  auto registerInfo = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(info);
161 
162  // 1D or scalar register
163  if(info->getNumberOfDimensions() <= 1) {
164  if((registerInfo->dataType == RegisterInfoMap::RegisterInfo::Type::FIXED_POINT) ||
165  (registerInfo->dataType == RegisterInfoMap::RegisterInfo::Type::VOID)) {
166  if(flags.has(AccessMode::raw)) {
167  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
168  new NumericAddressedBackendRegisterAccessor<UserType, FixedPointConverter, true>(
169  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
170  }
171  else {
172  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
173  new NumericAddressedBackendRegisterAccessor<UserType, FixedPointConverter, false>(
174  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
175  }
176  }
177  else if(registerInfo->dataType == RegisterInfoMap::RegisterInfo::Type::IEEE754) {
178  if(flags.has(AccessMode::raw)) {
179  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
180  new NumericAddressedBackendRegisterAccessor<UserType, IEEE754_SingleConverter, true>(
181  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
182  }
183  else {
184  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
185  new NumericAddressedBackendRegisterAccessor<UserType, IEEE754_SingleConverter, false>(
186  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
187  }
188  }
189  else {
190  throw ChimeraTK::logic_error("NumericAddressedBackend:: trying to get "
191  "accessor for unsupported data type");
192  }
193  }
194  // 2D multiplexed register
195  else {
196  flags.checkForUnknownFlags({});
197  accessor =
198  boost::shared_ptr<NDRegisterAccessor<UserType>>(new NumericAddressedBackendMuxedRegisterAccessor<UserType>(
199  registerPathName, numberOfWords, wordOffsetInRegister, shared_from_this()));
200  }
201 
202  accessor->setExceptionBackend(shared_from_this());
203  return accessor;
204  }
