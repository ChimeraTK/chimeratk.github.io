31  boost::shared_ptr<RegisterInfoMap::RegisterInfo> NumericAddressedBackend::getRegisterInfo(
32  const RegisterPath& registerPathName) {
33  if(!registerPathName.startsWith(numeric_address::BAR)) {
34  boost::shared_ptr<RegisterInfo> info = _catalogue.getRegister(registerPathName);
35  return boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(info);
36  }
37  else {
38  auto components = registerPathName.getComponents();
39  if(components.size() != 3) {
40  throw ChimeraTK::logic_error("Illegal numeric address: '" + (registerPathName) + "'");
41  }
42  auto bar = std::stoi(components[1]);
43  size_t pos = components[2].find_first_of("*");
44  auto address = std::stoi(components[2].substr(0, pos));
45  size_t nBytes;
46  if(pos != std::string::npos) {
47  nBytes = std::stoi(components[2].substr(pos + 1));
48  }
49  else {
50  nBytes = sizeof(int32_t);
51  }
52  auto nElements = nBytes / sizeof(int32_t);
53  if(nBytes == 0 || nBytes % sizeof(int32_t) != 0) {
54  throw ChimeraTK::logic_error("Illegal numeric address: '" + (registerPathName) + "'");
55  }
56  return boost::make_shared<RegisterInfoMap::RegisterInfo>(registerPathName, nElements, address, nBytes, bar);
57  }
58  }

89  void NumericAddressedBackend::checkRegister(const std::string& regName, const std::string& regModule, size_t dataSize,
90  uint32_t addRegOffset, uint32_t& retDataSize, uint32_t& retRegOff, uint8_t& retRegBar) const {
91  RegisterInfoMap::RegisterInfo registerInfo;
92  _registerMap->getRegisterInfo(regName, registerInfo, regModule);
93  if(addRegOffset % 4) {
94  throw ChimeraTK::logic_error("Register offset must be divisible by 4");
95  }
96  if(dataSize) {
97  if(dataSize % 4) {
98  throw ChimeraTK::logic_error("Data size must be divisible by 4");
99  }
100  if(dataSize > registerInfo.nBytes - addRegOffset) {
101  throw ChimeraTK::logic_error("Data size exceed register size");
102  }
103  retDataSize = dataSize;
104  }
105  else {
106  retDataSize = registerInfo.nBytes;
107  }
108  retRegBar = registerInfo.bar;
109  retRegOff = registerInfo.address + addRegOffset;
110  }

114  template<typename UserType>
115  boost::shared_ptr<NDRegisterAccessor<UserType>> NumericAddressedBackend::getRegisterAccessor_impl(
116  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
117  boost::shared_ptr<NDRegisterAccessor<UserType>> accessor;
118  // obtain register info
119  boost::shared_ptr<RegisterInfo> info = getRegisterInfo(registerPathName);
120  auto registerInfo = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(info);
121 
122  // 1D or scalar register
123  if(info->getNumberOfDimensions() <= 1) {
124  if(registerInfo->dataType == RegisterInfoMap::RegisterInfo::Type::FIXED_POINT) {
125  if(flags.has(AccessMode::raw)) {
126  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
127  new NumericAddressedBackendRegisterAccessor<UserType, FixedPointConverter, true>(
128  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
129  }
130  else {
131  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
132  new NumericAddressedBackendRegisterAccessor<UserType, FixedPointConverter, false>(
133  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
134  }
135  }
136  else if(registerInfo->dataType == RegisterInfoMap::RegisterInfo::Type::IEEE754) {
137  if(flags.has(AccessMode::raw)) {
138  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
139  new NumericAddressedBackendRegisterAccessor<UserType, IEEE754_SingleConverter, true>(
140  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
141  }
142  else {
143  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
144  new NumericAddressedBackendRegisterAccessor<UserType, IEEE754_SingleConverter, false>(
145  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
146  }
147  }
148  else {
149  throw ChimeraTK::logic_error("NumericAddressedBackend:: trying to get "
150  "accessor for unsupported data type");
151  }
152  }
153  // 2D multiplexed register
154  else {
155  flags.checkForUnknownFlags({});
156  accessor =
157  boost::shared_ptr<NDRegisterAccessor<UserType>>(new NumericAddressedBackendMuxedRegisterAccessor<UserType>(
158  registerPathName, numberOfWords, wordOffsetInRegister, shared_from_this()));
159  }
160 
161  accessor->setExceptionBackend(shared_from_this());
162  return accessor;
163  }
