31  boost::shared_ptr<RegisterInfoMap::RegisterInfo> NumericAddressedBackend::getRegisterInfo(
32  const RegisterPath& registerPathName) {
33  if(!registerPathName.startsWith(numeric_address::BAR)) {
34  boost::shared_ptr<RegisterInfo> info = _catalogue.getRegister(registerPathName);
35  return boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(info);
36  }
37  else {
38  auto components = registerPathName.getComponents();
39  if(components.size() != 3) {
40  throw ChimeraTK::logic_error("Illegal numeric address: '" + (registerPathName) + "'");
41  }
42  auto bar = std::stoi(components[1]);
43  size_t pos = components[2].find_first_of("*");
44  auto address = std::stoi(components[2].substr(0, pos));
45  size_t nBytes;
46  if(pos != std::string::npos) {
47  nBytes = std::stoi(components[2].substr(pos + 1));
48  }
49  else {
50  nBytes = sizeof(int32_t);
51  }
52  auto nElements = nBytes / sizeof(int32_t);
53  if(nBytes == 0 || nBytes % sizeof(int32_t) != 0) {
54  throw ChimeraTK::logic_error("Illegal numeric address: '" + (registerPathName) + "'");
55  }
56  return boost::make_shared<RegisterInfoMap::RegisterInfo>(registerPathName, nElements, address, nBytes, bar);
57  }
58  }

97  void NumericAddressedBackend::checkRegister(const std::string& regName, const std::string& regModule, size_t dataSize,
98  uint32_t addRegOffset, uint32_t& retDataSize, uint32_t& retRegOff, uint8_t& retRegBar) const {
99  RegisterInfoMap::RegisterInfo registerInfo;
100  _registerMap->getRegisterInfo(regName, registerInfo, regModule);
101  if(addRegOffset % 4) {
102  throw ChimeraTK::logic_error("Register offset must be divisible by 4");
103  }
104  if(dataSize) {
105  if(dataSize % 4) {
106  throw ChimeraTK::logic_error("Data size must be divisible by 4");
107  }
108  if(dataSize > registerInfo.nBytes - addRegOffset) {
109  throw ChimeraTK::logic_error("Data size exceed register size");
110  }
111  retDataSize = dataSize;
112  }
113  else {
114  retDataSize = registerInfo.nBytes;
115  }
116  retRegBar = registerInfo.bar;
117  retRegOff = registerInfo.address + addRegOffset;
118  }

122  template<typename UserType>
123  boost::shared_ptr<NDRegisterAccessor<UserType>> NumericAddressedBackend::getRegisterAccessor_impl(
124  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
125  boost::shared_ptr<NDRegisterAccessor<UserType>> accessor;
126  // obtain register info
127  boost::shared_ptr<RegisterInfo> info = getRegisterInfo(registerPathName);
128  auto registerInfo = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(info);
129 
130  // 1D or scalar register
131  if(info->getNumberOfDimensions() <= 1) {
132  if(registerInfo->dataType == RegisterInfoMap::RegisterInfo::Type::FIXED_POINT) {
133  if(flags.has(AccessMode::raw)) {
134  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
135  new NumericAddressedBackendRegisterAccessor<UserType, FixedPointConverter, true>(
136  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
137  }
138  else {
139  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
140  new NumericAddressedBackendRegisterAccessor<UserType, FixedPointConverter, false>(
141  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
142  }
143  }
144  else if(registerInfo->dataType == RegisterInfoMap::RegisterInfo::Type::IEEE754) {
145  if(flags.has(AccessMode::raw)) {
146  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
147  new NumericAddressedBackendRegisterAccessor<UserType, IEEE754_SingleConverter, true>(
148  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
149  }
150  else {
151  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
152  new NumericAddressedBackendRegisterAccessor<UserType, IEEE754_SingleConverter, false>(
153  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
154  }
155  }
156  else {
157  throw ChimeraTK::logic_error("NumericAddressedBackend:: trying to get "
158  "accessor for unsupported data type");
159  }
160  }
161  // 2D multiplexed register
162  else {
163  accessor =
164  boost::shared_ptr<NDRegisterAccessor<UserType>>(new NumericAddressedBackendMuxedRegisterAccessor<UserType>(
165  registerPathName, numberOfWords, wordOffsetInRegister, shared_from_this()));
166  }
167  return accessor;
168  }
