20  NumericAddressedBackend::NumericAddressedBackend(
21  const std::string& mapFileName, std::unique_ptr<NumericAddressedRegisterCatalogue> registerMapPointer)
22  : _registerMapPointer(std::move(registerMapPointer)), _registerMap(*_registerMapPointer) {
23  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(getRegisterAccessor_impl);
24  if(!mapFileName.empty()) {
25  MapFileParser parser;
26  std::tie(_registerMap, _metadataCatalogue) = parser.parse(mapFileName);
27 
28  // Add information about interrupt controller handlers from the map file meta data to the factory.
29  for(auto const& metaDataEntry : _metadataCatalogue) {
30  auto const& key = metaDataEntry.first;
31  if(key[0] == '!') {
32  auto jkey = nlohmann::json::parse(std::string({++key.begin(), key.end()})); // key without the !
33  auto interruptId = jkey.get<std::vector<uint32_t>>();
34 
35  auto jdescriptor = nlohmann::json::parse(metaDataEntry.second);
36  auto controllerType = jdescriptor.begin().key();
37  auto controllerDescription = jdescriptor.front().dump();
38  _interruptControllerHandlerFactory.addControllerDescription(
39  interruptId, controllerType, controllerDescription);
40  }
41  }
42 
43  // Create the containers for the AsyncDomains. We have the AsyncDomainsContainer for the
44  // base class (shared) pointer. It is filled dynamically and has a lock because it is only accessed when creating
45  // accessors and sending exceptions.
46  _asyncDomainsContainer = std::make_unique<AsyncDomainsContainer<uint32_t>>();
47  // There also is the container for the implementations. It is only accessed by the const reference after
48  // filling it here, so we don't need a lock for all interrupts during interrupt distribution.
49  for(const auto& interruptID : _registerMap.getListOfInterrupts()) {
50  _asyncDomainImplsNonConst.try_emplace(interruptID.front(), std::make_unique<AsyncDomainPtr_t>());
51  }
52  }
53  }

57  NumericAddressedRegisterInfo NumericAddressedBackend::getRegisterInfo(const RegisterPath& registerPathName) {
58  if(!registerPathName.startsWith(numeric_address::BAR())) {
59  return _registerMap.getBackendRegister(registerPathName);
60  }
62 
63  auto components = registerPathName.getComponents();
64  if(components.size() != 3) {
65  throw ChimeraTK::logic_error("Illegal numeric address: '" + (registerPathName) + "'");
66  }
67  auto bar = std::stoi(components[1]);
68  size_t pos = components[2].find_first_of('*');
69  auto address = std::stoi(components[2].substr(0, pos));
70  size_t nBytes;
71  if(pos != std::string::npos) {
72  nBytes = std::stoi(components[2].substr(pos + 1));
73  }
74  else {
75  nBytes = sizeof(int32_t);
76  }
77  auto nElements = nBytes / sizeof(int32_t);
78  if(nBytes == 0 || nBytes % sizeof(int32_t) != 0) {
79  throw ChimeraTK::logic_error("Illegal numeric address: '" + (registerPathName) + "'");
80  }
81  return NumericAddressedRegisterInfo(registerPathName, nElements, address, nBytes, bar);
82  }

127  template<typename UserType>
128  boost::shared_ptr<NDRegisterAccessor<UserType>> NumericAddressedBackend::getRegisterAccessor_impl(
129  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
130  if(flags.has(AccessMode::wait_for_new_data)) {
131  // get the interrupt information from the map file
132  auto registerInfo = _registerMap.getBackendRegister(registerPathName);
133  if(!registerInfo.getSupportedAccessModes().has(AccessMode::wait_for_new_data)) {
134  throw ChimeraTK::logic_error(
135  "Register " + registerPathName + " does not support AccessMode::wait_for_new_data.");
136  }
137 
138  auto asyncDomain = _asyncDomainImpls.at(registerInfo.interruptId.front())->lock();
139 
140  if(!asyncDomain) {
141  auto creatorFct = [&](boost::shared_ptr<AsyncDomain> domain) {
142  return boost::make_shared<TriggerDistributor>(shared_from_this(), &_interruptControllerHandlerFactory,
143  std::vector<uint32_t>({registerInfo.interruptId.front()}), nullptr, domain);
144  };
145  asyncDomain = boost::make_shared<AsyncDomainImpl<TriggerDistributor, std::nullptr_t>>(creatorFct);
146  *_asyncDomainImpls.at(registerInfo.interruptId.front()) = asyncDomain;
147  auto& domainsContainer = dynamic_cast<AsyncDomainsContainer<uint32_t>&>(*_asyncDomainsContainer);
148  domainsContainer.addAsyncDomain(registerInfo.interruptId.front(), asyncDomain);
149  if(_asyncIsActive) {
150  auto subscriptionDone = activateSubscription(registerInfo.interruptId.front());
151  // wait until the backend reports that the subscription is ready before proceeding with the creation of the
152  // accessor, which will poll the initial value.
153  subscriptionDone.wait();
154  asyncDomain->activate(nullptr);
155  }
156  }
157 
158  auto newSubscriber =
159  asyncDomain->subscribe<UserType>(registerPathName, numberOfWords, wordOffsetInRegister, flags);
160  // The new subscriber might already be activated. Hence the exception backend is already set during creation.
161  return newSubscriber;
162  }
163  return getSyncRegisterAccessor<UserType>(registerPathName, numberOfWords, wordOffsetInRegister, flags);
164  }

168  template<typename UserType>
169  boost::shared_ptr<NDRegisterAccessor<UserType>> NumericAddressedBackend::getSyncRegisterAccessor(
170  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
171  boost::shared_ptr<NDRegisterAccessor<UserType>> accessor;
172  // obtain register info
173  auto registerInfo = getRegisterInfo(registerPathName);
174 
175  // 1D or scalar register
176  if(registerInfo.getNumberOfDimensions() <= 1) {
177  if((registerInfo.channels.front().dataType == NumericAddressedRegisterInfo::Type::FIXED_POINT) ||
178  (registerInfo.channels.front().dataType == NumericAddressedRegisterInfo::Type::VOID)) {
179  if(flags.has(AccessMode::raw)) {
180  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
181  new NumericAddressedBackendRegisterAccessor<UserType, FixedPointConverter, true>(
182  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
183  }
184  else {
185  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
186  new NumericAddressedBackendRegisterAccessor<UserType, FixedPointConverter, false>(
187  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
188  }
189  }
190  else if(registerInfo.channels.front().dataType == NumericAddressedRegisterInfo::Type::IEEE754) {
191  if(flags.has(AccessMode::raw)) {
192  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
193  new NumericAddressedBackendRegisterAccessor<UserType, IEEE754_SingleConverter, true>(
194  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
195  }
196  else {
197  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
198  new NumericAddressedBackendRegisterAccessor<UserType, IEEE754_SingleConverter, false>(
199  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
200  }
201  }
202  else if(registerInfo.channels.front().dataType == NumericAddressedRegisterInfo::Type::ASCII) {
203  if constexpr(!std::is_same<UserType, std::string>::value) {
204  throw ChimeraTK::logic_error("NumericAddressedBackend: ASCII data must be read with std::string UserType.");
205  }
206  else {
207  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(new NumericAddressedBackendASCIIAccessor(
208  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
209  }
210  }
211  else {
212  throw ChimeraTK::logic_error("NumericAddressedBackend: trying to get accessor for unsupported data type");
213  }
214  }
215  // 2D multiplexed register
216  else {
217  flags.checkForUnknownFlags({});
218  if(registerInfo.channels.front().dataType == NumericAddressedRegisterInfo::Type::IEEE754) {
219  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
220  new NumericAddressedBackendMuxedRegisterAccessor<UserType, IEEE754_SingleConverter>(
221  registerPathName, numberOfWords, wordOffsetInRegister, shared_from_this()));
222  }
223  else {
224  accessor = boost::shared_ptr<NDRegisterAccessor<UserType>>(
225  new NumericAddressedBackendMuxedRegisterAccessor<UserType, FixedPointConverter>(
226  registerPathName, numberOfWords, wordOffsetInRegister, shared_from_this()));
227  }
228  }
229 
230  accessor->setExceptionBackend(shared_from_this());
231  return accessor;
232  }

236  void NumericAddressedBackend::activateAsyncRead() noexcept {
237  _asyncIsActive = true;
238  for(const auto& it : _asyncDomainImpls) {
239  auto asyncDomain = it.second->lock();
240  if(asyncDomain) {
241  auto subscriptionDone = activateSubscription(it.first);
242  // Wait until the backends reports that the subscription is complete (typically set from inside another thread)
243  // before polling the initial values when activating the async domain. This is necessary to make sure we don't
244  // miss an update that came in after polling the initial value.
245  subscriptionDone.wait();
246  asyncDomain->activate(nullptr);
247  }
248  }
249  }
