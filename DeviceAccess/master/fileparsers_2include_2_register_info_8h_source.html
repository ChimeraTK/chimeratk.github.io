236  inline RegisterInfo::DataDescriptor::DataDescriptor(DataType type) {
237  if(type.isNumeric()) {
238  _fundamentalType = FundamentalType::numeric;
239  }
240  else {
241  _fundamentalType = FundamentalType::string;
242  }
243  _isIntegral = type.isIntegral();
244  _isSigned = type.isSigned();
245 
246  // the defaults
247  _nDigits = 0; // invalid for non handled types
248  _nFractionalDigits = 0; // 0 for integers
249 
250  if(type == DataType::int8) { // 8 bit signed
251  _nDigits = 4; // -127 .. 128
252  }
253  if(type == DataType::uint8) { // 8 bit unsigned
254  _nDigits = 3; // 0..256
255  }
256  else if(type == DataType::int16) { // 16 bit signed
257  _nDigits = 6; // -32000 .. 32000 (approx)
258  }
259  else if(type == DataType::uint16) { // 16 bit unsigned
260  _nDigits = 6; // 0 .. 65000 (approx)
261  }
262  else if(type == DataType::int32) { // 32 bit signed
263  _nDigits = 11; // -2e9 .. 2e9 (approx)
264  }
265  else if(type == DataType::uint32) { // 32 bit unsigned
266  _nDigits = 10; // 0 .. 4e9 (approx)
267  }
268  else if(type == DataType::int64) { // 64 bit signed
269  _nDigits = 20; // -9e18 .. 9e18 (approx)
270  }
271  else if(type == DataType::uint64) { // 64 bit unsigned
272  _nDigits = 20; // 0 .. 2e19 (approx)
273  }
274  else if(type == DataType::float32) { // 32 bit float IEEE 754
275  _nDigits = 3 + 45; // see RegisterInfoMap::RegisterInfo::RegisterInfo
276  _nFractionalDigits = 45;
277  }
278  else if(type == DataType::float64) { // 64 bit float IEEE 754
279  _nDigits = 3 + 325; // see RegisterInfoMap::RegisterInfo::RegisterInfo
280  _nFractionalDigits = 325;
281  }
282  }
