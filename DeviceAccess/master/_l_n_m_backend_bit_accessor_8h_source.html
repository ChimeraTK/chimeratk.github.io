20  template<typename UserType>
21  class LNMBackendBitAccessor : public NDRegisterAccessor<UserType> {
22  public:
23  LNMBackendBitAccessor(boost::shared_ptr<DeviceBackend> dev, const RegisterPath& registerPathName,
24  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
25  : NDRegisterAccessor<UserType>(registerPathName, flags), _registerPathName(registerPathName),
26  _fixedPointConverter(registerPathName, 32, 0, 1) {
27  // check for unknown flags
28  flags.checkForUnknownFlags({AccessMode::raw});
29  // check for illegal parameter combinations
30  if(flags.has(AccessMode::raw)) {
31  throw ChimeraTK::logic_error("LNMBackendBitAccessor: raw access not supported!");
32  }
33  _dev = boost::dynamic_pointer_cast<LogicalNameMappingBackend>(dev);
34  // copy the register info and create the internal accessors, if needed
35  auto info = _dev->_catalogue_mutable.getBackendRegister(_registerPathName);
36 
37  // check for incorrect usage of this accessor
38  if(info.targetType != LNMBackendRegisterInfo::TargetType::BIT) {
39  throw ChimeraTK::logic_error("LNMBackendBitAccessor used for wrong register type."); // LCOV_EXCL_LINE
40  // (impossible to test...)
41  }
42  if(wordOffsetInRegister != 0) {
43  throw ChimeraTK::logic_error("LNMBackendBitAccessors cannot have a word offset.");
44  }
45  if(numberOfWords > 1) {
46  throw ChimeraTK::logic_error("LNMBackendBitAccessors must have size 1.");
47  // The case that the target size actually is 1 if numberOfWords == 0 cannot be checked here.
48  // 0 is allowed. It is tested after the target has created the accessor with 0 as length parameter.
49  }
50 
51  // get target device and accessor
52  std::string devName = info.deviceName;
53  boost::shared_ptr<DeviceBackend> targetDevice;
54  if(devName != "this") {
55  targetDevice = _dev->_devices[devName];
56  }
57  else {
58  targetDevice = dev;
59  }
60  {
61  std::unique_lock<std::mutex> l{_dev->sharedAccessorMap_mutex};
62  auto& map = boost::fusion::at_key<uint64_t>(_dev->sharedAccessorMap.table);
63  // we need an identifier of the device in the key, in case the logical name mapping accesses more than one
64  // device with same set of register names
65  LogicalNameMappingBackend::AccessorKey key(targetDevice.get(), RegisterPath(info.registerName));
66  auto it = map.find(key);
67  // Obtain accessor if not found in the map or if weak pointer has expired
68  // Note: we must not use boost::weak_ptr::expired() here, because we have to check the status and obtain the
69  // accessor in one atomic step.
70  if(it == map.end() || (_accessor = map[key].accessor.lock()) == nullptr) {
71  _accessor = targetDevice->getRegisterAccessor<uint64_t>(key.second, numberOfWords, wordOffsetInRegister, {});
72  if(_accessor->getNumberOfSamples() != 1) {
73  throw ChimeraTK::logic_error("LNMBackendBitAccessors only work with registers of size 1");
74  }
75  map[key].accessor = _accessor;
76  }
77  lock = std::unique_lock<std::recursive_mutex>(map[key].mutex, std::defer_lock);
78  }
79  // allocate and initialise the buffer
80  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
81  NDRegisterAccessor<UserType>::buffer_2D[0].resize(1);
82  NDRegisterAccessor<UserType>::buffer_2D[0][0] = numericToUserType<UserType>(false);
83  // set the bit mask
84  _bitMask = 1 << info.bit;
85  }
86 
87  void doReadTransferSynchronously() override {
88  assert(lock.owns_lock());
89  _accessor->readTransfer();
90  }
91 
92  bool doWriteTransfer(ChimeraTK::VersionNumber) override {
93  assert(lock.owns_lock());
94  return _accessor->writeTransfer(_versionNumberTemp);
95  }
96 
97  void doPreRead(TransferType type) override {
98  lock.lock();
99  _accessor->preRead(type);
100  }
101 
102  void doPostRead(TransferType type, bool hasNewData) override {
103  auto unlock = cppext::finally([this] { this->lock.unlock(); });
104  _accessor->postRead(type, hasNewData);
105  if(!hasNewData) return;
106  if(_accessor->accessData(0) & _bitMask) {
107  NDRegisterAccessor<UserType>::buffer_2D[0][0] = numericToUserType<UserType>(true);
108  }
109  else {
110  NDRegisterAccessor<UserType>::buffer_2D[0][0] = numericToUserType<UserType>(false);
111  }
112  this->_versionNumber = {}; // VersionNumber needs to be decoupled from target accessor
113  this->_dataValidity = _accessor->dataValidity();
114  }
115 
116  void doPreWrite(TransferType type, VersionNumber) override {
117  lock.lock();
118 
119  if(!_fixedPointConverter.toRaw<UserType>(NDRegisterAccessor<UserType>::buffer_2D[0][0])) {
120  _accessor->accessData(0) &= ~(_bitMask);
121  }
122  else {
123  _accessor->accessData(0) |= _bitMask;
124  }
125 
126  _versionNumberTemp = {};
127  _accessor->setDataValidity(this->_dataValidity);
128  _accessor->preWrite(type, _versionNumberTemp);
129  }
130 
131  void doPostWrite(TransferType type, VersionNumber) override {
132  auto unlock = cppext::finally([this] { this->lock.unlock(); });
133  _accessor->postWrite(type, _versionNumberTemp);
134  }
135 
136  bool mayReplaceOther(const boost::shared_ptr<TransferElement const>& other) const override {
137  auto rhsCasted = boost::dynamic_pointer_cast<const LNMBackendBitAccessor<UserType>>(other);
138  if(!rhsCasted) return false;
139  if(_registerPathName != rhsCasted->_registerPathName) return false;
140  if(_dev != rhsCasted->_dev) return false;
141  return true;
142  }
143 
144  bool isReadOnly() const override {
145  std::lock_guard<std::recursive_mutex> guard(*lock.mutex());
146  return _accessor->isReadOnly();
147  }
148 
149  bool isReadable() const override {
150  std::lock_guard<std::recursive_mutex> guard(*lock.mutex());
151  return _accessor->isReadable();
152  }
153 
154  bool isWriteable() const override {
155  std::lock_guard<std::recursive_mutex> guard(*lock.mutex());
156  return _accessor->isWriteable();
157  }
158 
159  void setExceptionBackend(boost::shared_ptr<DeviceBackend> exceptionBackend) override {
160  std::lock_guard<std::recursive_mutex> guard(*lock.mutex());
161  this->_exceptionBackend = exceptionBackend;
162  _accessor->setExceptionBackend(exceptionBackend);
163  }
164 
165  protected:
167  boost::shared_ptr<NDRegisterAccessor<uint64_t>> _accessor;
168 
171  std::unique_lock<std::recursive_mutex> lock;
172 
174  RegisterPath _registerPathName;
175 
179  VersionNumber _versionNumberTemp{nullptr};
180 
182  boost::shared_ptr<LogicalNameMappingBackend> _dev;
183 
187  FixedPointConverter _fixedPointConverter;
188 
190  size_t _bitMask;
191 
192  std::vector<boost::shared_ptr<TransferElement>> getHardwareAccessingElements() override {
193  std::lock_guard<std::recursive_mutex> guard(*lock.mutex());
194  return _accessor->getHardwareAccessingElements();
195  }
196 
197  std::list<boost::shared_ptr<TransferElement>> getInternalElements() override {
198  std::lock_guard<std::recursive_mutex> guard(*lock.mutex());
199  auto result = _accessor->getInternalElements();
200  result.push_front(_accessor);
201  return result;
202  }
203 
204  void replaceTransferElement(boost::shared_ptr<TransferElement> newElement) override {
205  std::lock_guard<std::recursive_mutex> guard(*lock.mutex());
206  auto casted = boost::dynamic_pointer_cast<NDRegisterAccessor<uint64_t>>(newElement);
207  if(casted && _accessor->mayReplaceOther(newElement)) {
208  _accessor = detail::createCopyDecorator<uint64_t>(casted);
209  }
210  else {
211  _accessor->replaceTransferElement(newElement);
212  }
213  _accessor->setExceptionBackend(this->_exceptionBackend);
214  }
215  };
