25  template<typename UserType>
26  class LNMBackendBitAccessor : public NDRegisterAccessor<UserType> {
27  public:
28  LNMBackendBitAccessor(boost::shared_ptr<DeviceBackend> dev, const RegisterPath& registerPathName,
29  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
30  : NDRegisterAccessor<UserType>(registerPathName, flags), _registerPathName(registerPathName),
31  _fixedPointConverter(registerPathName, 32, 0, 1) {
32  // check for unknown flags
33  flags.checkForUnknownFlags({AccessMode::raw});
34  // check for illegal parameter combinations
35  if(flags.has(AccessMode::raw)) {
36  throw ChimeraTK::logic_error("LNMBackendBitAccessor: raw access not supported!");
37  }
38  _dev = boost::dynamic_pointer_cast<LogicalNameMappingBackend>(dev);
39  // copy the register info and create the internal accessors, if needed
40  auto info = boost::static_pointer_cast<LNMBackendRegisterInfo>(
41  _dev->getRegisterCatalogue().getRegister(_registerPathName));
42  // check for incorrect usage of this accessor
43  if(info->targetType != LNMBackendRegisterInfo::TargetType::BIT) {
44  throw ChimeraTK::logic_error(
45  "LNMBackendBitAccessor used for wrong register type."); // LCOV_EXCL_LINE (impossible to test...)
46  }
47  if(wordOffsetInRegister != 0) {
48  throw ChimeraTK::logic_error("LNMBackendBitAccessors cannot have a word offset.");
49  }
50  if(numberOfWords > 1) {
51  throw ChimeraTK::logic_error("LNMBackendBitAccessors must have size 1.");
52  // The case that the target size actually is 1 if numberOfWords == 0 cannnot be checked here.
53  // 0 is allowed. It is tested after the target has created the accessor with 0 as length parameter.
54  }
55 
56  // get target device and accessor
57  std::string devName = info->deviceName;
58  boost::shared_ptr<DeviceBackend> targetDevice;
59  if(devName != "this") {
60  targetDevice = _dev->_devices[devName];
61  }
62  else {
63  targetDevice = dev;
64  }
65  auto& map = boost::fusion::at_key<uint64_t>(_dev->sharedAccessorMap.table);
66  auto it = map.find(RegisterPath(info->registerName));
67  // Obtain accessor if not found in the map or if weak pointer has expired
68  // Note: we must not use boost::weak_ptr::expired() here, because we have to check the status and obtain the
69  // accessor in one atomic step.
70  if(it == map.end() || (_accessor = map[RegisterPath(info->registerName)].accessor.lock()) == nullptr) {
71  _accessor = targetDevice->getRegisterAccessor<uint64_t>(
72  RegisterPath(info->registerName), numberOfWords, wordOffsetInRegister, {});
73  if(_accessor->getNumberOfSamples() != 1) {
74  throw ChimeraTK::logic_error("LNMBackendBitAccessors only work with registers of size 1");
75  }
76  map[RegisterPath(info->registerName)].accessor = _accessor;
77  }
78  lock = std::unique_lock<std::recursive_mutex>(map[RegisterPath(info->registerName)].mutex, std::defer_lock);
79  // allocate and initialise the buffer
80  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
81  NDRegisterAccessor<UserType>::buffer_2D[0].resize(1);
82  NDRegisterAccessor<UserType>::buffer_2D[0][0] = numericToUserType<UserType>(false);
83  // set the bit mask
84  _bitMask = 1 << info->bit;
85  }
86 
87  void doReadTransferSynchronously() override {
88  assert(lock.owns_lock());
89  _accessor->readTransfer();
90  }
91 
92  bool doWriteTransfer(ChimeraTK::VersionNumber) override {
93  assert(lock.owns_lock());
94  return _accessor->writeTransfer(_versionNumberTemp);
95  }

97  void doPreRead(TransferType type) override {
98  lock.lock();
99  _accessor->preRead(type);
100  }
101 
102  void doPostRead(TransferType type, bool hasNewData) override {
103  auto unlock = cppext::finally([this] { this->lock.unlock(); });
104  _accessor->postRead(type, hasNewData);
105  if(!hasNewData) return;
106  if(_accessor->accessData(0) & _bitMask) {
107  NDRegisterAccessor<UserType>::buffer_2D[0][0] = numericToUserType<UserType>(true);
108  }
109  else {
110  NDRegisterAccessor<UserType>::buffer_2D[0][0] = numericToUserType<UserType>(false);
111  }
112  this->_versionNumber = {}; // VersionNumber needs to be decoupled from target accessor
113  this->_dataValidity = _accessor->dataValidity();
114  }
115 
116  void doPreWrite(TransferType type, VersionNumber) override {
117  lock.lock();
118 
119  if(!_fixedPointConverter.toRaw<UserType>(NDRegisterAccessor<UserType>::buffer_2D[0][0])) {
120  _accessor->accessData(0) &= ~(_bitMask);
121  }
122  else {
123  _accessor->accessData(0) |= _bitMask;
124  }
125 
126  _versionNumberTemp = {};
127  _accessor->setDataValidity(this->_dataValidity);
128  _accessor->preWrite(type, _versionNumberTemp);
129  }
130 
131  void doPostWrite(TransferType type, VersionNumber) override {
132  auto unlock = cppext::finally([this] { this->lock.unlock(); });
133  _accessor->postWrite(type, _versionNumberTemp);
134  }

136  bool mayReplaceOther(const boost::shared_ptr<TransferElement const>& other) const override {
137  auto rhsCasted = boost::dynamic_pointer_cast<const LNMBackendBitAccessor<UserType>>(other);
138  if(!rhsCasted) return false;
139  if(_registerPathName != rhsCasted->_registerPathName) return false;
140  if(_dev != rhsCasted->_dev) return false;
141  return true;
142  }

203  void replaceTransferElement(boost::shared_ptr<TransferElement> newElement) override {
204  std::lock_guard<std::recursive_mutex> guard(*lock.mutex());
205  auto casted = boost::dynamic_pointer_cast<NDRegisterAccessor<uint64_t>>(newElement);
206  if(casted && _accessor->mayReplaceOther(newElement)) {
207  _accessor = detail::createCopyDecorator<uint64_t>(casted);
208  }
209  else {
210  _accessor->replaceTransferElement(newElement);
211  }
212  _accessor->setExceptionBackend(this->_exceptionBackend);
213  }
