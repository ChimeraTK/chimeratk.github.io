181  std::string removeExtraSeparators(std::string string, std::string sep = separator) const {
182  std::size_t pos;
183  while((pos = string.find(std::string(sep) + sep)) != std::string::npos) {
184  string.erase(pos, 1);
185  }
186  if(string.length() > 1 && string.substr(string.length() - 1, 1) == sep) string.erase(string.length() - 1, 1);

192  std::string getWithOtherSeparatorReplaced(const std::string& otherSeparator) const {
193  // no replacement needed, of other separator is empty
194  if(otherSeparator.length() == 0) return path;
195  // replace all occurrences of otherSeparator with separator
196  std::string path_other = path;
197  size_t pos = 0;
198  while((pos = path_other.find(otherSeparator)) != std::string::npos) {
199  path_other.replace(pos, 1, std::string(separator));
200  }
201  // replace duplicate separators and remove one of them. Also remove a
202  // trailing separator, if present.
203  return removeExtraSeparators(path_other);
204  }
