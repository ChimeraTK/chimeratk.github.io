90  template<typename UserType>
91  LNMBackendVariableAccessor<UserType>::LNMBackendVariableAccessor(boost::shared_ptr<DeviceBackend> dev,
92  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
93  : NDRegisterAccessor<UserType>(registerPathName, flags), _registerPathName(registerPathName),
94  _wordOffsetInRegister(wordOffsetInRegister), _flags(flags) {
95  // cast device
96  _dev = boost::dynamic_pointer_cast<LogicalNameMappingBackend>(dev);
97 
98  // obtain the register info
99  _info =
100  boost::static_pointer_cast<LNMBackendRegisterInfo>(_dev->getRegisterCatalogue().getRegister(_registerPathName));
101 
102  // check for unknown flags
103  if(_info->targetType == LNMBackendRegisterInfo::TargetType::VARIABLE) {
104  flags.checkForUnknownFlags({AccessMode::wait_for_new_data});
105  }
106  else {
107  // no flags are supported for constants
108  flags.checkForUnknownFlags({});
109  }
110 
111  // numberOfWords default to full register length
112  if(numberOfWords == 0) numberOfWords = _info->length;
113 
114  // check for illegal parameter combinations
115  if(wordOffsetInRegister + numberOfWords > _info->length) {
116  throw ChimeraTK::logic_error(
117  "Requested number of words and/or offset exceeds length of register '" + registerPathName + "'.");
118  }
119 
120  // check for incorrect usage of this accessor
121  if(_info->targetType != LNMBackendRegisterInfo::TargetType::CONSTANT &&
122  _info->targetType != LNMBackendRegisterInfo::TargetType::VARIABLE) {
123  throw ChimeraTK::logic_error(
124  "LNMBackendVariableAccessor used for wrong register type."); // LCOV_EXCL_LINE (impossible to test...)
125  }
126 
127  // if wait_for_new_data is specified, make subscription
128  if(flags.has(AccessMode::wait_for_new_data)) {
129  // alocate _queueValue buffer
130  this->_queueValue.value.resize(numberOfWords);
131 
132  std::lock_guard<std::mutex> lock(_info->valueTable_mutex);
133  callForType(_info->valueType, [&, this](auto arg) {
134  typedef decltype(arg) T;
135  auto& vtEntry = boost::fusion::at_key<T>(_info->valueTable.table);
136  // create subscription queue
137  cppext::future_queue<typename LNMBackendRegisterInfo::ValueTable<T>::QueuedValue> queue(3);
138  // place queue in map
139  vtEntry.subscriptions[this->getId()] = queue;
140  // make void-typed continuationof subscription queue, which stores the received value into the _queueValue
141  this->_readQueue = queue.template then<void>(
142  [this](const typename LNMBackendRegisterInfo::ValueTable<T>::QueuedValue& queueValue) {
143  this->_queueValue.validity = queueValue.validity;
144  this->_queueValue.version = queueValue.version;
145  for(size_t i = 0; i < queueValue.value.size(); ++i) {
146  this->_queueValue.value[i] =
147  userTypeToUserType<UserType>(queueValue.value[i + this->_wordOffsetInRegister]);
148  }
149  },
150  std::launch::deferred);
151  // put initial value into queue, if async reads activated
152  if(_dev->_asyncReadActive) {
153  queue.push({vtEntry.latestValue, vtEntry.latestValidity, vtEntry.latestVersion});
154  }
155  });
156  }
157 
158  // allocate application buffer
159  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
160  NDRegisterAccessor<UserType>::buffer_2D[0].resize(numberOfWords);
161  }

165  template<typename UserType>
166  LNMBackendVariableAccessor<UserType>::~LNMBackendVariableAccessor() {
167  if(_flags.has(AccessMode::wait_for_new_data)) {
168  // unsubscribe the update queue
169  std::lock_guard<std::mutex> lock(_info->valueTable_mutex);
170  callForType(_info->valueType, [&, this](auto arg) {
171  typedef decltype(arg) T;
172  auto vtEntry = boost::fusion::at_key<T>(_info->valueTable.table);
173  vtEntry.subscriptions.erase(this->getId());
174  });
175  }
176  }

203  template<typename UserType>
204  bool LNMBackendVariableAccessor<UserType>::doWriteTransfer(ChimeraTK::VersionNumber v) {
205  if(_dev->_hasException) {
206  throw ChimeraTK::runtime_error("previous, unrecovered fault");
207  }
208  std::lock_guard<std::mutex> lock(_info->valueTable_mutex);
209  callForType(_info->valueType, [&, this](auto arg) {
210  auto& vtEntry = boost::fusion::at_key<decltype(arg)>(_info->valueTable.table);
211 
212  // store new value as latest value
213  for(size_t i = 0; i < this->buffer_2D[0].size(); ++i) {
214  vtEntry.latestValue[i + _wordOffsetInRegister] = userTypeToUserType<decltype(arg)>(this->buffer_2D[0][i]);
215  }
216  vtEntry.latestValidity = this->dataValidity();
217  vtEntry.latestVersion = v;
218 
219  // push new value to subscription queues, if async read is activated
220  if(_dev->_asyncReadActive) {
221  for(auto& sub : vtEntry.subscriptions) {
222  sub.second.push_overwrite({vtEntry.latestValue, this->dataValidity(), v});
223  }
224  }
225  });
226  return false;
227  }

268  template<typename UserType>
269  void LNMBackendVariableAccessor<UserType>::doPostRead(TransferType, bool hasNewData) {
270  if(!hasNewData) return;
271 
272  if(!_flags.has(AccessMode::wait_for_new_data)) {
273  // poll-type read transfer: fetch latest value from ValueTable
274  std::lock_guard<std::mutex> lock(_info->valueTable_mutex);
275  callForType(_info->valueType, [&, this](auto arg) {
276  auto& vtEntry = boost::fusion::at_key<decltype(arg)>(_info->valueTable.table);
277  for(size_t i = 0; i < this->buffer_2D[0].size(); ++i) {
278  this->buffer_2D[0][i] = userTypeToUserType<UserType>(vtEntry.latestValue[i + _wordOffsetInRegister]);
279  this->_dataValidity = vtEntry.latestValidity;
280  }
281  });
282  this->_versionNumber = {};
283  }
284  else {
285  // push-type read transfer: received value is in _queueValue (cf. readQueue continuation in constructor)
286  this->buffer_2D[0].swap(_queueValue.value);
287  this->_versionNumber = _queueValue.version;
288  this->_dataValidity = _queueValue.validity;
289  }
290  }
