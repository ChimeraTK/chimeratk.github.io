158  template<typename UserType>
159  size_t NumericAddressedAsyncVariableImpl<UserType>::unsubscribe() {
160  for(auto it = _subscribers.begin(); it != _subscribers.end(); ++it) {
161  // This code is called from the destructor of an AsyncNDRegisterAccessor inside a boost::shared_ptr.
162  // When this code is called the weak_ptr is already not lockable any more. We just use this to identify
163  // which element is to be removed. If we get the wrong one it does not matter because then the other destructor will get it.
164  if(it->lock().get() == nullptr) {
165  _subscribers.erase(it);
166  return _subscribers.size();
167  }
168  }
169 
170  throw ChimeraTK::logic_error("NumericAddressedAsyncVariable::unsibscribe must only be called from the "
171  "destructor of an AsyncDNRegisterAccessor!");
172  }

214  template<typename Function>
215  void NumericAddressedAsyncVariableImpl<UserType>::executeWithCopy(Function function, VersionNumber const& version) {
216  assert(!_subscribers.empty());
217  assert(syncAccessor);
218  try {
219  syncAccessor->read();
220  _sendBuffer.value.swap(syncAccessor->accessChannels());
221  _sendBuffer.dataValidity = syncAccessor->dataValidity();
222  _sendBuffer.versionNumber = version;
223 
224  for(auto it = _subscribers.begin(); it != --(_subscribers.end()); ++it) {
225  auto subscriber = it->lock();
226  if(subscriber.get() != nullptr) { // Possible race condition: The subscriber is being destructed.
227  function(subscriber, _sendBuffer);
228  }
229  }
230  auto subscriber = _subscribers.back().lock();
231  if(subscriber.get() != nullptr) { // Possible race condition: The subscriber is being destructed.
232  function(subscriber, _sendBuffer);
233  }
234  }
235  catch(ChimeraTK::runtime_error&) {
236  // no action needed. The synchronous read() already triggered backend->setException();
237  }
238  }

276  NumericAddressedInterruptDispatcher::subscribe(boost::shared_ptr<NumericAddressedBackend> backend,
277  RegisterPath name, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
278  std::lock_guard<std::recursive_mutex> variablesLock(_variablesMutex);
279 
280  AccessorInstanceDescriptor descriptor(name, typeid(UserType), numberOfWords, wordOffsetInRegister, flags);
281  if(!_asyncVariables[descriptor].get()) {
282  // Variable does not exist yet. Create it.
283  auto synchronousFlags = flags;
284  synchronousFlags.remove(AccessMode::wait_for_new_data);
285  // Don't call backend->getSyncRegisterAccessor() here. It might skip the overriding of a backend.
286  _asyncVariables[descriptor] = std::make_unique<NumericAddressedAsyncVariableImpl<UserType>>(
287  backend->getRegisterAccessor<UserType>(name, numberOfWords, wordOffsetInRegister, synchronousFlags),
288  _isActive);
289  }
290 
291  auto asyncVariable = dynamic_cast<NumericAddressedAsyncVariableImpl<UserType>*>(_asyncVariables[descriptor].get());
292  // we just take all the information we need for the async accessor from the sync accessor which has already done all the parsing
293  auto newSubscriber = boost::make_shared<AsyncNDRegisterAccessor<UserType, NumericAddressedInterruptDispatcher,
294  NumericAddressedInterruptDispatcher::AccessorInstanceDescriptor>>(backend, shared_from_this(), name,
295  asyncVariable->syncAccessor->getNumberOfChannels(), asyncVariable->syncAccessor->getNumberOfSamples(), flags,
296  descriptor, asyncVariable->syncAccessor->getUnit(), asyncVariable->syncAccessor->getDescription());
297  // Set the exception backend here. It might be that the accessor is already activated during subscription, and the backend shoud be set at that point
298  newSubscriber->setExceptionBackend(backend);
299 
300  if(asyncVariable->syncAccessor->isWriteable()) {
301  // for writeable variables we add another synchronous accessors (which knows how to access the data) to the asyncAccessor (which is generic)
302  auto synchronousFlags = flags;
303  synchronousFlags.remove(AccessMode::wait_for_new_data);
304 
305  // Don't call backend->getSyncRegisterAccessor() here. It might skip the overriding of a backend.
306  newSubscriber->setWriteAccessor(
307  backend->getRegisterAccessor<UserType>(name, numberOfWords, wordOffsetInRegister, synchronousFlags));
308  }
309 
310  asyncVariable->subscribe(newSubscriber);
311  return newSubscriber;
312  }
