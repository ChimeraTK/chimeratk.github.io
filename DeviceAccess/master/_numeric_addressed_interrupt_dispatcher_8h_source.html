88  std::unique_ptr<AsyncVariable> NumericAddressedInterruptDispatcher::createAsyncVariable(
89  const boost::shared_ptr<DeviceBackend>& backend, AccessorInstanceDescriptor const& descriptor, bool isActive) {
90  auto synchronousFlags = descriptor.flags;
91  synchronousFlags.remove(AccessMode::wait_for_new_data);
92  // Don't call backend->getSyncRegisterAccessor() here. It might skip the overriding of a backend.
93  auto syncAccessor = backend->getRegisterAccessor<UserType>(
94  descriptor.name, descriptor.numberOfWords, descriptor.wordOffsetInRegister, synchronousFlags);
95  // read the initial value before adding it to the transfer group
96  if(isActive) {
97  try {
98  syncAccessor->read();
99  }
100  catch(ChimeraTK::runtime_error&) {
101  isActive = false;
102  }
103  }
104 
105  _transferGroup->addAccessor(syncAccessor);
106  return std::make_unique<NumericAddressedAsyncVariableImpl<UserType>>(syncAccessor);
107  }
