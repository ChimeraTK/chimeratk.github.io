222  template<typename T>
223  class DummyMultiplexedRegisterAccessor {
224  public:
231  DummyMultiplexedRegisterAccessor(DummyBackend* dev, std::string module, std::string name)
232  : _dev(dev), _path(module + "/" + name), pitch(0) {
233  registerInfo = _dev->_registerMap.getBackendRegister(module + "." + name);
234 
235  // create fixed point converters for each channel
236  for(auto& c : registerInfo.channels) {
237  // create fixed point converter for sequence
238  fpc.emplace_back(registerInfo.pathName, c.width, c.nFractionalBits, c.signedFlag);
239  // store offsets and number of bytes per word
240  assert(c.bitOffset % 8 == 0);
241  offsets.push_back(registerInfo.address + c.bitOffset / 8);
242  nbytes.push_back((c.width - 1) / 8 + 1); // width/8 rounded up
243  }
244 
245  if(fpc.empty()) {
246  throw ChimeraTK::logic_error("No sequences found for name \"" + name + "\".");
247  }
248 
249  // cache some information
250  nElements = registerInfo.nElements;
251  assert(registerInfo.elementPitchBits % 8 == 0);
252  pitch = registerInfo.elementPitchBits / 8;
253  }
254 
255  // declare that we want the default copy constructor. Needed because we have a custom = operator
256  DummyMultiplexedRegisterAccessor(const DummyMultiplexedRegisterAccessor&) = default;
257 
259  unsigned int getNumberOfElements() { return nElements; }
260 
262  unsigned int getNumberOfSequences() { return fpc.size(); }
263 
269  inline proxies::DummyRegisterSequence<T> operator[](unsigned int sequence) {
270  int8_t* basePtr = reinterpret_cast<int8_t*>(_dev->_barContents[registerInfo.bar].data());
271  int32_t* seq = reinterpret_cast<int32_t*>(basePtr + offsets[sequence]);
272  return proxies::DummyRegisterSequence<T>(&(fpc[sequence]), nbytes[sequence], pitch, seq);
273  }

276  DummyBackend& getBackend() const { return *_dev; }
277 
279  const RegisterPath& getRegisterPath() const { return _path; }
280 
281  const NumericAddressedRegisterInfo& getRegisterInfo() { return registerInfo; }

353

362  int32_t* buffer;
