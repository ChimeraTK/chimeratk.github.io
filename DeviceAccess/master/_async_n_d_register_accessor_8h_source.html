159  void AsyncNDRegisterAccessor<UserType, AsyncAccessorManager, AsyncAccessorID>::doPostRead(
160  [[maybe_unused]] TransferType type, bool updateDataBuffer) {
161  if(updateDataBuffer) {
162  // do not update meta data if updateDataBuffer == false, since this is the equivalent to a backend
163  // implementation, not a decorator
164  this->_versionNumber = _receiveBuffer.versionNumber;
165  this->_dataValidity = _receiveBuffer.dataValidity;
166  // Do not overwrite the vectors in the first layer of the 2D array. Accessing code might have stored them.
167  // Instead, swap the received data into the channel vectors.
168  auto source = _receiveBuffer.value.begin(); // the received data is the source as it is moved into the user buffer
169  auto destination = this->buffer_2D.begin();
170  for(; source != _receiveBuffer.value.end(); ++source, ++destination) {
171  destination->swap(*source);
172  }
173  }
