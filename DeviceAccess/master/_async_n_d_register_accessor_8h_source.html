173  template<typename UserType, typename AsyncAccessorManager, typename AsyncAccessorID>
174  AsyncNDRegisterAccessor<UserType, AsyncAccessorManager, AsyncAccessorID>::AsyncNDRegisterAccessor(
175  const boost::shared_ptr<DeviceBackend>& backend, const boost::shared_ptr<AsyncAccessorManager>& manager,
176  std::string const& name, size_t nChannels, size_t nElements, AccessModeFlags accessModeFlags, AsyncAccessorID id,
177  std::string const& unit, std::string const& description)
178 
179  : NDRegisterAccessor<UserType>(name, accessModeFlags, unit, description), _backend(backend),
180  _accessorManager(manager), _receiveBuffer(nChannels, nElements), _id(id) {
181  // Don't throw a ChimeraTK::logic_error here. They are for mistakes an application is doing when using DeviceAccess.
182  // If an AsyncNDRegisterAccessor is created without wait_for_new_data it is a mistake in the backend, which is not part
183  // of the application.
184  assert(accessModeFlags.has(AccessMode::wait_for_new_data));
185  buffer_2D.resize(nChannels);
186  for(auto& chan : buffer_2D) chan.resize(nElements);
187 
188  // The sequence to initialise the queue:
189  // * write once and then read,
190  // * repeat n+1 times to make sure all buffers inside the queue have been replaced with
191  // a properly sized buffer, so it can be swapped out and used for data
192  for(size_t i = 0; i < _queueSize + 1; ++i) {
193  Buffer b1(nChannels, nElements);
194  _dataTransportQueue.push(std::move(b1));
195  Buffer b2(nChannels, nElements);
196  _dataTransportQueue.pop(b2); // here b2 is swapped into the queue and transported "backwards"
197  }
198 
199  this->_readQueue = _dataTransportQueue.template then<void>(
200  [&](Buffer& buf) { std::swap(_receiveBuffer, buf); }, std::launch::deferred);
201  }

204  template<typename UserType, typename AsyncAccessorManager, typename AsyncAccessorID>
205  void AsyncNDRegisterAccessor<UserType, AsyncAccessorManager, AsyncAccessorID>::doPostRead(
206  [[maybe_unused]] TransferType type, bool updateDataBuffer) {
207  if(updateDataBuffer) {
208  // do not update meta data if updateDataBuffer == false, since this is the equivalent to a backend
209  // implementation, not a decorator
210  this->_versionNumber = _receiveBuffer.versionNumber;
211  this->_dataValidity = _receiveBuffer.dataValidity;
212  // Do not overwrite the vectors in the first layer of the 2D array. Accessing code might have stored them.
213  // Instead, swap the received data into the channel vectors.
214  auto source = _receiveBuffer.value.begin(); // the received data is the source as it is moved into the user buffer
215  auto destination = this->buffer_2D.begin();
216  for(; source != _receiveBuffer.value.end(); ++source, ++destination) {
217  destination->swap(*source);
218  }
219  }
220  }
