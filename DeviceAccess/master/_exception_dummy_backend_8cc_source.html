21  void ExceptionDummy::triggerPush(RegisterPath path, VersionNumber v) {
22  path.setAltSeparator(".");
23  std::unique_lock<std::mutex> lk(_pushDecoratorsMutex);
24  _pushVersions[path] = v;
25  for(auto& acc_weak : _pushDecorators[path]) {
26  auto acc = acc_weak.lock();
27  if(!acc->_isActive) continue;
28  lk.unlock(); // next line might call setException()...
29  acc->trigger();
30  lk.lock();
31  }
32  }

39  std::unique_lock<std::mutex> lk(_pushDecoratorsMutex);
40  for(auto& pair : _pushDecorators) {
