10  void ExceptionDummy::triggerPush(RegisterPath path, VersionNumber v) {
11  path.setAltSeparator(".");
12  std::unique_lock<std::mutex> lk(_pushDecoratorsMutex);
13  _pushVersions[path] = v;
14  for(auto& acc_weak : _pushDecorators[path]) {
15  auto acc = acc_weak.lock();
16  if(!acc->_isActive) continue;
17  lk.unlock(); // next line might call setException()...
18  acc->trigger();
19  lk.lock();
20  }
21  }

25  void ExceptionDummy::activateAsyncRead() noexcept {
26  DummyBackend::activateAsyncRead();
27 
28  std::unique_lock<std::mutex> lk(_pushDecoratorsMutex);
29  for(auto& pair : _pushDecorators) {
30  _pushVersions[pair.first] = {};
31  for(auto& acc_weak : pair.second) {
32  auto acc = acc_weak.lock();
33  if(acc->_isActive) continue;
34  lk.unlock(); // next line might call setException()...
35  acc->trigger(); // initial value
36  lk.lock();
37  acc->_isActive = true;
38  acc->_hasException = false;
39  }
40  }
41  _activateNewPushAccessors = true;
42  }

46  void ExceptionDummy::setException() {
47  DummyBackend::setException();
48 
49  // deactivate async transfers
50  std::unique_lock<std::mutex> lk(_pushDecoratorsMutex);
51  for(auto& pair : _pushDecorators) {
52  _pushVersions[pair.first] = {};
53  for(auto& acc_weak : pair.second) {
54  auto acc = acc_weak.lock();
55  if(!acc->_isActive) continue;
56  acc->_isActive = false;
57  if(acc->_hasException) continue;
58  acc->_hasException = true;
59  lk.unlock(); // next line might call setException()...
60  acc->trigger();
61  lk.lock();
62  }
63  }
64  _activateNewPushAccessors = false;
65  }

132  void ExceptionDummy::write(uint64_t bar, uint64_t address, int32_t const* data, size_t sizeInBytes) {
133  if(throwExceptionWrite) {
134  thereHaveBeenExceptions = true;
135  throw(ChimeraTK::runtime_error("DummyException: write throws by request"));
136  }
137  ChimeraTK::DummyBackend::write(bar, address, data, sizeInBytes);
138 
139  // increment write counter and update write order (only if address points to beginning of a register!)
140  auto itWriteOrder = _writeOrderMap.find(std::make_pair(bar, address));
141  if(itWriteOrder != _writeOrderMap.end()) {
142  // update write order
143  auto generatedOrderNumber = ++_writeOrderCounter;
144  auto& orderNumberInMap = itWriteOrder->second;
145  // atomically update order number in the map only if the generated order number is bigger. This will be always
146  // the case, unless there is a concurrent write operation updating the order number in between.
147  size_t current;
148  while((current = orderNumberInMap.load()) < generatedOrderNumber) {
149  orderNumberInMap.compare_exchange_weak(current, generatedOrderNumber);
150  }
151 
152  // increment write counter
153  auto itWriteCounter = _writeCounterMap.find(std::make_pair(bar, address));
154  assert(itWriteCounter != _writeCounterMap.end()); // always inserted together
155  itWriteCounter->second++;
156  }
157  }

161  bool ExceptionDummy::isFunctional() const {
162  // thereHaveBeenExceptions is different from _hasActiceException
163  // * thereHaveBeenExceptions is set when this class originally raised an exception
164  // * _hasActiveException is raised externally via setException. This can happen if a transfer element from another
165  // backend,
166  // which is in the same logical name mapping backend than this class, has seen an exception.
167  return (_opened && !throwExceptionOpen && !thereHaveBeenExceptions && !_hasActiveException);
168  }
