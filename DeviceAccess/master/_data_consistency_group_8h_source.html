113  inline bool DataConsistencyGroup::update(TransferElementID transferElementID) {
114  // if matching mode is none, always return true
115  if(mode == MatchingMode::none) return true;
116  assert(mode == MatchingMode::exact);
117 
118  // ignore transferElementID does not belong to DataConsistencyGroup
119  if(push_elements.find(transferElementID) == push_elements.end()) {
120  return false;
121  }
122  auto getVNFromElement = push_elements[transferElementID].getVersionNumber();
123  if(getVNFromElement < versionNumberToBeConsistentTo){
124  return false;
125  }
126  if(versionNumberToBeConsistentTo != getVNFromElement) {
127  versionNumberToBeConsistentTo = getVNFromElement;
128  consistentElements.clear();
129  }
130  consistentElements.insert(transferElementID);
131  if(consistentElements.size() == push_elements.size()) {
132  lasteSateOfConsistentElements = consistentElements;
133  return true;
134  }
135  return false;
136  }
