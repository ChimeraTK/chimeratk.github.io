77  inline bool DataConsistencyGroup::update(const TransferElementID& transferElementID) {
78  // ignore transferElementID does not belong to DataConsistencyGroup
79  if(push_elements.find(transferElementID) == push_elements.end()) {
80  return false;
81  }
82  // if matching mode is none, always return true
83  if(mode == MatchingMode::none) return true;
84  assert(mode == MatchingMode::exact);
85 
86  auto getVNFromElement = push_elements[transferElementID].getVersionNumber();
87  assert(getVNFromElement != VersionNumber{nullptr});
88  if(getVNFromElement < versionNumberToBeConsistentTo) {
89  return false;
90  }
91  if(versionNumberToBeConsistentTo != getVNFromElement) {
92  versionNumberToBeConsistentTo = getVNFromElement;
93  consistentElements.clear();
94  }
95  consistentElements.insert(transferElementID);
96  if(consistentElements.size() == push_elements.size()) {
97  lasteSateOfConsistentElements = consistentElements;
98  return true;
99  }
100  return false;
101  }
