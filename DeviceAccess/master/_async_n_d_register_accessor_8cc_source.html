16  AsyncNDRegisterAccessor<UserType>::AsyncNDRegisterAccessor(boost::shared_ptr<DeviceBackend> backend,
17  boost::shared_ptr<AsyncAccessorManager> manager, std::string const& name, size_t nChannels, size_t nElements,
18  AccessModeFlags accessModeFlags, std::string const& unit, std::string const& description)
19 
20  : NDRegisterAccessor<UserType>(name, accessModeFlags, unit, description), _backend(std::move(backend)),
21  _accessorManager(std::move(manager)), _receiveBuffer(nChannels, nElements) {
22  // Don't throw a ChimeraTK::logic_error here. They are for mistakes an application is doing when using DeviceAccess.
23  // If an AsyncNDRegisterAccessor is created without wait_for_new_data it is a mistake in the backend, which is not
24  // part of the application.
25  assert(accessModeFlags.has(AccessMode::wait_for_new_data));
26  buffer_2D.resize(nChannels);
27  for(auto& chan : buffer_2D) chan.resize(nElements);
28 
29  // The sequence to initialise the queue:
30  // * write once and then read,
31  // * repeat n+1 times to make sure all buffers inside the queue have been replaced with
32  // a properly sized buffer, so it can be swapped out and used for data
33  for(size_t i = 0; i < _queueSize + 1; ++i) {
34  Buffer b1(nChannels, nElements);
35  _dataTransportQueue.push(std::move(b1));
36  Buffer b2(nChannels, nElements);
37  _dataTransportQueue.pop(b2); // here b2 is swapped into the queue and transported "backwards"
38  }
39 
40  this->_readQueue = _dataTransportQueue.template then<void>(
41  [&](Buffer& buf) { std::swap(_receiveBuffer, buf); }, std::launch::deferred);
42  }

45  template<typename UserType>
46  void AsyncNDRegisterAccessor<UserType>::doPostRead([[maybe_unused]] TransferType type, bool updateDataBuffer) {
47  if(updateDataBuffer) {
48  // do not update meta data if updateDataBuffer == false, since this is the equivalent to a backend
49  // implementation, not a decorator
50  this->_versionNumber = _receiveBuffer.versionNumber;
51  this->_dataValidity = _receiveBuffer.dataValidity;
52  // Do not overwrite the vectors in the first layer of the 2D array. Accessing code might have stored them.
53  // Instead, swap the received data into the channel vectors.
54  auto source = _receiveBuffer.value.begin(); // the received data is the source as it is moved into the user buffer
55  auto destination = this->buffer_2D.begin();
56  for(; source != _receiveBuffer.value.end(); ++source, ++destination) {
57  destination->swap(*source);
58  }
59  }
60  }
