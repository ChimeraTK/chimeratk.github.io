124  template<class UserType, class ConverterType>
125  NumericAddressedBackendMuxedRegisterAccessor<UserType, ConverterType>::NumericAddressedBackendMuxedRegisterAccessor(
126  const RegisterPath& registerPathName, size_t numberOfElements, size_t elementsOffset,
127  boost::shared_ptr<DeviceBackend> _backend)
128  : NDRegisterAccessor<UserType>(registerPathName, {}),
129  _ioDevice(boost::dynamic_pointer_cast<NumericAddressedBackend>(_backend)) {
130  // Obtain information about the area
131  _registerInfo = _ioDevice->_registerMap.getBackendRegister(registerPathName);
132  assert(!_registerInfo.channels.empty());
133 
134  // Create a fixed point converter for each channel
135  for(size_t i = 0; i < _registerInfo.getNumberOfChannels(); ++i) {
136  if(_registerInfo.channels[i].bitOffset % 8 != 0) {
137  throw ChimeraTK::logic_error("NumericAddressedBackendMuxedRegisterAccessor: elements must be byte aligned.");
138  }
139  _converters.emplace_back(detail::createDataConverter<ConverterType>(_registerInfo, i));
140  }
141  // check information
142  if(_registerInfo.elementPitchBits % 8 != 0) {
143  throw ChimeraTK::logic_error("NumericAddressedBackendMuxedRegisterAccessor: blocks must be byte aligned.");
144  }
145 
146  // compute effective numberOfElements
147  if(numberOfElements == 0) {
148  numberOfElements = _registerInfo.nElements;
149  }
150 
151  // check number of words
152  if(numberOfElements + elementsOffset > _registerInfo.nElements) {
153  throw ChimeraTK::logic_error("Requested number of elements exceeds the size of the register! Requested end: " +
154  std::to_string(numberOfElements + elementsOffset) +
155  ", register length: " + std::to_string(_registerInfo.nElements));
156  }
157 
158  // update register info
159  _registerInfo.nElements = numberOfElements;
160  assert(_registerInfo.elementPitchBits % 8 == 0);
161  _registerInfo.address += elementsOffset * _registerInfo.elementPitchBits / 8;
162 
163  // allocate the buffer for the converted data
164  NDRegisterAccessor<UserType>::buffer_2D.resize(_converters.size());
165  for(size_t i = 0; i < _converters.size(); ++i) {
166  NDRegisterAccessor<UserType>::buffer_2D[i].resize(_registerInfo.nElements);
167  }
168 
169  // allocate the raw io buffer. Make it one element larger to make sure we can access the last byte via int32_t*
170  _ioBuffer.resize(_registerInfo.elementPitchBits / 8 * _registerInfo.nElements / sizeof(int32_t) + 1);
171 
172  // compute pitched iterators for accessing the channels
173  auto* ioBuffer = reinterpret_cast<uint8_t*>(&_ioBuffer[0]);
174  for(auto& c : _registerInfo.channels) {
175  assert(c.bitOffset % 8 == 0);
176  _startIterators.emplace_back(ioBuffer + c.bitOffset / 8, _registerInfo.elementPitchBits / 8);
177  _endIterators.push_back(_startIterators.back() + _registerInfo.nElements);
178  }
179  }

185  assert(_registerInfo.elementPitchBits % 8 == 0);

188  nbt = ((nbt - 1) / 4 + 1) * 4; // round up to multiple of 4 bytes

190  _ioDevice->read(_registerInfo.bar, _registerInfo.address, _ioBuffer.data(), nbt);

226  template<class UserType, class ConverterType>
227  void NumericAddressedBackendMuxedRegisterAccessor<UserType, ConverterType>::doPreWrite(TransferType, VersionNumber) {
228  if(!_ioDevice->isOpen()) throw ChimeraTK::logic_error("Device not opened.");
229 
230  assert(_registerInfo.channels.size() == _converters.size());
231 
232  // Convert channel by channel
233  for(size_t i = 0; i < _registerInfo.channels.size(); ++i) {
234  callForRawType(_registerInfo.channels[i].getRawType(), [&](auto x) {
235  using RawType = decltype(x);
236 
237  // Call FixedPointConverter::toRaw() for each value in the channel. The result (C++ type int32) is written to
238  // the target buffer through the pitched iterators after converting it into the RawType matching the actual
239  // bit width of the channel. This is important to avoid overwriting data of other channels.
240  std::transform(buffer_2D[i].begin(), buffer_2D[i].end(), detail::pitched_iterator<RawType>(_startIterators[i]),
241  [&](UserType cookedValue) { return _converters[i].toRaw(cookedValue); });
242  });
243  }
244  }
