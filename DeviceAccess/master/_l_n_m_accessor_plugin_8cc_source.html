45  void AccessorPluginBase::updateRegisterInfo(BackendRegisterCatalogue<LNMBackendRegisterInfo>& catalogue) {
46  // First update the info so we have the latest version from the catalogue.
47  // At this point we also get a list of shared_ptrs of all plugins inside the info object.
48  _info = catalogue.getBackendRegister(_info.name);
49  // Do the actual info modifications as implemented by the plugin.
50  doRegisterInfoUpdate();
51  // Write the modifications back to the catalogue (still including plugins).
52  catalogue.modifyRegister(_info);
53  // Remove the list of plugins from the copy inside the plugin, which otherwise would hold a shared_ptr to itself.
54  // For abstraction reasons it also must not know about other plugins, so it is safe to remove the whole list.
55  _info.plugins.clear();
56  }
