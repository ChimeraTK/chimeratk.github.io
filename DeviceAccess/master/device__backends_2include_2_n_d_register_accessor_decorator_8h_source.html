48  template<typename UserType>
49  class NDRegisterAccessorDecoratorImpl<UserType, UserType> : public NDRegisterAccessor<UserType> {
50  public:
51  NDRegisterAccessorDecoratorImpl(std::string const& name, AccessModeFlags accessModeFlags,
52  std::string const& unit = std::string(TransferElement::unitNotSet),
53  std::string const& description = std::string())
54  : NDRegisterAccessor<UserType>(name, accessModeFlags, unit, description) {
55  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(getAsCooked_impl);
56  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(setAsCooked_impl);

59  void doPreRead(TransferType type) override { _target->preRead(type); }
60 
61  void doPostRead(TransferType type, bool updateDataBuffer) override {
62  _target->setActiveException(this->_activeException);
63  _target->postRead(type, updateDataBuffer);
64 
65  // Decorators have to copy meta data even if updateDataBuffer is false
66  this->_dataValidity = _target->dataValidity();
67  this->_versionNumber = _target->getVersionNumber();
68 
69  if(!updateDataBuffer) return;
70  for(size_t i = 0; i < _target->getNumberOfChannels(); ++i) buffer_2D[i].swap(_target->accessChannel(i));
71  }
72 
73  void doPreWrite(TransferType type, VersionNumber versionNumber) override {
74  for(size_t i = 0; i < _target->getNumberOfChannels(); ++i) buffer_2D[i].swap(_target->accessChannel(i));
75  _target->setDataValidity(this->_dataValidity);
76  _target->preWrite(type, versionNumber);
77  }
78 
79  void doPostWrite(TransferType type, VersionNumber versionNumber) override {
80  // swap back buffers unconditionally (even if postWrite() throws) at the end of this function
81  auto _ = cppext::finally([&] {
82  for(size_t i = 0; i < _target->getNumberOfChannels(); ++i) buffer_2D[i].swap(_target->accessChannel(i));
83  });
84  _target->setActiveException(this->_activeException);
85  _target->postWrite(type, versionNumber);
86  }

88  template<typename COOKED_TYPE>
89  COOKED_TYPE getAsCooked_impl(unsigned int channel, unsigned int sample) {
90  // Swap the user buffer back into the target, so we can call the target's getAsCooked()
91  // Channels might cached. Just swap the channel content.
92  this->buffer_2D[channel].swap(_target->accessChannel(channel));
93  auto retVal = _target->template getAsCooked<COOKED_TYPE>(channel, sample);
94  // Swap the buffer back
95  this->buffer_2D[channel].swap(_target->accessChannel(channel));
96 
97  return retVal;
98  }
99 
100  template<typename COOKED_TYPE>
101  void setAsCooked_impl(unsigned int channel, unsigned int sample, COOKED_TYPE value) {
102  // Swap the user buffer back into the target, so the target's setAsCooked() puts it in,
103  // then swap the buffer back out of the target.
104  // Channels might cached. Just swap the channel content.
105  this->buffer_2D[channel].swap(_target->accessChannel(channel));
106  _target->template setAsCooked<COOKED_TYPE>(channel, sample, value);
107  this->buffer_2D[channel].swap(_target->accessChannel(channel));
108  }
109 
110  protected:
111  using ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D;
112 
114  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> _target;
115  DEFINE_VIRTUAL_FUNCTION_TEMPLATE_VTABLE_FILLER(NDRegisterAccessorDecorator_impl<UserType>, getAsCooked_impl, 2);
116  DEFINE_VIRTUAL_FUNCTION_TEMPLATE_VTABLE_FILLER(NDRegisterAccessorDecorator_impl<UserType>, setAsCooked_impl, 3);
117  };

208 template<typename UserType, typename TargetUserType>
209 void ChimeraTK::NDRegisterAccessorDecorator<UserType, TargetUserType>::replaceTransferElement(
210  boost::shared_ptr<ChimeraTK::TransferElement> newElement) {
211  auto casted = boost::dynamic_pointer_cast<ChimeraTK::NDRegisterAccessor<TargetUserType>>(newElement);
212  if(casted && newElement->mayReplaceOther(_target)) {
213  if(_target != newElement) {
214  _target = detail::createCopyDecorator<TargetUserType>(casted);
215  }
216  }
217  else {
218  _target->replaceTransferElement(newElement);
219  }
220  _target->setExceptionBackend(this->_exceptionBackend);
221 }
