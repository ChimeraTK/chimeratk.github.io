67  bool RegisterInfoMap::check(ErrorList& err, RegisterInfoMap::ErrorList::ErrorElem::TYPE level) {
68  std::vector<addresses> v_addresses;
69  std::vector<addresses>::iterator v_iter;
70  addresses address;
71 
72  std::vector<RegisterInfo> map_file;
73  for(auto it = _catalogue.begin(); it != _catalogue.end(); ++it) {
74  auto infoCast = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(it.get());
75  map_file.push_back(*infoCast);
76  }
77 
78  std::vector<RegisterInfo>::iterator iter_p, iter_n;
79  bool ret = true;
80 
81  err.clear();
82  if(map_file.size() < 2) return true;
83  std::sort(map_file.begin(), map_file.end(), compareRegisterInfosByName_functor());
84  iter_p = map_file.begin();
85  iter_n = iter_p + 1;
86  while(1) {
87  if((iter_p->name == iter_n->name) && (iter_p->module == iter_n->module)) {
88  err.insert(ErrorList::ErrorElem(ErrorList::ErrorElem::ERROR, ErrorList::ErrorElem::NONUNIQUE_REGISTER_NAME,
89  *iter_p, *iter_n, _mapFileName));
90  ret = false;
91  }
92  iter_p = iter_n;
93  iter_n = ++iter_n;
94  if(iter_n == map_file.end()) {
95  break;
96  }
97  }
98 
99  for(iter_p = map_file.begin(); iter_p != map_file.end(); ++iter_p) {
100  address.start = iter_p->address;
101  address.end = iter_p->address + iter_p->nBytes;
102  address.iter = iter_p;
103  for(v_iter = v_addresses.begin(); v_iter != v_addresses.end(); ++v_iter) {
104  // only addresses in the same module are considered to overlap
105  if(iter_p->module != v_iter->iter->module) {
106  continue;
107  }
108 
109  if(iter_p->bar == (v_iter->iter)->bar) {
110  if((v_iter->start >= iter_p->address) && (v_iter->start < (iter_p->address + iter_p->nBytes))) {
111  if(level >= ErrorList::ErrorElem::WARNING) {
112  err.insert(ErrorList::ErrorElem(ErrorList::ErrorElem::WARNING,
113  ErrorList::ErrorElem::WRONG_REGISTER_ADDRESSES, *iter_p, *(v_iter->iter), _mapFileName));
114  ret = false;
115  }
116  }
117  else if((v_iter->start <= iter_p->address) && (iter_p->address < v_iter->end)) {
118  if(level >= ErrorList::ErrorElem::WARNING) {
119  err.insert(ErrorList::ErrorElem(ErrorList::ErrorElem::WARNING,
120  ErrorList::ErrorElem::WRONG_REGISTER_ADDRESSES, *iter_p, *(v_iter->iter), _mapFileName));
121  ret = false;
122  }
123  }
124  }
125  }
126  v_addresses.push_back(address);
127  }
128  return ret;
129  }

246  std::list<RegisterInfoMap::RegisterInfo> RegisterInfoMap::getRegistersInModule(std::string const& moduleName) {
247  // first sort all elements accordind the names (module first, then register in
248  // module) make a copy to keep the original order from the map file
249  std::vector<RegisterInfo> sortedRegisterInfos;
250  for(auto it = _catalogue.begin(); it != _catalogue.end(); ++it) {
251  auto infoCast = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(it.get());
252  sortedRegisterInfos.push_back(*infoCast);
253  }
254  std::sort(sortedRegisterInfos.begin(), sortedRegisterInfos.end(), compareRegisterInfosByName_functor());
255 
256  // The vector is sorted, first module, than register name.
257  // Find the first iterator with the module name, starting at the beginning,
258  // then search for the first iterator with another name, starting from the
259  // previousy found first match.
260  std::vector<RegisterInfo>::iterator firstMatchingIterator =
261  std::find_if(sortedRegisterInfos.begin(), sortedRegisterInfos.end(), compareModuleName_pred(moduleName));
262  std::vector<RegisterInfo>::iterator firstNotMatchingIterator =
263  std::find_if(firstMatchingIterator, sortedRegisterInfos.end(), std::not1(compareModuleName_pred(moduleName)));
264 
265  // fill the list
266  std::list<RegisterInfoMap::RegisterInfo> registerInfoList;
267  for(std::vector<RegisterInfo>::iterator it = firstMatchingIterator; it != firstNotMatchingIterator; ++it) {
268  registerInfoList.push_back(*it);
269  }
270 
271  return registerInfoList;
272  }

277  RegisterInfoMap::RegisterInfo::RegisterInfo(std::string const& name_, uint32_t nElements_, uint32_t address_,
278  uint32_t nBytes_, uint32_t bar_, uint32_t width_, int32_t nFractionalBits_, bool signedFlag_,
279  std::string const& module_, uint32_t nChannels_, bool is2DMultiplexed_, Access dataAccess_, Type dataType_)
280  : name(name_), nElements(nElements_), nChannels(nChannels_), is2DMultiplexed(is2DMultiplexed_), address(address_),
281  nBytes(nBytes_), bar(bar_), width(width_), nFractionalBits(nFractionalBits_), signedFlag(signedFlag_),
282  module(module_), registerAccess(dataAccess_), dataType(dataType_) {
283  if(dataType == IEEE754) {
284  if(width == 32) {
285  // Largest possible number +- 3e38, smallest possible number 1e-45
286  // However, the actual precision is only 23+1 bit, which is < 1e9 relevant
287  // digits. Hence, we don't have to add the 3e38 and the 1e45, but just add
288  // the leading 0. comma and sign to the largest 45 digits
289  dataDescriptor = DataDescriptor(RegisterInfo::FundamentalType::numeric, // fundamentalType
290  false, // isIntegral
291  true, // isSigned
292  3 + 45, // nDigits
293  45, // nFractionalDigits
294  DataType::int32); // we have integer in the transport layer
295  }
296  else if(width == 64) {
297  // smallest possible 5e-324, largest 2e308
298  dataDescriptor = DataDescriptor(RegisterInfo::FundamentalType::numeric, // fundamentalType
299  false, // isIntegral
300  true, // isSigned
301  3 + 325, // nDigits
302  325, // nFractionalDigits
303  DataType::int64);
304  }
305  else {
306  throw logic_error("Wrong data width for data type IEEE754. Check your map file!");
307  }
308  }
309  else if(dataType == FIXED_POINT) {
310  if(width > 1) { // numeric type
311  DataType rawDataInfo = (is2DMultiplexed_ ? DataType::none : DataType::int32);
312 
313  if(nFractionalBits_ > 0) {
314  size_t nDigits =
315  std::ceil(std::log10(std::pow(2, width_))) + (signedFlag_ ? 1 : 0) + (nFractionalBits_ != 0 ? 1 : 0);
316  size_t nFractionalDigits = std::ceil(std::log10(std::pow(2, nFractionalBits_)));
317 
318  dataDescriptor = DataDescriptor(RegisterInfo::FundamentalType::numeric, // fundamentalType
319  false, // isIntegral
320  signedFlag_, // isSigned
321  nDigits, nFractionalDigits, rawDataInfo);
322  }
323  else {
324  size_t nDigits =
325  std::ceil(std::log10(std::pow(2, width_))) + (signedFlag_ ? 1 : 0) + (nFractionalBits_ != 0 ? 1 : 0);
326 
327  dataDescriptor = DataDescriptor(RegisterInfo::FundamentalType::numeric, // fundamentalType
328  true, // isIntegral
329  signedFlag_, // isSigned
330  nDigits, 0, rawDataInfo);
331  }
332  }
333  else if(width == 1) { // boolean
334  dataDescriptor = DataDescriptor(RegisterInfo::FundamentalType::boolean);
335  }
336  else { // width == 0 -> nodata
337  dataDescriptor = DataDescriptor(RegisterInfo::FundamentalType::nodata);
338  }
339  }
340  else if(dataType == ASCII) {
341  dataDescriptor = DataDescriptor(RegisterInfo::FundamentalType::string);
342  }
