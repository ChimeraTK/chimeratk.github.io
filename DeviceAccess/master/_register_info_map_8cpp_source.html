70  bool RegisterInfoMap::check(ErrorList& err, RegisterInfoMap::ErrorList::ErrorElem::TYPE level) {
71  std::vector<addresses> v_addresses;
72  std::vector<addresses>::iterator v_iter;
73  addresses address;
74 
75  std::vector<RegisterInfo> map_file;
76  for(auto it = _catalogue.begin(); it != _catalogue.end(); ++it) {
77  auto infoCast = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(it.get());
78  map_file.push_back(*infoCast);
79  }
80 
81  std::vector<RegisterInfo>::iterator iter_p, iter_n;
82  bool ret = true;
83 
84  err.clear();
85  if(map_file.size() < 2) return true;
86  std::sort(map_file.begin(), map_file.end(), compareRegisterInfosByName_functor());
87  iter_p = map_file.begin();
88  iter_n = iter_p + 1;
89  while(1) {
90  if((iter_p->name == iter_n->name) && (iter_p->module == iter_n->module)) {
91  err.insert(ErrorList::ErrorElem(ErrorList::ErrorElem::ERROR, ErrorList::ErrorElem::NONUNIQUE_REGISTER_NAME,
92  *iter_p, *iter_n, _mapFileName));
93  ret = false;
94  }
95  iter_p = iter_n;
96  iter_n = ++iter_n;
97  if(iter_n == map_file.end()) {
98  break;
99  }
100  }
101 
102  for(iter_p = map_file.begin(); iter_p != map_file.end(); ++iter_p) {
103  address.start = iter_p->address;
104  address.end = iter_p->address + iter_p->nBytes;
105  address.iter = iter_p;
106  for(v_iter = v_addresses.begin(); v_iter != v_addresses.end(); ++v_iter) {
107  // only addresses in the same module are considered to overlap
108  if(iter_p->module != v_iter->iter->module) {
109  continue;
110  }
111 
112  if(iter_p->bar == (v_iter->iter)->bar) {
113  if((v_iter->start >= iter_p->address) && (v_iter->start < (iter_p->address + iter_p->nBytes))) {
114  if(level >= ErrorList::ErrorElem::WARNING) {
115  err.insert(ErrorList::ErrorElem(ErrorList::ErrorElem::WARNING,
116  ErrorList::ErrorElem::WRONG_REGISTER_ADDRESSES, *iter_p, *(v_iter->iter), _mapFileName));
117  ret = false;
118  }
119  }
120  else if((v_iter->start <= iter_p->address) && (iter_p->address < v_iter->end)) {
121  if(level >= ErrorList::ErrorElem::WARNING) {
122  err.insert(ErrorList::ErrorElem(ErrorList::ErrorElem::WARNING,
123  ErrorList::ErrorElem::WRONG_REGISTER_ADDRESSES, *iter_p, *(v_iter->iter), _mapFileName));
124  ret = false;
125  }
126  }
127  }
128  }
129  v_addresses.push_back(address);
130  }
131  return ret;
132  }

256  std::list<RegisterInfoMap::RegisterInfo> RegisterInfoMap::getRegistersInModule(std::string const& moduleName) {
257  // first sort all elements accordind the names (module first, then register in
258  // module) make a copy to keep the original order from the map file
259  std::vector<RegisterInfo> sortedRegisterInfos;
260  for(auto it = _catalogue.begin(); it != _catalogue.end(); ++it) {
261  auto infoCast = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(it.get());
262  sortedRegisterInfos.push_back(*infoCast);
263  }
264  std::sort(sortedRegisterInfos.begin(), sortedRegisterInfos.end(), compareRegisterInfosByName_functor());
265 
266  // The vector is sorted, first module, than register name.
267  // Find the first iterator with the module name, starting at the beginning,
268  // then search for the first iterator with another name, starting from the
269  // previousy found first match.
270  std::vector<RegisterInfo>::iterator firstMatchingIterator =
271  std::find_if(sortedRegisterInfos.begin(), sortedRegisterInfos.end(), compareModuleName_pred(moduleName));
272  std::vector<RegisterInfo>::iterator firstNotMatchingIterator =
273  std::find_if(firstMatchingIterator, sortedRegisterInfos.end(), std::not1(compareModuleName_pred(moduleName)));
274 
275  // fill the list
276  std::list<RegisterInfoMap::RegisterInfo> registerInfoList;
277  for(std::vector<RegisterInfo>::iterator it = firstMatchingIterator; it != firstNotMatchingIterator; ++it) {
278  registerInfoList.push_back(*it);
279  }
280 
281  return registerInfoList;
282  }

287  RegisterInfoMap::RegisterInfo::RegisterInfo(std::string const& name_, uint32_t nElements_, uint64_t address_,
288  uint32_t nBytes_, uint64_t bar_, uint32_t width_, int32_t nFractionalBits_, bool signedFlag_,
289  std::string const& module_, uint32_t nChannels_, bool is2DMultiplexed_, Access dataAccess_, Type dataType_,
290  uint32_t interruptCtrlNumber_, uint32_t interruptNumber_)
291  : name(name_), nElements(nElements_), nChannels(nChannels_), is2DMultiplexed(is2DMultiplexed_), address(address_),
292  nBytes(nBytes_), bar(bar_), width(width_), nFractionalBits(nFractionalBits_), signedFlag(signedFlag_),
293  module(module_), registerAccess(dataAccess_), dataType(dataType_), interruptCtrlNumber(interruptCtrlNumber_),
294  interruptNumber(interruptNumber_) {
295  if(nBytes_ > 0 && nElements_ > 0) {
296  if(nBytes_ % nElements_ != 0) {
297  // nBytes_ must be divisible by nElements_
298  throw logic_error(
299  "Number of bytes is not a multiple of number of elements for register " + name_ + ". Check your map file!");
300  }
301  }
302 
303  DataType rawDataInfo;
304  if(nBytesPerElement() == 0 && !is2DMultiplexed_) {
305  rawDataInfo = DataType::Void;
306  }
307  else if(nBytesPerElement() == 1 && !is2DMultiplexed_) {
308  rawDataInfo = DataType::int8;
309  }
310  else if(nBytesPerElement() == 2 && !is2DMultiplexed_) {
311  rawDataInfo = DataType::int16;
312  }
313  else if(nBytesPerElement() == 4 && !is2DMultiplexed_) {
314  rawDataInfo = DataType::int32;
315  }
316  else {
317  rawDataInfo = DataType::none;
318  }
319 
320  if(dataType == IEEE754) {
321  if(width == 32) {
322  // Largest possible number +- 3e38, smallest possible number 1e-45
323  // However, the actual precision is only 23+1 bit, which is < 1e9 relevant
324  // digits. Hence, we don't have to add the 3e38 and the 1e45, but just add
325  // the leading 0. comma and sign to the largest 45 digits
326  dataDescriptor = DataDescriptor(RegisterInfo::FundamentalType::numeric, // fundamentalType
327  false, // isIntegral
328  true, // isSigned
329  3 + 45, // nDigits
330  45, // nFractionalDigits
331  DataType::int32); // we have integer in the transport layer
332  }
333  else if(width == 64) {
334  // smallest possible 5e-324, largest 2e308
335  dataDescriptor = DataDescriptor(RegisterInfo::FundamentalType::numeric, // fundamentalType
336  false, // isIntegral
337  true, // isSigned
338  3 + 325, // nDigits
339  325, // nFractionalDigits
340  DataType::int64);
341  }
342  else {
343  throw logic_error("Wrong data width for data type IEEE754 for register " + name_ + ". Check your map file!");
344  }
345  }
346  else if(dataType == FIXED_POINT) {
347  if(width > 1) { // numeric type
348 
349  if(nFractionalBits_ > 0) {
350  size_t nDigits =
351  std::ceil(std::log10(std::pow(2, width_))) + (signedFlag_ ? 1 : 0) + (nFractionalBits_ != 0 ? 1 : 0);
352  size_t nFractionalDigits = std::ceil(std::log10(std::pow(2, nFractionalBits_)));
353 
354  dataDescriptor = DataDescriptor(RegisterInfo::FundamentalType::numeric, // fundamentalType
355  false, // isIntegral
356  signedFlag_, // isSigned
357  nDigits, nFractionalDigits, rawDataInfo);
358  }
359  else {
360  size_t nDigits = std::ceil(std::log10(std::pow(2, width_ + nFractionalBits_))) + (signedFlag_ ? 1 : 0);
361 
362  dataDescriptor = DataDescriptor(RegisterInfo::FundamentalType::numeric, // fundamentalType
363  true, // isIntegral
364  signedFlag_, // isSigned
365  nDigits, 0, rawDataInfo);
366  }
367  }
368  else if(width == 1) { // boolean
369  dataDescriptor = DataDescriptor(RegisterInfo::FundamentalType::boolean, true, false, 1, 0, rawDataInfo);
370  }
371  else { // width == 0 -> nodata
372  dataDescriptor = DataDescriptor(RegisterInfo::FundamentalType::nodata, false, false, 0, 0, rawDataInfo);
373  }
374  }
375  else if(dataType == ASCII) {
376  dataDescriptor = DataDescriptor(RegisterInfo::FundamentalType::string, false, false, 0, 0, rawDataInfo);
377  }
378  else if(dataType == VOID) {
379  dataDescriptor = DataDescriptor(RegisterInfo::FundamentalType::nodata, false, false, 0, 0, rawDataInfo);
380  }
