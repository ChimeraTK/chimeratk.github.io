22  template<typename UserType>
23  class LNMBackendChannelAccessor : public NDRegisterAccessor<UserType> {
24  public:
25  LNMBackendChannelAccessor(boost::shared_ptr<DeviceBackend> dev, const RegisterPath& registerPathName,
26  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
27  : NDRegisterAccessor<UserType>(registerPathName, flags), _registerPathName(registerPathName) {
28  // check for unknown flags
29  flags.checkForUnknownFlags({AccessMode::wait_for_new_data});
30 
31  // FIXME: use right type in constructor argument...
32  _dev = boost::dynamic_pointer_cast<LogicalNameMappingBackend>(dev);
33 
34  // copy the register info and create the internal accessors, if needed
35  _info = boost::static_pointer_cast<LNMBackendRegisterInfo>(
36  _dev->getRegisterCatalogue().getRegister(_registerPathName));
37 
38  // check for incorrect usage of this accessor
39  assert(_info->targetType == LNMBackendRegisterInfo::TargetType::CHANNEL);
40 
41  // get target device and accessor
42  std::string devName = _info->deviceName;
43  boost::shared_ptr<DeviceBackend> targetDevice;
44  if(devName != "this") {
45  targetDevice = _dev->_devices[devName];
46  }
47  else {
48  targetDevice = dev;
49  }
50  _accessor = targetDevice->getRegisterAccessor<UserType>(
51  RegisterPath(_info->registerName), numberOfWords, wordOffsetInRegister, flags);
52 
53  // verify channel number
54  if(_info->channel >= _accessor->getNumberOfChannels()) {
55  throw ChimeraTK::logic_error("LNMBackendChannelAccessor: Requested channel number " +
56  std::to_string(_info->channel) +
57  " exceeds number of channels of target register,"
58  " in accesor for register '" +
59  registerPathName + "'.");
60  }
61 
62  // allocate the buffer
63  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
64  NDRegisterAccessor<UserType>::buffer_2D[0].resize(_accessor->getNumberOfSamples());
65 
66  // set readQueue
67  TransferElement::_readQueue = _accessor->getReadQueue();
68  }
69 
70  void doReadTransferSynchronously() override { _accessor->readTransfer(); }
71 
72  bool doWriteTransfer(ChimeraTK::VersionNumber /*versionNumber*/) override {
73  assert(false); // writing not allowed
74  return true;
75  }
76 
77  void doPreWrite(TransferType type, VersionNumber) override {
78  std::ignore = type;
79  throw ChimeraTK::logic_error("Writing to channel-type registers of logical "
80  "name mapping devices is not supported.");
81  }
82 
83  void doPreRead(TransferType type) override { _accessor->preRead(type); }
84 
85  void doPostRead(TransferType type, bool hasNewData) override {
86  _accessor->postRead(type, hasNewData);
87  if(!hasNewData) return;
88  _accessor->accessChannel(_info->channel).swap(NDRegisterAccessor<UserType>::buffer_2D[0]);
89  this->_versionNumber = _accessor->getVersionNumber();
90  this->_dataValidity = _accessor->dataValidity();
91  }
92 
93  bool mayReplaceOther(const boost::shared_ptr<TransferElement const>& other) const override {
94  auto rhsCasted = boost::dynamic_pointer_cast<const LNMBackendChannelAccessor<UserType>>(other);
95  if(!rhsCasted) return false;
96  if(_registerPathName != rhsCasted->_registerPathName) return false;
97  if(_dev != rhsCasted->_dev) return false;
98  return true;
99  }
100 
101  bool isReadOnly() const override { return true; }
102 
103  bool isReadable() const override { return true; }
104 
105  bool isWriteable() const override { return false; }

110  }

129  }
130 
131  std::list<boost::shared_ptr<TransferElement>> getInternalElements() override {
132  auto result = _accessor->getInternalElements();
133  result.push_front(_accessor);
134  return result;
135  }
136 
137  void replaceTransferElement(boost::shared_ptr<TransferElement> newElement) override {
138  auto casted = boost::dynamic_pointer_cast<NDRegisterAccessor<UserType>>(newElement);
139  if(casted && _accessor->mayReplaceOther(newElement)) {
140  _accessor = casted;
141  }
142  else {
143  _accessor->replaceTransferElement(newElement);
144  }
145  _accessor->setExceptionBackend(this->_exceptionBackend);
146  }
147  };
