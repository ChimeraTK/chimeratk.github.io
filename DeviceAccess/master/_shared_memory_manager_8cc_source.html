57  SharedDummyBackend::SharedMemoryManager::~SharedMemoryManager() {
58  // stop and delete dispatcher thread first since it uses shm and mutex
59  intDispatcherIf.reset();
60  int pidSetSize;
61  {
62  // lock guard with the interprocess mutex
63  std::lock_guard<boost::interprocess::named_mutex> lock(interprocessMutex);
64 
65  // Clean up
66  checkPidSetConsistency();
67 
68  int32_t ownPid = static_cast<int32_t>(getOwnPID());
69  for(auto it = pidSet->begin(); it != pidSet->end();) {
70  if(*it == ownPid) {
71  it = pidSet->erase(it);
72  }
73  else {
74  ++it;
75  }
76  }
77  pidSetSize = pidSet->size();
78  }
79  // If size of pidSet is 0 (i.e, the instance belongs to the last accessing
80  // process), destroy shared memory and the interprocess mutex

82  boost::interprocess::shared_memory_object::remove(name.c_str());

107  bool SharedDummyBackend::SharedMemoryManager::checkPidSetConsistency() {
108  unsigned pidSetSizeBeforeCleanup = pidSet->size();
109 
110  for(auto it = pidSet->begin(); it != pidSet->end();) {
111  if(!processExists(*it)) {
112  // std::cout << "Nonexistent PID " << *it << " found. " <<std::endl;
113  it = pidSet->erase(it);
114  }
115  else {
116  it++;
117  }
118  }
119 
120  if(pidSetSizeBeforeCleanup != 0 && pidSet->size() == 0) {
121  return true;
122  }

126  void SharedDummyBackend::SharedMemoryManager::reInitMemory() {
127  std::vector<std::string> nameList = listNamedElements();
128 
129  for(auto item = nameList.begin(); item != nameList.end(); ++item) {
130  if(item->compare(SHARED_MEMORY_REQUIRED_VERSION_NAME) == 0) {
131  segment.destroy<unsigned>(item->c_str());
132  }
133  // reset the BAR vectors in shm.

135  else if(item->compare(SHARED_MEMORY_PID_SET_NAME) != 0) {

137  }
138  }
139  InterruptDispatcherInterface::cleanupShm(segment);
