21  template<typename DistributorType>
22  boost::shared_ptr<DistributorType> TriggerDistributor::getDistributorRecursive(
23  std::vector<uint32_t> const& interruptID) {
24  if(interruptID.size() == 1) {
25  // return the distributor from this instance, not a from further down the tree
26 
27  boost::weak_ptr<DistributorType>* weakDistributor{
28  nullptr}; // Cannot create a reference here, but references in the "if constexpr" scope are not seen outside
29 
30  if constexpr(std::is_same<DistributorType, TriggeredPollDistributor>::value) {
31  weakDistributor = &_pollDistributor;
32  }
33  else if constexpr(std::is_same<DistributorType, VariableDistributor<std::nullptr_t>>::value) {
34  weakDistributor = &_variableDistributor;
35  }
36  else {
37  throw ChimeraTK::logic_error("TriggerDistributor::getDistributorRecursive(): Wrong template parameter.");
38  }
39 
40  auto distributor = weakDistributor->lock();
41  if(!distributor) {
42  distributor = boost::make_shared<DistributorType>(_backend, shared_from_this(), _asyncDomain);
43  *weakDistributor = distributor;
44  if(_asyncDomain->unsafeGetIsActive()) {
45  distributor->distribute(nullptr, {});
46  }
47  }
48  return distributor;
49  }
50  // get a distributor from further down the tree, behind one or more InterruptControllerHandlers
51  auto controllerHandler = _interruptControllerHandler.lock();
52  if(!controllerHandler) {
53  controllerHandler = _interruptControllerHandlerFactory->createInterruptControllerHandler(_id, shared_from_this());
54  _interruptControllerHandler = controllerHandler;
55  }
56 
57  return controllerHandler->getDistributorRecursive<DistributorType>({++interruptID.begin(), interruptID.end()});
58  }

77  void TriggerDistributor::distribute([[maybe_unused]] std::nullptr_t, VersionNumber version) {
78  if(!_asyncDomain->unsafeGetIsActive()) {
79  return;
80  }
81  auto pollDistributor = _pollDistributor.lock();
82  if(pollDistributor) {
83  pollDistributor->distribute(nullptr, version);
84  }
85  auto controllerHandler = _interruptControllerHandler.lock();
86  if(controllerHandler) {
87  controllerHandler->handle(version);
88  }
89  auto variableDistributor = _variableDistributor.lock();
90  if(variableDistributor) {
91  variableDistributor->distribute(nullptr, version);
92  }
93  }

131  template<typename UserType>
132  boost::shared_ptr<AsyncNDRegisterAccessor<UserType>> TriggerDistributor::SubscriptionImplementor<
133  UserType>::subscribeTo(TriggerDistributor& triggerDistributor, RegisterPath name, size_t numberOfWords,
134  size_t wordOffsetInRegister, AccessModeFlags flags) {
135  auto catalogue = triggerDistributor._backend->getRegisterCatalogue(); // need to store the clone you get
136  const auto& backendCatalogue = catalogue.getImpl();
137  // This code only works for backends which use the NumericAddressedRegisterCatalogue because we need the
138  // interrupt description which is specific for those backends and not in the general catalogue.
139  // If the cast fails, it will throw an exception.
140  auto removeMe = backendCatalogue.getRegister(name);
141  const auto& numericCatalogue = dynamic_cast<const NumericAddressedRegisterCatalogue&>(backendCatalogue);
142  auto registerInfo = numericCatalogue.getBackendRegister(name);
143 
144  // Find the right place in the distribution tree to subscribe
145  boost::shared_ptr<AsyncAccessorManager> distributor;
146  if(registerInfo.getDataDescriptor().fundamentalType() == DataDescriptor::FundamentalType::nodata) {
147  distributor =
148  triggerDistributor.getDistributorRecursive<VariableDistributor<std::nullptr_t>>(registerInfo.interruptId);
149  }
150  else {
151  distributor = triggerDistributor.getDistributorRecursive<TriggeredPollDistributor>(registerInfo.interruptId);
152  }
153 
154  return distributor->template subscribe<UserType>(name, numberOfWords, wordOffsetInRegister, flags);
155  }
