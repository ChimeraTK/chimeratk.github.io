106  template<typename BackendSpecificDataType>
107  template<typename DistributorType>
108  boost::shared_ptr<DistributorType> TriggerDistributor<BackendSpecificDataType>::getDistributorRecursive(
109  std::vector<uint32_t> const& interruptID) {
110  if(interruptID.size() == 1) {
111  // return the distributor from this instance, not a from further down the tree
112 
113  boost::weak_ptr<DistributorType>* weakDistributor{
114  nullptr}; // Cannot create a reference here, but references in the "if constexpr" scope are not seen outside
115 
116  if constexpr(std::is_same<DistributorType, TriggeredPollDistributor>::value) {
117  weakDistributor = &_pollDistributor;
118  }
119  else if constexpr(std::is_same<DistributorType, VariableDistributor<BackendSpecificDataType>>::value) {
120  weakDistributor = &_variableDistributor;
121  }
122  else {
123  throw ChimeraTK::logic_error("TriggerDistributor::getDistributorRecursive(): Wrong template parameter.");
124  }
125 
126  auto distributor = weakDistributor->lock();
127  if(!distributor) {
128  distributor = boost::make_shared<DistributorType>(_backend, this->shared_from_this(), _asyncDomain);
129  *weakDistributor = distributor;
130  if(_asyncDomain->unsafeGetIsActive()) {
131  distributor->distribute(nullptr, {});
132  }
133  }
134  return distributor;
135  }
136  // get a distributor from further down the tree, behind one or more InterruptControllerHandlers
137  auto controllerHandler = _interruptControllerHandler.lock();
138  if(!controllerHandler) {
139  controllerHandler = _backend->createInterruptControllerHandler(_id, this->shared_from_this());
140  _interruptControllerHandler = controllerHandler;
141  }
142 
143  return controllerHandler->getDistributorRecursive<DistributorType>({++interruptID.begin(), interruptID.end()});
144  }

148  template<typename BackendSpecificDataType>
149  void TriggerDistributor<BackendSpecificDataType>::distribute(BackendSpecificDataType data, VersionNumber version) {
150  if(!_asyncDomain->unsafeGetIsActive()) {
151  return;
152  }
153  auto pollDistributor = _pollDistributor.lock();
154  if(pollDistributor) {
155  pollDistributor->distribute(nullptr, version);
156  }
157  auto controllerHandler = _interruptControllerHandler.lock();
158  if(controllerHandler) {
159  controllerHandler->handle(version);
160  }
161  auto variableDistributor = _variableDistributor.lock();
162  if(variableDistributor) {
163  variableDistributor->distribute(data, version);
164  }
165  }

207  template<typename UserType, typename BackendSpecificDataType>
208  boost::shared_ptr<AsyncNDRegisterAccessor<UserType>> AsyncDataAdapterSubscriptionImplementor<UserType,
209  BackendSpecificDataType>::subscribeTo(TriggerDistributor<BackendSpecificDataType>& triggerDistributor,
210  RegisterPath name, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
211  auto catalogue = triggerDistributor._backend->getRegisterCatalogue(); // need to store the clone you get
212  const auto& backendCatalogue = catalogue.getImpl();
213  // This code only works for backends which use the NumericAddressedRegisterCatalogue because we need the
214  // interrupt description which is specific for those backends and not in the general catalogue.
215  // If the cast fails, it will throw an exception.
216  const auto& numericCatalogue = dynamic_cast<const NumericAddressedRegisterCatalogue&>(backendCatalogue);
217  auto registerInfo = numericCatalogue.getBackendRegister(name);
218 
219  // Find the right place in the distribution tree to subscribe
220  boost::shared_ptr<AsyncAccessorManager> distributor;
221  if constexpr(std::is_same<BackendSpecificDataType, std::nullptr_t>::value) {
222  // Special implementation for data type nullptr_t: Use a poll distributor if the data is not
223  // FundamentalType::nodata itself
224  if(registerInfo.getDataDescriptor().fundamentalType() == DataDescriptor::FundamentalType::nodata) {
225  distributor = triggerDistributor.template getDistributorRecursive<VariableDistributor<std::nullptr_t>>(
226  registerInfo.interruptId);
227  }
228  else {
229  distributor =
230  triggerDistributor.template getDistributorRecursive<TriggeredPollDistributor>(registerInfo.interruptId);
231  }
232  }
233  else {
234  // For all other BackendSpecificDataType use the according VariableDistributor.
235  // This scheme might need some improvement later.
236  triggerDistributor.template getDistributorRecursive<VariableDistributor<BackendSpecificDataType>>(
237  registerInfo.interruptId);
238  }
239 
240  return distributor->template subscribe<UserType>(name, numberOfWords, wordOffsetInRegister, flags);
241  }
