26  bool SubdeviceRegisterAccessor::doWriteTransfer(ChimeraTK::VersionNumber) {
27  std::lock_guard<decltype(_backend->mutex)> lockGuard(_backend->mutex);
28  size_t nTransfers;
29  if(_backend->type == SubdeviceBackend::Type::areaHandshake)
30  // for areaHandshake case with 1D array
31  nTransfers = _numberOfWords;
32  else {
33  assert(_backend->type == SubdeviceBackend::Type::threeRegisters ||
34  _backend->type == SubdeviceBackend::Type::twoRegisters);
35  // This is "_numberOfWords / _accData->getNumberOfSamples()" rounded up:
36  nTransfers = (_numberOfWords + _accDataArea->getNumberOfSamples() - 1) / _accDataArea->getNumberOfSamples();
37  }
38  try {
39  size_t idx = 0;
40  for(size_t adr = _startAddress; adr < _startAddress + nTransfers; ++adr) {
41  if(_backend->type == SubdeviceBackend::Type::threeRegisters ||
42  _backend->type == SubdeviceBackend::Type::areaHandshake) {
43  size_t retry = 0;
44  size_t max_retry = _backend->timeout * 1000 / _backend->sleepTime;
45  while(true) {
46  _accStatus->read();
47  if(_accStatus->accessData(0) == 0) break;
48  if(++retry > max_retry) {
49  throw ChimeraTK::runtime_error("Write to register '" + _name +
50  "' failed: timeout waiting for cleared busy flag (" + _accStatus->getName() + ")");
51  }
52  usleep(_backend->sleepTime);
53  }
54  }
55  else { // twoRegisters type
56  usleep(_backend->sleepTime);
57  }
58  if(_backend->type == SubdeviceBackend::Type::areaHandshake) {
59  int32_t val = (idx < _numberOfWords) ? _buffer[idx] : 0;
60  _accDataArea->accessData(0, idx) = val;
61  ++idx;
62  }
63  else {
64  _accAddress->accessData(0) = adr;
65  _accAddress->write();
66  usleep(_backend->addressToDataDelay);
67  for(size_t innerOffset = 0; innerOffset < _accDataArea->getNumberOfSamples(); ++innerOffset) {
68  // pad data with zeros, if _numberOfWords isn't an integer multiple of _accData->getNumberOfSamples()
69  int32_t val = (idx < _numberOfWords) ? _buffer[idx] : 0;
70  _accDataArea->accessData(0, innerOffset) = val;
71  ++idx;
72  }
73  }
74  _accDataArea->write();
75  }
76  }
77  catch(ChimeraTK::runtime_error&) {
78  _exceptionBackend->setException();
79  throw;
80  }
81  return false;
82  }
