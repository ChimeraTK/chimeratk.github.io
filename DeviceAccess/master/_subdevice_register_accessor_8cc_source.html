27  bool SubdeviceRegisterAccessor::doWriteTransfer(ChimeraTK::VersionNumber) {
28  std::lock_guard<decltype(_backend->mutex)> lockGuard(_backend->mutex);
29  try {
30  size_t idx = 0;
31  for(size_t adr = _byteOffset; adr < _byteOffset + 4 * _numberOfWords; adr += 4) {
32  if(_backend->type == SubdeviceBackend::Type::threeRegisters) {
33  while(true) {
34  _accStatus->read();
35  if(_accStatus->accessData(0) == 0) break;
36  usleep(_backend->sleepTime);
37  }
38  }
39  else {
40  usleep(_backend->sleepTime);
41  }
42  _accAddress->accessData(0) = adr;
43  _accAddress->write();
44  _accData->accessData(0) = _buffer[idx];
45  _accData->write();
46  ++idx;
47  }
48  }
49  catch(ChimeraTK::runtime_error&) {
50  _exceptionBackend->setException();
51  throw;
52  }
53  return false;
54  }

80  if(!_accAddress->isWriteable()) {

83  }
84  if(!_accData->isWriteable()) {
85  throw ChimeraTK::logic_error("SubdeviceRegisterAccessor[" + this->getName() + "]: data register '" +
86  _accData->getName() + "' is not writeable.");

88  if(_backend->type == SubdeviceBackend::Type::threeRegisters) {
89  if(!_accStatus->isReadable()) {

93  }
