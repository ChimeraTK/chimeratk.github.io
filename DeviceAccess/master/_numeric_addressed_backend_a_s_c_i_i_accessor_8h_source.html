18  class NumericAddressedBackendASCIIAccessor : public NDRegisterAccessor<std::string> {
19  public:
20  NumericAddressedBackendASCIIAccessor(const boost::shared_ptr<DeviceBackend>& dev,
21  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister,
22  const AccessModeFlags& flags)
23  : NDRegisterAccessor<std::string>(registerPathName, flags),
24  _dev(boost::dynamic_pointer_cast<NumericAddressedBackend>(dev)) {
25  // check for unknown flags
26  flags.checkForUnknownFlags({});
27 
28  // check device backend
29  _dev = boost::dynamic_pointer_cast<NumericAddressedBackend>(dev);
30  if(!_dev) {
31  throw ChimeraTK::logic_error("NumericAddressedBackendASCIIAccessor is used with a backend which is not "
32  "a NumericAddressedBackend.");
33  }
34 
35  // obtain register information
36  _registerInfo = _dev->getRegisterInfo(registerPathName);
37  assert(!_registerInfo.channels.empty());
38 
39  if(_registerInfo.elementPitchBits % 8 != 0) {
40  throw ChimeraTK::logic_error("NumericAddressedBackendASCIIAccessor: Elements must be byte aligned.");
41  }
42 
43  if(_registerInfo.channels.size() > 1) {
44  throw ChimeraTK::logic_error("NumericAddressedBackendASCIIAccessor is used with a 2D register.");
45  }
46 
47  if(_registerInfo.channels.front().bitOffset > 0) {
48  throw ChimeraTK::logic_error("NumericAddressedBackendASCIIAccessor: Registers must be byte aligned.");
49  }
50 
51  if(_registerInfo.channels.front().dataType != NumericAddressedRegisterInfo::Type::ASCII) {
52  throw ChimeraTK::logic_error("NumericAddressedBackendASCIIAccessor: Cannot be used with non-ASCII registers.");
53  }
54 
55  // check number of words
56  if(numberOfWords == 0) {
57  numberOfWords = _registerInfo.getNumberOfElements();
58  }
59  if(numberOfWords + wordOffsetInRegister > _registerInfo.getNumberOfElements()) {
60  throw ChimeraTK::logic_error(
61  "Requested number of words exceeds the size of the register '" + registerPathName + "'!");
62  }
63  if(wordOffsetInRegister >= _registerInfo.getNumberOfElements()) {
64  throw ChimeraTK::logic_error("Requested offset exceeds the size of the register'" + registerPathName + "'!");
65  }
66 
67  // change registerInfo (local copy!) to account for given offset and length override
68  _registerInfo.address += wordOffsetInRegister * _registerInfo.elementPitchBits / 8;
69  _registerInfo.nElements = numberOfWords;
70 
71  // create low-level transfer element handling the actual data transfer to the hardware with raw data
72  assert(_registerInfo.elementPitchBits % 8 == 0);
73  _rawAccessor = boost::make_shared<NumericAddressedLowLevelTransferElement>(
74  _dev, _registerInfo.bar, _registerInfo.address, _registerInfo.nElements * _registerInfo.elementPitchBits / 8);
75 
76  // allocated the buffers
77  NDRegisterAccessor<std::string>::buffer_2D.resize(1);
78  NDRegisterAccessor<std::string>::buffer_2D[0].resize(_registerInfo.nElements);
79 
80  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(getAsCooked_impl);
81  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(setAsCooked_impl);
82  }
83 
84  void doReadTransferSynchronously() override { _rawAccessor->readTransfer(); }
85 
86  bool doWriteTransfer(ChimeraTK::VersionNumber versionNumber) override {
87  assert(!TransferElement::_isInTransferGroup);
88  _rawAccessor->writeTransfer(versionNumber);
89  return false;
90  }
91 
92  void doPostRead(TransferType type, bool hasNewData) override {
93  if(!_dev->isOpen()) return; // do not delegate if exception was thrown by us in doPreWrite
94 
95  _rawAccessor->setActiveException(this->_activeException);
96  _rawAccessor->postRead(type, hasNewData);
97 
98  if(!hasNewData) return;
99 
100  const size_t nCharsPerElement = _registerInfo.elementPitchBits / 8;
101  std::string temp;
102  temp.reserve(nCharsPerElement);
103  auto* itsrc = (char*)_rawAccessor->begin(_registerInfo.address);
104  for(size_t el = 0; el < _registerInfo.nElements; ++el) {
105  auto len = strnlen(itsrc, nCharsPerElement);
106  temp.resize(len);
107  std::strncpy(temp.data(), itsrc, nCharsPerElement);
108  itsrc += nCharsPerElement;
109  buffer_2D[0][el] = temp;
110  }
111 
112  // we don't put the setting of the version number into the PrePostActionImplementor
113  // because it does not need template specialisation, and the implementer does not
114  // know about _versionNumber. It's just easier here.
115  this->_versionNumber = _rawAccessor->getVersionNumber();
116  this->_dataValidity = _rawAccessor->dataValidity();
117  }
118 
119  void doPreWrite(TransferType type, VersionNumber versionNumber) override {
120  if(!_dev->isOpen()) throw ChimeraTK::logic_error("Device not opened.");
121  // raw accessor preWrite must be called before our _prePostActionsImplementor.doPreWrite(), as it needs to
122  // prepare the buffer in case of unaligned access and acquire the lock.
123  _rawAccessor->preWrite(type, versionNumber);
124 
125  const size_t nCharsPerElement = _registerInfo.elementPitchBits / 8;
126  auto* itdst = (char*)_rawAccessor->begin(_registerInfo.address);
127  for(size_t el = 0; el < _registerInfo.nElements; ++el) {
128  std::memset(itdst, 0, nCharsPerElement);
129  std::strncpy(itdst, buffer_2D[0][el].data(), nCharsPerElement);
130  itdst += nCharsPerElement;
131  }
132 
133  _rawAccessor->setDataValidity(this->_dataValidity);
134  }
135 
136  void doPreRead(TransferType type) override {
137  if(!_dev->isOpen()) throw ChimeraTK::logic_error("Device not opened.");
138  _rawAccessor->preRead(type);
139  }
140 
141  void doPostWrite(TransferType type, VersionNumber versionNumber) override {
142  if(!_dev->isOpen()) return; // do not delegate if exception was thrown by us in doPreWrite
143  _rawAccessor->setActiveException(this->_activeException);
144  _rawAccessor->postWrite(type, versionNumber);
145  }

148  auto rhsCasted = boost::dynamic_pointer_cast<const NumericAddressedBackendASCIIAccessor>(other);
149  if(!rhsCasted) return false;

152  return true;
153  }

198  void replaceTransferElement(boost::shared_ptr<TransferElement> newElement) override {
199  auto casted = boost::dynamic_pointer_cast<NumericAddressedLowLevelTransferElement>(newElement);
200  if(casted && casted->isMergeable(_rawAccessor)) {
201  size_t newStartAddress = std::min(casted->_startAddress, _rawAccessor->_startAddress);
202  size_t newStopAddress = std::max(
203  casted->_startAddress + casted->_numberOfBytes, _rawAccessor->_startAddress + _rawAccessor->_numberOfBytes);
204  size_t newNumberOfBytes = newStopAddress - newStartAddress;
205  casted->changeAddress(newStartAddress, newNumberOfBytes);
206  _rawAccessor = casted;
207  }
208  _rawAccessor->setExceptionBackend(this->_exceptionBackend);
209  }
