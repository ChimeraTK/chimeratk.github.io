270  boost::shared_ptr<LogicalNameMappingBackend>& backend, size_t numberOfWords, size_t wordOffsetInRegister,
271  AccessModeFlags flags, size_t pluginIndex) {
272  boost::shared_ptr<NDRegisterAccessor<UserType>> decorated;
273 
274  // obtain desired target type from plugin implementation
275  auto type = getTargetDataType(typeid(UserType));
276  if((_info._dataDescriptor.rawDataType() == DataType::none) and flags.has(AccessMode::raw)) {
277  throw ChimeraTK::logic_error(
278  "Access mode 'raw' is not supported for register '" + std::string(_info.getRegisterName()) + "'");
279  }
280 
281  callForType(type, [&](auto T) {
282  // obtain target accessor with desired type
283  auto target = backend->getRegisterAccessor_impl<decltype(T)>(
284  _info.getRegisterName(), numberOfWords, wordOffsetInRegister, flags, pluginIndex + 1);
285  decorated = static_cast<Derived*>(this)->template decorateAccessor<UserType>(backend, target);
286  });
287 
288  decorated->setExceptionBackend(backend);
