301  template<typename Derived>
302  template<typename UserType>
303  boost::shared_ptr<NDRegisterAccessor<UserType>> AccessorPlugin<Derived>::getAccessor_impl(
304  boost::shared_ptr<LogicalNameMappingBackend>& backend, size_t numberOfWords, size_t wordOffsetInRegister,
305  AccessModeFlags flags, size_t pluginIndex) {
306  boost::shared_ptr<NDRegisterAccessor<UserType>> decorated;
307 
308  // obtain desired target type from plugin implementation
309  auto type = getTargetDataType(typeid(UserType));
310  if((_info._dataDescriptor.rawDataType() == DataType::none) and flags.has(AccessMode::raw)) {
311  throw ChimeraTK::logic_error(
312  "Access mode 'raw' is not supported for register '" + std::string(_info.getRegisterName()) + "'");
313  }
314 
315  callForType(type, [&](auto T) {
316  // obtain target accessor with desired type
317  auto target = backend->getRegisterAccessor_impl<decltype(T)>(
318  _info.getRegisterName(), numberOfWords, wordOffsetInRegister, flags, pluginIndex + 1);
319  // double buffering plugin needs numberOfWords, wordOffsetInRegister of already existing accessor
320  UndecoratedParams accessorParams(numberOfWords, wordOffsetInRegister, flags);
321  decorated = static_cast<Derived*>(this)->template decorateAccessor<UserType>(backend, target, accessorParams);
322  });
323 
324  decorated->setExceptionBackend(backend);
325  return decorated;
326  }
