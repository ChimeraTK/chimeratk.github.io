33 BOOST_AUTO_TEST_CASE(testParseCDD) {
34  {
35  // check standard case
36  auto r = Utilities::parseDeviceDesciptor("(myBackendType:some/weired*address 234?par1=someValue with "
37  "spaces&map=file)");
38  BOOST_CHECK_EQUAL(r.backendType, "myBackendType");
39  BOOST_CHECK_EQUAL(r.address, "some/weired*address 234");
40  BOOST_CHECK_EQUAL(r.parameters.size(), 2);
41  BOOST_CHECK_EQUAL(r.parameters["par1"], "someValue with spaces");
42  BOOST_CHECK_EQUAL(r.parameters["map"], "file");
43  }
44  {
45  // check proper trimming
46  auto r = Utilities::parseDeviceDesciptor(" ( myBackendType : some/weired*address 234 ? par1 = "
47  "someValue with spaces & map = file ) ");
48  BOOST_CHECK_EQUAL(r.backendType, "myBackendType");
49  BOOST_CHECK_EQUAL(r.address, "some/weired*address 234");
50  BOOST_CHECK_EQUAL(r.parameters.size(), 2);
51  BOOST_CHECK_EQUAL(r.parameters["par1"], "someValue with spaces");
52  BOOST_CHECK_EQUAL(r.parameters["map"], "file");
53  }
54  {
55  // check only backend type
56  auto r = Utilities::parseDeviceDesciptor("(someStrangeBackendType)");
57  BOOST_CHECK_EQUAL(r.backendType, "someStrangeBackendType");
58  BOOST_CHECK_EQUAL(r.address, "");
59  BOOST_CHECK_EQUAL(r.parameters.size(), 0);
60  }
61  {
62  // check only backend type with address
63  auto r = Utilities::parseDeviceDesciptor("(pci:pcieunis6)");
64  BOOST_CHECK_EQUAL(r.backendType, "pci");
65  BOOST_CHECK_EQUAL(r.address, "pcieunis6");
66  BOOST_CHECK_EQUAL(r.parameters.size(), 0);
67  }
68  {
69  // check explicitly empty parameter list
70  auto r = Utilities::parseDeviceDesciptor("(pci:pcieunis6?)");
71  BOOST_CHECK_EQUAL(r.backendType, "pci");
72  BOOST_CHECK_EQUAL(r.address, "pcieunis6");
73  BOOST_CHECK_EQUAL(r.parameters.size(), 0);
74  }
75  {
76  // check explicitly empty parameter list with more empty parameters
77  auto r = Utilities::parseDeviceDesciptor("(pci:pcieunis6?&&)");
78  BOOST_CHECK_EQUAL(r.backendType, "pci");
79  BOOST_CHECK_EQUAL(r.address, "pcieunis6");
80  BOOST_CHECK_EQUAL(r.parameters.size(), 0);
81  }
82  {
83  // check only backend type with parameters
84  auto r = Utilities::parseDeviceDesciptor("(logicalNameMapper?map=myMapFile.xlmap)");
85  BOOST_CHECK_EQUAL(r.backendType, "logicalNameMapper");
86  BOOST_CHECK_EQUAL(r.address, "");
87  BOOST_CHECK_EQUAL(r.parameters.size(), 1);
88  BOOST_CHECK_EQUAL(r.parameters["map"], "myMapFile.xlmap");
89  }
90  {
91  // check explicitly empty address
92  auto r = Utilities::parseDeviceDesciptor("(logicalNameMapper:?map=myMapFile.xlmap)");
93  BOOST_CHECK_EQUAL(r.backendType, "logicalNameMapper");
94  BOOST_CHECK_EQUAL(r.address, "");
95  BOOST_CHECK_EQUAL(r.parameters.size(), 1);
96  BOOST_CHECK_EQUAL(r.parameters["map"], "myMapFile.xlmap");
97  }
98  {
99  // check explicitly empty parameters
100  auto r = Utilities::parseDeviceDesciptor("(logicalNameMapper?&map=myMapFile.xlmap& &a=b&)");
101  BOOST_CHECK_EQUAL(r.backendType, "logicalNameMapper");
102  BOOST_CHECK_EQUAL(r.address, "");
103  BOOST_CHECK_EQUAL(r.parameters.size(), 2);
104  BOOST_CHECK_EQUAL(r.parameters["map"], "myMapFile.xlmap");
105  BOOST_CHECK_EQUAL(r.parameters["a"], "b");

107  {

109  auto r = Utilities::parseDeviceDesciptor("(x?a=b=c)");
110  BOOST_CHECK_EQUAL(r.backendType, "x");
111  BOOST_CHECK_EQUAL(r.address, "");
112  BOOST_CHECK_EQUAL(r.parameters.size(), 1);
113  BOOST_CHECK_EQUAL(r.parameters["a"], "b=c");
114  }
115  {
116  // check escaping special characters
117  auto r = Utilities::parseDeviceDesciptor("(x:address\\?withQuestionmark?para=value\\&with\\&ampersand\\&&x="
118  "y\\\\&y=\\))");
119  BOOST_CHECK_EQUAL(r.backendType, "x");
120  BOOST_CHECK_EQUAL(r.address, "address?withQuestionmark");
121  BOOST_CHECK_EQUAL(r.parameters.size(), 3);
122  BOOST_CHECK_EQUAL(r.parameters["para"], "value&with&ampersand&");
123  BOOST_CHECK_EQUAL(r.parameters["x"], "y\\");
124  BOOST_CHECK_EQUAL(r.parameters["y"], ")");
125  }
126  {
127  // check nesting CDDs
128  auto r = Utilities::parseDeviceDesciptor("(nested:(pci:pcieunis6?map=dummy.map)?"

155 BOOST_AUTO_TEST_CASE(testParseSdm) {
156  Sdm sdm = Utilities::parseSdm(VALID_SDM);
157  BOOST_CHECK(sdm.host == ".");
158  BOOST_CHECK(sdm.interface == "pci");
159  BOOST_CHECK(sdm.instance == "pcieunidummys6");
160  BOOST_CHECK(sdm.parameters.size() == 0);
161  BOOST_CHECK(sdm.protocol == "undefined");
162 
163  sdm = Utilities::parseSdm(VALID_SDM_WITH_PARAMS);
164  BOOST_CHECK(sdm.host == ".");
165  BOOST_CHECK(sdm.interface == "dummy");
166  BOOST_CHECK(sdm.parameters.size() == 1);
167  BOOST_CHECK(sdm.parameters.front() == "goodMapFile.map");
168  BOOST_CHECK_THROW(Utilities::parseSdm(""),
169  ChimeraTK::logic_error); // Empty string
170  BOOST_CHECK_THROW(Utilities::parseSdm("sdm:"),
171  ChimeraTK::logic_error); // shorter than sdm:// signature
172  BOOST_CHECK_THROW(Utilities::parseSdm(INVALID_SDM), ChimeraTK::logic_error);
173  BOOST_CHECK_THROW(Utilities::parseSdm(INVALID_SDM_2), ChimeraTK::logic_error);
174  BOOST_CHECK_THROW(Utilities::parseSdm(INVALID_SDM_3), ChimeraTK::logic_error);
175  BOOST_CHECK_THROW(Utilities::parseSdm(INVALID_SDM_4), ChimeraTK::logic_error);
176  BOOST_CHECK_THROW(Utilities::parseSdm(INVALID_SDM_5), ChimeraTK::logic_error);
177 }
