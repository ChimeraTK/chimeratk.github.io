157  size_t npos = path.find_first_of(separator, pos + 1);

160  if(npos_alt != std::string::npos && npos_alt < npos) npos = npos_alt;

163  pos = npos;

191  std::string removeExtraSeparators(std::string string, std::string sep = separator) const {
192  std::size_t pos;
193  while((pos = string.find(std::string(sep) + sep)) != std::string::npos) {
194  string.erase(pos, 1);

202  std::string getWithOtherSeparatorReplaced(const std::string& otherSeparator) const {
203  // no replacement needed, of other separator is empty
204  if(otherSeparator.length() == 0) return path;
205  // replace all occurrences of otherSeparator with separator
206  std::string path_other = path;
207  size_t pos = 0;
208  while((pos = path_other.find(otherSeparator)) != std::string::npos) {
209  path_other.replace(pos, 1, std::string(separator));
210  }
211  // replace duplicate separators and remove one of them. Also remove a
212  // trailing separator, if present.
213  return removeExtraSeparators(path_other);
214  }
