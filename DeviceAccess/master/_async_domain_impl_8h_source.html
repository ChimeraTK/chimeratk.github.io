74  VersionNumber AsyncDomainImpl<BackendDataType>::distribute(BackendDataType data, VersionNumber version) {
75  std::lock_guard l(_mutex);

77  if(version == VersionNumber(nullptr)) {

103  template<typename BackendDataType>
104  VersionNumber AsyncDomainImpl<BackendDataType>::activate(BackendDataType data, VersionNumber version) {
105  std::lock_guard l(_mutex);
106  // everything incl. potential creation of a new version number must happen under the lock
107  if(version == VersionNumber(nullptr)) {
108  version = {};
109  }
110 
111  _isActive = true;
112 
113  auto distributor = _distributor.lock();
114  if(!distributor) {
115  return VersionNumber{nullptr};
116  }
117 
118  if(version >= _notDistributedVersion) {
119  distributor->activate(data, version);
120  _activationVersion = version;
121  }
122  else {
123  // Due to a race condition, it has been tried to distribute newer data before activate was called.
124  distributor->activate(_notDistributedData, _notDistributedVersion);
125  _activationVersion = _notDistributedVersion;
126  }
127  return _activationVersion;
128  }
