76  void AsyncDomainsContainer<KeyType>::distributeExceptions() {
77  std::string exceptionMessage;
78  while(true) {
79  // block to wait until setException has been called
80  try {
81  _startExceptionDistribution.pop_wait(exceptionMessage);
82  }
83  catch(StopThread&) {
84  return;

88 
89  {

91  for(auto& keyAndDomain : _asyncDomains) {

93  if(domain) {

96  }

100  }
