107  boost::shared_ptr<AsyncDomainImpl<BackendSpecificDataType>> AsyncDomainsContainer::getOrCreateAsyncDomain(
108  boost::shared_ptr<BackendType> backend, size_t key, bool activate) {
109  std::lock_guard<std::mutex> domainsLock(_domainsMutex);
110 
111  auto domain = _asyncDomains[key].lock();
112 
113  if(domain) {
114  auto domainImpl = boost::dynamic_pointer_cast<AsyncDomainImpl<BackendSpecificDataType>>(domain);
115  assert(domainImpl);
116  return domainImpl;
117  }
118 
119  // The domain does not exist, create it
120  auto domainImpl = boost::make_shared<AsyncDomainImpl<BackendSpecificDataType>>(backend, key);
121  // start the thread if not already running
122  { // thread lock scope
123  auto threadCreationLock = std::lock_guard(_threadCreationMutex);
124 
125  if(!_distributorThread.joinable()) {
126  _distributorThread = std::thread([&] { distributeExceptions(); });
127  _threadIsRunning = true;
128  }
129  } // end of thread lock scope
130 
131  if(activate) {
132  auto subscriptionDone = backend->activateSubscription(key, domainImpl);
133  // wait until the backend reports that the subscription is ready before proceeding with the creation of the
134  // accessor, which will poll the initial value.
135  subscriptionDone.wait();
136  auto [value, version] = backend->template getAsyncDomainInitialValue<BackendSpecificDataType>(key);
137  domainImpl->activate(value, version);
138  }
139 
140  _asyncDomains[key] = domainImpl;
141  return domainImpl;
142  }
