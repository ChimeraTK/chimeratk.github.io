50  template<typename KeyType>
51  void AsyncDomainsContainer<KeyType>::distributeExceptions() {
52  std::string exceptionMessage;
53  while(true) {
54  // block to wait until setException has been called
55  try {
56  _startExceptionDistribution.pop_wait(exceptionMessage);
57  }
58  catch(StopThread&) {
59  return;
60  }
61 
62  auto ex = std::make_exception_ptr(ChimeraTK::runtime_error(exceptionMessage));
63  for(auto& keyAndDomain : asyncDomains) {
64  auto domain = keyAndDomain.second.lock();
65  if(domain) {
66  domain->sendException(ex);
67  }
68  }
69 
70  _isSendingExceptions = false;
71  }
72  }
