69  void MathPlugin::openHook(boost::shared_ptr<LogicalNameMappingBackend> backend) {
70  // make sure backend catalogue is updated with target backend information
71  backend->getRegisterCatalogue();
72 
73  // check if this is the first call to openHook: perform tasks which could not be done in the constructor because
74  // the backend is not yet available there.
75  if(_backend._empty()) {
76  // make sure the register info is up to date
77  updateRegisterInfo();
78 
79  // store backend as weak pointer for later use
80  _backend = backend;
81 
82  // If write direction, check for push-type parameters if enabled
83  if(_isWrite && _enablePushParameters) {
84  for(auto& parpair : _parameters) {
85  auto paramFlags = backend->getRegisterCatalogue().getRegister(parpair.second)->getSupportedAccessModes();
86  if(paramFlags.has(AccessMode::wait_for_new_data)) {
87  _hasPushParameter = true;
88  break;
89  }
90  }
91 
92  // perpare for updates via push-type parameters
93  if(_hasPushParameter) {
94  // fill the _pushParameterReadGroup
95  for(auto& parpair : _parameters) {
96  // push-type parameters need to be obtained with wait_for_new_data, others without
97  AccessModeFlags flags{};
98  auto paramFlags = backend->getRegisterCatalogue().getRegister(parpair.second)->getSupportedAccessModes();
99  if(paramFlags.has(AccessMode::wait_for_new_data)) {
100  flags = {AccessMode::wait_for_new_data};
101  }
102  auto acc = backend->getRegisterAccessor<double>(parpair.second, 0, 0, flags);
103  if(acc->getNumberOfChannels() != 1) {
104  throw ChimeraTK::logic_error(
105  "The LogicalNameMapper MathPlugin supports only scalar or 1D array registers. Register name: '" +
106  _info.lock()->name + "', parameter name: '" + parpair.first + "'");
107  }
108  _pushParameterReadGroup.add(acc);
109  _pushParameterAccessorMap[parpair.first] = acc;
110  }
111  _pushParameterReadGroup.finalise();
112  _lastWrittenValue.resize(_info.lock()->length);
113  }
114 
115  // compile formula
116  try {
117  _h->compileFormula(_formula, _backend.lock(), _pushParameterAccessorMap, _info.lock()->length);
118  }
119  catch(...) {
120  // Do not throw errors at this point, only when accessing the register directly
121  _hasPushParameter = false;
122  }
123  }
124  }
125 
126  // if we have push-type parameters triggering a write of the target, start the _pushParameterWriteThread
127  if(_hasPushParameter) {
128  assert(_isWrite); // we do not set hasPushParameter if !isWrite
129 
130  // If the thread is still running, it will terminate soon by itself. We must not attempt to terminate it here,
131  // because it would be subject to the race condition that the thread still sees an exception from before recovery
132  // which results in putting the device back into exception state.
133  // The reason why we might need to wait here is that we cannot wait for the thread to terminate in the exception
134  // hook, because it might be called from inside the thread.
135  while(_pushParameterWriteThread.joinable()) {
136  _pushParameterReadGroup.interrupt();
137  _pushParameterWriteThread.try_join_for(boost::chrono::milliseconds(100));
138  }
139 
140  // start write thread
141  boost::barrier waitUntilThreadLaunched(2);
142  _pushParameterWriteThread = boost::thread([this, &waitUntilThreadLaunched] {
143  // obtain target accessor
144  auto targetDevice = BackendFactory::getInstance().createBackend(_info.lock()->deviceName);
145  auto info = _info.lock();
146  auto target = targetDevice->getRegisterAccessor<double>(info->registerName, info->length, info->firstIndex, {});
147  // empty all queues (initial values, remaining exceptions from previous thread runs). Ignore all exceptions.
148  while(true) {
149  auto notfy = _pushParameterReadGroup.waitAnyNonBlocking();
150  if(!notfy.isReady()) break;
151  try {
152  notfy.accept();
153  }
154  catch(...) {
155  continue;
156  }
157  }
158  assert(!_backend.lock()->_asyncReadActive);
159  assert(_lastWrittenValue.size() == target->getNumberOfSamples());
160 
161  // need to get to this point before the openHook completes
162  waitUntilThreadLaunched.wait();
163 
164  // start processing loop
165  while(true) {
166  try {
167  {
168  std::unique_lock<std::mutex> lk(_mx_lastWrittenValue);
169  _h->computeResult(_lastWrittenValue, target->accessChannel(0));
170  }
171  target->writeDestructively();
172  _pushParameterReadGroup.readAny();
173  }
174  catch(ChimeraTK::runtime_error& e) {
175  // no need to report the exception, as the accessor should have already done it
176  return;
177  }
178  }
179  });
180 
181  // do not proceed before the thread is ready to receive new data
182  waitUntilThreadLaunched.wait();
183  }
184  }

292  template<typename UserType>
293  void MathPluginDecorator<UserType>::doPreWrite(TransferType type, ChimeraTK::VersionNumber versionNumber) {
294  if(!_p->_isWrite) {
295  throw ChimeraTK::logic_error("This register with MathPlugin enabled is not writeable: " + _target->getName());
296  }
297 
298  // update parameters
299  for(auto& p : h.params) p.first->readLatest();
300 
301  // convert from UserType to double - use the target accessor's buffer as a temporary buffer (this is a bit a hack,
302  // but it is safe to overwrite the buffer and we can avoid the need for an additional permanent buffer which might
303  // not even be used if the register is never written).
304  for(size_t k = 0; k < buffer_2D[0].size(); ++k) {
305  _target->accessData(0, k) = userTypeToNumeric<double>(buffer_2D[0][k]);
306  }
307 
308  // update last written data buffer for updater thread if needed
309  if(_p->_hasPushParameter) {
310  std::unique_lock<std::mutex> lk(_p->_mx_lastWrittenValue);
311  _p->_lastWrittenValue = _target->accessChannel(0);
312  }
313 
314  // evaluate the expression and store into target accessor
315  h.computeResult(_target->accessChannel(0), _target->accessChannel(0));
316 
317  // pass validity to target and delegate preWrite
318  _target->setDataValidity(this->_dataValidity);
319  _target->preWrite(type, versionNumber);

369  void MathPluginFormulaHelper::compileFormula(const std::string& formula,
370  const boost::shared_ptr<LogicalNameMappingBackend>& backend,
371  const std::map<std::string, boost::shared_ptr<ChimeraTK::NDRegisterAccessor<double>>>& parameters,
372  size_t nElements) {
373  // create exprtk parser
374  exprtk::parser<double> parser;
375 
376  // add basic constants like pi
377  symbols.add_constants();
378 
379  // Create vector view for the value and add it to the symbol table. We need to use a vector view instead of adding
380  // the buffer directly as a vector, since our buffers might be swapped and hence the address of the data can
381  // change. The pointer used for the view is for now to a temporary vector and will become invalid once this function
382  // returns. This is acceptable, since before using the view the pointer will be updated to the right buffer.
383  // Using a nullptr instead of the temporary buffer does not work, as the buffer seems to be accessible during
384  // compilation of the formula.
385  std::vector<double> temp(nElements);
386  valueView = std::make_unique<exprtk::vector_view<double>>(exprtk::make_vector_view(temp, nElements));
387  symbols.add_vector("x", *valueView);
388 
389  // iterate parameters, add all but 'formula' parameter as a variable.
390  for(auto& parpair : parameters) {
391  if(parpair.first == "formula") continue;
392  auto& acc = parpair.second;
393  acc->setExceptionBackend(backend);
394  if(acc->getNumberOfChannels() != 1) {
395  throw ChimeraTK::logic_error(
396  "The LogicalNameMapper MathPlugin supports only scalar or 1D array registers. Register name: '" + varName +
397  "', parameter name: '" + parpair.first + "'");
398  }
399  auto view = std::make_unique<exprtk::vector_view<double>>(
400  exprtk::make_vector_view(acc->accessChannel(0), acc->getNumberOfSamples()));
401  symbols.add_vector(parpair.first, *view);
402  params[acc] = std::move(view);
403  }
404 
405  // compile the expression
406  expression.register_symbol_table(symbols);
407  bool success = parser.compile(formula, expression);
408  if(!success) {
409  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + varName +
410  "': failed to compile expression '" + formula + "': " + parser.error());
411  }
412  }

416  template<typename T>
417  void MathPluginFormulaHelper::computeResult(std::vector<double>& x, std::vector<T>& resultBuffer) {
418  // inform the value view of the new data pointer - the buffer might have been swapped
419  valueView->rebase(x.data());
420 
421  // update parameter buffers
422  for(auto& p : params) {
423  p.second->rebase(p.first->accessChannel(0).data());
424  }
425 
426  // evaluate the expression, obtain the result in a way so it also works when using the return statement
427  double valueWhenNotUsingReturn = expression.value();
428  exprtk::results_context<double> results = expression.results();
429 
430  // extract result depending on how it was returned
431  if(results.count() == 0) {
432  // if results.count() is 0, the return statement presumably has not been used
433  if(resultBuffer.size() != 1) {
434  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + varName +
435  "': The expression returns a scalar but " + std::to_string(resultBuffer.size()) + " expected.");
436  }
437  resultBuffer[0] = numericToUserType<T>(valueWhenNotUsingReturn);
438  }
439  else if(results.count() == 1) {
440  // return statement has been used to return exactly one value (note: this value might be an array)
441  exprtk::type_store<double> result = results[0];
442 
443  // make sure we got a numeric result
444  if(result.type != exprtk::type_store<double>::e_scalar && result.type != exprtk::type_store<double>::e_vector) {
445  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + varName +
446  "': The expression did not return a numeric result.");
447  }
448 
449  // create vector view and check that its size matches
450  exprtk::type_store<double>::type_view<double> view(result);
451  if(view.size() != resultBuffer.size()) {
452  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + varName +
453  "': The expression returns " + std::to_string(view.size()) + " elements but " +
454  std::to_string(resultBuffer.size()) + " expected.");
455  }
456 
457  // convert and copy data into target buffer
458  for(size_t k = 0; k < view.size(); ++k) {
459  resultBuffer[k] = numericToUserType<T>(view[k]);
460  }
461  }
462  else {
463  // multiple results in return statement are unexpected
464  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + varName +
465  "': The expression returned " + std::to_string(results.count()) + " results, expect exactly one result.");
466  }
467  }
