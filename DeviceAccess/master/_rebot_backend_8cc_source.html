22  std::unique_ptr<RebotProtocolImplementor> getProtocolImplementor(boost::shared_ptr<Rebot::Connection>& c) {
23  auto serverVersion = getProtocolVersion(c);
24  if(serverVersion == 0) {
25  return std::make_unique<RebotProtocol0>(c);
26  }
27  if(serverVersion == 1) {
28  return std::make_unique<RebotProtocol1>(c);
29  }
30  c->close();
31  std::stringstream errorMessage;
32  errorMessage << "Server protocol version " << serverVersion << " not supported!";
33  throw ChimeraTK::runtime_error(errorMessage.str());
34  }

36  uint32_t getProtocolVersion(boost::shared_ptr<Rebot::Connection>& c) {
37  // send a negotiation to the server:
38  // sendClientProtocolVersion
39  std::vector<uint32_t> clientHelloMessage;
40  clientHelloMessage.push_back(Rebot::HELLO_TOKEN);
41  clientHelloMessage.push_back(Rebot::MAGIC_WORD);
42  clientHelloMessage.push_back(Rebot::CLIENT_PROTOCOL_VERSION);
43 
44  c->write(clientHelloMessage);
45 
46  // Kludge is needed to work around server bug.
47  // We have a bug with the old version were only one word is returned for
48  // multiple unrecognized command. Fetching one word for the 3 words send is a
49  // workaround.
50  auto serverHello = c->read(1);
51 
52  if(serverHello.at(0) == static_cast<uint32_t>(Rebot::UNKNOWN_INSTRUCTION)) {
53  return 0; // initial protocol version 0.0
54  }
55 
56  auto remainingBytesOfAValidServerHello = c->read(Rebot::LENGTH_OF_HELLO_TOKEN_MESSAGE - 1);
57 
58  serverHello.insert(
59  serverHello.end(), remainingBytesOfAValidServerHello.begin(), remainingBytesOfAValidServerHello.end());
60  return parseRxServerHello(serverHello);
61  }

63  uint32_t parseRxServerHello(const std::vector<uint32_t>& serverHello) {
64  // 3 rd element/word is the version word
65  return serverHello.at(2);
66  }

170  void RebotBackend::heartbeatLoop(const boost::shared_ptr<ThreadInformerMutex>& threadInformerMutex) {
171  while(true) {
172  try {
173  // only send a heartbeat if the connection was inactive for half of the
174  // timeout period
175 
176  // We can only calculate this while holding the lock (because we need _lastSendTime), but we have to use
177  // it when not holding the lock because we are calling sleep. Hence we have to store the next wakekup time in a
178  // variable.
179  boost::chrono::steady_clock::time_point wakeupTime;
180 
181  { // scope of the lock guard. We must hold it to safely access _lastSendTime, which we need in the if statement
182  std::lock_guard<std::mutex> lock(_threadInformerMutex->mutex);
183  if((testable_rebot_sleep::now() - _lastSendTime) > boost::chrono::milliseconds(_connectionTimeout / 2)) {
184  // To handle the race condition that the thread woke up while the
185  // desructor was holding the lock and closes the socket: Check the flag
186  // and quit if set
187  if(threadInformerMutex->quitThread) {
188  break;
189  }
190  // always update the last send time. Otherwise the sleep will be
191  // ineffective for a closed connection and go to 100 & CPU load
192  _lastSendTime = testable_rebot_sleep::now();
193  if(_protocolImplementor) {
194  _protocolImplementor->sendHeartbeat();
195  }
196  }
197  // Sleep for half of the connection timeout (plus 1 ms)
198  wakeupTime = _lastSendTime + boost::chrono::milliseconds(_connectionTimeout / 2 + 1);
199  } // scope of the lock guard
200 
201  // sleep without holding the lock.
202  testable_rebot_sleep::sleep_until(wakeupTime);
203  }
204  catch(ChimeraTK::runtime_error& e) {
205  std::cerr << "RebotBackend: Sending heartbeat failed. Caught exception: " << e.what() << std::endl;
206  std::cerr << "Closing connection." << std::endl;
207  close();
208  }
209  } // while true
210  }
