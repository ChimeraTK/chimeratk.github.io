156  void RebotBackend::heartbeatLoop(boost::shared_ptr<ThreadInformerMutex> threadInformerMutex) {
157  while(true) {
158  try {
159  // only send a heartbeat if the connection was inactive for half of the
160  // timeout period
161 
162  //We can only calculate this while holding the lock (because we need _lastSendTime), but we have to use
163  //it when not holding the lock because we are calling sleep. Hence we have to store the next wakekup time in a variable.
164  boost::chrono::steady_clock::time_point wakeupTime;
165 
166  { // scope of the lock guard. We must hold it to safely access _lastSendTime, which we need in the if statement
167  std::lock_guard<std::mutex> lock(_threadInformerMutex->mutex);
168  if((testable_rebot_sleep::now() - _lastSendTime) > boost::chrono::milliseconds(_connectionTimeout / 2)) {
169  // To handle the race condition that the thread woke up while the
170  // desructor was holding the lock and closes the socket: Check the flag
171  // and quit if set
172  if(threadInformerMutex->quitThread) {
173  break;
174  }
175  // always update the last send time. Otherwise the sleep will be
176  // ineffective for a closed connection and go to 100 & CPU load
177  _lastSendTime = testable_rebot_sleep::now();
178  if(_protocolImplementor) {
179  _protocolImplementor->sendHeartbeat();
180  }
181  }
182  // Sleep for half of the connection timeout (plus 1 ms)
183  wakeupTime = _lastSendTime + boost::chrono::milliseconds(_connectionTimeout / 2 + 1);
184  } //scope of the lock guard
185 
186  // sleep without holding the lock.
187  testable_rebot_sleep::sleep_until(wakeupTime);
188  }
189  catch(ChimeraTK::runtime_error& e) {
190  std::cerr << "RebotBackend: Sending heartbeat failed. Caught exception: " << e.what() << std::endl;
191  std::cerr << "Closing connection." << std::endl;
192  close();
193  }
194  } // while true
195  }
