308  inline bool ReadAnyGroup::Notification::accept() {
309  if(!this->valid) {
310  throw std::logic_error("This notification object is invalid.");
311  }
312  if(this->accepted) {
313  throw std::logic_error("This notification has already been accepted.");
314  }
315  this->accepted = true;
316  try {
317  _owner->push_elements[index].getHighLevelImplElement()->_readQueue.pop_wait();
318  }
319  catch(detail::DiscardValueException&) {
320  // we must not call postRead() in this case, hence we do not call preRead()
321  _owner->_lastOperationIndex = std::numeric_limits<size_t>::max() - 1;
322  return false;
323  }
324  _owner->_lastOperationIndex = index;
325  _owner->push_elements[index].getHighLevelImplElement()->postRead(TransferType::read, true);
326  return true;
327  }

416  inline void ReadAnyGroup::finalise() {
417  if(isFinalised)
418  throw std::logic_error("ReadAnyGroup has already been finalised, calling "
419  "finalise() is no longer allowed.");
420  std::vector<cppext::future_queue<void>> queueList;
421  bool groupEmpty = true;
422  for(auto& e : push_elements) {
423  queueList.push_back(e.getHighLevelImplElement()->_readQueue);
424  groupEmpty = false;
425  }
426  if(groupEmpty) {
427  throw std::logic_error("ReadAnyGroup has no element with AccessMode::wait_for_new_data.");
428  }
429  notification_queue = cppext::when_any(queueList.begin(), queueList.end());
430  isFinalised = true;
431  }

464  inline void ReadAnyGroup::handlePreRead() {
465  // preRead() and postRead() must be called in pairs. Hence we call all preReads here before waiting for transfers to
466  // finish. postRead() will be called when accepting the notification. We can call preRead() repeatedly on the same
467  // element, even if no transfer and call to postRead() have happened. It is just ignored (see Transfer element spec
468  // B.5.2). Since this has a performance impact which might be significant on big applications, we try to avoid
469  // unnecessary calls anyway.
470 
471  // Notice : This has the side effect that decorators can block here, for instance for the setup phase. This is used
472  // by ApplicationCore in testable mode.
473  if(_lastOperationIndex == std::numeric_limits<size_t>::max()) {
474  for(auto& elem : push_elements) {
475  elem.getHighLevelImplElement()->preRead(TransferType::read);
476  }
477  }
478  else if(_lastOperationIndex != std::numeric_limits<size_t>::max() - 1) {
479  // Note: _lastOperationIndex is set to std::numeric_limits<size_t>::max() - 1 in case a DiscardValueException
480  // has been seen, in which case no postRead() is called.
481  push_elements[_lastOperationIndex].getHighLevelImplElement()->preRead(TransferType::read);
482  }
483  }

498  inline ReadAnyGroup::Notification ReadAnyGroup::waitAnyNonBlocking() {
499  // check if update is available
500  if(notification_queue.empty()) {
501  // If no notification is present, do not even execute preRead. This is necessary for two reasons:
502  // - We always used TransferType::read to avoid mixing TransferType::read and TransferType::readNonBlocking
503  // in the same transfer of the same variable. We can do this even in this non-blocking case, if we already know
504  // that there will be an update read, since there will not be any difference beyond this point.
505  // - In ApplicationCore testable mode, the testable mode lock must be released in a preRead before any blocking
506  // read operation. If preRead is called here when no update is available, no preRead will be called in a
507  // possible subsequend blocking readAny(), hence there would be no way to release the testable mode lock in the
508  // right place.
509  return {};
510  }
511 
512  // now that we know that an update is available, we can defer to waitAny()
513  return waitAny();
514  }

540  inline void ReadAnyGroup::readUntilAll(const std::vector<TransferElementID>& ids) {
541  std::map<TransferElementID, bool> found;
542  for(auto& id : ids)
543  found[id] = false; // initialise map so we can tell from the map if a
544  // variable is in the vector
545  size_t leftToFind = ids.size();
546  while(true) {
547  auto read = readAny();
548  if(found.count(read) == 0) continue; // variable is not in the vector ids
549  if(found[read] == true) continue; // variable has been read already
550  found[read] = true;
551  --leftToFind;
552  if(leftToFind == 0) return;
553  }
554  }
