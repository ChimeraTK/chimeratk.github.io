345  inline bool ReadAnyGroup::Notification::accept() {
346  if(!this->valid) {
347  throw std::logic_error("This notification object is invalid.");
348  }
349  if(this->accepted) {
350  throw std::logic_error("This notification has already been accepted.");
351  }
352  this->accepted = true;
353  try {
354  _owner->push_elements[index].getHighLevelImplElement()->_readQueue.pop_wait();
355  }
356  catch(ChimeraTK::runtime_error&) {
357  _owner->push_elements[index].getHighLevelImplElement()->_activeException = std::current_exception();
358  }
359  catch(boost::thread_interrupted&) {
360  _owner->push_elements[index].getHighLevelImplElement()->_activeException = std::current_exception();
361  }
362  catch(detail::DiscardValueException&) {
363  // we must not call postRead() in this case, hence we do not call preRead()
364  _owner->_lastOperationIndex = std::numeric_limits<size_t>::max() - 1;
365  return false;
366  }
367  _owner->_lastOperationIndex = index;
368  _owner->push_elements[index].getHighLevelImplElement()->postRead(TransferType::read, true);
369  return true;
370  }

465  inline void ReadAnyGroup::finalise() {
466  if(isFinalised) {
467  throw std::logic_error("ReadAnyGroup has already been finalised, calling "
468  "finalise() is no longer allowed.");
469  }
470  std::vector<cppext::future_queue<void>> queueList;
471  bool groupEmpty = true;
472  for(auto& e : push_elements) {

477  throw std::logic_error("ReadAnyGroup has no element with AccessMode::wait_for_new_data.");

514  inline void ReadAnyGroup::handlePreRead() {
515  // preRead() and postRead() must be called in pairs. Hence we call all preReads here before waiting for transfers to
516  // finish. postRead() will be called when accepting the notification. We can call preRead() repeatedly on the same
517  // element, even if no transfer and call to postRead() have happened. It is just ignored (see Transfer element spec
518  // B.5.2). Since this has a performance impact which might be significant on big applications, we try to avoid
519  // unnecessary calls anyway.
520 
521  // Notice : This has the side effect that decorators can block here, for instance for the setup phase. This is used
522  // by ApplicationCore in testable mode.
523  if(_lastOperationIndex == std::numeric_limits<size_t>::max()) {
524  for(auto& elem : push_elements) {
525  elem.getHighLevelImplElement()->preRead(TransferType::read);
526  }
527  }
528  else if(_lastOperationIndex != std::numeric_limits<size_t>::max() - 1) {
529  // Note: _lastOperationIndex is set to std::numeric_limits<size_t>::max() - 1 in case a DiscardValueException
530  // has been seen, in which case no postRead() is called.
531  push_elements[_lastOperationIndex].getHighLevelImplElement()->preRead(TransferType::read);
532  }
533  }

550  inline ReadAnyGroup::Notification ReadAnyGroup::waitAnyNonBlocking() {
551  restart_after_discard_value:
552  // check if update is available
553  if(notification_queue.empty()) {
554  // If no notification is present, do not even execute preRead. This is necessary for two reasons:
555  // - We always used TransferType::read to avoid mixing TransferType::read and TransferType::readNonBlocking
556  // in the same transfer of the same variable. We can do this even in this non-blocking case, if we already know
557  // that there will be an update read, since there will not be any difference beyond this point.
558  // - In ApplicationCore testable mode, the testable mode lock must be released in a preRead before any blocking
559  // read operation. If preRead is called here when no update is available, no preRead will be called in a
560  // possible subsequend blocking readAny(), hence there would be no way to release the testable mode lock in the
561  // right place.
562  return {};
563  }
564 
565  // if update is available, peek into the queue to check whether a DiscardValueException will be read
566  auto id = notification_queue.front();
567  try {
568  // call to empty() necessary before call to front(), to gain ownership of front element
569  if(push_elements[id].getHighLevelImplElement()->_readQueue.empty()) {
570  // cannot place the call to empty() into the assert, since it must be executed also in release builds
571  assert(false);
572  }
573  push_elements[id].getHighLevelImplElement()->_readQueue.front();
574  }
575  catch(detail::DiscardValueException&) {
576  // Remove discarded transfer from the queues and go back to square one
577  notification_queue.pop();
578  try {
579  push_elements[id].getHighLevelImplElement()->_readQueue.pop();
580  }
581  catch(detail::DiscardValueException&) {
582  goto restart_after_discard_value;
583  }
584  assert(false); // we must never end up at this point
585  }
586  catch(ChimeraTK::runtime_error&) {
587  // While peeking we found another runtime which is stored in the queue.
588  // Don't let it through, but leave it on the queue and continue with the waitAny().
589  // It will be handled later.
590  }
591  catch(boost::thread_interrupted&) {
592  // Also suppress the thread_interrupted exception here.
593  // It will stay on the queue, hence it's not lost and will be handled later.
594  }
595  catch(...) {
596  std::cout << "Fatal ERROR in ReadAnyGroup: Found unexpected exception on the read queue. Terminating!"
597  << std::endl;
598  std::terminate();
599  }
600 
601  // now that we know that an update is available, we can defer to waitAny()
602  return waitAny();
603  }

631  inline void ReadAnyGroup::readUntilAll(const std::vector<TransferElementID>& ids) {
632  std::map<TransferElementID, bool> found;
633  for(const auto& id : ids) {
634  found[id] = false; // initialise map so we can tell from the map if a
635  // variable is in the vector
636  }
637  size_t leftToFind = ids.size();
638  while(true) {
639  auto read = readAny();
640  if(found.count(read) == 0) continue; // variable is not in the vector ids
641  if(found[read]) continue; // variable has been read already
642  found[read] = true;
643  --leftToFind;
644  if(leftToFind == 0) return;
645  }
646  }
