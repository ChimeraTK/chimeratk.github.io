24  void DummyInterruptControllerHandler::handle(VersionNumber version) {
25  try {
26  _activeInterrupts->read();
27  for(uint32_t i = 0; i < 32; ++i) {
28  if(_activeInterrupts->accessData(0) & 0x1U << i) {
29  try {
30  auto distributor = _distributors.at(i).lock();
31  if(distributor) {
32  distributor->distribute(nullptr, version);
33  }
34  }
35  catch(std::out_of_range&) {
36  _backend->setException(
37  "ERROR: DummyInterruptControllerHandler reports unknown active interrupt " + std::to_string(i));
38  }
39  }
40  }
41  }
42  catch(ChimeraTK::runtime_error&) {
43  // Nothing to do. The transferElement part of _activeInterrupts has already called the backend's setException
44  }
45  }
