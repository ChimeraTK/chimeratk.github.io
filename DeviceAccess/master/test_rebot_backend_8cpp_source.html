135 void RebotTestClass::testReadWriteAPIOfRebotBackend() {
136  ChimeraTK::RebotBackend rebotBackend(_rebotServer.ip, _rebotServer.port);
137  rebotBackend.open();
138 
139  /*****************
140  * The dummy server wites 0xDEADDEAD to the start address 0x04. Use this for
141  * testing
142  */
143  uint32_t address = 0x04;
144  int32_t readValue = 0;
145  rebotBackend.read(uint64_t(0), address, &readValue, sizeof(readValue));
146  BOOST_CHECK_EQUAL(0xDEADBEEF, readValue);
147 
148  /****************************************************************************/
149  // Single word read - Hardcoding addresses for now
150  uint64_t word_status_register_address = 0x8;
151  int32_t data = -987;
152  // Register
153  rebotBackend.write(0, word_status_register_address, &data, sizeof(data));
154 
155  rebotBackend.read(0, word_status_register_address, &readValue, sizeof(readValue));
156 
157  BOOST_CHECK_EQUAL(data, readValue);
158  /****************************************************************************/
159 
160  // Multiword read/write
161  uint32_t word_clk_mux_addr = 28;
162  int32_t dataToWrite[4] = {rand(), rand(), rand(), rand()};
163  int32_t readInData[4];
164 
165  rebotBackend.write(0, word_clk_mux_addr, dataToWrite, sizeof(dataToWrite));
166  rebotBackend.read(0, word_clk_mux_addr, readInData, sizeof(readInData));
167 
168  for(int i = 0; i < 4; i++) {
169  BOOST_CHECK_EQUAL(dataToWrite[i], readInData[i]);
170  }
171 
172  uint32_t test_area_Addr = 0x00000030;
173  std::vector<int32_t> test_area_data(1024);
174  for(uint32_t i = 0; i < test_area_data.size(); ++i) {
175  test_area_data.at(i) = i;
176  }
177  std::vector<int32_t> test_area_ReadIndata(1024);
178 
179  rebotBackend.write(0, test_area_Addr, test_area_data.data(), sizeof(int32_t) * test_area_data.size());
180  rebotBackend.read(0, test_area_Addr, test_area_ReadIndata.data(), sizeof(int32_t) * test_area_ReadIndata.size());
181 
182  for(uint32_t i = 0; i < test_area_ReadIndata.size(); i++) {
183  BOOST_CHECK_EQUAL(test_area_data[i], test_area_ReadIndata[i]);
184  }
