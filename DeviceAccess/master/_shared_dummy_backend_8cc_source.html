36  void SharedDummyBackend::setupBarContents() {
37  for(auto& _barSizesInByte : _barSizesInBytes) {
38  std::string barName = SHARED_MEMORY_BAR_PREFIX + std::to_string(_barSizesInByte.first);
39 
40  size_t barSizeInWords = (_barSizesInByte.second + sizeof(int32_t) - 1) / sizeof(int32_t);

42  try {

67  void SharedDummyBackend::read(uint64_t bar, uint64_t address, int32_t* data, size_t sizeInBytes) {
68  if(!_opened) {
69  throw ChimeraTK::logic_error("Device is closed.");
70  }
71  if(_hasActiveException) {
72  throw ChimeraTK::runtime_error("previous, unrecovered fault");
73  }
74  checkSizeIsMultipleOfWordSize(sizeInBytes);
75  uint64_t wordBaseIndex = address / sizeof(int32_t);
76 
77  std::lock_guard<boost::interprocess::named_mutex> lock(sharedMemoryManager.interprocessMutex);
78  for(uint64_t wordIndex = 0; wordIndex < sizeInBytes / sizeof(int32_t); ++wordIndex) {
79  TRY_REGISTER_ACCESS(data[wordIndex] = _barContents[bar]->at(wordBaseIndex + wordIndex););
80  }
81  }

226  void SharedDummyBackend::InterruptDispatcherThread::run() {
227  // copy interrupt counts at the beginning, and then
228  // only look for different values. count up all values till they match
229  // map (controller,intNumber) -> count
230  // use map instead of vector because search is more efficient
231  std::map<std::pair<int, int>, std::uint32_t> lastInterruptState;
232  {
233  std::lock_guard<boost::interprocess::named_mutex> lock(_dispatcherInterf->_shmMutex);
234  for(auto& entry : _semShm->interruptEntries) {
235  assert(entry._controllerId == 0);
236  if(!entry.used) continue;
237  lastInterruptState[std::make_pair(entry._controllerId, entry._intNumber)] = entry._counter;
238  }
239  // we register a semaphore only after being ready
240  _sem = _semShm->addSem(_semId);
241  _started = true;
242  }
243 
244  // local copy of shm contents, used to reduce lock time
245  InterruptEntry interruptEntries[maxInterruptEntries];
246 
247  while(!_stop) {
248  _sem->wait();
249  {
250  std::lock_guard<boost::interprocess::named_mutex> lock(_dispatcherInterf->_shmMutex);
251  std::memcpy(interruptEntries, _semShm->interruptEntries, sizeof(interruptEntries));
252  }
253  for(auto& entry : interruptEntries) {
254  assert(entry._controllerId == 0);
255  if(!entry.used) continue;
256 
257  // find match with controllerId and intNumber
258  auto key = std::make_pair(entry._controllerId, entry._intNumber);
259  auto it = lastInterruptState.find(key);
260  if(it != lastInterruptState.end()) {
261  while(it->second != entry._counter) {
262  // call trigger/dispatch
263 #ifdef _DEBUG
264  std::cout << "existing interrupt event for x,y = " << entry._controllerId << ", " << entry._intNumber
265  << std::endl;
266 #endif
267  handleInterrupt(entry._intNumber);
268  it->second++;
269  }
270  }
271  else {
272  // new interrupt number
273  // call trigger/dispatch count times
274 #ifdef _DEBUG
275  std::cout << "count = " << entry._counter << " interrupt events for x,y = " << entry._controllerId << ", "
276  << entry._intNumber << std::endl;
277 #endif
278  handleInterrupt(entry._intNumber);
279  lastInterruptState[key] = entry._counter;
280  }
281  }
282  }
283  }

285  void SharedDummyBackend::InterruptDispatcherThread::stop() noexcept {
286  _stop = true;
287  // we must wait until the semaphore is registered
288  try {
289  while(!_started) {
290  boost::this_thread::sleep_for(boost::chrono::milliseconds{10});
291  }
292  }
293  catch(const boost::thread_interrupted&) {
294  // Simply suppress the thread_interrupted here. This function is only called
295  // within a destructor, which anyway would have terminated the program when it sees the exception.
296  // There are two possible scenarios what can happen now.
297  // 1. _started is set and the destruction can continue normally.
298  // 2. _started is not set yet an we don't know if the semaphore is in the correct state.
299  // Again there are two possibilities:
300  // 2a_ The semaphore was set correctly and the destructor continues normally.
301  // 2b_ sem->post() throws and terminate() is called (which otherwise would have been called from the escaping
302  // thread_interrupted
303  }
304  catch(const boost::system::system_error&) {
305  // if something went really wrong we terminate here
306  std::terminate();
307  }
308 
309  try {
310  _sem->post();
311  }
312  catch(const boost::interprocess::interprocess_exception&) {
313  std::terminate();
314  }
315  }

328  SharedDummyBackend::ShmForSems::Sem* SharedDummyBackend::ShmForSems::addSem(SemId semId) {
329  // look up whether semaphore for id already exists and return error
330  for(auto& entry : semEntries) {
331  if(entry.used && entry.semId == semId) {
332  throw logic_error("error: semId already exists - check assumption about identifiers!");
333  }
334  }
335 
336  for(auto& entry : semEntries) {
337  if(!entry.used) {
338  entry.semId = semId;
339  entry.used = true;
340  // It would be nice to also reset semaphores state, but if
341  // interrupt dispatcher thread which last used it terminated property its not necessary
342  // (since it calls post in destructor)
343  return &entry.s;
344  }
345  }
346  // increasing size not implemented
347  throw runtime_error("error: semaphore array full - increase maxSems!");
348  }

371  void SharedDummyBackend::ShmForSems::addInterrupt(uint32_t interruptNumber) {
372  bool found = false;
373  for(auto& entry : interruptEntries) {
374  if(entry.used && entry._controllerId == 0 && static_cast<uint32_t>(entry._intNumber) == interruptNumber) {
375  entry._counter++;
376  found = true;
377  break;
378  }
379  }
380  if(!found) {
381  bool added = false;
382  for(auto& entry : interruptEntries) {
383  if(!entry.used) {
384  entry.used = true;
385  entry._controllerId = 0;
386  entry._intNumber = static_cast<int>(interruptNumber);
387  entry._counter = 1;
388  added = true;
389  break;
390  }
391  }
392  if(!added) {
393  throw runtime_error("no place left in interruptEntries!");
394  }
395  }
396  }
