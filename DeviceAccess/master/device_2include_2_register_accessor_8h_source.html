46  template<typename ConvertedDataType>
47  void read(ConvertedDataType* convertedData, size_t nWords = 1, uint32_t wordOffsetInRegister = 0) const {
48  if(nWords == 0) return;
49  // obtain the accessor through the handler
50  auto& accessorHandler = boost::fusion::at_key<ConvertedDataType>(_convertingAccessorHandlers.table);
51  accessorHandler.checkAndResize(nWords, wordOffsetInRegister, false /*not raw*/, _backend, _registerPathName);
52  // now we can be sure that the accessor is valid and has enough memory to
53  // perform the memcpy
54  accessorHandler.accessor->read();
55  // copy data to target buffer. The accessor might not start at index 0, so
56  // we have to calculate the correct offset inside the accessor
57  memcpy(convertedData,
58  accessorHandler.accessor->accessChannel(0).data() + wordOffsetInRegister - accessorHandler.beginIndex,
59  nWords * sizeof(ConvertedDataType));
60  }

203  template<class UserType>
204  class AccessorHandler {
205  public:
206  AccessorHandler() : beginIndex(0), endIndex(0) {}
207 
208  boost::shared_ptr<NDRegisterAccessor<UserType>> accessor;
209  size_t beginIndex; // the first index of the accessor;
210  size_t endIndex; // the end index (first index after the valid range, liken
211  // end iterator)
212  void checkAndResize(size_t nWords, size_t wordOffsetInRegister, bool isRaw,
213  boost::shared_ptr<DeviceBackend> const& backend, RegisterPath const& registerPathName) {
214  // no need for out of register size checking. getRegisterAccessor does the
215  // job for us and throws in case of error.
216  if((wordOffsetInRegister < beginIndex) || (wordOffsetInRegister + nWords > (endIndex) || !accessor)) {
217  // we have to (re)allocate
218  size_t newBeginIndex;
219  if(!accessor) {
220  newBeginIndex = wordOffsetInRegister;
221  }
222  else {
223  newBeginIndex = std::min(wordOffsetInRegister, beginIndex);
224  }
225  size_t newEndIndex = std::max(wordOffsetInRegister + nWords, endIndex);
226  accessor =
227  backend->getRegisterAccessor<UserType>(registerPathName, newEndIndex - newBeginIndex, newBeginIndex,
228  (isRaw ? AccessModeFlags{AccessMode::raw} : AccessModeFlags{})); // 0 offset, raw
229  // we have to update the accessor with the current hardware information.
230  // It might be that a partial write is going to happen.
231  accessor->read();
232  // only if creating the new accessor succeeds we change the index
233  // bookkeeping variables
234  beginIndex = newBeginIndex;
235  endIndex = newEndIndex;
236  }
237  }
238  };
