48  template<typename ConvertedDataType>
49  void read(ConvertedDataType* convertedData, size_t nWords = 1, uint32_t wordOffsetInRegister = 0) const {
50  if(nWords == 0) return;
51  // obtain the accessor through the handler
52  auto& accessorHandler = boost::fusion::at_key<ConvertedDataType>(_convertingAccessorHandlers.table);
53  accessorHandler.checkAndResize(nWords, wordOffsetInRegister, false /*not raw*/, _backend, _registerPathName);
54  // now we can be sure that the accessor is valid and has enough memory to
55  // perform the memcpy
56  accessorHandler.accessor->read();
57  // copy data to target buffer. The accessor might not start at index 0, so
58  // we have to calculate the correct offset inside the accessor
59  memcpy(convertedData,
60  accessorHandler.accessor->accessChannel(0).data() + wordOffsetInRegister - accessorHandler.beginIndex,
61  nWords * sizeof(ConvertedDataType));

205  template<class UserType>
206  class AccessorHandler {
207  public:
208  AccessorHandler() : beginIndex(0), endIndex(0) {}
209 
210  boost::shared_ptr<NDRegisterAccessor<UserType>> accessor;
211  size_t beginIndex; // the first index of the accessor;
212  size_t endIndex; // the end index (first index after the valid range, liken
213  // end iterator)
214  void checkAndResize(size_t nWords, size_t wordOffsetInRegister, bool isRaw,
215  boost::shared_ptr<DeviceBackend> const& backend, RegisterPath const& registerPathName) {
216  // no need for out of register size checking. getRegisterAccessor does the
217  // job for us and throws in case of error.
218  if((wordOffsetInRegister < beginIndex) || (wordOffsetInRegister + nWords > (endIndex) || !accessor)) {
219  // we have to (re)allocate
220  size_t newBeginIndex;
221  if(!accessor) {
222  newBeginIndex = wordOffsetInRegister;
223  }
224  else {
225  newBeginIndex = std::min(wordOffsetInRegister, beginIndex);
226  }
227  size_t newEndIndex = std::max(wordOffsetInRegister + nWords, endIndex);
228  accessor =
229  backend->getRegisterAccessor<UserType>(registerPathName, newEndIndex - newBeginIndex, newBeginIndex,
230  (isRaw ? AccessModeFlags{AccessMode::raw} : AccessModeFlags{})); // 0 offset, raw
231  // we have to update the accessor with the current hardware information.
232  // It might be that a partial write is going to happen.
233  accessor->read();
234  // only if creating the new accessor succeeds we change the index
235  // bookkeeping variables
236  beginIndex = newBeginIndex;
237  endIndex = newEndIndex;
238  }
239  }
240  };
