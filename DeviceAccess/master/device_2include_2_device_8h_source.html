641  size_t wordOffsetInRegister, const AccessModeFlags& flags) const {
642  auto acc = getOneDRegisterAccessor<UserType>(registerPathName, numberOfWords, wordOffsetInRegister, flags);

660  template<typename UserType>
661  void Device::write(const RegisterPath& registerPathName, const std::vector<UserType>& vector,
662  size_t wordOffsetInRegister, const AccessModeFlags& flags) {
663  auto acc = getOneDRegisterAccessor<UserType>(registerPathName, vector.size(), wordOffsetInRegister, flags);
664  // the vector stays constant, but we have to work around so we can swap it and
665  // later swap it back...
666  std::vector<UserType>& mutable_vector = const_cast<std::vector<UserType>&>(vector);
667  acc.swap(mutable_vector);
668  acc.write();
669  acc.swap(mutable_vector);
670  }
