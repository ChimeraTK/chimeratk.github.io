637  std::vector<UserType> Device::read(const RegisterPath& registerPathName, size_t numberOfWords,
638  size_t wordOffsetInRegister, const AccessModeFlags& flags) const {
639  auto acc = getOneDRegisterAccessor<UserType>(registerPathName, numberOfWords, wordOffsetInRegister, flags);
640  acc.read();
641  std::vector<UserType> vector(acc.getNElements());
642  acc.swap(vector);

657  template<typename UserType>
658  void Device::write(const RegisterPath& registerPathName, const std::vector<UserType>& vector,
659  size_t wordOffsetInRegister, const AccessModeFlags& flags) {
660  auto acc = getOneDRegisterAccessor<UserType>(registerPathName, vector.size(), wordOffsetInRegister, flags);
661  // the vector stays constant, but we have to work around so we can swap it and
662  // later swap it back...
663  std::vector<UserType>& mutable_vector = const_cast<std::vector<UserType>&>(vector);
664  acc.swap(mutable_vector);
665  acc.write();
666  acc.swap(mutable_vector);
667  }
