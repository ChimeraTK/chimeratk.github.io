581  template<typename UserType>
582  std::vector<UserType> Device::read(const RegisterPath& registerPathName, size_t numberOfWords,
583  size_t wordOffsetInRegister, const AccessModeFlags& flags) const {
584  auto acc = getOneDRegisterAccessor<UserType>(registerPathName, numberOfWords, wordOffsetInRegister, flags);
585  acc.read();
586  std::vector<UserType> vector(acc.getNElements());
587  acc.swap(vector);
588  return vector;

602  template<typename UserType>
603  void Device::write(const RegisterPath& registerPathName, const std::vector<UserType>& vector,
604  size_t wordOffsetInRegister, const AccessModeFlags& flags) {
605  auto acc = getOneDRegisterAccessor<UserType>(registerPathName, vector.size(), wordOffsetInRegister, flags);
606  // the vector stays constant, but we have to work around so we can swap it and
607  // later swap it back...
608  std::vector<UserType>& mutable_vector = const_cast<std::vector<UserType>&>(vector);
609  acc.swap(mutable_vector);
610  acc.write();
611  acc.swap(mutable_vector);
612  }
