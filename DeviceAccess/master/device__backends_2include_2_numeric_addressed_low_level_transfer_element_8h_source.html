84  bool isMergeable(const boost::shared_ptr<TransferElement const>& other) const {
85  if(!_dev->canMergeRequests()) return false;
86 
87  // accessor type, device and bar must be the same
88  auto rhsCasted = boost::dynamic_pointer_cast<const NumericAddressedLowLevelTransferElement>(other);
89  if(!rhsCasted) return false;
90  if(_dev != rhsCasted->_dev) return false;
91  if(_bar != rhsCasted->_bar) return false;
92 
93  // only allow adjacent and overlapping address areas to be merged
94  if(_startAddress + _numberOfBytes < rhsCasted->_startAddress) return false;
95  if(_startAddress > rhsCasted->_startAddress + rhsCasted->_numberOfBytes) return false;
96  return true;
97  }

139  void setAddress(size_t startAddress, size_t numberOfBytes) {
140  // change address
141  _startAddress = startAddress;
142  _numberOfBytes = numberOfBytes;
143 
144  // make sure access is properly aligned
145  _isUnaligned = false;
146  auto alignment = _dev->minimumTransferAlignment();
147  auto start_padding = _startAddress % alignment;
148  assert(_startAddress >= start_padding);
149  if(start_padding > 0) _isUnaligned = true;
150  _startAddress -= start_padding;
151  _numberOfBytes += start_padding;
152  auto end_padding = alignment - _numberOfBytes % alignment;
153  if(end_padding != alignment) {
154  _isUnaligned = true;
155  _numberOfBytes += end_padding;
156  }
157 
158  // Allocated the buffer
159  rawDataBuffer.resize(_numberOfBytes);
160 
161  // update the name
162  _name = "NALLTE:" + std::to_string(_startAddress) + "+" + std::to_string(_numberOfBytes);
163  }
