303  std::unique_ptr<BackendRegisterCatalogueBase> BackendRegisterCatalogue<BackendRegisterInfo>::clone() const {
304  // FIXME: Change BackendRegisterCatalogue to CRTP, i.e. it has the DERRIVED class as template parameter.
305  // Like this the correct catalogue type is already created here and inherriting backends can call this clone,
306  // then cast to the actual type and fill the rest of it's properties. Would get rid of the "fillFromThis()" workaround and
307  // safe clone() implementations in case no data members are added.
308 
309  // Create a new instance of a BackendRegisterCatalogue with the correct type.
310  // Inheriting backends will create the derrived type here.
311  auto* c = new BackendRegisterCatalogue<BackendRegisterInfo>();
312  // Fill the contents of the BackendRegisterCatalogue base class into the target c. This is accessing the
313  // private variables and ensures consistency.
314  fillFromThis(c);
315  // Derrived backends will copy/clone their additional data members here, before
316  // returning a unique_ptr of the new catalogue which from now on takes the ownership.
317  return std::unique_ptr<BackendRegisterCatalogueBase>(c);

348  template<typename BackendRegisterInfo>
349  void BackendRegisterCatalogue<BackendRegisterInfo>::removeRegister(const RegisterPath& name) {
350  // check existence
351  if(!hasRegister(name)) {
352  throw ChimeraTK::logic_error(
353  "BackendRegisterCatalogue::removeRegister(): Register '" + name + "' does not exist.");
354  }
355 
356  // remove from insertion-ordered vector
357  auto it = std::find_if(insertionOrderedCatalogue.begin(), insertionOrderedCatalogue.end(),
358  [&](auto reg) { return reg->getRegisterName() == name; });
359  assert(it != insertionOrderedCatalogue.end());
360  insertionOrderedCatalogue.erase(it);
361 
362  // remove from catalogue map
363  auto removed = catalogue.erase(name);
364  assert(removed == 1);
365  }
