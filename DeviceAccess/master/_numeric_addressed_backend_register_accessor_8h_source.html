24  template<typename UserType, typename DataConverterType, bool isRaw>
25  class NumericAddressedBackendRegisterAccessor : public NDRegisterAccessor<UserType> {
26  public:
27  NumericAddressedBackendRegisterAccessor(const boost::shared_ptr<DeviceBackend>& dev,
28  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
29  : NDRegisterAccessor<UserType>(registerPathName, flags), _dataConverter(registerPathName),
30  _dev(boost::dynamic_pointer_cast<NumericAddressedBackend>(dev)) {
31  // check for unknown flags
32  flags.checkForUnknownFlags({AccessMode::raw});
33 
34  // check device backend
35  _dev = boost::dynamic_pointer_cast<NumericAddressedBackend>(dev);
36  if(!_dev) {
37  throw ChimeraTK::logic_error("NumericAddressedBackendRegisterAccessor is used with a backend which is not "
38  "a NumericAddressedBackend.");
39  }
40 
41  // obtain register information
42  _registerInfo = _dev->getRegisterInfo(registerPathName);
43  assert(!_registerInfo.channels.empty());
44 
45  if(_registerInfo.elementPitchBits % 8 != 0) {
46  throw ChimeraTK::logic_error("NumericAddressedBackendRegisterAccessor: Elements must be byte aligned.");
47  }
48 
49  if(_registerInfo.channels.size() > 1) {
50  throw ChimeraTK::logic_error("NumericAddressedBackendRegisterAccessor is used with a 2D register.");
51  }
52 
53  if(_registerInfo.channels.front().bitOffset > 0) {
54  throw ChimeraTK::logic_error("NumericAddressedBackendRegisterAccessor: Registers must be byte aligned.");
55  }
56 
57  // check number of words
58  if(_registerInfo.channels.front().dataType == NumericAddressedRegisterInfo::Type::VOID) {
59  // in void registers we always create one element
60  if(numberOfWords == 0) {
61  numberOfWords = 1;
62  }
63  if(numberOfWords > 1) {
64  throw ChimeraTK::logic_error(
65  "Requested number of words is larger than 1 in VOID register '" + registerPathName + "'!");
66  }
67  if(wordOffsetInRegister > 0) {
68  throw ChimeraTK::logic_error("No offset allowed in VOID register '" + registerPathName + "'!");
69  }
70  }
71  else { // do the regular consistency check
72  if(numberOfWords == 0) {
73  numberOfWords = _registerInfo.getNumberOfElements();
74  }
75  if(numberOfWords + wordOffsetInRegister > _registerInfo.getNumberOfElements()) {
76  throw ChimeraTK::logic_error(
77  "Requested number of words exceeds the size of the register '" + registerPathName + "'!");
78  }
79  if(wordOffsetInRegister >= _registerInfo.getNumberOfElements()) {
80  throw ChimeraTK::logic_error("Requested offset exceeds the size of the register'" + registerPathName + "'!");
81  }
82  }
83 
84  // change registerInfo (local copy!) to account for given offset and length override
85  _registerInfo.address += wordOffsetInRegister * _registerInfo.elementPitchBits / 8;
86  _registerInfo.nElements = numberOfWords;
87 
88  // create low-level transfer element handling the actual data transfer to the hardware with raw data
89  assert(_registerInfo.elementPitchBits % 8 == 0);
90  _rawAccessor = boost::make_shared<NumericAddressedLowLevelTransferElement>(
91  _dev, _registerInfo.bar, _registerInfo.address, _registerInfo.nElements * _registerInfo.elementPitchBits / 8);
92 
93  // allocated the buffers
94  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
95  NDRegisterAccessor<UserType>::buffer_2D[0].resize(_registerInfo.nElements);
96 
97  // We don't have to fill it in a special way if the accessor is raw
98  // because we have an overloaded, more efficient implementation
99  // in this case. So we can use it in setAsCooked() and getAsCooked()
100  _dataConverter = detail::createDataConverter<DataConverterType>(_registerInfo);
101 
102  if(flags.has(AccessMode::raw)) {
103  if(DataType(typeid(UserType)) != _registerInfo.getDataDescriptor().rawDataType()) {
104  throw ChimeraTK::logic_error("Given UserType when obtaining the NumericAddressedBackendRegisterAccessor in "
105  "raw mode does not match the expected type. Use an " +
106  _registerInfo.getDataDescriptor().rawDataType().getAsString() +
107  " instead! (Register name: " + registerPathName + "')");
108  }
109  }
110 
111  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(getAsCooked_impl);
112  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(setAsCooked_impl);
113  }
114 
115  void doReadTransferSynchronously() override { _rawAccessor->readTransfer(); }
116 
117  bool doWriteTransfer(ChimeraTK::VersionNumber versionNumber) override {
118  assert(!TransferElement::_isInTransferGroup);
119  _rawAccessor->writeTransfer(versionNumber);
120  return false;
121  }

123  void doPostRead(TransferType type, bool hasNewData) override {
124  if(!_dev->isOpen()) return; // do not delegate if exception was thrown by us in doPreWrite
125 
126  _rawAccessor->setActiveException(this->_activeException);
127  _rawAccessor->postRead(type, hasNewData);
128 
129  if(!hasNewData) return;
130 
131  if constexpr(!isRaw || std::is_same<UserType, std::string>::value) {
132  callForRawType(_registerInfo.getDataDescriptor().rawDataType(), [this](auto t) {
133  typedef decltype(t) RawType;
134  auto itsrc = (RawType*)_rawAccessor->begin(_registerInfo.address);
135  _dataConverter.template vectorToCooked<UserType>(itsrc, itsrc + buffer_2D[0].size(), buffer_2D[0].begin());
136  });
137  }
138  else {
139  // optimised variant for raw transfers (unless type is a string)
140  auto* itsrc = _rawAccessor->begin(_registerInfo.address);
141  auto* itdst = buffer_2D[0].data();
142  memcpy(itdst, itsrc, buffer_2D[0].size() * sizeof(UserType));
143  }
144 
145  // we don't put the setting of the version number into the PrePostActionImplementor
146  // because it does not need template specialisation, and the implementer does not
147  // know about _versionNumber. It's just easier here.
148  this->_versionNumber = _rawAccessor->getVersionNumber();
149  this->_dataValidity = _rawAccessor->dataValidity();
150  }
151 
152  void doPreWrite(TransferType type, VersionNumber versionNumber) override {
153  if(!_dev->isOpen()) throw ChimeraTK::logic_error("Device not opened.");
154  // raw accessor preWrite must be called before our _prePostActionsImplementor.doPreWrite(), as it needs to
155  // prepare the buffer in case of unaligned access and acquire the lock.
156  _rawAccessor->preWrite(type, versionNumber);
157 
158  if constexpr(!isRaw || std::is_same<UserType, std::string>::value) {
159  if(!_registerInfo.isWriteable()) {
160  throw ChimeraTK::logic_error(
161  "NumericAddressedBackend: Writeing to a non-writeable register is not allowed (Register name: " +
162  _registerInfo.getRegisterName() + ").");
163  }
164  callForRawType(_registerInfo.getDataDescriptor().rawDataType(), [this](auto t) {
165  typedef decltype(t) RawType;
166  auto itsrc = (RawType*)_rawAccessor->begin(_registerInfo.address);
167  for(auto itdst = buffer_2D[0].begin(); itdst != buffer_2D[0].end(); ++itdst) {
168  *itsrc = _dataConverter.template toRaw<UserType>(*itdst);
169  ++itsrc;
170  }
171  });
172  }
173  else {
174  // optimised variant for raw transfers (unless type is a string)
175  auto* itdst = _rawAccessor->begin(_registerInfo.address);
176  auto itsrc = buffer_2D[0].begin();
177  memcpy(&(*itdst), &(*itsrc), buffer_2D[0].size() * sizeof(UserType));
178  }
179 
180  _rawAccessor->setDataValidity(this->_dataValidity);
181  }
182 
183  void doPreRead(TransferType type) override {

250  void replaceTransferElement(boost::shared_ptr<TransferElement> newElement) override {
251  auto casted = boost::dynamic_pointer_cast<NumericAddressedLowLevelTransferElement>(newElement);
252  if(casted && casted->isMergeable(_rawAccessor)) {
253  size_t newStartAddress = std::min(casted->_startAddress, _rawAccessor->_startAddress);
254  size_t newStopAddress = std::max(
255  casted->_startAddress + casted->_numberOfBytes, _rawAccessor->_startAddress + _rawAccessor->_numberOfBytes);
256  size_t newNumberOfBytes = newStopAddress - newStartAddress;
257  casted->changeAddress(newStartAddress, newNumberOfBytes);
258  _rawAccessor = casted;
259  }
260  _rawAccessor->setExceptionBackend(this->_exceptionBackend);

314  template<typename COOKED_TYPE>
315  COOKED_TYPE NumericAddressedBackendRegisterAccessor<UserType, DataConverterType, isRaw>::getAsCooked_impl(
316  unsigned int channel, unsigned int sample) {
317  if(isRaw) {
318  std::vector<COOKED_TYPE> cookedData(1);
319  dataConverterTemplateSpecialisationHelper<UserType, COOKED_TYPE>::vectorToCooked(_dataConverter,
320  NDRegisterAccessor<UserType>::buffer_2D[channel].begin() + sample,
321  NDRegisterAccessor<UserType>::buffer_2D[channel].begin() + sample + 1, cookedData.begin());
322  return cookedData[0];
323  }
324  throw ChimeraTK::logic_error("Getting as cooked is only available for raw accessors!");
325  }
