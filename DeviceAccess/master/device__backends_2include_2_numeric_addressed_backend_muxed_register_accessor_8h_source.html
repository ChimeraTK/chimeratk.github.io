141  NumericAddressedBackendMuxedRegisterAccessor<UserType>::NumericAddressedBackendMuxedRegisterAccessor(
142  const RegisterPath& registerPathName, size_t numberOfElements, size_t elementsOffset,
143  boost::shared_ptr<DeviceBackend> _backend)
144  : NDRegisterAccessor<UserType>(registerPathName, {}),
145  _ioDevice(boost::dynamic_pointer_cast<NumericAddressedBackend>(_backend)), _registerPathName(registerPathName),
146  _numberOfElements(numberOfElements), _elementsOffset(elementsOffset) {
147  // re-split register and module after merging names by the last dot (to
148  // allow module.register in the register name)
149  _registerPathName.setAltSeparator(".");
150  auto moduleAndRegister = MapFileParser::splitStringAtLastDot(_registerPathName.getWithAltSeparator());
151  _moduleName = moduleAndRegister.first;
152  _registerName = moduleAndRegister.second;
153 
154  // build name of area as written in the map file
155  std::string areaName = MULTIPLEXED_SEQUENCE_PREFIX + _registerName;
156 
157  // Obtain information about the area
158  auto registerMapping = _ioDevice->getRegisterMap();
159  registerMapping->getRegisterInfo(areaName, _areaInfo, _moduleName);
160 
161  // Cache writeability
162  _isNotWriteable = !_areaInfo.isWriteable();
163 
164  // Obtain information for each sequence (= channel) in the area:
165  // Create a fixed point converter for each sequence and store the sequence
166  // information in a vector
167  size_t iSeq = 0;
168  while(true) {
169  // build name of the next sequence as written in the map file
170  RegisterInfoMap::RegisterInfo sequenceInfo;
171  std::stringstream sequenceNameStream;
172  sequenceNameStream << SEQUENCE_PREFIX << _registerName << "_" << iSeq++;
173 
174  // try to find sequence
175  try {
176  registerMapping->getRegisterInfo(sequenceNameStream.str(), sequenceInfo, _moduleName);
177  }
178  catch(ChimeraTK::logic_error&) {
179  // no sequence found: we are done
180  break;
181  }
182 
183  // check consistency
184  if(sequenceInfo.nElements != 1) {
185  throw ChimeraTK::logic_error("Sequence words must have exactly one element");
186  }
187 
188  // store sequence info and fixed point converter
189  if(sequenceInfo.width > sequenceInfo.nBytes * 8) sequenceInfo.width = sequenceInfo.nBytes * 8;
190  _sequenceInfos.push_back(sequenceInfo);
191  _converters.push_back(FixedPointConverter(
192  registerPathName, sequenceInfo.width, sequenceInfo.nFractionalBits, sequenceInfo.signedFlag));
193  }
194 
195  // check if no sequences were found
196  if(_converters.empty()) {
197  throw ChimeraTK::logic_error("No sequences found for name \"" + _registerName + "\".");
198  }
199 
200  // compute size of one block in bytes (one sample for all channels)
201  bytesPerBlock = 0;
202  for(unsigned int i = 0; i < _converters.size(); i++) {
203  uint32_t nbt = _sequenceInfos[i].nBytes;
204  bytesPerBlock += nbt;
205  if(nbt != 1 && nbt != 2 && nbt != 4) {
206  throw ChimeraTK::logic_error("Sequence word size must correspond to a primitive type");
207  }
208  }
209 
210  // compute number of blocks (number of samples for each channel)
211  _nBlocks = std::floor(_areaInfo.nBytes / bytesPerBlock);
212 
213  // check number of words
214  if(_elementsOffset >= _nBlocks) {
215  throw ChimeraTK::logic_error("Requested element offset exceeds the size of the register!");
216  }
217  if(_numberOfElements == 0) {
218  _numberOfElements = _nBlocks;
219  }
220  if(_numberOfElements + _elementsOffset > _nBlocks) {
221  throw ChimeraTK::logic_error("Requested number of elements exceeds the size of the register! Requested end: " +
222  std::to_string(_numberOfElements + _elementsOffset) + ", register length: " + std::to_string(_nBlocks));
223  }
224 
225  // compute the address taking into account the selected area of interest
226  _bar = _areaInfo.bar;
227  _address = _areaInfo.address + bytesPerBlock * _elementsOffset;
228  _nBytes = bytesPerBlock * _numberOfElements;
229  if(_nBytes % sizeof(int32_t) > 0) _nBytes += 4 - _nBytes % sizeof(int32_t); // round up to the next multiple of 4
230  if(_nBytes > _areaInfo.nBytes) {
231  throw ChimeraTK::logic_error(
232  "Requested number of elements exceeds the size of the register (late, redundant safety check)!");
233  }
234  _nBlocks = _numberOfElements;
235 
236  // allocate the buffer for the converted data
237  NDRegisterAccessor<UserType>::buffer_2D.resize(_converters.size());
238  for(size_t i = 0; i < _converters.size(); ++i) {
239  NDRegisterAccessor<UserType>::buffer_2D[i].resize(_nBlocks);
240  }
241 
242  // allocate the raw io buffer. Make it one element larger to make sure we can access the last byte via int32_t*
243  _ioBuffer.resize(_nBytes / sizeof(int32_t) + 1);
244 
245  // compute pitched iterators for accessing the channels
246  uint8_t* standOfMyioBuffer = reinterpret_cast<uint8_t*>(&_ioBuffer[0]);
247  for(size_t sequenceIndex = 0; sequenceIndex < _converters.size(); ++sequenceIndex) {
248  _startIterators.push_back(detail::pitched_iterator<int32_t>(standOfMyioBuffer, bytesPerBlock));
249  _endIterators.push_back(_startIterators.back() + _nBlocks);
250  standOfMyioBuffer += _sequenceInfos[sequenceIndex].nBytes;
251  }
252  }

287  template<class UserType>
288  void NumericAddressedBackendMuxedRegisterAccessor<UserType>::doPreWrite(TransferType, VersionNumber) {
289  if(!_ioDevice->isOpen()) throw ChimeraTK::logic_error("Device not opened.");
290  uint8_t* standOfMyioBuffer = reinterpret_cast<uint8_t*>(&_ioBuffer[0]);
291  for(size_t blockIndex = 0; blockIndex < _nBlocks; ++blockIndex) {
292  for(size_t sequenceIndex = 0; sequenceIndex < _converters.size(); ++sequenceIndex) {
293  switch(_sequenceInfos[sequenceIndex].nBytes) {
294  case 1: // 8 bit variables
295  *(standOfMyioBuffer) =
296  _converters[sequenceIndex].toRaw(NDRegisterAccessor<UserType>::buffer_2D[sequenceIndex][blockIndex]);
297  standOfMyioBuffer++;
298  break;
299  case 2: // 16 bit variables
300  *(reinterpret_cast<uint16_t*>(standOfMyioBuffer)) =
301  _converters[sequenceIndex].toRaw(NDRegisterAccessor<UserType>::buffer_2D[sequenceIndex][blockIndex]);
302  standOfMyioBuffer = standOfMyioBuffer + 2;
303  break;
304  case 4: // 32 bit variables
305  *(reinterpret_cast<uint32_t*>(standOfMyioBuffer)) =
306  _converters[sequenceIndex].toRaw(NDRegisterAccessor<UserType>::buffer_2D[sequenceIndex][blockIndex]);
307  standOfMyioBuffer = standOfMyioBuffer + 4;
308  break;
309  }
310  }
311  }
