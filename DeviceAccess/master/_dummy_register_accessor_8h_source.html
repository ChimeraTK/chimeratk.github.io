216  template<typename T>
217  class DummyMultiplexedRegisterAccessor {
218  public:
225  DummyMultiplexedRegisterAccessor(DummyBackend* dev, std::string module, std::string name)
226  : _dev(dev), _path(module + "/" + name), pitch(0) {
227  registerInfo = _dev->_registerMap.getBackendRegister(module + "." + name);
228 
229  // create fixed point converters for each channel
230  for(auto& c : registerInfo.channels) {
231  // create fixed point converter for sequence
232  fpc.emplace_back(registerInfo.pathName, c.width, c.nFractionalBits, c.signedFlag);
233  // store offsets and number of bytes per word
234  assert(c.bitOffset % 8 == 0);
235  offsets.push_back(registerInfo.address + c.bitOffset / 8);
236  nbytes.push_back((c.width - 1) / 8 + 1); // width/8 rounded up
237  }
238 
239  if(fpc.empty()) {
240  throw ChimeraTK::logic_error("No sequences found for name \"" + name + "\".");
241  }
242 
243  // cache some information
244  nElements = registerInfo.nElements;
245  assert(registerInfo.elementPitchBits % 8 == 0);
246  pitch = registerInfo.elementPitchBits / 8;
247  }
248 
249  // declare that we want the default copy constructor. Needed because we have a custom = operator
250  DummyMultiplexedRegisterAccessor(const DummyMultiplexedRegisterAccessor&) = default;
251 
253  unsigned int getNumberOfElements() { return nElements; }
254 
256  unsigned int getNumberOfSequences() { return fpc.size(); }
257 
263  inline proxies::DummyRegisterSequence<T> operator[](unsigned int sequence) {
264  int8_t* basePtr = reinterpret_cast<int8_t*>(_dev->_barContents[registerInfo.bar].data());
265  int32_t* seq = reinterpret_cast<int32_t*>(basePtr + offsets[sequence]);
266  return proxies::DummyRegisterSequence<T>(&(fpc[sequence]), nbytes[sequence], pitch, seq);
267  }
268 
270  DummyBackend& getBackend() const { return *_dev; }
271 
273  const RegisterPath& getRegisterPath() const { return _path; }
274 
275  const NumericAddressedRegisterInfo& getRegisterInfo() { return registerInfo; }
276

280

283
