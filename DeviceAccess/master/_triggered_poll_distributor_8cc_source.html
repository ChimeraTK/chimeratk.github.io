19  void TriggeredPollDistributor::trigger(VersionNumber version) {
20  if(!_asyncDomain->unsafeGetIsActive()) {
21  return;
22  }
23 
24  _version = version;
25  try {
26  _transferGroup->read();
27 
28  for(auto& var : _asyncVariables) {
29  var.second->fillSendBuffer();
30  var.second->send(); // send function from the AsyncVariable base class
31  }
32  }
33  catch(ChimeraTK::runtime_error&) {
34  // Nothing to do. Backend's set exception has already been called by the accessor in the transfer group that
35  // raised it.
36  }
37  }
38

41  trigger(version);
