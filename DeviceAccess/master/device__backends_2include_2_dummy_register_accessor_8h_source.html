232  template<typename T>
233  class DummyMultiplexedRegisterAccessor : public DummyRegisterAddressChecker {
234  public:
241  DummyMultiplexedRegisterAccessor(DummyBackend* dev, std::string module, std::string name)
242  : _dev(dev), _path(module + "/" + name), pitch(0) {
243  _dev->_registerMapping->getRegisterInfo(MULTIPLEXED_SEQUENCE_PREFIX + name, registerInfo, module);
244 
245  int i = 0;
246  while(true) {
247  // obtain register information for sequence
248  RegisterInfoMap::RegisterInfo elem;
249  std::stringstream sequenceNameStream;
250  sequenceNameStream << SEQUENCE_PREFIX << name << "_" << i++;
251  try {
252  _dev->_registerMapping->getRegisterInfo(sequenceNameStream.str(), elem, module);
253  }
254  catch(ChimeraTK::logic_error&) {
255  break;
256  }
257  // create fixed point converter for sequence
258  fpc.push_back(FixedPointConverter(module + "/" + name, elem.width, elem.nFractionalBits, elem.signedFlag));
259  // store offsets and number of bytes per word
260  offsets.push_back(elem.address);
261  nbytes.push_back(elem.nBytes);
262  // determine pitch
263  pitch += elem.nBytes;
264  }
265 
266  if(fpc.empty()) {
267  throw ChimeraTK::logic_error("No sequenes found for name \"" + name + "\".");
268  }
269 
270  // compute number of elements per sequence
271  nElements = registerInfo.nBytes / pitch;
272  }
273 
274  // declare that we want the default copy constructor. Needed because we have a custom = operator
275  DummyMultiplexedRegisterAccessor(const DummyMultiplexedRegisterAccessor&) = default;
276 
278  unsigned int getNumberOfElements() { return nElements; }
279 
281  unsigned int getNumberOfSequences() { return fpc.size(); }
282 
288  inline proxies::DummyRegisterSequence<T> operator[](unsigned int sequence) {
289  int8_t* basePtr = reinterpret_cast<int8_t*>(_dev->_barContents[registerInfo.bar].data());
290  int32_t* seq = reinterpret_cast<int32_t*>(basePtr + offsets[sequence]);
291  return proxies::DummyRegisterSequence<T>(&(fpc[sequence]), nbytes[sequence], pitch, seq);
292  }
293 
295  DummyBackend& getBackend() const { return *_dev; }
296 
298  const RegisterPath& getRegisterPath() const { return _path; }
299 
300  protected:
302  DummyBackend* _dev;
303 
305  RegisterPath _path;
306 
308  std::vector<FixedPointConverter> fpc;
309 
311  std::vector<int> offsets;
312 
314  std::vector<int> nbytes;
315 
317  int pitch;
318 
320  unsigned int nElements;
321 
322  private:
324  void operator=(const DummyMultiplexedRegisterAccessor& rightHandSide) const;
325  };
