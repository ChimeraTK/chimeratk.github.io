231  template<typename T>
232  class DummyMultiplexedRegisterAccessor : public DummyRegisterAddressChecker {
233  public:
240  DummyMultiplexedRegisterAccessor(DummyBackend* dev, std::string module, std::string name)
241  : _dev(dev), _path(module + "/" + name), pitch(0) {
242  _dev->_registerMapping->getRegisterInfo(MULTIPLEXED_SEQUENCE_PREFIX + name, registerInfo, module);
243 
244  int i = 0;
245  while(true) {
246  // obtain register information for sequence
247  RegisterInfoMap::RegisterInfo elem;
248  std::stringstream sequenceNameStream;
249  sequenceNameStream << SEQUENCE_PREFIX << name << "_" << i++;
250  try {
251  _dev->_registerMapping->getRegisterInfo(sequenceNameStream.str(), elem, module);
252  }
253  catch(ChimeraTK::logic_error&) {
254  break;
255  }
256  // create fixed point converter for sequence
257  fpc.push_back(FixedPointConverter(module + "/" + name, elem.width, elem.nFractionalBits, elem.signedFlag));
258  // store offsets and number of bytes per word
259  offsets.push_back(elem.address);
260  nbytes.push_back(elem.nBytes);
261  // determine pitch
262  pitch += elem.nBytes;
263  }
264 
265  if(fpc.empty()) {
266  throw ChimeraTK::logic_error("No sequenes found for name \"" + name + "\".");
267  }
268 
269  // compute number of elements per sequence
270  nElements = registerInfo.nBytes / pitch;
271  }
272 
273  // declare that we want the default copy constructor. Needed because we have a custom = operator
274  DummyMultiplexedRegisterAccessor(const DummyMultiplexedRegisterAccessor&) = default;
275 
277  unsigned int getNumberOfElements() { return nElements; }
278 
280  unsigned int getNumberOfSequences() { return fpc.size(); }
281 
287  inline proxies::DummyRegisterSequence<T> operator[](unsigned int sequence) {
288  int8_t* basePtr = reinterpret_cast<int8_t*>(_dev->_barContents[registerInfo.bar].data());
289  int32_t* seq = reinterpret_cast<int32_t*>(basePtr + offsets[sequence]);
290  return proxies::DummyRegisterSequence<T>(&(fpc[sequence]), nbytes[sequence], pitch, seq);
291  }
292 
294  DummyBackend& getBackend() const { return *_dev; }
295 
297  const RegisterPath& getRegisterPath() const { return _path; }
298 
299  protected:
301  DummyBackend* _dev;
302 
304  RegisterPath _path;
305 
307  std::vector<FixedPointConverter> fpc;
308 
310  std::vector<int> offsets;
311 
313  std::vector<int> nbytes;
314 
316  int pitch;
317 
319  unsigned int nElements;
320 
321  private:
323  void operator=(const DummyMultiplexedRegisterAccessor& rightHandSide) const;
324  };
