40  void LogicalNameMappingBackend::open() {
41  if(isFunctional()) return;
42  parse();
43 
44  // open all referenced devices (unconditionally, open() is also used for recovery)
45  for(auto device = _devices.begin(); device != _devices.end(); ++device) {
46  device->second->open();
47  }
48 
49  // flag as opened
50  _opened = true;
51  _hasException = false;
52  _asyncReadActive = false;
53 
54  // make sure to update the catalogue from target devices in case they change their catalogue upon open
55  catalogueCompleted = false;
56 
57  // call the openHook for all plugins
58  for(auto& reg : _catalogue_mutable) {
59  for(auto& plug : dynamic_cast<LNMBackendRegisterInfo&>(reg).plugins) {
60  plug->openHook(boost::dynamic_pointer_cast<LogicalNameMappingBackend>(shared_from_this()));
61  }
62  }
63  }

101  template<typename UserType>
102  boost::shared_ptr<NDRegisterAccessor<UserType>> LogicalNameMappingBackend::getRegisterAccessor_impl(
103  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags,
104  size_t omitPlugins) {
105  parse();
106  // check if accessor plugin present
107  boost::shared_ptr<NDRegisterAccessor<UserType>> returnValue;
108  auto info = boost::static_pointer_cast<LNMBackendRegisterInfo>(_catalogue_mutable.getRegister(registerPathName));
109  if(info->plugins.size() <= omitPlugins) {
110  // no plugin: directly return the accessor
111  returnValue =
112  getRegisterAccessor_internal<UserType>(registerPathName, numberOfWords, wordOffsetInRegister, flags);
113  }
114  else {
115  returnValue = info->plugins[omitPlugins]->getAccessor<UserType>(
116  boost::static_pointer_cast<LogicalNameMappingBackend>(shared_from_this()), numberOfWords,
117  wordOffsetInRegister, flags, omitPlugins);
118  }
119 
120  returnValue->setExceptionBackend(shared_from_this());
121 
122  return returnValue;
123  }

127  template<typename UserType>
128  boost::shared_ptr<NDRegisterAccessor<UserType>> LogicalNameMappingBackend::getRegisterAccessor_internal(
129  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
130  // obtain register info
131  auto info = boost::static_pointer_cast<LNMBackendRegisterInfo>(_catalogue_mutable.getRegister(registerPathName));
132 
133  // Check that the requested requested accessor fits into the register as described by the info. It is not enough to let
134  // the target do the check. It might be a sub-register of a much larger one and for the target it is fine.
135  if(info->length != 0) {
136  // If info->length is 0 we let the target device do the checking. Nothing we can decide here.
137  if(numberOfWords == 0) numberOfWords = info->length;
138  if((numberOfWords + wordOffsetInRegister) > info->length) {
139  throw ChimeraTK::logic_error(
140  std::string(
141  "LogicalNameMappingBackend: Error creating accessor. Number of words plus offset too large in ") +
142  registerPathName);
143  }
144  }
145 
146  // determine the offset and length
147  size_t actualOffset = size_t(info->firstIndex) + wordOffsetInRegister;
148  size_t actualLength = (numberOfWords > 0 ? numberOfWords : size_t(info->length));
149 
150  // implementation for each type
151  boost::shared_ptr<NDRegisterAccessor<UserType>> ptr;
152  if(info->targetType == LNMBackendRegisterInfo::TargetType::REGISTER) {
153  DeviceBackend* _targetDevice;
154  std::string devName = info->deviceName;
155  if(devName != "this") {
156  _targetDevice = _devices[info->deviceName].get();
157  }
158  else {
159  _targetDevice = this;
160  }
161  // make sure the target device exists
162  if(_targetDevice == nullptr) {
163  throw ChimeraTK::logic_error("Target device for this logical register is not opened. See "
164  "exception thrown in open()!");
165  }
166  // obtain underlying register accessor
167  ptr = _targetDevice->getRegisterAccessor<UserType>(
168  RegisterPath(info->registerName), actualLength, actualOffset, flags);
169  }
170  else if(info->targetType == LNMBackendRegisterInfo::TargetType::CHANNEL) {
171  ptr = boost::shared_ptr<NDRegisterAccessor<UserType>>(new LNMBackendChannelAccessor<UserType>(
172  shared_from_this(), registerPathName, actualLength, actualOffset, flags));
173  }
174  else if(info->targetType == LNMBackendRegisterInfo::TargetType::BIT) {
175  ptr = boost::shared_ptr<NDRegisterAccessor<UserType>>(
176  new LNMBackendBitAccessor<UserType>(shared_from_this(), registerPathName, actualLength, actualOffset, flags));
177  }
178  else if(info->targetType == LNMBackendRegisterInfo::TargetType::CONSTANT ||
179  info->targetType == LNMBackendRegisterInfo::TargetType::VARIABLE) {
180  ptr = boost::shared_ptr<NDRegisterAccessor<UserType>>(new LNMBackendVariableAccessor<UserType>(
181  shared_from_this(), registerPathName, actualLength, actualOffset, flags));
182  }
183  else {
184  throw ChimeraTK::logic_error("For this register type, a RegisterAccessor cannot be obtained (name "
185  "of logical register: " +
186  registerPathName + ").");
187  }
188 
189  return ptr;
190  }

194  const RegisterCatalogue& LogicalNameMappingBackend::getRegisterCatalogue() const {
195  if(catalogueCompleted) return _catalogue_mutable;
196  parse();
197 
198  // fill in information to the catalogue from the target devices
199  for(auto& info : _catalogue_mutable) {
200  LNMBackendRegisterInfo& info_cast = static_cast<LNMBackendRegisterInfo&>(info);
201  auto targetType = info_cast.targetType;
202  if(targetType != LNMBackendRegisterInfo::TargetType::REGISTER &&
203  targetType != LNMBackendRegisterInfo::TargetType::CHANNEL &&
204  targetType != LNMBackendRegisterInfo::TargetType::BIT)
205  continue;
206 
207  std::string devName = info_cast.deviceName;
208  boost::shared_ptr<RegisterInfo> target_info;
209  if(devName != "this") {
210  auto& cat = _devices.at(devName)->getRegisterCatalogue();
211  if(!cat.hasRegister(info_cast.registerName)) continue;
212  target_info = cat.getRegister(info_cast.registerName);
213  }
214  else {
215  target_info = _catalogue_mutable.getRegister(std::string(info_cast.registerName));
216  }
217 
218  if(targetType != LNMBackendRegisterInfo::TargetType::BIT) {
219  info_cast._dataDescriptor = target_info->getDataDescriptor();
220  }
221  else {
222  info_cast._dataDescriptor =
223  RegisterInfo::DataDescriptor(RegisterInfo::FundamentalType::boolean, true, false, 1, 0);
224  }
225  info_cast.readable = target_info->isReadable();
226  info_cast.writeable = target_info->isWriteable();
227  info_cast.supportedFlags = target_info->getSupportedAccessModes();
228 
229  if(targetType == LNMBackendRegisterInfo::TargetType::CHANNEL) {
230  info_cast.writeable = false;
231  }
232 
233  if(targetType == LNMBackendRegisterInfo::TargetType::REGISTER) {
234  info_cast.nDimensions = target_info->getNumberOfDimensions();
235  info_cast.nChannels = target_info->getNumberOfChannels();
236  }
237  if((int)info_cast.length == 0) info_cast.length = target_info->getNumberOfElements();
238  }
239 
240  // update catalogue info by plugins
241  for(auto& info : _catalogue_mutable) {
242  LNMBackendRegisterInfo& info_cast = static_cast<LNMBackendRegisterInfo&>(info);
243  for(auto& plugin : info_cast.plugins) {
244  plugin->updateRegisterInfo();
245  }
246  }
247 
248  catalogueCompleted = true;
249  return _catalogue_mutable;
250  }

288  void LogicalNameMappingBackend::setException() {
289  _hasException = true;
290  for(auto& d : _devices) {
291  d.second->setException();
292  }
293 
294  // iterate all push subscriptions of variables and place exception into queue
295  if(_asyncReadActive) {
296  VersionNumber v{};
297  for(auto& r : _catalogue_mutable) {
298  auto& info = dynamic_cast<LNMBackendRegisterInfo&>(r);
299  if(info.targetType == LNMBackendRegisterInfo::TargetType::VARIABLE) {
300  callForType(info.valueType, [&](auto arg) {
301  auto& vtEntry = boost::fusion::at_key<decltype(arg)>(info.valueTable.table);
302  for(auto& sub : vtEntry.subscriptions) {
303  try {
304  throw ChimeraTK::runtime_error("previous, unrecovered fault");
305  }
306  catch(...) {
307  sub.second.push_overwrite_exception(std::current_exception());
308  }
309  }
310  });
311  }
312  }
313  }
314 
315  // deactivate async read
316  _asyncReadActive = false;
317 
318  // call the exceptionHook for all plugins
319  for(auto& reg : _catalogue_mutable) {
320  for(auto& plug : dynamic_cast<LNMBackendRegisterInfo&>(reg).plugins) {
321  plug->exceptionHook();
322  }
323  }
324  }

326  /********************************************************************************************************************/
327  void LogicalNameMappingBackend::activateAsyncRead() noexcept {
328  if(!_opened || _hasException) return;
329 
330  // store information locally, as variable accessors have async read
331  _asyncReadActive = true;
332 
333  // delegate to target devices
334  for(auto& d : _devices) {
335  d.second->activateAsyncRead();
336  }
337 
338  // iterate all push subscriptions of variables and place initial value into queue
339  VersionNumber v{};
340  for(auto& r : _catalogue_mutable) {
341  auto& info = dynamic_cast<LNMBackendRegisterInfo&>(r);
342  if(info.targetType == LNMBackendRegisterInfo::TargetType::VARIABLE) {
343  callForType(info.valueType, [&](auto arg) {
344  auto& vtEntry = boost::fusion::at_key<decltype(arg)>(info.valueTable.table);
345  for(auto& sub : vtEntry.subscriptions) {
346  sub.second.push_overwrite({vtEntry.latestValue, vtEntry.latestValidity, v});
347  }
348  });
349  }
350  }
351  }
