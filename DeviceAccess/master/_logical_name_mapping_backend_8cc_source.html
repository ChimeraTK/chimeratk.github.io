23  void LogicalNameMappingBackend::parse() const {
24  // don't run, if already parsed
25  if(hasParsed) return;
26  hasParsed = true;
27 
28  // parse the map fle
29  LogicalNameMapParser parser = LogicalNameMapParser(_lmapFileName, _parameters);
30  _catalogue_mutable = parser.getCatalogue();
31 
32  // create all devices referenced in the map
33  for(auto& devName : parser.getTargetDevices()) {
34  _devices[devName] = BackendFactory::getInstance().createBackend(devName);
35  }
36  }

40  void LogicalNameMappingBackend::open() {
41  parse();
42 
43  // open all referenced devices (unconditionally, open() is also used for recovery)

46  }

50  _hasException = false;
51 
52  // make sure to update the catalogue from target devices in case they change their catalogue upon open
53  catalogueCompleted = false;

82  template<typename UserType>
83  boost::shared_ptr<NDRegisterAccessor<UserType>> LogicalNameMappingBackend::getRegisterAccessor_impl(
84  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags,
85  size_t omitPlugins) {
86  parse();
87  // check if accessor plugin present
88  boost::shared_ptr<NDRegisterAccessor<UserType>> returnValue;
89  auto info = boost::static_pointer_cast<LNMBackendRegisterInfo>(_catalogue_mutable.getRegister(registerPathName));
90  if(info->plugins.size() <= omitPlugins) {
91  // no plugin: directly return the accessor
92  returnValue =
93  getRegisterAccessor_internal<UserType>(registerPathName, numberOfWords, wordOffsetInRegister, flags);
94  }
95  else {
96  returnValue = info->plugins[omitPlugins]->getAccessor<UserType>(
97  boost::static_pointer_cast<LogicalNameMappingBackend>(shared_from_this()), numberOfWords,
98  wordOffsetInRegister, flags, omitPlugins);
99  }
100 
101  returnValue->setExceptionBackend(shared_from_this());
102 
103  return returnValue;
104  }

108  template<typename UserType>
109  boost::shared_ptr<NDRegisterAccessor<UserType>> LogicalNameMappingBackend::getRegisterAccessor_internal(
110  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
111  // obtain register info
112  auto info = boost::static_pointer_cast<LNMBackendRegisterInfo>(_catalogue_mutable.getRegister(registerPathName));
113 
114  // Check that the requested requested accessor fits into the register as described by the info. It is not enough to let
115  // the target do the check. It might be a sub-register of a much larger one and for the target it is fine.
116  if(info->length != 0) {
117  // If info->length is 0 we let the target device do the checking. Nothing we can decide here.
118  if(numberOfWords == 0) numberOfWords = info->length;
119  if((numberOfWords + wordOffsetInRegister) > info->length) {
120  throw ChimeraTK::logic_error(
121  std::string(
122  "LogicalNameMappingBackend: Error creating accessor. Number of words plus offset too large in ") +
123  registerPathName);
124  }
125  }
126 
127  // implementation for each type
128  boost::shared_ptr<NDRegisterAccessor<UserType>> ptr;
129  if(info->targetType == LNMBackendRegisterInfo::TargetType::REGISTER) {
130  DeviceBackend* _targetDevice;
131  std::string devName = info->deviceName;
132  if(devName != "this") {
133  _targetDevice = _devices[info->deviceName].get();
134  }
135  else {
136  _targetDevice = this;
137  }
138  // make sure the target device exists
139  if(_targetDevice == nullptr) {
140  throw ChimeraTK::logic_error("Target device for this logical register is not opened. See "
141  "exception thrown in open()!");
142  }
143  // determine the offset and length
144  size_t actualOffset = size_t(info->firstIndex) + wordOffsetInRegister;
145  size_t actualLength = (numberOfWords > 0 ? numberOfWords : size_t(info->length));
146  // obtain underlying register accessor
147  ptr = _targetDevice->getRegisterAccessor<UserType>(
148  RegisterPath(info->registerName), actualLength, actualOffset, flags);
149  }
150  else if(info->targetType == LNMBackendRegisterInfo::TargetType::CHANNEL) {
151  ptr = boost::shared_ptr<NDRegisterAccessor<UserType>>(new LNMBackendChannelAccessor<UserType>(
152  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
153  }
154  else if(info->targetType == LNMBackendRegisterInfo::TargetType::BIT) {
155  ptr = boost::shared_ptr<NDRegisterAccessor<UserType>>(new LNMBackendBitAccessor<UserType>(
156  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
157  }
158  else if(info->targetType == LNMBackendRegisterInfo::TargetType::CONSTANT ||
159  info->targetType == LNMBackendRegisterInfo::TargetType::VARIABLE) {
160  ptr = boost::shared_ptr<NDRegisterAccessor<UserType>>(new LNMBackendVariableAccessor<UserType>(
161  shared_from_this(), registerPathName, numberOfWords, wordOffsetInRegister, flags));
162  }
163  else {
164  throw ChimeraTK::logic_error("For this register type, a RegisterAccessor cannot be obtained (name "
165  "of logical register: " +
166  registerPathName + ").");
167  }
168 
169  return ptr;
170  }

174  const RegisterCatalogue& LogicalNameMappingBackend::getRegisterCatalogue() const {
175  if(catalogueCompleted) return _catalogue_mutable;
176  parse();
177 
178  // fill in information to the catalogue from the target devices
179  for(auto& info : _catalogue_mutable) {
180  LNMBackendRegisterInfo& info_cast = static_cast<LNMBackendRegisterInfo&>(info);
181  auto targetType = info_cast.targetType;
182  if(targetType != LNMBackendRegisterInfo::TargetType::REGISTER &&
183  targetType != LNMBackendRegisterInfo::TargetType::CHANNEL &&
184  targetType != LNMBackendRegisterInfo::TargetType::BIT)
185  continue;
186 
187  std::string devName = info_cast.deviceName;
188  boost::shared_ptr<RegisterInfo> target_info;
189  if(devName != "this") {
190  auto& cat = _devices.at(devName)->getRegisterCatalogue();
191  if(!cat.hasRegister(info_cast.registerName)) continue;
192  target_info = cat.getRegister(info_cast.registerName);
193  }
194  else {
195  target_info = _catalogue_mutable.getRegister(std::string(info_cast.registerName));
196  }
197 
198  if(targetType != LNMBackendRegisterInfo::TargetType::BIT) {
199  info_cast._dataDescriptor = target_info->getDataDescriptor();
200  }
201  else {
202  info_cast._dataDescriptor =
203  RegisterInfo::DataDescriptor(RegisterInfo::FundamentalType::boolean, true, false, 1, 0);
204  }
205  info_cast.readable = target_info->isReadable();
206  info_cast.writeable = target_info->isWriteable();
207  info_cast.supportedFlags = target_info->getSupportedAccessModes();
208 
209  if(targetType == LNMBackendRegisterInfo::TargetType::CHANNEL) {
210  info_cast.writeable = false;
211  }
212 
213  if(targetType == LNMBackendRegisterInfo::TargetType::REGISTER) {
214  info_cast.nDimensions = target_info->getNumberOfDimensions();
215  info_cast.nChannels = target_info->getNumberOfChannels();
216  }
217  if((int)info_cast.length == 0) info_cast.length = target_info->getNumberOfElements();
218  }
219 
220  // update catalogue info by plugins
221  for(auto& info : _catalogue_mutable) {
222  LNMBackendRegisterInfo& info_cast = static_cast<LNMBackendRegisterInfo&>(info);
223  for(auto& plugin : info_cast.plugins) {
224  plugin->updateRegisterInfo();
225  }
226  }
227 
228  catalogueCompleted = true;
229  return _catalogue_mutable;
230  }

254  return false;
255  }
