53  void PcieBackend::determineDriverAndConfigureIoctl() {
54  // determine the driver by trying the physical slot ioctl
55  device_ioctrl_data ioctlData = {0, 0, 0, 0};
56 
57  if(ioctl(_deviceID, PCIEDEV_PHYSICAL_SLOT, &ioctlData) >= 0) {
58  // it's the pciedev driver
59  _ioctlPhysicalSlot = PCIEDEV_PHYSICAL_SLOT;
60  _ioctlDriverVersion = PCIEDEV_DRIVER_VERSION;
61  _ioctlDMA = PCIEDEV_READ_DMA;
62  _readDMAFunction = [&](uint8_t bar, uint32_t address, int32_t* data, size_t sizeInBytes) {
63  PcieBackend::readDMAViaIoctl(bar, address, data, sizeInBytes);
64  };
65  _writeFunction = [&](uint8_t bar, uint32_t address, int32_t const* data, size_t sizeInBytes) {
66  writeWithStruct(bar, address, data, sizeInBytes);
67  };
68  _readFunction = [&](uint8_t bar, uint32_t address, int32_t* data, size_t sizeInBytes) {
69  readWithStruct(bar, address, data, sizeInBytes);
70  };
71  return;
72  }
73 
74  if(ioctl(_deviceID, LLRFDRV_PHYSICAL_SLOT, &ioctlData) >= 0) {
75  // it's the llrf driver
76  _ioctlPhysicalSlot = LLRFDRV_PHYSICAL_SLOT;
77  _ioctlDriverVersion = LLRFDRV_DRIVER_VERSION;
78  _ioctlDMA = 0;
79  _readDMAFunction = [&](uint8_t bar, uint32_t address, int32_t* data, size_t sizeInBytes) {
80  PcieBackend::readDMAViaStruct(bar, address, data, sizeInBytes);
81  };
82  _writeFunction = [&](uint8_t bar, uint32_t address, int32_t const* data, size_t sizeInBytes) {
83  writeWithStruct(bar, address, data, sizeInBytes);
84  };
85  _readFunction = [&](uint8_t bar, uint32_t address, int32_t* data, size_t sizeInBytes) {
86  readWithStruct(bar, address, data, sizeInBytes);
87  };
88  return;
89  }
90 
91  if(ioctl(_deviceID, PCIEUNI_PHYSICAL_SLOT, &ioctlData) >= 0) {
92  // it's the pcieuni
93  _ioctlPhysicalSlot = PCIEUNI_PHYSICAL_SLOT;
94  _ioctlDriverVersion = PCIEUNI_DRIVER_VERSION;
95  _ioctlDMA = PCIEUNI_READ_DMA;
96  _readDMAFunction = [&](uint8_t bar, uint32_t address, int32_t* data, size_t sizeInBytes) {
97  PcieBackend::readDMAViaIoctl(bar, address, data, sizeInBytes);
98  };
99  _writeFunction = [&](uint8_t bar, uint32_t address, int32_t const* data, size_t sizeInBytes) {
100  directWrite(bar, address, data, sizeInBytes);
101  };
102  _readFunction = [&](uint8_t bar, uint32_t address, int32_t* data, size_t sizeInBytes) {
103  directRead(bar, address, data, sizeInBytes);
104  };
105  return;
106  }
107 
108  // No working driver. Close the device and throw an exception.
109  std::cerr << "Unsupported driver. " << createErrorStringWithErrnoText("Error is ") << std::endl;
110  ::close(_deviceID);
111  throw ChimeraTK::runtime_error("Unsupported driver in device" + _deviceNodeName);
112  }

121  bool PcieBackend::isFunctional() const {
122  if(!_opened) return false;
123  if(_hasActiveException) return false;
124 
125  // Note: This expects byte 0 of bar 0 to be readable. This is currently guaranteed by our firmware framework. If
126  // other firmware needs to be supported, this should be made configurable (via CDD). If a map file is used, we could
127  // also use the first readable address specified in the map file.
128 
129  // read word 0 from bar 0 to check if device works
130  device_rw l_RW;
131  l_RW.barx_rw = 0;
132  l_RW.mode_rw = RW_D8;
133  l_RW.offset_rw = 0;
134  l_RW.size_rw = 0; // does not overwrite the struct but writes one word back to data
135  l_RW.data_rw = -1;
136  l_RW.rsrvd_rw = 0;
137  return ::read(_deviceID, &l_RW, sizeof(device_rw)) == sizeof(device_rw);
138  }

140  void PcieBackend::readInternal(uint8_t bar, uint32_t address, int32_t* data) {
141  device_rw l_RW;
142  assert(_opened);
143  l_RW.barx_rw = bar;
144  l_RW.mode_rw = RW_D32;
145  l_RW.offset_rw = address;
146  l_RW.size_rw = 0; // does not overwrite the struct but writes one word back to data
147  l_RW.data_rw = -1;
148  l_RW.rsrvd_rw = 0;
149 
150  if(::read(_deviceID, &l_RW, sizeof(device_rw)) != sizeof(device_rw)) {
151  throw ChimeraTK::runtime_error(createErrorStringWithErrnoText("Cannot read data from device: "));
152  }
153  *data = static_cast<int32_t>(l_RW.data_rw);
154  }

156  void PcieBackend::directRead(uint8_t bar, uint32_t address, int32_t* data, size_t sizeInBytes) {
157  assert(_opened);
158  assert(bar <= 5);
159  loff_t virtualOffset = PCIEUNI_BAR_OFFSETS[bar] + address;
160 
161  if(pread(_deviceID, data, sizeInBytes, virtualOffset) != static_cast<int>(sizeInBytes)) {
162  throw ChimeraTK::runtime_error(createErrorStringWithErrnoText("Cannot read data from device: "));
163  }
164  }

200  void PcieBackend::read(uint8_t bar, uint32_t address, int32_t* data, size_t sizeInBytes) {
201  if(_hasActiveException) {
202  throw ChimeraTK::runtime_error("previous, unrecovered fault");
203  }
204  if(bar != 0xD) {
205  _readFunction(bar, address, data, sizeInBytes);

227  void PcieBackend::readDMAViaStruct(uint8_t /*bar*/, uint32_t address, int32_t* data, size_t sizeInBytes) {
228  ssize_t ret;
229  device_rw l_RW;
230  device_rw* pl_RW;
231 
232  assert(_opened);
233 
234  if(sizeInBytes < sizeof(device_rw)) {
235  pl_RW = &l_RW;
236  }
237  else {
238  // We are reaching down into the C-interface for the ioctl. There is not we can do about reinterpret-casting into
239  // the correct type. NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
240  pl_RW = reinterpret_cast<device_rw*>(data);
241  }
242 
243  pl_RW->data_rw = 0;
244  pl_RW->barx_rw = 0;
245  pl_RW->size_rw = sizeInBytes;
246  pl_RW->mode_rw = RW_DMA;
247  pl_RW->offset_rw = address;
248  pl_RW->rsrvd_rw = 0;
249 
250  ret = ::read(_deviceID, pl_RW, sizeof(device_rw));
251  if(ret != (ssize_t)sizeInBytes) {
252  throw ChimeraTK::runtime_error(createErrorStringWithErrnoText("Cannot read data from device: "));
253  }
254  if(sizeInBytes < sizeof(device_rw)) {
255  memcpy(data, pl_RW, sizeInBytes);
256  }
257  }

259  void PcieBackend::readDMAViaIoctl(uint8_t /*bar*/, uint32_t address, int32_t* data, size_t sizeInBytes) {
260  assert(_opened);
261 
262  // prepare the struct
263  device_ioctrl_dma DMA_RW;
264  DMA_RW.dma_cmd = 0; // FIXME: Why is it 0? => read driver code
265  DMA_RW.dma_pattern = 0; // FIXME: Why is it 0? => read driver code
266  DMA_RW.dma_size = sizeInBytes;
267  DMA_RW.dma_offset = address;
268  DMA_RW.dma_reserved1 = 0; // FIXME: is this a correct value?
269  DMA_RW.dma_reserved2 = 0; // FIXME: is this a correct value?
270 
271  if(sizeInBytes >= sizeof(device_ioctrl_dma)) {
272  // the ioctrl_dma struct is copied to the beginning of the data buffer,
273  // so the information about size and offset are passed to the driver.
274  memcpy((void*)data, &DMA_RW, sizeof(device_ioctrl_dma));
275  int ret = ioctl(_deviceID, _ioctlDMA, (void*)data);
276  if(ret) {
277  throw ChimeraTK::runtime_error(createErrorStringWithErrnoText("Cannot read data from device "));
278  }
279  }
280  else {
281  // the ioctrl_dma struct is used as a dma buffer and the read data is later copied out
282  int ret = ioctl(_deviceID, _ioctlDMA, (void*)&DMA_RW);
283  if(ret) {
284  throw ChimeraTK::runtime_error(createErrorStringWithErrnoText("Cannot read data from device "));
285  }
286  memcpy(&DMA_RW, (void*)data, sizeInBytes);
287  }
288  }

290  std::string PcieBackend::readDeviceInfo() {
291  if(!_opened) throw ChimeraTK::logic_error("Device not opened.");
292  std::ostringstream os;
293  device_ioctrl_data ioctlData = {0, 0, 0, 0};
294  if(ioctl(_deviceID, _ioctlPhysicalSlot, &ioctlData) < 0) {
295  throw ChimeraTK::runtime_error(createErrorStringWithErrnoText("Cannot read device info: "));
296  }
297  os << "SLOT: " << ioctlData.data;
298  if(ioctl(_deviceID, _ioctlDriverVersion, &ioctlData) < 0) {
299  throw ChimeraTK::runtime_error(createErrorStringWithErrnoText("Cannot read device info: "));
300  }
301  // major and minor version are in data and offset, respectively
302  os << " DRV VER: " << ioctlData.data << "." << ioctlData.offset;
303  return os.str();
304  }
