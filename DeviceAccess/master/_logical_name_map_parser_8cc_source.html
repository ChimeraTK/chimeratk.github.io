18  template<>
19  std::string LogicalNameMapParser::getValueFromXmlSubnode<std::string>(
20  const xmlpp::Node* node, const std::string& subnodeName, bool hasDefault, std::string defaultValue) {
21  auto list = node->find(subnodeName);
22  if(list.size() < 1 && hasDefault) return defaultValue;
23  if(list.size() != 1) {
24  parsingError(node,
25  "Expected exactly one subnode of the type '" + subnodeName + "' below node '" + node->get_name() + "'.");
26  }
27  auto childList = list[0]->get_children();
28 
29  std::string value;
30  for(auto& child : childList) {
31  // check for plain text
32  const xmlpp::TextNode* textNode = dynamic_cast<xmlpp::TextNode*>(child);
33  if(textNode) {
34  // put to stream buffer
35  value += textNode->get_content();
36  continue;
37  }
38 
39  // check for reference
40  if(child->get_name() == "ref") {
41  auto childChildList = child->get_children();
42  const xmlpp::TextNode* refNameNode = dynamic_cast<xmlpp::TextNode*>(childChildList.front());
43  if(refNameNode && childChildList.size() == 1) {
44  std::string regName = refNameNode->get_content();
45  if(!_catalogue.hasRegister(regName)) {
46  parsingError(child, "Reference to constant '" + regName + "' could not be resolved.");
47  }
48  auto reg = _catalogue.getRegister(regName);
49  auto reg_casted = boost::dynamic_pointer_cast<LNMBackendRegisterInfo>(reg);
50  assert(reg_casted != nullptr); // this is our own catalogue
51  // fetch the value of the target constant
52  if(reg_casted->targetType == LNMBackendRegisterInfo::TargetType::CONSTANT) {
53  if(reg_casted->plugins.size() > 0) {
54  parsingError(childList.front(), "'" + regName + "' uses plugins which is not supported for <ref>");
55  }
56  // put to stream buffer
57  callForType(reg_casted->valueType, [&](auto arg) {
58  std::stringstream buf;
59  buf << boost::fusion::at_key<decltype(arg)>(reg_casted->valueTable.table)[0];
60  value = buf.str();
61  });
62  continue;
63  }
64  else {
65  parsingError(child, "Reference to '" + regName + "' does not refer to a constant.");
66  }
67  }
68  else {
69  parsingError(child, "The <ref> node must contain only text.");
70  }
71  }
72 
73  // check for parameter
74  if(child->get_name() == "par") {
75  auto childChildList = child->get_children();
76  const xmlpp::TextNode* parNameNode = dynamic_cast<xmlpp::TextNode*>(childChildList.front());
77  if(parNameNode && childChildList.size() == 1) {
78  std::string parName = parNameNode->get_content();
79  if(_parameters.find(parName) == _parameters.end()) {
80  parsingError(child, "Parameter '" + parName + "' could not be resolved.");
81  }
82  // put to stream buffer
83  value += _parameters[parName];
84  continue;
85  }
86  else {
87  parsingError(child, "The <par> node must contain only text.");
88  }
89  }
90 
91  // neither found: throw error
92  parsingError(node,
93  "Node '" + subnodeName + "' should contain only text, references or parameters. Instead child '" +
94  child->get_name() + "' was found.");
95  }
96  return value;
97  }

103  const xmlpp::Node* node, const std::string& subnodeName, bool hasDefault, T defaultValue) {
104  // obtain result as string an put into stream

107

116  template<typename T>
117  std::vector<T> LogicalNameMapParser::getValueVectorFromXmlSubnode(
118  const xmlpp::Node* node, const std::string& subnodeName) {
119  auto list = node->find(subnodeName);
120  if(list.size() < 1) {
121  parsingError(node,
122  "Expected at least one subnode of the type '" + subnodeName + "' below node '" + node->get_name() + "'.");
123  }
124 
125  std::vector<T> valueVector;
126 
127  for(auto& child : list) {
128  const xmlpp::Element* childElement = dynamic_cast<const xmlpp::Element*>(child);
129  if(!childElement) continue; // ignore comments etc.
130 
131  // obtain index and resize valueVector if necessary
132  auto indexAttr = childElement->get_attribute("index");
133  size_t index = 0;
134  if(indexAttr) {
135  index = std::stoi(indexAttr->get_value());
136  }
137  if(index >= valueVector.size()) valueVector.resize(index + 1);
138 
139  // get content
140  auto childList = childElement->get_children();
141  if(childList.size() != 1) {
142  parsingError(childElement,
143  "Node '" + subnodeName +
144  "' should contain only text or exactly one reference, "
145  "instead multiple childs "
146  "were found.");
147  }
148 
149  // check for plain text
150  const xmlpp::TextNode* textNode = dynamic_cast<xmlpp::TextNode*>(childList.front());
151  if(textNode) {
152  std::stringstream buf(textNode->get_content());
153  buf >> valueVector[index];
154  continue;
155  }
156 
157  // check for reference
158  if(childList.front()->get_name() == "ref") {
159  auto childChildList = childList.front()->get_children();
160  const xmlpp::TextNode* refNameNode = dynamic_cast<xmlpp::TextNode*>(childChildList.front());
161  if(refNameNode && childChildList.size() == 1) {
162  std::string regName = refNameNode->get_content();
163  if(!_catalogue.hasRegister(regName)) {
164  parsingError(childList.front(), "Reference to constant '" + regName + "' could not be resolved.");
165  }
166  auto reg = _catalogue.getRegister(regName);
167  auto reg_casted = boost::dynamic_pointer_cast<LNMBackendRegisterInfo>(reg);
168  assert(reg_casted != nullptr); // this is our own catalogue
169  // fetch the value of the target constant
170  if(reg_casted->targetType == LNMBackendRegisterInfo::TargetType::CONSTANT) {
171  if(reg_casted->plugins.size() > 0) {
172  parsingError(childList.front(), "'" + regName + "' uses plugins which is not supported for <ref>");
173  }
174  // convert via string
175  std::stringstream buf;
176  callForType(reg_casted->valueType,
177  [&](auto arg) { buf << boost::fusion::at_key<decltype(arg)>(reg_casted->valueTable.table)[0]; });
178  buf >> valueVector[index];
179  continue;
180  }
181  else {
182  parsingError(childList.front(), "Reference to '" + regName + "' does not refer to a constant.");
183  }
184  }
185  else {
186  parsingError(childList.front(), "The <ref> node must contain only text.");
187  }
188  }
189 
190  if(childList.front()->get_name() == "par") {
191  auto childChildList = childList.front()->get_children();
192  const xmlpp::TextNode* parNameNode = dynamic_cast<xmlpp::TextNode*>(childChildList.front());
193  if(parNameNode && childChildList.size() == 1) {
194  std::string parName = parNameNode->get_content();
195  if(_parameters.find(parName) == _parameters.end()) {
196  parsingError(childList.front(), "Parameter '" + parName + "' could not be resolved.");
197  }
198  // put to stream buffer
199  std::stringstream buf;
200  buf << _parameters[parName];
201  buf >> valueVector[index];
202  continue;
203  }
204  else {
205  parsingError(childList.front(), "The <par> node must contain only text.");
206  }
207  }
208 
209  // neither found: throw error
210  parsingError(node,
211  "Node '" + subnodeName + "' should contain only text or exactly one reference, instead child '" +
212  childList.front()->get_name() + "' was found.");
213  }
214 
215  return valueVector;
216  }

220  void LogicalNameMapParser::parseFile(const std::string& fileName) {
221  _fileName = fileName;
222 
223  // parse the file into a DOM structure
224  xmlpp::DomParser parser;
225  try {
226  parser.parse_file(fileName);
227  }
228  catch(xmlpp::exception& e) {
229  throw ChimeraTK::logic_error("Error opening the xlmap file '" + fileName + "': " + e.what());
230  }
231 
232  // get root element
233  const auto root = parser.get_document()->get_root_node();
234  if(root->get_name() != "logicalNameMap") {
235  parsingError(root, "Expected 'logicalNameMap' tag instead of: " + root->get_name());
236  }
237 
238  // parsing loop
239  for(const auto& child : root->get_children()) {
240  // cast into element, ignore if not an element (e.g. comment)
241  const xmlpp::Element* element = dynamic_cast<const xmlpp::Element*>(child);
242  if(!element) continue;
243 
244  // parse the element
245  parseElement(RegisterPath(), element);
246  }

249  void LogicalNameMapParser::parseElement(RegisterPath currentPath, const xmlpp::Element* element) {
250  // module tag found: look for registers and sub-modules in module
251  if(element->get_name() == "module") {
252  // obtain name of the module
253  auto nameAttr = element->get_attribute("name");
254  if(!nameAttr) {
255  parsingError(element, "Missing name attribute of 'module' tag.");
256  }
257  std::string moduleName = nameAttr->get_value();
258 
259  // iterate over childs in module
260  for(const auto& child : element->get_children()) {
261  // cast into element, ignore if not an element (e.g. comment)
262  const xmlpp::Element* childElement = dynamic_cast<const xmlpp::Element*>(child);
263  if(!childElement) continue;
264 
265  // parse the element
266  parseElement(currentPath / moduleName, childElement);
267  }
268  }
269 
270  // register tag found: create new entry in map
271  else {
272  // obtain the type
273  std::string type = element->get_name();
274 
275  // create new RegisterInfo object
276  auto info = boost::shared_ptr<LNMBackendRegisterInfo>(new LNMBackendRegisterInfo());
277 
278  if(type == "redirectedRegister") {
279  info->targetType = LNMBackendRegisterInfo::TargetType::REGISTER;
280  info->deviceName = getValueFromXmlSubnode<std::string>(element, "targetDevice");
281  info->registerName = getValueFromXmlSubnode<std::string>(element, "targetRegister");
282  info->firstIndex = getValueFromXmlSubnode<unsigned int>(element, "targetStartIndex", true, 0);
283  info->length = getValueFromXmlSubnode<unsigned int>(element, "numberOfElements", true, 0);
284  info->nDimensions = 0;
285  info->nChannels = 0;
286  }
287  else if(type == "redirectedChannel") {
288  info->targetType = LNMBackendRegisterInfo::TargetType::CHANNEL;
289  info->deviceName = getValueFromXmlSubnode<std::string>(element, "targetDevice");
290  info->registerName = getValueFromXmlSubnode<std::string>(element, "targetRegister");
291  info->channel = getValueFromXmlSubnode<unsigned int>(element, "targetChannel");
292  info->firstIndex = 0;
293  info->length = 0;
294  info->nDimensions = 1;
295  info->nChannels = 1;
296  }
297  else if(type == "redirectedBit") {
298  info->targetType = LNMBackendRegisterInfo::TargetType::BIT;
299  info->deviceName = getValueFromXmlSubnode<std::string>(element, "targetDevice");
300  info->registerName = getValueFromXmlSubnode<std::string>(element, "targetRegister");
301  info->bit = getValueFromXmlSubnode<unsigned int>(element, "targetBit");
302  info->firstIndex = 0;
303  info->length = 0;
304  info->nDimensions = 1;
305  info->nChannels = 1;
306  }
307  else if(type == "constant") {
308  std::string constantType = getValueFromXmlSubnode<std::string>(element, "type");
309  if(constantType == "integer") constantType = "int32";
310  info->targetType = LNMBackendRegisterInfo::TargetType::CONSTANT;
311  info->valueType = DataType(constantType);
312  callForType(info->valueType, [&](auto arg) {
313  boost::fusion::at_key<decltype(arg)>(info->valueTable.table) =
314  this->getValueVectorFromXmlSubnode<decltype(arg)>(element, "value");
315  });
316  info->firstIndex = 0;
317  info->length = getValueFromXmlSubnode<unsigned int>(element, "numberOfElements", true, 1);
318  info->nDimensions = info->length > 1 ? 1 : 0;
319  info->nChannels = 1;
320  info->writeable = false;
321  info->readable = true;
322  info->_dataDescriptor = ChimeraTK::RegisterInfo::DataDescriptor(info->valueType);
323  }
324  else if(type == "variable") {
325  std::string constantType = getValueFromXmlSubnode<std::string>(element, "type");
326  if(constantType == "integer") constantType = "int32";
327  info->targetType = LNMBackendRegisterInfo::TargetType::VARIABLE;
328  info->valueType = DataType(constantType);
329  callForType(info->valueType, [&](auto arg) {
330  boost::fusion::at_key<decltype(arg)>(info->valueTable.table) =
331  this->getValueVectorFromXmlSubnode<decltype(arg)>(element, "value");
332  });
333  info->firstIndex = 0;
334  info->length = getValueFromXmlSubnode<unsigned int>(element, "numberOfElements", true, 1);
335  info->nDimensions = info->length > 1 ? 1 : 0;
336  info->nChannels = 1;
337  info->writeable = true;
338  info->readable = true;
339  info->_dataDescriptor = ChimeraTK::RegisterInfo::DataDescriptor(info->valueType);
340  }
341  else {
342  parsingError(element, "Wrong logical register type: " + type);
343  }
344 
345  // obtain name of logical register
346  auto nameAttr = element->get_attribute("name");
347  if(!nameAttr) {
348  parsingError(element, "Missing name attribute of '" + type + "' tag.");
349  }
350  RegisterPath registerName = currentPath / std::string(nameAttr->get_value());
351  info->name = registerName;
352 
353  // iterate over childs of the register to find plugins
354  for(const auto& child : element->get_children()) {
355  // cast into element, ignore if not an element (e.g. comment)
356  const xmlpp::Element* childElement = dynamic_cast<const xmlpp::Element*>(child);
357  if(!childElement) continue;
358  if(childElement->get_name() != "plugin") continue; // look only for plugins
359 
360  // get name of plugin
361  auto pluginNameAttr = childElement->get_attribute("name");
362  if(!pluginNameAttr) {
363  parsingError(childElement, "Missing name attribute of 'plugin' tag.");
364  }
365  std::string pluginName = pluginNameAttr->get_value();
366 
367  // collect parameters
368  std::map<std::string, std::string> parameters;
369  for(const auto& paramchild : childElement->get_children()) {
370  // cast into element, ignore if not an element (e.g. comment)
371  const xmlpp::Element* paramElement = dynamic_cast<const xmlpp::Element*>(paramchild);
372  if(!paramElement) continue;
373  if(paramElement->get_name() != "parameter") {
374  parsingError(paramElement, "Unexpected element '" + paramElement->get_name() + "' inside plugin tag.");
375  }
376 
377  // get name of parameter
378  auto parameterNameAttr = paramElement->get_attribute("name");
379  if(!pluginNameAttr) {
380  parsingError(paramElement, "Missing name attribute of 'parameter' tag.");
381  }
382  std::string parameterName = parameterNameAttr->get_value();
383 
384  // get value of parameter and store in map
385  parameters[parameterName] =
386  getValueFromXmlSubnode<std::string>(childElement, "parameter[@name='" + parameterName + "']");
387  }
388 
389  // create instance of plugin and add to the list in the register info
390  info->plugins.push_back(LNMBackend::makePlugin(info, pluginName, parameters));
391  }
392 
393  // add register to catalogue
394  _catalogue.addRegister(info);
395  }
396  }

400  std::unordered_set<std::string> LogicalNameMapParser::getTargetDevices() const {
401  std::unordered_set<std::string> ret;
402  for(auto it = _catalogue.begin(); it != _catalogue.end(); ++it) {
403  auto info = boost::static_pointer_cast<const LNMBackendRegisterInfo>(it.get());
