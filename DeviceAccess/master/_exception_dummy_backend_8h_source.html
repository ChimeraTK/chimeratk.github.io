249  void ExceptionDummy::triggerPush(RegisterPath path, VersionNumber v) {
250  path.setAltSeparator(".");
251  std::unique_lock<std::mutex> lk(_pushDecoratorsMutex);
252  _pushVersions[path] = v;
253  for(auto& acc_weak : _pushDecorators[path]) {

266  for(auto& pair : _pushDecorators) {
267  _pushVersions[pair.first] = {};

272  acc->trigger(); // initial value
