60  void write(uint64_t bar, uint64_t address, int32_t const* data, size_t sizeInBytes) override {
61  if(throwExceptionWrite) {
62  thereHaveBeenExceptions = true;
63  throw(ChimeraTK::runtime_error("DummyException: write throws by request"));
64  }
65  ChimeraTK::DummyBackend::write(bar, address, data, sizeInBytes);
66 
67  // increment write counter and update write order (only if address points to beginning of a register!)
68  auto itWriteOrder = _writeOrderMap.find(std::make_pair(bar, address));
69  if(itWriteOrder != _writeOrderMap.end()) {
70  // update write order
71  auto generatedOrderNumber = ++_writeOrderCounter;
72  auto& orderNumberInMap = itWriteOrder->second;
73  // atomically update order number in the map only if the generated order number is bigger. This will be always
74  // the case, unless there is a concurrent write operation updating the order number in between.
75  size_t current;
76  while((current = orderNumberInMap.load()) < generatedOrderNumber) {
77  orderNumberInMap.compare_exchange_weak(current, generatedOrderNumber);
78  }
79 
80  // increment write counter
81  auto itWriteCounter = _writeCounterMap.find(std::make_pair(bar, address));
82  assert(itWriteCounter != _writeCounterMap.end()); // always inserted together
83  itWriteCounter->second++;
84  }
85  }

87  bool isFunctional() const override {
88  // thereHaveBeenExceptions is different from _hasActiceException
89  // * thereHaveBeenExceptions is set when this class originally raised an exception
90  // * _hasActiveException is raised externally via setException. This can happen if a transfer element from another
91  // backend,
92  // which is in the same logical name mapping backend than this class, has seen an exception.
93  return (_opened && !throwExceptionOpen && !thereHaveBeenExceptions && !_hasActiveException);

97  template<typename UserType>
98  boost::shared_ptr<NDRegisterAccessor<UserType>> getRegisterAccessor_impl(const RegisterPath& registerPathName,
99  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
100  auto path = registerPathName;
101  path.setAltSeparator(".");
102  auto pathComponents = path.getComponents();
103  bool pushRead = false;
104  if(pathComponents[pathComponents.size() - 1] == "PUSH_READ") {
105  if(flags.has(AccessMode::wait_for_new_data)) {
106  pushRead = true;
107  flags.remove(AccessMode::wait_for_new_data);
108  }
109  path--; // remove last component
110  }
111 
112  auto acc = DummyBackendBase::getRegisterAccessor_impl<UserType>(path, numberOfWords, wordOffsetInRegister, flags);
113  if(pushRead) {
114  std::unique_lock<std::mutex> lk(_pushDecoratorsMutex);
115 
116  auto decorator = boost::make_shared<ExceptionDummyPushDecorator<UserType>>(
117  acc, boost::dynamic_pointer_cast<ExceptionDummy>(this->shared_from_this()));
118 
119  _pushDecorators[registerPathName].push_back(decorator);
120 
121  if(_activateNewPushAccessors) {
122  decorator->_isActive = true;
123  decorator->trigger(); // initial value
124  }
125 
126  acc = decorator;
127  }
128  else {
129  // decorate all poll-type variable so returned validity of the data can be controlled
130  auto decorator = boost::make_shared<ExceptionDummyPollDecorator<UserType>>(
131  acc, boost::dynamic_pointer_cast<ExceptionDummy>(this->shared_from_this()));
132  acc = decorator;
133  }
134 
135  // create entry in _writeOrderMap and _writeCounterMap if necessary
136  if(pathComponents[pathComponents.size() - 1] != "DUMMY_WRITEABLE" &&
137  (pathComponents[0].find("DUMMY_INTERRUPT_") != 0)) {
138  auto info = getRegisterInfo(path);
139  auto adrPair = std::make_pair(info.bar, info.address);
140  if(_writeOrderMap.find(adrPair) == _writeOrderMap.end()) {
141  _writeOrderMap[adrPair] = 0;
142  _writeCounterMap[adrPair] = 0;
143  }
144  }
145 
146  return acc;
147  }

242  : NDRegisterAccessorDecorator<UserType>(target), _backend(backend) {

244 
245  this->_accessModeFlags = _target->getAccessModeFlags();
246  this->_accessModeFlags.add(AccessMode::wait_for_new_data);

256  ~ExceptionDummyPushDecorator() override {
257  std::unique_lock<std::mutex> lk(_backend->_pushDecoratorsMutex);
258  auto& list = _backend->_pushDecorators.at(_path);
259  for(auto it = list.begin(); it != list.end(); ++it) {
260  if(it->lock().get() == nullptr) { // weak_ptr is already not lockable any more
261  _backend->_pushDecorators.at(_path).erase(it);
262  return;
263  }
264  }

366  void ExceptionDummy::triggerPush(RegisterPath path, VersionNumber v) {
367  path.setAltSeparator(".");
368  std::unique_lock<std::mutex> lk(_pushDecoratorsMutex);
369  _pushVersions[path] = v;
370  for(auto& acc_weak : _pushDecorators[path]) {
371  auto acc = acc_weak.lock();
372  if(!acc->_isActive) continue;
373  lk.unlock(); // next line might call setException()...
374  acc->trigger();
375  lk.lock();
376  }
377  }

381  void ExceptionDummy::activateAsyncRead() noexcept {
382  DummyBackend::activateAsyncRead();
383 
384  std::unique_lock<std::mutex> lk(_pushDecoratorsMutex);
385  for(auto& pair : _pushDecorators) {
386  _pushVersions[pair.first] = {};
387  for(auto& acc_weak : pair.second) {
388  auto acc = acc_weak.lock();
389  if(acc->_isActive) continue;
390  lk.unlock(); // next line might call setException()...
391  acc->trigger(); // initial value
392  lk.lock();
393  acc->_isActive = true;
394  acc->_hasException = false;
395  }
396  }
397  _activateNewPushAccessors = true;
398  }

402  void ExceptionDummy::setException() {
403  DummyBackend::setException();
404 
405  // deactivate async transfers
406  std::unique_lock<std::mutex> lk(_pushDecoratorsMutex);
407  for(auto& pair : _pushDecorators) {
408  _pushVersions[pair.first] = {};
409  for(auto& acc_weak : pair.second) {
410  auto acc = acc_weak.lock();
411  if(!acc->_isActive) continue;
412  acc->_isActive = false;
413  if(acc->_hasException) continue;
414  acc->_hasException = true;
415  lk.unlock(); // next line might call setException()...
416  acc->trigger();
417  lk.lock();
418  }
419  }
420  _activateNewPushAccessors = false;
421  }
