217  std::pair<std::string, std::string> MapFileParser::splitStringAtLastDot(std::string moduleDotName) {
218  size_t lastDotPosition = moduleDotName.rfind('.');
219 
220  // some special case handlings to avoid string::split from throwing exceptions
221  if(lastDotPosition == std::string::npos) {
222  // no dot found, the whole string is the second argument
223  return std::make_pair(std::string(), moduleDotName);
224  }
225 
226  if(lastDotPosition == 0) {
227  // the first character is a dot, everything from pos 1 is the second
228  // argument
229  if(moduleDotName.size() == 1) {
230  // it's just a dot, return two empty strings
231  return std::make_pair(std::string(), std::string());
232  }
233 
234  // split after the first character
235  return std::make_pair(std::string(), moduleDotName.substr(1));
236  }
237 
238  if(lastDotPosition == (moduleDotName.size() - 1)) {
239  // the last character is a dot. The second argument is empty
240  return std::make_pair(moduleDotName.substr(0, lastDotPosition), std::string());
241  }
242 
243  return std::make_pair(moduleDotName.substr(0, lastDotPosition), moduleDotName.substr(lastDotPosition + 1));
244  }
