249  size_t lastDotPosition = moduleDotName.rfind('.');
250 
251  // some special case handlings to avoid string::split from throwing exceptions
252  if(lastDotPosition == std::string::npos) {
253  // no dot found, the whole string is the second argument
254  return std::make_pair(std::string(), moduleDotName);
255  }
256 
257  if(lastDotPosition == 0) {
258  // the first character is a dot, everything from pos 1 is the second
259  // argument
260  if(moduleDotName.size() == 1) {
261  // it's just a dot, return two empty strings
262  return std::make_pair(std::string(), std::string());
263  }
264 
265  // split after the first character
266  return std::make_pair(std::string(), moduleDotName.substr(1));
267  }
268 
269  if(lastDotPosition == (moduleDotName.size() - 1)) {
270  // the last character is a dot. The second argument is empty
271  return std::make_pair(moduleDotName.substr(0, lastDotPosition), std::string());
272  }
273 
274  return std::make_pair(moduleDotName.substr(0, lastDotPosition), moduleDotName.substr(lastDotPosition + 1));
