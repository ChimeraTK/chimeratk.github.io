80  void UioBackend::startInterruptHandlingThread(uint32_t interruptNumber) {
81  if(interruptNumber != 0) {
82  throw ChimeraTK::logic_error("UIO: Backend only uses interrupt number 0");
83  }
84 
85  if(_launchThreadMutex.try_lock()) {
86  /* Mutex was found to be unlocked: Mutex is now locked so that handling thread will
87  be execute once and run until the destructor is called. */
88  _interruptWaitingThread = std::thread(&UioBackend::waitForInterruptLoop, this);
89  }
90  }

101  void UioBackend::waitForInterruptLoop() {
102  uint32_t numberOfInterrupts;
103 
104  _uioAccess->clearInterrupts();
105 
106  while(!_stopInterruptLoop) {
107  try {
108  numberOfInterrupts = _uioAccess->waitForInterrupt(100);
109  }
110  catch(ChimeraTK::runtime_error& ex) {
111  setException();
112  continue;
113  }
114 
115  if(numberOfInterrupts > 0) {
116  _uioAccess->clearInterrupts();
117 
118  if(_hasActiveException) {
119  // Don't dispatch interrupts in case of exceptions
120  continue;
121  }
122 
123  for(uint32_t i = 0; i < numberOfInterrupts; i++) {
124  dispatchInterrupt(0);
125  }
126  }
127  }
