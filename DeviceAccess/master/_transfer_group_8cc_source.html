21  void TransferGroup::runPostReads(
22  std::set<boost::shared_ptr<TransferElement>>& elements, std::exception_ptr firstDetectedRuntimeError) noexcept {
23  for(auto& elem : elements) {
24  // check for exceptions on any of the element's low level elements
25  for(auto& lowLevelElem : elem->getHardwareAccessingElements()) {
26  // In case there are mupliple exceptions we take the last one, but this does not matter. They are all
27  // ChimeraTK::runtime_errors and the first detected runtime error, which is re-thrown, has already been
28  // determined by previously.
29  if(lowLevelElem->_activeException) {
30  // copy the runtime error from low level element into the high level element so it is processed in post-read
31  elem->_activeException = lowLevelElem->_activeException;
32  }
33  }
34 
35  try {
36  // call with updateDataBuffer = false if there has been any ChimeraTK::runtime_error in the transfer phase,
37  // true otherwise
38  elem->postReadAndHandleExceptions(TransferType::read, firstDetectedRuntimeError == nullptr);
39  }
40  catch(ChimeraTK::runtime_error&) {
41  ++_nRuntimeErrors;
42  }
43  }
44  }

46  void TransferGroup::read() {
47  // reset exception flags
48  for(auto& it : _lowLevelElementsAndExceptionFlags) {
49  it.second = false;
50  }
51 
52  // check pre-conditions so preRead() does not throw logic errors
53  for(auto& backend : _exceptionBackends) {
54  if(backend && !backend->isOpen()) {
55  throw ChimeraTK::logic_error("DeviceBackend " + backend->readDeviceInfo() + "is not opened!");
56  }
57  }
58  if(!isReadable()) {
59  throw ChimeraTK::logic_error("TransferGroup::read() called, but not all elements are readable.");
60  }
61 
62  std::exception_ptr firstDetectedRuntimeError{nullptr};
63 
64  for(auto& elem : _highLevelElements) {
65  elem->preReadAndHandleExceptions(TransferType::read);
66  if((elem->_activeException != nullptr) && (firstDetectedRuntimeError == nullptr)) {
67  firstDetectedRuntimeError = elem->_activeException;
68  }
69  }
70 
71  for(auto& elem : _copyDecorators) {
72  elem->preReadAndHandleExceptions(TransferType::read);
73  if((elem->_activeException != nullptr) && (firstDetectedRuntimeError == nullptr)) {
74  firstDetectedRuntimeError = elem->_activeException;
75  }
76  }
77 
78  if(firstDetectedRuntimeError == nullptr) {
79  // only execute the transfers if there has been no exception yet
80  for(auto& it : _lowLevelElementsAndExceptionFlags) {
81  auto& elem = it.first;
82  elem->handleTransferException([&] { elem->readTransfer(); });
83  if((elem->_activeException != nullptr) && (firstDetectedRuntimeError == nullptr)) {
84  firstDetectedRuntimeError = elem->_activeException;
85  }
86  }
87  }
88 
89  // Exceptions from copy decorators are ignored. The same exception will be thrown by their target accessors in the
90  // second call to runPostReads() anyway. In case the target is decorated with an ExceptionHandlingDecorator (in
91  // ApplicationCore), the copy decorators exception would not be handled by an ExceptionHandlingDecorator and hence
92  // terminate the application.
93  _nRuntimeErrors = 0;
94  runPostReads(_copyDecorators, firstDetectedRuntimeError);
95  _nRuntimeErrors = 0;
96  runPostReads(_highLevelElements, firstDetectedRuntimeError);
97 
98  // re-throw exceptions in the order of occurence
99 
100  // The runtime error which was seen as _activeException in the pre or transfer phase might
101  // have been handled in postRead, and thus become invalid (for instance because it
102  // is suppressed by the ApplicationCore::ExceptionHandlingDecorator).
103  // Only re-throw here if an exception has been re-thrown in the postRead step.
104  // FIXME(?): If the firstDetectedRuntimeError has been handled, but another runtime_error
105  // has been thrown, this logic will throw the wrong exception here. It could be
106  // prevented with a complicated logic that loops all lowLevelElements in the order
107  // that was used to execute the transfer, for each of them it loops all its associated high
108  // level element, and for each high level element it loops all the low level elements again
109  // because some might and others might not have exceptions, and if there was an exception for
110  // that high level element on any of its low level elements, postRead must be called with
111  // an exception set each time. (I gues this sentense in close to not understandabel, that's
112  // why I am not trying to implement it).
113  // In practice this will not happen because either all elements will have an ExceptionHandlingDecorator, or none.
114 
115  if(_nRuntimeErrors != 0) {
116  assert(firstDetectedRuntimeError !=
117  nullptr); // postRead must only rethrow, so there must be a detected runtime_error
118  _cachedReadableWriteableIsValid = false;
119  std::rethrow_exception(firstDetectedRuntimeError);
120  }

125  void TransferGroup::write(VersionNumber versionNumber) {
126  // check pre-conditions so preRead() does not throw logic errors
127  for(auto& backend : _exceptionBackends) {
128  if(backend && !backend->isOpen()) {
129  throw ChimeraTK::logic_error("DeviceBackend " + backend->readDeviceInfo() + "is not opened!");
130  }
131  }
132  if(!isWriteable()) {
133  throw ChimeraTK::logic_error("TransferGroup::write() called, but not all elements are writeable.");
134  }
135 
136  for(auto& it : _lowLevelElementsAndExceptionFlags) {
137  it.second = false;
138  }
139 
140  std::exception_ptr firstDetectedRuntimeError{nullptr};
141  for(auto& elem : _highLevelElements) {
142  elem->preWriteAndHandleExceptions(TransferType::write, versionNumber);
143  if((elem->_activeException != nullptr) && (firstDetectedRuntimeError == nullptr)) {
144  firstDetectedRuntimeError = elem->_activeException;
145  }
146  }
147 
148  if(firstDetectedRuntimeError == nullptr) {
149  for(auto& it : _lowLevelElementsAndExceptionFlags) {
150  auto& elem = it.first;
151  elem->handleTransferException([&] { elem->writeTransfer(versionNumber); });
152  if((elem->_activeException != nullptr) && (firstDetectedRuntimeError == nullptr)) {
153  firstDetectedRuntimeError = elem->_activeException;
154  }
155  }
156  }
157 
158  _nRuntimeErrors = 0;
159  for(auto& elem : _highLevelElements) {
160  // check for exceptions on any of the element's low level elements
161  for(auto& lowLevelElem : elem->getHardwareAccessingElements()) {
162  // In case there are mupliple exceptions we take the last one, but this does not matter. They are all ChimeraTK::runtime_errors and
163  // the first detected runtime error, which is re-thrown, has already been determined by previously.
164  if(lowLevelElem->_activeException) {
165  // copy the runtime error from low level element into the high level element so it is processed in post-read
166  elem->_activeException = lowLevelElem->_activeException;
167  }
168  }
169 
170  try {
171  elem->postWrite(TransferType::write, versionNumber);
172  }
173  catch(ChimeraTK::runtime_error&) {
174  ++_nRuntimeErrors;
175  }
176  }
177 
178  if(_nRuntimeErrors != 0) {
179  assert(firstDetectedRuntimeError != nullptr);
180  _cachedReadableWriteableIsValid = false;
181  std::rethrow_exception(firstDetectedRuntimeError);
182  }
183  }

209  void TransferGroup::addAccessor(TransferElementAbstractor& accessor) {
210  // check if accessor is already in a transfer group
211  if(accessor.getHighLevelImplElement()->_isInTransferGroup) {
212  throw ChimeraTK::logic_error("The given accessor is already in a TransferGroup and cannot be added "
213  "to another.");
214  }
215 
216  // Only accessors without wait_for_new_data can be used in a transfer group.
217  if(accessor.getAccessModeFlags().has(AccessMode::wait_for_new_data)) {
218  throw ChimeraTK::logic_error(
219  "A TransferGroup can only be used with transfer elements that don't have aAccessMode::wait_for_new_data.");
220  }
221 
222  // set flag on the accessors that it is now in a transfer group
223  accessor.getHighLevelImplElement()->_isInTransferGroup = true;
224 
225  _exceptionBackends.insert(accessor.getHighLevelImplElement()->getExceptionBackend());
226 
227  auto highLevelElementsWithNewAccessor = _highLevelElements;
228  highLevelElementsWithNewAccessor.insert(accessor.getHighLevelImplElement());
229 
230  // try replacing all internal elements in all high-level elements
231  for(auto& hlElem1 : highLevelElementsWithNewAccessor) {
232  auto list = hlElem1->getInternalElements();
233  list.push_front(hlElem1);
234 
235  for(auto& replacement : list) {
236  // try on the abstractor first, to make sure we replace at the highest
237  // level if possible
238  accessor.replaceTransferElement(replacement);
239  // try on all high-level elements already stored in the list
240  for(auto& hlElem : highLevelElementsWithNewAccessor) {
241  hlElem->replaceTransferElement(replacement); // note: this does nothing, if the replacement cannot
242  // be used by the hlElem!
243  }
244  }
245  }
246 
247  // store the accessor in the list of high-level elements
248  // this must be done only now, since it may have been replaced during the
249  // replacement process
250  _highLevelElements.insert(accessor.getHighLevelImplElement());
251 
252  // update the list of hardware-accessing elements, since we might just have
253  // made some of them redundant since we are using a set to store the elements,
254  // duplicates are intrinsically avoided.
255  _lowLevelElementsAndExceptionFlags.clear();
256  for(auto& hlElem : _highLevelElements) {
257  for(auto& hwElem : hlElem->getHardwareAccessingElements())
258  _lowLevelElementsAndExceptionFlags.insert({hwElem, false});
259  }
260 
261  // update the list of CopyRegisterDecorators
262  _copyDecorators.clear();
263  for(auto& hlElem : _highLevelElements) {
264  if(boost::dynamic_pointer_cast<ChimeraTK::CopyRegisterDecoratorTrait>(hlElem) != nullptr) {
265  _copyDecorators.insert(hlElem);
266  }
267  for(auto& hwElem : hlElem->getInternalElements()) {
268  if(boost::dynamic_pointer_cast<ChimeraTK::CopyRegisterDecoratorTrait>(hwElem) != nullptr) {
269  _copyDecorators.insert(hwElem);
270  }
271  }
272  }
273 
274  // read-write flag may need to be updated
275  _cachedReadableWriteableIsValid = false;
276  }
