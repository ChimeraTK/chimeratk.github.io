81  bool isMergeable(const boost::shared_ptr<TransferElement const>& other) const {
82  if(!_dev->canMergeRequests()) return false;
83 
84  // accessor type, device and bar must be the same
85  auto rhsCasted = boost::dynamic_pointer_cast<const NumericAddressedLowLevelTransferElement>(other);
86  if(!rhsCasted) return false;
87  if(_dev != rhsCasted->_dev) return false;
88  if(_bar != rhsCasted->_bar) return false;
89 
90  // only allow adjacent and overlapping address areas to be merged
91  if(_startAddress + _numberOfBytes < rhsCasted->_startAddress) return false;
92  if(_startAddress > rhsCasted->_startAddress + rhsCasted->_numberOfBytes) return false;
93  return true;
94  }

136  void setAddress(size_t startAddress, size_t numberOfBytes) {
137  // change address
138  _startAddress = startAddress;
139  _numberOfBytes = numberOfBytes;
140 
141  // make sure access is properly aligned
142  _isUnaligned = false;
143  auto alignment = _dev->minimumTransferAlignment(_bar);
144  auto start_padding = _startAddress % alignment;
145  assert(_startAddress >= start_padding);
146  if(start_padding > 0) _isUnaligned = true;
147  _startAddress -= start_padding;
148  _numberOfBytes += start_padding;
149  auto end_padding = alignment - _numberOfBytes % alignment;
150  if(end_padding != alignment) {
151  _isUnaligned = true;
152  _numberOfBytes += end_padding;
153  }
154 
155  // Allocated the buffer
156  rawDataBuffer.resize(_numberOfBytes);
157 
158  // update the name
159  _name = "NALLTE:" + std::to_string(_startAddress) + "+" + std::to_string(_numberOfBytes);
160  }
