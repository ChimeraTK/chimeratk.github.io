96  void Device::readReg(const std::string& regName, const std::string& regModule, int32_t* data, size_t dataSize,
97  uint32_t addRegOffset) const {
98  if(dataSize % sizeof(int32_t) != 0) {
99  throw ChimeraTK::logic_error("Wrong data size - must be dividable by 4");
100  }
101  if(addRegOffset % sizeof(int32_t) != 0) {
102  throw ChimeraTK::logic_error("Wrong additional register offset - must be dividable by 4");
103  }
104  auto vec = read<int32_t>(
105  RegisterPath(regModule) / regName, dataSize / sizeof(int32_t), addRegOffset / sizeof(int32_t), true);
106  memcpy(data, vec.data(), vec.size() * sizeof(int32_t));
107  }

126  std::vector<int32_t> vec(dataSize / sizeof(int32_t));
