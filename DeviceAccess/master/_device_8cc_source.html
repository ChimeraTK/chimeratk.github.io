70  void Device::readReg(const std::string& regName, const std::string& regModule, int32_t* data, size_t dataSize,
71  uint32_t addRegOffset) const {
72  if(dataSize % sizeof(int32_t) != 0) {
73  throw ChimeraTK::logic_error("Wrong data size - must be dividable by 4");
74  }
75  if(addRegOffset % sizeof(int32_t) != 0) {
76  throw ChimeraTK::logic_error("Wrong additional register offset - must be dividable by 4");
77  }
78  auto vec = read<int32_t>(
79  RegisterPath(regModule) / regName, dataSize / sizeof(int32_t), addRegOffset / sizeof(int32_t), true);
80  memcpy(data, vec.data(), vec.size() * sizeof(int32_t));
81  }

91  void Device::writeReg(const std::string& regName, const std::string& regModule, int32_t const* data, size_t dataSize,
92  uint32_t addRegOffset) {
93  if(dataSize == 0) dataSize = sizeof(int32_t);

96  }
97  if(addRegOffset % sizeof(int32_t) != 0) {
98  throw ChimeraTK::logic_error("Wrong additional register offset - must be dividable by 4");
99  }
100  std::vector<int32_t> vec(dataSize / sizeof(int32_t));
101  memcpy(vec.data(), data, dataSize);
102  write(RegisterPath(regModule) / regName, vec, addRegOffset / sizeof(int32_t), true);
