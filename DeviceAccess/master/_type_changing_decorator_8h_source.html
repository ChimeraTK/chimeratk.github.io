452  template<class T, class IMPL_T>
453  void TypeChangingRangeCheckingDecorator<T, IMPL_T>::convertAndCopyFromImpl() {
454  typedef boost::numeric::converter<T, IMPL_T, boost::numeric::conversion_traits<T, IMPL_T>,
455  boost::numeric::def_overflow_handler, csa_helpers::Round<double>>
456  FromImplConverter;
457  // fixme: are iterartors more efficient?
458  for(size_t i = 0; i < this->buffer_2D.size(); ++i) {
459  for(size_t j = 0; j < this->buffer_2D[i].size(); ++j) {
460  try {
461  this->buffer_2D[i][j] = FromImplConverter::convert(_target->accessChannel(i)[j]);
462  }
463  catch(boost::numeric::positive_overflow&) {
464  this->buffer_2D[i][j] = std::numeric_limits<T>::max();
465  }
466  catch(boost::numeric::negative_overflow&) {
467  this->buffer_2D[i][j] = std::numeric_limits<T>::min();
468  }
469  }
470  }
471  }

595  struct DecoratorMapKey {
596  boost::shared_ptr<ChimeraTK::TransferElement> element;
597  const std::type_info& dataType;
598  ChimeraTK::DecoratorType conversionType;
599  bool operator<(const DecoratorMapKey& other) const {
600  if(element < other.element) return true;
601  if(element != other.element) return false;
602  if(&dataType < &other.dataType) return true;
603  if(&dataType != &other.dataType) return false;
604  if(conversionType < other.conversionType) return true;
605  return false;
606  }
607  bool operator==(const DecoratorMapKey& other) const {
608  return element == other.element && dataType == other.dataType && conversionType == other.conversionType;
609  }

619  template<class UserType>
620  class DecoratorFactory {
621  public:
622  DecoratorFactory(boost::shared_ptr<ChimeraTK::TransferElement> theImpl_, DecoratorType wantedDecoratorType_)
623  : theImpl(theImpl_), wantedDecoratorType(wantedDecoratorType_) {}
624  boost::shared_ptr<ChimeraTK::TransferElement> theImpl;
625  DecoratorType wantedDecoratorType;
626  mutable boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> createdDecorator;
627 
628  DecoratorFactory(const DecoratorFactory&) = delete;
629 
630  template<typename PAIR>
631  void operator()(PAIR&) const {
632  typedef typename PAIR::first_type TargetImplType;
633  if(typeid(TargetImplType) != theImpl->getValueType()) return;
634 
635  if(wantedDecoratorType == DecoratorType::limiting) {
636  createdDecorator.reset(new TypeChangingRangeCheckingDecorator<UserType, TargetImplType>(
637  boost::dynamic_pointer_cast<ChimeraTK::NDRegisterAccessor<TargetImplType>>(theImpl)));
638  }
639  else if(wantedDecoratorType == DecoratorType::C_style_conversion) {
640  createdDecorator.reset(new TypeChangingDirectCastDecorator<UserType, TargetImplType>(
641  boost::dynamic_pointer_cast<ChimeraTK::NDRegisterAccessor<TargetImplType>>(theImpl)));
642  }
643  else {
644  throw ChimeraTK::logic_error("TypeChangingDecorator with range "
645  "limitation is not implemented yet.");
646  }
647  }

651  // requesting a decorator. Implementation, declared at the top of this file
652  template<class UserType>
653  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> getTypeChangingDecorator(
654  const boost::shared_ptr<ChimeraTK::TransferElement>& transferElement, DecoratorType decoratorType) {
655  // check if there already is a decorator for the transfer element with the right type
656  DecoratorMapKey key{transferElement, typeid(UserType), decoratorType};
657  auto decoratorMapEntry = getGlobalDecoratorMap().find(key);
658  if(decoratorMapEntry != getGlobalDecoratorMap().end()) {
659  // There already is a decorator for this transfer element
660  // The decorator has to have a matching type, otherwise we can only throw
661  auto castedType = boost::dynamic_pointer_cast<ChimeraTK::NDRegisterAccessor<UserType>>(decoratorMapEntry->second);
662  assert(castedType);
663  assert(boost::dynamic_pointer_cast<DecoratorTypeHolder>(decoratorMapEntry->second));
664  return castedType;
665  }
666 
667  DecoratorFactory<UserType> factory(transferElement, decoratorType);
668  boost::fusion::for_each(ChimeraTK::userTypeMap(), std::ref(factory));
669  if(factory.createdDecorator == nullptr) {
670  throw ChimeraTK::logic_error("ChimeraTK::ControlSystemAdapter: Decorator for TransferElement " +
671  transferElement->getName() + " has been requested for an unknown user type: " + typeid(UserType).name());
672  }
673  getGlobalDecoratorMap()[key] = factory.createdDecorator;
674  return factory.createdDecorator;
675  }
