98  template<typename UserType, typename NUMERIC>
99  UserType detail::numericToUserType_impl<UserType, NUMERIC>::impl(NUMERIC value) {
100  typedef boost::numeric::converter<UserType, NUMERIC, boost::numeric::conversion_traits<UserType, NUMERIC>,
101  boost::numeric::def_overflow_handler, Round<NUMERIC>>
102  converter;
103  // There seems to be no other way to alter the value on negative/positive overflow than using a try-catch-block
104  // here. The overflow_handler is a stateless class and hence can either throw another exception or do nothing.
105  // It does not have any influence on the converted value, and it cannot transport the information out differently.
106  try {
107  return converter::convert(value);
108  }
109  catch(boost::numeric::negative_overflow&) {
110  return std::numeric_limits<UserType>::min();
111  }
112  catch(boost::numeric::positive_overflow&) {
113  return std::numeric_limits<UserType>::max();
114  }
115  }
