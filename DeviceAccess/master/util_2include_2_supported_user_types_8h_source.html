216  template<typename UserType, typename NUMERIC>
217  UserType detail::numericToUserType_impl<UserType, NUMERIC>::impl(NUMERIC value) {
218  typedef boost::numeric::converter<UserType, NUMERIC, boost::numeric::conversion_traits<UserType, NUMERIC>,
219  boost::numeric::def_overflow_handler, Round<NUMERIC>>
220  converter;
221  // There seems to be no other way to alter the value on negative/positive overflow than using a try-catch-block
222  // here. The overflow_handler is a stateless class and hence can either throw another exception or do nothing.
223  // It does not have any influence on the converted value, and it cannot transport the information out differently.
224  try {
225  return converter::convert(value);
226  }
227  catch(boost::numeric::negative_overflow&) {
228  return std::numeric_limits<UserType>::min();
229  }
230  catch(boost::numeric::positive_overflow&) {
231  return std::numeric_limits<UserType>::max();
232  }
233  }

286  template<typename NUMERIC>
287  NUMERIC detail::userTypeToNumeric_impl<std::string, NUMERIC>::impl(const std::string& value) {
288  if constexpr(std::is_same<NUMERIC, Boolean>::value) {
289  // special treatment for Boolean
290  std::stringstream ss(value);
291  Boolean converted;
292  ss >> converted;
293  return converted;
294  }
295 
296  if constexpr(!std::is_same<NUMERIC, int8_t>::value && !std::is_same<NUMERIC, uint8_t>::value) {
297  NUMERIC v;
298  std::stringstream ss(value);
299  ss >> v;
300  return v;
301  }
302  else {
303  // int8_t and uint8_t are interpreted as char resp. unsigned char, in which case the stream operator does the
304  // wrong thing...
305  auto temp = std::stoi(value);
306  return detail::userTypeToNumeric_impl<decltype(temp), NUMERIC>::impl(temp);
307  }
308  }
