196  template<typename UserType, typename NUMERIC>
197  UserType detail::numericToUserType_impl<UserType, NUMERIC>::impl(NUMERIC value) {
198  typedef boost::numeric::converter<UserType, NUMERIC, boost::numeric::conversion_traits<UserType, NUMERIC>,
199  boost::numeric::def_overflow_handler, Round<NUMERIC>>
200  converter;
201  // There seems to be no other way to alter the value on negative/positive overflow than using a try-catch-block
202  // here. The overflow_handler is a stateless class and hence can either throw another exception or do nothing.
203  // It does not have any influence on the converted value, and it cannot transport the information out differently.
204  try {
205  return converter::convert(value);
206  }
207  catch(boost::numeric::negative_overflow&) {
208  return std::numeric_limits<UserType>::min();
209  }
210  catch(boost::numeric::positive_overflow&) {
211  return std::numeric_limits<UserType>::max();
212  }
213  }

265  template<typename NUMERIC>
266  NUMERIC detail::userTypeToNumeric_impl<std::string, NUMERIC>::impl(const std::string& value) {
267  if constexpr(!std::is_same<NUMERIC, int8_t>::value && !std::is_same<NUMERIC, uint8_t>::value) {
268  NUMERIC v;
269  std::stringstream ss(value);
270  ss >> v;
271  return v;
272  }
273  else {
274  // int8_t and uint8_t are interpreted as char resp. unsigned char, in which case the stream operator does the
275  // wrong thing...
276  auto temp = std::stoi(value);
277  return detail::userTypeToNumeric_impl<decltype(temp), NUMERIC>::impl(temp);
278  }
279  }
