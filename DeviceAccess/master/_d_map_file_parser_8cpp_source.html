18  DeviceInfoMapPointer DMapFileParser::parse(const std::string& file_name) {
19  std::ifstream file;
20  std::string line;
21  uint32_t line_nr = 0;
22 
23  std::string absPathToDMapFile = utilities::convertToAbsolutePath(file_name);
24 
25  file.open(absPathToDMapFile.c_str());
26  if(!file) {
27  throw ChimeraTK::logic_error("Cannot open dmap file: \"" + absPathToDMapFile + "\"");
28  }
29 
30  DeviceInfoMapPointer dmap(new DeviceInfoMap(absPathToDMapFile));
31  while(std::getline(file, line)) {
32  line_nr++;
33  line.erase(line.begin(), std::find_if(line.begin(), line.end(), [](int c) { return !isspace(c); }));
34  if(line.empty()) {
35  continue;
36  }
37  if(line[0] == '#') {
38  continue;
39  }
40  if(line[0] == '@') {
41  parseForLoadLib(file_name, line, line_nr, dmap);
42  continue;
43  }
44  parseRegularLine(file_name, line, line_nr, dmap);
45  }
46  file.close();
47  if(dmap->getSize() == 0) {
48  // need to throw special exception for the DMapFilesParser - this is
49  // actually a ChimeraTK::logic_error exception!
50  throw ChimeraTK::detail::EmptyDMapFileException("No data in in dmap file: \"" + file_name + "\"");
51  }
52  return dmap;
53  }

55  void DMapFileParser::parseForLoadLib(
56  const std::string& file_name, const std::string& line, uint32_t line_nr, const DeviceInfoMapPointer& dmap) {
57  // we expect at leat two tokens: the key and the value
58  std::istringstream s;
59  std::string key, value;
60  s.str(line);
61  s >> key >> value;
62  if(s && (key == "@LOAD_LIB")) {
63  dmap->addPluginLibrary(absPathOfDMapContent(value, file_name));
64  }
65  else {
66  raiseException(file_name, line, line_nr);
67  }
68  }

70  void DMapFileParser::parseRegularLine(
71  const std::string& file_name, const std::string& line, uint32_t line_nr, const DeviceInfoMapPointer& dmap) {
72  std::istringstream inStream;
73  DeviceInfoMap::DeviceInfo deviceInfo;
74 
75  inStream.str(line);
76  inStream >> deviceInfo.deviceName >> deviceInfo.uri >> deviceInfo.mapFileName;
77 
78  if(inStream) {
79  std::string absPathToDMapFile = utilities::convertToAbsolutePath(file_name);
80  std::string absPathToMapFile = absPathOfDMapContent(deviceInfo.mapFileName, file_name);
81  deviceInfo.mapFileName = absPathToMapFile;
82  deviceInfo.dmapFileName = absPathToDMapFile;
83  deviceInfo.dmapFileLineNumber = line_nr;
84  dmap->insert(deviceInfo);
85  }
86  else {
87  std::istringstream inStream2;
88  inStream2.str(line);
89  inStream2 >> deviceInfo.deviceName >> deviceInfo.uri;
90 
91  if(inStream2) {
92  std::string absPathToDMapFile = utilities::convertToAbsolutePath(file_name);
93  deviceInfo.mapFileName = "";
94  deviceInfo.dmapFileName = absPathToDMapFile;
95  deviceInfo.dmapFileLineNumber = line_nr;
96  dmap->insert(deviceInfo);
97  }
98  else {
99  raiseException(file_name, line, line_nr);
100  }
101  }
102  }

104  std::string DMapFileParser::absPathOfDMapContent(const std::string& dmapContent, const std::string& dmapFileName) {
105  // first determine the absolute path to the dmap file
106  std::string absPathToDMapFile = utilities::convertToAbsolutePath(dmapFileName);
107  // then extract the directory
108  std::string absPathToDmapDirectory = utilities::extractDirectory(absPathToDMapFile);
109  // now concatenate the dmap diretory to the entry in the dmap file (if the
110  // latter is not absolute)
111  return utilities::concatenatePaths(absPathToDmapDirectory, dmapContent);
112  }
