124  void read() {
125  if(TransferElement::_isInTransferGroup) {
126  throw ChimeraTK::logic_error("Calling read() or write() on an accessor "
127  "which is part of a TransferGroup "
128  "is not allowed.");
129  }
130  this->readTransactionInProgress = false;
131 
132  preReadAndHandleExceptions(TransferType::read);
133  if(!_activeException) {
134  handleTransferException([&] { readTransfer(); });
135  }
136 
137  postRead(TransferType::read, !_activeException);

151  bool readNonBlocking() {
152  if(TransferElement::_isInTransferGroup) {
153  throw ChimeraTK::logic_error("Calling read() or write() on an accessor "
154  "which is part of a TransferGroup "
155  "is not allowed.");
156  }
157  this->readTransactionInProgress = false;
158  preReadAndHandleExceptions(TransferType::readNonBlocking);
159  bool updateDataBuffer = false;
160  if(!_activeException) {
161  handleTransferException([&] { updateDataBuffer = readTransferNonBlocking(); });
162  }
163 
164  bool retVal = updateDataBuffer;
165  if(_activeException) {
166  auto previousVersionNumber = _versionNumber;
167  auto previousDataValidity = _dataValidity;
168  // always call postRead with updateDataBuffer = false in case of an exception
169  postRead(TransferType::readNonBlocking, false);
170  // Usually we do not reach this point because postRead() is re-throwing the _activeExeption.
171  // If we reach this point the exception has been suppressed. We have to calculate a
172  // new return value because the dataBuffer has not changed, but the meta data
173  // could have, in which case we have to return true.
174  if((previousVersionNumber != _versionNumber) || (previousDataValidity != _dataValidity)) {
175  retVal = true;
176  }
177  else {
178  retVal = false;
179  }
180  }
181  else {
182  // call postRead with updateDataBuffer as returned by readTransferNonBlocking
183  postRead(TransferType::readNonBlocking, updateDataBuffer);
184  }
185  return retVal;
186  }

192  bool readLatest() {
193  if(_accessModeFlags.has(AccessMode::wait_for_new_data)) {
194  bool updateDataBuffer = false;
195  // Call readNonBlocking until there is no new data to be read any more
196  while(readNonBlocking()) {
197  // remember whether we have new data
198  updateDataBuffer = true;
199  }
200  return updateDataBuffer;
201  }
202  else {
203  // Without wait_for_new_data readNonBlocking always returns true, and the while loop above would never end.
204  // Hence we just call the (synchronous) read and return true;
205  read();
206  return true;
207  }
208  }

482  void postRead(TransferType type, bool updateDataBuffer) {
483  // only delegate to doPostRead() the first time postRead() is called in a row.
484  if(readTransactionInProgress) {
485  readTransactionInProgress = false;
486  doPostRead(type, updateDataBuffer);
487  }
488 
489  // Throw on each call of postRead(). All high-level elements for a shared low-level transfer element must see the exception.
490  // Note: doPostRead can throw an exception, but in that case _activeException must be false (we can only have one
491  // exception at a time). In case other code is added here later which needs to be executed after doPostRead()
492  // always, a try-catch block may be necessary.
493  if(_activeException) {
494  // don't clear the active connection. This is done in preRead().
495  std::rethrow_exception(_activeException);
496  }
