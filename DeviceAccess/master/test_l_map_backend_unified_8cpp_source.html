105 template<typename Derived>
106 struct OneDRegisterDescriptorBase : RegisterDescriptorBase<Derived> {
107  using RegisterDescriptorBase<Derived>::derived;
108 
109  size_t nChannels() { return 1; }
110 
111  size_t myOffset() { return 0; }
112 
113  // T is always minimumUserType, but C++ doesn't allow to use Derived::minimumUserType here (circular dependency)
114  template<typename T, typename Traw>
115  T convertRawToCooked(Traw value) {
116  return static_cast<T>(value);
117  }
118 
119  template<typename UserType>
120  void generateValueHook(std::vector<UserType>&) {} // override in derived if needed
121 
122  // type can be user type or raw type
123  template<typename Type>
124  std::vector<std::vector<Type>> generateValue(bool getRaw = false) {
125  std::vector<Type> v;
126  typedef typename Derived::rawUserType Traw;
127  typedef typename Derived::minimumUserType T;
128  auto cv = derived->template getRemoteValue<Traw>(true)[0];
129  for(size_t i = 0; i < derived->nElementsPerChannel(); ++i) {
130  Traw e = cv[i] + derived->increment * (static_cast<Traw>(i) + 1);
131  if(!getRaw) {
132  v.push_back(derived->template convertRawToCooked<T, Traw>(e));
133  }
134  else {
135  v.push_back(static_cast<T>(e));
136  }
137  }
138  derived->generateValueHook(v);
139  return {v};
140  }
141 
142  template<typename UserType>
143  std::vector<std::vector<UserType>> getRemoteValue(bool getRaw = false) {
144  std::vector<UserType> v;
145  typedef typename Derived::rawUserType Traw;
146  typedef typename Derived::minimumUserType T;
147  auto bufferLock = derived->acc.getBufferLock();
148  for(size_t i = 0; i < derived->nElementsPerChannel(); ++i) {
149  Traw e = derived->acc[i + derived->myOffset()];
150  if(!getRaw) {
151  v.push_back(derived->template convertRawToCooked<T, Traw>(e));
152  }
153  else {
154  v.push_back(static_cast<T>(e)); // you can only use raw if user type and raw type are the same, so the static
155  // cast is a no-op
156  }
157  }
158  return {v};
159  }
160 
161  void setRemoteValue() {
162  auto v = generateValue<typename Derived::rawUserType>(true)[0];
163  { // scope for the buffer lock
164  auto bufferLock = derived->acc.getBufferLock();
165  for(size_t i = 0; i < derived->nElementsPerChannel(); ++i) {
166  derived->acc[i + derived->myOffset()] = v[i];
167  }
168  } // release the buffer lock before triggering another thread
169  if(derived->isPush()) {
170  dynamic_cast<ExceptionDummy&>(derived->acc.getBackend()).triggerInterrupt(5, 6);
171  }
172  }
173 };

213 template<typename Derived>
214 struct VariableRegisterDescriptorBase : OneDRegisterDescriptorBase<Derived> {
215  using RegisterDescriptorBase<Derived>::derived;
216  static constexpr auto capabilities = RegisterDescriptorBase<Derived>::capabilities.disableTestRawTransfer();
217 
218  size_t nChannels() { return 1; }
219  ChimeraTK::AccessModeFlags supportedFlags() { return {ChimeraTK::AccessMode::wait_for_new_data}; }
220 
221  size_t nRuntimeErrorCases() { return 0; }
222 
223  template<typename UserType>
224  std::vector<std::vector<UserType>> getRemoteValue(bool = false) {
225  // For Variables we don't have a backdoor. We have to use the normal read and write
226  // functions which are good enough. It seems like a self consistency test, but all
227  // functionality the variable has to provide is that I can write something, and
228  // read it back, which is tested with it.
229 
230  // We might have to open/recover the backend to perform the operation. We have to remember
231  // that we did so and close/set-exception it again it we did. Some tests require the backend to be closed.
232  bool backendWasOpened = lmapBackend->isOpen();
233  bool backendWasFunctional = lmapBackend->isFunctional();
234  if(!backendWasOpened || !backendWasFunctional) {
235  lmapBackend->open();
236  }
237  auto acc = lmapBackend->getRegisterAccessor<typename Derived::minimumUserType>(derived->path(), 0, 0, {});
238  acc->read();
239  if(!backendWasOpened) {
240  lmapBackend->close();
241  }
242  else if(!backendWasFunctional) {
243  lmapBackend->setException();
244  }
245  std::vector<UserType> v;
246  for(size_t k = 0; k < derived->nElementsPerChannel(); ++k) {
247  v.push_back(acc->accessData(k));
248  }
249  return {v};
250  }
251 
252  void setRemoteValue() {
253  auto acc = lmapBackend->getRegisterAccessor<typename Derived::minimumUserType>(derived->path(), 0, 0, {});
254  auto v = derived->template generateValue<typename Derived::minimumUserType>()[0];
255  for(size_t k = 0; k < derived->nElementsPerChannel(); ++k) {
256  acc->accessData(k) = v[k];
257  }
258  bool backendWasOpened = lmapBackend->isOpen();
259  if(!backendWasOpened) {
260  lmapBackend->open();
261  }
262  try {
263  acc->write();
264  }
265  catch(...) {
266  // ignore any exceptions: if the device is in an exception state, this write must not take place but the exception
267  // must not get through.
268  }
269  if(!backendWasOpened) {
270  lmapBackend->close();
271  }
272  }
273 
274  void setForceRuntimeError(bool, size_t) { assert(false); }
275 };
