323  template<typename UserType>
324  uint32_t FixedPointConverter::toRaw(UserType cookedValue) const {
325  // Do a range check first. The later overflow check in the conversion is not
326  // sufficient, since we can have non-standard word sizes like 12 bits.
327  if(cookedValue < boost::fusion::at_key<UserType>(_minCookedValues)) {
328  return _minRawValue;
329  }
330  if(cookedValue > boost::fusion::at_key<UserType>(_maxCookedValues)) {
331  return _maxRawValue;
332  }
333 
334  // handle integer and floating-point types differently
335  if(std::numeric_limits<UserType>::is_integer && _fractionalBits == 0) {
336  // extract the sign and leave the positive number
337  bool isNegative = isNegativeUserType(cookedValue);
338  if(isNegative && !_isSigned) return _minRawValue;
339  if(isNegative)
340  cookedValue = -(cookedValue + 1); // bit inversion, ~ operator cannot be
341  // used as UserType might be double
342 
343  // cast into raw type
344  uint32_t rawValue = static_cast<uint32_t>(cookedValue);
345 
346  // handle sign
347  if(_isSigned && isNegative) {
348  rawValue = ~rawValue;
349  }
350 
351  // return with bit mask applied
352  return rawValue & _usedBitsMask;
353  }
354  else {
355  // convert into double and scale by fractional bit coefficient
356  double d_cooked = _inverseFractionalBitsCoefficient * static_cast<double>(cookedValue);
357 
358  // Convert into either signed or uinsigned int32_t, depending on _isSigned,
359  // so the conversion handls the sign correctly. Store always in a uint32_t,
360  // since this is our raw type. The conversion will properly round, when
361  // needed. Negative overflow exceptions need to be cought for some corner
362  // cases (e.g. number of fractional bits >= number of bits in total).
363  // Positive overflow cannot happen due to the rangecheck above (the negative
364  // branch has one more possible value).
365  uint32_t raw;
366  try {
367  if(_isSigned) {
368  typedef boost::numeric::converter<int32_t, double, boost::numeric::conversion_traits<int32_t, double>,
369  boost::numeric::def_overflow_handler, Round<double>>
370  converter_signed;
371  raw = converter_signed::convert(d_cooked);
372  }
373  else {
374  typedef boost::numeric::converter<uint32_t, double, boost::numeric::conversion_traits<uint32_t, double>,
375  boost::numeric::def_overflow_handler, Round<double>>
376  converter_unsigned;
377  raw = converter_unsigned::convert(d_cooked);
378  }
379  }
380  catch(boost::numeric::negative_overflow& e) {
381  raw = _minRawValue;
382  }
383 
384  // apply bit mask
385  return raw & _usedBitsMask;
386  }
387  }
