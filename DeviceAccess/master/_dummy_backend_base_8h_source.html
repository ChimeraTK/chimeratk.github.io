103  template<typename UserType>
104  boost::shared_ptr<NDRegisterAccessor<UserType>> getRegisterAccessor_impl(const RegisterPath& registerPathName,
105  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
106  // Suffix to mark writeable references to read-only registers
107  static const std::string DUMMY_WRITEABLE_SUFFIX{".DUMMY_WRITEABLE"};
108 
109  bool isDummyWriteableAccessor = false;
110  RegisterPath actualRegisterPath{registerPathName};
111 
112  // Check if register name ends on DUMMY_WRITEABLE_SUFFIX,
113  // in that case, set actual path to the "real" register
114  // which exists in the catalogue.
115  const std::string regPathNameStr{registerPathName};
116  std::smatch match;
117  const std::regex re{DUMMY_WRITEABLE_SUFFIX + "$"};
118  std::regex_search(regPathNameStr, match, re);
119 
120  if(!match.empty()) {
121  isDummyWriteableAccessor = true;
122  actualRegisterPath = RegisterPath{match.prefix()};
123  }
124 
125  auto accessor = NumericAddressedBackend::getRegisterAccessor_impl<UserType>(
126  actualRegisterPath, numberOfWords, wordOffsetInRegister, flags);
127 
128  // Modify writeability of the synchronous NumericAddressedBackendRegisterAccessor
129  if(isDummyWriteableAccessor) {
130  // the accessor might be synchronous or asynchronous. If it is an async accessor we have to add a dummy-writable accessor
131  boost::shared_ptr<NDRegisterAccessor<UserType>> syncAccessor;
132  if(flags.has(AccessMode::wait_for_new_data)) {
133  auto syncFlags = flags;
134  syncFlags.remove(AccessMode::wait_for_new_data);
135  syncAccessor = NumericAddressedBackend::getRegisterAccessor_impl<UserType>(
136  actualRegisterPath, numberOfWords, wordOffsetInRegister, syncFlags);
137  // we cannot add the new sync accessor to the async accessor as writeAccessor here, because it is not writeable yet
138  }
139  else {
140  syncAccessor = accessor;
141  }
142 
143  const auto info{getRegisterInfo(actualRegisterPath)};
144 
145  if(info->dataType == RegisterInfoMap::RegisterInfo::Type::FIXED_POINT) {
146  if(flags.has(AccessMode::raw)) {
147  boost::dynamic_pointer_cast<NumericAddressedBackendRegisterAccessor<UserType, FixedPointConverter, true>>(
148  syncAccessor)
149  ->makeWriteable();
150  }
151  else {
152  if(info->getNumberOfDimensions() < 2) {
153  boost::dynamic_pointer_cast<
154  NumericAddressedBackendRegisterAccessor<UserType, FixedPointConverter, false>>(syncAccessor)
155  ->makeWriteable();
156  }
157  else {
158  boost::dynamic_pointer_cast<NumericAddressedBackendMuxedRegisterAccessor<UserType>>(syncAccessor)
159  ->makeWriteable();
160  }
161  }
162  }
163  else if(info->dataType == RegisterInfoMap::RegisterInfo::Type::IEEE754) {
164  if(flags.has(AccessMode::raw)) {
165  boost::dynamic_pointer_cast<
166  NumericAddressedBackendRegisterAccessor<UserType, IEEE754_SingleConverter, true>>(syncAccessor)
167  ->makeWriteable();
168  }
169  else {
170  boost::dynamic_pointer_cast<
171  NumericAddressedBackendRegisterAccessor<UserType, IEEE754_SingleConverter, false>>(syncAccessor)
172  ->makeWriteable();
173  }
174  }
175 
176  if(flags.has(AccessMode::wait_for_new_data)) {
177  // We still have to set the now writeable synchronous accessor in the asynchronous accessor to enable writing
178  boost::dynamic_pointer_cast<AsyncNDRegisterAccessor<UserType>>(accessor)->setWriteAccessor(syncAccessor);
179  }
180  }
181 
182  return accessor;
183  }

201  template<typename T, typename... Args>
202  static boost::shared_ptr<DeviceBackend> returnInstance(const std::string& instanceId, Args&&... arguments) {
203  if(instanceId == "") {
204  // std::forward because template accepts forwarding references
205  // (Args&&) and this can have both lvalue and rvalue references passed
206  // as arguments.
207  return boost::shared_ptr<DeviceBackend>(new T(std::forward<Args>(arguments)...));
208  }
209  auto instanceMap = DerivedBackendType::getInstanceMap();
210 
211  // search instance map and create new instanceId, if not found under the
212  // name
213  if(DerivedBackendType::getInstanceMap().find(instanceId) == DerivedBackendType::getInstanceMap().end()) {
214  boost::shared_ptr<DeviceBackend> ptr(new T(std::forward<Args>(arguments)...));
215  DerivedBackendType::getInstanceMap().insert(std::make_pair(instanceId, ptr));
216  return ptr;
217  }
218  // return existing instanceId from the map
219  return boost::shared_ptr<DeviceBackend>(DerivedBackendType::getInstanceMap()[instanceId]);
