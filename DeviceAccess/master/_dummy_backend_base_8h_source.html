80  boost::shared_ptr<NDRegisterAccessor<UserType>> getRegisterAccessor_impl(const RegisterPath& registerPathName,
81  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
82  // First check if the request is for one of the special DUMMY_INTEERRUPT_X registers. if so, early return
83  // this special accessor.
84  if(registerPathName.startsWith("DUMMY_INTERRUPT_")) {
85  bool interruptFound;
86  uint32_t interrupt;
87 
88  auto* dummyCatalogue = dynamic_cast<DummyBackendRegisterCatalogue*>(_registerMapPointer.get());
89  assert(dummyCatalogue);
90  std::tie(interruptFound, interrupt) = dummyCatalogue->extractControllerInterrupt(registerPathName);
91  if(!interruptFound) {
92  throw ChimeraTK::logic_error("Unknown dummy interrupt " + registerPathName);
93  }
94 
95  // Delegate the other parameters down to the accessor which will throw accordingly, to satisfy the specification
96  // Since the accessor will keep a shared pointer to the backend, we can safely capture "this"
97  auto d = new DummyInterruptTriggerAccessor<UserType>(
98  shared_from_this(), [this, interrupt]() { return triggerInterrupt(interrupt); }, registerPathName,
99  numberOfWords, wordOffsetInRegister, flags);
100 
101  return boost::shared_ptr<NDRegisterAccessor<UserType>>(d);
102  }
103 
104  return NumericAddressedBackend::getRegisterAccessor_impl<UserType>(
105  registerPathName, numberOfWords, wordOffsetInRegister, flags);
106  }
