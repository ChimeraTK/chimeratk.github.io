34  void doPreRead(TransferType type) override {
35  _transferType = type;
36  if(this->_accessModeFlags.has(AccessMode::wait_for_new_data) && type == TransferType::readNonBlocking) {
37  // if the access mode has wait_for_new_data then readNonBlocking can be called multiple times by readLatest,
38  // without the test re-setting the counter. In this case just check that the number of calls is smaller or euqal
39  // the queue size i.e. called maximum size + 1 times.
40  BOOST_CHECK(_preRead_counter <= (this->_readQueue.size()));
41  }
42  else {
43  // in all other pre-read must be called exactly once
44  BOOST_CHECK_EQUAL(_preRead_counter, 0);
45  }
46  // in all cases, doPreRead and doPostRead must be called in pairs
47  BOOST_CHECK_EQUAL(_preRead_counter, _postRead_counter);
48  BOOST_CHECK(_preWrite_counter == 0);
49  BOOST_CHECK_EQUAL(_readTransfer_counter, 0);
50  BOOST_CHECK(_writeTransfer_counter == 0);
51  BOOST_CHECK(_postWrite_counter == 0);
52  ++_preRead_counter;
53  if(_throwLogicErr) throw ChimeraTK::logic_error("Test");
54  if(_throwRuntimeErrInPre) throw ChimeraTK::runtime_error("Test");
55  if(_throwThreadInterruptedInPre) throw boost::thread_interrupted();
56  }

138  void doPostRead(TransferType type, bool updateDataBuffer) override {
139  // doPreRead and doPortRead must always be called in pairs.
140  // This can happen multiple times in readLatest. The absolute counting is done in doPreRead()
141  BOOST_CHECK_EQUAL(_preRead_counter, _postRead_counter + 1);
142  BOOST_CHECK(_preWrite_counter == 0);
143  if(!_throwLogicErr && !_throwRuntimeErrInPre && !_throwThreadInterruptedInPre) {
144  if(!this->_accessModeFlags.has(AccessMode::wait_for_new_data)) {
145  BOOST_CHECK_EQUAL(_readTransfer_counter, 1);
146  }
147  else {
148  BOOST_CHECK_EQUAL(_readTransfer_counter, 0);
149  }
150  }
151  else {
152  /* Here B.6.1 is tested for read operations */
153  BOOST_CHECK_EQUAL(_readTransfer_counter, 0);
154  }
155  if(_throwLogicErr || _throwRuntimeErrInPre || _throwThreadInterruptedInPre || _throwRuntimeErrInTransfer ||
156  _throwThreadInterruptedInTransfer) {
157  BOOST_CHECK(this->_activeException != nullptr);
158  }
159  /* Check B.7.4 */
160  if(this->_activeException) {
161  BOOST_CHECK(updateDataBuffer == false);
162  }
163  BOOST_CHECK(_writeTransfer_counter == 0);
164  BOOST_CHECK(_postWrite_counter == 0);
165  BOOST_CHECK(_transferType == type);
166  ++_postRead_counter;
167  _hasNewData = updateDataBuffer;
168  if(_throwNumericCast) throw boost::numeric::bad_numeric_cast();
169  if(_throwThreadInterruptedInPost) throw boost::thread_interrupted();
170  }

180  void doPostWrite(TransferType type, VersionNumber versionNumber) override {
181  BOOST_CHECK(_preRead_counter == 0);
182  BOOST_CHECK(_preWrite_counter == 1);
183  BOOST_CHECK_EQUAL(_readTransfer_counter, 0);
184  if(!_throwLogicErr && !_throwRuntimeErrInPre && !_throwNumericCast && !_throwThreadInterruptedInPre) {
185  BOOST_CHECK(_writeTransfer_counter == 1);
186  }
187  else {
188  /* Here B.6.1 is tested for write operations */
189  BOOST_CHECK(_writeTransfer_counter == 0);
190  }
191  // Exceptions must be passed on to the level which is throwing it (B.6.3 This actually tests the
192  // NDRegisterAccessorDecorator)
193  if(_throwLogicErr || _throwRuntimeErrInPre || _throwThreadInterruptedInPre || _throwRuntimeErrInTransfer ||
194  _throwThreadInterruptedInTransfer || _throwNumericCast) {
195  BOOST_CHECK(this->_activeException != nullptr);
196  }
197  BOOST_CHECK(versionNumber == _newVersion);
198  BOOST_CHECK(_postRead_counter == 0);
199  BOOST_CHECK(_postWrite_counter == 0);
200  BOOST_CHECK(_transferType == type);
201  ++_postWrite_counter;
202  if(_throwThreadInterruptedInPost) throw boost::thread_interrupted();
203  }
