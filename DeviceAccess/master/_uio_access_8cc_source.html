16  UioAccess::UioAccess(std::string deviceFilePath) : _deviceFilePath(deviceFilePath.c_str()) {
17  std::string fileName = _deviceFilePath.filename().string();
18  _deviceKernelBase = (void*)readUint64HexFromFile("/sys/class/uio/" + fileName + "/maps/map0/addr");
19  _deviceMemSize = readUint64HexFromFile("/sys/class/uio/" + fileName + "/maps/map0/size");
20  _lastInterruptCount = readUint32FromFile("/sys/class/uio/" + fileName + "/event");
21 
22  // Open UIO device file here, so that interrupt thread can run before calling open()
23  _deviceFileDescriptor = ::open(_deviceFilePath.c_str(), O_RDWR);
24  if(_deviceFileDescriptor < 0) {
25  throw ChimeraTK::runtime_error("UIO: Failed to open device file '" + getDeviceFilePath() + "'");
26  }
27  }

46  void UioAccess::read(uint64_t map, uint64_t address, int32_t* data, size_t sizeInBytes) {
47  if(map > 0) {
48  throw ChimeraTK::logic_error("UIO: Multiple memory regions are not supported");
49  }
50 
51  // This is a temporary work around, because register nodes of current map use absolute bus addresses.
52  address = address % reinterpret_cast<uint64_t>(_deviceKernelBase);
53 
54  if(address + sizeInBytes > _deviceMemSize) {
55  throw ChimeraTK::logic_error("UIO: Read request exceeds device memory region");
56  }
57 
58  void* targetAddress = static_cast<uint8_t*>(_deviceUserBase) + address;
59 
60  // Is inherently thread-safe
61  std::memcpy(data, targetAddress, sizeInBytes);
62  }

81  uint32_t UioAccess::waitForInterrupt(int timeoutMs) {
82  // Represents the total interrupt count since system uptime.
83  uint32_t totalInterruptCount = 0;
84  // Will hold the number of new interrupts
85  uint32_t occurredInterruptCount = 0;
86 
87  struct pollfd pfd;
88  pfd.fd = _deviceFileDescriptor;
89  pfd.events = POLLIN;
90 
91  int ret = poll(&pfd, 1, timeoutMs);
92 
93  if(ret >= 1) {
94  // No timeout, start reading
95  ret = ::read(_deviceFileDescriptor, &totalInterruptCount, sizeof(totalInterruptCount));
96 
97  if(ret != (ssize_t)sizeof(totalInterruptCount)) {
98  throw ChimeraTK::runtime_error("UIO - Reading interrupt failed: " + std::string(std::strerror(errno)));
99  }
100 
101  // Prevent overflow of interrupt count value
102  occurredInterruptCount = subtractUint32OverflowSafe(totalInterruptCount, _lastInterruptCount);
103  _lastInterruptCount = totalInterruptCount;
104  }
105  else if(ret == 0) {
106  // Timeout
107  occurredInterruptCount = 0;
108  }
109  else {
110  throw ChimeraTK::runtime_error("UIO - Waiting for interrupt failed: " + std::string(std::strerror(errno)));
111  }
112  return occurredInterruptCount;
113  }

128  void UioAccess::UioMMap() {
129  _deviceUserBase = mmap(NULL, _deviceMemSize, PROT_READ | PROT_WRITE, MAP_SHARED, _deviceFileDescriptor, 0);
130  if(_deviceUserBase == MAP_FAILED) {
131  ::close(_deviceFileDescriptor);
132  throw ChimeraTK::runtime_error("UIO: Cannot allocate memory for UIO device '" + getDeviceFilePath() + "'");
133  }
134  return;
135  }

141  uint32_t UioAccess::subtractUint32OverflowSafe(uint32_t minuend, uint32_t subtrahend) {
142  if(subtrahend > minuend) {
143  return minuend +
144  (uint32_t)(static_cast<uint64_t>(std::numeric_limits<uint32_t>::max()) - static_cast<uint64_t>(subtrahend));
145  }
146  else {
147  return minuend - subtrahend;
148  }
