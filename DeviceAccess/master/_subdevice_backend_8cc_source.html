285  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
286  assert(type == Type::area);
287

290

292  if(info->bar != 0) {
293 // throw ChimeraTK::logic_error("SubdeviceBackend: BARs other then 0 are not supported. Register '" +
294 // registerPathName + "' is in BAR " + std::to_string(info->bar) + ".");
295  std::cout << "SubdeviceBackend: WARNING: BAR others then 0 detected. BAR 0 will be used instead. Register " << registerPathName << " is in BAR " << std::to_string(info->bar) << "." << std::endl;
296 
297  }
298 
299  // check that the register is not a 2D multiplexed register, which is not yet
300  // supported
301  if(info->is2DMultiplexed) {
302  throw ChimeraTK::logic_error("SubdeviceBackend: 2D multiplexed registers are not yet supported.");
303  }
304 
305  // compute full offset (from map file and function arguments)
306  size_t byteOffset = info->address + sizeof(int32_t) * wordOffsetInRegister;
307  if(byteOffset % 4 != 0) {
308  throw ChimeraTK::logic_error("SubdeviceBackend: Only addresses which are a "
309  "multiple of 4 are supported.");
310  }
311  size_t wordOffset = byteOffset / 4;
312 
313  // compute effective length
314  if(numberOfWords == 0) {
315  numberOfWords = info->nElements;
316  }
317  else if(numberOfWords > info->nElements) {
318  throw ChimeraTK::logic_error("SubdeviceBackend: Requested " + std::to_string(numberOfWords) +
319  " elements from register '" + registerPathName + "', which only has a length of " +
320  std::to_string(info->nElements) + " elements.");
321  }
322 
323  // check if raw transfer?
324  bool isRaw = flags.has(AccessMode::raw);
325 
326  // obtain target accessor in raw mode
327  flags.add(AccessMode::raw);
328  auto rawAcc = targetDevice->getRegisterAccessor<int32_t>(targetArea, numberOfWords, wordOffset, flags);
329 
330  // decorate with appropriate FixedPointConvertingDecorator. This is done even
331  // when in raw mode so we can properly implement getAsCooked()/setAsCooked().
332  if(!isRaw) {
333  return boost::make_shared<FixedPointConvertingDecorator<UserType, int32_t>>(
334  rawAcc, FixedPointConverter(registerPathName, info->width, info->nFractionalBits, info->signedFlag));
335  }
336  else {
337  // this is handled by the template specialisation for int32_t
338  throw ChimeraTK::logic_error("Given UserType when obtaining the SubdeviceBackend in raw mode does "

342  }
