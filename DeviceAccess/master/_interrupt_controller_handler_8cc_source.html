27  boost::shared_ptr<InterruptControllerHandler> InterruptControllerHandlerFactory::createInterruptControllerHandler(
28  std::vector<uint32_t> const& controllerID, boost::shared_ptr<TriggerDistributor<std::nullptr_t>> parent) {
29  assert(!controllerID.empty());
30  std::string name, description;
31  try {
32  std::tie(name, description) = _controllerDescriptions.at(controllerID);
33  }
34  catch(std::out_of_range&) {
35  std::string idAsString;
36  for(auto i : controllerID) {
37  idAsString += std::to_string(i) + ":";
38  }
39  idAsString.pop_back(); // remove last ":"
40  throw ChimeraTK::logic_error("Unknown interrupt controller ID " + idAsString);
41  }
42  auto creatorFunctionIter = _creatorFunctions.find(name);
43  if(creatorFunctionIter == _creatorFunctions.end()) {
44  throw ChimeraTK::logic_error("Unknown interrupt controller type \"" + name + "\"");
45  }
46  return creatorFunctionIter->second(this, controllerID, description, std::move(parent));
47  }

61  template<typename DistributorType>
62  boost::shared_ptr<DistributorType> InterruptControllerHandler::getDistributorRecursive(
63  std::vector<uint32_t> const& interruptID) {
64  // assert(false); // FIXME: needs container lock!
65  assert(!interruptID.empty());
66  auto qualifiedInterruptId = _id;
67  qualifiedInterruptId.push_back(interruptID.front());
68 
69  // we can't use try_emplace because the map contains weak pointers
70  boost::shared_ptr<TriggerDistributor<std::nullptr_t>> distributor;
71  auto distributorIter = _distributors.find(interruptID.front());
72  if(distributorIter != _distributors.end()) {
73  distributor = distributorIter->second.lock();
74  }
75  if(!distributor) {
76  distributor = boost::make_shared<TriggerDistributor<std::nullptr_t>>(
77  _backend, qualifiedInterruptId, shared_from_this(), _asyncDomain);
78  _distributors[interruptID.front()] = distributor;
79  if(_asyncDomain->unsafeGetIsActive()) {
80  // Creating a new version here is correct. Nothing has been distributed to any accessor connected to this
81  // sub-interrupt yet because we just created this distributor.
82  distributor->activate(nullptr, {});
83  }
84  }
85  return distributor->getDistributorRecursive<DistributorType>(interruptID);
86  }
