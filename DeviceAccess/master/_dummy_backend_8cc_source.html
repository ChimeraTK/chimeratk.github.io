120  std::list<boost::function<void(void)>> DummyBackend::findCallbackFunctionsForAddressRange(AddressRange addressRange) {
121  // as callback functions are not sortable, we want to loop the multimap only
122  // once.
123  // FIXME: If the same function is registered more than one, it may be executed
124  // multiple times
125 
126  // we only want the start address of the range, so we set size to 0
127  AddressRange firstAddressInBar(addressRange.bar, 0, 0);
128  AddressRange endAddress(addressRange.bar, addressRange.offset + addressRange.sizeInBytes, 0);
129 
130  std::multimap<AddressRange, boost::function<void(void)>>::iterator startIterator =
131  _writeCallbackFunctions.lower_bound(firstAddressInBar);
132 
133  std::multimap<AddressRange, boost::function<void(void)>>::iterator endIterator =
134  _writeCallbackFunctions.lower_bound(endAddress);
135 
136  std::list<boost::function<void(void)>> returnList;
137  for(std::multimap<AddressRange, boost::function<void(void)>>::iterator callbackIter = startIterator;
138  callbackIter != endIterator; ++callbackIter) {
139  if(isWriteRangeOverlap(callbackIter->first, addressRange)) {
140  returnList.push_back(callbackIter->second);
141  }
142  }
143 
144  return returnList;
145  }

147  bool DummyBackend::isWriteRangeOverlap(AddressRange firstRange, AddressRange secondRange) {
148  if(firstRange.bar != secondRange.bar) {
149  return false;
150  }
151 
152  uint64_t startAddress = std::max(firstRange.offset, secondRange.offset);
153  uint64_t endAddress =
154  std::min(firstRange.offset + firstRange.sizeInBytes, secondRange.offset + secondRange.sizeInBytes);
155 
156  // if at least one register is writable there is an overlap of writable
157  // registers
158  for(uint64_t address = startAddress; address < endAddress; address += sizeof(int32_t)) {
159  if(isReadOnly(firstRange.bar, address) == false) {
160  return true;
161  }
162  }
163 
164  // we looped all possible registers, none is writable
165  return false;
166  }
