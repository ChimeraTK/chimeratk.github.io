131  std::list<boost::function<void(void)>> DummyBackend::findCallbackFunctionsForAddressRange(AddressRange addressRange) {
132  // as callback functions are not sortable, we want to loop the multimap only
133  // once.
134  // FIXME: If the same function is registered more than one, it may be executed
135  // multiple times
136 
137  // we only want the start address of the range, so we set size to 0
138  AddressRange firstAddressInBar(addressRange.bar, 0, 0);
139  AddressRange endAddress(addressRange.bar, addressRange.offset + addressRange.sizeInBytes, 0);
140 
141  std::multimap<AddressRange, boost::function<void(void)>>::iterator startIterator =
142  _writeCallbackFunctions.lower_bound(firstAddressInBar);
143 
144  std::multimap<AddressRange, boost::function<void(void)>>::iterator endIterator =
145  _writeCallbackFunctions.lower_bound(endAddress);
146 
147  std::list<boost::function<void(void)>> returnList;
148  for(std::multimap<AddressRange, boost::function<void(void)>>::iterator callbackIter = startIterator;
149  callbackIter != endIterator; ++callbackIter) {
150  if(isWriteRangeOverlap(callbackIter->first, addressRange)) {
151  returnList.push_back(callbackIter->second);
152  }
153  }
154 
155  return returnList;
156  }

158  bool DummyBackend::isWriteRangeOverlap(AddressRange firstRange, AddressRange secondRange) {
159  if(firstRange.bar != secondRange.bar) {
160  return false;
161  }
162 
163  uint32_t startAddress = std::max(firstRange.offset, secondRange.offset);
164  uint32_t endAddress =
165  std::min(firstRange.offset + firstRange.sizeInBytes, secondRange.offset + secondRange.sizeInBytes);
166 
167  // if at least one register is writable there is an overlap of writable
168  // registers
169  for(uint32_t address = startAddress; address < endAddress; address += sizeof(int32_t)) {
170  if(isReadOnly(firstRange.bar, address) == false) {
171  return true;
172  }
173  }
174 
175  // we looped all possible registers, none is writable
176  return false;
177  }
