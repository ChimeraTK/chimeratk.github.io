44  void read(uint64_t bar, uint64_t address, int32_t* data, size_t sizeInBytes) override {
45  // Note, although ExceptionDummy::read() cannot be called concurrently with read or write from the
46  // fw simulating side, this limitation should not matter here since we only interrupt
47  // DummyForDoubleBuffering::read() and not it's base implementation
48 
49  for(unsigned i = 0; i < 2; i++) {
50  if(blockNextRead[i]) {
51  blockedInRead[i].wait();
52  unblockRead[i].wait();
53  blockNextRead[i] = false;
54  }
55  }
56  // finalize reading by calling Exception backend read
57  ChimeraTK::ExceptionDummy::read(bar, address, data, sizeInBytes);
58  }
59  // use this to request that next read blocks.
60  // array index corresponds to that of barrier arrays
61  // We know that read is called only 2nd after write (for the buffer-switching enable ctrl register),
62  // so in this sense it requests blocking after only part of the double-buffer read operation is done
63  static thread_local bool blockNextRead[2];
64  // one pair of barriers per reader thread
65  // after request that read blocks, you must wait on this
66  std::array<boost::barrier, 2> blockedInRead{boost::barrier{2}, boost::barrier{2}};
67  // use this to unblock the read
68  std::array<boost::barrier, 2> unblockRead{boost::barrier{2}, boost::barrier{2}};
69 };

242  /*
243  * Test race condition: slow reader, which blocks the Firmware from buffer switching.
244  */
245  auto accessor = d.getOneDRegisterAccessor<uint32_t>("/doubleBuffer");
246 
247  // make double buffer operation block after write to ctrl register, at read of buffer number
248  std::thread s([&] {
249  // this thread reads from double-buffered region

264

266  doubleBufferingEnabled->readLatest();
267  BOOST_CHECK(doubleBufferingEnabled->accessData(0));

270 BOOST_FIXTURE_TEST_CASE(testConcurrentRead, DeviceFixture) {
271  /*
272  * A test which exposes the dangerous race condition of two readers
273  * - reader A deactivates buffer switching, starts reading buffer0
274  * - reader B (again) deactivates buffer switching, starts reading buffer0
275  * - reader A finished with reading, activates buffer switching already, which is too early
276  * here the correct double buffering implementation would need to wait on reader B
277  * - firmware writes into buffer1 and when done, switches buffers
278  * the writing may have started earlier (e.g. before reader A started reading), important here is
279  * only buffer switch at end
280  * - firmware writes into buffer0 and corrupts data
281  * - reader B finishes reading, and gets corrupt data, enables buffer switching.
282  */
283 
284  // wait on "reader B has started" and then wait on "reader A has finished" inside reader B
285 
286  std::thread readerA([&] {
287  auto accessor = d.getOneDRegisterAccessor<uint32_t>("/doubleBuffer");
288  // begin read
289  doubleBufDummy->blockNextRead[0] = true;
290  accessor.read();
291  });
292  std::thread readerB([&] {
293  auto accessor = d.getOneDRegisterAccessor<uint32_t>("/doubleBuffer");
294  // wait that readerA is in blocked double-buffer read
295  doubleBufDummy->blockedInRead[0].wait();
296  // begin read
297  doubleBufDummy->blockNextRead[1] = true;
298  accessor.read();
299  });
300  doubleBufDummy->blockedInRead[1].wait(); // wait that reader B also in blocked read
301  doubleBufDummy->unblockRead[0].wait(); // this is for reader A
302  readerA.join();
303 
304  // check that after reader A returned, buffer switching is still disabled
305  doubleBufferingEnabled->readLatest();
306  BOOST_CHECK(!doubleBufferingEnabled->accessData(0));
307 
308  doubleBufDummy->unblockRead[1].wait(); // this is for reader B
309  // check that after reader B returned, buffer switching is enabled
310  readerB.join();
311  doubleBufferingEnabled->readLatest();
312  BOOST_CHECK(doubleBufferingEnabled->accessData(0));
313 }

380  void checkExtractedChannels(std::string readerAReg, std::string readerBReg) {
381  /*
382  * test access to extracted channels of multiplexed 2D region
383  * this is an application of concurrent readers
384  */
385 
386  writingBufferNum->accessData(0) = 1;
387  writingBufferNum->write();
388 
389  float modulation = 4.2; // example data series 1
390  float correction = 10.1; // example data series 2
391  buf0->accessData(3, 0) = modulation;
392  buf1->accessData(3, 0) = 2 * modulation;
393  buf0->accessData(1, 0) = correction;
394  buf1->accessData(1, 0) = 2 * correction;
395  buf0->write();
396  buf1->write();
397 
398  boost::barrier waitForBufferSwap{3};
399 
400  // since BOOST.Test is not thread-safe, need to buffer check results in main thread
401  bool readerA_ok1, readerA_ok2, readerB_ok1, readerB_ok2;
402 
403  std::thread readerA([&] {
404  auto accessorA = d.getOneDRegisterAccessor<float>(readerAReg);
405  accessorA.readLatest();
406  readerA_ok1 = abs(accessorA[0] - modulation) < 1e-4;
407  waitForBufferSwap.wait();
408  waitForBufferSwap.wait();
409  accessorA.readLatest();
410  readerA_ok2 = abs(accessorA[0] - 2 * modulation) < 1e-4;
411  });
412  std::thread readerB([&] {
413  auto accessorB = d.getOneDRegisterAccessor<float>(readerBReg);
414  accessorB.read();
415  readerB_ok1 = abs(accessorB[0] - correction) < 1e-4;
416  waitForBufferSwap.wait();
417  waitForBufferSwap.wait();
418  accessorB.read();
419  readerB_ok2 = abs(accessorB[0] - 2 * correction) < 1e-4;
420  });
421 
422  waitForBufferSwap.wait();
423  writingBufferNum->accessData(0) = 0;
424  writingBufferNum->write();
425  waitForBufferSwap.wait();
426 
427  readerA.join();
428  readerB.join();
429  BOOST_CHECK(readerA_ok1);
430  BOOST_CHECK(readerA_ok2);
431  BOOST_CHECK(readerB_ok1);
432  BOOST_CHECK(readerB_ok2);
433 
434  // Check that also reading from a TransferGroup works
435  TransferGroup tg;
436  auto accessorA = d.getOneDRegisterAccessor<float>(readerAReg);
437  auto accessorB = d.getOneDRegisterAccessor<float>(readerBReg);
438  tg.addAccessor(accessorA);
439  tg.addAccessor(accessorB);
440  tg.read();
441  BOOST_CHECK_CLOSE(accessorA[0], 2 * modulation, 1e-4);
442  BOOST_CHECK_CLOSE(accessorB[0], 2 * correction, 1e-4);

446  tg.read();
