119  template<typename UserType, typename DataConverterType, bool isRaw>
120  void NumericAddressedPrePostActionsImplementor<UserType, DataConverterType, isRaw>::doPostRead() {
121  callForRawType(_registerInfo->getDataDescriptor().rawDataType(), [this](auto t) {
122  typedef decltype(t) RawType;
123  auto itsrc = (RawType*)_rawAccessor->begin(_startAddress);
124  _dataConverter.template vectorToCooked<UserType>(itsrc, itsrc + _buffer_2D[0].size(), _buffer_2D[0].begin());
125  });
126  }

128  template<typename UserType, typename DataConverterType, bool isRaw>
129  void NumericAddressedPrePostActionsImplementor<UserType, DataConverterType, isRaw>::doPreWrite() {
130  if(_isNotWriteable) {
131  throw ChimeraTK::logic_error(
132  "NumericAddressedBackend: Writeing to a non-writeable register is not allowed (Register name: " +
133  _registerInfo->getRegisterName() + ").");
134  }
135  callForRawType(_registerInfo->getDataDescriptor().rawDataType(), [this](auto t) {
136  typedef decltype(t) RawType;
137  auto itsrc = (RawType*)_rawAccessor->begin(_startAddress);
138  for(auto itdst = _buffer_2D[0].begin(); itdst != _buffer_2D[0].end(); ++itdst) {
139  *itsrc = _dataConverter.template toRaw<UserType>(*itdst);
140  ++itsrc;
141  }
142  });
143  }

196  template<typename UserType, typename DataConverterType, bool isRaw>
197  class NumericAddressedBackendRegisterAccessor : public NDRegisterAccessor<UserType> {
198  public:
199  NumericAddressedBackendRegisterAccessor(boost::shared_ptr<DeviceBackend> dev, const RegisterPath& registerPathName,
200  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
201  : NDRegisterAccessor<UserType>(registerPathName, flags), _dataConverter(registerPathName),
202  _registerPathName(registerPathName), _numberOfWords(numberOfWords),
203  _prePostActionsImplementor(
204  NDRegisterAccessor<UserType>::buffer_2D, _rawAccessor, _startAddress, _dataConverter, _isNotWriteable) {
205  // check for unknown flags
206  flags.checkForUnknownFlags({AccessMode::raw});
207 
208  // check device backend
209  _dev = boost::dynamic_pointer_cast<NumericAddressedBackend>(dev);
210  if(!_dev) {
211  throw ChimeraTK::logic_error("NumericAddressedBackendRegisterAccessor "
212  "is used with a backend which is not "
213  "a NumericAddressedBackend.");
214  }
215 
216  // obtain register information
217  boost::shared_ptr<RegisterInfo> info = _dev->getRegisterInfo(registerPathName);
218  _prePostActionsImplementor._registerInfo = info;
219  _registerInfo = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(info);
220  _bar = _registerInfo->bar;
221  _startAddress = _registerInfo->address + wordOffsetInRegister * _registerInfo->nBytesPerElement();
222 
223  // check number of words
224  if(_registerInfo->dataType == RegisterInfoMap::RegisterInfo::VOID) {
225  // in void registers we always create one element
226  if(_numberOfWords == 0) {
227  _numberOfWords = 1;
228  }
229  if(_numberOfWords > 1) {
230  throw ChimeraTK::logic_error(
231  "Requested number of words is larger than 1 in VOID register '" + _registerPathName + "'!");
232  }
233  if(wordOffsetInRegister > 0) {
234  throw ChimeraTK::logic_error("No offset allowed in VOID register '" + _registerPathName + "'!");
235  }
236  }
237  else { // do the regular consistency check
238  if(_numberOfWords == 0) {
239  _numberOfWords = _registerInfo->getNumberOfElements();
240  }
241  if(_numberOfWords + wordOffsetInRegister > _registerInfo->getNumberOfElements()) {
242  throw ChimeraTK::logic_error(
243  "Requested number of words exceeds the size of the register '" + _registerPathName + "'!");
244  }
245  if(wordOffsetInRegister >= _registerInfo->getNumberOfElements()) {
246  throw ChimeraTK::logic_error("Requested offset exceeds the size of the register'" + _registerPathName + "'!");
247  }
248  }
249 
250  // Cache writeability
251  _isNotWriteable = !_registerInfo->isWriteable();
252 
253  // create low-level transfer element handling the actual data transfer to
254  // the hardware with raw data
255  _rawAccessor.reset(new NumericAddressedLowLevelTransferElement(
256  _dev, _bar, _startAddress, _numberOfWords * _registerInfo->nBytesPerElement()));
257 
258  // allocated the buffers
259  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
260  NDRegisterAccessor<UserType>::buffer_2D[0].resize(_numberOfWords);
261 
262  // We don't have to fill it in a special way if the accessor is raw
263  // because we have an overloaded, more efficient implementation
264  // in this case. So we can use it in setAsCooked() and getAsCooked()
265  _dataConverter = detail::createDataConverter<DataConverterType>(_registerInfo);
266 
267  if(flags.has(AccessMode::raw)) {
268  if(DataType(typeid(UserType)) != _registerInfo->getDataDescriptor().rawDataType()) {
269  throw ChimeraTK::logic_error("Given UserType when obtaining the NumericAddressedBackendRegisterAccessor in "
270  "raw mode does not match the expected type. Use an " +
271  _registerInfo->getDataDescriptor().rawDataType().getAsString() +
272  " instead! (Register name: " + _registerPathName + "')");
273  }
274  // FIXME: this has to move to the creation
275  //isRaw = true;
276  }
277 
278  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(getAsCooked_impl);
279  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(setAsCooked_impl);
280  }

291  if(!_dev->isOpen()) return; // do not delegate if exception was thrown by us in doPreWrite

401  void replaceTransferElement(boost::shared_ptr<TransferElement> newElement) override {
402  auto casted = boost::dynamic_pointer_cast<NumericAddressedLowLevelTransferElement>(newElement);
403  if(casted && casted->isMergeable(_rawAccessor)) {
404  size_t newStartAddress = std::min(casted->_startAddress, _rawAccessor->_startAddress);
405  size_t newStopAddress = std::max(
406  casted->_startAddress + casted->_numberOfBytes, _rawAccessor->_startAddress + _rawAccessor->_numberOfBytes);
407  size_t newNumberOfBytes = newStopAddress - newStartAddress;
408  casted->changeAddress(newStartAddress, newNumberOfBytes);
409  _rawAccessor = casted;
410  }
411  _rawAccessor->setExceptionBackend(this->_exceptionBackend);
