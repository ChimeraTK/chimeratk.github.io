16  struct ReferenceCountedUniqueLock {
17  ReferenceCountedUniqueLock() = default;
18 
19  ReferenceCountedUniqueLock(std::recursive_mutex& mutex, int* useCounter)
20  : _lock(mutex, std::defer_lock), _targetUseCount(useCounter) {}
21 
22  void lock() {
23  _lock.lock();
24  *_targetUseCount = *_targetUseCount + 1;
25  }
26 
27  void unlock() {
28  *_targetUseCount = *_targetUseCount - 1;
29  _lock.unlock();
30  }
31 
32  [[nodiscard]] int useCount() const {
33  assert(_lock.owns_lock());
34  return *_targetUseCount;
35  }
36 
37  std::unique_lock<std::recursive_mutex> _lock;
38  int* _targetUseCount{nullptr};
39  };

43  // From https://stackoverflow.com/questions/1392059/algorithm-to-generate-bit-mask
44  constexpr uint64_t getMaskForNBits(uint64_t numberOfBits) {
45  // Prevent warning about undefined behavior if shifting right by 64 bit below
46  if(numberOfBits == 0) {
47  return 0;
48  }
49 
50  return (static_cast<uint64_t>(-(numberOfBits != 0)) &
51  (static_cast<uint64_t>(-1) >> ((sizeof(uint64_t) * CHAR_BIT) - numberOfBits)));
52  }

57  struct BitRangeAccessPluginDecorator : ChimeraTK::NDRegisterAccessorDecorator<UserType, TargetType> {

60  /********************************************************************************************************************/
61  BitRangeAccessPluginDecorator(boost::shared_ptr<LogicalNameMappingBackend>& backend,
62  const boost::shared_ptr<ChimeraTK::NDRegisterAccessor<TargetType>>& target, const std::string& name,
63  uint64_t shift, uint64_t numberOfBits)
64  : ChimeraTK::NDRegisterAccessorDecorator<UserType, TargetType>(target), _shift(shift),
65  _numberOfBits(numberOfBits), _writeable{_target->isWriteable()} {
66  if(_target->getNumberOfChannels() > 1 || _target->getNumberOfSamples() > 1) {
67  throw ChimeraTK::logic_error("LogicalNameMappingBackend BitRangeAccessPluginDecorator: " +
68  TransferElement::getName() + ": Cannot target non-scalar registers.");
69  }
70 
71  auto& map = boost::fusion::at_key<TargetType>(backend->sharedAccessorMap.table);
72  RegisterPath path{name};
73  path.setAltSeparator(".");
74  LogicalNameMappingBackend::AccessorKey key{backend.get(), path};
75 
76  auto it = map.find(key);
77  if(it != map.end()) {
78  _lock = {it->second.mutex, &(it->second.useCount)};
79  }
80  else {
81  assert(false);
82  }
83 
84  _baseBitMask = getMaskForNBits(_numberOfBits);
85  _maskOnTarget = _baseBitMask << _shift;
86  }

90  void doPreRead(TransferType type) override {
91  _lock.lock();
92 
93  _target->preRead(type);
94  }

97 
98  void doPostRead(TransferType type, bool hasNewData) override {
99  auto unlock = cppext::finally([this] { this->_lock.unlock(); });
100  _target->postRead(type, hasNewData);
101  if(!hasNewData) return;
102  auto validity = _target->dataValidity();
103 
104  if constexpr(!std::is_integral<UserType>::value) {
105  // This should never be reached, the decorate function should make sure that we only have integral types here
106  assert(false);
107  }
108  else {
109  uint64_t v{};
110  static_assert(sizeof(TargetType) <= sizeof(uint64_t), "Target data type too big.");
111 
112  auto targetValue = _target->accessData(0);
113  memcpy(std::addressof(v), std::addressof(targetValue), sizeof(TargetType));
114  v = (v & _maskOnTarget) >> _shift;
115 
116  // There are bits set outside of the range of the UserType
117  // Clamping according to B.2.4 and setting the faulty flag
118  // FIXME: Probably easier once the FixpointConverter is supporting 64bit raw types
119  if((v & ~_userTypeMask) != 0) {
120  v = std::numeric_limits<UserType>::max();
121  validity = DataValidity::faulty;
122  }
123  memcpy(buffer_2D[0].data(), &v, sizeof(UserType));
124  }
125 
126  this->_versionNumber = std::max(this->_versionNumber, _target->getVersionNumber());
127  this->_dataValidity = validity;
128  }
129 
130  /********************************************************************************************************************/
131 
132  void doPreWrite(TransferType type, VersionNumber versionNumber) override {
133  _lock.lock();
134 
135  if(!_writeable) {
136  throw ChimeraTK::logic_error(
137  "Register \"" + TransferElement::getName() + "\" with BitRange plugin is not writeable.");
138  }
139 
140  uint64_t value{};
141  if constexpr(!std::is_integral<UserType>::value) {
142  // This should never be reached, the decorate function should make sure that we only have integral types here
143  assert(false);
144  }
145  else {
146  memcpy(&value, buffer_2D[0].data(), sizeof(UserType));
147  }
148 
149  // We have received more data than we actually have bits for
150  if((value & ~_baseBitMask) != 0) {
151  this->_dataValidity = DataValidity::faulty;
152  value = _baseBitMask;
153  }
154  else {
155  this->_dataValidity = DataValidity::ok;
156  }
157 
158  // When in a transfer group, only the first accessor to write to the _target can call read() in its preWrite()
159  // Otherwise it will overwrite the
160  if(_target->isReadable() && (!TransferElement::_isInTransferGroup || _lock.useCount() == 1)) {
161  _target->read();
162  }
163 
164  _target->accessData(0) &= ~_maskOnTarget;
165  _target->accessData(0) |= (value << _shift);
166 
167  _temporaryVersion = std::max(versionNumber, _target->getVersionNumber());
168  _target->setDataValidity(this->_dataValidity);
169  _target->preWrite(type, _temporaryVersion);
170  }
171

173 
174  void doPostWrite(TransferType type, VersionNumber /*versionNumber*/) override {
175  auto unlock = cppext::finally([this] { this->_lock.unlock(); });
176  _target->postWrite(type, _temporaryVersion);
177  }

181  void replaceTransferElement(boost::shared_ptr<ChimeraTK::TransferElement> newElement) override {
182  auto casted = boost::dynamic_pointer_cast<BitRangeAccessPluginDecorator<UserType, TargetType>>(newElement);
183 
184  // In a transfer group, we are trying to replaced with an accessor. Check if this accessor is for the
185  // same target and not us and check for overlapping bit range afterwards. If they overlap, switch us and
186  // the replacement read-only which switches the transfergroup read-only since we cannot guarantee the write order
187  // for overlapping bit ranges
188  if(casted && casted.get() != this && casted->_target == _target) {
189  if((casted->_maskOnTarget & _maskOnTarget) != 0) {
190  casted->_writeable = false;
191  _writeable = false;
192  }
193  }
194  NDRegisterAccessorDecorator<UserType, TargetType>::replaceTransferElement(newElement);
195  }

202  uint64_t _userTypeMask{getMaskForNBits(sizeof(UserType) * CHAR_BIT)};
203  uint64_t _targetTypeMask{getMaskForNBits(sizeof(TargetType) * CHAR_BIT)};
204  uint64_t _baseBitMask;

208  bool _writeable{false};

210  using ChimeraTK::NDRegisterAccessorDecorator<UserType, TargetType>::_target;
211  };

215  BitRangeAccessPlugin::BitRangeAccessPlugin(
216  const LNMBackendRegisterInfo& info, size_t pluginIndex, const std::map<std::string, std::string>& parameters)
217  : AccessorPlugin<BitRangeAccessPlugin>(info, pluginIndex, true) {
218  try {
219  const auto& shift = parameters.at("shift");
220 
221  // This is how you are supposed to use std::from_chars with std::string
222  // NOLINTNEXTLINE(unsafe-buffer-usage)
223  auto [suffix, ec]{std::from_chars(shift.data(), shift.data() + shift.size(), _shift)};
224  if(ec != std::errc()) {
225  throw ChimeraTK::logic_error("LogicalNameMappingBackend BitRangeAccessPlugin: " + info.getRegisterName() +
226  R"(: Unparseable parameter "shift".)");
227  }
228  }
229  catch(std::out_of_range&) {
230  throw ChimeraTK::logic_error("LogicalNameMappingBackend BitRangeAccessPlugin: " + info.getRegisterName() +
231  R"(: Missing parameter "shift".)");
232  }
233 
234  try {
235  const auto& numberOfBits = parameters.at("numberOfBits");
236  // This is how you are supposed to use std::from_chars with std::string
237  // NOLINTNEXTLINE(unsafe-buffer-usage)
238  auto [suffix, ec]{std::from_chars(numberOfBits.data(), numberOfBits.data() + numberOfBits.size(), _numberOfBits)};
239  if(ec != std::errc()) {
240  throw ChimeraTK::logic_error("LogicalNameMappingBackend BitRangeAccessPlugin: " + info.getRegisterName() +
241  R"(: Unparseable parameter "numberOfBits".)");
242  }
243  }
244  catch(std::out_of_range&) {
245  throw ChimeraTK::logic_error("LogicalNameMappingBackend BitRangeAccessPlugin: " + info.getRegisterName() +
246  R"(: Missing parameter "numberOfBits".)");
247  }
248  }
