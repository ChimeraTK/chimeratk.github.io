72  void MathPlugin::openHook(boost::shared_ptr<LogicalNameMappingBackend> backend) {
73  // make sure backend catalogue is updated with target backend information
74  backend->getRegisterCatalogue();
75 
76  // check if this is the first call to openHook: perform tasks which could not be done in the constructor because
77  // the backend is not yet available there.
78  if(_backend._empty()) {
79  // make sure the register info is up to date
80  updateRegisterInfo(backend->_catalogue_mutable);
81 
82  // store backend as weak pointer for later use
83  _backend = backend;
84 
85  // If write direction, check for push-type parameters if enabled
86  if(_isWrite && _enablePushParameters) {
87  for(auto& parpair : _parameters) {
88  auto paramFlags = backend->getRegisterCatalogue().getRegister(parpair.second).getSupportedAccessModes();
89  if(paramFlags.has(AccessMode::wait_for_new_data)) {
90  _hasPushParameter = true;
91  break;
92  }
93  }
94 
95  // perpare for updates via push-type parameters
96  if(_hasPushParameter) {
97  // fill the _pushParameterReadGroup
98  for(auto& parpair : _parameters) {
99  // push-type parameters need to be obtained with wait_for_new_data, others without
100  AccessModeFlags flags{};
101  auto paramFlags = backend->getRegisterCatalogue().getRegister(parpair.second).getSupportedAccessModes();
102  if(paramFlags.has(AccessMode::wait_for_new_data)) {
103  flags = {AccessMode::wait_for_new_data};
104  }
105  auto acc = backend->getRegisterAccessor<double>(parpair.second, 0, 0, flags);
106  if(acc->getNumberOfChannels() != 1) {
107  throw ChimeraTK::logic_error(
108  "The LogicalNameMapper MathPlugin supports only scalar or 1D array registers. Register name: '" +
109  _info.name + "', parameter name: '" + parpair.first + "'");
110  }
111  _pushParameterReadGroup.add(acc);
112  _pushParameterAccessorMap[parpair.first] = acc;
113  }
114  _pushParameterReadGroup.finalise();
115  _lastWrittenValue.resize(_info.length);
116  }
117 
118  // compile formula
119  try {
120  _lastWrittenValue.resize(_info.length);
121  _h->compileFormula(_formula, _backend.lock(), _pushParameterAccessorMap, _info.length);
122  }
123  catch(...) {
124  // Do not throw errors at this point, only when accessing the register directly
125  _hasPushParameter = false;
126  }
127  }
128  }
129 
130  // The main value has to be written once before the parameter thread will write anything,
131  // and all parameters have to be written before the accessor itself will write to the device.
132  // Set this before launching the thread, so we don't have to fiddle with the mutex here.
133  _mainValueWrittenAfterOpen = false;
134  _allParametersWrittenAfterOpen = false;
135 
136  // if we have push-type parameters triggering a write of the target, start the _pushParameterWriteThread
137  if(_hasPushParameter) {
138  assert(_isWrite); // we do not set hasPushParameter if !isWrite
139 
140  // If the thread is still running, it will terminate soon by itself. We must not attempt to terminate it here,
141  // because it would be subject to the race condition that the thread still sees an exception from before recovery
142  // which results in putting the device back into exception state.
143  // The reason why we might need to wait here is that we cannot wait for the thread to terminate in the exception
144  // hook, because it might be called from inside the thread.
145  while(_pushParameterWriteThread.joinable()) {
146  _pushParameterReadGroup.interrupt();
147  _pushParameterWriteThread.try_join_for(boost::chrono::milliseconds(100));
148  }
149

151  _pushParameterWriteThread = boost::thread([&] { parameterReadLoop(); });
152 
153  // do not proceed before the thread is ready to receive new data
154  _waitUntilParameterThreadLaunched.wait();

160  void MathPlugin::parameterReadLoop() {
161  boost::shared_ptr<NDRegisterAccessor<double>> target;
162  { // scope for the barrier wait caller
163  // Use finally to ensure that the barrier::wait is always called, even if the thread kicks out with an exception
164  auto barrierWaitCaller = cppext::finally([&] { _waitUntilParameterThreadLaunched.wait(); });
165 
166  // obtain target accessor
167  auto targetDevice = BackendFactory::getInstance().createBackend(_info.deviceName);
168  target = targetDevice->getRegisterAccessor<double>(_info.registerName, _info.length, _info.firstIndex, {});
169  // empty all queues (initial values, remaining exceptions from previous thread runs). Ignore all exceptions.
170  while(true) {
171  auto notfy = _pushParameterReadGroup.waitAnyNonBlocking();
172  if(!notfy.isReady()) break;
173  try {
174  notfy.accept();
175  }
176  catch(...) {
177  continue;
178  }
179  }
180  assert(!_backend.lock()->_asyncReadActive);
181  assert(_lastWrittenValue.size() == target->getNumberOfSamples());
182 
183  } // end scope of the barrierWaitCaller. At this point the finally is executed and other thread is notified that
184  // this thread has started and the opend hook can complete.
185 
186  // start processing loop
187  while(true) {
188  try {
189  _pushParameterReadGroup.readAny();
190  {
191  std::unique_lock<std::recursive_mutex> lk(_writeMutex);
192  if(!checkAllParametersWritten(_pushParameterAccessorMap)) {
193  continue;
194  }
195  if(!_mainValueWrittenAfterOpen) {
196  continue;
197  }
198 
199  _h->computeResult(_lastWrittenValue, target->accessChannel(0));
200  target->writeDestructively();
201  }
202  }
203  catch([[maybe_unused]] ChimeraTK::runtime_error& e) {
204  // no need to report the exception, as the accessor should have already done it
205  return;
206  }
207  }

235  bool MathPlugin::checkAllParametersWritten(
236  std::map<std::string, boost::shared_ptr<NDRegisterAccessor<double>>> const& accessorsMap) {
237  if(_allParametersWrittenAfterOpen == true) {
238  return true;
239  }
240 
241  // we can assign temporary values to _allParametersWrittenAfterOpen because it's protected by the mutex
242  // (must be locked before calling this function)
243  _allParametersWrittenAfterOpen = true;
244  auto backend = _backend.lock(); // No need to check the lock of the weak pointer, it cannot fail.

246  // shared pointer of it self.

350  template<typename UserType>
351  void MathPluginDecorator<UserType>::doPreWrite(TransferType type, ChimeraTK::VersionNumber versionNumber) {
352  if(!_p->_isWrite) {
353  throw ChimeraTK::logic_error("This register with MathPlugin enabled is not writeable: " + _target->getName());
354  }
355 
356  // The readLatest might throw an exception. In this case preWrite() is never delegated and we must not call the target's portWrite().
357  _skipWriteDelegation = true;
358 
359  // update parameters
360  for(auto& p : h.params) p.first->readLatest();
361 
362  // convert from UserType to double - use the target accessor's buffer as a temporary buffer (this is a bit a hack,
363  // but it is safe to overwrite the buffer and we can avoid the need for an additional permanent buffer which might
364  // not even be used if the register is never written).
365  for(size_t k = 0; k < buffer_2D[0].size(); ++k) {
366  _target->accessData(0, k) = userTypeToNumeric<double>(buffer_2D[0][k]);
367  }
368 
369  // update last written data buffer for updater thread if needed
370  if(_p->_hasPushParameter) {
371  // Accquire the lock and hold it until the transaction is completed in postWrite.
372  // This is save because it is guaranteed by the frameworn that pre- and post actions are called in pairs.
373  _p->_writeMutex.lock();
374  _p->_lastWrittenValue = _target->accessChannel(0);
375  _p->_mainValueWrittenAfterOpen = true; // We have stored the value for the parameters thread,
376  // even if it's not written yet because not all parameters are there.
377 
378  // Note: There is a potential race condition that the parameter thread has not processed a parameter yet but the poll
379  // registers here have already seen it and could in principle run. However, this would lead to inconsistent behaviour
380  // because sometimes writing all parameters and then the accessor itself exactly once after opening would lead to
381  // two writes, because eventually the thread will also write.
382  // So there are two reasons we must wait for the flag from the thread
383  // 1. Ensure a consistent number of write operations
384  // 2. We cannot determine the correct version number from the poll-type accessors anyway because they always get
385  // a fresh version number. (We could use read_latest on push-type, but this would be less efficient,
386  // and still leave point 1.)
387  if(!_p->_allParametersWrittenAfterOpen) {
388  return;
389  }
390  }
391 
392  // There either are onyl poll-type parameters, or all push-type parameters have been received.
393  // We are OK to go through with the transfer
394  _skipWriteDelegation = false;
395 
396  // evaluate the expression and store into target accessor
397  h.computeResult(_target->accessChannel(0), _target->accessChannel(0));
398 
399  // pass validity to target and delegate preWrite
400  // FIXME: #9622 The data validity of the parameters should be considered
401  _target->setDataValidity(this->_dataValidity);
402  _target->preWrite(type, versionNumber);
403  }

426  void MathPluginDecorator<UserType>::doPostWrite(TransferType type, ChimeraTK::VersionNumber versionNumber) {
427  if(_skipWriteDelegation && (this->_activeException != nullptr)) {
428  // Something has thrown before the target's preWrite was called. Re-throw it here.
429  // Do not unlock the mutex. It never has been locked.
430  std::rethrow_exception(this->_activeException);

432 
433  // make sure the mutex is released, even if the delegated postWrite kicks out with an exception
434  auto _ = cppext::finally([&] {
435  if(_p->_hasPushParameter) {
436  _p->_writeMutex.unlock();
437  }

497  void MathPluginFormulaHelper::compileFormula(const std::string& formula,
498  const boost::shared_ptr<LogicalNameMappingBackend>& backend,
499  const std::map<std::string, boost::shared_ptr<ChimeraTK::NDRegisterAccessor<double>>>& parameters,
500  size_t nElements) {
501  // create exprtk parser
502  exprtk::parser<double> parser;
503 
504  // add basic constants like pi
505  symbols.add_constants();
506 
507  // Create vector view for the value and add it to the symbol table. We need to use a vector view instead of adding
508  // the buffer directly as a vector, since our buffers might be swapped and hence the address of the data can
509  // change. The pointer used for the view is for now to a temporary vector and will become invalid once this function
510  // returns. This is acceptable, since before using the view the pointer will be updated to the right buffer.
511  // Using a nullptr instead of the temporary buffer does not work, as the buffer seems to be accessible during
512  // compilation of the formula.
513  std::vector<double> temp(nElements);
514  valueView = std::make_unique<exprtk::vector_view<double>>(exprtk::make_vector_view(temp, nElements));
515  symbols.add_vector("x", *valueView);
516 
517  // iterate parameters, add all but 'formula' parameter as a variable.
518  for(auto& parpair : parameters) {
519  if(parpair.first == "formula") continue;
520  auto& acc = parpair.second;
521  acc->setExceptionBackend(backend);
522  if(acc->getNumberOfChannels() != 1) {
523  throw ChimeraTK::logic_error(
524  "The LogicalNameMapper MathPlugin supports only scalar or 1D array registers. Register name: '" + varName +
525  "', parameter name: '" + parpair.first + "'");
526  }
527  auto view = std::make_unique<exprtk::vector_view<double>>(
528  exprtk::make_vector_view(acc->accessChannel(0), acc->getNumberOfSamples()));
529  symbols.add_vector(parpair.first, *view);
530  params[acc] = std::move(view);
531  }
532 
533  // compile the expression
534  expression.register_symbol_table(symbols);
535  bool success = parser.compile(formula, expression);
536  if(!success) {
537  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + varName +
538  "': failed to compile expression '" + formula + "': " + parser.error());
539  }
540  }

544  template<typename T>
545  void MathPluginFormulaHelper::computeResult(std::vector<double>& x, std::vector<T>& resultBuffer) {
546  // inform the value view of the new data pointer - the buffer might have been swapped
547  valueView->rebase(x.data());
548 
549  // update parameter buffers
550  for(auto& p : params) {
551  p.second->rebase(p.first->accessChannel(0).data());
552  }
553 
554  // evaluate the expression, obtain the result in a way so it also works when using the return statement
555  double valueWhenNotUsingReturn = expression.value();
556  exprtk::results_context<double> results = expression.results();
557 
558  // extract result depending on how it was returned
559  if(results.count() == 0) {
560  // if results.count() is 0, the return statement presumably has not been used
561  if(resultBuffer.size() != 1) {
562  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + varName +
563  "': The expression returns a scalar but " + std::to_string(resultBuffer.size()) + " expected.");
564  }
565  resultBuffer[0] = numericToUserType<T>(valueWhenNotUsingReturn);
566  }
567  else if(results.count() == 1) {
568  // return statement has been used to return exactly one value (note: this value might be an array)
569  exprtk::type_store<double> result = results[0];
570 
571  // make sure we got a numeric result
572  if(result.type != exprtk::type_store<double>::e_scalar && result.type != exprtk::type_store<double>::e_vector) {
573  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + varName +
574  "': The expression did not return a numeric result.");
575  }
576 
577  // create vector view and check that its size matches
578  exprtk::type_store<double>::type_view<double> view(result);
579  if(view.size() != resultBuffer.size()) {
580  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + varName +
581  "': The expression returns " + std::to_string(view.size()) + " elements but " +
582  std::to_string(resultBuffer.size()) + " expected.");
583  }
584 
585  // convert and copy data into target buffer
586  for(size_t k = 0; k < view.size(); ++k) {
587  resultBuffer[k] = numericToUserType<T>(view[k]);
588  }
589  }
590  else {
591  // multiple results in return statement are unexpected
592  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + varName +
593  "': The expression returned " + std::to_string(results.count()) + " results, expect exactly one result.");
594  }
595  }
