43  void ApplicationModule::run() {
44  // start the module thread
45  assert(!moduleThread.joinable());
46  moduleThread = boost::thread(&ApplicationModule::mainLoopWrapper, this);
47  //Wait until the thread has launched and acquired and released the testable mode lock at least once.
48  if(Application::getInstance().isTestableModeEnabled()) {
49  while(!testableModeReached) {
50  Application::getInstance().testableModeUnlock("releaseForReachTestableMode");
51  usleep(100);
52  Application::getInstance().testableModeLock("acquireForReachTestableMode");
53  }
54  }
55  }

59  void ApplicationModule::terminate() {
60  if(moduleThread.joinable()) {
61  moduleThread.interrupt();
62  // try joining the thread
63  while(!moduleThread.try_join_for(boost::chrono::milliseconds(10))) {
64  // if thread is not yet joined, send interrupt() to all variables.
65  for(auto& var : getAccessorListRecursive()) {
66  var.getAppAccessorNoType().getHighLevelImplElement()->interrupt();
67  }
68  // it may not suffice to send interrupt() once, as the exception might get
69  // overwritten in the queue, thus we repeat this until the thread was
70  // joined.
71  }
72  }
73  assert(!moduleThread.joinable());
74  }

82  void ApplicationModule::mainLoopWrapper() {
83  Application::registerThread("AM_" + getName());
84 
85  // Acquire testable mode lock, so from this point on we are running only one user thread concurrently
86  Application::testableModeLock("start");
87 
88  // Read all variables once to obtain the initial values from the devices and from the control system persistency
89  // layer. This is done in two steps, first for all poll-type variables and then for all push-types, because
90  // poll-type reads might trigger distribution of values to push-type variables via a ConsumingFanOut.
91  for(auto& variable : getAccessorListRecursive()) {
92  if(variable.getDirection().dir != VariableDirection::consuming) continue;
93  if(variable.getMode() == UpdateMode::poll) {
94  auto hasInitialValue = variable.hasInitialValue();
95  if(hasInitialValue != VariableNetworkNode::InitialValueMode::None) {
96  variable.getAppAccessorNoType().readLatest();
97  }
98  }
99  }
100  for(auto& variable : getAccessorListRecursive()) {
101  if(variable.getDirection().dir != VariableDirection::consuming) continue;
102  if(variable.getMode() == UpdateMode::push) {
103  auto hasInitialValue = variable.hasInitialValue();
104  if(hasInitialValue == VariableNetworkNode::InitialValueMode::Poll) {
105  variable.getAppAccessorNoType().readLatest();
106  }
107  else if(hasInitialValue == VariableNetworkNode::InitialValueMode::Push) {
108  variable.getAppAccessorNoType().read();
109  }
110  }
111  }
112 
113  // We are holding the testable mode lock, so we are sure the mechanism will work now.
114  testableModeReached = true;
115 
116  // enter the main loop
117  mainLoop();
118  Application::testableModeUnlock("terminate");
119  }
