34  void addSlave(boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> slave, VariableNetworkNode&) override {
35  // check if array shape is compatible, unless the receiver is a trigger
36  // node, so no data is expected
37  if(slave->getNumberOfSamples() != 0 &&
38  (slave->getNumberOfChannels() != 1 || slave->getNumberOfSamples() != this->getNumberOfSamples())) {
39  std::string what = "FeedingFanOut::addSlave(): Trying to add a slave '" + slave->getName();
40  what += "' with incompatible array shape! Name of fan out: '" + this->getName() + "'";
41  throw ChimeraTK::logic_error(what.c_str());
42  }
43 
44  // make sure slave is writeable
45  if(!slave->isWriteable()) {
46  throw ChimeraTK::logic_error("FeedingFanOut::addSlave() has been called "
47  "with a receiving implementation!");
48  }
49 
50  // handle return channels
51  if(_withReturn) {
52  if(slave->isReadable()) {
53  if(_hasReturnSlave) {
54  throw ChimeraTK::logic_error("FeedingFanOut: Cannot add multiple slaves with return channel!");
55  }
56  _hasReturnSlave = true;
57  _returnSlave = slave;
58  }
59  }
60

63  }

92  void doPostRead() override {
93  if(!_withReturn) throw ChimeraTK::logic_error("Read operation called on write-only variable.");
94  assert(_hasReturnSlave);
95  _returnSlave->postRead();
96  _returnSlave->accessChannel(0).swap(ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0]);
97  // distribute return-channel update to the other slaves
98  for(auto& slave : FanOut<UserType>::slaves) { // send out copies to slaves
99  if(slave == _returnSlave) continue;
100  if(slave->getNumberOfSamples() != 0) { // do not send copy if no data is expected (e.g. trigger)
101  slave->accessChannel(0) = ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0];
102  }
103  slave->writeDestructively(_returnSlave->getVersionNumber());
104  }
105  }

112  void doPreWrite() override {
113  for(auto& slave : FanOut<UserType>::slaves) { // send out copies to slaves
114  if(slave->getNumberOfSamples() != 0) { // do not send copy if no data is expected (e.g. trigger)
115  if(slave == FanOut<UserType>::slaves.front()) { // in case of first slave, swap instead of copy
116  slave->accessChannel(0).swap(ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0]);
117  }
118  else { // not the first slave: copy the data from the first slave
119  slave->accessChannel(0) = FanOut<UserType>::slaves.front()->accessChannel(0);
120  }
121  }
122  slave->setDataValidity(this->dataValidity());
123  }
124  // pre write may only be called on the target accessors after we have filled
125  // them all, otherwise the first accessor might take us the data away...
126  for(auto& slave : FanOut<UserType>::slaves) {
127  slave->preWrite();
128  }
129  }
