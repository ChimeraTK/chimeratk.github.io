271  template<typename UserType>
272  VariableNetworkNode VariableNetworkNode::makeConstant(bool makeFeeder, UserType value, size_t length) {
273  VariableNetworkNode node;
274  node.pdata = boost::make_shared<VariableNetworkNode_data>();
275  node.pdata->constNode.reset(new ConstantAccessor<UserType>(value, length));
276  node.pdata->type = NodeType::Constant;
277  node.pdata->valueType = &typeid(UserType);
278  node.pdata->nElements = length;
279  node.pdata->name = "*UNNAMED CONSTANT*";
280  if(makeFeeder) {
281  node.pdata->direction = {VariableDirection::feeding, false};
282  node.pdata->mode = UpdateMode::push;
283  }
284  else {
285  node.pdata->direction = {VariableDirection::consuming, false};
286  node.pdata->mode = UpdateMode::poll;
287  }
288  return node;
289  }
