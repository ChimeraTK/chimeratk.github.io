646  auto acc = getOneDRegisterAccessor<UserType>(registerPathName, numberOfWords, wordOffsetInRegister, flags);

664  template<typename UserType>
665  void Device::write(const RegisterPath& registerPathName, const std::vector<UserType>& vector,
666  size_t wordOffsetInRegister, const AccessModeFlags& flags) {
667  auto acc = getOneDRegisterAccessor<UserType>(registerPathName, vector.size(), wordOffsetInRegister, flags);
668  // the vector stays constant, but we have to work around so we can swap it and
669  // later swap it back...
670  std::vector<UserType>& mutable_vector = const_cast<std::vector<UserType>&>(vector);
671  acc.swap(mutable_vector);
672  acc.write();
673  acc.swap(mutable_vector);
674  }
