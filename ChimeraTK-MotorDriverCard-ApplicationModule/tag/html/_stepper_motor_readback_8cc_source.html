15 void ReadbackHandler::prepare() {
16 // The module starts with the fake device state in FAIL mode since we
17 // did not open the motor yet and know nothing about it
18 moduleError.status = StatusOutput::Status::FAULT;
19 moduleError.message = "Initial bringup of device";
20 moduleError.setCurrentVersionNumber({});
21 incrementDataFaultCounter();
22 moduleError.status.setDataValidity(DataValidity::ok);
23 moduleError.message.setDataValidity(DataValidity::ok);
24 writeAll();
25 decrementDataFaultCounter();
26 }

28 /********************************************************************************************************************/
29
30 void ReadbackHandler::mainLoop() {
31 auto group = ReadAnyGroup({trigger, deviceBecameFunctional});
32
33 // Report an exception once on startup. If the device is already working, we get an initial toggle of
34 // deviceBecameFunctional that way, otherwise we would sit in the wait for deviceBecameFunctional until the first
35 // error happens.
36 _deviceModule->reportException("Bootstrap module startup");
37
38 while(true) {
39 switch(static_cast<StatusOutput::Status>(moduleError.status)) {
40 case StatusOutput::Status::FAULT:
41 group.readUntil(deviceBecameFunctional.getId());
42
43 tryMotorRenew();
44 tryReadingFromMotor();
45 break;
46 case StatusOutput::Status::OK: {
47 group.readUntil(trigger.getId());
48 tryReadingFromMotor();
49 break;
50 }
51 default:
52 assert(false);
53 break;
54 }
55
56 // If we are (still) in FAULT mode after this round of either readout or trying to open the motor
57 // flag everything as invalid and shove it out the door
58 if(moduleError.status == StatusOutput::Status::FAULT) {
59 incrementDataFaultCounter();
60 }
61
62 // Flag the status and message outputs of the module errors as valid
63 moduleError.status.setDataValidity(DataValidity::ok);
64 moduleError.message.setDataValidity(DataValidity::ok);
65 writeAll();
66
67 if(moduleError.status == StatusOutput::Status::FAULT) {
68 decrementDataFaultCounter();
69 }
70 }
71 }

97 void ReadbackHandler::readback() {
98 status.isEnabled = _motor->get()->getEnabled();
99
100 auto calibMode = _motor->get()->getCalibrationMode();
101 auto error = _motor->get()->getError();
102
103 status.calibrationMode = static_cast<int>(calibMode);
104 status.errorId = static_cast<int32_t>(error);
105
106 status.isFullStepping = _motor->get()->isFullStepping();
107 status.autostartEnabled = _motor->get()->getAutostart();
108
109 status.encoderReadoutMode = _motor->get()->getEncoderReadoutMode();
110
111 position.actualValueInSteps = _motor->get()->getCurrentPositionInSteps();
112 position.encoderReadback = _motor->get()->getEncoderPosition();
113 position.targetValueInSteps = _motor->get()->getTargetPositionInSteps();
114
115 // Update values that have a static relation to HW readback
116 position.actualValue = _motor->get()->recalculateStepsInUnits(position.actualValueInSteps);
117 position.targetValue = _motor->get()->recalculateStepsInUnits(position.targetValueInSteps);
118
119 status.isIdle = _motor->get()->isSystemIdle();
120 swLimits.isEnabled = _motor->get()->getSoftwareLimitsEnabled();
121 swLimits.maxPositionInSteps = _motor->get()->getMaxPositionLimitInSteps();
122 swLimits.minPositionInSteps = _motor->get()->getMinPositionLimitInSteps();
123 swLimits.maxPosition = _motor->get()->recalculateStepsInUnits(swLimits.maxPositionInSteps);
124 swLimits.minPosition = _motor->get()->recalculateStepsInUnits(swLimits.minPositionInSteps);
125
126 if(!_motorIsDummy) {
127 currentLimit.userValue = _motor->get()->getUserCurrentLimit(); // Include when this method gets implemented
128 }
129 speedLimit.userValue = _motor->get()->getUserSpeedLimit();
130 }

135 void ReadbackHandler::readEndSwitchData() {
136 if(_motor->get()->hasHWReferenceSwitches()) {
137 negativeEndSwitch.available = true;
138 positiveEndSwitch.available = true;
139
140 negativeEndSwitch.isActive = _motor->get()->isNegativeReferenceActive();
141 positiveEndSwitch.isActive = _motor->get()->isPositiveReferenceActive();
142
143 positiveEndSwitch.positionInSteps = _motor->get()->getPositiveEndReferenceInSteps();
144 negativeEndSwitch.positionInSteps = _motor->get()->getNegativeEndReferenceInSteps();
145 positiveEndSwitch.position = _motor->get()->recalculateStepsInUnits(positiveEndSwitch.positionInSteps);
146 negativeEndSwitch.position = _motor->get()->recalculateStepsInUnits(negativeEndSwitch.positionInSteps);
147
148 positiveEndSwitch.tolerance = _motor->get()->getTolerancePositiveEndSwitch();
149 negativeEndSwitch.tolerance = _motor->get()->getToleranceNegativeEndSwitch();
150 }
151 else {
152 positiveEndSwitch.available = false;
153 negativeEndSwitch.available = false;
154
155 positiveEndSwitch.isActive = 0;
156 negativeEndSwitch.isActive = 0;
157 positiveEndSwitch.isActive.setDataValidity(DataValidity::faulty);
158 negativeEndSwitch.isActive.setDataValidity(DataValidity::faulty);
159
160 positiveEndSwitch.positionInSteps.setDataValidity(DataValidity::faulty);
161 negativeEndSwitch.positionInSteps.setDataValidity(DataValidity::faulty);
162
163 positiveEndSwitch.position.setDataValidity(DataValidity::faulty);
164 negativeEndSwitch.position.setDataValidity(DataValidity::faulty);
165
166 positiveEndSwitch.tolerance = 0.F;
167 negativeEndSwitch.tolerance = 0.F;
168 positiveEndSwitch.tolerance.setDataValidity(DataValidity::faulty);
169 negativeEndSwitch.tolerance.setDataValidity(DataValidity::faulty);
170 }

175 bool ReadbackHandler::motorIsDummy() {

177
178 try {
179 // Throws if dummy is used

191 void ReadbackHandler::tryMotorRenew() {
192 try {
193 std::ignore = motorIsDummy();
194 readConstData();
195 moduleError.status = StatusOutput::Status::OK;
196 moduleError.message = "";
197 moduleError.setCurrentVersionNumber({});
198 _spiErrorCounter = 0;
199 }
200 catch(ChimeraTK::runtime_error& e) {
201 if(std::string(moduleError.message) != e.what()) {
202 setStatusFromException(e);
203 }
204 _deviceModule->reportException(e.what());
205 }

210 void ReadbackHandler::tryReadingFromMotor() {
211 constexpr unsigned int SPI_RETRY_COUNT{5};
212
213 // Don't event try reading if there is no motor
214 if(!_motor->isOpen()) {
215 return;
216 }
217
218 try {
219 // FIXME This is only to evaluate the timer
220 if(!_execTimer.isInitialized()) {
221 _execTimer.initializeMeasurement();
222 }
223 else {
224 _execTimer.measureIterativeMean();
225 }
226 auto ct = std::chrono::duration_cast<std::chrono::microseconds>(_execTimer.getMeasurementResult());
227 actualCycleTime = static_cast<float>(ct.count()) / 1000.F;
228
229 _receiveTimer.initializeMeasurement();
230
231 readback();
232 readEndSwitchData();
233
234 _receiveTimer.measureOnce();
235 auto rt = std::chrono::duration_cast<std::chrono::microseconds>(_receiveTimer.getMeasurementResult());
236 actualReceiveTime = static_cast<float>(rt.count()) / 1000.F;
237
238 // We have apparently communicated successfully. Reset the error counter
239 _spiErrorCounter = 0;
240 }
241 catch(ChimeraTK::runtime_error& e) {
242 if(_spiErrorCounter > SPI_RETRY_COUNT) {
243 _motor->close();
244 setStatusFromException(e);
245 _deviceModule->reportException(e.what());
246
247 // Reset counter so that there is a longer period in trying to read after re-open
248 _spiErrorCounter = 0;
249 }
250 else {
251 _spiErrorCounter++;
252 }
253 }
254 }
