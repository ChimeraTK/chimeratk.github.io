67BOOST_FIXTURE_TEST_CASE(testMoving, TestFixture) {
68 std::cout << "testMoving" << std::endl;
69
70 TestServer testServer;
71 ChimeraTK::TestFacility testFacility{testServer};
72 testFacility.runApplication();
73
74 auto trigger = testFacility.getVoid("/Motor/readback/tick");
75 auto motorState = testFacility.getScalar<std::string>("/Motor/readback/status/state");
76
77 // Enable the dummy motor
78 // FIXME: If the dummy is enabled here or not does not make a difference. The test still passes. Check the test code.
79 _motorControlerDummy->setEnabled(true);
80 _motorControlerDummy->setMotorCurrentEnabled(true);
81
82 BOOST_CHECK(motorState.dataValidity() == ctk::DataValidity::ok);
83
84 // Trigger readout loop once, so the motor device "wakes up" and actually initializes itself
85 testServer.motor->motorProxyDevice.reportException("Exception trigger from test to toggle deviceBecameFunctional");
86 testFacility.stepApplication();
87 // Get rid of all the intermediate changes due to exception reporting
88 motorState.readLatest();
89
90 // Application should start with disabled motor (initial value being send)
91 // Directly check. Initial values are already received.
92 BOOST_CHECK_EQUAL(static_cast<std::string>(motorState), "disabled");
93
94 testFacility.writeScalar<int>("Motor/controlInput/referenceSettings/positionInSteps", 0);
95 testFacility.stepApplication();
96
97 // Enable stepper motor and set a reference position
98 // (the latter should yield ChimeraTK::MotorDriver::CalibrationMode::SIMPLE)
99 testFacility.writeScalar<int>("Motor/controlInput/control/enable", 1);
100 testFacility.stepApplication();
101 motorState.read();
102
103 testFacility.writeScalar<int>("Motor/controlInput/control/enableAutostart", 1);
104 testFacility.stepApplication();
105 trigger.write();
106 testFacility.stepApplication();
107
108 BOOST_CHECK_EQUAL(testFacility.readScalar<int>("Motor/readback/status/autostartEnabled"), 1);
109 BOOST_CHECK_EQUAL(static_cast<std::string>(motorState), "idle");
110 BOOST_CHECK_EQUAL(testFacility.readScalar<ChimeraTK::Boolean>("Motor/controlInput/notification/hasMessage"), false);
111 BOOST_CHECK_EQUAL(testFacility.readScalar<int>("Motor/readback/status/calibrationMode"),
112 static_cast<int>(ChimeraTK::MotorDriver::CalibrationMode::SIMPLE));
113
114 // Apply a position setpoint
115 int targetPosition{1000};
116 testFacility.writeScalar<int>("Motor/controlInput/positionSetpoint/positionInSteps", targetPosition);
117
118 testFacility.stepApplication();
119 trigger.write();
120 testFacility.stepApplication();
121 motorState.read();
122
123 // Target value should have been applied and module transitioned ito state "moving"
124 BOOST_CHECK_EQUAL(testFacility.readScalar<ChimeraTK::Boolean>("Motor/controlInput/notification/hasMessage"), false);
125 BOOST_CHECK_EQUAL(testFacility.readScalar<int>("Motor/readback/position/targetValueInSteps"), targetPosition);
126 BOOST_CHECK_EQUAL(static_cast<std::string>(motorState), "moving");
127
128 // Move the dummy
129 // Afterwards target position should be read back as actual value and system in state "idle"
130 _motorControlerDummy->moveTowardsTarget(1.F);
131
132 trigger.write();
133 testFacility.stepApplication();
134 motorState.read();
135
136 BOOST_CHECK_EQUAL(testFacility.readScalar<int>("Motor/readback/position/actualValueInSteps"), targetPosition);
137 BOOST_CHECK_EQUAL(static_cast<std::string>(motorState), "idle");
138}

206BOOST_FIXTURE_TEST_CASE(testCalibrationEmergencyStop, TestFixture) {
207 std::cout << "testCalibrationEmergencyStop" << std::endl;
208
209 TestServer testServer;
210 ChimeraTK::TestFacility testFacility{testServer};
211 testFacility.runApplication();
212
213 auto trigger = testFacility.getVoid("/Motor/readback/tick");
214 auto motorState = testFacility.getScalar<std::string>("/Motor/readback/status/state");
215
216 // Enable the dummy motor
217 // FIXME: If the dummy is enabled here or not does not make a difference. The test still passes. Check the test code.
218 _motorControlerDummy->setEnabled(true);
219 _motorControlerDummy->setMotorCurrentEnabled(true);
220
221 // Remove all calibration on the controller
222 _motorControlerDummy->setCalibrationTime(0);
223 _motorControlerDummy->setPositiveReferenceSwitchCalibration(std::numeric_limits<int>::max());
224 _motorControlerDummy->setNegativeReferenceSwitchCalibration(std::numeric_limits<int>::min());
225
226 // This is used to make the move below stop before the endswitch
227 // The default end switch position in the dummy is at +/-10000
228 // while POS_BEYOND_POSITIVE_ENDSWITCH is 50000. So the moveTowardsTarget(1.) will hit the endswitch and the
229 // emergency stop will not have any effect at all - I think.
230 _motorControlerDummy->setPositiveEndSwitch(60000);
231 _motorControlerDummy->setNegativeEndSwitch(-60000);
232
233 BOOST_CHECK(motorState.dataValidity() == ctk::DataValidity::ok);
234
235 // Trigger readout loop once, so the motor device "wakes up" and actually initializes itself
236 testServer.motor->motorProxyDevice.reportException("Exception trigger from test to toggle deviceBecameFunctional");
237 testFacility.stepApplication();
238 // Get rid of all the intermediate changes due to exception reporting
239 motorState.readLatest();
240
241 // Application should start with disabled motor (initial value being send)
242 // Directly check. Initial values are already received.
243 BOOST_CHECK_EQUAL(static_cast<std::string>(motorState), "disabled");
244
245 // Enable stepper motor and set a reference position
246 // (the latter should yield ChimeraTK::MotorDriver::CalibrationMode::SIMPLE)
247 testFacility.writeScalar<int>("Motor/controlInput/control/enable", 1);
248 testFacility.writeScalar<int>("Motor/controlInput/control/enableAutostart", 1);
249 testFacility.stepApplication();
250 motorState.readLatest();
251 trigger.write();
252 testFacility.stepApplication();
253
254 BOOST_CHECK_EQUAL(testFacility.readScalar<int>("Motor/readback/status/autostartEnabled"), 1);
255 BOOST_CHECK_EQUAL(static_cast<std::string>(motorState), "idle");
256 BOOST_CHECK_EQUAL(testFacility.readScalar<ChimeraTK::Boolean>("Motor/controlInput/notification/hasMessage"), false);
257
258 testFacility.writeScalar<int>("Motor/controlInput/control/calibrate", 1);
259 testFacility.stepApplication();
260 trigger.write();
261 testFacility.stepApplication();
262 motorState.read();
263
264 // Move the dummy
265 // Afterwards target position should be read back as actual value and system in state "idle"
266 BOOST_CHECK_EQUAL(static_cast<std::string>(motorState), "calibrating");
267
268 int i = 0;
269 do {
270 if(i < 50) {
271 _motorControlerDummy->moveTowardsTarget(i / 100.F);
272 }
273 trigger.write();
274 if(i == 50) {
275 std::cout << "Performing emergency stop" << std::endl;
276 testFacility.writeScalar<int>("Motor/controlInput/control/emergencyStop", 1);
277 }
278 testFacility.stepApplication();
279 i++;
280 } while(!motorState.readNonBlocking() && (std::string)motorState == std::string{"calibrating"});
281
282 BOOST_CHECK_EQUAL(static_cast<std::string>(motorState), "error");
283 BOOST_CHECK_EQUAL(testFacility.readScalar<int>("Motor/readback/status/calibrationMode"),
284 static_cast<int>(ChimeraTK::MotorDriver::CalibrationMode::NONE));
285}
