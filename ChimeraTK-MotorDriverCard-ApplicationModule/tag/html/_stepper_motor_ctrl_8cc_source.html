237  void ControlInputHandler::mainLoop() {
238  auto inputGroup = this->readAnyGroup();
239 
240  // before anything else, wait for the deviceBecameFunctional trigger
241  // then flush out all values that are written during recovery
242 
243  motorState.setDataValidity(DataValidity::faulty);
244  motorState.write();
245 
246  inputGroup.readUntil(deviceBecameFunctional.getId());
247  writeRecoveryValues();
248 
249  // Write once to propagate inital values
250  motorState.setDataValidity(DataValidity::ok);
251  motorState.writeIfDifferent(_motor->get()->getState());
252  dummySignals.writeAll();
253  message.write();
254 
255  while(true) {
256  try {
257  std::string notificationMessage;
258 
259  auto changedVarId = inputGroup.readAny();
260  if(changedVarId == deviceBecameFunctional.getId()) {
261  writeRecoveryValues();
262  // Flush state of notifications
263  message.setAndWrite("");
264 
265  continue;
266  }
267 
268  // Some state changes are only triggered by the readout of the state, so we regularly
269  // read it out and publish any changes
270  if(changedVarId == trigger.getId() && _motor->isOpen()) {
271  motorState.writeIfDifferent(_motor->get()->getState());
272  continue;
273  }
274 
275  if(not _motor->isOpen()) {
276  notificationMessage = "Motor device is in recovery, not doing control step";
277  }
278  else {
279  // FIXME Keep this only as long as we rely on the dummy for tests
280  try {
281  notificationMessage = std::get<2>(_funcMap.at(changedVarId))();
282  }
283  catch(ChimeraTK::logic_error& e) {
284  notificationMessage = "Exception: " + std::string(e.what());
285  }
286  }
287 
288  dummySignals.dummyMotorStop.writeIfDifferent(control.stop || control.emergencyStop);
289  dummySignals.dummyMotorTrigger++;
290  dummySignals.dummyMotorTrigger.write();
291  message.setAndWrite(notificationMessage);
292  }
293  catch(ChimeraTK::runtime_error& e) {
294  // Comes from the state readout. Do nothing here. either it was a single glitch
295  // or the readout module will see it as well and start the recovery procedure
296  }
297  }
298  }

348  std::string ControlInputHandler::calibrateCallback() {
349  if(!_motor->get()->hasHWReferenceSwitches()) {
350  return {};
351  }
352 
353  if(control.calibrationCtrl.calibrateMotor) {
354  if(_motor->get()->isSystemIdle()) {
355  auto code = _motor->get()->calibrate();
356  if(code != ExitStatus::SUCCESS) {
357  return "Could not calibrate motor: " + ChimeraTK::MotorDriver::toString(code);
358  }
359  }
360  else {
361  return "WARNING: Called calibrateMotor while motor is not in IDLE state.";
362  }
363  }
364 
365  return {};
366  }

383  return "WARNING: Called determineTolerance while motor is not in IDLE state.";
384  }
