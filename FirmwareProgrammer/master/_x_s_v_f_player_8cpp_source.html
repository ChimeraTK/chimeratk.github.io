247 /*****************************************************************************
248  * Function: xsvfGotoTapState
249  * Description: From the current TAP state, go to the named TAP state.
250  * A target state of RESET ALWAYS causes TMS reset sequence.
251  * All SVF standard stable state paths are supported.
252  * All state transitions are supported except for the following
253  * which cause an XSVF_ERROR_ILLEGALSTATE:
254  * - Target==DREXIT2; Start!=DRPAUSE
255  * - Target==IREXIT2; Start!=IRPAUSE
256  * Parameters: pucTapState - Current TAP state; returns final TAP state.
257  * ucTargetState - New target TAP state.
258  * Returns: int - 0 = success; otherwise error.
259  *****************************************************************************/
260 int XSVFPlayer::xsvfGotoTapState(unsigned char* pucTapState, unsigned char ucTargetState) {
261  int iErrorCode;
262 
263  iErrorCode = XSVF_ERROR_NONE;
264  if(ucTargetState == XTAPSTATE_RESET) {
265  /* If RESET, always perform TMS reset sequence to reset/sync TAPs */
266  xsvfTmsTransition(1);
267  for(int i = 0; i < 5; ++i) {
268  SET_PORT(TCK, 0);
269  SET_PORT(TCK, 1);
270  }
271  *pucTapState = XTAPSTATE_RESET;
272  XSVFDBG_PRINTF(3, " TMS Reset Sequence -> Test-Logic-Reset\n");
273  XSVFDBG_PRINTF1(3, " TAP State = %s\n", xsvf_pzTapState[*pucTapState]);
274  }
275  else if((ucTargetState != *pucTapState) &&
276  (((ucTargetState == XTAPSTATE_EXIT2DR) && (*pucTapState != XTAPSTATE_PAUSEDR)) ||
277  ((ucTargetState == XTAPSTATE_EXIT2IR) && (*pucTapState != XTAPSTATE_PAUSEIR)))) {
278  /* Trap illegal TAP state path specification */
279  iErrorCode = XSVF_ERROR_ILLEGALSTATE;
280  }
281  else {
282  if(ucTargetState == *pucTapState) {
283  /* Already in target state. Do nothing except when in DRPAUSE
284  or in IRPAUSE to comply with SVF standard */
285  if(ucTargetState == XTAPSTATE_PAUSEDR) {
286  xsvfTmsTransition(1);
287  *pucTapState = XTAPSTATE_EXIT2DR;
288  XSVFDBG_PRINTF1(3, " TAP State = %s\n", xsvf_pzTapState[*pucTapState]);
289  }
290  else if(ucTargetState == XTAPSTATE_PAUSEIR) {
291  xsvfTmsTransition(1);
292  *pucTapState = XTAPSTATE_EXIT2IR;
293  XSVFDBG_PRINTF1(3, " TAP State = %s\n", xsvf_pzTapState[*pucTapState]);
294  }
295  }
296 
297  /* Perform TAP state transitions to get to the target state */
298  while(ucTargetState != *pucTapState) {
299  switch(*pucTapState) {
300  case XTAPSTATE_RESET:
301  xsvfTmsTransition(0);
302  *pucTapState = XTAPSTATE_RUNTEST;
303  break;
304  case XTAPSTATE_RUNTEST:
305  xsvfTmsTransition(1);
306  *pucTapState = XTAPSTATE_SELECTDR;
307  break;
308  case XTAPSTATE_SELECTDR:
309  if(ucTargetState >= XTAPSTATE_IRSTATES) {
310  xsvfTmsTransition(1);
311  *pucTapState = XTAPSTATE_SELECTIR;
312  }
313  else {
314  xsvfTmsTransition(0);
315  *pucTapState = XTAPSTATE_CAPTUREDR;
316  }
317  break;
318  case XTAPSTATE_CAPTUREDR:
319  if(ucTargetState == XTAPSTATE_SHIFTDR) {
320  xsvfTmsTransition(0);
321  *pucTapState = XTAPSTATE_SHIFTDR;
322  }
323  else {
324  xsvfTmsTransition(1);
325  *pucTapState = XTAPSTATE_EXIT1DR;
326  }
327  break;
328  case XTAPSTATE_SHIFTDR:
329  xsvfTmsTransition(1);
330  *pucTapState = XTAPSTATE_EXIT1DR;
331  break;
332  case XTAPSTATE_EXIT1DR:
333  if(ucTargetState == XTAPSTATE_PAUSEDR) {
334  xsvfTmsTransition(0);
335  *pucTapState = XTAPSTATE_PAUSEDR;
336  }
337  else {
338  xsvfTmsTransition(1);
339  *pucTapState = XTAPSTATE_UPDATEDR;
340  }
341  break;
342  case XTAPSTATE_PAUSEDR:
343  xsvfTmsTransition(1);
344  *pucTapState = XTAPSTATE_EXIT2DR;
345  break;
346  case XTAPSTATE_EXIT2DR:
347  if(ucTargetState == XTAPSTATE_SHIFTDR) {
348  xsvfTmsTransition(0);
349  *pucTapState = XTAPSTATE_SHIFTDR;
350  }
351  else {
352  xsvfTmsTransition(1);
353  *pucTapState = XTAPSTATE_UPDATEDR;
354  }
355  break;
356  case XTAPSTATE_UPDATEDR:
357  if(ucTargetState == XTAPSTATE_RUNTEST) {
358  xsvfTmsTransition(0);
359  *pucTapState = XTAPSTATE_RUNTEST;
360  }
361  else {
362  xsvfTmsTransition(1);
363  *pucTapState = XTAPSTATE_SELECTDR;
364  }
365  break;
366  case XTAPSTATE_SELECTIR:
367  xsvfTmsTransition(0);
368  *pucTapState = XTAPSTATE_CAPTUREIR;
369  break;
370  case XTAPSTATE_CAPTUREIR:
371  if(ucTargetState == XTAPSTATE_SHIFTIR) {
372  xsvfTmsTransition(0);
373  *pucTapState = XTAPSTATE_SHIFTIR;
374  }
375  else {
376  xsvfTmsTransition(1);
377  *pucTapState = XTAPSTATE_EXIT1IR;
378  }
379  break;
380  case XTAPSTATE_SHIFTIR:
381  xsvfTmsTransition(1);
382  *pucTapState = XTAPSTATE_EXIT1IR;
383  break;
384  case XTAPSTATE_EXIT1IR:
385  if(ucTargetState == XTAPSTATE_PAUSEIR) {
386  xsvfTmsTransition(0);
387  *pucTapState = XTAPSTATE_PAUSEIR;
388  }
389  else {
390  xsvfTmsTransition(1);
391  *pucTapState = XTAPSTATE_UPDATEIR;
392  }
393  break;
394  case XTAPSTATE_PAUSEIR:
395  xsvfTmsTransition(1);
396  *pucTapState = XTAPSTATE_EXIT2IR;
397  break;
398  case XTAPSTATE_EXIT2IR:
399  if(ucTargetState == XTAPSTATE_SHIFTIR) {
400  xsvfTmsTransition(0);
401  *pucTapState = XTAPSTATE_SHIFTIR;
402  }
403  else {
404  xsvfTmsTransition(1);
405  *pucTapState = XTAPSTATE_UPDATEIR;
406  }
407  break;
408  case XTAPSTATE_UPDATEIR:
409  if(ucTargetState == XTAPSTATE_RUNTEST) {
410  xsvfTmsTransition(0);
411  *pucTapState = XTAPSTATE_RUNTEST;
412  }
413  else {
414  xsvfTmsTransition(1);
415  *pucTapState = XTAPSTATE_SELECTDR;
416  }
417  break;
418  default:
419  iErrorCode = XSVF_ERROR_ILLEGALSTATE;
420  *pucTapState = ucTargetState; /* Exit while loop */
421  break;
422  }
423  XSVFDBG_PRINTF1(3, " TAP State = %s\n", xsvf_pzTapState[*pucTapState]);
424  }
425  }
426 
427  return (iErrorCode);
428 }

430 /*****************************************************************************
431  * Function: xsvfShiftOnly
432  * Description: Assumes that starting TAP state is SHIFT-DR or SHIFT-IR.
433  * Shift the given TDI data into the JTAG scan chain.
434  * Optionally, save the TDO data shifted out of the scan chain.
435  * Last shift cycle is special: capture last TDO, set last TDI,
436  * but does not pulse TCK. Caller must pulse TCK and optionally
437  * set TMS=1 to exit shift state.
438  * Parameters: lNumBits - number of bits to shift.
439  * plvTdi - ptr to lenval for TDI data.
440  * plvTdoCaptured - ptr to lenval for storing captured TDO data.
441  * iExitShift - 1=exit at end of shift; 0=stay in Shift-DR.
442  * Returns: void.
443  *****************************************************************************/
444 void XSVFPlayer::xsvfShiftOnly(long lNumBits, lenVal* plvTdi, lenVal* plvTdoCaptured, int iExitShift) {
445  unsigned char* pucTdi;
446  unsigned char* pucTdo;
447  unsigned char ucTdiByte;
448  unsigned char ucTdoByte;
449  unsigned char ucTdoBit;
450  int i;
451 
452  /* assert( ( ( lNumBits + 7 ) / 8 ) == plvTdi->len ); */
453 
454  /* Initialize TDO storage len == TDI len */
455  pucTdo = 0;
456  if(plvTdoCaptured) {
457  plvTdoCaptured->len = plvTdi->len;
458  pucTdo = plvTdoCaptured->val + plvTdi->len;
459  }
460 
461  /* Shift LSB first. val[N-1] == LSB. val[0] == MSB. */
462  pucTdi = plvTdi->val + plvTdi->len;
463  while(lNumBits) {
464  /* Process on a byte-basis */
465  ucTdiByte = (*(--pucTdi));
466  ucTdoByte = 0;
467  for(i = 0; (lNumBits && (i < 8)); ++i) {
468  --lNumBits;
469  if(iExitShift && !lNumBits) {
470  /* Exit Shift-DR state */
471  SET_PORT(TMS, 1);
472  }
473 
474  /* Set the new TDI value */
475  SET_PORT(TDI, (short)(ucTdiByte & 1));
476  ucTdiByte >>= 1;
477 
478  /* Set TCK low */
479  SET_PORT(TCK, 0);
480 
481  if(pucTdo) {
482  /* Save the TDO value */
483  ucTdoBit = READ_TDO_BIT;
484  ucTdoByte |= (ucTdoBit << i);
485  }
486 
487  /* Set TCK high */
488  SET_PORT(TCK, 1);
489  }
490 
491  /* Save the TDO byte value */
492  if(pucTdo) {
493  (*(--pucTdo)) = ucTdoByte;
494  }
495  }
496 }

499  * Function: xsvfShift
500  * Description: Goes to the given starting TAP state.
501  * Calls xsvfShiftOnly to shift in the given TDI data and
502  * optionally capture the TDO data.
503  * Compares the TDO captured data against the TDO expected
504  * data.
505  * If a data mismatch occurs, then executes the exception
506  * handling loop upto ucMaxRepeat times.
507  * Parameters: pucTapState - Ptr to current TAP state.
508  * ucStartState - Starting shift state: Shift-DR or Shift-IR.
509  * lNumBits - number of bits to shift.
510  * plvTdi - ptr to lenval for TDI data.
511  * plvTdoCaptured - ptr to lenval for storing TDO data.
512  * plvTdoExpected - ptr to expected TDO data.
513  * plvTdoMask - ptr to TDO mask.
514  * ucEndState - state in which to end the shift.
515  * lRunTestTime - amount of time to wait after the shift.
516  * ucMaxRepeat - Maximum number of retries on TDO mismatch.
517  * Returns: int - 0 = success; otherwise TDO mismatch.
518  * Notes: XC9500XL-only Optimization:
519  * Skip the waitTime() if plvTdoMask->val[0:plvTdoMask->len-1]
520  * is NOT all zeros and sMatch==1.
521  *****************************************************************************/
522 int XSVFPlayer::xsvfShift(unsigned char* pucTapState, unsigned char ucStartState, long lNumBits, lenVal* plvTdi,
523  lenVal* plvTdoCaptured, lenVal* plvTdoExpected, lenVal* plvTdoMask, unsigned char ucEndState, long lRunTestTime,
524  unsigned char ucMaxRepeat) {
525  int iErrorCode;
526  int iMismatch;
527  unsigned char ucRepeat;
528  int iExitShift;
529 
530  iErrorCode = XSVF_ERROR_NONE;
531  iMismatch = 0;
532  ucRepeat = 0;
533  iExitShift = (ucStartState != ucEndState);
534 
535  XSVFDBG_PRINTF1(3, " Shift Length = %ld\n", lNumBits);
536  XSVFDBG_PRINTF(4, " TDI = ");
537  XSVFDBG_PRINTLENVAL(4, plvTdi);
538  XSVFDBG_PRINTF(4, "\n");
539  XSVFDBG_PRINTF(4, " TDO Expected = ");
540  XSVFDBG_PRINTLENVAL(4, plvTdoExpected);
541  XSVFDBG_PRINTF(4, "\n");
542 
543  if(!lNumBits) {
544  /* Compatibility with XSVF2.00: XSDR 0 = no shift, but wait in RTI */
545  if(lRunTestTime) {
546  /* Wait for prespecified XRUNTEST time */
547  xsvfGotoTapState(pucTapState, XTAPSTATE_RUNTEST);
548  XSVFDBG_PRINTF1(3, " Wait = %ld usec\n", lRunTestTime);
549  WAIT_TIME(lRunTestTime);
550  }
551  }
552  else {
553  do {
554  /* Goto Shift-DR or Shift-IR */
555  xsvfGotoTapState(pucTapState, ucStartState);
556 
557  /* Shift TDI and capture TDO */
558  xsvfShiftOnly(lNumBits, plvTdi, plvTdoCaptured, iExitShift);
559 
560  if(plvTdoExpected) {
561  /* Compare TDO data to expected TDO data */
562  iMismatch = !EqualLenVal(plvTdoExpected, plvTdoCaptured, plvTdoMask);
563  }
564 
565  if(iExitShift) {
566  /* Update TAP state: Shift->Exit */
567  ++(*pucTapState);
568  XSVFDBG_PRINTF1(3, " TAP State = %s\n", xsvf_pzTapState[*pucTapState]);
569 
570  if(iMismatch && lRunTestTime && (ucRepeat < ucMaxRepeat)) {
571  XSVFDBG_PRINTF(4, " TDO Expected = ");
572  XSVFDBG_PRINTLENVAL(4, plvTdoExpected);
573  XSVFDBG_PRINTF(4, "\n");
574  XSVFDBG_PRINTF(4, " TDO Captured = ");
575  XSVFDBG_PRINTLENVAL(4, plvTdoCaptured);
576  XSVFDBG_PRINTF(4, "\n");
577  XSVFDBG_PRINTF(4, " TDO Mask = ");
578  XSVFDBG_PRINTLENVAL(4, plvTdoMask);
579  XSVFDBG_PRINTF(4, "\n");

599  }

649 /*****************************************************************************
650  * Function: xsvfDoSDRMasking
651  * Description: Update the data value with the next XSDRINC data and address.
652  * Example: dataVal=0x01ff, nextData=0xab, addressMask=0x0100,
653  * dataMask=0x00ff, should set dataVal to 0x02ab
654  * Parameters: plvTdi - The current TDI value.
655  * plvNextData - the next data value.
656  * plvAddressMask - the address mask.
657  * plvDataMask - the data mask.
658  * Returns: void.
659  *****************************************************************************/
660 #ifdef XSVF_SUPPORT_COMPRESSION
661 void xsvfDoSDRMasking(lenVal* plvTdi, lenVal* plvNextData, lenVal* plvAddressMask, lenVal* plvDataMask) {
662  int i;
663  unsigned char ucTdi;
664  unsigned char ucTdiMask;
665  unsigned char ucDataMask;
666  unsigned char ucNextData;
667  unsigned char ucNextMask;
668  short sNextData;
669 
670  /* add the address Mask to dataVal and return as a new dataVal */
671  addVal(plvTdi, plvTdi, plvAddressMask);
672 
673  ucNextData = 0;
674  ucNextMask = 0;
675  sNextData = plvNextData->len;
676  for(i = plvDataMask->len - 1; i >= 0; --i) {
677  /* Go through data mask in reverse order looking for mask (1) bits */
678  ucDataMask = plvDataMask->val[i];
679  if(ucDataMask) {
680  /* Retrieve the corresponding TDI byte value */
681  ucTdi = plvTdi->val[i];
682 
683  /* For each bit in the data mask byte, look for 1's */
684  ucTdiMask = 1;
685  while(ucDataMask) {
686  if(ucDataMask & 1) {
687  if(!ucNextMask) {
688  /* Get the next data byte */
689  ucNextData = plvNextData->val[--sNextData];
690  ucNextMask = 1;
691  }
692 
693  /* Set or clear the data bit according to the next data */
694  if(ucNextData & ucNextMask) {
695  ucTdi |= ucTdiMask; /* Set bit */
696  }
697  else {
698  ucTdi &= (~ucTdiMask); /* Clear bit */
699  }
700 
701  /* Update the next data */
702  ucNextMask <<= 1;
703  }
704  ucTdiMask <<= 1;
705  ucDataMask >>= 1;
706  }
707 
708  /* Update the TDI value */
709  plvTdi->val[i] = ucTdi;
710  }
711  }
