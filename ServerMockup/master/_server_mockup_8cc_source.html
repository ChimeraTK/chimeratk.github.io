14 void TemplateGroup::addElement(const xmlpp::Element* element) {
15  for(const auto& child : element->get_children()) {
16  const xmlpp::Element* child_element = dynamic_cast<const xmlpp::Element*>(child);
17  if(!child_element) continue;
18  if(child_element->get_name() == "directory") {
19 #ifdef DEBUG
20  std::cout << this->getName() << "::Adding subdirectory: " << child_element->get_attribute_value("name").c_str()
21  << std::endl;
22 #endif
23  groups.emplace_back(TemplateGroup{this, child_element->get_attribute_value("name").c_str(), ""});
24  groups.back().addElement(child_element);
25  }
26  else if(child_element->get_name() == "variable") {
27  xml_parser::addElement(*child_element, this, &_outputAccessorListMap, &_outputArrayAccessorListMap,
28  &_inputAccessorListMap, &_inputArrayAccessorListMap);
