30 variable xml_parser::analyseNode(const xmlpp::Element& element) {
31  variable var;
32  var.name = element.get_attribute_value("name").c_str();
33  for(const auto& child : element.get_children()) {
34  const xmlpp::Element* node = dynamic_cast<const xmlpp::Element*>(child);
35  const xmlpp::TextNode* txt = dynamic_cast<const xmlpp::TextNode*>(child);
36  if(!node) {
37  continue;
38  }
39  // check for additional "text" elements
40  if(txt && txt->is_white_space()) {
41  continue;
42  }
43  // check for empty elements e.g. <unit></unit>
44  if(!node->has_child_text()) {
45  continue;
46  }
47  if(child->get_name() == "description") {
48  try {
49  var.description = node->get_child_text()->get_content().c_str();
50  }
51  catch(xmlpp::exception& e) {
52  std::cerr << "While reading description data:" << e.what() << std::endl;
53  var.description = "";
54  }
55  }
56  else if(child->get_name() == "unit") {
57  try {
58  var.unit = node->get_child_text()->get_content().c_str();
59  }
60  catch(xmlpp::exception& e) {
61  std::cerr << "While reading unit data:" << e.what() << std::endl;
62  var.unit = "";
63  }
64  }
65  else if(child->get_name() == "value_type") {
66  /*
67  * \ToDo: Get rid of this and use SupportedUserTypes for doing that in a generic way
68  * \remark In the XMLGeneretorVisitor that produces the XML files it is done in a similar way
69  * -> find common solution using e.g. boost bimap.
70  */
71  if(std::string(node->get_child_text()->get_content().c_str()).compare("string") == 0) {
72  var.type = ctk::DataType::string;
73  }
74  else if(std::string(node->get_child_text()->get_content().c_str()).compare("int8") == 0) {
75  var.type = ctk::DataType::int8;
76  }
77  else if(std::string(node->get_child_text()->get_content().c_str()).compare("uint8") == 0) {
78  var.type = ctk::DataType::uint8;
79  ;
80  }
81  else if(std::string(node->get_child_text()->get_content().c_str()).compare("int16") == 0) {
82  var.type = ctk::DataType::int16;
83  }
84  else if(std::string(node->get_child_text()->get_content().c_str()).compare("uint16") == 0) {
85  var.type = ctk::DataType::uint16;
86  ;
87  }
88  else if(std::string(node->get_child_text()->get_content().c_str()).compare("int32") == 0) {
89  var.type = ctk::DataType::int32;
90  }
91  else if(std::string(node->get_child_text()->get_content().c_str()).compare("uint32") == 0) {
92  var.type = ctk::DataType::uint32;
93  ;
94  }
95  else if(std::string(node->get_child_text()->get_content().c_str()).compare("int64") == 0) {
96  var.type = ctk::DataType::int64;
97  }
98  else if(std::string(node->get_child_text()->get_content().c_str()).compare("uint64") == 0) {
99  var.type = ctk::DataType::uint64;
100  ;
101  }
102  else if(std::string(node->get_child_text()->get_content().c_str()).compare("double") == 0) {
103  var.type = ctk::DataType::float64;
104  }
105  else if(std::string(node->get_child_text()->get_content().c_str()).compare("float") == 0) {
106  var.type = ctk::DataType::float32;
107  }
108  else if(std::string(node->get_child_text()->get_content().c_str()).compare("Boolean") == 0) {
109  var.type = ctk::DataType::Boolean;
110  }
111  else {
112  std::cerr << "Read unknown type: " << node->get_child_text()->get_content().c_str() << std::endl;
113  }
114  }
115  else if(child->get_name() == "direction") {
116  std::string type = node->get_child_text()->get_content().c_str();
117  if(type.compare("application_to_control_system") == 0) {
118  var.dir = xml_parser::direction::OUT;
119  }
120  else if(type.compare("control_system_to_application") == 0 ||
121  type.compare("control_system_to_application_with_return") == 0) {
122  var.dir = xml_parser::direction::IN;
123  }
124  else {
125  throw xmlpp::exception("Unknown data flow direction: " + type);
126  }
127  }
128  else if(child->get_name() == "numberOfElements") {
129  var.nElements = atoi(node->get_child_text()->get_content().c_str());
130  }
131  }
132  return var;
133 }

135 std::string xml_parser::getRootDir(std::string inputFile) {
136  xmlpp::DomParser parser;
137  parser.parse_file(inputFile);
138  const auto root = parser.get_document()->get_root_node();
139  if(root->get_name() != "application") {
140  throw xmlpp::exception("Expected root tag 'application' instead of: " + root->get_name());
141  }
142  return std::string(root->get_attribute_value("name").c_str());
143 }

156  auto myLambda = [&](auto arg) {
157  auto& accessorList = boost::fusion::at_key<decltype(arg)>(l2->table);

159  };
160  ctk::callForTypeNoVoid(tmp.type, myLambda);
161  }
162  else {

165  accessorList.emplace_back(
166  ctk::ArrayPollInput<decltype(arg)>{owner, tmp.name, "", tmp.nElements, tmp.description});
167  };
168  ctk::callForTypeNoVoid(tmp.type, myLambda);
169  }

171  else {
172  if(tmp.dir == xml_parser::direction::OUT) {

174  auto& accessorList = boost::fusion::at_key<decltype(arg)>(l1->table);

183  };
