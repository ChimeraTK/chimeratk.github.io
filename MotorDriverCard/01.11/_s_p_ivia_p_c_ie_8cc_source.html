45  boost::lock_guard<boost::recursive_mutex> guard(_spiMutex);
46 
47  // try three times to mittigate effects of a firmware bug
48  for(int i = 0; i < 3; ++i) {
49  if(i > 0) {
50  std::cerr << "Warning, SPI handshake timed out. Retrying..." << std::endl;
51  }
52 
53  // Implement the write handshake
54  // 1. write 0xff to the synch register
55  _synchronisationRegister = SPI_SYNC_REQUESTED;
56  _synchronisationRegister.write();
57 
58  // 2. write the spi command
59  _writeRegister = spiCommand;
60  _writeRegister.write();
61 
62  // 3. wait for the handshake
63  _synchronisationRegister.read();
64 
65  for(size_t syncCounter = 0; (_synchronisationRegister == SPI_SYNC_REQUESTED) && (syncCounter < 10);
66  ++syncCounter) {
67  sleepMicroSeconds(_spiWaitingTime);
68 
69  _synchronisationRegister.read();
70  }
71 
72  if(_synchronisationRegister != SPI_SYNC_REQUESTED) {
73  // break on either error or ok. If still in SPI_SYNC_REQUESTED repeat (up
74  // to 3 times)
75  break;
76  }
77  }
78

80  // needed, but it minimises code duplication.
81  std::stringstream errorDetails;
82  errorDetails << "PCIe register " << _moduleName << "." << _writeRegisterName << ", sync register " << _moduleName

87  break;
