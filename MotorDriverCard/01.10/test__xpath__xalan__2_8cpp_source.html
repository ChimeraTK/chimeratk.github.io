135 static const char_t* number_to_string_unsafe(int number) {
136  static char_t buffer[16];
137 
138  // construct number in reverse
139  char_t* it = buffer;
140 
141  while(number) {
142  *it++ = static_cast<char_t>('0' + number % 10);
143  number /= 10;
144  }
145 
146  // zero terminate
147  *it = 0;
148 
149  // reverse to get final result
150  std::reverse(buffer, it);
151 
152  return buffer;
153 }
