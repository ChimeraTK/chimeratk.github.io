189  void DFMC_MD22Dummy::handleControlerSpiWrite(){
190  //debug functionality: cause timeouts by ignoring spi writes
191  if (checkSpiTimeoutsCounter()){
192  return;
193  }
194  int32_t value = SPI_SYNC_ERROR;
195  //debug functionality: cause errors
196  if (_causeSpiErrors){
197  write(_controlerSpiBar, _controlerSpiSyncAddress, &value, 4);
198  return;
199  }
200 
201  int32_t controlerSpiSyncWord;
202  read(_controlerSpiBar, _controlerSpiSyncAddress, &controlerSpiSyncWord, 4);
203 
204  if (controlerSpiSyncWord != SPI_SYNC_REQUESTED){
205  write(_controlerSpiBar, _controlerSpiSyncAddress, &value, 4);
206  return;
207  }
208 
209  int32_t controlerSpiWriteWord;
210  read(_controlerSpiBar, _controlerSpiWriteAddress, &controlerSpiWriteWord, 4);
211 
212  TMC429InputWord inputWord(controlerSpiWriteWord);
213  uint32_t controlerSpiAddress = inputWord.getADDRESS();
214  if (inputWord.getRW() == RW_READ){
215  writeControlerSpiContentToReadbackRegister(controlerSpiAddress);
216  // it takes some time to perform this in firmware. Implement a short delay,
217  // which proabaly will cause rescheduling, twice for write and read
218  boost::this_thread::sleep(boost::posix_time::microseconds(2*_microsecondsControllerSpiDelay));
219  }
220  else{
221  writeContentToControlerSpiRegister(inputWord.getDATA(), controlerSpiAddress);
222  // it takes some time to perform this in firmware. Implement a short delay,
223  // which proabaly will cause rescheduling.
224  boost::this_thread::sleep(boost::posix_time::microseconds(_microsecondsControllerSpiDelay));
225  }
226 
227  triggerActionsOnControlerSpiWrite(inputWord);
228  value = SPI_SYNC_OK;
229  // SPI action done, write sync ok to finish the handshake
230  write(_controlerSpiBar, _controlerSpiSyncAddress, &value, 4);
231  }

233  void DFMC_MD22Dummy::handleDriverSpiWrite(unsigned int ID){
234  //debug functionality: cause timeouts by ignoring spi writes
235  if (checkSpiTimeoutsCounter()){
236  return;
237  }
238 
239  unsigned int bar = _driverSPIs[ID].bar;
240  unsigned int writeIndex = _driverSPIs[ID].pcieWriteAddress/sizeof(int32_t);
241  unsigned int syncIndex = _driverSPIs[ID].pcieSyncAddress/sizeof(int32_t);
242 
243  //debug functionality: cause errors
244  if (_causeSpiErrors){
245  _barContents[bar].at( syncIndex ) = SPI_SYNC_ERROR;
246  return;
247  }
248 
249  // check the sync register
250  if ( _barContents[bar].at( syncIndex ) != SPI_SYNC_REQUESTED){
251  _barContents[bar].at( syncIndex ) = SPI_SYNC_ERROR;
252  return;
253  }
254 
255  // perform the SPI transfer
256  unsigned int writtenSpiWord = _barContents[bar].at(writeIndex);
257  unsigned int spiAddress = tmc260::spiAddressFromDataWord(writtenSpiWord);
258  unsigned int payloadDataMask = tmc260::dataMaskFromSpiAddress(spiAddress);
259 
260  boost::this_thread::sleep(boost::posix_time::microseconds(_microsecondsDriverSpiDelay));
261 
262  _driverSPIs[ID].addressSpace.at(spiAddress) = writtenSpiWord & payloadDataMask;
263 
264  driverSpiActions(ID, spiAddress, writtenSpiWord & payloadDataMask);
265 
266  // SPI transfer ready, set the sync register to ok
267  _barContents[bar].at( syncIndex ) = SPI_SYNC_OK;
268  }
