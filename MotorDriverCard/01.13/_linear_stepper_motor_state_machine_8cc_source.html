78  void LinearStepperMotor::StateMachine::calibrationThreadFunction() {
79  _motor._calibrationFailed.exchange(false);
80  _stopAction.exchange(false);
81  _moveInterrupted.exchange(false);
82 
83  // Local variables for calibrated position
84  // TODO Move to else scope to remove cppcheck warning
85  int calibPositiveEndSwitchInSteps = 0;
86  int calibNegativeEndSwitchInSteps = 0;
87 
88  // At least one end switch disabled -> calibration not possible
89  if(!(_motor._positiveEndSwitchEnabled.load() && _motor._negativeEndSwitchEnabled.load())) {
90  _motor._motorControler->setCalibrationTime(0);
91 
92  _motor._calibrationFailed.exchange(true);
93  _motor._calibrationMode.exchange(CalibrationMode::NONE);
94  }
95  else {
96  findEndSwitch(Sign::POSITIVE);
97  calibPositiveEndSwitchInSteps = _motor.getCurrentPositionInSteps();
98  findEndSwitch(Sign::NEGATIVE);
99  calibNegativeEndSwitchInSteps = _motor.getCurrentPositionInSteps();
100 
101  if(_moveInterrupted.load() || _stopAction.load()) {
102  _motor._motorControler->setCalibrationTime(0);
103 
104  _motor._calibrationFailed.exchange(true);
105  _motor._errorMode.exchange(Error::CALIBRATION_ERROR);
106  _motor._calibrationMode.exchange(CalibrationMode::NONE);
107  }
108  else {
109  // Define positive axis with negative end switch as zero
110  calibPositiveEndSwitchInSteps = calibPositiveEndSwitchInSteps - calibNegativeEndSwitchInSteps;
111  calibNegativeEndSwitchInSteps = 0;
112 
113  _motor._motorControler->setCalibrationTime(time(nullptr));
114  _motor._motorControler->setPositiveReferenceSwitchCalibration(calibPositiveEndSwitchInSteps);
115  _motor._motorControler->setNegativeReferenceSwitchCalibration(0);
116  _motor._calibPositiveEndSwitchInSteps.exchange(calibPositiveEndSwitchInSteps);
117  _motor._calibNegativeEndSwitchInSteps.exchange(0);
118 
119  _motor._calibrationMode.exchange(CalibrationMode::FULL);
120  _motor.resetMotorControllerPositions(0);
121  }
122  }
123 
124  _asyncActionActive.exchange(false);
125  return;
126  }

128  void LinearStepperMotor::StateMachine::findEndSwitch(Sign sign) {
129  while(!_motor.isEndSwitchActive(sign)) {
130  if(_stopAction.load() || _moveInterrupted.load()) {
131  return;
132  }
133  else if(_motor.getTargetPositionInSteps() != _motor.getCurrentPositionInSteps() &&
134  !_motor.isPositiveReferenceActive() && !_motor.isNegativeReferenceActive()) {
135  _moveInterrupted.exchange(true);
136  return;
137  }
138  moveToEndSwitch(sign);
139  }
140  return;

199  double LinearStepperMotor::StateMachine::getToleranceEndSwitch(Sign sign) {
200  double meanMeasurement = 0;
201  double stdMeasurement = 0;
202  const int N_TOLERANCE_CALC_SAMPLES = 10;
203  double measurements[N_TOLERANCE_CALC_SAMPLES];
204 
205  int endSwitchPosition = getPositionEndSwitch(sign);
206 
207  // Get 10 samples for tolerance calculation
208  for(unsigned i = 0; i < N_TOLERANCE_CALC_SAMPLES; i++) {
209  if(_stopAction.load() || _moveInterrupted.load()) {
210  break;
211  }
212 
213  // Move close to end switch
214  {
215  boost::lock_guard<boost::mutex> lck(_motor._mutex);
216  _motor._motorControler->setTargetPosition(endSwitchPosition - static_cast<int>(sign) * 1000);
217  }
218  while(_motor._motorControler->isMotorMoving()) {
219  std::this_thread::sleep_for(std::chrono::milliseconds(wakeupPeriodInMilliseconds));
220  }
221 
222  // Check if in expected position
223  if(!_motor.verifyMoveAction()) {
224  _moveInterrupted.exchange(true);
225  break;
226  }
227 
228  // Try to move beyond end switch
229  {
230  boost::lock_guard<boost::mutex> lck(_motor._mutex);
231  _motor._motorControler->setTargetPosition(endSwitchPosition + static_cast<int>(sign) * 1000);
232  }
233  while(_motor._motorControler->isMotorMoving()) {
234  std::this_thread::sleep_for(std::chrono::milliseconds(wakeupPeriodInMilliseconds));
235  }
236  if(!_motor.isEndSwitchActive(sign)) {
237  _moveInterrupted.exchange(true);
238  break;
239  }
240 
241  // Mean calculation
242  meanMeasurement += static_cast<double>(_motor.getCurrentPositionInSteps()) / N_TOLERANCE_CALC_SAMPLES;
243  measurements[i] = _motor.getCurrentPositionInSteps();
244  } /* for (i in [0,9]) */
245 
246  // Compute variance
247  if(!(_stopAction.load() || _moveInterrupted.load())) {
248  for(unsigned int i = 0; i < N_TOLERANCE_CALC_SAMPLES; i++) {
249  stdMeasurement += ((measurements[i] - meanMeasurement) / (N_TOLERANCE_CALC_SAMPLES - 1)) *
250  (measurements[i] - meanMeasurement);
251  }
252  }
253 
254  return sqrt(stdMeasurement);
255  }
