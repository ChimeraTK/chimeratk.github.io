38  StepperMotorStatusAndError StepperMotor::moveToPositionInSteps(int newPositionInSteps) {
39  StepperMotorStatusAndError statusAndError = this->determineMotorStatusAndError();
40  if(statusAndError.status == StepperMotorStatusTypes::M_ERROR ||
41  statusAndError.status == StepperMotorStatusTypes::M_DISABLED) {
42  return statusAndError;
43  }
44 
45  _blockingFunctionActive = true;
46  // Ask the hardware to move the motor.
47  this->setTargetPositionInSteps(newPositionInSteps);
48  if(!this->getAutostart()) {
49  this->start();
50  }
51 
52  // Block till the motor has finished moving.
53  while(StepperMotor::isMoving()) {
54  usleep(1000);
55  }
56 
57  // update and return status and error once at target position.
58  _blockingFunctionActive = false;
59 
60  return (this->determineMotorStatusAndError());
61  }

274  int StepperMotor::truncateMotorPosition(int newPosition) {
275  int position = newPosition;

290  StepperMotorStatusAndError StepperMotor::determineMotorStatusAndError() {
291  // InternalMutex intMutex(this);
292  boost::lock_guard<boost::mutex> guard(_mutex);
293 
294  _motorStatus = StepperMotorStatusTypes::M_OK;
295  _motorError = StepperMotorErrorTypes::M_NO_ERROR;
296 
297  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
298  // ERROR CONDITION CHECK - it should be done first.
299  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
300 
301  if(_softwareLimitsEnabled && (_maxPositionLimitInSteps <= _minPositionLimitInSteps)) {
302  _motorStatus = StepperMotorStatusTypes::M_ERROR;
303  _motorError = StepperMotorErrorTypes::M_CONFIG_ERROR_MIN_POS_GRATER_EQUAL_TO_MAX;
304 
305  return StepperMotorStatusAndError(_motorStatus, _motorError);
306  }
307 
308  // first check if disabled
309  if(!_motorControler->isEnabled()) {
310  _motorStatus = StepperMotorStatusTypes::M_DISABLED;
311  return StepperMotorStatusAndError(_motorStatus, _motorError);
312  }
313 
314  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
315  // STATUS CHECK
316  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
317 
318  if(_motorControler->getStatus().getStandstillIndicator() == 0) {
319  _motorStatus = StepperMotorStatusTypes::M_IN_MOVE;
320  return StepperMotorStatusAndError(_motorStatus, _motorError);
321  }
322 
323  if(_targetPositionInSteps != _motorControler->getActualPosition()) {
324  _motorStatus = StepperMotorStatusTypes::M_NOT_IN_POSITION;
325  return StepperMotorStatusAndError(_motorStatus, _motorError);
326  }
327 
328  if(_softwareLimitsEnabled) {
329  if(_targetPositionInSteps >= _maxPositionLimitInSteps) {
330  _motorStatus = StepperMotorStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON;
331  return StepperMotorStatusAndError(_motorStatus, _motorError);
332  }
333 
334  if(_targetPositionInSteps <= _minPositionLimitInSteps) {
335  _motorStatus = StepperMotorStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON;
336  return StepperMotorStatusAndError(_motorStatus, _motorError);
337  }
338  }
339 
340  return StepperMotorStatusAndError(_motorStatus, _motorError);
341  }
