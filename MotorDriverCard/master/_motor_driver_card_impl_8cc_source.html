149  void MotorDriverCardImpl::checkFirmwareVersion() {
150  ChimeraTK::ScalarRegisterAccessor<int32_t> firmwareVersion =
151  _device->getScalarRegisterAccessor<int32_t>(_moduleName + "/" + PROJECT_VERSION_ADDRESS_STRING);
152  firmwareVersion.read();
153 
154  // only allow firmware versions with the same major number
155  uint32_t maxFirmwareVersion = (MINIMAL_FIRMWARE_VERSION & 0xFF000000) | 0x00FFFFFF;
156 
157  // check the version
158  if((static_cast<uint32_t>(firmwareVersion) < static_cast<uint32_t>(MINIMAL_FIRMWARE_VERSION)) ||
159  (static_cast<uint32_t>(firmwareVersion) > maxFirmwareVersion)) {
160  std::stringstream errorMessage;
161  errorMessage << "MotorDriverCardImpl detected wrong firmware version. "
162  << "Current firmware is 0x" << std::hex << firmwareVersion << ", minumum required version is 0x"
163  << MINIMAL_FIRMWARE_VERSION << ", maximum allowed version is 0x" << maxFirmwareVersion << std::dec
164  << ".";
165  throw ChimeraTK::runtime_error(errorMessage.str());
166  }
