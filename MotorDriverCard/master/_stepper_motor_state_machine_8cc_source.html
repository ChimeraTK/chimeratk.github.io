48  void BasicStepperMotor::StateMachine::waitForStandstill() {
49  if(!_motorControler->isMotorMoving()) {
50  _asyncActionActive.exchange(false);
51 
52  // Move may be stopped by
53  // user requesting state transition
54  if(hasRequestedState()) {
55  moveToRequestedState();
56  return;
57  }
58 
59  Event stateExitEvnt = stopEvent;
60 
61  // Motor stopped by itself
62  if(!_stepperMotor.verifyMoveAction()) {
63  _stepperMotor._errorMode.exchange(Error::MOVE_INTERRUPTED);
64 
65  _motorControler->setTargetPosition(_motorControler->getActualPosition());
66  stateExitEvnt = errorEvent;
67  }
68  performTransition(stateExitEvnt);
69  return;
70  }
71  }
