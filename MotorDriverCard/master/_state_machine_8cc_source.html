60 
61  const State::TransitionTable& transitionTable = _currentState->getTransitionTable();
62  it = transitionTable.find(event);
63  if(it != transitionTable.end()) {
64  _requestedState = ((it->second).targetState);
65  _requestedInternalCallback = (it->second).internalCallbackAction;
66 
67  // Apply new state right away, if no async action active
68  if(!_asyncActionActive.load()) {
69  _currentState = _requestedState;
