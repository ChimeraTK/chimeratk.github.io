226  void MotorControlerDummy::roundToNextFullStep(int& targetPosition) {
227  int delta = targetPosition - _currentPosition;
228  int deltaMicroStep = delta * _userMicroStepSize;
229  unsigned int actualMicroStepCount = ((_currentPosition * _userMicroStepSize) - 1) & 0x3FF;
230  unsigned int newActualMicroStepCount = (actualMicroStepCount + deltaMicroStep) & 0x3FF;
231  unsigned int distanceToPreviousFullStep = (newActualMicroStepCount + 1) % 256;
232  if(distanceToPreviousFullStep < 128) {
233  targetPosition = targetPosition - distanceToPreviousFullStep / _userMicroStepSize;
234  }
235  else {
236  targetPosition = targetPosition + (256 - distanceToPreviousFullStep) / _userMicroStepSize;
237  }

385  void MotorControlerDummy::moveTowardsTarget(
386  float fraction, bool block, bool bothEndSwitchesAlwaysOn, bool zeroPositions) {
387  lock_guard guard(_motorControllerDummyMutex);
388  if(zeroPositions) {
389  _absolutePosition = 0;
390  _targetPosition = 0;
391  _currentPosition = 0;
392  _positiveEndSwitchEnabled = true;
393  _negativeEndSwitchEnabled = true;
394  }
395 
396  _bothEndSwitchesAlwaysOn = bothEndSwitchesAlwaysOn;
397 
398  _blockMotor = block;
399  if(_blockMotor) {
400  return;
401  }
402 
403  // check if the motor is should to be moving if it was enabled
404  // otherwise return immediately
405  if(!isStepping()) {
406  return;
407  }
408 
409  // range check for the input parameter 'fraction', just force it to 0..1
410  if(fraction <= 0) {
411  return;
412  }
413  if(fraction > 1) {
414  fraction = 1.;
415  }
416 
417  // calculate the new abolute target position for this move operation
418  int relativeSteps = (_targetPosition - _currentPosition) * fraction;
419  int absoluteTargetInThisMove = _absolutePosition + relativeSteps;
420  int targetInThisMove = _currentPosition + relativeSteps;
421 
422  if(!_motorCurrentEnabled) {
423  // the motor is stepping to the target position, but the actual positon
424  // does not change. Thus no end switch will be hit.
425  _currentPosition = targetInThisMove;
426  return;
427  }
428 
429  // ok, so the motor is really moving.
430 
431  // check whether an end switch would be hit
432  if(_positiveEndSwitchEnabled && (absoluteTargetInThisMove > _positiveEndSwitchPosition)) {
433  absoluteTargetInThisMove = _positiveEndSwitchPosition;
434  // also the current positon stops if an end switch is reached
435  int stepsToEndSwitch = _positiveEndSwitchPosition - _absolutePosition;
436  targetInThisMove = _currentPosition + stepsToEndSwitch;
437  }
438  if(_negativeEndSwitchEnabled && (absoluteTargetInThisMove < _negativeEndSwitchPosition)) {
439  absoluteTargetInThisMove = _negativeEndSwitchPosition;
440  // also the current positon stops if an end switch is reached
441  int stepsToEndSwitch = _negativeEndSwitchPosition - _absolutePosition;
442  targetInThisMove = _currentPosition + stepsToEndSwitch;
443  }
444 
445  // finally 'move' the motor
446  _absolutePosition = absoluteTargetInThisMove;
447  _currentPosition = targetInThisMove;
448  }
