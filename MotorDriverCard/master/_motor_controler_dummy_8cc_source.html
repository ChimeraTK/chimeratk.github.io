227  void MotorControlerDummy::roundToNextFullStep(int& targetPosition) {
228  int delta = targetPosition - _currentPosition;
229  int deltaMicroStep = delta * _userMicroStepSize;
230  unsigned int actualMicroStepCount = ((_currentPosition * _userMicroStepSize) - 1) & 0x3FF;
231  unsigned int newActualMicroStepCount = (actualMicroStepCount + deltaMicroStep) & 0x3FF;
232  unsigned int distanceToPreviousFullStep = (newActualMicroStepCount + 1) % 256;
233  if(distanceToPreviousFullStep < 128) {
234  targetPosition = targetPosition - distanceToPreviousFullStep / _userMicroStepSize;
235  }
236  else {
237  targetPosition = targetPosition + (256 - distanceToPreviousFullStep) / _userMicroStepSize;
238  }

374  void MotorControlerDummy::moveTowardsTarget(
375  float fraction, bool block, bool bothEndSwitchesAlwaysOn, bool zeroPositions) {
376  lock_guard guard(_motorControllerDummyMutex);
377  if(zeroPositions) {
378  _absolutePosition = 0;
379  _targetPosition = 0;
380  _currentPosition = 0;
381  _positiveEndSwitchEnabled = true;
382  _negativeEndSwitchEnabled = true;
383  }
384 
385  _bothEndSwitchesAlwaysOn = bothEndSwitchesAlwaysOn;
386 
387  _blockMotor = block;
388  if(_blockMotor) {
389  return;
390  }
391 
392  // check if the motor is should to be moving if it was enabled
393  // otherwise return immediately
394  if(!isStepping()) {
395  return;
396  }
397 
398  // range check for the input parameter 'fraction', just force it to 0..1
399  if(fraction <= 0) {
400  return;
401  }
402  if(fraction > 1) {
403  fraction = 1.;
404  }
405 
406  // calculate the new abolute target position for this move operation
407  int relativeSteps = (_targetPosition - _currentPosition) * fraction;
408  int absoluteTargetInThisMove = _absolutePosition + relativeSteps;
409  int targetInThisMove = _currentPosition + relativeSteps;
410 
411  if(!_motorCurrentEnabled) {
412  // the motor is stepping to the target position, but the actual positon
413  // does not change. Thus no end switch will be hit.
414  _currentPosition = targetInThisMove;
415  return;
416  }
417 
418  // ok, so the motor is really moving.
419 
420  // check whether an end switch would be hit
421  if(_positiveEndSwitchEnabled && (absoluteTargetInThisMove > _positiveEndSwitchPosition)) {
422  absoluteTargetInThisMove = _positiveEndSwitchPosition;
423  // also the current positon stops if an end switch is reached
424  int stepsToEndSwitch = _positiveEndSwitchPosition - _absolutePosition;
425  targetInThisMove = _currentPosition + stepsToEndSwitch;
426  }
427  if(_negativeEndSwitchEnabled && (absoluteTargetInThisMove < _negativeEndSwitchPosition)) {
428  absoluteTargetInThisMove = _negativeEndSwitchPosition;
429  // also the current positon stops if an end switch is reached
430  int stepsToEndSwitch = _negativeEndSwitchPosition - _absolutePosition;
431  targetInThisMove = _currentPosition + stepsToEndSwitch;
432  }
433 
434  // finally 'move' the motor
435  _absolutePosition = absoluteTargetInThisMove;
436  _currentPosition = targetInThisMove;
437  }
