278  bool BasicStepperMotor::checkIfOverflow(int termA, int termB) {
279  int signTermA = (termA > 0) - (termA < 0);
280  int signTermB = (termB > 0) - (termB < 0);
281  if(signTermA * signTermB <= 0) {
282  return false;
283  }
284  else if(signTermA > 0) {
285  if(termB > std::numeric_limits<int>::max() - termA) {
286  return true;
287  }
288  else {
289  return false;
290  }
291  }
292  else {
293  if(termB < std::numeric_limits<int>::min() - termA) {
294  return true;
295  }
296  else {
297  return false;
298  }
299  }
