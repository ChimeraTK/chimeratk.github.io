35  void SPIviaPCIe::write(int32_t spiCommand) {
36  boost::lock_guard<boost::recursive_mutex> guard(_spiMutex);
37 
38  // try three times to mitigate effects of a firmware bug
39  for(int i = 0; i < 3; ++i) {
40  // Implement the write handshake
41  // 1. write 0xff to the synch register
42  _synchronisationRegister = SPI_SYNC_REQUESTED;
43  _synchronisationRegister.write();
44 
45  // 2. write the spi command
46  _writeRegister = spiCommand;
47  _writeRegister.write();
48 
49  // 3. wait for the handshake
50  _synchronisationRegister.read();
51 
52  for(size_t syncCounter = 0; (_synchronisationRegister == SPI_SYNC_REQUESTED) && (syncCounter < 10);
53  ++syncCounter) {
54  boost::this_thread::sleep_for(_spiWaitingTime);
55 
56  _synchronisationRegister.read();
57  }
58 
59  if(_synchronisationRegister != SPI_SYNC_REQUESTED) {
60  // break on either error or ok. If still in SPI_SYNC_REQUESTED repeat (up
61  // to 3 times)
62  break;
63  }
64  }
65 
66  // It might be inefficient to always create the error message, even if not
67  // needed, but it minimises code duplication.
68  std::stringstream errorDetails;
69  errorDetails << "PCIe register " << _moduleName << "." << _writeRegisterName << ", sync register " << _moduleName
70  << "." << _syncRegisterName << "= 0x" << std::hex << _synchronisationRegister << ", spi command 0x"
71  << std::hex << spiCommand << std::dec;
72  switch(_synchronisationRegister) {
73  case SPI_SYNC_OK:
74  break;
75  case SPI_SYNC_REQUESTED:
76  throw ChimeraTK::runtime_error("Timeout writing via SPI, " + errorDetails.str());
77  case SPI_SYNC_ERROR:
78  default:
79  throw ChimeraTK::runtime_error("Error writing via SPI, " + errorDetails.str());
80  }
81  }

83  uint32_t SPIviaPCIe::read(int32_t spiCommand) {
84  boost::lock_guard<boost::recursive_mutex> guard(_spiMutex);
85 
86  // this is easy: the write does all the sync. And after a successful sync we
87  // know that the readback word is valid.
88  write(spiCommand);
89 
90  _readbackRegister.read();
91  return static_cast<uint32_t>(_readbackRegister);
92  }
