212  void LinearStepperMotor::loadEndSwitchCalibration() {
213  LockGuard guard(_mutex);
214 
215  if(_motorControler->getCalibrationTime() != 0) {
216  int calibPositiveEndSwitchInSteps = _motorControler->getPositiveReferenceSwitchCalibration();
217  int calibNegativeEndSwitchInSteps = _motorControler->getNegativeReferenceSwitchCalibration();
218  _calibPositiveEndSwitchInSteps.exchange(calibPositiveEndSwitchInSteps);
219  _calibNegativeEndSwitchInSteps.exchange(calibNegativeEndSwitchInSteps);
220 
221  if(calibPositiveEndSwitchInSteps == std::numeric_limits<int>::max() &&
222  calibNegativeEndSwitchInSteps == std::numeric_limits<int>::min()) {
223  _calibrationMode.exchange(CalibrationMode::SIMPLE);
224  }
225  else {
226  _calibrationMode.exchange(CalibrationMode::FULL);

239  bool LinearStepperMotor::isEndSwitchActive(Sign sign) {
240  bool posActive = _motorControler->getReferenceSwitchData().getPositiveSwitchActive() == 1U;
241  bool negActive = _motorControler->getReferenceSwitchData().getNegativeSwitchActive() == 1U;
242 
243  if(posActive && negActive) {
244  _errorMode.exchange(Error::BOTH_END_SWITCHES_ON);
245  _stateMachine->setAndProcessUserEvent(StateMachine::errorEvent);
246  }
247  else {
248  // The error mode was BOTH_END_SWITCHES_ON, reset the error mode. Otherwise this will stick unless the application
249  // is re-started.
250  auto expected = Error::BOTH_END_SWITCHES_ON;
251  auto replace = Error::NO_ERROR;
252  _errorMode.compare_exchange_strong(expected, replace);
253  }
254 
255  if(_toleranceCalculated) {
256  if(posActive) {
257  if(std::abs(_motorControler->getActualPosition() - _calibPositiveEndSwitchInSteps.load()) >
258  3 * _tolerancePositiveEndSwitch.load()) {
259  _errorMode.exchange(Error::CALIBRATION_ERROR);
260  }
261  }
262  else if(negActive) {
263  if(std::abs(_motorControler->getActualPosition() - _calibNegativeEndSwitchInSteps.load()) >
264  3 * _toleranceNegativeEndSwitch.load()) {
265  _errorMode.exchange(Error::CALIBRATION_ERROR);
266  }
267  }
268  }
269 
270  if(sign == Sign::POSITIVE) {
271  return posActive;
272  }
273 
274  return negActive;
275  }
