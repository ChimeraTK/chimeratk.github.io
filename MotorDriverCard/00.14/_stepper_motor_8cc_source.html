336  StepperMotorStatusAndError StepperMotor::determineMotorStatusAndError() {
337 
338  //InternalMutex intMutex(this);
339  boost::lock_guard<boost::mutex> guard(_mutex);
340 
341  _motorStatus = StepperMotorStatusTypes::M_OK;
342  _motorError = StepperMotorErrorTypes::M_NO_ERROR;
343 
344  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
345  // ERROR CONDITION CHECK - it should be done first.
346  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
347 
348 
349  if (_softwareLimitsEnabled && (_maxPositionLimitInSteps <= _minPositionLimitInSteps)) {
350  _motorStatus = StepperMotorStatusTypes::M_ERROR;
351  _motorError = StepperMotorErrorTypes::M_CONFIG_ERROR_MIN_POS_GRATER_EQUAL_TO_MAX;
352 
353  return StepperMotorStatusAndError(_motorStatus, _motorError);
354  }
355 
356  //first check if disabled
357  if (!_motorControler->isEnabled()) {
358  _motorStatus = StepperMotorStatusTypes::M_DISABLED;
359  return StepperMotorStatusAndError(_motorStatus, _motorError);
360  }
361 
362  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
363  // STATUS CHECK
364  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
365 
366  if (_motorControler->getStatus().getStandstillIndicator() == 0) {
367  _motorStatus = StepperMotorStatusTypes::M_IN_MOVE;
368  return StepperMotorStatusAndError(_motorStatus, _motorError);
369  }
370 
371 
372  if (_targetPositionInSteps != _motorControler->getActualPosition()) {
373  _motorStatus = StepperMotorStatusTypes::M_NOT_IN_POSITION;
374  return StepperMotorStatusAndError(_motorStatus, _motorError);
375  }
376 
377  if (_softwareLimitsEnabled) {
378  if (_targetPositionInSteps >= _maxPositionLimitInSteps) {
379  _motorStatus = StepperMotorStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON;
380  return StepperMotorStatusAndError(_motorStatus, _motorError);
381  }
382 
383  if (_targetPositionInSteps <= _minPositionLimitInSteps) {
384  _motorStatus = StepperMotorStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON;
385  return StepperMotorStatusAndError(_motorStatus, _motorError);
386  }
387  }
388 
389  return StepperMotorStatusAndError(_motorStatus, _motorError);
390  }
