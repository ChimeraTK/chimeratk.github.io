69 void CsaStepperMotorTecRemote::initialize() {
70  this->_endSwitchesAvailable->setValue(1);
71  this->_speedControlAvailable->setValue(0);
72  this->_motorTypeId->setValue(_motorType);
73  this->setCsControlAllowed(_csControlAllowed); // just to set current allowance in Process variable
74 
75  _stepperMotorConfiurationError->setValue(0); // for this type this flag is always 0, it is used for panel logic
76 
77  _stepperMotor->requestInitialization(_address.getValue());
78 
79  _addressInUse.setValue(_stepperMotor->getName());
80  _initialized = true;
81  _isInitialized->setValue(1);
82 
83  //_logger.setChangeLogLevelCb(boost::bind(&StepperMotorPiezo::setLogLevel, this->_stepperMotor, _1));
84  _logger.init();
85  _logger.setPreamble(this->_moduleName);
86  _stepperMotor->_logger.setPreamble(this->_moduleName);
87  _stepperMotor->_logger.setLogLevel(_logger.getLogLevel());
88 
89  if(_initialized) {
90  if(_csControlAllowed) {
91  this->enableCb();
92  this->autostartCb();
93  this->enableSoftwareLimitsCb();
94 
95  this->setPositionMaxCb();
96  this->setPositionMinCb();
97 
98  this->setPositionCb();
99  }
100  _stepperMotorId->setValue(0);
101  _isInitialized->setValue(1);
102  }
103  _calibrationSuccessDate->setValue(std::time(0));
104 }
