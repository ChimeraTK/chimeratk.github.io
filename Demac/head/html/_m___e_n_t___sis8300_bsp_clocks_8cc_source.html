288 bool M_ENT_Sis8300BspClocks::checkClockStability(uint32_t channelId, uint32_t requestedFrequency) {
289  if(!_clkFrequencyMonitor) return false;
290 
291  int32_t errorCnt = 0;
292 
293  auto allowedError = _clockRequiredFreqError->getValue();
294 
295  if(allowedError == 0) allowedError = 10000;
296 
297  for(uint32_t i = 0; i < 1; i++) {
298  std::this_thread::sleep_for(std::chrono::milliseconds(200));
299  int32_t clockFreqInternal = _clkFrequencyMonitor->getFrequency(channelId);
300  uint32_t delta = std::abs(clockFreqInternal - (int)requestedFrequency);
301  _logger(Loger::DETAIL) << __func__ << ": "
302  << "Clock frequency on channel " << channelId << " is: " << clockFreqInternal
303  << " Hz. Measurement delta is: " << delta << std::endl;
304 
305  if(clockFreqInternal < 50000000) {
306  errorCnt++;
307  _logger(Loger::DETAIL) << __func__ << ": "
308  << "Clock seems be missing (fclk < 50 MHz). Error counter: " << errorCnt << std::endl;
309  }
310  else if(clockFreqInternal > 126000000) {
311  errorCnt++;
312  _logger(Loger::DETAIL) << __func__ << ": "
313  << "Clock is to fast (fclk > 126 MHz). Error counter: " << errorCnt << std::endl;
314  }
315  else {
316  if(delta > allowedError) {
317  errorCnt++;
318  _logger(Loger::DETAIL) << __func__ << ": "
319  << "Clock is unstable. Error counter: " << errorCnt << std::endl;
320  }
321  else {
322  errorCnt--;
323  _logger(Loger::DETAIL) << __func__ << ": "
324  << "Clock is stable. Error counter: " << errorCnt << std::endl;
325  }
326  }
327  }
328 
329  if(errorCnt > 0)
330  return false;
331  else
332  return true;
333 }
