67 void CsaPidCtrlOnStateSpace::uploadCtrl() {
68  _logger(Loger::DETAIL) << __func__ << ": "
69  << "Uploading PID controller." << std::endl;
70 
71  bool enabled = _ssCtrl->isEnabled();
72  bool active = _ssCtrl->isActive();
73  if(enabled && !active)
74  _logger(Loger::WARNING) << __func__ << ": "
75  << "Controller enabled, but NOT active. Check safety." << std::endl;
76  else if(active && !enabled)
77  _logger(Loger::ERROR) << __func__ << ": "
78  << "Controller NOT enabled, but active." << std::endl;
79 
80  bool reactivate = false;
81  auto ctrlLoadedType = _ssCtrl->getType();
82  if(ctrlLoadedType != this->_ctrlTypeId) {
83  _logger(Loger::WARNING) << __func__ << ": "
84  << "Space State Controller type different than PID. Change type and upload. New type: "
85  << this->_ctrlTypeId << std::endl;
86  _logger(Loger::DETAIL) << __func__ << ": "
87  << "Space State Controller type: " << ctrlLoadedType << std::endl;
88  // in type of controller is not stored yet, it means that data are not taken from hardware
89  // then it may be that this is first execution of the method
90  // in this case when controller is active, we cannot disable it, because it may unlock the system
91  if(!(ctrlLoadedType == -1 && enabled && active)) {
92  // if controller was active and enable in the beginning and controller type has been change we need to reactivate it later
93  if(enabled && active) reactivate = true;
94  _ssCtrl->setEnable(false);
95  _ssCtrl->setHold(false);
96  _ssCtrl->reset();
97  _logger(Loger::DETAIL) << __func__ << ": "
98  << "Space State Controller reset allowed and done." << std::endl;
99  }
100  else
101  _logger(Loger::DETAIL)
102  << __func__ << ": "
103  << "Space State Controller reset not allowed - controller enabled and active, type not stored in hwd."
104  << std::endl;
105 
106  _ssCtrl->setType(this->_ctrlTypeId);
107  }
108  else {
109  _logger(Loger::DETAIL) << __func__ << ": "
110  << "Space State Controller type is PID. Upload. Hold is: " << _ssCtrl->getHold()
111  << std::endl;
112  if(_holdDuringLoadingOn && _holdDuringLoadingOn->getValue()) _ssCtrl->setHold(true);
113  }
114  // disable coefficients for loading time
115  _ssCtrl->setCoefficientsValid(false);
116 
117  // Upload PI controller to Space State part
118  _ssCtrl->setParameter("A", 0, 0, 1);
119  _ssCtrl->setParameter("D", 0, 0, _pParam->getValue());
120  _ssCtrl->setParameter("B", 0, 0, _iParam->getValue());
121  _ssCtrl->setParameter("C", 0, 0, 1);
122 
123  // Upload PI controller to Pure PI part if available
124  if(_ssCtrl->isPurePICtrlAvailable()) {
125  _ssCtrl->setPurePICtrlP(_pParam->getValue());
126  _ssCtrl->setPurePICtrlI(_iParam->getValue());
127  }
128 
129  // enable coefficients after loading
130  _ssCtrl->setCoefficientsValid(true);
131 
132  // Pipe input trough as debug y2 = e
133  //_ssCtrl->setParameter("D", 1, 0, 1); -- 6.12.17 - commented on Lukasz request
134  _ssCtrl->setHold(false);
135 
136  if(reactivate) _ssCtrl->setEnable(true);
137 }
