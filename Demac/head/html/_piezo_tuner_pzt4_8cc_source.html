81 bool PiezoTunerPzt4::initialize() {
82  if(_pzt4Channel->getDacResetStatus()) {
83  _logger(Loger::WARNING) << __func__ << ": "
84  << "DAC reset enabled for PZT4 channel: " << _channelId << ". Initialization failed. "
85  << std::endl;
86  return false;
87  }
88 
89  if(_pzt4Channel->getHwdDacRangeControlOn()) {
90  _logger(Loger::WARNING) << __func__ << ": "
91  << "DAC hardware range (span) control enabled for PZT4 module. PZT4 channel: " << _channelId
92  << ". Initialization failed. " << std::endl;
93  return false;
94  }
95 
96  _logger(Loger::DETAIL) << __func__ << ": "
97  << "Starting PZT4 channel " << _channelId << " initialization for Piezo Tuner." << std::endl;
98  _pzt4Channel->setCsControlAllowed(false);
99 
100  _logger(Loger::DETAIL) << __func__ << ": "
101  << "Check status of PZT4 channel " << _channelId << "." << std::endl;
102 
103  if(this->checkIfInitNeeded()) {
104  _logger(Loger::DETAIL) << __func__ << ": "
105  << "Disabling PZT4 channel " << _channelId << "." << std::endl;
106  _pzt4Channel->setEnable(false);
107 
108  _logger(Loger::DETAIL) << __func__ << ": "
109  << "Set mode of operation to CONSTANT_M for channel: " << _channelId << "." << std::endl;
110  _pzt4Channel->setOperationMode(M_ENT_PZT4Channel::OP_MODE::CONSTANT_M);
111 
112  if(_isUnipolar) {
113  _logger(Loger::DETAIL) << __func__ << ": "
114  << "Set range of DAC to unipolar mode (0V to 10V) for channel: " << _channelId << "."
115  << std::endl;
116  _pzt4Channel->setDacRange(M_ENT_PZT4Channel::DAC_RANGE::RNG_0_TO_P10);
117  }
118  else {
119  _logger(Loger::DETAIL) << __func__ << ": "
120  << "Set range of DAC to bipolar mode (-10V to 10V) for channel: " << _channelId << "."
121  << std::endl;
122  _pzt4Channel->setDacRange(M_ENT_PZT4Channel::DAC_RANGE::RNG_N10_TO_P10);
123  }
124 
125  _logger(Loger::DETAIL) << __func__ << ": "
126  << "Set range of filter 300 Hz for channel: " << _channelId << "." << std::endl;
127  _pzt4Channel->setFilterMode(M_ENT_PZT4Channel::FILTER_MODE::HZ_300);
128 
129  _logger(Loger::DETAIL) << __func__ << ": "
130  << "Enable the channel: " << _channelId << ". Ramping aim " << getDacValueInSteps()
131  << " counts." << std::endl;
132 
133  _pzt4Channel->setEnable(true, getDacValueInSteps());
134 
135  uint32_t timeoutCounter = 0;
136  while(_pzt4Channel->getEnabled() == 0) {
137  _logger(Loger::DETAIL) << __func__ << ": "
138  << "Wait for channel to be activated." << std::endl;
139  std::this_thread::sleep_for(std::chrono::microseconds(5000));
140  timeoutCounter++;
141  if(timeoutCounter > 200) { // 2 seconds
142  _logger(Loger::WARNING) << __func__ << ": "
143  << "Wait for channel to be done TIMEOUT. Channel not initialized correctly."
144  << std::endl;
145  return false;
146  }
147  }
148 
149  if(_pzt4Channel->isActive())
150  _logger(Loger::DETAIL) << __func__ << ": "
151  << "Channel " << _channelId
152  << " of PZT4 is now active and initialized to operate the Piezo Tuner. " << std::endl;
153  else
154  _logger(Loger::DETAIL) << __func__ << ": "
155  << "Channel " << _channelId
156  << " of PZT4 is NOT active and initialized to operate the Piezo Tuner. " << std::endl;
157 
158  _logger(Loger::DETAIL) << __func__ << ": "
159  << "Channel " << _channelId
160  << " Control System input PVs will be synchronized with physical values." << std::endl;
161  _pzt4Channel->synchronize();
162  }
163  return true;
164 }
