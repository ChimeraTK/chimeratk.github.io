340 void M_RapidX_ENT_StateSpaceCtrl::rampFeedforward(double aim, double step, int delayLoopWaitMs) {
341  if(_rampingActive) {
342  _logger(Loger::WARNING) << __func__ << ": "
343  << "Ramping FF already in progress. Returning " << std::endl;
344  return;
345  }
346  _rampingActive = true;
347  if(aim > _MAX_OUT_VALUE)
348  aim = _MAX_OUT_VALUE;
349  else if(aim < -_MAX_OUT_VALUE)
350  aim = -_MAX_OUT_VALUE;
351  double currentFF = getFeedforwardValue();
352  if(step == 0) step = aim / 25.0;
353  step = std::fabs(step);
354  // ramp process
355  while(std::fabs(currentFF - aim) > 1.1 * step) {
356  if(currentFF > aim)
357  _ffValue->setValue(currentFF - step);
358  else
359  _ffValue->setValue(currentFF + step);
360  std::this_thread::sleep_for(std::chrono::milliseconds(delayLoopWaitMs));
361  currentFF = getFeedforwardValue();
362  _logger(Loger::DETAIL) << __func__ << ": "
363  << "Ramping FF: Current: " << currentFF << " aim: " << aim
364  << ", distance: " << std::fabs(currentFF - aim) << std::endl;
365  }
366  _ffValue->setValue(aim);
367  _rampingActive = false;
368 }

544 /*
545  This method ramps controller to the current FF value using ' constant controller output ' register.
546  It reset firmware safety if requested by 'resetHold' parameter.
547  It always disables the controller.
548  */
549 void M_RapidX_ENT_StateSpaceCtrl::rampCtrlToFf(bool resetHold, double step, uint32_t delayLoopWaitMs) {
550  _rampingActive = true;
551  _logger(Loger::INFO) << __func__ << ": " << _moduleName << ": Starting ramp controller." << std::endl;
552 
553  // auto step = 0.1;
554  auto ff = _ffValue->getValueFromDevice();
555  auto currentOut = _outputU->getValueFromDevice();
556 
557  _logger(Loger::INFO) << __func__ << ": " << _moduleName << ": Current output: " << currentOut
558  << ", Feedforward value (aim): " << ff << std::endl;
559 
560  _outputConstValue->setValue(currentOut); // set 'constant' to current value on output of the controller
561  _outputSelect->setValue(1); // set controller output to 'constant' register instead of 'internal-controller'
562  if(resetHold) {
563  _logger(Loger::INFO) << __func__ << ": " << _moduleName << ": Reset AOFF HOLD." << std::endl;
564  _aOffHoldReset->setValue(1); // reset hold to unblock controller
565  }
566 
567  _logger(Loger::INFO) << __func__ << ": " << _moduleName << ": Disable the controller." << std::endl;
568  _enable->setValue(0); // disable controller
569 
570  // ramp process
571  while(std::fabs(currentOut - ff) > 1.1 * step) {
572  if(currentOut > ff)
573  _outputConstValue->setValue(currentOut - step);
574  else
575  _outputConstValue->setValue(currentOut + step);
576  std::this_thread::sleep_for(std::chrono::milliseconds(delayLoopWaitMs));
577  currentOut = _outputU->getValueFromDevice();
578  _logger(Loger::DETAIL) << __func__ << ": " << _moduleName << ": Ramping: Current: " << currentOut << " aim: " << ff
579  << ", distance: " << std::fabs(currentOut - ff) << std::endl;
580  }
581  _outputConstValue->setValue(ff);
582  _logger(Loger::DETAIL) << __func__ << ": " << _moduleName << ": Ramping: Last set to: " << ff << std::endl;
583 
584  _outputSelect->setValue(0); // set controller output to 'internal-controller' again
585  _outputConstValue->setValue(0);
586  _logger(Loger::INFO) << __func__ << ": " << _moduleName << ": Finish ramp controller." << std::endl;
587  _rampingActive = false;
588 }
