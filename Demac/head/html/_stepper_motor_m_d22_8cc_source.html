53StepperMotorMD22StatusAndError StepperMotorMD22::moveToPositionInSteps(int newPositionInSteps) {
54 StepperMotorMD22StatusAndError statusAndError = this->determineMotorStatusAndError();
55 if(statusAndError.status == SMAStatusTypes::M_ERROR || statusAndError.status == SMAStatusTypes::M_DISABLED) {
56 return statusAndError;
57 }
58
59 // Ask the hardware to move the motor.
60 this->setTargetPositionInSteps(newPositionInSteps);
61 if(!this->getAutostart()) {
62 this->start();
63 }
64
65 // Block till the motor has finished moving.
66 do {
67 usleep(1000);
68 } while(StepperMotorMD22::isMoving());
69
70 if(_useEncoderAsPositionSource) {
71 do {
72 this->determineMotorStatusAndError();
73 std::this_thread::sleep_for(std::chrono::milliseconds(100));
74 } while(_motorStatus == SMAStatusTypes::M_IN_MOVE);
75 }
76
77 return (this->determineMotorStatusAndError());
78}

390 int position = truncateStepsToMaxRange(newPosition);

406 else if(position < _minPositionLimitInSteps) {
407 position = _minPositionLimitInSteps;
408 }
409
410 return position;

413StepperMotorMD22StatusAndError StepperMotorMD22::determineMotorStatusAndError() {
414 boost::lock_guard<boost::mutex> guard(_motorControlerMutex);
415
416 _motorStatus = SMAStatusTypes::M_OK;
417 _motorError = SMAErrorTypes::M_NO_ERROR;
418
419 // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
420 // ERROR CONDITION CHECK - it should be done first.
421 // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
422
423 if(_softwareLimitsEnabled && (_maxPositionLimitInSteps <= _minPositionLimitInSteps)) {
424 _motorStatus = SMAStatusTypes::M_ERROR;
425 _motorError = SMAErrorTypes::M_CONFIG_ERROR_MIN_POS_GRATER_EQUAL_TO_MAX;
426
427 return {_motorStatus, _motorError};
428 }
429
430 // first check if disabled
431 // need to check whether "automatic holding current OFF" feature is enabled
432 // only then one can use status from MotorController
433 if(!_holdCurrOffWhenNotMoving && !_motorControler->isMotorCurrentEnabled()) {
434 _motorStatus = SMAStatusTypes::M_DISABLED;
435 return {_motorStatus, _motorError};
436 }
437
438 if(!_md22IsEnabled) {
439 _motorStatus = SMAStatusTypes::M_DISABLED;
440 return {_motorStatus, _motorError};
441 }
442
443 // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
444 // STATUS CHECK
445 // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
446
447 if(_motorControler->isMotorMoving()) {
448 _motorStatus = SMAStatusTypes::M_IN_MOVE;
449 return {_motorStatus, _motorError};
450 }
451
452 if(_useEncoderAsPositionSource && _encoderInMove) {
453 _motorStatus = SMAStatusTypes::M_IN_MOVE;
454 return {_motorStatus, _motorError};
455 }
456
457 if((_useEncoderAsPositionSource && _encoderTargetPosition != _targetPositionInSteps) ||
458 (!_useEncoderAsPositionSource && _targetPositionInSteps != getActualPosition())) {
459 _motorStatus = SMAStatusTypes::M_NOT_IN_POSITION;
460 return {_motorStatus, _motorError};
461 }
462
463 if(_softwareLimitsEnabled) {
464 if(_targetPositionInSteps >= _maxPositionLimitInSteps) {
465 _motorStatus = SMAStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON;
466 return {_motorStatus, _motorError};
467 }
468
469 if(_targetPositionInSteps <= _minPositionLimitInSteps) {
470 _motorStatus = SMAStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON;
471 return {_motorStatus, _motorError};
472 }
473 }
474
475 return {_motorStatus, _motorError};
476}

555void StepperMotorMD22::encoderThread() {
556 _logger(Loger::INFO) << __func__ << ": " << "Start encoder thread." << std::endl;
557 // delay thread by 1 second to be sure that encoder readout are working - I do not know how to check if readout is valid
558 std::this_thread::sleep_for(std::chrono::milliseconds(1000));
559
560 StatisticsForStdVector<unsigned int> stats;
561 stats.setEnableMean(true);
562 stats.setEnableStdDev(true);
563
564 int cnt = 0;
565 int currentMotorControllerPos = 0, targetMotorControllerPos = 0;
566 bool triggerAutoCorrection = false;
567 while(!_stopEncoderThread) {
568 if(!_pauseEncoderThread) {
569 _encoderPosAvgBuffer.add(_motorControler->getDecoderPosition());
570 auto vec = _encoderPosAvgBuffer.toStdVector();
571 stats.setVector(vec);
572 stats.calculate();
573 _encoderPositionAveraged = stats.getMean();
574
575 _logger(Loger::MAX_DETAIL) << __func__ << ": " << " Buff size: " << _encoderPosAvgBuffer.getCurrentLength()
576 << " encoderPositionAveraged " << _encoderPositionAveraged << std::endl;
577
578 if(_useEncoderAsPositionSource) {
579 _encoderCtrlError = int(_encoderTargetPosition - _encoderPositionAveraged);
580 currentMotorControllerPos = _motorControler->getActualPosition();
581
582 if(this->getEnabled()) {
583 if(!_encoderInMove && !triggerAutoCorrection && _encoderMaxCtrlAllowedError > 0 &&
584 _encoderMaxCtrlAllowedError > std::fabs(_encoderToMicrostepsConvertion) &&
585 std::abs(_encoderCtrlError) > _encoderMaxCtrlAllowedError) {
586 _logger(Loger::INFO) << __func__ << ": " << "Encoder ctrl error crossed. Error: " << _encoderCtrlError
587 << " , allowed error: " << _encoderMaxCtrlAllowedError
588 << " . Trigger automatic correction. " << std::endl;
589 triggerAutoCorrection = true;
590 }
591
592 if(((_encoderInMove || triggerAutoCorrection) && targetMotorControllerPos == currentMotorControllerPos) ||
593 _encoderControllerForcedExecute) {
594 if(_encoderCtrOutput) { // this condition is needed to clean buffer when stepping of motor is done. After
595 // movement we must calculate fresh average in a new motor position.
596 _encoderPosAvgBuffer.clear();
597 _encoderCtrOutput = 0;
598 }
599 else if(_encoderPosAvgBuffer.isFull()) { // controller should be only executed when buffer is full. Then we
600 // have average which filters out the jitter of the readout
601 if(_encoderCtrlError > _encoderToMicrostepsConvertion)
602 _encoderCtrOutput = (int)((1.0 / _encoderToMicrostepsConvertion) * _encoderCtrlError);
603 else
604 _encoderCtrOutput = std::round((1.0 / _encoderToMicrostepsConvertion) * _encoderCtrlError);
605 targetMotorControllerPos = currentMotorControllerPos + _encoderCtrOutput;
606
607 if(truncateStepsToMaxRangeNeeded(targetMotorControllerPos)) {
608 currentMotorControllerPos = 0;
609 targetMotorControllerPos = _encoderCtrOutput;
610 _motorControler->setActualPosition(currentMotorControllerPos);
611 }
612
613 if(!_motorControler->isMotorCurrentEnabled()) {
614 _logger(Loger::INFO) << __func__ << ": "
615 << "Switch ON holding current before movement in encoder mode." << std::endl;
616 _motorControler->setMotorCurrentEnabled(true);
617 std::this_thread::sleep_for(std::chrono::milliseconds(10));
618 }
619 _motorControler->setTargetPosition(targetMotorControllerPos);
620 _encoderControllerForcedExecute = false;
621
622 _logger(Loger::DETAIL) << __func__ << ": " << ++cnt << ") set: " << (int)_encoderTargetPosition
623 << " , now: " << (int)_encoderPositionAveraged
624 << " , ctrl_err: " << (int)_encoderCtrlError
625 << " , ctrl_out: " << _encoderCtrOutput << " , tMC: " << targetMotorControllerPos
626 << " , cMC " << currentMotorControllerPos
627 << " , delta: " << (targetMotorControllerPos - currentMotorControllerPos)
628 << std::endl;
629 }
630 }
631 // else if (targetMotorControllerPos != currentMotorControllerPos) {
632 // _logger(Loger::MAX_DETAIL) << __func__ << ": " << "Wait motor to finish movement. Current: " <<
633 // currentMotorControllerPos << " target: " << targetMotorControllerPos << std::endl;
634 // }
635
636 if(_encoderPosAvgBuffer.isFull() && _encoderCtrOutput == 0) {
637 if(_encoderInMove) {
638 _encoderInMove = false;
639 }
640 if(triggerAutoCorrection) {
641 triggerAutoCorrection = false;
642 }
643 cnt = 0;
644 }
645 }
646 }
647 std::this_thread::sleep_for(std::chrono::milliseconds(10));
648 }
649 else {
650 _encoderPosAvgBuffer.clear();
651 _encoderCtrOutput = 0;
652 std::this_thread::sleep_for(std::chrono::milliseconds(100));
653 }
654
655 if(_holdCurrOffWhenNotMoving) {
656 if(_motorControler->isMotorCurrentEnabled()) {
657 // encoder is used as source of position
658 if(_useEncoderAsPositionSource && !_encoderInMove && !triggerAutoCorrection) {
659 _logger(Loger::INFO) << __func__ << ": "
660 << "Automatic holding current disabling is ON. Encoder movement has finised. Switch "
661 "OFF holding current."
662 << std::endl;
663 _motorControler->setMotorCurrentEnabled(false);
664 }
665 // motor is driven by steps internally by MD22 MotorController
666 else if(_targetPositionInSteps == getActualPosition() && !_motorControler->isMotorMoving() &&
667 _motorControler->isMotorCurrentEnabled()) {
668 _logger(Loger::INFO)
669 << __func__ << ": "
670 << "Automatic holding current disabling is ON. Movement has finised. Switch OFF holding current."
671 << std::endl;
672 _motorControler->setMotorCurrentEnabled(false);
673 }
674 }
675 }
676 else if(_md22IsEnabled && !_motorControler->isMotorCurrentEnabled()) {
677 _logger(Loger::INFO) << __func__ << ": "
678 << "Automatic holding current disabling is OFF, motor is ENABLED, but holding current is "
679 "OFF. Switch ON holding current."
680 << std::endl;
681 _motorControler->setMotorCurrentEnabled(true);
682 }
683 }
684 _logger(Loger::INFO) << __func__ << ": " << "Stop encoder thread." << std::endl;
685}
