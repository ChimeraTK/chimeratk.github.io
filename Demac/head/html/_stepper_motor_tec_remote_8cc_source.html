365void StepperMotorTecRemote::updateFromDevice() {
366 auto time_start = std::chrono::high_resolution_clock::now();
367 try {
368 if(_triggerSynchInit) {
369 initializationCore();
370 _triggerSynchInit = false;
371 }
372
373 bool remoteDataLoaded = getRemoteData();
374 evaluateStatusAndError();
375 if(_waitForRemoteDataAndStatusEvaluation && remoteDataLoaded) _waitForRemoteDataAndStatusEvaluation = false;
376
377 // FIX ME - execute every 10 updates, not every update
378 // when communication exceptions occurred, try to reinitialize
379 if(!_initialized) initializationCore();
380
381 if(!_initialized || !_enableFlag) {
382 _sendStartCommand = false;
383 _sendStopCommand = false;
384 return;
385 }
386
387 //'send start command' should be executed always after 'send target position'
388 if(_sendStartCommand) {
389 _targetPositionRemote.setValue(_targetPositionLocal);
390 _setPositionWriteCommand.setValue(WRITE_TARGET_TEMP_CMD);
391 _sendStartCommand = false;
392 _targetPositionWasSend = true;
393 }
394
395 if(_sendStopCommand) {
396 _targetPositionLocal = _positionLocal = _positionRemote.getValue();
397 _targetPositionRemote.setValue(_targetPositionLocal);
398 _targetPositionReadbackLocal = _targetPositionReadbackRemote.getValue();
399 _setPositionWriteCommand.setValue(WRITE_TARGET_TEMP_CMD);
400 _sendStopCommand = false;
401 }
402 }
403 catch(CsaExternalProcessException& ex) {
404 _initialized = false;
405 }
406
407 auto time_stop = std::chrono::high_resolution_clock::now();
408 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(time_stop - time_start).count();
409 _logger(Loger::MAX_DETAIL) << __func__ << ": "
410 << "Update takes: " << duration << std::endl;
411}

413bool StepperMotorTecRemote::getRemoteData() {
414 try {
415 if(!_initialized) return false;
416
417 auto now =
418 std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch());
419 auto timeFromLastUpdateMs =
420 std::chrono::duration_cast<std::chrono::milliseconds>(now - this->_lastUpdateTime).count();
421 if(timeFromLastUpdateMs < _periodicReadbackPeriodLocal) return false;
422
423 _positionLocal = _positionRemote.getValue();
424 _targetPositionReadbackLocal = _targetPositionReadbackRemote.getValue();
425
426 _communicationStatusLocal = _communicationStatus.getValue();
427 _driverStatusLocal = _driverStatusRemote.getValue();
428 _isTemperatureStableLocal = _isTemperatureStableRemote.getValue();
429
430 _periodicReadbackConfigMonCh1Local = _periodicReadbackConfigMonCh1Remote.getValue();
431 _periodicReadbackConfigMonLocal = _periodicReadbackConfigMonRemote.getValue();
432 _allowedTemperatureDeviationLocal = _allowedTemperatureDeviationRemote.getValue();
433 _periodicReadbackPeriodLocal = _periodicReadbackPeriodRemote.getValue();
434
435 _lastUpdateTime = now;
436 }
437 catch(CsaExternalProcessException& ex) {
438 _initialized = false;
439 }
440 return _initialized;
441}

443void StepperMotorTecRemote::evaluateStatusAndError() {
444 if(!_enableFlag) {
445 _statusLocal = SMAStatusTypes::M_DISABLED;
446 _errorLocal = SMAErrorTypes::M_NO_ERROR;
447 return;
448 }
449
450 if(_enableSoftwareLimits && _softLimitMax <= _softLimitMin) {
451 _statusLocal = SMAStatusTypes::M_ERROR;
452 _errorLocal = SMAErrorTypes::M_CONFIG_ERROR_MIN_POS_GRATER_EQUAL_TO_MAX;
453 return;
454 }
455
456 if(!_initialized) {
457 _statusLocal = SMAStatusTypes::M_ERROR;
458 _errorLocal = SMAErrorTypes::M_COMMUNICATION_LOST;
459 _logger(Loger::INFO) << __func__ << ": "
460 << "Communication with TEC location lost. Check address and status of TEC location."
461 << std::endl;
462 return;
463 }
464
465 if(_communicationStatusLocal == 1) {
466 _statusLocal = SMAStatusTypes::M_ERROR;
467 _errorLocal = SMAErrorTypes::M_COMMUNICATION_LOST;
468 _logger(Loger::ERROR) << __func__ << ": "
469 << "Communication status of TEC location is ERROR. No communication with TEC hardware."
470 << std::endl;
471 return;
472 }
473
474 if(_communicationStatusLocal == 2) {
475 _statusLocal = SMAStatusTypes::M_ERROR;
476 _errorLocal = SMAErrorTypes::M_CONFIG_ERROR;
477 _logger(Loger::ERROR)
478 << __func__ << ": "
479 << "Communication status of TEC location is DISABLED. Heartbeat not enabled. Change setting of TEC location."
480 << std::endl;
481 return;
482 }
483
484 if(((_periodicReadbackConfigMonCh1Local & 5) != 5) || ((_periodicReadbackConfigMonLocal & 16416) != 16416)) {
485 _statusLocal = SMAStatusTypes::M_ERROR;
486 _errorLocal = SMAErrorTypes::M_CONFIG_ERROR;
487 _logger(Loger::ERROR)
488 << __func__ << ": "
489 << "Periodic read of TEC location misconfigured. Enable periodic reads for 'Object Temperature' , 'Target "
490 "Object Temperature', 'Driver status' and 'Temperture is stable' indicator."
491 << std::endl;
492 return;
493 }
494
495 /*
496 3.3.2.12 Driver Status
497 ID Name Format Value Range Description
498 1080 Driver Status INT32 ..
499 0: Init
500 1: Ready
501 2: Run
502 3: Error
503 4: Bootloader
504 5: Device will Reset within the next 200ms
505 */
506 if(_driverStatusLocal != 2) {
507 _statusLocal = SMAStatusTypes::M_ERROR;
508 _errorLocal = SMAErrorTypes::M_HARDWARE_ERROR;
509 _logger(Loger::INFO) << __func__ << ": "
510 << "TEC driver status different than 'Run'. TEC driver status ID: " << _driverStatusLocal
511 << ". Check TEC hardware." << std::endl;
512 return;
513 }
514 /*
515 1200 Temperature is Stable INT32 ..
516 0: Temperature regulation is not active
517 1: Is not stable
518 2: Is stable
519 */
520 if(_isTemperatureStableLocal == 0) {
521 _statusLocal = SMAStatusTypes::M_ERROR;
522 _errorLocal = SMAErrorTypes::M_HARDWARE_ERROR;
523 return;
524 }
525
526 if(_isTemperatureStableLocal == 1) {
527 _statusLocal = SMAStatusTypes::M_IN_MOVE;
528 _errorLocal = SMAErrorTypes::M_NO_ERROR;
529 if(_targetPositionWasSend) _targetPositionWasSend = false;
530
531 return;
532 }
533
534 if(_targetPositionWasSend && std::fabs(_positionLocal - _targetPositionLocal) > _allowedTemperatureDeviationLocal) {
535 _statusLocal = SMAStatusTypes::M_IN_MOVE;
536 _errorLocal = SMAErrorTypes::M_NO_ERROR;
537 return;
538 }
539
540 if(std::fabs(_targetPositionReadbackLocal - _targetPositionLocal) > 0.0001) {
541 _statusLocal = SMAStatusTypes::M_NOT_IN_POSITION;
542 _errorLocal = SMAErrorTypes::M_NO_ERROR;
543 return;
544 }
545
546 if(_enableSoftwareLimits) {
547 if(_positionLocal >= _softLimitMax) {
548 _statusLocal = SMAStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON;
549 _errorLocal = SMAErrorTypes::M_NO_ERROR;
550 return;
551 }
552
553 if(_positionLocal <= _softLimitMin) {
554 _statusLocal = SMAStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON;
555 _errorLocal = SMAErrorTypes::M_NO_ERROR;
556 return;
557 }
558 }
559
560 if(_positionLocal >= RNG_TEMP_MAX) {
561 _statusLocal = SMAStatusTypes::M_POSITIVE_END_SWITCHED_ON;
562 _errorLocal = SMAErrorTypes::M_NO_ERROR;
563 return;
564 }
565
566 if(_positionLocal <= RNG_TEMP_MIN) {
567 _statusLocal = SMAStatusTypes::M_NEGATIVE_END_SWITCHED_ON;
568 _errorLocal = SMAErrorTypes::M_NO_ERROR;
569 return;
570 }
571
572 _statusLocal = SMAStatusTypes::M_OK;
573 _errorLocal = SMAErrorTypes::M_NO_ERROR;
574}

576SMAStatus StepperMotorTecRemote::moveToPositionCore(float newPosition) {
577 _finishBlockingFunctions = false;
578
579 if(!_initialized) {
580 _statusLocal = SMAStatusTypes::M_ERROR;
581 _errorLocal = SMAErrorTypes::M_COMMUNICATION_LOST;
582 return _statusLocal;
583 }
584 _waitForRemoteDataAndStatusEvaluation = true;
585 while(_waitForRemoteDataAndStatusEvaluation) {
586 std::this_thread::sleep_for(std::chrono::milliseconds(1));
587 if(!_initialized) return SMAStatusTypes::M_ERROR;
588 if(_finishBlockingFunctions) return SMAStatusTypes::M_OK;
589 }
590 _logger(Loger::DETAIL) << __func__ << ": "
591 << "Current status " << _statusLocal << std::endl;
592 if(_statusLocal == SMAStatusTypes::M_ERROR || _statusLocal == SMAStatusTypes::M_DISABLED) return _statusLocal;
593
594 _logger(Loger::DETAIL) << __func__ << ": "
595 << "Trigger start command." << std::endl;
596 this->start(newPosition);
597 while(_sendStartCommand) {
598 std::this_thread::sleep_for(std::chrono::milliseconds(1));
599 if(_finishBlockingFunctions) return SMAStatusTypes::M_OK;
600 }
601 _logger(Loger::DETAIL) << __func__ << ": "
602 << "New position send. Wait 2x TEC location update: " << 2 * _periodicReadbackPeriodLocal
603 << "ms." << std::endl;
604 std::this_thread::sleep_for(std::chrono::milliseconds(2 * _periodicReadbackPeriodLocal));
605 do {
606 _waitForRemoteDataAndStatusEvaluation = true;
607 while(_waitForRemoteDataAndStatusEvaluation) {
608 std::this_thread::sleep_for(std::chrono::milliseconds(1));
609 if(!_initialized) return SMAStatusTypes::M_ERROR;
610 if(_finishBlockingFunctions) return SMAStatusTypes::M_OK;
611 }
612 _logger(Loger::DETAIL) << __func__ << ": "
613 << "In loop: Current status: " << _statusLocal.asString()
614 << " , position: " << _positionLocal << " target: " << _targetPositionReadbackLocal
615 << " delta: " << std::fabs(_positionLocal - _targetPositionReadbackLocal) << std::endl;
616 if(_statusLocal == SMAStatusTypes::M_DISABLED || _statusLocal == SMAStatusTypes::M_ERROR) return _statusLocal;
617
618 if(_statusLocal == SMAStatusTypes::M_IN_MOVE) {
619 std::this_thread::sleep_for(std::chrono::milliseconds(1000));
620 }
621 else if(_statusLocal == SMAStatusTypes::M_NOT_IN_POSITION) {
622 _logger(Loger::DETAIL) << __func__ << ": "
623 << "SMAStatusTypes::M_NOT_IN_POSITION reported. Re-send start command." << std::endl;
624 this->start(newPosition);
625 }
626 else if(std::fabs(_positionLocal - _targetPositionReadbackLocal) < 0.02) {
627 return _statusLocal;
628 }
629 } while(!_finishBlockingFunctions);
630
631 return SMAStatusTypes::M_OK;
632}
