300 template<typename DataType>
301 void CsaRegisterAccesor<DataType>::determineCreatedPv(bool& createToControlSystem, bool& createFromControlSystem) {
302  this->_lastUpdateTime = std::chrono::milliseconds::zero();
303 
304  // check if name is not empty
305  if(_nameInControlSystem.compare("") == 0) {
306  CsaAdapter::getInstance().globalData.getLogger()(Loger::INFO)
307  << __func__ << ": " << "Control System name is empty. No Control System variables will be created."
308  << std::endl;
309  createFromControlSystem = false;
310  createToControlSystem = false;
311  this->_csControlAllowed = false;
312  }
313  else if(this->_controlSystemAccessMode == CsAccessMode::NO_ACCESS) {
314  createFromControlSystem = false;
315  createToControlSystem = false;
316  this->_csControlAllowed = false;
317  }
318  else if(this->_deviceAccessMode == DevAccessMode::READ_WRITE) {
319  // std::cout << "CsaRegisterAccesor<DataType>::CsaRegisterAccesor: INFO: Create read/write register: " << regName <<
320  // " as: " << nameForCs << std::endl;
321  if(this->_controlSystemAccessMode == CsAccessMode::READ_WRITE) {
322  createToControlSystem = true;
323  createFromControlSystem = true;
324  }
325  else if(this->_controlSystemAccessMode == CsAccessMode::READ_ONLY) {
326  createToControlSystem = true;
327  this->_csControlAllowed = false;
328  }
329  else if(this->_controlSystemAccessMode == CsAccessMode::WRITE_ONLY) {
330  createFromControlSystem = true;
331  }
332  }
333  else if(this->_deviceAccessMode == DevAccessMode::READ_ONLY &&
334  this->_controlSystemAccessMode != CsAccessMode::WRITE_ONLY) {
335  // std::cout << "CsaRegisterAccesor<DataType>::CsaRegisterAccesor: INFO: Create read only register: " << regName <<
336  // " as: " << nameForCs <<std::endl;
337  createToControlSystem = true;
338  }
339  else if(this->_deviceAccessMode == DevAccessMode::WRITE_ONLY &&
340  this->_controlSystemAccessMode != CsAccessMode::READ_ONLY) {
341  // std::cout << "CsaRegisterAccesor<DataType>::CsaRegisterAccesor: INFO: Create write only register: " << regName <<
342  // " as: " << nameForCs <<std::endl;
343  createFromControlSystem = true;
344  }
345  else {
346  if(this->_deviceAccessMode == DevAccessMode::READ_ONLY &&
347  this->_controlSystemAccessMode == CsAccessMode::WRITE_ONLY) {
348  CsaAdapter::getInstance().globalData.getLogger()()
349  << __func__ << ": " << this->_nameInControlSystem
350  << " -> Device Access Mode defined as: READ_ONLY and Control System Access Mode defined as WRITE_ONLY. "
351  "Communication direction collision. No PV will be created."
352  << std::endl;
353  }
354  else if(this->_deviceAccessMode == DevAccessMode::WRITE_ONLY &&
355  this->_controlSystemAccessMode == CsAccessMode::READ_ONLY) {
356  CsaAdapter::getInstance().globalData.getLogger()()
357  << __func__ << ": " << this->_nameInControlSystem
358  << " -> Device Access Mode defined as: WRITE_ONLY and Control System Access Mode defined as READ_ONLY. "
359  "Communication direction collision. No PV will be created."
360  << std::endl;
361  }
362 
363  createFromControlSystem = false;
364  createToControlSystem = false;
365  }
366 
367  updateFromDeviceCnt = 0;
368 }

553 template<typename DataType>
554 bool CsaRegisterAccesor<DataType>::setValueInternal(DataType value, bool executeCallback) {
555  if(_isConstantFromFile) return false;
556 
557  if(this->_deviceAccessMode == DevAccessMode::READ_ONLY) return false;
558 
559  if(!this->_device)
560  throw D_AdvancedRegisterAccessorException(
561  "CsaRegisterAccesor::setValueInternal: Exception detected. Device is NULL.");
562 
563  if(!this->_deviceInitDone)
564  throw D_AdvancedRegisterAccessorException(
565  "CsaRegisterAccesor::setValueInternal: Exception detected. Device not initialized.");
566 
567  if(this->_limiterCallback) value = this->_limiterCallback(value);
568 
569  if(this->_converter) value = this->_converter->convert(value);
570 
571  CsaAdapter::getInstance().globalData.getLogger()(Loger::DETAIL)
572  << __func__ << ": " << "Set value: " << value << " to register: " << _registerName
573  << " represented as: " << _nameInControlSystem << std::endl;
574 
575  // define scope for mutex guard
576  {
577  std::lock_guard<std::mutex> guard(_deviceAccessorMutex);
578  _currentValueRegister = ((DataType)value);
579  readWriteHwd(
580  true); // physical write to hardware with exception handling, replacement for: _currentValueRegister.write();
581  this->_lastUpdateTime = CsaAdapter::getInstance().globalData.getGlobalTimeMillisec();
582  }
583  // end of scope for mutex guard
584  if(executeCallback && this->_writeCallback) this->_writeCallback();
585 
586  return true;
587 }

597 template<typename DataType>
598 DataType CsaRegisterAccesor<DataType>::getValueFromDevice() {
599  if(_isConstantFromFile) {
600  if(_csaDataToCs) _csaDataToCs->setValueNoCallback(_constantFromFileValue);
601  return _constantFromFileValue;
602  }
603  // local copy for multi-threads access. Each thread will have its own copy when calling the method
604  DataType localDataCopy = 0;
605  try {
606  if(this->_device && this->_deviceInitDone) {
607  // define scope for mutex
608  {
609  std::lock_guard<std::mutex> guard(_deviceAccessorMutex);
610  // store old value from previous read in case read triggers boost::numeric::negative_overflow or
611  // boost::numeric::positive_overflow
612  localDataCopy = _currentValueRegister;
613  // physical read from hardware with exception handling
614  readWriteHwd(false);
615  // this is also accessing a device in case read data is '-1'
616  checkDeviceAvailability();
617  // in case converter is set, pass data to do a conversion. We can pass _currentValueRegister protected by
618  // mutex, because unconvert() takes a copy (not a pointer or reference)
619  if(this->_converter) _currentValueRegister = this->_converter->unconvert(_currentValueRegister);
620  localDataCopy = _currentValueRegister;
621  this->_lastUpdateTime = CsaAdapter::getInstance().globalData.getGlobalTimeMillisec();
622  }
623  // end of scope for mutex guard
624  // send data to Control System if object available
625  if(_csaDataToCs) _csaDataToCs->setValueNoCallback(localDataCopy);
626  }
627  else {
628  if(this->_deviceInitDone == false)
629  throw D_AdvancedRegisterAccessorException(
630  "CsaRegisterAccesor::getValueFromDevice: Exception detected. Device not initialized for " +
631  this->_registerName);
632  else
633  throw D_AdvancedRegisterAccessorException(
634  "CsaRegisterAccesor::getValueFromDevice: Exception detected. Device is NULL for " + this->_registerName);
635  }
636  }
637  catch(boost::numeric::negative_overflow& e) {
638  CsaAdapter::getInstance().globalData.getLogger()(Loger::ERROR)
639  << __func__ << ": " << e.what() << " exception on: " << this->_registerName
640  << ", module name: " << this->_mapModuleName << std::endl;
641  }
642  catch(boost::numeric::positive_overflow& e) {
643  CsaAdapter::getInstance().globalData.getLogger()(Loger::ERROR)
644  << __func__ << ": " << e.what() << " exception on: " << this->_registerName
645  << ", module name: " << this->_mapModuleName << std::endl;
646  }
647  return localDataCopy;
648 }

651 void CsaRegisterAccesor<DataType>::checkDeviceAvailability() {
652  if(_checkDevAvailableWhenNegativeOne && _currentValueRegister == static_cast<DataType>(-1)) {
653  try {
654  int controlData =
655  _device->read<int>(_checkDevAvailablePath, ChimeraTK::AccessModeFlags(_checkDevAvailableRawAccessMode));
656  if(controlData == -1)
657  throw D_AdvancedRegisterAccessorException("CsaRegisterAccesor::checkDeviceAvailability: Exception detected. "

659  }

662  }

664  try {
665  if(!this->_device->isOpened())
666  this->_device->open();
667  else
668  throw ex;
669  }
670  catch(ChimeraTK::runtime_error& ex_reopen) {

672  }
