108void CsaStepperMotorSawPiezoPZT4::initialize(boost::shared_ptr<M_BSP_PZT4> pzt4Board, uint32_t channelId) {
109 _logger.setChangeLogLevelCb(boost::bind(&StepperMotorSawPiezoPZT4::setLogLevel, this->_stepperMotor, _1));
110 _logger.init();
111 //_logger.setPreamble("ActPiezoMotor::" + this->_moduleName);
112 _logger.setPreamble(this->_moduleName);
113
114 _stepperMotorConfiurationError->setValue(1);
115 _initialized = false;
116 _isInitialized->setValue(0);
117
118 pztLowPassFilterCb();
119 motorModelCb(); // motor model needs to be set before _stepperMotor->initialize, it must be know during init process
120 this->_stepperMotorUnitsConverter->settingChangedReset();
121 this->_stepperMotor->initialize(pzt4Board, channelId, _position->getValue(), this->_stepperMotorUnitsConverter);
122 _initialized = this->_stepperMotor->getInitialized();
123
124 if(_initialized) {
125 _stepperMotorConfiurationError->setValue(0);
126 if(_csControlAllowed) {
127 this->enableCb();
128 this->autostartCb();
129 this->enableSoftwareLimitsCb();
130
131 this->setPositionMaxCb();
132 this->setPositionMinCb();
133
134 this->parameterCb();
135 }
136 // std::cout << "CsaStepperMotorSawPiezoPZT4::init . Current value of enable flag: " << _enable->getValue() << " cs
137 // system allowed: " << _csControlAllowed << " !" << std::endl;
138 _stepperMotorId->setValue(channelId);
139 _isInitialized->setValue(1);
140 }
141
142 this->_endSwitchesAvailable->setValue(0);
143 this->_speedControlAvailable->setValue(0);
144 this->_motorTypeId->setValue(_motorType);
145 this->setCsControlAllowed(_csControlAllowed); // just to set current allowance in Process variable
146
147 this->_pztChannelNo->setValue(channelId + 1);
148}
