547 void StepperMotorMD22::encoderThread() {
548  _logger(Loger::INFO) << __func__ << ": " << "Start encoder thread." << std::endl;
549  // delay thread by 1 second to be sure that encoder readout are working - I do not know how to check if readout is valid
550  std::this_thread::sleep_for(std::chrono::milliseconds(1000));
551 
552  StatisticsForStdVector<unsigned int> stats;
553  stats.setEnableMean(true);
554  stats.setEnableStdDev(true);
555 
556  int cnt = 0;
557  int currentMotorControllerPos = 0, targetMotorControllerPos = 0;
558  bool triggerAutoCorrection = false;
559  while(!_stopEncoderThread) {
560  if(!_pauseEncoderThread) {
561  _encoderPosAvgBuffer.add(_motorControler->getDecoderPosition());
562  auto vec = _encoderPosAvgBuffer.toStdVector();
563  stats.setVector(vec);
564  stats.calculate();
565  _encoderPositionAveraged = stats.getMean();
566 
567  _logger(Loger::MAX_DETAIL) << __func__ << ": " << "Test Buff size: " << _encoderPosAvgBuffer.getCurrentLength()
568  << " encoderPositionAveraged " << _encoderPositionAveraged << std::endl;
569 
570  if(_useEncoderAsPositionSource) {
571  _encoderCtrlError = int(_encoderTargetPosition - _encoderPositionAveraged);
572  currentMotorControllerPos = _motorControler->getActualPosition();
573 
574  if(this->getEnabled()) {
575  if(!_encoderInMove && !triggerAutoCorrection && _encoderMaxCtrlAllowedError > 0 &&
576  _encoderMaxCtrlAllowedError > std::fabs(_encoderToMicrostepsConvertion) &&
577  std::abs(_encoderCtrlError) > _encoderMaxCtrlAllowedError) {
578  _logger(Loger::INFO) << __func__ << ": " << "Encoder ctrl error crossed. Error: " << _encoderCtrlError
579  << " , allowed error: " << _encoderMaxCtrlAllowedError
580  << " . Trigger automatic correction. " << std::endl;
581  triggerAutoCorrection = true;
582  }
583 
584  if(((_encoderInMove || triggerAutoCorrection) && targetMotorControllerPos == currentMotorControllerPos) ||
585  _encoderControllerForcedExecute) {
586  if(_encoderCtrOutput) { // this condition is needed to clean buffer when stepping of motor is done. After
587  // movement we must calculate fresh average in a new motor position.
588  _encoderPosAvgBuffer.clear();
589  _encoderCtrOutput = 0;
590  }
591  else if(_encoderPosAvgBuffer.isFull()) { // controller should be only executed when buffer is full. Then we
592  // have average which filters out the jitter of readout
593  if(_encoderCtrlError > _encoderToMicrostepsConvertion)
594  _encoderCtrOutput = (int)((1.0 / _encoderToMicrostepsConvertion) * _encoderCtrlError);
595  else
596  _encoderCtrOutput = std::round((1.0 / _encoderToMicrostepsConvertion) * _encoderCtrlError);
597  targetMotorControllerPos = currentMotorControllerPos + _encoderCtrOutput;
598 
599  if(truncateStepsToMaxRangeNeeded(targetMotorControllerPos)) {
600  currentMotorControllerPos = 0;
601  targetMotorControllerPos = _encoderCtrOutput;
602  _motorControler->setActualPosition(currentMotorControllerPos);
603  }
604  _motorControler->setTargetPosition(targetMotorControllerPos);
605  _encoderControllerForcedExecute = false;
606 
607  _logger(Loger::DETAIL) << __func__ << ": " << ++cnt << ") set: " << (int)_encoderTargetPosition
608  << " , now: " << (int)_encoderPositionAveraged
609  << " , ctrl_err: " << (int)_encoderCtrlError
610  << " , ctrl_out: " << _encoderCtrOutput << " , tMC: " << targetMotorControllerPos
611  << " , cMC " << currentMotorControllerPos
612  << " , delta: " << (targetMotorControllerPos - currentMotorControllerPos)
613  << std::endl;
614  }
615  }
616  // else if (targetMotorControllerPos != currentMotorControllerPos) {
617  // _logger(Loger::MAX_DETAIL) << __func__ << ": " << "Wait motor to finish movement. Current: " <<
618  // currentMotorControllerPos << " target: " << targetMotorControllerPos << std::endl;
619  // }
620 
621  if(_encoderPosAvgBuffer.isFull() && _encoderCtrOutput == 0) {
622  if(_encoderInMove) {
623  _encoderInMove = false;
624  }
625  if(triggerAutoCorrection) {
626  triggerAutoCorrection = false;
627  }
628  cnt = 0;
629  }
630  }
631  }
632  std::this_thread::sleep_for(std::chrono::milliseconds(10));
633  }
634  else {
635  _encoderPosAvgBuffer.clear();
636  _encoderCtrOutput = 0;
637  std::this_thread::sleep_for(std::chrono::milliseconds(1000));
638  }
639  }
640  _logger(Loger::INFO) << __func__ << ": " << "Stop encoder thread." << std::endl;
641 }
