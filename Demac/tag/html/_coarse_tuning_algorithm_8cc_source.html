76void CoarseTuningAlgorithm::checkTuning() {
77 if(!_detector) {
78 _logger() << __func__ << ": " << _logIdString << ": _detector not set. exit(1)." << std::endl;
79 exit(1);
80 return;
81 }
82 if(!_actuator) {
83 _logger() << __func__ << ": " << _logIdString << ": _actuator not set. exit(1)." << std::endl;
84 exit(1);
85 return;
86 }
87
88 if(!_enabled) {
89 // std::cout << "disabled";
90 if(_tuningInProgress) {
91 this->stopTuning();
92 }
93 return;
94 }
95
96 _detector->getCurrentValue();
97
98 if(_detector->checkSafetyCondition(false) != CsaCoarseTuningDetector::LIMIT::IN_RANGE) {
99 _logger(Loger::ERROR) << __func__ << ": " << _logIdString << ": Safety limits crossed." << std::endl;
100 if(_tuningInProgress) {
101 _logger(Loger::ERROR) << __func__ << ": " << _logIdString
102 << ": Stopping current tuning process due to safety reasons." << std::endl;
103 this->stopTuning();
104 }
105 _safetyAutoresetCnt = 0;
106 _safetyLimitsCrossedFlag = true;
107 }
108 else if(_safetyLimitsCrossedFlag) {
109 _logger(Loger::INFO) << __func__ << ": " << _logIdString
110 << ": Safety flag auto-reset in: " << (10 - _safetyAutoresetCnt) << std::endl;
111 if(_safetyAutoresetCnt > 10) {
112 _safetyLimitsCrossedFlag = false;
113 }
114 _safetyAutoresetCnt++;
115 }
116
117 if(_safetyLimitsCrossedFlag) {
118 return;
119 }
120
121 if(_tuningInProgress) {
122 if(!_multithreadEnabled) tuneInternal();
123 return;
124 }
125
126 CsaCoarseTuningDetector::LIMIT limitCrossed = _detector->checkStartCondition();
127 if(limitCrossed != CsaCoarseTuningDetector::LIMIT::IN_RANGE && !_pauseFlag) {
128 _logger(Loger::INFO) << __func__ << ": " << _logIdString
129 << ": Start condition for coarse tuning start fulfilled. Start tuning." << std::endl;
130 if(_multithreadEnabled) {
131 _logger(Loger::DETAIL) << __func__ << ": " << _logIdString << ": Tuning is done in separate thread." << std::endl;
132 std::thread t1(&CoarseTuningAlgorithm::tuningThread, this);
133 t1.detach();
134 }
135 else {
136 // prepareActuatorToMove(false);
137 _actHelper.enableActuatorWithNoAutostart();
138 _actHelper.storeAndchangeActuatorSpeedSettings(ACT_SPEED_SETTING::SLOW); /*0-user,1-fast,2-slow*/
139 tuneInternal();
140 }
141 }

144void CoarseTuningAlgorithm::tuneInternal() {
145 _tuningInProgress = true;
146
147 if(_controller->isSafetyHold()) {
148 _logger(Loger::INFO)
149 << __func__ << ": " << _logIdString
150 << ": Safety hold enabled. Need to ramp controller and reset safety flag before tuning. Start ramping."
151 << std::endl;
152 _controller->rampAndResetSafetyHold();
153 while(_controller->isRampingInProgress() == false) {
154 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Wait for ramping thread to start." << std::endl;
155 }
156 return;
157 }
158
159 if(_controller->isRampingInProgress()) {
160 _logger(Loger::INFO) << __func__ << ": " << _logIdString
161 << ": Controller ramping in progress. Tune algortihm paused." << std::endl;
162 return;
163 }
164
165 auto currentStopCondition = _detector->checkStopCondition();
166 if(currentStopCondition == CsaCoarseTuningDetector::LIMIT::IN_RANGE) {
167 this->stopTuning();
168 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Tune done." << std::endl;
169 return;
170 }
171
172 auto currentPos = _actuator->getCurrentPosition();
173 auto currentDetVal = _detector->getCurrentValue();
174 auto newPosition = 0;
175 if(currentStopCondition == CsaCoarseTuningDetector::LIMIT::MAX_CROSSED) {
176 newPosition = currentPos - _stepScaling * _stepSize;
177 _logger(Loger::DETAIL) << __func__ << ": " << _logIdString
178 << ": Detector stop max limit crossed. Doing step down. Current detector value: "
179 << currentDetVal << std::endl;
180 }
181 else if(currentStopCondition == CsaCoarseTuningDetector::LIMIT::MIN_CROSSED) {
182 newPosition = currentPos + _stepScaling * _stepSize;
183 _logger(Loger::DETAIL) << __func__ << ": " << _logIdString
184 << ": Detector stop min limit crossed. Doing step up. Current detector value: "
185 << currentDetVal << std::endl;
186 }
187
188 auto status = _actuator->getStatus();
189 if(status == SMAStatusTypes::M_ERROR) {
190 if(_actuatorErrorCouter > 10) {
191 _logger(Loger::ERROR) << __func__ << ": " << _logIdString
192 << ": Stepper Actuator status is error. Error counter: " << _actuatorErrorCouter
193 << ".Stop tuning. Error is: " << _actuator->getError() << std::endl;
194 this->stopTuning();
195 }
196 _actuatorErrorCouter++;
197 _logger(Loger::ERROR) << __func__ << ": " << _logIdString
198 << ": Stepper Actuator status is error. Error counter: " << _actuatorErrorCouter
199 << ".Try to reset actuator. Error is: " << _actuator->getError() << std::endl;
200 _actuator->resetMotorFlags();
201 return;
202 }
203 else if(status == SMAStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON ||
204 status == SMAStatusTypes::M_POSITIVE_END_SWITCHED_ON) {
205 if(newPosition > currentPos) {
206 _logger(Loger::WARNING) << __func__ << ": " << _logIdString
207 << ": Positive end-switch active. Cannot continue tuning in this direction." << std::endl;
208 this->stopTuning();
209 return;
210 }
211 trigerActMoveToPosition(newPosition);
212 }
213 else if(status == SMAStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON ||
214 status == SMAStatusTypes::M_NEGATIVE_END_SWITCHED_ON) {
215 if(newPosition < currentPos) {
216 _logger(Loger::WARNING) << __func__ << ": " << _logIdString
217 << ": Negative end-switch active. Cannot continue tuning in this direction." << std::endl;
218 this->stopTuning();
219 return;
220 }
221 trigerActMoveToPosition(newPosition);
222 }
223 else if(status == SMAStatusTypes::M_OK) {
224 trigerActMoveToPosition(newPosition);
225 }
226 else if(status == SMAStatusTypes::M_NOT_IN_POSITION) {
227 if(_lastNewPositionSet)
228 trigerActMoveToPosition(_lastNewPosition);
229 else
230 trigerActMoveToPosition(newPosition);
231 _logger(Loger::WARNING) << __func__ << ": " << _logIdString
232 << ": Motor not in position. Repear command to sent to: " << _lastNewPosition << std::endl;
233 }
234 else if(status != SMAStatusTypes::M_IN_MOVE) {
235 _logger(Loger::ERROR) << __func__ << ": " << _logIdString
236 << ": Cannot tune. Stepper Actuator status is different than OK. Status ID: "
237 << _actuator->getStatus() << std::endl;
238 }
239 _actuatorErrorCouter = 0;
240}

274void CoarseTuningAlgorithm::tuningThread() {
275 if(_tuningInProgress) {
276 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Tuning thread already active. Ignore command."
277 << std::endl;
278 return;
279 }
280 _tuningInProgress = true;
281
282 // check whether controller need ramping and ramp it if needed
283 if(_controller->isSafetyHold()) {
284 _logger(Loger::INFO)
285 << __func__ << ": " << _logIdString
286 << ": Safety hold enabled. Need to ramp controller and reset safety flag before OXC search. Start ramping."
287 << std::endl;
288 _controller->rampAndResetSafetyHold();
289 while(_controller->isRampingInProgress() == false) {
290 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Wait for ramping thread to start." << std::endl;
291 if(!_tuningInProgress) return;
292 }
293 while(_controller->isRampingInProgress()) {
294 _logger(Loger::INFO) << __func__ << ": " << _logIdString
295 << ": Controller ramping in progress. Tune algortihm paused." << std::endl;
296 std::this_thread::sleep_for(std::chrono::milliseconds(15));
297 if(!_tuningInProgress) return;
298 }
299 }
300 // prepare actuator to movement with slow motion
301 _actHelper.enableActuatorWithNoAutostart();
302 _actHelper.storeAndchangeActuatorSpeedSettings(ACT_SPEED_SETTING::SLOW); /*0-user,1-fast,2-slow*/
303
304 bool tuningNotDone = true;
305 do {
306 if(!_tuningInProgress) {
307 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Tuning stopped by user or due to safety reasons."
308 << std::endl;
309 tuningNotDone = false;
310 }
311 else {
312 // check stop condition
313 auto currentStopCondition = _detector->checkStopCondition();
314 if(currentStopCondition == CsaCoarseTuningDetector::LIMIT::IN_RANGE) {
315 tuningNotDone = false;
316 _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Tune done. Signal in range" << std::endl;
317 }
318 else if(_detector->checkSafetyCondition() != CsaCoarseTuningDetector::LIMIT::IN_RANGE) {
319 _logger(Loger::ERROR) << __func__ << ": " << _logIdString << ": Safety limits crossed." << std::endl;
320 _logger(Loger::ERROR) << __func__ << ": " << _logIdString
321 << ": Stopping current tuning process due to safety reasons." << std::endl;
322 tuningNotDone = false;
323 _safetyAutoresetCnt = 0;
324 _safetyLimitsCrossedFlag = true;
325 }
326 else {
327 auto currentDetVal = _detector->getCurrentValue();
328 auto currentPos = _actuator->getCurrentPosition();
329 auto newPosition = currentPos;
330 if(currentStopCondition == CsaCoarseTuningDetector::LIMIT::MAX_CROSSED) {
331 newPosition = currentPos - _stepScaling * _stepSize;
332 _logger(Loger::DETAIL) << __func__ << ": " << _logIdString
333 << ": Detector stop max limit crossed. Doing step down. Current detector value: "
334 << currentDetVal << std::endl;
335 }
336 else if(currentStopCondition == CsaCoarseTuningDetector::LIMIT::MIN_CROSSED) {
337 newPosition = currentPos + _stepScaling * _stepSize;
338 _logger(Loger::DETAIL) << __func__ << ": " << _logIdString
339 << ": Detector stop min limit crossed. Doing step up. Current detector value: "
340 << currentDetVal << std::endl;
341 }
342 _logger(Loger::DETAIL) << __func__ << ": " << _logIdString << ": Send motor from : " << currentPos
343 << "(hwd read: " << _actuator->getCurrentPositionInSteps() << ") to: " << newPosition
344 << std::endl;
345 auto status = _actuator->moveToPosition(newPosition);
346 if(status != SMAStatusTypes::M_OK) {
347 _logger(Loger::ERROR) << __func__ << ": " << _logIdString << ": Send to position: " << newPosition
348 << " failed with status: " << status << ". Stop tuning. " << std::endl;
349 tuningNotDone = false;
350 }
351
352 if(_controller->isSafetyHold()) {
353 _logger(Loger::ERROR) << __func__ << ": " << _logIdString
354 << ": Controller safety triggered. Tuning failed. Finish thread." << std::endl;
355 tuningNotDone = false;
356 }
357 }
358 std::this_thread::sleep_for(std::chrono::milliseconds(50));
359 }
360
361 } while(tuningNotDone);
362 _actuator->stop();
363 // restore speed settings
364 _actHelper.restoreSpeedSettings();
365 _tuningInProgress = false;
366}

368CoarseTuningAlgorithmBase::STATUS CoarseTuningAlgorithm::evaluateStatus() {
369 // if disabled then status is just Disabled
370 if(!_enabled) {
371 _status = CoarseTuningAlgorithmBase::STATUS::DISABLED;
372 _statusInfo = "Coarse tuning disabled.";
373 return _status;
374 }
375
376 // assume that status will be OK, store value in local variable
377 CoarseTuningAlgorithmBase::STATUS status = CoarseTuningAlgorithmBase::STATUS::OK;
378 std::string statusInfo("");
379
380 if(_actuator->getStatus() == SMAStatusTypes::M_DISABLED) {
381 status = CoarseTuningAlgorithmBase::STATUS::ERROR;
382 statusInfo.append("Actuator is DISABLED. ");
383 }
384
385 if(_actuator->getStatus() == SMAStatusTypes::M_ERROR) {
386 status = CoarseTuningAlgorithmBase::STATUS::ERROR;
387 statusInfo.append("Actuator in ERROR. ");
388 }
389
390 if(_detector->getError() == SMAStatusTypes::M_ERROR) {
391 status = CoarseTuningAlgorithmBase::STATUS::ERROR;
392 statusInfo.append("Detector in ERROR. ");
393 }
394
395 if(!_detector->checkSettingsValidity()) {
396 status = CoarseTuningAlgorithmBase::STATUS::ERROR;
397 statusInfo.append("Invalid thresholds. ");
398 }
399
400 if(_stepSize == 0) {
401 status = CoarseTuningAlgorithmBase::STATUS::ERROR;
402 statusInfo.append("Step size is 0. ");
403 }
404
405 if(_stepScaling == 0) {
406 status = CoarseTuningAlgorithmBase::STATUS::ERROR;
407 statusInfo.append("Step size scaling is 0. ");
408 }
409
410 if(status == CoarseTuningAlgorithmBase::STATUS::OK) {
411 _statusInfo = "Coarse tuning status is OK.";
412 }
413 else
414 _statusInfo = statusInfo;
415 _status = status;
416 return _status;
417}
