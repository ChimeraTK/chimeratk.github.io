226void M_APP_FMC20_LbSync::decodeFmcType() {
227 uint32_t regValue = _fmcsIds->getValue();
228
229 uint32_t fmc1Id = regValue & 0xFFFF;
230 uint32_t fmc2Id = (regValue >> 16) & 0xFFFF;
231
232 _fmc1LpcId->setValue(fmc1Id);
233 _fmc2HpcId->setValue(fmc2Id);
