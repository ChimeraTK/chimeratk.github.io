236int32_t StepperMotorUNIO::correctSoftTargetPosition(int32_t positionInSteps) {
237 int32_t newPositionInSteps = positionInSteps;
238 int32_t regulationError = _positionInSteps - _targetHardwarePositionInStepsRB;
239 int32_t dissCpToNp = std::abs(_positionInSteps - newPositionInSteps);
240 if(_targetSoftwarePositionInSteps == newPositionInSteps && regulationError &&
241 static_cast<uint32_t>(std::abs(regulationError)) < _minimumStepSize &&
242 dissCpToNp <= int32_t(_minimumStepSize + std::abs(regulationError))) {
243 newPositionInSteps -= regulationError;
244 _logger(Loger::INFO) << __func__ << ": "
245 << "New soft target pos. equal to current soft target pos. (both: "
246 << _targetSoftwarePositionInSteps << "), but regulation error found (" << regulationError
247 << "). New soft target pos. was corrected with regulation error and equals now: "
248 << newPositionInSteps << std::endl;
249 }
250
251 if(_targetSoftwarePositionInSteps != newPositionInSteps) {
252 uint32_t dissNpToTsp = std::abs(newPositionInSteps - _targetSoftwarePositionInSteps);
253 uint32_t dissNpToThp = std::abs(newPositionInSteps - _targetHardwarePositionInStepsRB);
254 uint32_t dissTspToThp = std::abs(_targetSoftwarePositionInSteps - _targetHardwarePositionInStepsRB);
255 if(dissNpToTsp < _minimumStepSize && dissNpToThp < _minimumStepSize && dissTspToThp <= _minimumStepSize) {
256 if(newPositionInSteps > _targetSoftwarePositionInSteps) {
257 newPositionInSteps = _targetSoftwarePositionInSteps + _minimumStepSize;
258 }
259 else if(newPositionInSteps < _targetSoftwarePositionInSteps) {
260 newPositionInSteps = _targetSoftwarePositionInSteps - _minimumStepSize;
261 }
262
263 _logger(Loger::INFO) << __func__ << ": "
264 << "New target pos. (" << positionInSteps << ") and current target pos.("
265 << _targetSoftwarePositionInSteps << ") distance smaller than min. step size("
266 << _minimumStepSize << "). Modify new target pos. to: " << newPositionInSteps << std::endl;
267 }
268 }
269 else
270 _logger(Loger::INFO) << __func__ << ": "
271 << "New soft target pos. equal to current soft target pos.: " << _targetSoftwarePositionInSteps
272 << std::endl;
273
274 return newPositionInSteps;
275}

277int32_t StepperMotorUNIO::truncateMotorPositionInSteps(int32_t positionInSteps) {
278 int32_t newPositionInSteps = correctSoftTargetPosition(positionInSteps);
279
280 if(!_enableSoftwareLimits) return newPositionInSteps;
281
282 // truncate new position according to the internal limits.
283 if(newPositionInSteps > _softLimitMaxInSteps) {
284 newPositionInSteps = _softLimitMaxInSteps;
285 }
286 else if(newPositionInSteps < _softLimitMinInSteps) {
287 newPositionInSteps = _softLimitMinInSteps;
288 }
289
290 return newPositionInSteps;
291}

344bool StepperMotorUNIO::prepareForStepping() {
345 if(!_initialized) {
346 _logger(Loger::ERROR) << __func__ << ": "
347 << "UNIO motor is not initialized. Return." << std::endl;
348 return false;
349 }
350
351 if(!_enable) {
352 _logger(Loger::WARNING) << __func__ << ": "
353 << "UNIO motor is disabled. Return." << std::endl;
354 return false;
355 }
356
357 if(!_readbackValid) {
358 _logger(Loger::WARNING) << __func__ << ": "
359 << "UNIO position and target data readback not available. Return." << std::endl;
360 return false;
361 }
362
363 int32_t distance = _targetSoftwarePositionInSteps - _positionInSteps;
364 if(static_cast<uint32_t>(std::abs(distance)) < _minimumStepSize) {
365 int32_t regulationDifferece = _positionInSteps - _targetHardwarePositionInStepsRB;
366 if(regulationDifferece && (uint32_t(std::abs(regulationDifferece) + std::abs(distance)) >= _minimumStepSize)) {
367 _logger(Loger::WARNING) << __func__ << ": "
368 << "Target soft position (" << _targetSoftwarePositionInSteps
369 << ") too close to current position (" << _positionInSteps
370 << "). Distance: " << abs(distance) << ", but UNIO regulation error found ("
371 << regulationDifferece << "). We can continue." << std::endl;
372 }
373 else {
374 _logger(Loger::WARNING) << __func__ << ": "
375 << "Target soft position (" << _targetSoftwarePositionInSteps
376 << ") is too close too current position (" << _positionInSteps
377 << "). Distance: " << abs(distance) << ". Minimal distance: " << _minimumStepSize
378 << " Regulation error is: " << abs(regulationDifferece)
379 << ". Cannot perform now step. Return." << std::endl;
380 return false;
381 }
382 }
383
384 if(_stopMovement) {
385 _logger(Loger::INFO) << __func__ << ": "
386 << "Stop motor detected." << std::endl;
387 return false;
388 }
389
390 _logger(Loger::INFO) << __func__ << ": "
391 << "Distance between current pos. (" << _positionInSteps << ") and target soft pos.("
392 << _targetSoftwarePositionInSteps << ") is: " << distance << ". Start stepping." << std::endl;
393 return true;
394}

396void StepperMotorUNIO::moveToPositionInternal() {
397 _stopMovement = false;
398 if(!prepareForStepping()) return;
399
400 bool doMoving = true;
401 int waitForOkTimeout = 0;
402 int targetPositionIsNotChangingErrorCnt = 0;
403 unsigned int zeroDistanceCnt = 0;
404 int32_t dir = 0;
405 int32_t distance = 0;
406
407 do {
408 dir = 0;
409 if(_readbackValid) {
410 distance = _targetSoftwarePositionInSteps - _positionInSteps;
411 int32_t dissStpToHsp = _targetSoftwarePositionInSteps - _targetHardwarePositionInStepsRB;
412 /*
413 if (std::abs(dissStpToHsp) < _minimumStepSize) {
414 _logger(Loger::INFO) << __func__ << ": " << "Small soft ("<< _targetSoftwarePositionInSteps <<") and hard ("<<
415 _targetHardwarePositionInStepsRB <<") target pos. difference (" << dissStpToHsp << "). Movement not needed. Wait
416 for position stabilization." << std::endl; } else if (distance > 0) dir = 1; else if (distance < 0) dir = -1;
417 */
418 if(distance > 0)
419 dir = 1;
420 else if(distance < 0)
421 dir = -1;
422
423 if(dir &&
424 ((static_cast<uint32_t>(std::abs(distance)) < _minimumStepSize) ||
425 (static_cast<uint32_t>(std::abs(dissStpToHsp)) < _minimumStepSize))) {
426 int32_t currHSp = _targetHardwarePositionInStepsRB;
427 int32_t nextHSp = _targetHardwarePositionInStepsRB + dir * _minimumStepSize;
428 _logger(Loger::INFO) << __func__ << ": "
429 << "Distance between soft target pos. (" << _targetSoftwarePositionInSteps
430 << ") and current pos. (" << _positionInSteps << ") is " << distance
431 << " and it is smaller than the step size (" << _minimumStepSize
432 << "). Check if step is needed." << std::endl;
433 if(std::abs(_targetSoftwarePositionInSteps - nextHSp) >= std::abs(_targetSoftwarePositionInSteps - currHSp)) {
434 _logger(Loger::INFO)
435 << __func__ << ": "
436 << "next hard target pos: (" << nextHSp << "), current hard target pos. (" << currHSp
437 << "), soft target pos.(" << _targetSoftwarePositionInSteps
438 << "). Current hard target pos. is closer to software target than next hard target pos. Step not needed."
439 << std::endl;
440 dir = 0;
441 }
442 else {
443 _logger(Loger::INFO)
444 << __func__ << ": "
445 << "next hard target pos: (" << nextHSp << "), current hard target pos. (" << currHSp
446 << "), soft target pos.(" << _targetSoftwarePositionInSteps
447 << "). Next hard target pos. is closer to soft target pos. than current hard target pos. Step needed."
448 << std::endl;
449 }
450 }
451 }
452
453 if(_stopMovement) {
454 _logger(Loger::INFO) << __func__ << ": "
455 << "Stop motor detected. Set 'dir' to 0." << std::endl;
456 dir = 0;
457 }
458
459 if(dir) {
460 auto status = determineMotorStatusInternal();
461 if(_firstHardwareTargetPositionAvailable ==
462 false) { // if this is a first step we need to do it in order to trigger setpoint transmission from hardware
463 // because status is not OK
464 step(dir);
465 } // all is ok, we can step
466 else if(status == SMAStatusTypes::M_OK || status == SMAStatusTypes::M_NOT_IN_POSITION) {
467 int32_t targetPositionBeforeStep = _targetHardwarePositionInStepsRB;
468 step(dir);
469 _logger(Loger::INFO) << __func__ << ": "
470 << "Hardware target position before: " << targetPositionBeforeStep << " and after step ("
471 << dir << ") : " << _targetHardwarePositionInStepsRB << std::endl;
472 if(_stopMovement == false && targetPositionBeforeStep == _targetHardwarePositionInStepsRB) {
473 targetPositionIsNotChangingErrorCnt++;
474 _logger(Loger::ERROR) << __func__ << ": "
475 << "No reaction on step signal. Hardware target position didn't change. Possible "
476 "hardware problem. Error counter: "
477 << targetPositionIsNotChangingErrorCnt << std::endl;
478 if(targetPositionIsNotChangingErrorCnt > 3) {
479 _logger(Loger::ERROR) << __func__ << ": "
480 << "No reaction on step signal after 4 tries. ERROR. Finish movement. " << std::endl;
481 _noReactionOnTuningSignalErrorFlag = true;
482 this->stop();
483 }
484 }
485 else
486 targetPositionIsNotChangingErrorCnt = 0;
487
488 waitForOkTimeout = 0;
489 }
490 else if(status == SMAStatusTypes::M_IN_MOVE) { // we need to wait one more waiting loop
491 _logger(Loger::INFO) << __func__ << ": "
492 << "Motor still in move. Waiting for MOTOR_OK. Timeout counter: " << waitForOkTimeout
493 << std::endl;
494 this->stepInProgressWaitingLoop();
495 if(waitForOkTimeout > 10) {
496 _logger(Loger::ERROR) << __func__ << ": "
497 << "Movement cannot be done. Timeout when waiting for MOTOR_OK status." << std::endl;
498 return;
499 }
500 waitForOkTimeout++;
501 }
502 else if(status == SMAStatusTypes::M_ERROR) {
503 _logger(Loger::ERROR) << __func__ << ": "
504 << "Movement cannot be done. Motor status is ERROR: " << determineMotorErrorInternal()
505 << std::endl;
506 return;
507 }
508 else { // some different condition occured, we cannot continue movement /
509 _logger(Loger::ERROR) << __func__ << ": "
510 << "Movement cannot be done. Motor status is: " << status << std::endl;
511 return;
512 }
513 }
514 else {
515 uint32_t stablizeDistance = std::abs(_targetHardwarePositionInStepsRB - _positionInSteps);
516 uint32_t maxStablizeDistance = _minimumStepSize / 2;
517 if(!_readbackValid) {
518 _logger(Loger::ERROR) << __func__ << ": "
519 << "Data from UNIO not available. Stopping move from safety reason." << std::endl;
520 doMoving = false;
521 }
522 else if(_stopMovement) { // stopped by user or algorithm (external source)
523 doMoving = false;
524 }
525 else {
526 // stabilizing waiting loop - ignored when stop command triggered
527 // averaged current_position needs to be in range at least for 5 seconds, if not it means that controller
528 // overshoot and we need to wait more time.
529 uint32_t waitCntMax = 50;
530 uint32_t loopDelayMs = 100;
531 if(stablizeDistance <= maxStablizeDistance) {
532 _logger(Loger::DETAIL) << __func__ << ": "
533 << "Finish move. Distance is in range ( <=" << maxStablizeDistance
534 << " ). Distance needs to be " << std::round(1.0 * waitCntMax * loopDelayMs / 1000)
535 << " seconds in range to finish movement." << std::endl;
536 do {
537 stablizeDistance = std::abs(_targetHardwarePositionInStepsRB - _positionInSteps);
538 if(stablizeDistance <= maxStablizeDistance) {
539 zeroDistanceCnt++;
540 std::this_thread::sleep_for(std::chrono::milliseconds(loopDelayMs));
541 }
542 else
543 zeroDistanceCnt = 0;
544 } while(zeroDistanceCnt < waitCntMax && !_stopMovement);
545 doMoving = false;
546 }
547 else {
548 zeroDistanceCnt = 0;
549 _logger(Loger::INFO) << __func__ << ": "
550 << "Wait to stabilize current position with hardware target. Distance: "
551 << stablizeDistance << ", allowed: " << maxStablizeDistance << std::endl;
552 this->stepInProgressWaitingLoop();
553 }
554 }
555 }
556
557 if(_stopMovement) {
558 _logger(Loger::INFO) << __func__ << ": "
559 << "Stop step detected. Set 'doMoving' flag to false." << std::endl;
560 doMoving = false;
561 }
562
563 } while(doMoving);
564 _logger(Loger::INFO) << __func__ << ": "
565 << "Finish movement." << std::endl;
566}

570void StepperMotorUNIO::step(int32_t direction) {
571 _stepInProgress = true;
572 _tuningPort->setCsControlAllowed(false);
573 _tuningReadbackPort->setCsControlAllowed(false);
574
575 _logger(Loger::MAX_DETAIL) << __func__ << ": "
576 << "Activate step signal." << std::endl;
577 if(direction == 1) {
578 _logger(Loger::INFO) << __func__ << ": "
579 << "Step in positive direction. Send signal to 'step_up' pin." << std::endl;
580 _tuningPort->setPinState(_tuneUpPinNo, true);
581 }
582 else if(direction == -1) {
583 _logger(Loger::INFO) << __func__ << ": "
584 << "Step in negative direction. Send signal to 'step_down' pin." << std::endl;
585 _tuningPort->setPinState(_tuneDownPinNo, true);
586 }
587 else {
588 _stepInProgress = false;
589 return;
590 }
591
592 std::this_thread::sleep_for(std::chrono::milliseconds(_highTimeMs));
593 _logger(Loger::MAX_DETAIL) << __func__ << ": "
594 << "Deactivate step signal." << std::endl;
595 _tuningPort->setPinState(_tuneUpPinNo, false);
596 _tuningPort->setPinState(_tuneDownPinNo, false);
597
598 this->stepInProgressWaitingLoop();
599
600 _tuningPort->setPinState(_tuneUpPinNo, false);
601 _tuningPort->setPinState(_tuneDownPinNo, false);
602 _stepInProgress = false;
603}

605void StepperMotorUNIO::stepInProgressWaitingLoop() {
606 _logger(Loger::INFO) << __func__ << ": "
607 << "Wait step period: " << _tuningPeriodSec << " seconds." << std::endl;
608
609 int cntDisplay = 0;
610 bool waitInProgress = true;
611 _tuningStartPoint = std::chrono::system_clock::now();
612
613 do {
614 auto nowTime = std::chrono::system_clock::now();
615 auto seconds = std::chrono::duration_cast<std::chrono::seconds>(nowTime - _tuningStartPoint).count();
616 _remainBusyTime = std::abs(_tuningPeriodSec - seconds);
617 if(_tuningPeriodSec < 5 || _tuningPeriodSec > 600) {
618 _tuningPeriodSec = 45;
619 }
620
621 if(seconds > _tuningPeriodSec) {
622 _logger(Loger::INFO) << __func__ << ": "
623 << "Step done. Step period: " << _tuningPeriodSec << " seconds." << std::endl;
624 waitInProgress = false;
625 }
626
627 if(_stopMovement) {
628 _logger(Loger::INFO) << __func__ << ": "
629 << "Stop step detected." << std::endl;
630 waitInProgress = false;
631 }
632
633 std::this_thread::sleep_for(std::chrono::milliseconds(50));
634
635 if(cntDisplay > 100) {
636 _logger(Loger::DETAIL) << __func__ << ": "

671SMAStatus StepperMotorUNIO::determineMotorStatusInternal() {
672 // assume that everything is ok
673 // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
674 // ERROR CONDITION CHECK - it should be done first.
675 // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
676
677 // first check if disabled
678 if(!_enable) {
679 return SMAStatusTypes::M_DISABLED;
680 }
681
682 if(!_initialized || !_readbackValid || _noReactionOnTuningSignalErrorFlag) {
683 return SMAStatusTypes::M_ERROR;
684 }
685
686 if(_enableSoftwareLimits && (_softLimitMaxInSteps <= _softLimitMinInSteps)) {
687 return SMAStatusTypes::M_ERROR;
688 }
689
690 // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
691 // STATUS CHECK
692 // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
693
694 if(_enableSoftwareLimits) {
695 if(_positionInSteps >= _softLimitMaxInSteps) {
696 return SMAStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON;
697 }
698
699 if(_positionInSteps <= _softLimitMinInSteps) {
700 return SMAStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON;
701 }
702 }
703
704 if(_stepInProgress) return SMAStatusTypes::M_IN_MOVE;
705
706 if(static_cast<uint32_t>(std::abs(_targetSoftwarePositionInSteps - _positionInSteps)) > _minimumStepSize)
707 return SMAStatusTypes::M_NOT_IN_POSITION;
708
709 if(static_cast<uint32_t>(std::abs(_targetHardwarePositionInStepsRB - _positionInSteps)) > _minimumStepSize / 2)
710 return SMAStatusTypes::M_NOT_IN_POSITION;
711
712 return SMAStatusTypes::M_OK;
713}

737void StepperMotorUNIO::readbackReceiverThread() {
738 _logger(Loger::INFO) << __func__ << ": "
739 << "Starting receiver thread." << std::endl;
740 uint32_t timeoutCnt = 0;
741
742 uint32_t loopSleepInMs = 10;
743
744 while(_stopReadbackReceiverThread == false) {
745 _tuningReadbackPort->updateFromDevice();
746 // std::cout << "StepperMotorUNIO::readbackReceiverThread: Execute receiver thread " << std::endl;
747 std::string readback;
748 if(_tuningReadbackPort->receiveStringNoBlocking(readback)) {
749 _tuningReadbackString += readback;
750
751 if((timeoutCnt == 0) &&
752 ((_tuningReadbackString.find("s") != std::string::npos) ||
753 (_tuningReadbackString.find("a") != std::string::npos))) {
754 if((_tuningReadbackString.find("s") == 0) || (_tuningReadbackString.find("a") == 0)) {
755 _logger(Loger::MAX_DETAIL) << __func__ << ": "
756 << "'s' or 'a' marker detected. Current string: " << _tuningReadbackString
757 << std::endl;
758 timeoutCnt = 1;
759 }
760 else {
761 _logger(Loger::MAX_DETAIL) << __func__ << ": "
762 << "'s' or 'a' marker detected on wrong position. Delete string: "
763 << _tuningReadbackString << std::endl;
764 _tuningReadbackString.clear();
765 }
766 }
767 else if(timeoutCnt > uint32_t(100 / loopSleepInMs)) {
768 _logger(Loger::MAX_DETAIL) << __func__ << ": "
769 << "Timeout for current string. Delete string: " << _tuningReadbackString
770 << std::endl;
771 _tuningReadbackString.clear();
772 timeoutCnt = 0;
773 }
774
775 if(_tuningReadbackString.length() >= 3) {
776 _logger(Loger::MAX_DETAIL) << __func__ << ": "
777 << "String length: " << _tuningReadbackString.length()
778 << ". Timeout cnt: " << timeoutCnt
779 << ". Start parse. String: " << _tuningReadbackString << std::endl;
780 parseTuningReadbackString(_tuningReadbackString);
781 _tuningReadbackString.clear();
782 timeoutCnt = 0;
783 }
784 }
785
786 if(timeoutCnt) timeoutCnt++;
787
788 auto nowTime = std::chrono::system_clock::now();
789 auto secondFromLastData =
790 std::chrono::duration_cast<std::chrono::seconds>(nowTime - _lastReadBackUpdateTime).count();
791 if(secondFromLastData > 10) {
792 // std::cout << "StepperMotorUNIO::parseTuningReadbackString: No new data available since 10 seconds. Readback
793 // data expired and will be zero." << std::endl;
794 _positionInSteps = _targetHardwarePositionInStepsRB = 0;
795 _readbackValid = false;
796 }
797
798 std::this_thread::sleep_for(std::chrono::milliseconds(loopSleepInMs));
799 }
800 _positionInSteps = _targetHardwarePositionInStepsRB = 0;
801 _readbackValid = false;
802 _logger(Loger::INFO) << __func__ << ": "
803 << "Stopping receiver thread closing." << std::endl;
804}

806void StepperMotorUNIO::parseTuningReadbackString(std::string& tuningReadbackString) {
807 bool aFound = false;
808 uint32_t setpoint = 0;
809 uint32_t readback = 0;
810 auto sPos = tuningReadbackString.find("s");
811 auto aPos = tuningReadbackString.find("a");
812 if(sPos != std::string::npos) {
813 _logger(Loger::MAX_DETAIL) << __func__ << ": "
814 << "'s' marker found on position: " << sPos << std::endl;
815 if(tuningReadbackString.length() > sPos + 2) {
816 setpoint =
817 (uint32_t(tuningReadbackString.at(sPos + 1)) << 8 | (uint32_t(tuningReadbackString.at(sPos + 2)) & 0xFF));
818 _logger(Loger::MAX_DETAIL) << __func__ << ": "
819 << "Current setpoint: " << setpoint
820 << " Data 0: " << (uint32_t(tuningReadbackString.at(sPos + 1)) << 8)
821 << " Data 1: " << (uint32_t(tuningReadbackString.at(sPos + 2)) & 0xFF) << std::endl;
822 _targetHardwarePositionInStepsRB = setpoint;
823
824 if(_firstHardwareTargetPositionAvailable == false) {
825 _firstHardwareTargetPositionAvailable = true;
826 _targetSoftwarePositionInSteps = _targetHardwarePositionInStepsRB;
827 }
828 }
829 else
830 _logger(Loger::MAX_DETAIL) << __func__ << ": "
831 << "'s' marker found on position: " << aPos << ", but string is too short."
832 << std::endl;
833 }
834
835 if(aPos != std::string::npos) {
836 if(tuningReadbackString.length() > aPos + 2) {
837 _logger(Loger::MAX_DETAIL) << __func__ << ": "
838 << "'a' marker found on position: " << aPos << std::endl;
839 readback =
840 (uint32_t(tuningReadbackString.at(aPos + 1)) << 8 | (uint32_t(tuningReadbackString.at(aPos + 2)) & 0xFF));
841 aFound = true;
842 }
843 else
844 _logger(Loger::MAX_DETAIL) << __func__ << ": "
845 << "'a' marker found on position: " << aPos << ", but string is too short."
846 << std::endl;
847 }
848
849 if(aFound) {
850 _logger(Loger::MAX_DETAIL) << __func__ << ": "
851 << "Current readback: " << readback
852 << " Data 0: " << (uint32_t(tuningReadbackString.at(aPos + 1)) << 8)
853 << " Data 1: " << (uint32_t(tuningReadbackString.at(aPos + 2)) & 0xFF) << std::endl;
854
855 if(_positionInStepsAverageLen <= 1) {
856 _positionInSteps = readback;
857 }
858 else {
859 if(_positionInStepsAvgQueue.size() >= _positionInStepsAverageLen) {
860 while(_positionInStepsAvgQueue.size() >= _positionInStepsAverageLen) _positionInStepsAvgQueue.pop_front();
861 }
862
863 _positionInStepsAvgQueue.push_back(readback);
864 float avg = 0;
865 for(auto& element : _positionInStepsAvgQueue) {
866 avg += element;
867 }
868 avg /= _positionInStepsAvgQueue.size();
869 _positionInSteps = (int32_t)(avg);
870 _logger(Loger::MAX_DETAIL) << __func__ << ": "
871 << "Current readback: " << readback << " avg. len: " << _positionInStepsAverageLen
872 << ", queue size: " << _positionInStepsAvgQueue.size() << ", avg value: " << avg
873 << ", position in steps: " << _positionInSteps << std::endl;
874 }
875
876 _readbackValid = true;
877 _lastReadBackUpdateTime = std::chrono::system_clock::now();
878 }
