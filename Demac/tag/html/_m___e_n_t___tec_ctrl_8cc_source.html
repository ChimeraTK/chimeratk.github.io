186 void M_ENT_TecCtrl::commandCb() {
187  auto cmd = _command->getValue();
188 
189  if(cmd == CMD_RESET_TEC_DEVICE) {
190  _logger(Loger::WARNING) << __func__ << ": "
191  << "Reset command will be sent to the TEC device. " << std::endl;
192  _writeVector.clear();
193  TecFrame resetFrame(this->_tecAddressLocal, TecFrame::TecFrameType::TecReset);
194  _writeVector.push_back(resetFrame);
195  TecFrame statusReqFrame(
196  0xFF, TecFrame::TecParameter::MonDriverStatus, TecFrame::TecFrameType::TecParameterValueRead, 1);
197  _writeVector.push_back(statusReqFrame);
198  _writeVectorHasChanged = true;
199  _autoTuningInProgress->setValue(0);
200  _tecResetSend = true;
201  return;
202  }
203 
204  if(cmd == CMD_RESET_TEC_AUTO_TUNE_START) {
205  _logger(Loger::WARNING) << __func__ << ": "
206  << "Start AUTOTUNE will be sent to the TEC device. " << std::endl;
207  _writeVector.clear();
208  this->createWriteVector(2, false);
209  TecFrame startTuneFrame(this->_tecAddressLocal, TecFrame::TecParameter::OthAtmAutoTuningStart,
210  TecFrame::TecFrameType::TecParameterValueSet, 1, 1);
211  _writeVector.push_back(startTuneFrame);
212  _writeVectorHasChanged = true;
213  return;
214  }
215 
216  if(cmd == CMD_RESET_TEC_AUTO_TUNE_CANCEL) {
217  _logger(Loger::WARNING) << __func__ << ": "
218  << "Cancel AUTOTUNE will be sent to the TEC device. " << std::endl;
219  _writeVector.clear();
220  TecFrame startTuneFrame(this->_tecAddressLocal, TecFrame::TecParameter::OthAtmAutoTuningCancel,
221  TecFrame::TecFrameType::TecParameterValueSet, 1, 1);
222  _writeVector.push_back(startTuneFrame);
223  _writeVectorHasChanged = true;
224  return;
225  }
226 
227  if(cmd >= CMD_CREATE_READ_VECTOR_FOR_GROUP_1 && cmd <= CMD_CREATE_READ_VECTOR_FOR_GROUP_15) {
228  int groupId = cmd - CMD_CREATE_READ_VECTOR_FOR_GROUP_1 + 1;
229  this->createReadVector(groupId);
230  return;
231  }
232 
233  if(cmd >= CMD_CREATE_WRITE_VECTOR_FOR_GROUP_1 && cmd <= CMD_CREATE_WRITE_VECTOR_FOR_GROUP_15) {
234  int groupId = cmd - CMD_CREATE_WRITE_VECTOR_FOR_GROUP_1 + 1;
235  this->createWriteVector(groupId);
236  return;
237  }
238 
239  if(cmd >= CMD_CREATE_WRITE_VECTOR_FOR_GROUP_1_TO_FLASH && cmd <= CMD_CREATE_WRITE_VECTOR_FOR_GROUP_15_TO_FLASH) {
240  int groupId = cmd - CMD_CREATE_WRITE_VECTOR_FOR_GROUP_1_TO_FLASH + 1;
241  this->createWriteVector(groupId, true);
242  return;
243  }
244 
245  if(cmd >= CMD_SYNCH_WRITE_TO_READ_VALUES_FOR_GROUP_1 && cmd <= CMD_SYNCH_WRITE_TO_READ_VALUES_FOR_GROUP_15) {
246  int groupId = cmd - CMD_SYNCH_WRITE_TO_READ_VALUES_FOR_GROUP_1 + 1;
247  _logger(Loger::INFO) << __func__ << ": "
248  << "Command SYNCH_IN_WITH_OUT. Command ID: " << cmd << " Group ID: " << groupId << std::endl;
249  this->synchronizeInWithOut(groupId);
250  return;
251  }
252 
253  /*
254  if (cmd == CMD_APPLY_CHANGES_OF_PERIODIC_VECTOR) {
255  int32_t creationMask = _enablePeriodicReadreadback->getValue();
256  if (creationMask != _periodicReadreadbackIsEnabled->getValue()) {
257  createPeriodicVector(creationMask);
258  _periodicReadreadbackIsEnabled->setValue(creationMask);
259  }
260  return;
261  }
262  */
263  _logger(Loger::ERROR) << __func__ << ": "
264  << "Command unknown. Command ID: " << cmd << std::endl;
265  return;

268 void M_ENT_TecCtrl::analizeAnswerFrame(TecFrame frame) {
269  if(frame.getState() == TecFrame::TecFrameParsed) {
270  if(frame.getFrameType() == TecFrame::TecFrameType::TecParameterValueSet) {
271  if(frame.getParameterOffset() == TecFrame::TecParameter::OthAtmAutoTuningStart) {
272  _logger(Loger::INFO) << __func__ << ": "
273  << "AUTOTUNE start command accepted by TEC. Tuning has started. " << std::endl;
274  _autoTuningInProgress->setValue(1);
275  _atTriggered = true;
276  _atCanceled = false;
277  _atTimeout = 0;
278  //_atTuningStatus->setValue(0); // just after start we need to clear status, not to get "false" status until
279  //next update is done
280  this->createReadVector(4);
281  return;
282  }
283  else if(frame.getParameterOffset() == TecFrame::TecParameter::OthAtmAutoTuningCancel) {
284  _logger(Loger::INFO) << __func__ << ": "
285  << "AUTOTUNE cancel accepted by TEC. Tuning has been stopped by user. " << std::endl;
286  _autoTuningInProgress->setValue(0);
287  _atCanceled = true;
288  _atTriggered = false;
289  _atTimeout = 0;
290  this->createReadVector(4);
291  return;
292  }
293  }
