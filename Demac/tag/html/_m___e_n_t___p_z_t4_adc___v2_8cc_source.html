327double M_ENT_PZT4Adc_V2::getConvertFactor(uint32_t channelId) {
328 double cal = 100.0 / 9.09; // 11.0001
329 double convFactor = 0.0;
330 if(channelId < 4) { // sensors channels
331 convFactor = 1.0 / 13107.2 / 9.99;
332 }
333 else if(channelId >= 4 && channelId < 8) { // current channels
334 convFactor = 1.0 / 13107.2 / 100.0;
335 }
336 else if(channelId >= 8 && channelId < 12) { // voltage channels
337 convFactor = 1.0 / 13107.2 * cal; // convFactor = 1.0 / 13107.2 * 11.0001
338 }
339 else if(channelId == 12) { // HV+ voltage channels
340 convFactor = 1.0 / 13107.2 * 100.0 / 8.47; // convFactor = 1.0 / 13107.2 * 11,80
341 }
342 else if(channelId == 14) { // HV- voltage channels
343 convFactor = 1.0 / 13107.2 * cal; // convFactor = 1.0 / 13107.2 * 11.0001;
344 }
345 else if(channelId == 13 || channelId == 15) {
346 convFactor = 1.0 / 13107.2;
347 }
348 return convFactor;
349}
