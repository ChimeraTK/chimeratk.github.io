471 void StepperMotorSawPiezoPZT4::moveToPositionInternal() {
472  if(!_initialized) {
473  _logger(Loger::ERROR) << __func__ << ": " << "Piezo motor is not initialized. Return." << std::endl;
474  return;
475  }
476 
477  if(!_enable) {
478  _logger(Loger::ERROR) << __func__ << ": " << "Piezo motor is disabled. Return." << std::endl;
479  return;
480  }
481  _stopMovement = false;
482 
483  int32_t distance = _targetPositionInStepsInternal - _positionInSteps;
484  int32_t dir = 1;
485 
486  if(distance == 0) {
487  _logger(Loger::INFO) << __func__ << ": " << "Target position equal current position. Return." << std::endl;
488  return;
489  }
490  _logger(Loger::INFO) << __func__ << ": " << "Current distance is: " << distance << ". Start stepping. " << std::endl;
491  do {
492  distance = _targetPositionInStepsInternal - _positionInSteps;
493  if(distance > 0) {
494  dir = 1;
495  }
496  else {
497  dir = -1;
498  }
499  if(distance == 0) break;
500 
501  step(dir);
502  _positionInSteps += dir;
503  if(_stopMovement) {
504  _targetPositionInStepsInternal = _positionInSteps;
505  _stopMovement = false;
506  distance = 0;
507  }
508  _logger(Loger::DETAIL) << __func__ << ": " << "Current distance is: " << distance
509  << ". Current position is steps: " << _positionInSteps << std::endl;
510  } while(distance != 0);
511  _logger(Loger::INFO) << __func__ << ": " << "Current distance is: " << distance
512  << ". Current position is steps: " << _positionInSteps << ". Movement is done." << std::endl;
513 }

644 void StepperMotorSawPiezoPZT4::determineMotorStatusAndError() {
645  // assume that everything is OK
646  _motorStatus = SMAStatusTypes::M_OK;
647  _motorError = SMAErrorTypes::M_NO_ERROR;
648 
649  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
650  // ERROR CONDITION CHECK - it should be done first.
651  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
652 
653  if(_pzt4Channel->isFunctional() == false) {
654  _motorStatus = SMAStatusTypes::M_ERROR;
655  _motorError = SMAErrorTypes::M_HARDWARE_ERROR;
656  if(_pzt4FailDetected == false)
657  _logger(Loger::ERROR) << __func__ << ": " << "Critical error on PZT4 channel. Cannot operate." << std::endl;
658  _pzt4FailDetected = true;
659  _initialized = false;
660  return;
661  }
662  else if(_pzt4FailDetected) {
663  _logger(Loger::ERROR) << __func__ << ": "
664  << "PZT4 channel is back for standard operation. Try to reinitialize motor." << std::endl;
665  _pzt4FailDetected = false;
666  initializeInternal();
667  }
668 
669  // first check if disabled
670  if(!_enable) {
671  _motorStatus = SMAStatusTypes::M_DISABLED;
672  return;
673  }
674 
675  if(!_initialized) {
676  _motorStatus = SMAStatusTypes::M_ERROR;
677  _motorError = SMAErrorTypes::M_HARDWARE_ERROR;
678  return;
679  }
680 
681  if(_enableSoftwareLimits && (_softLimitMaxInSteps <= _softLimitMinInSteps)) {
682  _motorStatus = SMAStatusTypes::M_ERROR;
683  _motorError = SMAErrorTypes::M_CONFIG_ERROR_MIN_POS_GRATER_EQUAL_TO_MAX;
684  return;
685  }
686 
687  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
688  // STATUS CHECK
689  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
690 
691  if(_enableSoftwareLimits) {
692  if(_positionInSteps >= _softLimitMaxInSteps) {
693  _motorStatus = SMAStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON;
694  return;
695  }
696 
697  if(_positionInSteps <= _softLimitMinInSteps) {
698  _motorStatus = SMAStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON;
699  return;
700  }
701  }
702 
703  if(_moveInProgress) {
704  _motorStatus = SMAStatusTypes::M_IN_MOVE;
705  return;
706  }
707 
708  if(_targetPositionInStepsInternal != _positionInSteps) {
709  // if (_targetPositionInSteps != _positionInSteps) {
710  _motorStatus = SMAStatusTypes::M_NOT_IN_POSITION;
711  return;
712  }
