124bool CsaCtrlOnStateSpaceManager::addController(boost::shared_ptr<CsaCtrlOnStateSpace> controller) {
125 std::uintptr_t ctrlAddress = reinterpret_cast<std::uintptr_t>(controller.get());
126 for(auto& item : _controllersUniqueIdVec) {
127 if(item == ctrlAddress) {
128 _logger(Loger::WARNING) << __func__ << ": "
129 << "CsaCtrlOnStateSpaceManager::addController: Controller with address: " << ctrlAddress
130 << " already added." << std::endl;
131 return false; // TO_DO - could be changed to exception
132 }
133 }
134
135 // all added controller to the manager must manage the same physical controller, then can be different types eg. PID
136 // or USER_FILE, but then need the same M_RapidX_ENT_StateSpaceCtrl object
137 if(_controllersVec.size() > 0 &&
138 _controllersVec[0]->getStateSpaceController() == controller->getStateSpaceController()) {
139 _logger(Loger::ERROR) << __func__ << ": "
140 << "CsaCtrlOnStateSpaceManager::addController: Added controller is based on different "
141 "physical controller. Add ignored. "
142 << std::endl;
143 return false; // TO_DO - could be changed to exception
144 }
145
146 _controllersUniqueIdVec.push_back(ctrlAddress);
147 _controllersVec.push_back(controller);
148 return true;

162void CsaCtrlOnStateSpaceManager::enableInternal(bool enable) {
163 if(_waitForRampingAndEnableActive.load()) {
164 _logger(Loger::WARNING) << __func__ << ": "
165 << "Previous call still in progress. Wait for ramping to finish" << std::endl;
166 return;
167 }
168
169 auto activeCtrlId = _activeControllerReadback->getValue();
170 if(activeCtrlId < _controllersVec.size()) {
171 if(enable && _controllersVec[activeCtrlId]->getStateSpaceController()->isSafetyHold()) {
172 _logger(Loger::INFO)
173 << __func__ << ": "
174 << "Safety hold enabled. Need to ramp controller and reset safety flag before enabling. Start ramping."
175 << std::endl;
176 _controllersVec[activeCtrlId]->getStateSpaceController()->rampAndResetSafetyHold();
177 _waitForRampingAndEnableActive = true;
178 while(_controllersVec[activeCtrlId]->getStateSpaceController()->isRampingInProgress() == false) {
179 _logger(Loger::DETAIL) << __func__ << ": "
180 << "Wait for ramping thread to start." << std::endl;
181 }
182 std::thread t(&CsaCtrlOnStateSpaceManager::waitForRampingAndEnable, this, activeCtrlId);
183 t.detach();
184 return;
185 }
186
187 if(_daq) {
188 if(_daq->getTriggerSource() == M_ENT_DAQCtrlBase::TRG_SRC::MANUAL) {
189 _daq->trigger();
190 }
191 }
192 _controllersVec[activeCtrlId]->setEnable(enable);
193 }

200void CsaCtrlOnStateSpaceManager::waitForRampingAndEnable(uint32_t activeCtrlId) {
201 uint32_t timeoutCnt = 0;
202 while(_controllersVec[activeCtrlId]->getStateSpaceController()->isRampingInProgress() == true) {
203 if(timeoutCnt > 2000) {
204 _logger(Loger::ERROR) << __func__ << ": "
205 << "CsaCtrlOnStateSpaceManager::waitForRampingAndEnable() : Wait for ramping thread to "
206 "finish timeout. Wait more than 2 seconds. Controller ID: "
207 << activeCtrlId << " not enabled." << std::endl;
208 _waitForRampingAndEnableActive = false;
209 return;
210 }
211 std::this_thread::sleep_for(std::chrono::milliseconds(1));
212 timeoutCnt++;
213 }
214
215 if(_daq) {
216 if(_daq->getTriggerSource() == M_ENT_DAQCtrlBase::TRG_SRC::MANUAL) {
217 _daq->trigger();
218 }
219 }
220 _controllersVec[activeCtrlId]->setEnable(true);
221 _waitForRampingAndEnableActive = false;
222}
