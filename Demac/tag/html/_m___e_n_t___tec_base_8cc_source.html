37void M_ENT_TecBase::addToReadMap(TecFrame::TecParameter parameterId, boost::shared_ptr<Csa_double> csObject,
38 uint32_t frameType, uint32_t valueType, uint32_t parameterGroupId) {
39 boost::shared_ptr<CsTecFrame> csTecFrame(new CsTecFrame());
40 csTecFrame->parameterId = parameterId;
41 csTecFrame->csObject = csObject;
42
43 if(frameType == IT_IS_READ)
44 csTecFrame->type = TecFrame::TecFrameType::TecParameterValueRead;
45 else if(frameType == IT_IS_LIMIT) {
46 return;
47 csTecFrame->type = TecFrame::TecFrameType::TecParameterLimitRead;
48 }
49 else {
50 _logger(Loger::WARNING) << __func__ << ": "
51 << "This is not read frame. Return." << std::endl;
52 return;
53 }
54 if(valueType == IT_IS_FLOAT)
55 csTecFrame->retValueType = TecFrame::TecParameterValueType::TecValueIsFloat;
56 else if(valueType == IT_IS_INT)
57 csTecFrame->retValueType = TecFrame::TecParameterValueType::TecValueIsInteger;
58 else
59 csTecFrame->retValueType = TecFrame::TecParameterValueType::TecValueIsRaw;
60
61 csTecFrame->maskId = this->_tecReadFramesMap.size();
62
63 this->_tecReadFramesMap.insert(
64 std::pair<TecFrame::TecParameter, boost::shared_ptr<CsTecFrame>>(csTecFrame->parameterId, csTecFrame));
65 this->_tecFramesVector.push_back(csTecFrame);
66 this->_tecFramesGroupsMap[parameterGroupId].push_back(csTecFrame);
67 // std::cout << "SIZE OF MAP READ IS: " << _tecReadFramesMap.size() << ", size of vector: " << _tecFramesVector.size()
68 // << ", size of group map is: " << _tecFramesGroupsMap.size() << std::endl;
69}

141void M_ENT_TecBase::createWriteVector(uint32_t groupId, bool writeToFlash) {
142 if(this->_tecFramesGroupsMap[groupId].size() != 0) {
143 _writeVector.clear();
144 if(writeToFlash) {
145 _logger(Loger::INFO) << __func__ << ": "
146 << "Data will be written to FLASH. Enable storage in FLASH memory." << std::endl;
147 TecFrame newFrame(
148 _tecAddressLocal, TecFrame::ParameterSystemFlashSaveOff, TecFrame::TecParameterValueSet, 1, (int)0);
149 _writeVector.push_back(newFrame);
150 }
151 else {
152 _logger(Loger::INFO) << __func__ << ": "
153 << "Data will NOT be written to FLASH. Disable storage in FLASH memory." << std::endl;
154 TecFrame newFrame(
155 _tecAddressLocal, TecFrame::ParameterSystemFlashSaveOff, TecFrame::TecParameterValueSet, 1, (int)1);
156 _writeVector.push_back(newFrame);
157 }
158
159 auto& vec = this->_tecFramesGroupsMap[groupId];
160 for(auto& item : vec) {
161 auto writeFrame = _tecWriteFramesMap[item->parameterId];
162 double newValue = writeFrame->csObject->getValue();
163
164 if(newValue > writeFrame->writeLimitMax) {
165 _logger(Loger::WARNING) << __func__ << ": "
166 << "Max limit crossed. Requested value: " << newValue
167 << ", limit value: " << writeFrame->writeLimitMax << std::endl;
168 newValue = writeFrame->writeLimitMax;
169 }
170
171 if(newValue < writeFrame->writeLimitMin) {
172 _logger(Loger::WARNING) << __func__ << ": "
173 << "Min limit crossed. Requested value: " << newValue
174 << ", limit value: " << writeFrame->writeLimitMin << std::endl;
175 newValue = writeFrame->writeLimitMin;
176 }
177
178 if(writeFrame->retValueType == TecFrame::TecParameterValueType::TecValueIsFloat) {
179 TecFrame newFrame(_tecAddressLocal, writeFrame->parameterId, writeFrame->type, _channelId, (float)newValue);
180 _writeVector.push_back(newFrame);
181 }
182 else if(writeFrame->retValueType == TecFrame::TecParameterValueType::TecValueIsInteger) {
183 TecFrame newFrame(_tecAddressLocal, writeFrame->parameterId, writeFrame->type, _channelId, (int)newValue);
184 _writeVector.push_back(newFrame);
185 }
186 }
187 if(writeToFlash) {
188 _logger(Loger::INFO) << __func__ << ": "
189 << "Disable writes to FLASH." << std::endl;
190 TecFrame newFrame(
191 _tecAddressLocal, TecFrame::ParameterSystemFlashSaveOff, TecFrame::TecParameterValueSet, 1, (int)1);
192 _writeVector.push_back(newFrame);
193 }
194
195 _writeVectorHasChanged = true;
196 // std::cout << "Write vector size is: " << _writeVector.size() << std::endl;
197 }

238void M_ENT_TecBase::analizeAnswerFrame(TecFrame frame) {
239 if(frame.getState() == TecFrame::TecFrameParsed) {
240 /*
241 if (_logger.getLogLevel() >= Loger::DETAIL) {
242 _logger(Loger::DETAIL) << __func__ << ": " << "TecFrameParsed state frame: " << std::endl;
243 frame.print();
244 _logger(Loger::DETAIL) << __func__ << ": " << "Frame printed. " << std::endl;
245 }
246 */
247
248 if(frame.getFrameType() == TecFrame::TecFrameType::TecParameterValueSet) {
249 if(this->_tecWriteFramesMap.find(frame.getParameterOffset()) != this->_tecWriteFramesMap.end()) {
250 // std::cout << "M_ENT_TecBase::analizeAnswerFrame: Frame with parameter offset: " << frame.getParameterOffset()
251 // << " found in WriteFrames map." << std::endl;
252 auto& csTecFramePtr = _tecWriteFramesMap[frame.getParameterOffset()];
253 // bool isInt = false;
254 double val = 0;
255 if(csTecFramePtr->retValueType == TecFrame::TecParameterValueType::TecValueIsFloat) {
256 _logger(Loger::INFO) << __func__ << ": "
257 << "Frame : " << frame.getParameterOffset()
258 << " -> Answer to write frame (float): " << frame.getFloatParameterValue() << std::endl;
259 val = (double)frame.getFloatParameterValue();
260 }
261 else if(csTecFramePtr->retValueType == TecFrame::TecParameterValueType::TecValueIsInteger) {
262 _logger(Loger::INFO) << __func__ << ": "
263 << "Frame : " << frame.getParameterOffset()
264 << " -> Answer to write frame (int): " << frame.getIntParameterValue() << std::endl;
265 val = (double)frame.getIntParameterValue();
266 }
267 if(csTecFramePtr->csObject) {
268 bool loadToProperty = false;
269 if(fabs(csTecFramePtr->csObject->getValue() - val) < 0.0001) {
270 loadToProperty = true;
271 }
272 else if((fabs(csTecFramePtr->writeLimitMax) - val) < 0.0001) {
273 _logger(Loger::INFO) << __func__ << ": "
274 << "Answer value mismatch with set value. Answer: " << val
275 << " set value: " << csTecFramePtr->csObject->getValue()
276 << ", but max limit has been reached." << std::endl;
277 loadToProperty = true;
278 }
279 else if((fabs(csTecFramePtr->writeLimitMin) - val) < 0.0001) {
280 _logger(Loger::INFO) << __func__ << ": "
281 << "Answer value mismatch with set value. Answer: " << val
282 << " set value: " << csTecFramePtr->csObject->getValue()
283 << ", but min limit has been reached." << std::endl;
284 loadToProperty = true;
285 }
286 else {
287 _logger(Loger::ERROR) << __func__ << ": "
288 << "Answer value mismatch with set value. Answer: " << val
289 << " set value: " << csTecFramePtr->csObject->getValue() << std::endl;
290 }
291
292 if(loadToProperty &&
293 this->_tecReadFramesMap.find(frame.getParameterOffset()) != this->_tecReadFramesMap.end()) {
294 auto& csTecFramePtr = _tecReadFramesMap[frame.getParameterOffset()];
295 csTecFramePtr->csObject->setValue(val);
296 }
297 }
298 }
299 else {
300 _logger(Loger::ERROR) << __func__ << ": "
301 << "Frame with parameter: " << frame.getParameterOffset() << " not found in write map."
302 << std::endl;
303 }
304 }
305 else {
306 if(this->_tecReadFramesMap.find(frame.getParameterOffset()) != this->_tecReadFramesMap.end()) {
307 auto& csTecFramePtr = _tecReadFramesMap[frame.getParameterOffset()];
308
309 if(csTecFramePtr->type == TecFrame::TecFrameType::TecParameterValueRead) {
310 if(csTecFramePtr->retValueType == TecFrame::TecParameterValueType::TecValueIsFloat) {
311 _logger(Loger::DETAIL) << __func__ << ": "
312 << "Frame : " << frame.getParameterOffset()
313 << " -> Value is float: " << frame.getFloatParameterValue() << std::endl;
314 double val = frame.getFloatParameterValue();
315 if(csTecFramePtr->csObject) csTecFramePtr->csObject->setValue(val);
316 }
317 else if(csTecFramePtr->retValueType == TecFrame::TecParameterValueType::TecValueIsInteger) {
318 _logger(Loger::DETAIL) << __func__ << ": "
319 << "Frame : " << frame.getParameterOffset()
320 << " -> Value is int: " << frame.getIntParameterValue() << std::endl;
321 int val = frame.getIntParameterValue();
322 if(csTecFramePtr->csObject) csTecFramePtr->csObject->setValue(val);
323 }
324
325 if(csTecFramePtr->readScalingFactor && (csTecFramePtr->csObject))
326 csTecFramePtr->csObject->setValue(csTecFramePtr->csObject->getValue() * csTecFramePtr->readScalingFactor);
327 }
328 else if(csTecFramePtr->type == TecFrame::TecFrameType::TecParameterLimitRead) {
329 _logger(Loger::WARNING) << __func__ << ": "
330 << "This is answer to read limit frame. Not supported yet. " << std::endl;
331 }
332 else {
333 _logger() << __func__
334 << "Upsss !! .. it seems to be an answer to write frame, but you should never see this message. "
335 "Something is screwed. "
336 << std::endl;
337 }
338 }
339 else {
340 _logger(Loger::ERROR) << __func__ << ": "
341 << "Frame with parameter: " << frame.getParameterOffset() << " not found in read map."
342 << std::endl;
343 }
344 }
345 }
346 else {
347 switch(frame.getState()) {
348 case TecFrame::TecFrameCRCError:
349 _logger(Loger::ERROR) << __func__ << ": "
350 << "Frame <" << frame.getParameterOffset()
351 << "> with invalid CRC. Frame type: " << frame.getFrameType() << std::endl;
352 if(_logger.getLogLevel() >= Loger::DETAIL) {
353 _logger(Loger::DETAIL) << __func__ << ": "
354 << "Frame with Invalid CRC:" << std::endl;
355 frame.print();
356 _logger(Loger::DETAIL) << __func__ << ": "
357 << "Frame printed. " << std::endl;
358 }
359 break;
360 case TecFrame::TecFrameTimedOut:
361 _logger(Loger::ERROR) << __func__ << ": "
362 << "Frame <" << frame.getParameterOffset()
363 << "> timed out. Frame type: " << frame.getFrameType() << std::endl;
364 break;
365 case TecFrame::TecFrameWithError:
366 _logger(Loger::ERROR) << __func__ << ": "
367 << "Frame <" << frame.getParameterOffset() << "> with error | Code <"
368 << (uint16_t)frame.getErrorCode() << ">. Frame type: " << frame.getFrameType()
369 << std::endl;
370 break;
371 default:
372 _logger(Loger::ERROR) << __func__ << ": "
373 << "Frame <" << frame.getParameterOffset()
374 << "> in undefined state. Frame type: " << frame.getFrameType() << std::endl;
375 if(_logger.getLogLevel() >= Loger::DETAIL) {
376 _logger(Loger::DETAIL) << __func__ << ": "
377 << "Undefined state frame: " << std::endl;
378 frame.print();
379 _logger(Loger::DETAIL) << __func__ << ": "
380 << "Frame printed. " << std::endl;
381 }
382
383 break;
384 }
385 }
