134void M_VHDL_ENT_DAQCtrl::initXAxisTimingParameters() {
135 _logger.setLogLevel(_daq->getLogLevel());
136
137 if(_mainFrequency == 0) {
138 _mainFrequency = 81250000;
139 _logger(Loger::WARNING) << __func__ << ": " << "Main frequency NOT set. Using default value: " << _mainFrequency
140 << std::endl;
141 }
142
143 if(_daqDataBusWidth == 0) {
144 _daqDataBusWidth = 256;
145 _logger(Loger::WARNING) << __func__ << ": "
146 << "DAQ DMA bus width not set. Using default value: " << _daqDataBusWidth << std::endl;
147 }
148
149 std::string units;
150 long int timeScale = 1;
151 switch(_plotXAxisTimeUnits->getValue()) { // 0 - sec, 1 - ms, 2 - us, 3 - ns, 4 - ps, 5 - fs
152 case 1: // ms
153 units = "ms";
154 timeScale = 1e3;
155 break;
156 case 2: // us
157 units = "us";
158 timeScale = 1e6;
159 break;
160 case 3: // ns
161 units = "ns";
162 timeScale = 1e9;
163 break;
164 case 4: // ps
165 units = "ps";
166 timeScale = 1e12;
167 break;
168 default:
169 units = "s";
170 timeScale = 1;
171 }
172
173 /*
174 uint32_t hwDaqSamplePeriod = 1;
175 if (_hwSamplePeriod)
176 hwDaqSamplePeriod = _hwSamplePeriod->getValue();
177
178 if (hwDaqSamplePeriod == 0)
179 hwDaqSamplePeriod = 1;
180 */
181
182 auto dws = _downSampleRatio->getValue() + 1;
183 if(dws < 1) dws = 1;
184
185 uint32_t bitWidth = _daq->getSequencesNumber() * 32;
186 if(_algDivider == 0) _algDivider = 1;
187
188 uint32_t finalDivider = dws * _algDivider;
189 if(finalDivider < (bitWidth / _daqDataBusWidth)) finalDivider = (bitWidth / _daqDataBusWidth);
190
191 // double timeStep = timeScale * 1.0 / (_mainFrequency / dws / hwDaqSamplePeriod) * (bitWidth / _daqDataBusWidth);
192 double timeStep = timeScale * 1.0 / (_mainFrequency / finalDivider);
193
194 _logger(Loger::INFO) << ": " << "Freq: " << _mainFrequency << ". Down.smpl: " << dws << ". Alg.div.: " << _algDivider
195 << " bitWidth: " << bitWidth << ", dataBusWidth: " << _daqDataBusWidth
196 << ", final div.: " << finalDivider << ". Time factor : " << timeScale
197 << ". Final time step: " << timeStep << " " << units << std::endl;
198 ;
199
200 _daq->setXAxisSettings(0, timeStep, units);
