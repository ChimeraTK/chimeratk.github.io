37CsaCoarseTuningDetector::LIMIT CsaCoarseTuningDetectorCtrlOutValue::checkSafetyCondition(bool updateInput) {
38 if((_disable->getValue() & DISABLE_MASK::ALL) || (_disable->getValue() & DISABLE_MASK::SAFETY)) {
39 // std::cout << "checkSafetyCondition::disabled" << std::endl;
40 return LIMIT::IN_RANGE;
41 }
42
43 // tests
44 if(triggerSafetyAlwaysTest) return LIMIT::MAX_CROSSED;
45 if(triggerSafetyOnceTest) {
46 triggerSafetyOnceTest = false;
47 return LIMIT::MAX_CROSSED;
48 }
49 //
50 double maxVal = _maxSafetyValue->getValue();
51 double minVal = _minSafetyValue->getValue();
52
53 // when parameters are equal safety check is disabled
54 if(maxVal == minVal) return LIMIT::IN_RANGE;
55
56 // auto detectorValue = _currentValue->getValue();
57 if(updateInput) this->getCurrentValue();
58
59 if(_ctdError.value()) {
60 _logger(Loger::INFO) << __func__ << ": "
61 << "Detector error. Trigger safety. Error: " << _ctdError.msg() << std::endl;
62 return LIMIT::MAX_CROSSED;
63 }
64
65 if(_safetyMode->getValue() == 0) // detector output
66 return CsaCoarseTuningDetector::checkSafetyCondition();
67 else if(_safetyMode->getValue() == 1) { // STD
68 if(_stdValue->getValue() > maxVal) {
69 _logger(Loger::INFO) << __func__ << ": "
70 << "Safety std value crossed. std: " << _stdValue->getValue() << ", max value: " << maxVal
71 << std::endl;
72 return LIMIT::MAX_CROSSED;
73 }
74 }
75 else if(_safetyMode->getValue() == 2) { // RMS
76 if(_rmsValue->getValue() > maxVal) {
77 _logger(Loger::INFO) << __func__ << ": "
78 << "Safety rms value crossed. rms: " << _rmsValue->getValue() << ", max value: " << maxVal
79 << std::endl;
80 return LIMIT::MAX_CROSSED;
81 }
82 }
83 else if(_safetyMode->getValue() == 3) { // MEAN
84 if(_meanValue->getValue() > maxVal) {
85 _logger(Loger::INFO) << __func__ << ": "
86 << "Safety mean value crossed. mean: " << _meanValue->getValue() << ", max value: " << maxVal
87 << std::endl;
88 return LIMIT::MAX_CROSSED;
89 }
90 else if(_meanValue->getValue() < minVal) {
91 _logger(Loger::INFO) << __func__ << ": "
92 << "Safety mean value crossed. mean: " << _meanValue->getValue() << ", min value: " << minVal
93 << std::endl;
94 return LIMIT::MIN_CROSSED;
95 }
96 }
97 else if(_safetyMode->getValue() == 4) { // RMS and STD
98 if(_rmsValue->getValue() > maxVal || _stdValue->getValue() > minVal) {
99 _logger(Loger::INFO) << __func__ << ": "
100 << "Safety rms or std values crossed. rms: " << _rmsValue->getValue()
101 << " - allowed: " << maxVal << ", std: " << _stdValue->getValue()
102 << " - allowed: " << minVal << std::endl;
103 return LIMIT::MAX_CROSSED;
104 }
105 }
106 else if(_safetyMode->getValue() == 5) { // RMS with offset and STD
107 if(_rmsValue->getValue() > maxVal || _rmsValue->getValue() < minVal ||
108 _stdValue->getValue() > _extraSafetyValue->getValue()) {
109 _logger(Loger::INFO) << __func__ << ": "
110 << "Safety rms or std values crossed. rms: " << _rmsValue->getValue() << " - allowed: ("
111 << minVal << "," << maxVal << "), std: " << _stdValue->getValue()
112 << " - allowed: " << _extraSafetyValue->getValue() << std::endl;
113 return LIMIT::MAX_CROSSED;
114 }
115 }
116 return LIMIT::IN_RANGE;
117}

119void CsaCoarseTuningDetectorCtrlOutValue::addDetector(
120 boost::shared_ptr<M_RapidX_ENT_StateSpaceCtrl> detector, boost::shared_ptr<CsaVector<float>> ctrlOutDataVector) {
121 std::uintptr_t address = reinterpret_cast<std::uintptr_t>(detector.get());
122 for(auto& item : _detectorUniqueIdVector) {
123 if(item == address) {
124 _logger(Loger::WARNING) << __func__ << ": "
125 << "Detector with address: " << address << " already added." << std::endl;
126 return;
127 }
128 }
129 _detectorUniqueIdVector.push_back(address);
130 _detectorVector.push_back(detector);
131

133 boost::shared_ptr<StatisticsForCsaVector<float>>(new StatisticsForCsaVector<float>()));
134 _ctrlOutDataVectorStats[_ctrlOutDataVectorStats.size() - 1]->setCsaVector(ctrlOutDataVector);
135 _ctrlOutDataVectorStats[_ctrlOutDataVectorStats.size() - 1]->setEnableMean(true);
136 _ctrlOutDataVectorStats[_ctrlOutDataVectorStats.size() - 1]->setEnableRms(true);
137 _ctrlOutDataVectorStats[_ctrlOutDataVectorStats.size() - 1]->setEnableStdDev(true);

140double CsaCoarseTuningDetectorCtrlOutValue::getCurrentValue() {

151 id = 0;
152
153 _ctrlOutDataVectorStats[id]->calculate();
154
155 _meanValue->setValue(_ctrlOutDataVectorStats[id]->getMean());
156 _rmsValue->setValue(_ctrlOutDataVectorStats[id]->getRms());
157 _stdValue->setValue(_ctrlOutDataVectorStats[id]->getStdDev());
158
159 if(_mode->getValue() == 1) {
160 _currentValue->setValue(_ctrlOutDataVectorStats[id]->getMean());
161 }
162 else {
163 _currentValue->setValue(_detectorVector[id]->getOutputValue());
164 }
165
