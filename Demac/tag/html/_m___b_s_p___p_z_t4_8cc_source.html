313// check FPGA clocks on the board
314bool M_BSP_PZT4::validateFpgaClocks() {
315 std::stringstream ss;
316 bool allClkOK = true;
317 bool clkOK[PZT4_CHANNEL_NUMBER] = {true, true, true, true};
318 int clkLimits[PZT4_CHANNEL_NUMBER * 2] = {
319 63000000, 62000000, 25500000, 24500000, 63000000, 62000000, 16000000, 15000000};
320 int currentClk[4];
321 for(int index = 0; index < PZT4_CHANNEL_NUMBER; index++) {
322 currentClk[index] = _clockStatus->getFrequency(index);
323 if(currentClk[index] > clkLimits[2 * index] || currentClk[index] < clkLimits[2 * index + 1]) {
324 allClkOK = false;
325 clkOK[index] = false;
326 }
327 }
328 if(!allClkOK) {
329 _status = STATUS::ST_INCORRECT_FPGA_CLOCK;
330 ss << ST_INCORRECT_FPGA_CLOCK_STR;
331 for(int index = 0; index < PZT4_CHANNEL_NUMBER; index++)
332 if(!clkOK[index]) ss << (index + 1) << ", ";
333 std::string str = ss.str();
334 str = str.substr(0, str.size() - 2);
335 str += std::string(" incorrect. Contact expert.");
336 _statusStr->setValue(str);
337
338 _logger(Loger::WARNING) << __func__ << ": "
339 << "Incorrect FPGA clock."
340 << " Clock 1: " << currentClk[0] << " Clock 2: " << currentClk[1]
341 << " Clock 3: " << currentClk[2] << " Clock 4: " << currentClk[3] << std::endl;
342 }
343 return allClkOK;
344}

346bool M_BSP_PZT4::validateStatusWithAdcData() {
347 // general validation code part - validation with ADC data
348 if(_adc->adcChannelAvailable()) { // this value is known on startup and is not changing during the runtime
349 // get ADC reset status
350 if(_adc->getResetStatus()) {
351 _statusStr->setValue(ST_ADC_IN_RESET_STR);
352 _status = STATUS::ST_ADC_IN_RESET;
353 }
354 else {
355 collectAdcData();
356 if(_currentVectorSize >= _mininumVectorSize) {
357 if(isAdcFrozen()) {
358 _logger(Loger::WARNING) << __func__ << ": "
359 << "ADC readout are frozen. Possible power lost." << std::endl;
360 _statusStr->setValue(ST_ADC_FROZEN_STR);
361 _status = STATUS::ST_ADC_FROZEN;
362 }
363 else if((_adc->getHighVoltage(0) < 85) || (_adc->getHighVoltage(2) > -85)) {
364 _logger(Loger::WARNING) << __func__ << ": "
365 << "PZT4 RTM Main Voltage is out of range. Check your hardware: HV+: "
366 << _adc->getHighVoltage(0) << ", HV-: " << _adc->getVoltage(2) << std::endl;
367 _statusStr->setValue(ST_PWR_SUPPLY_ERROR_STR);
368 _status = STATUS::ST_PWR_SUPPLY_ERROR;
369 }
370 }
371 }
372 }
373 else
374 _statusStr->setValue(ST_ADC_NOT_PRESENT_STR); // adc readout are not available in firmware, cannot validate
375
376 if(_status == STATUS::ST_ADC_FROZEN || _status == STATUS::ST_PWR_SUPPLY_ERROR) return false;
377 return true;
378}

380bool M_BSP_PZT4::validateOvercurrentRegister() {
381 std::stringstream ss;
382
383 // overcurrent validation code part - validation with OC register
384 _overCurrentAlarm = 0x0F; // no error;
385 if(_ovErrorAlarmReg) {
386 _overCurrentAlarm = _ovErrorAlarmReg->getValueFromDevice();
387 }
388 //_overCurrentAlarm = _hwdErrorAlarmTest->getValue(); //just for tests purpose
389 if(_overCurrentAlarm != 0xF) {

394 if((_overCurrentAlarm & (1 << index)) == 0) ss << (channelId + 1) << ", ";
395 }
396 std::string str = ss.str();
397 str = str.substr(0, str.size() - 2);
398 str += std::string(". Contact expert.");
399 _statusStr->setValue(str);
400 // set status
401 _status = STATUS::ST_HWD_CHANNEL_ERROR_ALARM;

417bool M_BSP_PZT4::isAdcFrozen() {
418 // check data - if any of channels had different data ADC is not frozen
419 bool dataFrozen = true;
420 for(int i = 0; i < PZT4_CHANNEL_NUMBER && dataFrozen; i++) {
421 for(uint32_t index = 0; index < _adcVoltageData[i].size() - 1 && dataFrozen; index++) {
422 if(_adcVoltageData[i].at(index) != _adcVoltageData[i].at(index + 1)) dataFrozen = false;
423 if(_adcCurrentData[i].at(index) != _adcCurrentData[i].at(index + 1)) dataFrozen = false;
424 if(_adcSensorData[i].at(index) != _adcSensorData[i].at(index + 1)) dataFrozen = false;
425 if(_adcHighVoltageData[i].at(index) != _adcHighVoltageData[i].at(index + 1)) dataFrozen = false;
426 }
427 }
428 // remove the oldest component from vector
429 for(int i = 0; i < PZT4_CHANNEL_NUMBER; i++) {
430 _adcVoltageData[i].erase(_adcVoltageData[i].begin());
431 _adcCurrentData[i].erase(_adcCurrentData[i].begin());
432 _adcSensorData[i].erase(_adcSensorData[i].begin());
433 _adcHighVoltageData[i].erase(_adcHighVoltageData[i].begin());
434 }
435
436 return dataFrozen;
437}

441void M_BSP_PZT4::passStatusToChannels() {
442 // if board is in full reset or FPGA clocks are not correct or ADC readout are frozen or power supply voltage not
443 // correct all channels are not operational
444 if(_status == STATUS::ST_PZT4_IN_RESET || _status == STATUS::ST_INCORRECT_FPGA_CLOCK ||
445 _status == STATUS::ST_ADC_FROZEN || _status == STATUS::ST_PWR_SUPPLY_ERROR || _status == ST_ADC_IN_RESET) {
446 for(auto& channel : _channels) channel->_pzt4HardwareOK = false;
447 if(_pzt4HardwareOK) {
448 _logger(Loger::ERROR) << __func__ << ": "
449 << "PZT4 BSP Hardware Status is ERROR. Pass information to all channels." << std::endl;
450 }
451 _pzt4HardwareOK = false;
452 return;
453 }
454
455 if(!_pzt4HardwareOK) {
456 _pzt4HardwareOK = true;
457 _logger(Loger::WARNING) << __func__ << ": "
458 << "PZT4 BSP Hardware Status is back to OK. Toggle PZT4 BSP reset." << std::endl;
459 setReset(true);
460 std::this_thread::sleep_for(std::chrono::milliseconds(1));
461 setReset(false);
462 std::this_thread::sleep_for(std::chrono::milliseconds(1));
463 }
464
465 // not check overcurrent and if set to false then hardware is OK, otherwise not
466 for(uint32_t index = 0; index < PZT4_CHANNEL_NUMBER; index++)
467 if((_overCurrentAlarm & (1 << (PZT4_CHANNEL_NUMBER - index - 1))) == 0)
468 _channels[index]->_pzt4HardwareOK = false;
469 else
470 _channels[index]->_pzt4HardwareOK = true;
471}
