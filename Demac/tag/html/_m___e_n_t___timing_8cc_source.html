138void M_ENT_Timing::measureMainTriggerFrequency() {
139 // auto readStart = std::chrono::high_resolution_clock::now();
140 auto index = _timingTriggerSelect->getValueFromDevice();
141 // auto readDuration = std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::high_resolution_clock::now()
142 // - readStart).count(); std::cout << "Register read takes: " << readDuration << " us." << std::endl;
143
144 if(index >= _channelAmount) index = _channelAmount - 1;
145 if(_prevTimingTriggerSelect == index) {
146 auto currMainTriggerCnt = _timingCounterVec[index]->getValueFromDevice();
147 auto diff = currMainTriggerCnt - _prevMainTriggerCnt;
148 auto currentUpdate = std::chrono::high_resolution_clock::now();
149 auto duration =
150 std::chrono::duration_cast<std::chrono::milliseconds>(currentUpdate - _prevMainTriggerUpdate).count();
151
152 if(diff > 0) {
153 // auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(currentUpdate - _prevMainTriggerUpdate).count();
154 if(!_firstFreqCalcInValid) {
155 double freq = 1.00 /
156 ((1.00 * duration / diff) /
157 1000.0); // multiply by 1.00 to convert to float/double, if not then for difference (diff) calculated
158 // value is infinity; divide by 1000 to scale from ms to seconds to get Hz
159 // std::cout << _moduleName << " : Main trigger: duration between changes " << duration << " ms, diff: " << diff
160 // << " , freq[Hz]: " << freq << std::endl;
161 _timingMainTriggerFrequency->setValue(freq);
162 }
163 else
164 _firstFreqCalcInValid = false;
165 _prevMainTriggerCnt = currMainTriggerCnt;
166 _prevMainTriggerUpdate = currentUpdate;
167 }
168 else if(duration > 10 * 1000) {
169 _timingMainTriggerFrequency->setValue(0);
170 _firstFreqCalcInValid = true;
171 }
172 }
173 else {
174 _prevMainTriggerCnt = _timingCounterVec[index]->getValueFromDevice();
175 _prevMainTriggerUpdate = std::chrono::high_resolution_clock::now();
176 _prevTimingTriggerSelect = index;
177 _firstFreqCalcInValid = true;
178 }
