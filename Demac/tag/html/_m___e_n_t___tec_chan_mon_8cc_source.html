224 void M_ENT_TecChanMon::commandCb() {
225  auto cmd = _command->getValue();
226 
227  if(cmd >= CMD_CREATE_READ_VECTOR_FOR_GROUP_1 && cmd <= CMD_CREATE_READ_VECTOR_FOR_GROUP_9) {
228  int groupId = cmd - CMD_CREATE_READ_VECTOR_FOR_GROUP_1 + 1;
229  this->createReadVector(groupId);
230  return;
231  }
232 
233  if(cmd == CMD_APPLY_CHANGES_OF_PERIODIC_VECTOR) {
234  int32_t creationMask = _enablePeriodicReadreadback->getValue();
235  if(creationMask != _periodicReadreadbackIsEnabled->getValue()) {
236  createPeriodicVector(creationMask);
237  _periodicReadreadbackIsEnabled->setValue(creationMask);
238  }
239  return;
240  }
241  _logger(Loger::ERROR) << __func__ << ": "
242  << "Command unknown. Command ID: " << cmd << std::endl;
243 }
