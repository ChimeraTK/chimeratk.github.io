179void CsaStepperMotor::setCsControlAllowedWithCnt(bool allowed) {
180 if(_csAllowedCnt == 0 && !allowed) setCsControlAllowed(false);
181 if(_csAllowedCnt == 1 && allowed) setCsControlAllowed(true);
182
183 if(!allowed) _csAllowedCnt++;
184
185 if(allowed && _csAllowedCnt) _csAllowedCnt--;
186}

188void CsaStepperMotor::setCsControlAllowed(bool allowed) {
189 _csControlAllowed = allowed;
190
191 if(_csControlAllowed) {
192 _autostart->synchronize();
193 _enableSoftwareLimits->synchronize();
194 _enable->synchronize();
195
196 _setPositionInSteps->setValue(_positionInSteps->getValue(), false);
197 _setPositionInSteps->synchronize();
198 _setPosition->setValue(_position->getValue(), false);
199 _setPosition->synchronize();
200
201 _csControlAllowedFlag->setValue(1);
202
203 _csAllowedCnt = 0;
204 }
205 else {
206 _csControlAllowedFlag->setValue(0);
207 }
208}
