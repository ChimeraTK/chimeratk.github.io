116void M_ENT_LowLatencyLinks::decodeStatus() {
117 bool updateString = true;
118
119 for(uint32_t i = 0; i < _lllStatusVec.size(); i++) {
120 uint32_t statusAsInt = _lllStatusVec[i]->getValue();
121
122 _statusVector[i]->PLL_OK = (statusAsInt & PLL_OK_MASK);
123 _statusVector[i]->RESET_DONE = (statusAsInt & RESET_DONE_MASK);
124 _statusVector[i]->EIDLE = (statusAsInt & EIDLE_MASK);
125 _statusVector[i]->ALIGN = (statusAsInt & ALIGN_MASK);
126 _statusVector[i]->SYNCH = (statusAsInt & SYNCH_MASK);
127 _statusVector[i]->FRAME = (statusAsInt & FRAME_MASK);
128 _statusVector[i]->FIFO = (statusAsInt & FIFO_MASK);
129 _statusVector[i]->MMLCLOCK = (statusAsInt & MMLCLOCK_MASK);
130
131 _statusVector[i]->LOSS_OFS = ((statusAsInt >> 6) & 0x03);
132 _statusVector[i]->RX_CLK_COR = ((statusAsInt >> 8) & 0x03);
133 _statusVector[i]->RX_DISP = ((statusAsInt >> 11) & 0x0F);
134 _statusVector[i]->BUF_ST = ((statusAsInt >> 15) & 0x03);
135 _statusVector[i]->ERROR_CNT = ((statusAsInt >> 18) & 0x1F);
136
137 // do update of string only every one second. Faster doesn't make sense
138
139 if(updateString) {
140 std::string status("");
141
142 std::string diffDelimiter(" ");
143 if(_statusVector[i]->FIFO) {
144 status += "-- " + diffDelimiter;
145 }
146 else {
147 status += "OK ";
148 }
