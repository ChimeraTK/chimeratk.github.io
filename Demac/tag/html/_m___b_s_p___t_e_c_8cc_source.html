217bool M_BSP_TEC::testCommDuringInit() {
218 //_tecController->addFrame(_tecAddress->getValue(), TecFrame::ParameterSystemFlashSaveOff, TecFrame::TecParameterValueRead, 1);
219 _tecController->clearAllFrames();
220 _tecController->addFrame(
221 _tecAddress->getValue(), TecFrame::ParameterSystemFlashSaveOff, TecFrame::TecParameterValueSet, 1, (int)1);
222 _tecController->constructAllFrames();
223 _tecController->copyCommandsToHardware();
224 _tecController->triggerHardwareCommands();
225
226 int timeoutCnt = 0;
227 int timeoutCntMax = 1000;
228 while(!_tecController->readHardwareResponses() && timeoutCnt < timeoutCntMax) {
229 std::this_thread::sleep_for(std::chrono::milliseconds(10));
230 timeoutCnt++;
231 }
232 if(timeoutCnt == timeoutCntMax) {
233 _logger(Loger::ERROR) << __func__ << ": "
234 << "Timeout in FPGA communication. Initialization failed" << std::endl;
235 return false;
236 }
237
238 _tecController->copyResponsesFromHardware();
239 _tecController->parseAllFrames();
240 for(uint8_t i = 0; i < _tecController->getFrameCount(); i++) {
241 std::shared_ptr<TecFrame> frame = _tecController->getFrameAt(i);
242 switch(frame->getState()) {
243 case TecFrame::TecFrameParsed:
244 //_flashStoringDisabledOnStart = true;
245 _logger(Loger::INFO) << __func__ << ": "
246 << "Frame <" << frame->getParameterOffset()
247 << "> (TecFrame::ParameterSystemFlashSaveOff) parsed successfully | Float value <"
248 << frame->getFloatParameterValue() << "> | Integer value <"
249 << frame->getIntParameterValue() << ">. Frame type: " << frame->getFrameType()
250 << std::endl;
251 break;
252 case TecFrame::TecFrameCRCError:
253 _logger(Loger::ERROR) << __func__ << ": "
254 << "Frame <" << frame->getParameterOffset()
255 << "> (TecFrame::ParameterSystemFlashSaveOff) with invalid CRC. Frame type: "
256 << frame->getFrameType() << std::endl;
257 break;
258 case TecFrame::TecFrameTimedOut:
259 _logger(Loger::ERROR) << __func__ << ": "
260 << "Frame <" << frame->getParameterOffset()
261 << "> (TecFrame::ParameterSystemFlashSaveOff) timed out. Frame type: "
262 << frame->getFrameType() << std::endl;
263 break;
264 case TecFrame::TecFrameWithError:
265 _logger(Loger::ERROR) << __func__ << ": "
266 << "Frame <" << frame->getParameterOffset()
267 << "> (TecFrame::ParameterSystemFlashSaveOff) with error | Code <"
268 << (uint16_t)frame->getErrorCode() << ">. Frame type: " << frame->getFrameType()
269 << std::endl;
270 break;
271 default:
272 _logger(Loger::ERROR) << __func__ << ": "
273 << "Frame <" << frame->getParameterOffset()
274 << "> (TecFrame::ParameterSystemFlashSaveOff) in undefined state. Frame type: "
275 << frame->getFrameType() << std::endl;
276 break;
277 }
278 }
279 return true;
280}

285 if(this->_moduleName.size() == 0) {

323 }

330

340 checkReadVector(_generalController, false);
341 }

349void M_BSP_TEC::updateFromDevice() {
350 if(!_initialized) {
351 _logger(Loger::ERROR) << __func__ << ": "
352 << "TEC BSP not initialized. Restart server." << std::endl;
353 return;
354 }
355
356 DeviceModuleBase::updateFromDevice();
357
358 if(_enableTec && _enableTec->getValue() == 0) {
359 _logger(Loger::WARNING) << __func__ << ": "
360 << "TEC is disabled." << std::endl;
361 _operationTypeInProgress->setValue(0);
362 _communicationStatus->setValue(2);
363 return;
364 }
365
366 if(_tecCommunicationThreadActive == true) {
367 // std::cout << "Commuunication thread still active" << std::endl;
368 return;
369 }
370 _operationTypeInProgress->setValue(0);
371
372 // WRITE SECTION
373 if(this->checkWriteVector(_generalController, _runCommInSeparateThread)) {
374 if(_runCommInSeparateThread) return;
375 }
376
377 if(this->checkWriteVector(_channel1Controller, _runCommInSeparateThread)) {
378 if(_runCommInSeparateThread) return;
379 }
380
381 if(this->checkWriteVector(_channel2Controller, _runCommInSeparateThread)) {
382 if(_runCommInSeparateThread) return;
383 }
384
385 // This section must be after write section, because in write section RESET command is physically send.
386 // It reads all parameters after TEC reset or after communication has been reestablished
387 if(_generalController->getTecResetSend() ||
388 (_communicationStatusPrevious != 0 && _communicationStatus->getValue() == 0)) {
389 auto execUpdate = this->isExecuteAutoUpdate();
390 if(!execUpdate) return;
391
392 int readAllDataCntMax = 5;
393 if(_generalController->getTecResetSend() && (readAllDataCntMax - _readAllDataCnt)) {
394 _logger(Loger::INFO) << __func__ << ": "
395 << "TEC has been reseted. Read all data to refresh values in "
396 << int(((readAllDataCntMax - _readAllDataCnt++) * this->_autoupdatePeriodMs) / 1000)
397 << " sec." << std::endl;
398 //_logger(Loger::INFO) << __func__ << ": " << "TEC communication established. Read all data to refresh values in "
399 //<< int(((readAllDataCntMax - _readAllDataCnt++)*this->_autoupdatePeriodMs)/1000) << " sec." << std::endl;
400 return;
401 }
402
403 if(_generalController->getTecResetSend()) {
404 _generalController->clearTecResetSend();
405 _readAllDataCnt = 0;
406 _logger(Loger::INFO) << __func__ << ": "
407 << "Send read to all readable parameters." << std::endl;
408 }
409 else
410 _logger(Loger::INFO) << __func__ << ": "
411 << "TEC communication established again. Send read to all readable parameters." << std::endl;
412
413 if(_generalMonitor) _generalMonitor->createReadVectorForAllReadFrames();
414 if(_channel1Monitor) _channel1Monitor->createReadVectorForAllReadFrames();
415 if(_channel2Monitor) _channel2Monitor->createReadVectorForAllReadFrames();
416 if(_channel1Controller) _channel1Controller->createReadVectorForAllReadFrames();
417 if(_channel2Controller) _channel2Controller->createReadVectorForAllReadFrames();
418 if(_generalController) _generalController->createReadVectorForAllReadFrames();
419 }
420 _communicationStatusPrevious = _communicationStatus->getValue();
421
422 // READ ON DEMAND SECTION
423 if(this->checkReadVector(_generalController, _runCommInSeparateThread)) {
424 _logger(Loger::MAX_DETAIL) << __func__ << ": "
425 << "checkReadVector for _generalController. " << std::endl;
426 if(_runCommInSeparateThread) return;
427 }
428 if(this->checkReadVector(_channel1Controller, _runCommInSeparateThread)) {
429 _logger(Loger::MAX_DETAIL) << __func__ << ": "
430 << "checkReadVector for _channel1Controller. " << std::endl;
431 if(_runCommInSeparateThread) return;
432 }
433 if(this->checkReadVector(_generalMonitor, _runCommInSeparateThread)) {
434 _logger(Loger::MAX_DETAIL) << __func__ << ": "
435 << "checkReadVector for _generalMonitor. " << std::endl;
436 if(_runCommInSeparateThread) return;
437 }
438 if(this->checkReadVector(_channel1Monitor, _runCommInSeparateThread) && _runCommInSeparateThread) {
439 _logger(Loger::MAX_DETAIL) << __func__ << ": "
440 << "checkReadVector for _channel1Monitor. " << std::endl;
441 return;
442 }
443 if(this->checkReadVector(_channel2Monitor, _runCommInSeparateThread) && _runCommInSeparateThread) {
444 _logger(Loger::MAX_DETAIL) << __func__ << ": "
445 << "checkReadVector for _channel2Monitor. " << std::endl;
446 return;
447 }
448
449 // PERIODIC READ SECTION
450
451 this->_autoupdatePeriodMs = _periodicReadPeriodMs->getValue();
452 if(_autoupdatePeriodMs < 1000) {
453 this->_autoupdatePeriodMs = 1000;
454 }
455 this->periodicRead(_runCommInSeparateThread);
456
457 // CHECK Driver Status and read errors in case of device is in ERROR state
458 this->driverStatusErrorCheck();
459}

485void M_BSP_TEC::periodicRead(bool runThread) {
486 _rebuildGlobalPeriodicVector = (_rebuildGlobalPeriodicVector | checkReadPeriodicVector(_channel1Controller) |
487 checkReadPeriodicVector(_channel2Controller) | checkReadPeriodicVector(_generalMonitor) |
488 checkReadPeriodicVector(_channel1Monitor) | checkReadPeriodicVector(_channel2Monitor));
489
490 if(_rebuildGlobalPeriodicVector) {
491 _globalPeriodicReadsVector.clear();
492 _infoVec.clear();
493
494 addToGlobalPeriodicReadsVector(_channel1Controller);
495 addToGlobalPeriodicReadsVector(_channel2Controller);
496 addToGlobalPeriodicReadsVector(_generalMonitor);
497 addToGlobalPeriodicReadsVector(_channel1Monitor);
498 addToGlobalPeriodicReadsVector(_channel2Monitor);
499 _rebuildGlobalPeriodicVector = false;
500 this->_lastUpdateTime = std::chrono::milliseconds::zero();
501 _logger(Loger::INFO) << __func__ << ": "
502 << "Rebuild global periodic vector. New vector size is: " << _globalPeriodicReadsVector.size()
503 << std::endl;
504 }
505
506 _execUpdate = this->isExecuteAutoUpdate();
507 if(_execUpdate) {
508 if(_globalPeriodicReadsVector.size()) {
509 startCommunicationToTec(_globalPeriodicReadsVector, _infoVec, OPERATION_TYPE::PERIODIC_READ, runThread);
510 }
511 else { // heart beat feature is checked when periodic read vector is empty
512 if(_disableHearbeat->getValue() == 0 &&
513 _heartBeatFrameVector.size() == 0) { // when not disabled add only once a single frame
514 TecFrame frame(_tecAddress->getValue(), TecFrame::DeviceType, TecFrame::TecParameterValueRead, 1);
515 _heartBeatFrameVector.push_back(frame);
516 _heartBeatInfoVec.push_back(
517 CallObjectInfo{0, static_cast<uint32_t>(_heartBeatFrameVector.size()), _generalMonitor});
518 }
519 else if(_disableHearbeat->getValue() &&
520 _heartBeatFrameVector.size()) { // when disabled and heartbeat frame still in vector then clean vectors
521 _heartBeatFrameVector.clear();
522 _heartBeatInfoVec.clear();
523 }
524 // it must be executed even when vectors are empty, because inside the Thread the communication status flag
525 // ("_communicationStatus") is updated.
526 startCommunicationToTec(_heartBeatFrameVector, _heartBeatInfoVec, OPERATION_TYPE::HEARTBEAT, runThread);
527 }
528 }
529 if(_generalController && _execUpdate) _generalController->update();
530}

532void M_BSP_TEC::startCommunicationToTec(std::vector<TecFrame>& framesVector,
533 std::vector<CallObjectInfo>& framesInfoVector, OPERATION_TYPE operationType, bool runThread) {
534 if(runThread) {
535 if(_tecCommunicationThread.joinable()) _tecCommunicationThread.join();
536 _logger(Loger::MAX_DETAIL) << __func__ << ": "
537 << "Starting communication thread." << std::endl;
538 _tecCommunicationThreadActive = true;
539 _tecCommunicationThread = std::thread(&M_BSP_TEC::tecCommunicationThread, this, framesVector, framesInfoVector);
540 _operationTypeInProgress->setValue(operationType);
541 }
542 else {
543 _operationTypeInProgress->setValue(operationType);

617// this buffer size is adjusted to
618// UNIO firmware where module is instantiated with generic set to 8
619// "INST_ENT_TEC1091_CTRL: entity work.ENT_TEC1091_CTRL" in "ENT_FMC2_UNIO_INST:ENT_FMC_UNIO_TOP"
620// UNIO firmware is used in FMC20 application for lbsync
621
622#define MAX_BUFFER_SIZE 8
623
624void M_BSP_TEC::tecCommunicationThreadCore(
625 const std::vector<TecFrame>& frames, std::vector<CallObjectInfo>& callObjectInfoVec) {
626 _logger(Loger::MAX_DETAIL) << __func__ << ": "
627 << "Start of communication." << std::endl;
628
629 _commErrorDetected = false; // we assume that communication will be fine
630 uint32_t currentFrameIndex = 0;
631 do {
632 _tecController->clearAllFrames();
633
634 for(uint32_t index = currentFrameIndex; index < (currentFrameIndex + MAX_BUFFER_SIZE) && index < frames.size();
635 index++) {
636 _tecController->addFrame((TecFrame&)(frames[index]));
637 }
638
639 _tecController->constructAllFrames();
640
641 auto start = std::chrono::system_clock::now().time_since_epoch();
642 _tecController->copyCommandsToHardware();
643 auto copyToHwd = std::chrono::system_clock::now().time_since_epoch();
644 _logger(Loger::DETAIL) << __func__ << ": "
645 << "Copy to hardware takes (us): "
646 << std::chrono::duration_cast<std::chrono::microseconds>(copyToHwd - start).count()
647 << std::endl;
648
649 _tecController->triggerHardwareCommands();
650 auto trigTime = std::chrono::system_clock::now().time_since_epoch();
651 _logger(Loger::DETAIL) << __func__ << ": "
652 << "Trigger (and physical copy to hardware) takes (ms): "
653 << std::chrono::duration_cast<std::chrono::milliseconds>(trigTime - copyToHwd).count()
654 << std::endl;
655
656 int timeoutCnt = 0;
657 int timeoutCntMax = 1000;
658 while(!_tecController->readHardwareResponses() && timeoutCnt < timeoutCntMax) {
659 std::this_thread::sleep_for(std::chrono::milliseconds(10));
660 timeoutCnt++;
661 }
662 if(timeoutCnt == timeoutCntMax) {
663 _logger(Loger::ERROR) << __func__ << ": "
664 << "Communication with FPGA Timeout detected." << std::endl;
665 _commErrorDetected = true;
666 return;
667 }
668
669 auto readHwdAns = std::chrono::system_clock::now().time_since_epoch();
670 _logger(Loger::DETAIL) << __func__ << ": "
671 << "Data processing by TEC (ms): "
672 << std::chrono::duration_cast<std::chrono::milliseconds>(readHwdAns - trigTime).count()
673 << std::endl;
674
675 _tecController->copyResponsesFromHardware();
676 auto copyFromHwd = std::chrono::system_clock::now().time_since_epoch();
677 _logger(Loger::DETAIL) << __func__ << ": "
678 << "Copy answer from hardware (us): "
679 << std::chrono::duration_cast<std::chrono::microseconds>(copyFromHwd - readHwdAns).count()
680 << std::endl;
681
682 _tecController->parseAllFrames();
683 auto parseAnswer = std::chrono::system_clock::now().time_since_epoch();
684 _logger(Loger::DETAIL) << __func__ << ": "
685 << "Parse answer (us): "
686 << std::chrono::duration_cast<std::chrono::microseconds>(parseAnswer - copyFromHwd).count()
687 << std::endl;
688
689 // std::cout << "callObjectInfoVec size: " << callObjectInfoVec.size() << std::endl;
690
691 for(uint8_t i = 0; i < _tecController->getFrameCount(); i++) {
692 TecFrame frame = *(_tecController->getFrameAt(i));
693 uint32_t correctObjectIndexInfo = 0;
694 uint32_t min = 0;
695 uint32_t max = 0;
696 for(correctObjectIndexInfo = 0; correctObjectIndexInfo < callObjectInfoVec.size(); correctObjectIndexInfo++) {
697 min = callObjectInfoVec[correctObjectIndexInfo].frameStartIndex;
698 max = callObjectInfoVec[correctObjectIndexInfo].frameEndIndex;
699 // std::cout << "MIN: " << min << " MAX: " << max << " Current Index: " << currentFrameIndex << std::endl;
700 if(currentFrameIndex + i >= min && currentFrameIndex + i < max) break;
701 }
702 // std::cout << "correctObjectIndexInfo: " << correctObjectIndexInfo << " min: " << min << " max: " << max << "
703 // currentFrameIndex : " << currentFrameIndex << " : ";
704 callObjectInfoVec[correctObjectIndexInfo].callObject->analizeAnswerFrame(frame);
705
706 // if any frame in current communication round is in CRC_ERROR or TIMEOUT state, then communication error is detected
707 if(_commErrorDetected == false &&
708 (frame.getState() == TecFrame::TecFrameTimedOut || frame.getState() == TecFrame::TecFrameCRCError)) {
709 _commErrorDetected = true;
710 }
711 }
712 currentFrameIndex = currentFrameIndex + MAX_BUFFER_SIZE;
713 } while(currentFrameIndex < frames.size());
714
715 // if after communication CRC or TIMEOUT has not been detected, we can clear ' timeout counter ' , because communication is fine
716 if(!_commErrorDetected) _commErrorsCnt = 0;
717
718 _logger(Loger::MAX_DETAIL) << __func__ << ": "
719 << "End of communication" << std::endl;
720}

722void M_BSP_TEC::tecCommunicationThread(
723 const std::vector<TecFrame>& frames, std::vector<CallObjectInfo> callObjectInfoVec) {
724 _tecCommunicationThreadActive = true;
725
726 if(frames.size()) {
727 _communicationActive->setValue(1);
728 tecCommunicationThreadCore(frames, callObjectInfoVec);
729 _communicationActive->setValue(0);
730 }
731
732 if(_disableHearbeat->getValue()) {
733 _communicationStatus->setValue(2);
734 _commErrorsCnt = 0;
735 _commErrorDetected = false;
736 }
737 else {
738 // const int COMM_ERRORS_CNT_MAX = 3;
739 if(_commErrorDetected && _communicationStatus->getValue() != 1) {
740 _commErrorsCnt++;
741 if(COMM_ERRORS_CNT_MAX == _commErrorsCnt)
742 _logger(Loger::ERROR) << __func__ << ": "
743 << "Timeout or CRC errors detected. Error counter: " << _commErrorsCnt
744 << ". Change communication status to ERROR." << std::endl;
745 else
746 _logger(Loger::WARNING) << __func__ << ": "
747 << "Timeout or CRC errors detected. Error counter: " << _commErrorsCnt
748 << ". Communication status WARNING." << std::endl;
749 _commErrorDetected = false;
750 }
751
752 if(_commErrorsCnt >= COMM_ERRORS_CNT_MAX)
753 _communicationStatus->setValue(1);
754 else if(_commErrorsCnt)
755 _communicationStatus->setValue(3);
756 else
757 _communicationStatus->setValue(0);
758 }
759 _tecCommunicationThreadActive = false;
760}
