90 void CsaStepperMotorPiezo::initialize(boost::shared_ptr<PiezoTuner> piezoActuator) {
91  this->_stepperMotorUnitsConverter->settingChangedReset();
92  this->_stepperMotor->initialize(piezoActuator, this->_stepperMotorUnitsConverter);
93 
94  _logger.setChangeLogLevelCb(boost::bind(&StepperMotorPiezo::setLogLevel, this->_stepperMotor, _1));
95  _logger.init();
96  _logger.setPreamble(this->_moduleName);
97 
98  _stepperMotorConfiurationError->setValue(1);
99  _initialized = false;
100  _isInitialized->setValue(0);
101 
102  _initialized = this->_stepperMotor->getInitialized();
103 
104  if(_initialized) {
105  _stepperMotorConfiurationError->setValue(0);
106  if(_csControlAllowed) {
107  this->enableCb();
108  this->autostartCb();
109  this->enableSoftwareLimitsCb();
110 
111  this->setPositionMaxCb();
112  this->setPositionMinCb();
113  this->setSpeedCb();
114  }
115  // std::cout << "CsaStepperMotorPiezo::init . Current value of enable flag: " << _enable->getValue() << " cs system
116  // allowed: " << _csControlAllowed << " !" << std::endl; _stepperMotorId->setValue(channelId);
117  _stepperMotorId->setValue(0);
118  _isInitialized->setValue(1);
119  }
120 
121  _maxSpeed->setValue(this->_stepperMotor->getMaxSpeedCapability());
122 
123  this->_endSwitchesAvailable->setValue(1);
124  this->_speedControlAvailable->setValue(1);
125  this->_motorTypeId->setValue(_motorType);
126  this->setCsControlAllowed(_csControlAllowed); // just to set current allowance in Process variable
127 
128  _pztChannelNo->setValue(this->_stepperMotor->getPiezoDacChannelId());
129 }
