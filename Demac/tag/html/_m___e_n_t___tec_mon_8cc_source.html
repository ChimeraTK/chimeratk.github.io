120 void M_ENT_TecMon::commandCb() {

122 
123  switch(cmd) {
124  case CMD_CREATE_READ_VECTOR_FOR_GROUP_1: {
125  createReadVector(1);
126  break;
127  }
128  case CMD_CREATE_READ_VECTOR_FOR_GROUP_2: {
129  createReadVector(2);
130  break;
131  }
132  case CMD_CREATE_READ_VECTOR_FOR_GROUP_3: {
133  createReadVector(3);
134  break;
135  }
136  case CMD_CREATE_READ_VECTOR_FOR_GROUP_4: {
137  createReadVector(4);
138  break;
139  }
140  case CMD_CREATE_READ_VECTOR_FOR_GROUP_5: {
141  createReadVector(5);
142  break;
143  }
144 
145  case CMD_APPLY_CHANGES_OF_PERIODIC_VECTOR: {
146  int32_t creationMask = _enablePeriodicReadreadback->getValue();
147  if(creationMask != _periodicReadreadbackIsEnabled->getValue()) {
148  createPeriodicVector(creationMask);
149  _periodicReadreadbackIsEnabled->setValue(creationMask);
150  }
151  break;
152  }
153  default:
154  _logger(Loger::ERROR) << __func__ << ": "
155  << "Command unknown. Command ID: " << cmd << std::endl;
156  }
