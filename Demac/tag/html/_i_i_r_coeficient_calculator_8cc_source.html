57IIRCoeficientCalculator::IIRParams IIRCoeficientCalculator::lowPassFilter(
58 double stopFreqInKHz, double samplingFreqInKHz) {
59 double c;
60 double h;
61 double fs = samplingFreqInKHz * 1000;
62 double stop = stopFreqInKHz * 1000;
63
64 IIRParams lowPassFilterParams;
65 lowPassFilterParams.valid = true;
66 lowPassFilterParams.samplingFreq = fs;
67 lowPassFilterParams.stopFreq = stop;
68 lowPassFilterParams.stopWidth = 0;
69
70 if(!validateParams(1, samplingFreqInKHz, stopFreqInKHz)) {
71 lowPassFilterParams.valid = false;
72 return lowPassFilterParams;
73 }
74
75 c = 1 / std::tan(M_PI * stop / fs);
76 h = c * c + std::sqrt(2.0) * c + 1;
77 lowPassFilterParams.b0 = 1 / h;
78 lowPassFilterParams.b1 = 2 / h;
79 lowPassFilterParams.b2 = lowPassFilterParams.b0;
80 lowPassFilterParams.a1 = -2 * (c * c - 1) / h;
81 lowPassFilterParams.a2 = (c * c - std::sqrt(2.0) * c + 1) / h;
82
83 return lowPassFilterParams;

86IIRCoeficientCalculator::IIRParams IIRCoeficientCalculator::notchFilter(
87 double stopFreqInKHz, double widthInKHz, double samplingFreqInKHz) {

89 double c;

91 double fs = samplingFreqInKHz * 1000;
92 double stop = stopFreqInKHz * 1000;

96 notchFilterParams.valid = true;
97 notchFilterParams.samplingFreq = fs;
98 notchFilterParams.stopFreq = stop;
99 notchFilterParams.stopWidth = width;
100
101 if(!validateParams(2, samplingFreqInKHz, stopFreqInKHz, widthInKHz)) {
102 notchFilterParams.valid = false;
103 return notchFilterParams;
104 }
105
106 q = stop / width / 2;
107 c = 1 / std::tan(M_PI * stop / fs);
108 h = c * c * q + c + q;
109 notchFilterParams.b0 = q * (c * c + 1) / h;
110 notchFilterParams.b1 = -2 * q * (c * c - 1) / h;
111 notchFilterParams.b2 = q * (c * c + 1) / h;
112 notchFilterParams.a1 = (-2 * q * (c * c - 1)) / h;
113 notchFilterParams.a2 = (c * c * q - c + q) / h;
114
115 return notchFilterParams;
