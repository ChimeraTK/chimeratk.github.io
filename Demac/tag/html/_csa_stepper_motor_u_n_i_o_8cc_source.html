88void CsaStepperMotorUNIO::initialize(
89 boost::shared_ptr<M_ENT_UNIODataPort> port, boost::shared_ptr<M_ENT_UNIOUartChannel> readbackChannel) {
90 _logger.setChangeLogLevelCb(boost::bind(&StepperMotorUNIO::setLogLevel, this->_stepperMotor, _1));
91 _logger.init();
92 //_logger.setPreamble("ActUNIO::" + this->_moduleName);
93 //_logger.setPreamble("" + this->_moduleName);
94
95 _stepperMotorConfiurationError->setValue(1);
96 _initialized = false;
97 _isInitialized->setValue(0);
98
99 this->_stepperMotorUnitsConverter->settingChangedReset();
100 this->_stepperMotor->initialize(port, readbackChannel, this->_stepperMotorUnitsConverter);
101 _initialized = this->_stepperMotor->getInitialized();
102
103 if(_initialized) {
104 //_stepperMotor->getFirstReadbackAvailable();
105
106 _stepperMotorConfiurationError->setValue(0);
107 if(_csControlAllowed) {
108 this->enableCb();
109 this->autostartCb();
110 this->enableSoftwareLimitsCb();
111
112 this->setPositionMaxCb();
113 this->setPositionMinCb();
114
115 this->parameterCb();
116 }
117
118 if(_loadSettingFromPropertiesOff->getValue() == 0) {
119 _logger(Loger::INFO) << __func__ << ": "
120 << "Load initial parameters from PV." << std::endl;
121 _stepperMotor->setCurrentPositionAsInSteps(_positionInSteps->getValue());
122 _stepperMotor->setCurrentTargetPositionAsInSteps(_positionInSteps->getValue());
123 }
124
125 _stepperMotorId->setValue(0);
126 _isInitialized->setValue(1);
127 }
128
129 this->_endSwitchesAvailable->setValue(0);
130 this->_speedControlAvailable->setValue(0);
131 this->_motorTypeId->setValue(_motorType);
132 this->setCsControlAllowed(_csControlAllowed); // just to set current allowance in Process variable
133}
