478  _logger(Loger::WARNING) << __func__ << ": " << "Hardware NOT OK flag set for channel " << _id

736  float currentVoltage = _adc->getVoltage(_id, true);

766 int32_t M_ENT_PZT4Channel::calcCorrectedDacValue(

769  _logger(Loger::INFO) << __func__ << ": " << "Range mode changed - old: " << (int32_t)oldRange << ", new "

777  calcCoefficientForRange(oldRange, aParForOld, bParForOld);
778 
779  float aParForNew = 0;
780  float bParForNew = 0;
781  calcCoefficientForRange(newRange, aParForNew, bParForNew);
782 
783  float currentVoltage = aParForOld * currentDacValue + bParForOld;
784  int32_t correctedDacValue = 1.0 * (currentVoltage - bParForNew) / aParForNew;
785 
786  _logger(Loger::INFO) << __func__ << ": " << "Range mode changed - old: " << (int32_t)oldRange << ", new "
787  << (int32_t)newRange << ". DAC counts: " << currentDacValue
788  << " . Voltage calculated from DAC counts: " << currentVoltage
789  << ". DAC counts which corresponds to the same value in new mode: " << correctedDacValue << " ."
790  << " Real ADC readout: " << _adc->getVoltage(_id, true) << " V)." << std::endl;
791 
792  if(correctedDacValue > 131071) {
793  _logger(Loger::ERROR) << __func__ << ": " << "Corrected value (" << correctedDacValue
794  << ") greater than 131071. Truncate to 131071." << std::endl;
795  correctedDacValue = 131071;
796  }
797  else if(correctedDacValue < -131072) {
798  _logger(Loger::ERROR) << __func__ << ": " << "Corrected value (" << correctedDacValue
799  << ") smaller than -131072. Truncate to -131072." << std::endl;
800  correctedDacValue = -131072;
801  }
802 
803  return correctedDacValue;

806 void M_ENT_PZT4Channel::dacRamping(int32_t currentDac, int32_t aimDac) {
807  if(_dacResetEnabled) {
808  _logger(Loger::WARNING) << __func__ << ": " << "DAC logic reset enabled. No actions possible returning."
809  << std::endl;
810  return;
811  }
812 
813  _logger(Loger::INFO) << __func__ << ": " << "Voltage before ramping is " << _adc->getVoltage(_id, true) << " V."
814  << std::endl;
815 
816  int32_t curDacValue = currentDac;
817  int32_t dacValueAim = aimDac;
818 
819  if(abs(dacValueAim - curDacValue) > 50) {
820  _logger(Loger::INFO) << __func__ << ": " << "Start DAC value: " << curDacValue
821  << ". Aim of ramping: " << dacValueAim
822  << ". Current distance: " << abs(dacValueAim - curDacValue) << std::endl;
823 
824  auto currentOperationMode = _operationMode->getValueFromDevice();
825  auto currentConstantValue = _constantValue->getValueFromDevice();
826 
827  // start ramping
828  // set this value to constant_value (ff mode)
829  _constantValue->setValue(curDacValue);
830  // change mode of operation to CONSTANT
831  _logger(Loger::INFO) << __func__ << ": " << "Enable OP_MODE::CONSTANT_M (Feedforward mode)." << std::endl;
832  _operationMode->setValue((uint32_t)M_ENT_PZT4Channel::OP_MODE::CONSTANT_M);
833  // activate the channel
834  _logger(Loger::INFO) << __func__ << ": " << "Enabling internal DAC and activating the channel." << std::endl;
835  _internalDacEnabled->setValue(1);
836  _activate->setValue(1);
837 
838  _logger(Loger::INFO) << __func__ << ": " << "Active status: " << _activate->getValueFromDevice()
839  << " . Internal DAC enable status: " << _internalDacEnabled->getValueFromDevice()
840  << " . Mode of operation: " << _operationMode->getValueFromDevice() << std::endl;
841 
842  while(abs(dacValueAim - curDacValue) > 50) {
843  int step = 600;
844  if(abs(dacValueAim - curDacValue) < 250)
845  step = 40;
846  else if(abs(dacValueAim - curDacValue) < 1000)
847  step = 150;
848  else if(abs(dacValueAim - curDacValue) < 2000)
849  step = 300;
850 
851  if((curDacValue - dacValueAim) > 0)
852  curDacValue = curDacValue - step;
853  else
854  curDacValue = curDacValue + step;
855 
856  _constantValue->setValue(curDacValue);
857  std::this_thread::sleep_for(std::chrono::microseconds(500));
858  _logger(Loger::DETAIL) << __func__ << ": " << "Current DAC (FF mode): " << _constantValue->getValueFromDevice()
859  << ". Distance: " << abs(dacValueAim - curDacValue) << " steps. Step: " << step
860  << ". Current DAC : " << (((int)_currentValue->getValueFromDevice()) - 131072)
861  << " ADC: " << _adc->getVoltage(_id, true) << " V." << std::endl;
862  }
863 
864  _logger(Loger::INFO) << __func__ << ": "
865  << "Value after ramping: " << ((int32_t)_currentValue->getValueFromDevice() - 131072)
866  << std::endl;
867  _logger(Loger::INFO) << __func__ << ": " << "Deactivating the channel." << std::endl;
868  _activate->setValue(0);
869 
870  _operationMode->setValue(currentOperationMode);
871  _constantValue->setValue(currentConstantValue);
872  std::this_thread::sleep_for(std::chrono::milliseconds(20));
873  }
874  else
875  _logger(Loger::INFO) << __func__ << ": " << "Current DAC value: " << curDacValue << " Aim is: " << dacValueAim
876  << ". Ramping not needed." << std::endl;
877 
878  _logger(Loger::INFO) << __func__ << ": " << "Voltage after ramping is " << _adc->getVoltage(_id, true) << " V."
879  << std::endl;
880 }
