469void M_ENT_PZT4Channel::enableInternal(bool enableFlag, int32_t rampingAim) {
470 auto dacBypassModeLocal = _dacBypassModeEnable;
471
472 if(_dacResetEnabled && !dacBypassModeLocal) {
473 _logger(Loger::WARNING) << __func__ << ": " << "DAC logic reset enabled. No actions possible returning."
474 << std::endl;
475 return;
476 }
477 if(!_pzt4HardwareOK && enableFlag) {
478 _logger(Loger::WARNING) << __func__ << ": " << "Hardware NOT OK flag set for channel " << _id
479 << ". Cannot enable. Return." << std::endl;
480 _enable->setValue(0, false);
481 _enable->synchronize();
482 return;
483 }
484
485 if(_enabledStatus && enableFlag) {
486 _logger(Loger::INFO) << __func__ << ": " << "Channel is already enabled." << std::endl;
487 return;
488 }
489
490 // enable channel routine
491 if(enableFlag) {
492 auto vol = _adc->getVoltage(_id, true);
493 if(std::fabs(vol) > 1.0) {
494 _logger(Loger::ERROR) << __func__ << ": "
495 << "Voltage on channel is greater than 1 V or smaller than -1 V. Current value: " << vol
496 << " V. Channel should be disabled, but is not. Disabling before enabling." << std::endl;
497 enableInternal(false);
498 }
499 else
500 _logger(Loger::INFO) << __func__ << ": " << "Voltage on channel is " << vol << " V." << std::endl;
501
502 _logger(Loger::INFO) << __func__ << ": " << "Channel will be enabled." << std::endl;
503 _pzt4BspOwner->channelIsEnabling(_id); // inform PZT4 BSP
504
505 _logger(Loger::INFO) << __func__ << ": " << "Current DAC count: " << (_currentValue->getValueFromDevice() - 131072)
506 << std::endl;
507
508 setInhibitWithAdcCheck(true);
509
510 _activate->setValue(0); // before applying dac range channel needs to be deactivated
511
512 M_ENT_PZT4Channel::DAC_RANGE oldDacRange = this->getDacRange();
513 // first set range of DAC
514 if(!_hardwareSpanEnabled) {
515 M_ENT_PZT4Channel::DAC_RANGE range = (M_ENT_PZT4Channel::DAC_RANGE)_dacRange->getValue();
516 _logger(Loger::INFO) << __func__ << ": " << "Setting new DAC range (span) to: " << _dacRange->getValue()
517 << " . Current dac range is: " << oldDacRange << std::endl;
518
519 // in case when DAC BYPASS is ON, then we need to use unipolar mode
520 if(dacBypassModeLocal && range > (int)M_ENT_PZT4Channel::DAC_RANGE::RNG_0_TO_P10) {
521 _logger(Loger::INFO) << __func__ << ": " << "DAC BYPASS mode enabled. Change DAC range to unipolar - 0 V to 5 V"
522 << std::endl;
523
524 _dacRange->setValue((int)M_ENT_PZT4Channel::DAC_RANGE::RNG_0_TO_P5, false);
525 _dacRange->synchronize();
526 range = M_ENT_PZT4Channel::DAC_RANGE::RNG_0_TO_P5;
527 }
528
529 if(range > (int)M_ENT_PZT4Channel::DAC_RANGE::RNG_N2_5_TO_P7_5) {
530 _logger(Loger::WARNING) << __func__ << ": " << "Wrong DAC range. Value " << (int)range
531 << " is not allowed. Select save unipolar mode 0 - 5V." << std::endl;
532 range = M_ENT_PZT4Channel::DAC_RANGE::RNG_0_TO_P5;
533 }
534 setDacRangeInternal(range);
535 }
536 else {
537 _logger(Loger::INFO) << __func__ << ": "
538 << "Hardware DAC range (span) in use. Selected hardware span is: " << _hardwareSpanMode
539 << std::endl;
540
541 if(dacBypassModeLocal && _hardwareSpanMode > (int)M_ENT_PZT4Channel::DAC_RANGE::RNG_0_TO_P10) {
542 _logger(Loger::ERROR)
543 << __func__ << ": "
544 << "DAC BYPASS mode enabled. Hardware SPAN enabled and set to BIPOLAR. Channel cannot be activated."
545 << std::endl;
546 return;
547 }
548 }
549
550 M_ENT_PZT4Channel::DAC_RANGE newDacRange = this->getDacRange();
551 _logger(Loger::INFO) << __func__ << ": " << "Applied DAC range (span) is " << newDacRange << std::endl;
552
553 // value which is present of DAC needs to be recalculated in case of bipolar/unipolar change. Recalculation is
554 // needed as for new mode, the voltage jump is created
555 int32_t curDacValue = _currentValue->getValueFromDevice() - 131072;
556 if(newDacRange != oldDacRange) {
557 auto dacValueAim = calcCorrectedDacValue(oldDacRange, newDacRange, curDacValue);
558 _logger(Loger::INFO)
559 << __func__ << ": "
560 << "Ramp DAC with open relays due to DAC range (span) changed to different mode. Internal voltage "
561 "jump possible, but not present on physical PZT4 output."
562 << std::endl;
563 dacRamping(curDacValue, dacValueAim);
564 curDacValue = dacValueAim;
565 }
566 // set relay to wanted state
567 _logger(Loger::INFO) << __func__ << ": "
568 << "Apply relay settings. Relay set in position: " << (_relayControlSoftware->getValue() & 0x1)
569 << std::endl;
570 _relayControl->setValue((_relayControlSoftware->getValue() & 0x1), false);
571
572 // if inhibit present - disable the inhibit
573 setInhibitWithAdcCheck(false);
574
575 // check if ramping in needed
576 // for all modes we want to go to virtual 0 so this is always middle of the range so: 0 for _constantValue
577 int32_t dacValueAim = rampingAim;
578 _logger(Loger::INFO) << __func__ << ": " << "Ramping aim DAC value: " << dacValueAim << std::endl;
579 if(((M_ENT_PZT4Channel::OP_MODE)_operationMode->getValueFromDevice()) == M_ENT_PZT4Channel::OP_MODE::LLL_M) {
580 dacValueAim = _lllDataMonitor->getValueFromDevice();
581 _logger(Loger::INFO) << __func__ << ": " << "Operation mode is LLL. Aim value changed to: " << dacValueAim
582 << std::endl;
583 }
584 else if(((M_ENT_PZT4Channel::OP_MODE)_operationMode->getValueFromDevice()) ==
585 M_ENT_PZT4Channel::OP_MODE::CONSTANT_M) {
586 dacValueAim = _constantValue->getValueFromDevice();
587 _logger(Loger::INFO) << __func__ << ": " << "Operation mode is CONSTANT. Aim value changed to: " << dacValueAim
588 << std::endl;
589 }
590
591 _logger(Loger::INFO) << __func__ << ": " << "Current DAC value: " << curDacValue << ". Aim is: " << dacValueAim
592 << " . Start ramping." << std::endl;
593 dacRamping(curDacValue, dacValueAim);
594
595 _logger(Loger::INFO) << __func__ << ": " << "Setting DAC as channel source and activating the channel."
596 << std::endl;
597
598 // in DAC BYPASS mode, we use unipolar mode so one needs to enable external voltage
599 if(dacBypassModeLocal) {
600 _logger(Loger::INFO)
601 << __func__ << ": "
602 << "DAC BYPASS mode enabled. Enable external voltage reference, PGA gain to 1 V/V and disable internal DAC."
603 << std::endl;
604 _externalReferenceVoltageEnabled->setValue(1, false);
605 _externalReferenceVoltageEnabled->synchronize();
606 _pgaGainMode->setValue(0, false);
607 _pgaGainMode->synchronize();
608 // give time for inhibit relay to physically switch - c.a 15 ms
609 std::this_thread::sleep_for(std::chrono::milliseconds(15));
610 _internalDacEnabled->setValue(0);
611 }
612 else {
613 _internalDacEnabled->setValue(1);
614 }
615
616 _activate->setValue(1);
617
618 _logger(Loger::INFO) << __func__ << ": " << "Channel is enabled now." << std::endl;
619 _enabledStatus = 1;
620 }
621
622 // disable channel routine
623 if(!enableFlag) {
624 if(!_enabledStatus) {
625 _logger(Loger::INFO) << __func__ << ": "
626 << "Channel seems to be disabled, but disabling process will be repeated." << std::endl;
627 }
628 else {
629 _logger(Loger::INFO) << __func__ << ": " << "Channel will be disabled." << std::endl;
630 }
631
632 _logger(Loger::INFO) << __func__ << ": " << "Deactivating the channel." << std::endl;
633 _activate->setValue(0);
634 // value which is present of DAC
635 int32_t curDacValue = _currentValue->getValueFromDevice() - 131072;
636
637 if(dacBypassModeLocal) {
638 _logger(Loger::INFO) << __func__ << ": "
639 << "DAC BYPASS mode enabled. Calcute DAC counts corresponding to voltage present on output."
640 << std::endl;
641 curDacValue = calculateDacCountFromAdc();
642 }
643
644 // get current DAC range
645 M_ENT_PZT4Channel::DAC_RANGE dacRange = this->getDacRange();
646 int32_t dacValueAim = 0;
647 // for bipolar range we need to ramp to curDacValue = 0 and for unipolar mode we need to ramp to curDacValue = -131072
648 if(dacRange == M_ENT_PZT4Channel::DAC_RANGE::RNG_0_TO_P5 ||
649 dacRange == M_ENT_PZT4Channel::DAC_RANGE::RNG_0_TO_P10) {
650 dacValueAim = -131072;
651 _logger(Loger::INFO) << __func__ << ": " << "Current DAC range is unipolar. Range: " << ((int32_t)dacRange)
652 << std::endl;
653 }
654 else {
655 _logger(Loger::INFO) << __func__ << ": " << "Current DAC range is bipolar. Range: " << ((int32_t)dacRange)
656 << std::endl;
657 }
658
659 if(dacRange == M_ENT_PZT4Channel::DAC_RANGE::RNG_UKNOWN) {
660 _logger(Loger::ERROR) << __func__ << ": " << "Current DAC range is unknown. Cannot ramp correctly." << std::endl;
661 }
662 else {
663 _logger(Loger::INFO) << __func__ << ": " << "Current DAC value: " << curDacValue << " . Aim is: " << dacValueAim
664 << std::endl;
665 dacRamping(curDacValue, dacValueAim);
666 }
667
668 // if inhibit present - enable the inhibit
669 setInhibitWithAdcCheck(true);
670
671 if(dacBypassModeLocal) {
672 _logger(Loger::INFO) << __func__ << ": " << "DAC BYPASS mode enabled. Disable external voltage reference."
673 << std::endl;
674 _externalReferenceVoltageEnabled->setValue(0, false);
675 _externalReferenceVoltageEnabled->synchronize();
676 }
677
678 _logger(Loger::INFO) << __func__ << ": " << "Channel is disabled now." << std::endl;
679 _enabledStatus = 0;
680 }
681
682 _enable->setValue(_enabledStatus, false);
683 _enable->synchronize();
684}

735int32_t M_ENT_PZT4Channel::calculateDacCountFromAdc() {
736 float currentVoltage = _adc->getVoltage(_id, true);
737 float aPar = 0;
738 float bPar = 0;
739 auto range = this->getDacRange();
740 // the coefficent has been measerured in 26A
741 if(range == RNG_0_TO_P5) {
742 calcCoefficient(22.4, -60000, 60.1, 60000, aPar, bPar);
743 }
744 else if(range == RNG_0_TO_P10) {
745 calcCoefficient(44.7, -60000, 19.6, -100000, aPar, bPar);
746 }
747 else if(range == RNG_N5_TO_P5) {
748 calcCoefficient(-37.5, -60000, 37.8, 60000, aPar, bPar);

750 else if(range == RNG_N10_TO_P10) {
751 calcCoefficient(-37.5, -30000, 37.8, 30000, aPar, bPar);

754 calcCoefficient(-9.3, -30000, 9.5, 30000, aPar, bPar);

756
757 int32_t calcDacCntValue = 1.0 * (currentVoltage - bPar) / aPar;
758 _logger(Loger::DETAIL) << __func__ << ": " << "Range " << range << ". a: " << aPar << " . b: " << bPar << std::endl;
759
760 _logger(Loger::INFO) << __func__ << ": " << "Current voltage is " << currentVoltage
761 << " V. Corresponding DAC count: " << calcDacCntValue << " . Range mode: " << range << std::endl;
762
763 return calcDacCntValue;

766int32_t M_ENT_PZT4Channel::calcCorrectedDacValue(
767 M_ENT_PZT4Channel::DAC_RANGE oldRange, M_ENT_PZT4Channel::DAC_RANGE newRange, int32_t currentDacValue) {
768 if(oldRange == newRange) {
769 _logger(Loger::INFO) << __func__ << ": " << "Range mode changed - old: " << (int32_t)oldRange << ", new "
770 << (int32_t)newRange << " are the same. Return input DAC counts: " << currentDacValue
771 << std::endl;
772 return currentDacValue;
773 }
774
775 float aParForOld = 0;
776 float bParForOld = 0;
777 calcCoefficientForRange(oldRange, aParForOld, bParForOld);
778
779 float aParForNew = 0;
780 float bParForNew = 0;
781 calcCoefficientForRange(newRange, aParForNew, bParForNew);
782
783 float currentVoltage = aParForOld * currentDacValue + bParForOld;
784 int32_t correctedDacValue = 1.0 * (currentVoltage - bParForNew) / aParForNew;
785
786 _logger(Loger::INFO) << __func__ << ": " << "Range mode changed - old: " << (int32_t)oldRange << ", new "
787 << (int32_t)newRange << ". DAC counts: " << currentDacValue
788 << " . Voltage calculated from DAC counts: " << currentVoltage
789 << ". DAC counts which corresponds to the same value in new mode: " << correctedDacValue << " ."
790 << " Real ADC readout: " << _adc->getVoltage(_id, true) << " V)." << std::endl;
791
792 if(correctedDacValue > 131071) {
793 _logger(Loger::ERROR) << __func__ << ": " << "Corrected value (" << correctedDacValue
794 << ") greater than 131071. Truncate to 131071." << std::endl;
795 correctedDacValue = 131071;
796 }
797 else if(correctedDacValue < -131072) {
798 _logger(Loger::ERROR) << __func__ << ": " << "Corrected value (" << correctedDacValue
799 << ") smaller than -131072. Truncate to -131072." << std::endl;
800 correctedDacValue = -131072;
801 }
802
803 return correctedDacValue;
804}

806void M_ENT_PZT4Channel::dacRamping(int32_t currentDac, int32_t aimDac) {
807 if(_dacResetEnabled) {
808 _logger(Loger::WARNING) << __func__ << ": " << "DAC logic reset enabled. No actions possible returning."
809 << std::endl;
810 return;
811 }
812
813 _logger(Loger::INFO) << __func__ << ": " << "Voltage before ramping is " << _adc->getVoltage(_id, true) << " V."
814 << std::endl;
815
816 int32_t curDacValue = currentDac;
817 int32_t dacValueAim = aimDac;
818
819 if(abs(dacValueAim - curDacValue) > 50) {
820 _logger(Loger::INFO) << __func__ << ": " << "Start DAC value: " << curDacValue
821 << ". Aim of ramping: " << dacValueAim
822 << ". Current distance: " << abs(dacValueAim - curDacValue) << std::endl;
823
824 auto currentOperationMode = _operationMode->getValueFromDevice();
825 auto currentConstantValue = _constantValue->getValueFromDevice();
826
827 // start ramping
828 // set this value to constant_value (ff mode)
829 _constantValue->setValue(curDacValue);
830 // change mode of operation to CONSTANT
831 _logger(Loger::INFO) << __func__ << ": " << "Enable OP_MODE::CONSTANT_M (Feedforward mode)." << std::endl;
832 _operationMode->setValue((uint32_t)M_ENT_PZT4Channel::OP_MODE::CONSTANT_M);
833 // activate the channel
834 _logger(Loger::INFO) << __func__ << ": " << "Enabling internal DAC and activating the channel." << std::endl;
835 _internalDacEnabled->setValue(1);
836 _activate->setValue(1);
837
838 _logger(Loger::INFO) << __func__ << ": " << "Active status: " << _activate->getValueFromDevice()
839 << " . Internal DAC enable status: " << _internalDacEnabled->getValueFromDevice()
840 << " . Mode of operation: " << _operationMode->getValueFromDevice() << std::endl;
841
842 while(abs(dacValueAim - curDacValue) > 50) {
843 int step = 600;
844 if(abs(dacValueAim - curDacValue) < 250)
845 step = 40;
846 else if(abs(dacValueAim - curDacValue) < 1000)
847 step = 150;
848 else if(abs(dacValueAim - curDacValue) < 2000)
849 step = 300;
850
851 if((curDacValue - dacValueAim) > 0)
852 curDacValue = curDacValue - step;
853 else
854 curDacValue = curDacValue + step;
855
856 _constantValue->setValue(curDacValue);
857 std::this_thread::sleep_for(std::chrono::microseconds(500));
858 _logger(Loger::DETAIL) << __func__ << ": " << "Current DAC (FF mode): " << _constantValue->getValueFromDevice()
859 << ". Distance: " << abs(dacValueAim - curDacValue) << " steps. Step: " << step
860 << ". Current DAC : " << (((int)_currentValue->getValueFromDevice()) - 131072)
861 << " ADC: " << _adc->getVoltage(_id, true) << " V." << std::endl;
862 }
863
864 _logger(Loger::INFO) << __func__ << ": "
865 << "Value after ramping: " << ((int32_t)_currentValue->getValueFromDevice() - 131072)
866 << std::endl;
867 _logger(Loger::INFO) << __func__ << ": " << "Deactivating the channel." << std::endl;
868 _activate->setValue(0);
869
870 _operationMode->setValue(currentOperationMode);
871 _constantValue->setValue(currentConstantValue);
872 std::this_thread::sleep_for(std::chrono::milliseconds(20));
873 }
874 else
875 _logger(Loger::INFO) << __func__ << ": " << "Current DAC value: " << curDacValue << " Aim is: " << dacValueAim
876 << ". Ramping not needed." << std::endl;
877
878 _logger(Loger::INFO) << __func__ << ": " << "Voltage after ramping is " << _adc->getVoltage(_id, true) << " V."
879 << std::endl;
880}
