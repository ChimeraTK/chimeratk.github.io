75 void CsaCoarseTuningSingleDetectorAlgorithm::triggerCommand(int cmd, bool printLog) {
76  if(printLog) _logger(Loger::DETAIL) << __func__ << ": " << _logIdString << ": Command id: " << cmd << std::endl;
77 
78  bool tuneInProgress = _tuneAlg.getTuningInProgres();
79  switch(cmd) {
80  case CsaCoarseTuningSingleDetectorAlgorithm::CMD::START_TUNE: {
81  if(_stepInProgress == false && tuneInProgress == false && _tuneAlg.getTuningPaused() == false) {
82  _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Start tuning." << std::endl;
83  _tuneAlg.triggerTuning();
84  }
85  else {
86  _logger(Loger::WARNING) << __func__ << ": " << _logIdString
87  << ": Tuning or step in progress. Tune command ignored." << std::endl;
88  }
89  break;
90  }
91  case CsaCoarseTuningSingleDetectorAlgorithm::CMD::STOP_TUNE: {
92  _tuneAlg.stopTuning();
93  break;
94  }
95 
96  case CsaCoarseTuningSingleDetectorAlgorithm::CMD::STEP_UP:
97  case CsaCoarseTuningSingleDetectorAlgorithm::CMD::STEP_DOWN: {
98  if(_stepInProgress == false && tuneInProgress == false) {
99  if(cmd == CsaCoarseTuningSingleDetectorAlgorithm::CMD::STEP_UP)
100  _stepRequest = 1;
101  else
102  _stepRequest = -1;
103  }
104  else if(_stepInProgress) {
105  _logger(Loger::WARNING) << __func__ << ": " << _logIdString
106  << ": Actuator step in progress. Step command ignored." << std::endl;
107  }
108  else {
109  _logger(Loger::WARNING) << __func__ << ": " << _logIdString
110  << ": Tuning step in progress. Step command ignored." << std::endl;
111  }
112  break;
113  }
114  case CsaCoarseTuningSingleDetectorAlgorithm::CMD::OPEN_FAST_LOOP:
115  this->_ctrlManager->setEnable(false);
116  _messageString.set("Lock was disable by user.");
117  break;
118 
119  case CsaCoarseTuningSingleDetectorAlgorithm::CMD::CLOSE_FAST_LOOP:
120  this->_ctrlManager->setEnable(true);
121  _messageString.set("Lock was enabled by user.");
122  break;
123 
124  default:
125  break;
126  }
127 }

160 void CsaCoarseTuningSingleDetectorAlgorithm::update() {
161  if(!_controller) {
162  _logger() << __func__ << ": " << _logIdString << ": _controller not set. exit(1)." << std::endl;
163  exit(1);
164  return;
165  }
166 
167  if(!_actuator) {
168  _logger() << __func__ << ": " << _logIdString << ": _actuator not set. exit(1)." << std::endl;
169  exit(1);
170  return;
171  }
172 
173  _tuneAlg.setStepSize(_stepSize->getValue());
174 
175  if(_stepInProgress) {
176  // std::cout << "Steps is in progress" << std::endl;
177  if(!_actuator->motorIsBusy()) {
178  _stepInProgress = false;
179  }
180  }
181 
182  auto ctrlEnabled = _controller->isActive();
183  bool tuneInProgress = false;
184  if(ctrlEnabled) {
185  _tuneAlg.checkTuning();
186  tuneInProgress = _tuneAlg.getTuningInProgres();
187  }
188  _prevCtrlEnabled = ctrlEnabled;
189 
190  if(!tuneInProgress && !_stepInProgress) {
191  if(_stepRequest) {
192  step(_stepRequest);
193  _stepRequest = 0;
194  }
195  _algorithmActive->setValue(0);
196  }
197  else {
198  _stepRequest = 0;
199  _algorithmActive->setValue(1);
200  }
201 
202  if(_updateDividerCnt == 10) {
203  _updateDividerCnt = 0;
204  if(!_resetBlockedActuatorThreadActive) {
205  std::thread t1(&CsaCoarseTuningSingleDetectorAlgorithm::resetBlockedActuatorThread, this);
206  t1.detach();
207  }
208  }
209  _updateDividerCnt++;
210 }

223 void CsaCoarseTuningSingleDetectorAlgorithm::step(int32_t direction) {
224  _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Start step in direction: " << direction << std::endl;
225 
226  _actuator->getStatus();
227 
228  if(!_actuator->motorIsBusy()) {
229  _stepInProgress = true;
230 
231  // step is done with curernt speed settings
232  _actuatorHelper.enableActuatorWithNoAutostart(); /*0-user,1-fast,2-slow*/
233 
234  auto currentPos = _actuator->getCurrentPosition();
235 
236  if(_stepSize->getValue() == 0)
237  _logger(Loger::WARNING) << __func__ << ": " << _logIdString << ": Step size is 0. Check settings." << std::endl;
238 
239  if(direction > 0) {
240  _actuator->setTargetPosition(currentPos + _stepSize->getValue());
241  _actuator->start();
242  }
243  else {
244  _actuator->setTargetPosition(currentPos - _stepSize->getValue());
245  _actuator->start();
246  }
247  _algorithmActive->setValue(1);
248  }
249  else {
250  _logger(Loger::WARNING) << __func__ << ": " << _logIdString << ": Motor is busy. " << direction << std::endl;
251  }
252 }

267 void CsaCoarseTuningSingleDetectorAlgorithm::coarseTuningEnabledCb() {
268  try {
269  if(_coarseTuningEnabled->getValue()) {
270  _tuneAlg.setEnable(true);
271  _actuator->setCsControlAllowed(false);
272  _tuneAlg.setTuningPaused(false);
273  if(!_actuator->isEnabled()) {
274  _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Motor is not enabled. Enabling." << std::endl;
275  _actuator->enable(true);
276  }
277 
278  if(_actuator->getAutostart()) {
279  _logger(Loger::INFO) << __func__ << ": " << _logIdString << ": Autostart is enabled. Disabling." << std::endl;
280  _actuator->setAutostart(false);
281  }
282  }
283  else {
284  _tuneAlg.setEnable(false);
285  _actuator->stop();
286  _actuator->setCsControlAllowed(true);
287  }
288  }
289  catch(StepperMotorActuatorException& ex) {
290  _logger(Loger::INFO) << __func__ << ": " << _logIdString
291  << ": StepperMotorActuatorException detected. Tuning enabling failed. Error is : " << ex.what()
292  << std::endl;
293  _tuneAlg.setEnable(false);
294  _actuator->setCsControlAllowed(true);
295  _coarseTuningEnabled->setValue(0, false);
296  _coarseTuningEnabled->synchronize();
297  }
