126  // std::cout << "CsaMultiBitFieldRegisterAccesor::init: Object: " << this->_nameInControlSystem << " (REGISTER): "

128  return;
129  }

132  CsaAdapter::getInstance().globalData.getLogger()(Loger::ERROR)
133  << __func__ << ": " << "Object: " << this->_nameInControlSystem
134  << ". Register name is empty during initialization. Error. Returning." << std::endl;
135  return;
136  }
137 
138  this->_device = device;
139 
140  if(this->_device) {
141  this->checkRegisterRename();
142 
143  boost::shared_ptr<ChimeraTK::Device> device =
144  this->_device; // this is because of compiler bug -> compiler cannot correctly resolve 'this->_device'
145  // this->_currentValueRegister =
146  // device->getBufferingRegisterAccessor<DataType>(this->_mapModuleName+"/"+this->_registerName,1,this->_registerOffset);
147  // this->_currentValueRegister = device->getScalarRegisterAccessor<DataType>(this->_mapModuleName + "/" +

151  ChimeraTK::RegisterPath(this->_mapModuleName) / ChimeraTK::RegisterPath(this->_registerName);
152  this->_currentValueRegister.replace(
153  device->getScalarRegisterAccessor<DataType>(realRegisterName, this->_registerOffset));
154 
155  this->_deviceInitDone = true;
156 
157  auto regInfo = device->getRegisterCatalogue().getRegister(realRegisterName);
158  auto dataDes = regInfo.getDataDescriptor();
159  if(dataDes.nFractionalDigits() != 0) {
160  CsaAdapter::getInstance().globalData.getLogger()(Loger::ERROR)
161  << __func__ << ": " << "Register: " << this->_registerName << " defined in a device as double. "
162  << this->_nameInControlSystem << " is integer. Types mismatch!" << std::endl;
163  this->_deviceInitDone = false;
164  }
165 
166  // boost::shared_ptr<ChimeraTK::RegisterInfoMap::RegisterInfo>
167  auto oldStyleRegisterInfo = dynamic_cast<const ChimeraTK::NumericAddressedRegisterInfo*>(&regInfo.getImpl());
168  if(oldStyleRegisterInfo) {
169  if(_fieldOffset > oldStyleRegisterInfo->channels[0].width || _fieldOffset > _maxIndexFromType) {
170  CsaAdapter::getInstance().globalData.getLogger()(Loger::ERROR)
171  << __func__ << ": " << "MultiBitFieldRegister: " << this->_registerName
172  << " Cs name: " << this->_nameInControlSystem
173  << " has field index out of range. Field index: " << _fieldOffset << " Reg. width: " << dataDes.nDigits()
174  << std::endl;
175  this->_deviceInitDone = false;
176  }
177 
178  if((_fieldWidth + _fieldOffset) > oldStyleRegisterInfo->channels[0].width ||
179  (_fieldWidth + _fieldOffset) > _maxIndexFromType) {
180  CsaAdapter::getInstance().globalData.getLogger()(Loger::ERROR)
181  << __func__ << ": " << "MultiBitFieldRegister: " << this->_registerName
182  << " Cs name: " << this->_nameInControlSystem
183  << " has field width out of range. Field index: " << _fieldOffset << ". Field width: " << _fieldWidth
184  << " Reg. width: " << dataDes.nDigits() << " Reg. width from type: " << _maxIndexFromType << std::endl;
185  this->_deviceInitDone = false;
186  }
187  }
188  else {
189  this->_deviceInitDone = false;
190  CsaAdapter::getInstance().globalData.getLogger()(Loger::ERROR)
191  << __func__ << ": " << "MultiBitFieldRegister: " << this->_registerName
192  << " Cs name: " << this->_nameInControlSystem
193  << ". is not bit-field value acceptable. Possible wrong backend. Number of bits not recognized." << std::endl;
194  throw D_AdvancedRegisterAccessorException("CsaMultiBitFieldRegisterAccesor::init: Exception detected. Not "
195  "correct back-end. Lack of support for bit-wise registers.");
196  }
197 
198  if(this->_deviceInitDone) {
199  _fieldMask = (~(0xFFFFFFFFF << _fieldWidth)) << _fieldOffset;
200  // std::cout << "CsaMultiBitFieldRegisterAccesor<DataType>::init: INFO: MultiBitFieldRegister: " << this->_registerName

202  // << " Field offset: " << _fieldOffset << " Reg. offset: " << this->_registerOffset <<std::endl;
203  }
204  }
205  // do one hardware readout to initialize value in the doocs property - it is after initialization with default value if enabled
206  if(this->_autoupdatePeriodMs == UPDATE_PERIOD_NOT_DEFINED && this->_deviceAccessMode == DevAccessMode::READ_WRITE) {
207  this->_autoupdatePeriodMs = UPDATE_PERIOD_FOR_READ_WRITE_REG_MS_DEFAULT;
208  }

210  this->_autoupdatePeriodMs = UPDATE_PERIOD_FOR_READ_ONLY_REG_MS_DEFAULT;

219 template<typename DataType>
220 bool CsaMultiBitFieldRegisterAccesor<DataType>::setValueInternal(DataType value, bool executeCallback) {
221  try {
222  if(this->_deviceAccessMode == DevAccessMode::READ_ONLY)
223  // std::cout << "CsaRegisterAccesor::setValue: Prop.name: " << this->_nameInControlSystem << ". Reg.name: " <<
224  // this->_registerName << " is read only register. Current hardware value: " << this->getValueFromDevice() << std::endl;
225  return false;
226 
227  if(!this->_device)
228  throw D_AdvancedRegisterAccessorException(
229  "CsaRegisterAccesor::setValueInternal: Exception detected. Device is NULL.");
230 
231  if(!this->_deviceInitDone)
232  throw D_AdvancedRegisterAccessorException(
233  "CsaRegisterAccesor::setValueInternal: Exception detected. Device not initialized.");
234 
235  if(this->_converter) value = this->_converter->convert(value);
236  // define scope for mutex guard
237  {
238  std::lock_guard<std::mutex> guard(this->_deviceAccessorMutex);
239  this->readWriteHwd(
240  false); // physical read from hardware with exception handling, replacement for: this->_currentValueRegister.read();
241  DataType _currentValueTemp = this->_currentValueRegister;
242  _currentValueTemp = _currentValueTemp & (~_fieldMask); // clear bit field
243  this->_currentValue = (value & (_fieldMask >> _fieldOffset));
244  this->_currentValueRegister = _currentValueTemp | (this->_currentValue << _fieldOffset); // set bit field
245 
246  CsaAdapter::getInstance().globalData.getLogger()(Loger::DETAIL)
247  << __func__ << ": " << "Set value: " << this->_currentValueRegister << " to register: " << this->_registerName
248  << "(field offset:" << _fieldOffset << ",mask:" << _fieldMask
249  << ") represented as: " << this->_nameInControlSystem << std::endl;
250  this->readWriteHwd(
251  true); // physical write from hardware with exception handling, replacement for: this->_currentValueRegister.write();
252  this->_lastUpdateTime = CsaAdapter::getInstance().globalData.getGlobalTimeMillisec();
253  }
254  // end of scope for mutex guard
255  if(executeCallback && this->_writeCallback) this->_writeCallback();
256 
257  return true;
258  }
259  catch(boost::numeric::negative_overflow& e) {
260  CsaAdapter::getInstance().globalData.getLogger()(Loger::ERROR)
261  << __func__ << ": " << e.what() << " exception on: " << this->_registerName
262  << ", module name: " << this->_mapModuleName << std::endl;
263  }
264  catch(boost::numeric::positive_overflow& e) {
265  CsaAdapter::getInstance().globalData.getLogger()(Loger::ERROR)
266  << __func__ << ": " << e.what() << " exception on: " << this->_registerName
267  << ", module name: " << this->_mapModuleName << std::endl;
268  }
269 
270  return false;
271 }

279 template<typename DataType>
280 DataType CsaMultiBitFieldRegisterAccesor<DataType>::getValueFromDevice() {
281  // local copy for multi-threads access. Each thread will have its own copy when calling the method
282  DataType localDataCopy = 0;
283  try {
284  if(this->_device && this->_deviceInitDone) {
285  // define scope for mutex guard
286  {
287  std::lock_guard<std::mutex> guard(this->_deviceAccessorMutex);
288  // store old value from previous read in case read triggers boost::numeric::negative_overflow or
289  // boost::numeric::positive_overflow
290  localDataCopy = this->_currentValue;
291  // physical read from hardware with exception handling, replacement for: this->_currentValueRegister.read();
292  this->readWriteHwd(false);
293  this->checkDeviceAvailability();
294  this->_currentValue = (this->_currentValueRegister & _fieldMask) >> _fieldOffset;
295  if(this->_converter) this->_currentValue = this->_converter->unconvert(this->_currentValue);
296  localDataCopy = this->_currentValue;
297  this->_lastUpdateTime = CsaAdapter::getInstance().globalData.getGlobalTimeMillisec();
298  }
299  // end of scope for mutex guard
300  // send data to Control System is object available
301  if(this->_csaDataToCs) this->_csaDataToCs->setValueNoCallback(localDataCopy);
302  }
303  else {
304  if(this->_deviceInitDone == false)
305  throw D_AdvancedRegisterAccessorException(
306  "CsaMultiBitFieldRegisterAccesor::getValueFromDevice: Exception detected. Device not initialized.");
307  else
308  throw D_AdvancedRegisterAccessorException(
309  "CsaMultiBitFieldRegisterAccesor::getValueFromDevice: Exception detected. Device is NULL.");
310  }
311  }
312  catch(boost::numeric::negative_overflow& e) {
313  std::cout << "CsaMultiBitFieldRegisterAccesor<DataType>::getValueFromDevice(): " << e.what()
314  << " exception on: " << this->_registerName << ", module name: " << this->_mapModuleName << std::endl;
315  }
316  catch(boost::numeric::positive_overflow& e) {
317  std::cout << "CsaMultiBitFieldRegisterAccesor<DataType>::getValueFromDevice(): " << e.what()
318  << " exception on: " << this->_registerName << ", module name: " << this->_mapModuleName << std::endl;
319  }
320 
321  return localDataCopy;
322 }
