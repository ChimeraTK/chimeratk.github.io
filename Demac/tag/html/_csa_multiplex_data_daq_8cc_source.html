253void CsaMultiplexDataDaq::updateFromDevice(uint32_t bufferToRead) {
254 auto startTimePointTotal = std::chrono::high_resolution_clock::now();
255
256 DeviceModuleBase::updateFromDevice();
257
258 if(_disableDaqReadouts->getValue()) {
259 _daqReadDuration->setValue(0.0);
260 _updateDuration->setValue(0.0);
261 return;
262 }
263
264 _timeStamp = CsaAdapter::getInstance().globalData.getGlobalUniqueStamp();
265 // read DAQ (via DMA usually)
266 _daqBufferId = 0;
267 if(_doubleBufferingAvailable && bufferToRead) // if double buffering is ON and current buffer is different than zero
268 // we need to use value, in other cases we need to read from buffer 0
269 _daqBufferId = bufferToRead;
270
271 auto startTimePoint = std::chrono::high_resolution_clock::now();
272 try {
273 this->applyDmaReadSize();
274 if(_daqBufferId)
275 _daqDataAccesorBuff1.read();
276 else
277 _daqDataAccesorBuff0.read();
278 }
279 catch(ChimeraTK::runtime_error& ex) {
280 _logger(Loger::ERROR) << __func__ << ": ChimeraTK runtime_error exception detected: " << ex.what() << std::endl;
281 if(this->_device->isOpened()) this->_device->close();
282 }
283 catch(ChimeraTK::logic_error& ex) {
284 _logger(Loger::ERROR) << __func__ << ": ChimeraTK logic_error exception detected: " << ex.what() << std::endl;
285 try {
286 if(!this->_device->isOpened())
287 this->_device->open();
288 else
289 throw ex;
290 }
291 catch(ChimeraTK::runtime_error& ex_reopen) {
292 return;
293 }
294 }
295
296 auto stopTimePoint = std::chrono::high_resolution_clock::now();
297 auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(stopTimePoint - startTimePoint).count();
298 _daqReadDuration->setValue(duration);
299 if(_doubleBufferingAvailable)
300 _logger(Loger::DETAIL) << __func__ << ": "
301 << "Double buffering available. Current buffer: " << _daqBufferId
302 << ", DMA read duration: " << duration << " ms." << std::endl;
303 else
304 _logger(Loger::DETAIL) << __func__ << ": "
305 << "No double buffering. DMA read duration: " << duration << " ms." << std::endl;
306
307 for(auto& i : _dSeqencesVec) {
308 i->update(_timeStamp, _daqBufferId);
309 }
310
311 for(auto& i : _dSeqencesAliasesMap) {
312 i.second->update(_timeStamp, _daqBufferId);
313 }
314
315 _dNumberOfSequences->setValue(_numberOfSequences);
316
317 auto stopTimePointTotal = std::chrono::high_resolution_clock::now();
318 duration = std::chrono::duration_cast<std::chrono::milliseconds>(stopTimePointTotal - startTimePointTotal).count();
319 _updateDuration->setValue(duration);
320 _logger(Loger::DETAIL) << __func__ << ": "
321 << "Total update from device duration: " << duration << " ms." << std::endl;
322}
