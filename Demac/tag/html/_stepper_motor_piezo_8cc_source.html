280 void StepperMotorPiezo::moveToPositionInternal() {
281  if(!_initialized) {
282  _logger(Loger::ERROR) << __func__ << ": "
283  << "Piezo motor is not initialized. Return." << std::endl;
284  return;
285  }
286 
287  if(!_enable) {
288  _logger(Loger::ERROR) << __func__ << ": "
289  << "Piezo motor is disabled. Return." << std::endl;
290  return;
291  }
292  _stopMovement = false;
293 
294  int32_t distance = _targetPositionInStepsInternal - _positionInSteps;
295 
296  if(distance == 0) {
297  _logger(Loger::INFO) << __func__ << ": "
298  << "Target position equal current position. Return." << std::endl;
299  return;
300  }
301  _logger(Loger::DETAIL) << __func__ << ": "
302  << "Current position: " << _positionInSteps
303  << ". Target position: " << _targetPositionInStepsInternal << ". Distance is: " << distance
304  << std::endl;
305 
306  int loopsPerSecond = 20;
307  do {
308  distance = _targetPositionInStepsInternal - _positionInSteps;
309  if(distance == 0) break;
310 
311  int maxStep = (int)(_speedStepPerSecond / loopsPerSecond);
312  _positionInSteps = _piezoActuator->getDacValueInSteps();
313 
314  _logger(Loger::MAX_DETAIL) << __func__ << ": "
315  << "Current distance is: " << distance
316  << ". Current position is steps: " << _positionInSteps << " , max step size: " << maxStep
317  << std::endl;
318 
319  if(abs(distance) > maxStep) {
320  if(distance > maxStep)
321  _positionInSteps += maxStep;
322  else
323  _positionInSteps -= maxStep;
324  }
325  else
326  _positionInSteps += distance;
327 
328  _piezoActuator->setDacValueInSteps(_positionInSteps);
329 
330  //_position = recalculateStepsToUnits(_positionInSteps);
331  if(_stopMovement) {
332  _targetPositionInStepsInternal = _positionInSteps;
333  _stopMovement = false;
334  distance = 0;
335  }
336 
337  std::this_thread::sleep_for(std::chrono::milliseconds((int)(1000 / loopsPerSecond)));
338 
339  } while(distance != 0);
340  _logger(Loger::DETAIL) << __func__ << ": "
341  << "Current distance is: " << distance << ". Current position is steps: " << _positionInSteps
342  << ". Movement is done." << std::endl;
343 }

345 void StepperMotorPiezo::determineMotorStatusAndError() {
346  // assume that everything is ok
347  _motorStatus = SMAStatusTypes::M_OK;
348  _motorError = SMAErrorTypes::M_NO_ERROR;
349 
350  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
351  // ERROR CONDITION CHECK - it should be done first.
352  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
353 
354  if(_piezoActuator->getHardwareOk() == false) {
355  _motorStatus = SMAStatusTypes::M_ERROR;
356  _motorError = SMAErrorTypes::M_HARDWARE_ERROR;
357  if(_piezoHwdFailDetected == false)
358  _logger(Loger::ERROR) << __func__ << ": "
359  << "Critical error on PZT4 channel. Cannot operate." << std::endl;
360  _piezoHwdFailDetected = true;
361  _initialized = false;
362  return;
363  }
364  else if(_enable && _piezoHwdFailDetected) {
365  _logger(Loger::ERROR) << __func__ << ": "
366  << "PZT4 channel is back for standard operation. Try to reinitialize motor." << std::endl;
367  _piezoHwdFailDetected = false;
368  initializeInternal();
369  }
370 
371  // first check if disabled
372  if(!_enable) {
373  _motorStatus = SMAStatusTypes::M_DISABLED;
374  return;
375  }
376 
377  if(!_initialized) {
378  _motorStatus = SMAStatusTypes::M_ERROR;
379  _motorError = SMAErrorTypes::M_HARDWARE_ERROR;
380  return;
381  }
382 
383  if(_enableSoftwareLimits && (_softLimitMaxInSteps <= _softLimitMinInSteps)) {
384  _motorStatus = SMAStatusTypes::M_ERROR;
385  _motorError = SMAErrorTypes::M_CONFIG_ERROR_MIN_POS_GRATER_EQUAL_TO_MAX;
386 
387  return;
388  }
389 
390  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
391  // STATUS CHECK
392  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
393 
394  if(_enableSoftwareLimits) {
395  if(_positionInSteps >= _softLimitMaxInSteps) {
396  _motorStatus = SMAStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON;
397  return;
398  }
399 
400  if(_positionInSteps <= _softLimitMinInSteps) {
401  _motorStatus = SMAStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON;
402  return;
403  }
404  }
405 
406  // Due to unit recalculations we can have rounding errors which are causing that current position is calculated with
407  // error and never can reach hardware limiter to avoid it, motor will report hard end switch active +/- 5 steps before real value
408  if(_positionInSteps >= _hardLimitMaxInSteps - 5) {
409  _motorStatus = SMAStatusTypes::M_POSITIVE_END_SWITCHED_ON;
410  return;
411  }
412  if(_positionInSteps <= _hardLimitMinInSteps + 5) {
413  _motorStatus = SMAStatusTypes::M_NEGATIVE_END_SWITCHED_ON;
414  return;
415  }
416 
417  if(_moveInProgress) {
418  _motorStatus = SMAStatusTypes::M_IN_MOVE;
419  return;
420  }
421 
422  if(_targetPositionInSteps != _positionInSteps) {
423  _motorStatus = SMAStatusTypes::M_NOT_IN_POSITION;
424  return;
425  }
426 }
