160CsaCoarseTuningDetector::LIMIT CsaCoarseTuningDetector::checkSafetyCondition(bool updateInput) {
161 if((_disable->getValue() & DISABLE_MASK::ALL) || (_disable->getValue() & DISABLE_MASK::SAFETY)) {
162 // std::cout << "checkSafetyCondition::disabled" << std::endl;
163 return LIMIT::IN_RANGE;
164 }
165
166 // tests
167 if(triggerSafetyAlwaysTest) return LIMIT::MAX_CROSSED;
168 if(triggerSafetyOnceTest) {
169 triggerSafetyOnceTest = false;
170 return LIMIT::MAX_CROSSED;
171 }
172 //
173
174 double maxVal = _maxSafetyValue->getValue();
175 double minVal = _minSafetyValue->getValue();
176
177 // when parameters are equal safety check is disabled
178 if(maxVal == minVal) return LIMIT::IN_RANGE;
179
180 auto detectorValue = _currentValue->getValue();
181 if(updateInput) detectorValue = this->getCurrentValue();
182
183 if(_ctdError.value()) {
184 _logger(Loger::WARNING) << __func__ << ": "
185 << "Detector error. Trigger safety. Error: " << _ctdError.msg() << std::endl;
186 return LIMIT::MAX_CROSSED;
187 }
188
189 if(detectorValue > maxVal) {
190 _logger(Loger::INFO) << __func__ << ": "
191 << "Safety max value crossed. Current value: " << detectorValue << ", max value: " << maxVal
192 << std::endl;
193 return LIMIT::MAX_CROSSED;
194 }
195 if(detectorValue < minVal) {
196 _logger(Loger::INFO) << __func__ << ": "
197 << "Safety min value crossed. Current value: " << detectorValue << ", min value: " << minVal
198 << std::endl;
199 return LIMIT::MIN_CROSSED;
200 }
201 return LIMIT::IN_RANGE;
202}

204CsaCoarseTuningDetector::LIMIT CsaCoarseTuningDetector::checkStartCondition(bool updateInput) {
205 if((_disable->getValue() & DISABLE_MASK::ALL) || (_disable->getValue() & DISABLE_MASK::START)) {

287CsaCoarseTuningDetector::LIMIT CsaCoarseTuningDetector::checkStopConditionInternal(double detectorValue) {
288 if(_stopMode->getValue() == 0) {
289 // just for code readability
290 auto max = _stopParam2->getValue();
291 auto min = _stopParam1->getValue();
292
293 if((detectorValue >= min) && (detectorValue <= max)) {
294 _logger(Loger::INFO) << __func__ << ": "
295 << "In range. Tuning can be stopped. Current value: " << detectorValue << ", range: <" << min
296 << "," << max << ">." << std::endl;
297 return LIMIT::IN_RANGE;
298 }
299 else if(detectorValue < min) {
300 return LIMIT::MIN_CROSSED;
301 }
302 else if(detectorValue > max) {
303 return LIMIT::MAX_CROSSED;
304 }
305 else {
306 _logger(Loger::ERROR) << __func__ << ": "
307 << "Wrong conditions settings. Stop tuning for safety." << std::endl;
308 return LIMIT::IN_RANGE;
309 }
310 }
311
312 if(_stopMode->getValue() == 1) {
313 auto mean = _stopParam1->getValue();
314 auto maxDistance = _stopParam2->getValue();
315
316 if(detectorValue > (mean + maxDistance)) return LIMIT::MAX_CROSSED;
317
318 if(detectorValue < (mean - maxDistance)) return LIMIT::MIN_CROSSED;
319
320 _logger(Loger::INFO) << __func__ << ": "
321 << "In range. Tuning can be stopped. Current value: " << detectorValue << ", range: <"
322 << (mean - maxDistance) << "," << (mean - maxDistance) << ">." << std::endl;
323 return LIMIT::IN_RANGE;
324 }
325
326 // for safety reason it is better to stop
327 return LIMIT::IN_RANGE;
328}

368bool CsaCoarseTuningDetector::checkSettingsValidity() {
369 double startMin, startMax, stopMin, stopMax;
370 int32_t modeInUse;
371 getStartStopThresholds(startMin, startMax, stopMin, stopMax, modeInUse);
372
373 double maxVal = _maxSafetyValue->getValue();
374 double minVal = _minSafetyValue->getValue();
375 if(std::fabs(maxVal - minVal) > 0.00005 &&
376 !(_disable->getValue() & DISABLE_MASK::SAFETY)) { // if safety start/stop different and safety is not disabled
377 if(maxVal < minVal || startMax > maxVal || minVal > startMin) {
378 _logger(Loger::WARNING) << __func__ << ": "
379 << "Safety ranges error: startMax: " << startMax << ", startMin: " << startMin
380 << ", stopMax: " << stopMax << ", stopMin: " << stopMin << ", satetyMax: " << maxVal
381 << ", safety min: " << minVal << ", mode: " << modeInUse << std::endl;
382 return false;
383 }
384 }
385
386 if(!((startMax > stopMax) && (stopMax > stopMin) && (stopMin > startMin))) {
387 _logger(Loger::WARNING) << __func__ << ": "
388 << "Start/stop thresholds ranges error: startMax: " << startMax
389 << ", startMin: " << startMin << ", stopMax: " << stopMax << ", stopMin: " << stopMin
390 << ", mode: " << modeInUse << std::endl;
391 return false;
392 }
393
394 return true;
395}
