29 // this is reinitialization - check new address

39 try {

56 stream.str("");

58 _currentValue.init(stream.str(), CsAccessDir::FROM_CS);
59 _currentValueLocal = _currentValue.getValue();

61 stream.str("");

63 _sensorValue.init(stream.str(), CsAccessDir::FROM_CS);
64 _sensorValueLocal = _sensorValue.getValue();
65
66 _initializedLocal->value = true;
