34bool M_BSP_SIS8300Base::validateScratchRegisterMagicValue() {
35 auto scratchRegisterValue = _bspCommon->getScratchRegister();
36
37 uint32_t readMagic = (scratchRegisterValue & 0xFFFF0000) >> 16;
38 if(readMagic == INIT_MAGIC_NUMBER) {
39 uint32_t dateValue = (scratchRegisterValue & 0xFFFF);
40 uint32_t day = (dateValue & 0xF800) >> 11;
41 uint32_t month = (dateValue & 0x780) >> 7;
42 uint32_t year = (dateValue & 0x7F);
43 // std::ostringstream stream;
44 // stream << day <<"."<<month<<".20"<<year;
45 _logger(Loger::DETAIL) << __func__ << ": "
46 << "Magic value correct. Initialization date: "
47 << buildStr(true, day, ".", month, ".20", year) << std::endl;
48 return true;
49 }
50 else {
51 _logger(Loger::DETAIL) << __func__ << ": "
52 << "Magic value NOT correct. Read magic value: " << readMagic
53 << ", register value: " << scratchRegisterValue << std::endl;
54 return false;
55 }

58void M_BSP_SIS8300Base::setScratchRegisterMagicValue() {
59 _logger(Loger::DETAIL) << __func__ << ": "
60 << "Setting magic value to the scratch register." << std::endl;
61
62 auto now = std::chrono::system_clock::now();
63 std::time_t now_c = std::chrono::system_clock::to_time_t(now);
64 struct tm* parts = std::localtime(&now_c);
65
66 auto day = parts->tm_mday;
67 auto month = 1 + parts->tm_mon;
68 auto year = 1900 + parts->tm_year;
69
70 uint32_t scratch = ((INIT_MAGIC_NUMBER << 16) & 0xFFFF0000) + ((day << 11) & 0xF800) + ((month << 7) & 0x780) +
71 ((year - 2000) & 0x7F);
72 _bspCommon->setCsControlAllowed(false);
73 _bspCommon->setScratchRegister(scratch);
74 _bspCommon->setCsControlAllowed(true);
75
76 _logger(Loger::DETAIL) << __func__ << ": "
77 << "Scratch register value: " << scratch << ". Current date is: " << day << "." << month << "."
78 << year << "." << std::endl;
79}
