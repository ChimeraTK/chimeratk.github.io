327bool M_RapidX_ENT_PhaseDetection::calculateBeatFrequency() {
328 if(_algorithmFrequency <= 0) {
329 std::cout << "M_RapidX_ENT_PhaseDetection::calculateBeatFrequency():: " << _moduleName
330 << ":: Fatal error. Algorithm frequency not set. Contact software expert. Exit program" << std::endl;
331 exit(1);
332 return false;
333 }
334
335 if(!signalIsValid()) return false;
336
337 if(_mainController) {
338 if(_mainController->isActive()) {
339 setError(ERROR_TYPE::CTRL_ACTIVE, "Beat freq. cannot be calculated. Controller active.");
340 return false;
341 }
342 }
343
344 _beatFequency->setValue(0.0);
345
346 auto wrapCntValue = 0;
347 if(_wrapCounterManualTests && _wrapCounterManualTests->getValue() != 1234) {
348 wrapCntValue = _wrapCounterManualTests->getValue();
349 }
350 else
351 wrapCntValue = _wrapCounter->getValueFromDevice();
352
353 if(wrapCntValue) {
354 double freq = _algorithmFrequency / wrapCntValue;
355 if(fabs(freq) > 0.2) {
356 _beatFequency->setValue(_algorithmFrequency / wrapCntValue);
357 setError(ERROR_TYPE::NO_ERROR, "");
358 _resetWrapCounterApplied = false;
359 return true;
360 }
361 else {
362 setError(ERROR_TYPE::DETECTOR, "Beat freq. too small (<0.2 Hz). Not reliable. Wait.");
363 if(_wrapCounterReset && !_resetWrapCounterApplied) {
364 // std::cout << "Try to reset RF beat counter. " << std::endl;
365 _wrapCounterReset->setValue(1);
366 _wrapCounterReset->setValue(0);
367 _resetWrapCounterApplied = true;
368 // std::this_thread::sleep_for(std::chrono::milliseconds(15));
369 }
370 }
371 }
372 else
373 setError(ERROR_TYPE::DETECTOR, "Wrapping counter is 0. Cannot compute. Wait.");
374
375 return false;
376}
