127template<typename DataType>
128CsaRegisteredMemoryAccesor<DataType>::CsaRegisteredMemoryAccesor(std::string nameForCs, std::string regName,
129 std::string regModuleName, uint32_t memSize, DevAccessMode registerAccessMode, bool exposeShapeGeneratorToCs)
130: ExRegAccBase(regName, regModuleName, 0), _accessMode(registerAccessMode), _nameInControlSystem(nameForCs),
131 _memorySize(memSize), _exposeShapeGeneratorToCs(exposeShapeGeneratorToCs) {
132 this->_autoupdatePeriodMs = -1;
133
134 this->setDataType();
135
136 std::string toCsName = nameForCs + std::string(".HW_MEM");
137 std::string shapeGenName = nameForCs + std::string(".SG");
138
139 if(_accessMode == DevAccessMode::READ_WRITE) {
140 // std::cout << "CsaRegisteredMemoryAccesor<DataType>::CsaRegisteredMemoryAccesor: INFO: Create read/write register:
141 // " << regName << " as: " << nameForCs << std::endl;
142 _dataVectorToCs.reset(new CsaVector<DataType>(toCsName, _memorySize));
143 _shapeGenerator.reset(new CsaShapeGenerator<DataType>(shapeGenName, _memorySize, 6, exposeShapeGeneratorToCs));
144 }
145 else if(_accessMode == DevAccessMode::READ_ONLY) {
146 // std::cout << "CsaRegisteredMemoryAccesor<DataType>::CsaRegisteredMemoryAccesor: INFO: Create read only register:
147 // " << regName << " as: " << nameForCs << std::endl;
148 _dataVectorToCs.reset(new CsaVector<DataType>(toCsName, _memorySize));
149 }
150 else if(_accessMode == DevAccessMode::WRITE_ONLY) {
151 // std::cout << "CsaRegisteredMemoryAccesor<DataType>::CsaRegisteredMemoryAccesor: INFO: Create write only register:
152 // " << regName << " as: " << nameForCs << std::endl;
153 _shapeGenerator.reset(new CsaShapeGenerator<DataType>(shapeGenName, _memorySize, 6, exposeShapeGeneratorToCs));
154 }
155
156 std::string cmdName = nameForCs + std::string(".CMD");
157 _command.reset(new Csa_int32_t(cmdName, CsAccessDir::FROM_CS));
158 _command->setWriteCallback(boost::bind(&CsaRegisteredMemoryAccesor<DataType>::commandCb, this));
159}

161template<typename DataType>
162void CsaRegisteredMemoryAccesor<DataType>::shapeGeneratorAction(COMMAND_ID command) {
163 switch(command) {
164 case COMMAND_ID::GENERATE: {
165 if(this->_shapeGenerator) this->_shapeGenerator->generate();
166 break;
167 }
168 case COMMAND_ID::GENERATE_WRITE: {
169 if(this->_shapeGenerator) {
170 this->_shapeGenerator->generate();
171 this->setValue(this->_shapeGenerator->getDataVector());
172 }
173 break;
174 }
175
176 case COMMAND_ID::READ: {
177 this->getValueFromDevice();
178 break;
179 }
180 default:
181 break;
182 }
183}

186void CsaRegisteredMemoryAccesor<DataType>::commandCb() {
187 COMMAND_ID cmd = static_cast<COMMAND_ID>(_command->getValue());
188 if(_exposeShapeGeneratorToCs || cmd == COMMAND_ID::READ) shapeGeneratorAction(cmd);

219template<typename DataType>
220void CsaRegisteredMemoryAccesor<DataType>::init(boost::shared_ptr<ChimeraTK::Device> device) {
221 if(_deviceInitDone) {
222 // std::cout << "CsaRegisteredMemoryAccesor::init: Object: " << this->_nameInControlSystem << " (REGISTER): " <<
223 // this->_registerName << " (MODULE): " << this->_mapModuleName << " is already initialized." << std::endl;
224 return;
225 }
226 // std::cout << "CsaRegisteredMemoryAccesor::init(): Init of Hardware Property: " << this->_csaDataFromCs.getName() << std::endl;
227 _device = device;
228
229 if(this->_device) {
230 checkRegisterRename();
231 auto realRegisterName =
232 ChimeraTK::RegisterPath(this->_mapModuleName) / ChimeraTK::RegisterPath(this->_registerName);
233 auto dataDes = device->getRegisterCatalogue().getRegister(realRegisterName).getDataDescriptor();
234 if(this->_isSigned != dataDes.isSigned()) {
235 if(this->_isSigned) {
236 std::cout << "CsaRegisteredMemoryAccesor<DataType>::init: WARNING: Memory: " << this->_registerName
237 << " defined in a device as unsigned. " << this->_nameInControlSystem
238 << " is signed. Types mismatch! "
239 << std::endl; //<< regInfo.nFractionalBits << " Integer flag: " << this->_isInteger << " " <<
240 //(regInfo.nFractionalBits == 0) << std::endl;
241 }
242 else {
243 std::cout << "CsaRegisteredMemoryAccesor<DataType>::init: WARNING: Memory: " << this->_registerName
244 << " defined in a device as signed. " << this->_nameInControlSystem
245 << " is unsigned. Types mismatch! "
246 << std::endl; //<< regInfo.nFractionalBits << " Integer flag: " << this->_isInteger << " " <<
247 //(regInfo.nFractionalBits == 0) << std::endl;
248 }
249 }
250
251 if(dataDes.isIntegral() != this->_isInteger) {
252 if(!this->_isInteger)
253 std::cout << "CsaRegisteredMemoryAccesor<DataType>::init: WARNING: Memory: " << this->_registerName
254 << " defined in a device as integer. " << this->_nameInControlSystem
255 << " is floating point. Types mismatch!" << std::endl;
256 else
257 std::cout << "CsaRegisteredMemoryAccesor<DataType>::init: WARNING: Memory: " << this->_registerName
258 << " defined in a device as floating point. " << this->_nameInControlSystem
259 << " is integer. Types mismatch!" << std::endl;
260 }
261 auto nElements = device->getRegisterCatalogue().getRegister(realRegisterName).getNumberOfElements();
262 if(nElements != this->_memorySize) {
263 std::cout << "CsaRegisteredMemoryAccesor<DataType>::init: WARNING: Memory: " << this->_registerName
264 << " size in device is: " << nElements << ". Object memory size is: " << this->_memorySize
265 << ". Sizes mismatch!" << std::endl;
266
267 if(nElements < this->_memorySize) {
268 this->_memorySize = nElements;
269 std::cout << "CsaRegisteredMemoryAccesor<DataType>::init: INFO: Adjust memory size to value from hardware. New "
270 "memory size is: "
271 << this->_memorySize << std::endl;
272 _shapeGenerator->setMaxLength(this->_memorySize);
273 _dataVectorToCs->resize(this->_memorySize);
274 }
275 }
276
277 _registerAccessor1D.replace(_device->getOneDRegisterAccessor<DataType>(
278 this->_mapModuleName + "/" + this->_registerName, this->_memorySize));
279 _deviceInitDone = true;
280 }
281
282 this->getValueFromDevice();
283}

285template<typename DataType>
286void CsaRegisteredMemoryAccesor<DataType>::checkRegisterRename() {
287 std::ostringstream stream;
288
289 stream.str("");
290 stream << this->_mapModuleName << "." << this->_registerName << ":" << this->_registerOffset;
291
292 std::string remapIdSearch = std::string("REG_RENAME::") + stream.str();
293 std::string newRegName("");
294 // std::cout << "Search for key word: " << remapIdSearch << std::endl;
295 if(CsaAppConfigFileParser::getInstance().getValueAsDataString(remapIdSearch, newRegName)) {
296 auto posModuleEnd = newRegName.find(".", newRegName.find(".", 0) + 1);
297 auto posOfRegOffset = newRegName.find(":");
298
299 this->_mapModuleName = newRegName.substr(0, posModuleEnd);
300 this->_registerName = newRegName.substr(posModuleEnd + 1, posOfRegOffset - posModuleEnd - 1);
301
302 std::cout << __func__ << ": "
303 << "Registered Memory: " << stream.str() << " mapped to: " << newRegName
304 << " , Mod.name:" << this->_mapModuleName << " , Reg.name:" << this->_registerName << std::endl;
305 }

308template<typename DataType>
309void CsaRegisteredMemoryAccesor<DataType>::setValue(std::vector<DataType>& valueVector) {
310 if(this->_accessMode == DevAccessMode::READ_WRITE || this->_accessMode == DevAccessMode::WRITE_ONLY) {
311 if(this->_deviceInitDone) {
312 std::lock_guard<std::mutex> guard(_deviceAccessorMutex);
313
314 uint32_t writeSize = valueVector.size();
315 if(writeSize > this->_memorySize) writeSize = this->_memorySize;
316
317 for(uint32_t i = 0; i < writeSize; i++) _registerAccessor1D[i] = valueVector[i];
318
319 readWriteHwd(
320 true); // physical write from hardware with exception handling, replacement for: _registerAccessor1D.write();
321 }
322 }

368template<typename DataType>
369void CsaRegisteredMemoryAccesor<DataType>::readWriteHwd(bool isWrite) {
370 try {
371 if(isWrite)
372 this->_registerAccessor1D.write();
373 else
374 this->_registerAccessor1D.read();
375
376 // send also to control system, we are doing it after ' _registerAccessor1D.write()/ _registerAccessor1D.read()'
377 // because in case of some exception
378 _registerAccessor1D.swap(_dataVectorToCs->getBuffer());
379 }
380 catch(ChimeraTK::runtime_error& ex) {
381 if(this->_device->isOpened()) this->_device->close();
382 }
383 catch(ChimeraTK::logic_error& ex) {
384 try {
385 if(!this->_device->isOpened())
386 this->_device->open();
387 else
388 throw ex;
389 }
390 catch(ChimeraTK::runtime_error& ex_reopen) {
391 }
392 }
