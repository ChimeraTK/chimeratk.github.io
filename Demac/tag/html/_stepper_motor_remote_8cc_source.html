433 SMAStatus StepperMotorRemote::moveToPosition(float newPosition) {
434  _finishBlockingFunctions = false;
435 
436  checkInitializedAndThrow(std::string(__func__));
437  int motorErrorCnt = 0;
438  uint32_t delayMs = 200;
439  bool inPosition = false;
440 
441  auto motorStatus = this->getStatus();
442  auto position = this->getCurrentPosition();
443 
444  // std::cout << __func__ << ": " << "Current position: " << position << " , new position: " << newPosition << " ,
445  // status: " << motorStatus << std::endl;
446  if(newPosition == position) return SMAStatusTypes::M_OK;
447 
448  if(newPosition > position &&
449  (motorStatus == SMAStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON ||
450  motorStatus == SMAStatusTypes::M_POSITIVE_END_SWITCHED_ON)) {
451  return SMAStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON;
452  }
453  if(newPosition < position &&
454  (motorStatus == SMAStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON ||
455  motorStatus == SMAStatusTypes::M_NEGATIVE_END_SWITCHED_ON)) {
456  return SMAStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON;
457  }
458 
459  this->start(newPosition);
460  std::this_thread::sleep_for(std::chrono::milliseconds(delayMs));
461  do {
462  // wait to update all flags
463  std::this_thread::sleep_for(std::chrono::milliseconds(50));
464  auto currentPosition = this->getCurrentPosition();
465  auto motorStatus = this->getStatus();
466 
467  if(motorStatus == SMAStatusTypes::M_OK) {
468  if(currentPosition == newPosition) {
469  // std::cout << __func__ << ": " << "Current position is equal to new requested position. In position." << std::endl;
470  inPosition = true;
471  }
472  else if((this->getCurrentPosition() == this->getTargetPosition()) && fabs(currentPosition - newPosition) <= 1) {
473  // TO DO - need to fix it, there should be no difference. It comes probably from calculation rounding error
474  // between steps and units. std::cout << __func__ << ": " << "Target position reached, but difference detected:
475  // " << (currentPosition - newPosition) << std::endl;
476  inPosition = true;
477  }
478  else {
479  logger() << __func__ << ": "
480  << "Motor not moving, but position is wrong. Repeat order. Requested position: " << newPosition
481  << " , current position: " << currentPosition << std::endl;
482  this->start(newPosition);
483  std::this_thread::sleep_for(std::chrono::milliseconds(2 * delayMs));
484  }
485  }
486  else if(motorStatus == SMAStatusTypes::M_NOT_IN_POSITION) {
487  if(!_finishBlockingFunctions) {
488  logger() << __func__ << ": "
489  << "Motor not moving. Not in position status reported. Repeat start order and extra wait."
490  << std::endl;
491  this->start(newPosition);
492  std::this_thread::sleep_for(std::chrono::milliseconds(2 * delayMs));
493  }
494  }
495  else if(motorStatus == SMAStatusTypes::M_ERROR) {
496  logger() << __func__ << ": "
497  << "Motor is in error: " << this->getError() << ". Try to reset motor/tuner. " << std::endl;
498  if(motorErrorCnt < 5) {
499  this->resetMotorFlags();
500  this->start(newPosition);
501  motorErrorCnt++;
502  }
503  else {
504  logger() << __func__ << ": "
505  << "Movement failed. Too many motor/tuner errors." << std::endl;
506  return SMAStatusTypes::M_ERROR;
507  }
508  }
509  else if((motorStatus == SMAStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON ||
510  motorStatus == SMAStatusTypes::M_POSITIVE_END_SWITCHED_ON)) {
511  std::this_thread::sleep_for(std::chrono::milliseconds(delayMs));
512  if(this->getCurrentPosition() == currentPosition) return SMAStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON;
513  }
514  else if((motorStatus == SMAStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON ||
515  motorStatus == SMAStatusTypes::M_NEGATIVE_END_SWITCHED_ON)) {
516  std::this_thread::sleep_for(std::chrono::milliseconds(delayMs));
517  if(this->getCurrentPosition() == currentPosition) return SMAStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON;
518  }
519  else if(motorStatus != SMAStatusTypes::M_IN_MOVE) {
520  logger() << __func__ << ": "
521  << "Movement failed. Motor/tuner status: " << motorStatus << std::endl;
522  return SMAStatusTypes::M_ERROR;
523  }
524  } while(inPosition == false && _finishBlockingFunctions == false);
525 
526  if(_finishBlockingFunctions) {
527  _finishBlockingFunctions = false;
528  logger() << __func__ << ": "
529  << "Motor movement stopped. Current remote status: " << motorStatus
530  << ". Locally return: SMAStatusTypes::M_OK ." << std::endl;
531  }
532  return SMAStatusTypes::M_OK;
533 }

616  motorStatus == SMAStatusTypes::M_POSITIVE_END_SWITCHED_ON)) {
617  return SMAStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON;
618  }

620  (motorStatus == SMAStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON ||
621  motorStatus == SMAStatusTypes::M_NEGATIVE_END_SWITCHED_ON)) {
622  return SMAStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON;
623  }
624 
625  this->startInSteps(newMotorPosInSteps);
626  std::this_thread::sleep_for(std::chrono::milliseconds(delayMs));
627  do {
628  // wait to update all flags
629  std::this_thread::sleep_for(std::chrono::milliseconds(50));
630  auto currentPosition = this->getCurrentPositionInSteps();
631  auto motorStatus = this->getStatus();
632 
633  if(motorStatus == SMAStatusTypes::M_OK) {
634  if(currentPosition == newMotorPosInSteps) {
635  // std::cout << __func__ << ": " << "Current position is equal to new requested position. In position." << std::endl;
636  inPosition = true;
637  }
638  else if((this->getCurrentPositionInSteps() == this->getTargetPositionInSteps()) &&
639  std::fabs(currentPosition - newMotorPosInSteps) <= 3) {
640  // TO DO - need to fix it, there should be no difference. It comes probably from calculation rounding error
641  // between steps and units. std::cout << __func__ << ": " << "Target position reached, but difference detected:
642  // " << (currentPosition - newMotorPosInSteps) << std::endl;
643  inPosition = true;
644  }
645  else {

647  << "Motor not moving, but position is wrong. Repeat order. Requested position: " << newMotorPosInSteps
648  << " , current position: " << currentPosition << std::endl;
649  this->startInSteps(newMotorPosInSteps);
650  std::this_thread::sleep_for(std::chrono::milliseconds(2 * delayMs));
651  }
652  }
653  else if(motorStatus == SMAStatusTypes::M_NOT_IN_POSITION) {
654  if(!_finishBlockingFunctions) {
655  logger() << __func__ << ": "
656  << "Motor not moving. Not in position status reported. Repeat start order and extra wait."
657  << std::endl;
658  this->startInSteps(newMotorPosInSteps);
659  std::this_thread::sleep_for(std::chrono::milliseconds(2 * delayMs));
660  }
661  }
662  else if(motorStatus == SMAStatusTypes::M_ERROR) {
663  logger() << __func__ << ": "
664  << "Motor is in error: " << this->getError() << ". Try to reset motor/tuner. " << std::endl;
665  if(motorErrorCnt < 5) {
666  this->resetMotorFlags();
667  this->startInSteps(newMotorPosInSteps);
668  motorErrorCnt++;
669  }
670  else {
671  logger() << __func__ << ": "
672  << "Movement failed. Too many motor/tuner errors." << std::endl;
673  return SMAStatusTypes::M_ERROR;
674  }
675  }
676  else if((motorStatus == SMAStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON ||
677  motorStatus == SMAStatusTypes::M_POSITIVE_END_SWITCHED_ON)) {
678  std::this_thread::sleep_for(std::chrono::milliseconds(delayMs));
679  if(this->getCurrentPositionInSteps() == currentPosition) return SMAStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON;
680  }
681  else if((motorStatus == SMAStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON ||
682  motorStatus == SMAStatusTypes::M_NEGATIVE_END_SWITCHED_ON)) {
683  std::this_thread::sleep_for(std::chrono::milliseconds(delayMs));
684  if(this->getCurrentPositionInSteps() == currentPosition) return SMAStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON;
685  }
686  else if(motorStatus != SMAStatusTypes::M_IN_MOVE) {
687  logger() << __func__ << ": "
688  << "Movement failed. Motor/tuner status: " << motorStatus << std::endl;
689  return SMAStatusTypes::M_ERROR;
690  }
691  } while(inPosition == false && _finishBlockingFunctions == false);
692 
693  if(_finishBlockingFunctions) {
694  _finishBlockingFunctions = false;
695  logger() << __func__ << ": "
696  << "Motor movement stopped. Current remote status: " << motorStatus
697  << ". Locally return: SMAStatusTypes::M_OK ." << std::endl;
698  }
