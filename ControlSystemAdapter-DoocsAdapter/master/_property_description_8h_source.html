18  struct PersistConfig {
19  enum {
20  OFF = 0,
21  ON = 1, // write long arrays to separate files below hist/, short arrays to config file
22  AUTO = 2 // default DOOCS behaviour: persist only short arrays, in config file
23  };
24  int val;
25 
26  PersistConfig(std::string txt) {
27  transform(txt.begin(), txt.end(), txt.begin(), ::tolower);
28  if(txt == "false" or txt == "off" or txt == "0") {
29  val = OFF;
30  }
31  else if(txt == "true" or txt == "on" or txt == "1") {
32  val = ON;
33  }
34  else if(txt == "auto") {
35  val = AUTO;
36  }
37  else {
38  throw std::invalid_argument(std::string("Error parsing xml file: invalid input for PersistConfig: ") + txt);
39  }
40  }
41  PersistConfig(int enumVal) { this->val = enumVal; }
42  bool operator==(const PersistConfig& other) const { return val == other.val; }
43  };

87  struct AutoPropertyDescription : public PropertyDescription, public PropertyAttributes {
88  enum class DataType { Byte, Short, Int, Long, Float, Double, Bool, Void, Auto };
89  ChimeraTK::RegisterPath source;
90  AutoPropertyDescription(ChimeraTK::RegisterPath const& source_ = "", std::string location_ = "",
91  std::string name_ = "", DataType dataType_ = DataType::Auto, bool hasHistory_ = true, bool isWriteable_ = true)
92  : PropertyDescription(location_, name_), PropertyAttributes(hasHistory_, isWriteable_), source(source_),
93  dataType(dataType_) {}
94  virtual bool operator==(PropertyDescription const& other) const override {
95  if(other.type() == typeid(AutoPropertyDescription)) {
96  auto casted_other = static_cast<AutoPropertyDescription const&>(other);
97  return dataType == casted_other.dataType && source == casted_other.source && location == other.location &&
98  name == other.name && static_cast<const PropertyAttributes*>(this)->operator==(casted_other);
99  }
100  else {
101  return false;
102  }
103  }
104  const std::type_info& type() const override { return typeid(AutoPropertyDescription); }
105  void print(std::ostream& os = std::cout) const override {
106  os << source << " -> " << location << " / " << name << std::endl;
107  }
108 
109  void deriveType(std::type_info const& info) {
110  if(info == typeid(uint8_t) || info == typeid(int8_t)) dataType = DataType::Byte;
111  if(info == typeid(uint16_t) || info == typeid(int16_t)) dataType = DataType::Short;
112  if(info == typeid(uint32_t) || info == typeid(int32_t)) dataType = DataType::Int;
113  if(info == typeid(uint64_t) || info == typeid(int64_t)) dataType = DataType::Long;
114  if(info == typeid(float)) dataType = DataType::Float;
115  if(info == typeid(double)) dataType = DataType::Double;
116  if(info == typeid(ChimeraTK::Boolean)) dataType = DataType::Bool;
117  if(info == typeid(ChimeraTK::Void)) dataType = DataType::Void;
118  }
119 
120  std::set<std::string> getSources() override { return {source}; };
121 
122  DataType dataType;
123  };
