40  void DoocsIfff::checkSourceConsistency() {
41  bool areAllSourcesWritable =
42  (_i1Value->isWriteable() && _f1Value->isWriteable() && _f2Value->isWriteable() && _f3Value->isWriteable());
43  _isWriteable = areAllSourcesWritable;
44  bool areAllSourcesReadOnly =
45  (_i1Value->isReadOnly() && _f1Value->isReadOnly() && _f2Value->isReadOnly() && _f3Value->isReadOnly());
46 
47  if(!areAllSourcesWritable && !areAllSourcesReadOnly) {
48  throw ChimeraTK::logic_error("Doocs Adapter IFFF configuration Error: some IFFF sources are not writable");
49  }
50  if(areAllSourcesReadOnly) {
51  this->set_ro_access();
52  }
53  }

63  void DoocsIfff::updateDoocsBuffer(const TransferElementID& transferElementId) {
64  if(!updateConsistency(transferElementId)) {
65  return;
66  }
67 
68  bool storeInHistory = true;
69  auto archiverStatus = ArchiveStatus::sts_ok;
70  if(_i1Value->dataValidity() != ChimeraTK::DataValidity::ok ||
71  _f1Value->dataValidity() != ChimeraTK::DataValidity::ok ||
72  _f2Value->dataValidity() != ChimeraTK::DataValidity::ok ||
73  _f3Value->dataValidity() != ChimeraTK::DataValidity::ok) {
74  if(this->d_error()) {
75  // data are already invalid, do not store in history
76  storeInHistory = false;
77  }
78 
79  archiverStatus = ArchiveStatus::sts_err;
80  this->d_error(stale_data);
81  }
82  else {
83  this->d_error(no_error);
84  }
85 
86  IFFF ifff;
87  ifff.i1_data = _i1Value->accessData(0);
88  ifff.f1_data = _f1Value->accessData(0);
89  ifff.f2_data = _f2Value->accessData(0);
90  ifff.f3_data = _f3Value->accessData(0);
91 
92  doocs::Timestamp timestamp = correctDoocsTimestamp();
93  if(_macroPulseNumberSource) {
94  this->set_mpnum(_macroPulseNumberSource->accessData(0));
95  }
96  // We should also checked if data should be stored (flag storeInHistory). Invalid data should NOT be stored except
97  // first invalid data point. (https://github.com/ChimeraTK/ControlSystemAdapter-DoocsAdapter/issues/40)
98  if(this->get_histPointer() && storeInHistory) {
99  /*FIXME: This set_and_archive does not support the timestamp yet (only sec and msec, and I guess m is milli?)*/
100  /*FIXME: This set_and_archive does not support eventIDs yet */
101  this->set_and_archive(&ifff, archiverStatus, 0, 0 /*msec*/);
102  }
103  else {
104  this->set_value(&ifff);
105  }
106  // Note, I guess above will in future look something like this:
107  // doocs::EventId eventid(_macroPulseNumberSource->accessData(0));
108  // this->set_value(ifff, timestamp, eventid, archiverStatus);
109  // because the feature "do not store invalid data in history" is effectily lost
110  // when set_value() now also stores to history.
111  // However, currently it does _not_ yet.
112 
113  sendZMQ(timestamp);
114  }
