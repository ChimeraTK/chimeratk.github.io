55  void DoocsSpectrum::auto_init() {
56  doocsAdapter.before_auto_init();
57 
58  // check if the macro pulse number source has been set if the spectrum is buffered
59  if(nBuffers > 1) {
60  if(_macroPulseNumberSource == nullptr) {
61  throw ChimeraTK::logic_error(
62  "D_spectrum '" + _processArray->getName() + "' has numberOfBuffers > 1 but not macro pulse number source.");
63  }
64  }
65 
66  // send the current value to the device
67  D_spectrum::read();
68  modified = false;
69  if(this->get_access() == 1) { // property is writeable
70  sendToDevice(false);
71  // set DOOCS time stamp, workaround for DOOCS bug (get() always gives current time stamp if no timestamp is set,
72  // which breaks consistency check in ZeroMQ subscriptions after the 4 minutes timeout)
73  D_spectrum::set_stamp();
74  }
75  }

77  void DoocsSpectrum::write(std::ostream& s) {
78  // DOOCS is normally keeping the location lock until everything is written for that location: all D_spectrum and all
79  // other properties. This can take too long (like seconds), which leads to noticable freezes of the UI. As a
80  // work-around we release the lock here, wait some time and acquire the lock again. Since this happens in a separate
81  // thread (svr_writer), this slow-down should not be of any harm.
82  get_eqfct()->unlock();
83  usleep(1000);
84  get_eqfct()->lock();
85  if(!modified || _processArray->isReadOnly()) return;
86  modified = false;
87  D_spectrum::write(s);
88  }

101  void DoocsSpectrum::updateDoocsBuffer(const TransferElementID& transferElementId) {
102  // Note: we already own the location lock by specification of the DoocsUpdater
103 
104  // There are only the processArray and the macro pulse number in the consistency
105  // group. The limits are coming asynchronously and not for every macro pulse,
106  // so we just take test latest we have.
107  if(!updateConsistency(transferElementId)) {
108  return;
109  }
110 
111  doocs::Timestamp timestamp = correctDoocsTimestamp();
112 
113  auto sinceEpoch = timestamp.get_seconds_and_microseconds_since_epoch();
114  auto seconds = sinceEpoch.seconds;
115  auto microseconds = sinceEpoch.microseconds;
116 
117  // set macro pulse number, buffer number and time stamp
118  size_t ibuf = 0;
119  if(_macroPulseNumberSource != nullptr) {
120  ibuf = _macroPulseNumberSource->accessData(0) % nBuffers;
121  macro_pulse(_macroPulseNumberSource->accessData(0), ibuf);
122  }
123  set_tmstmp(seconds, microseconds, ibuf);
124 
125  if(_processArray->dataValidity() != ChimeraTK::DataValidity::ok) {
126  this->d_error(stale_data);
127  }
128  else {
129  this->d_error(no_error);
130  }
131 
132  // fill the spectrum
133  std::vector<float>& processVector = _processArray->accessChannel(0);
134  if(nBuffers == 1) {
135  // We have to fill the spectrum differently if it is unbuffered, as the internal data structures seem to be
136  // completely different.
137  memcpy(spectrum()->d_spect_array.d_spect_array_val, processVector.data(), processVector.size() * sizeof(float));
138  spectrum()->d_spect_array.d_spect_array_len = processVector.size();
139  }
140  else {
141  fill_spectrum(processVector.data(), processVector.size(), ibuf);
142  }
143 
144  // mark property as modified, for (optional) persistence
145  modified = true;
146 
147  sendZMQ(timestamp);
148  }

150  void DoocsSpectrum::updateParameters() {
151  // Note: we already own the location lock by specification of the DoocsUpdater
152  float start, increment;
153  if(_startAccessor) {
154  start = _startAccessor->accessData(0);
155  }
156  else {
157  start = this->spec_start();
158  }
159  if(_incrementAccessor) {
160  increment = _incrementAccessor->accessData(0);
161  }
162  else {
163  increment = this->spec_inc();
164  }
165 
166  // WORKAROUND: spectrum_parameter modifies the internal timestamp of
167  // the spectrum, this confuses our code that it thinks it has already sent off the
168  // spectrum with this timestamp, bumps the timestamp to enable DOOCS to check for inconsistencies
169  // which then confuses the DAQ since it sees data for the same MP with different timestamps
170  // https://mcs-gitlab.desy.de/doocs/doocs-core-libraries/serverlib/-/issues/35
171  auto oldTimeStamp = this->get_timestamp();
172  spectrum_parameter(this->spec_time(), start, increment, this->spec_status());
173  this->set_timestamp(oldTimeStamp);
174  }
