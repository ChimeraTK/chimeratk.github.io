110  void DoocsSpectrum::write(std::ostream& s) {
111  // DOOCS is normally keeping the location lock until everything is written for that location: all D_spectrum and all
112  // other properties. This can take too long (like seconds), which leads to noticable freezes of the UI. As a
113  // work-around we release the lock here, wait some time and acquire the lock again. Since this happens in a separate
114  // thread (svr_writer), this slow-down should not be of any harm.
115  efp_->unlock();
116  usleep(1000);
117  efp_->lock();
118  if(!modified || _processArray->isReadOnly()) return;
119  modified = false;
120  D_spectrum::write(s);
121  }

123  void DoocsSpectrum::updateDoocsBuffer(TransferElementID transferElementId) {
124  // Note: we already own the location lock by specification of the DoocsUpdater
125 
126  // FIXME: A first implementation is checking the data consistency here. Later this should be
127  // before calling this function because calling this function through a function pointer is
128  // comparatively expensive.
129  // Only check the consistency group if there is a macro pulse number associated.
130  // There are only the processArray and the macro pulse number in the consistency
131  // group. The limits are coming asynchronously and not for every macro pulse,
132  // so we just take test latest we have.
133  if(_macroPulseNumberSource && !_consistencyGroup.update(transferElementId)) {
134  // data is not consistent (yet). Don't update the Doocs buffer.
135  // check if this will now throw away data and generate a warning
136  if(transferElementId == _processArray->getId()) {
137  if(!_doocsSuccessfullyUpdated) {
138  ++_nDataLossWarnings;
139  if(DoocsAdapter::checkPrintDataLossWarning(_nDataLossWarnings)) {
140  std::cout << "WARNING: Data loss in spectrum property " << _eqFct->name() << "/" << this->basename()
141  << " due to failed data matching between value and macro pulse number (repeated "
142  << _nDataLossWarnings << " times)." << std::endl;
143  }
144  }
145  }
146  _doocsSuccessfullyUpdated = false;
147  return;
148  }
149  _doocsSuccessfullyUpdated = true;
150 
151  // Convert time stamp from version number in Unix time (seconds and microseconds).
152  // Note that epoch of std::chrono::system_time might be different from Unix time, and Unix time omits leap seconds
153  // and hence is not the duration since the epoch! We have to convert to time_t and then find out the microseconds.
154  auto timestamp = _processArray->getVersionNumber().getTime();
155  auto seconds = std::chrono::system_clock::to_time_t(timestamp);
156  auto microseconds = std::chrono::duration_cast<std::chrono::microseconds>(
157  timestamp - std::chrono::system_clock::from_time_t(seconds))
158  .count();
159 
160  // set macro pulse number, buffer number and time stamp
161  size_t ibuf = 0;
162  if(_macroPulseNumberSource != nullptr) {
163  ibuf = _macroPulseNumberSource->accessData(0) % nBuffers;
164  macro_pulse(_macroPulseNumberSource->accessData(0), ibuf);
165  }
166  set_tmstmp(seconds, microseconds, ibuf);
167 
168  if(_processArray->dataValidity() != ChimeraTK::DataValidity::ok) {
169  this->d_error(stale_data);
170  }
171  else {
172  this->d_error(no_error);
173  }
174 
175  // fill the spectrum
176  std::vector<float>& processVector = _processArray->accessChannel(0);
177  if(nBuffers == 1) {
178  // We have to fill the spectrum differently if it is unbuffered, as the internal data structures seem to be
179  // completely different.
180  memcpy(spectrum()->d_spect_array.d_spect_array_val, processVector.data(), processVector.size() * sizeof(float));
181  spectrum()->d_spect_array.d_spect_array_len = processVector.size();
182  }
183  else {
184  fill_spectrum(processVector.data(), processVector.size(), ibuf);
185  }
186 
187  // mark property as modified, for (optional) persistence
188  modified = true;
189 
190  // send data via ZeroMQ if enabled and if DOOCS initialisation is complete
191  if(publishZMQ && ChimeraTK::DoocsAdapter::isInitialised) {
192  dmsg_info info;
193  memset(&info, 0, sizeof(info));
194  info.sec = seconds;
195  info.usec = microseconds;
196  if(_macroPulseNumberSource != nullptr) {
197  info.ident = _macroPulseNumberSource->accessData(0);
198  }
199  else {
200  info.ident = 0;
201  }
