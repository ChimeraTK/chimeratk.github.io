55  template<typename DOOCS_T, typename DOOCS_PRIMITIVE_T>
56  DoocsProcessArray<DOOCS_T, DOOCS_PRIMITIVE_T>::DoocsProcessArray(EqFct* eqFct, const std::string& doocsPropertyName,
57  const boost::shared_ptr<ChimeraTK::NDRegisterAccessor<DOOCS_PRIMITIVE_T>>& processArray, DoocsUpdater& updater)
58  : DOOCS_T(doocsPropertyName.c_str(), processArray->getNumberOfSamples(), eqFct),
59  PropertyBase(doocsPropertyName, updater), _processArray(processArray) {
60  // Check if the array length exceeds the maximum allowed length by DOOCS.
61  // DOOCS does not report this as an error and instead silently truncates the
62  // array length.
63  auto len = static_cast<size_t>(this->length());
64  if(processArray->getNumberOfSamples() != len) {
65  std::stringstream s;
66  s << "Error: The selected DOOCS data type for the variable '" << processArray->getName() << "' "
67  << "(mapped to the DOOCS name '" << _doocsPropertyName << "') seems not to support the requested length of "
68  << processArray->getNumberOfSamples() << " since the DOOCS property has a length of " << len
69  << ". Try selecting a different DOOCS type in the mappng XML file, e.g. a D_spectrum!";
70  throw ChimeraTK::logic_error(s.str());
71  }
72  setupOutputVar(processArray);
73  }

101  template<typename DOOCS_T, typename DOOCS_PRIMITIVE_T>
102  void DoocsProcessArray<DOOCS_T, DOOCS_PRIMITIVE_T>::updateDoocsBuffer(const TransferElementID& transferElementId) {
103  if(!updateConsistency(transferElementId)) {
104  return;
105  }
106 
107  // Note: we already own the location lock by specification of the
108  // DoocsUpdater
109  auto& processVector = _processArray->accessChannel(0);
110 
111  THE_DOOCS_TYPE* dataPtr;
112  if constexpr(std::is_same<THE_DOOCS_TYPE, DOOCS_PRIMITIVE_T>::value) {
113  // No cast necessary if types are identical.
114  dataPtr = processVector.data();
115  }
116  else if constexpr(std::is_same<DOOCS_PRIMITIVE_T, int64_t>::value &&
117  std::is_same<THE_DOOCS_TYPE, long long int>::value) {
118  // Cast from int64_t to long long int (which are different types!)
119  dataPtr = reinterpret_cast<long long int*>(processVector.data());
120  }
121  else if constexpr(std::is_same<DOOCS_PRIMITIVE_T, ChimeraTK::Boolean>::value &&
122  std::is_same<THE_DOOCS_TYPE, bool>::value) {
123  // FIXME: Is it really ok to use reinterpret_cast here?
124  static_assert(sizeof(ChimeraTK::Boolean) == sizeof(bool));
125  dataPtr = reinterpret_cast<bool*>(processVector.data());
126  }
127  else {
128  static_assert(std::is_same<THE_DOOCS_TYPE, DOOCS_PRIMITIVE_T>::value, "Bad type casting.");
129  }
130 
131  if(_processArray->dataValidity() != ChimeraTK::DataValidity::ok) {
132  this->d_error(stale_data);
133  }
134  else {
135  this->d_error(no_error);
136  }
137 
138  this->fill_array(dataPtr, processVector.size());
139  modified = true;
140 
141  doocs::Timestamp timestamp = correctDoocsTimestamp();
142 
143  if(_macroPulseNumberSource) {
144  this->set_mpnum(_macroPulseNumberSource->accessData(0));
145  }
146 
147  sendZMQ(timestamp);
148  }
