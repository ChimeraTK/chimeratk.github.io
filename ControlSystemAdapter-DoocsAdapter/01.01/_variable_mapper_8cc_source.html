366  void VariableMapper::import(std::string importSource, std::string importLocationName, std::string directory) {
367  // a slash will be added after the source, so we make the source empty for an
368  // import of everything
369  if(importSource == "/") {
370  importSource = "";
371  }
372 
373  // loop source tree, cut beginning, replace / with _ and add a property
374  for(auto const& processVariable : _inputVariables) {
375  if(_usedInputVariables.find(processVariable) != _usedInputVariables.end()) {
376  continue;
377  }
378 
379  if(processVariable.find(importSource + "/") == 0) {
380  // processVariable starts with wanted source
381  auto nameSource = processVariable.substr(importSource.size() + 1); // add the slash to be removed
382  // we use the register path because it removes duplicate separators and
383  // allows to use . as separater to replace all / with .
384  ChimeraTK::RegisterPath propertyName;
385  std::string locationName;
386  if(importLocationName.empty()) {
387  // a global import, try to get the location name from the source
388  auto locationAndPropertyName = splitStringAtFirstSlash(nameSource);
389  locationName = locationAndPropertyName.first;
390  propertyName = locationAndPropertyName.second;
391  if(locationName.empty()) {
392  throw std::logic_error(std::string("Invalid XML content in global import of ") +
393  (importSource.empty() ? "/" : importSource) + ": Cannot create location name from '" + nameSource +
394  "', one hirarchy level is missing.");
395  }
396  // convenience for the user: You get an error message is you try a
397  // global import with directory (in case you did not validate your xml
398  // against the schema).
399  if(!directory.empty()) {
400  throw std::logic_error(std::string("Invalid XML content in global import of ") +
401  (importSource.empty() ? "/" : importSource) + ": You cannot have a directory in a global import.");
402  }
403  }
404  else {
405  // import into a location, we know the location name.
406  // add the directory first, then add the name source property
407  propertyName /= directory;
408  propertyName /= nameSource;
409  locationName = importLocationName;
410  }
411 
412  // get the property name with . instead of /
413  propertyName.setAltSeparator(".");
414  auto autoPropertyDescription = std::make_shared<AutoPropertyDescription>(
415  processVariable, locationName, propertyName.getWithAltSeparator());
416 
417  // we are importing, so all properties get the intended defaults (not
418  // individual settings)
419  autoPropertyDescription->hasHistory = getHasHistoryDefault(locationName);
420  autoPropertyDescription->isWriteable = getIsWriteableDefault(locationName);
421  autoPropertyDescription->macroPulseNumberSource = getMacroPusleNumberSourceDefault(locationName);
422  autoPropertyDescription->dataMatching = getDataMatchingDefault(locationName);
423 
424  addDescription(autoPropertyDescription, {processVariable});
425  }
426  }
