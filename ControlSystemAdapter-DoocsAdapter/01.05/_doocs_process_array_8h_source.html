234  void sendToDevice(bool getLocks) {
235  // Brute force implementation with a loop. Works for all data types.
236  // always get a fresh reference
237  auto& processVector = _processArray->accessChannel(0);
238  size_t arraySize = processVector.size();
239  if(this->length() != arraySize) {
240  std::cout << "Warning: Array length mismatch in property " << this->getEqFct()->name() << "/"
241  << this->basename() << ": Property has length " << this->length() << " but " << arraySize
242  << " expected." << std::endl;
243  arraySize = std::min(arraySize, size_t(this->length()));
244  }
245  for(size_t i = 0; i < arraySize; ++i) {
246  processVector[i] = this->value(i);
247  }
248  _processArray->write();
249 
250  // Correct property length in case of a mismatch.
251  if(this->length() != processVector.size()) {
252  this->set_length(processVector.size());
253  // restore value from ProcessArray, as it may have been destroyed in set_length().
254  for(size_t i = 0; i < arraySize; ++i) {
255  this->set_value(processVector[i], i);
256  }
257  }
258

260  if(getLocks) this->get_eqfct()->unlock();

262  if(getLocks) prop->getEqFct()->lock();
263  prop->updateDoocsBuffer({});
264  if(getLocks) prop->getEqFct()->unlock();
