449  // requesting a decorator. Implementation, declared at the top of this file
450  template<class UserType>
451  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> getDecorator(
452  const boost::shared_ptr<ChimeraTK::TransferElement>& transferElement, DecoratorType decoratorType) {
453  // check if there already is a decorator for the transfer element with the right type
454  DecoratorMapKey key{transferElement, typeid(UserType), decoratorType};
455  auto decoratorMapEntry = getGlobalDecoratorMap().find(key);
456  if(decoratorMapEntry != getGlobalDecoratorMap().end()) {
457  // There already is a decorator for this transfer element
458  // The decorator has to have a matching type, otherwise we can only throw
459  auto castedType = boost::dynamic_pointer_cast<ChimeraTK::NDRegisterAccessor<UserType>>(decoratorMapEntry->second);
460  assert(castedType);
461  assert(boost::dynamic_pointer_cast<DecoratorTypeHolder>(decoratorMapEntry->second));
462  return castedType;
463  }
464 
465  DecoratorFactory<UserType> factory(transferElement, decoratorType);
466  boost::fusion::for_each(ChimeraTK::userTypeMap(), std::ref(factory));
467  if(factory.createdDecorator == nullptr) {
468  throw ChimeraTK::logic_error("ChimeraTK::ControlSystemAdapter: Decorator for TransferElement " +
469  transferElement->getName() + " has been requested for an unknown user type: " + typeid(UserType).name());
470  }
471  getGlobalDecoratorMap()[key] = factory.createdDecorator;
472  return factory.createdDecorator;
473  }
