55  void PersistentDataStorage::writeToFile() noexcept {
56  try {
57  // create XML document with root node and a flat list of variables below this root
58  xmlpp::Document doc;
59  xmlpp::Element* rootElement =
60  doc.create_root_node("PersistentData", "https://github.com/ChimeraTK/ControlSystemAdapter");
61  rootElement->set_attribute("application", _applicationName);
62 
63  for(size_t i = 0; i < _variableNames.size(); ++i) {
64  if(!_variableRegisteredFromApp[i]) continue; // exclude variables no longer present in the application
65 
66  // create XML element for the variable and set name attribute
67  xmlpp::Element* variable = rootElement->add_child("variable");
68  variable->set_attribute("name", static_cast<std::string>(_variableNames[i]));
69 
70  // generate value XML tags and set type name as a string
71  DataType dataType(*_variableTypes[i]);
72  callForType(dataType, [&](auto t) {
73  using UserType = decltype(t);
74  generateXmlValueTags<UserType>(variable, i);
75  });
76 
77  // set type attribute
78  variable->set_attribute("type", dataType.getAsString());
79  }
80 
81  // write out to file
82  auto tempfile = _filename + ".new";
83  doc.write_to_file_formatted(tempfile);
84  std::rename(tempfile.c_str(), _filename.c_str());
85  }
86  catch(const std::exception& e) {
87  std::cerr << "Error writing persistency file: " << e.what() << std::endl;
88  }
89  catch(...) {
90  std::cerr << "Error writing persistency file (unknown exception)" << std::endl;
91  }
92  }

97  void PersistentDataStorage::generateXmlValueTags(xmlpp::Element* parent, size_t id) {
98  std::vector<UserType>* pValue;
99  {
100  // obtain the data vector from the map
101  std::lock_guard<std::mutex> lock(_queueReadMutex);
102  auto& value = boost::fusion::at_key<UserType>(_dataMap.table)[id].read_latest();
103  pValue = &value;
104  }
105  // add one child element per element of the value
106  for(size_t idx = 0; idx < pValue->size(); ++idx) {
107  xmlpp::Element* valueElement = parent->add_child("val");
108  valueElement->set_attribute("i", userTypeToUserType<std::string>(idx));
109  valueElement->set_attribute("v", userTypeToUserType<std::string>((*pValue)[idx]));
110  }
111  }

115  void PersistentDataStorage::readFromFile() {
116  // check if file exists
117  struct stat buffer {};
118  if(stat(_filename.c_str(), &buffer) != 0) {
119  // file does not exist: print message and do nothing
120  std::cerr << "ChimeraTK::PersistentDataStorage: Persistency file '" << _filename
121  << "' does not exist. It will be created when exiting the application." << std::endl;
122  return;
123  }
124 
125  try {
126  xmlpp::DomParser parser;
127  // parser.set_validate();
128  parser.set_substitute_entities(); // We just want the text to be resolved/unescaped automatically.
129  parser.parse_file(_filename);
130  if(parser) {
131  // obtain root node
132  const xmlpp::Node* rootElement = parser.get_document()->get_root_node(); // object will be deleted by DomParser
134 
135  // iterate through variables
136  for(const auto& elem : rootElement->get_children()) {
137  const auto* child = dynamic_cast<const xmlpp::Element*>(elem);
138  if(!child) continue; // comment or white spaces...
139  std::string name = child->get_attribute("name")->get_value();
140  std::string type = child->get_attribute("type")->get_value();
141  // compatibility with old persistency files - remove after July 2023
142  if(type == "double") type = "float64";
143  if(type == "float") type = "float32";
144  DataType dataType(type);
145  if(dataType == DataType::none) {
146  std::cerr << "Unknown data type '" + type + "' found in persist file: " << name << std::endl;
147  continue;
148  }
149  callForType(dataType, [&](auto t) {
150  using UserType = decltype(t);
151  readXmlValueTags<UserType>(child, registerVariable<UserType>(name, 0, true));
152  });
153  }
154  }
155  else {
156  throw ChimeraTK::logic_error("Could not parse persist file " + _filename + ": Failed to create the parser.");
157  }
158  }
159  catch(const std::exception& ex) {
160  throw ChimeraTK::logic_error("Could not parse persist file " + _filename + ": " + ex.what());
161  }
162  }

166  template<typename UserType>
167  void PersistentDataStorage::readXmlValueTags(const xmlpp::Element* parent, size_t id) {
168  // obtain the data vector from the map
169  std::vector<UserType>& value = boost::fusion::at_key<UserType>(_dataMap.table)[id].read_latest();
170 
171  // collect values
172  for(const auto& valElems : parent->get_children()) {
173  const auto* valChild = dynamic_cast<const xmlpp::Element*>(valElems);
174  if(!valChild) continue; // comment or white spaces...
175 
176  // obtain index and value as string
177  std::string s_idx = valChild->get_attribute("i")->get_value();
178  std::string s_val = valChild->get_attribute("v")->get_value();
179 
180  // convert to data type
181  auto idx = userTypeToUserType<size_t>(s_idx);
182  auto val = userTypeToUserType<UserType>(s_val);
183 
184  // resize vector if needed
185  if(value.size() <= idx) value.resize(idx + 1);
186 
187  // store value
188  value[idx] = val;
189  }
190  }
