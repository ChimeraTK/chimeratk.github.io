281  void BidirectionalProcessArray<T>::doPostRead(ChimeraTK::TransferType type, bool hasNewData) {
282  _receiver->postRead(type, hasNewData);
283  if(hasNewData) {
284  this->accessChannel(0).swap(_receiver->accessChannel(0));
285  // After receiving, our new time stamp and version number are the ones
286  // that we got from the receiver.
287  TransferElement::_versionNumber = _receiver->getVersionNumber();
288 
289  // Pass on data validity flag from sender to receiver and make it our
290  // our internal validity flag
291  TransferElement::setDataValidity(_receiver->dataValidity());
292  _sender->setDataValidity(TransferElement::dataValidity());
293

295  // do this because a (new) value received from the other side should be
296  // treated like a value sent by this side.
297  if(_persistentDataStorage) {
298  _persistentDataStorage->updateValue(_persistentDataStorageID, this->accessChannel(0));
299  }
300  }

310  template<class T>
311  bool BidirectionalProcessArray<T>::doWriteTransfer(ChimeraTK::VersionNumber versionNumber) {
312  // We have to copy our current value to the sender. We cannot swap it
313  // because this would mean that we would lose the current value.
314  _sender->accessChannel(0) = this->accessChannel(0);
315  // We already copied the value, so the sender does not have to copy the
316  // value again.
317 
318  // Propagate validity flag
319  _sender->setDataValidity(TransferElement::dataValidity());
320 
321  bool lostData = _sender->writeDestructively(versionNumber);
322 
323  // If we have a persistent data-storage, we have to update it.
324  if(_persistentDataStorage) {
325  _persistentDataStorage->updateValue(_persistentDataStorageID, this->accessChannel(0));
326  }
327  return lostData;
328  }
