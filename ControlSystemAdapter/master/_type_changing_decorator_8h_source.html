427  // the factory function. You don't have to care about the IMPL_Type when
428  // requesting a decorator. Implementation, declared at the top of this file
429  template<class UserType>
430  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> getDecorator(
431  const boost::shared_ptr<ChimeraTK::TransferElement>& transferElement, DecoratorType decoratorType) {
432  // check if there already is a decorator for the transfer element
433  auto decoratorMapEntry = getGlobalDecoratorMap().find(transferElement);
434  if(decoratorMapEntry != getGlobalDecoratorMap().end()) {
435  // There already is a decorator for this transfer element
436  // The decorator has to have a matching type, otherwise we can only throw
437  auto castedType = boost::dynamic_pointer_cast<ChimeraTK::NDRegisterAccessor<UserType>>(decoratorMapEntry->second);
438  if(castedType) { // User type matches, but the decorator type also has to
439  // match
440  auto decoTypeHolder =
441  boost::dynamic_pointer_cast<DecoratorTypeHolder>(decoratorMapEntry->second);
442  assert(decoTypeHolder);
445  if(decoTypeHolder->getDecoratorType() == decoratorType) {
446  // decorator matches, return it
447  return castedType;
448  }
449  else {
450  // sorry there is a decorator, but it's the wrong user type
451  throw ChimeraTK::logic_error("ChimeraTK::ControlSystemAdapter: Decorator for TransferElement " +
452  transferElement->getName() + " already exists as a different decorator type.");
453  }
454  }
455  else {
456  // sorry there is a decorator, but it's the wrong user type
457  throw ChimeraTK::logic_error("ChimeraTK::ControlSystemAdapter: Decorator for TransferElement " +
458  transferElement->getName() + " already exists with a different user type.");
459  }
460  }
461 
462  DecoratorFactory<UserType> factory(transferElement, decoratorType);
463  boost::fusion::for_each(ChimeraTK::userTypeMap(), std::ref(factory));
464  if(factory.createdDecorator == nullptr) {
465  throw ChimeraTK::logic_error("ChimeraTK::ControlSystemAdapter: Decorator for TransferElement " +
466  transferElement->getName() + " has been requested for an unknown user type: " + typeid(UserType).name());
467  }
468  getGlobalDecoratorMap()[transferElement] = factory.createdDecorator;
469  return factory.createdDecorator;
470  }
