652  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> getDecorator(
653  const boost::shared_ptr<ChimeraTK::TransferElement>& transferElement, DecoratorType decoratorType) {
654  // check if there already is a decorator for the transfer element with the right type
655  DecoratorMapKey key{transferElement, typeid(UserType), decoratorType};
656  auto decoratorMapEntry = getGlobalDecoratorMap().find(key);
657  if(decoratorMapEntry != getGlobalDecoratorMap().end()) {
658  // There already is a decorator for this transfer element
659  // The decorator has to have a matching type, otherwise we can only throw
660  auto castedType = boost::dynamic_pointer_cast<ChimeraTK::NDRegisterAccessor<UserType>>(decoratorMapEntry->second);
661  assert(castedType);
662  assert(boost::dynamic_pointer_cast<DecoratorTypeHolder>(decoratorMapEntry->second));
663  return castedType;
664  }
665 
666  DecoratorFactory<UserType> factory(transferElement, decoratorType);
667  boost::fusion::for_each(ChimeraTK::userTypeMap(), std::ref(factory));
668  if(factory.createdDecorator == nullptr) {
669  throw ChimeraTK::logic_error("ChimeraTK::ControlSystemAdapter: Decorator for TransferElement " +
670  transferElement->getName() + " has been requested for an unknown user type: " + typeid(UserType).name());
671  }
672  getGlobalDecoratorMap()[key] = factory.createdDecorator;
673  return factory.createdDecorator;
674  }
