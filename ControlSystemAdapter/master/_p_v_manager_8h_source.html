265  ChimeraTK::RegisterPath const& processVariableName) const {
266  ProcessVariableSharedPtrPair processVariable = getProcessVariable(processVariableName);
267  typename ProcessArray<T>::SharedPtr csPV =
268  boost::dynamic_pointer_cast<ProcessArray<T>, ProcessVariable>(processVariable.first);
269  typename ProcessArray<T>::SharedPtr devPV =
270  boost::dynamic_pointer_cast<ProcessArray<T>, ProcessVariable>(processVariable.second);
271  if(!csPV || !devPV) {
272  throw ChimeraTK::logic_error("PVManager::getProcessArray() called for variable '" + processVariableName +
273  "' with type " + typeid(T).name() + " which is not the original type " +
274  processVariable.first->getValueType().name() + " of this process variable.");
275  }
276  return std::make_pair(csPV, devPV);
