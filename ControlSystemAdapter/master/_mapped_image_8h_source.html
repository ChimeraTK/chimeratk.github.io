189  auto* p = data();

191  header()->totalLength = sh; // minimal length, could be larger
192  memset(p + sh, 0, capacity() - sh);

197  inline size_t MappedImage::formatsDefinition(
198  ImgFormat fmt, unsigned width, unsigned height, unsigned& channels, unsigned& bytesPerPixel) {
199  switch(fmt) {
200  case ImgFormat::Unset:
201  assert(false && "ImgFormat::Unset not allowed");
202  break;
203  case ImgFormat::Gray8:
204  channels = 1;
205  bytesPerPixel = 1;
206  break;
207  case ImgFormat::Gray16:
208  channels = 1;
209  bytesPerPixel = 2;
210  break;
211  case ImgFormat::RGB24:
212  channels = 3;
213  bytesPerPixel = 3;
214  break;
215  case ImgFormat::RGBA32:
216  channels = 4;
217  bytesPerPixel = 4;
218  break;
219  }
220  return sizeof(ImgHeader) + (size_t)width * height * bytesPerPixel;
221  }

229  inline void MappedImage::setShape(unsigned width, unsigned height, ImgFormat fmt) {
230  unsigned channels;
231  unsigned bytesPerPixel;

247  template<typename ValType, ImgOptions OPTIONS>
248  ValType& ImgView<ValType, OPTIONS>::operator()(unsigned dx, unsigned dy, unsigned channel) {
249  auto* h = header();
250  assert(dy < h->height);
251  assert(dx < h->width);
252  assert(channel < h->channels);
253  // this is the only place where row-major / column-major storage is decided
254  // note, definition of row major/column major is confusing for images.
255  // - for a matrix M(i,j) we say it is stored row-major if rows are stored without interleaving: M11, M12,...
256  // - for the same Matrix, if we write M(x,y) for pixel value at coordite (x,y) of an image, this means
257  // that pixel _columns_ are stored without interleaving
258  // So definition used here is opposite to matrix definition.
259  if constexpr((unsigned)OPTIONS & (unsigned)ImgOptions::RowMajor) {
260  return vec()[(dy * h->width + dx) * h->channels + channel];
261  }
262  else {
263  return vec()[(dy + dx * h->height) * h->channels + channel];
264  }
265  }
