224  template<class StructHeader>
225  void MappedStruct<StructHeader>::helpInit(bool doInitData) {
226  static_assert(std::is_base_of<OpaqueStructHeader, StructHeader>::value,
227  "MappedStruct expects StructHeader to implement OpaqueStructHeader");
228  if(doInitData) {
229  if(capacity() < sizeof(StructHeader)) {
230  throw logic_error("buffer provided to MappedStruct is too small for correct initialization");
231  }
232 
233  auto* p = data();
234  memset(p, 0, capacity());
235  new(p) StructHeader;
236  header()->totalLength = sizeof(StructHeader); // minimal length, could be larger
237  }
238  else {
239  if(header()->totalLength > capacity()) {
240  throw logic_error("buffer provided to MappedStruct is too small for assumed content");
241  }
242  }

269  }

295  template<typename ValType, ImgOptions OPTIONS>
296  ValType& ImgView<ValType, OPTIONS>::operator()(unsigned dx, unsigned dy, unsigned channel) {
297  assert(dy < _h->height);
298  assert(dx < _h->width);
299  assert(channel < _h->channels);
300  // this is the only place where row-major / column-major storage is decided
301  // note, definition of row major/column major is confusing for images.
302  // - for a matrix M(i,j) we say it is stored row-major if rows are stored without interleaving: M11, M12,...
303  // - for the same Matrix, if we write M(x,y) for pixel value at coordite (x,y) of an image, this means
304  // that pixel _columns_ are stored without interleaving
305  // So definition used here is opposite to matrix definition.
306  if constexpr((unsigned)OPTIONS & (unsigned)ImgOptions::RowMajor) {
307  return _vec[(dy * _h->width + dx) * _h->channels + channel];
308  }
309  else {
310  return _vec[(dy + dx * _h->height) * _h->channels + channel];
311  }
312  }
