245  template<class StructHeader>
246  void MappedStruct<StructHeader>::initData()
247  {
248  size_t sh = sizeof(StructHeader);
249  if (capacity() < sh) {
250  throw logic_error(
251  "buffer provided to MappedStruct is too small for correct initialization");
252  }
253  auto* p = data();
254  new(p) StructHeader;
255  header()->totalLength = sh; // minimal length, could be larger
256  memset(p + sh, 0, capacity() - sh);
257  }

261  inline size_t MappedImage::formatsDefinition(
262  ImgFormat fmt, unsigned width, unsigned height, unsigned& channels, unsigned& bytesPerPixel) {
263  switch(fmt) {
264  case ImgFormat::Unset:

277  bytesPerPixel = 3;

279  case ImgFormat::RGBA32:
280  channels = 4;
281  bytesPerPixel = 4;
282  break;
283  }
284  return sizeof(ImgHeader) + (size_t)width * height * bytesPerPixel;

311  template<typename ValType, ImgOptions OPTIONS>
312  ValType& ImgView<ValType, OPTIONS>::operator()(unsigned dx, unsigned dy, unsigned channel) {
313  assert(dy < _h->height);
314  assert(dx < _h->width);
315  assert(channel < _h->channels);
316  // this is the only place where row-major / column-major storage is decided
317  // note, definition of row major/column major is confusing for images.
318  // - for a matrix M(i,j) we say it is stored row-major if rows are stored without interleaving: M11, M12,...
319  // - for the same Matrix, if we write M(x,y) for pixel value at coordite (x,y) of an image, this means
320  // that pixel _columns_ are stored without interleaving
321  // So definition used here is opposite to matrix definition.
322  if constexpr((unsigned)OPTIONS & (unsigned)ImgOptions::RowMajor) {
323  return _vec[(dy * _h->width + dx) * _h->channels + channel];
324  }
325  else {
326  return _vec[(dy + dx * _h->height) * _h->channels + channel];
327  }
328  }
