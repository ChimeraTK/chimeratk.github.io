28 std::pair<RegisterPath, DecoratorType> getPathAndType(RegisterPath path) {
29  DecoratorType type;
30  path.setAltSeparator(".");
31  auto typeName = path.getComponents().back();
32  if(typeName == "casted") {
33  type = DecoratorType::C_style_conversion;
34  }
35  else if(typeName == "limiting") {
36  type = DecoratorType::limiting;
37  }
38  else {
39  throw ChimeraTK::logic_error("Decorator type " + typeName + " not supported");
40  }
41 
42  // drop last component
43  path--;
44 
45  return {path, type};
46 }
