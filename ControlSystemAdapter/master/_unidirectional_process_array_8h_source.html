474  template<class T>
475  void UnidirectionalProcessArray<T>::doPostRead(ChimeraTK::TransferType, bool hasNewData) {
476  assert(checkThreadSafety());
477  if(hasNewData) {
478  // We have to check that the vector that we currently own still has the
479  // right size. Otherwise, the code using the sender might get into
480  // trouble when it suddenly experiences a vector of the wrong size.
481  assert(ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0].size() == _localBuffer._value.size());
482 
483  if(this->_accessModeFlags.has(AccessMode::wait_for_new_data)) {
484  // swap data out of the local buffer into the user buffer
485  ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0].swap(_localBuffer._value);
486  }
487  else {
488  // We have to mimic synchronous mode. Here we have to copy here because there might be multiple reads, and
489  // the reading code is allowed to swap out the user buffer, and has to get the correct value on the second read.
490  ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0] = _localBuffer._value;
491  }
492  TransferElement::_versionNumber = _localBuffer._versionNumber;
493  TransferElement::_dataValidity = _localBuffer._dataValidity;
494  }

533  template<class T>
534  bool UnidirectionalProcessArray<T>::writeInternal(VersionNumber newVersionNumber, bool shouldCopy) {
535  // thread safety check, if enabled (only active with debug flags enabled)
536  assert(checkThreadSafety());
537 
538  assert(this->isWriteable());
539 
540  // First update the persistent data storage, if any was associated. This
541  // cannot be done after sending, since the value might no longer be available
542  // within this instance.
543  if(_persistentDataStorage) {
544  _persistentDataStorage->updateValue(_persistentDataStorageID, _intermedateBuffer);
545  }
546 
547  // Set time stamp and version number
548  _localBuffer._versionNumber = newVersionNumber;
549  _localBuffer._dataValidity = TransferElement::dataValidity();
550 
551  // set the data by copying or swapping
552  assert(_localBuffer._value.size() == _intermedateBuffer.size());
553  if(shouldCopy) {
554  _localBuffer._value = _intermedateBuffer;
555  }
556  else {
557  _localBuffer._value.swap(_intermedateBuffer);
558  }
559 
560  // send the data to the queue
561  bool dataNotLost = _sharedState._queue.push_overwrite(std::move(_localBuffer));
562 
563  // if receiver does not have wait_for_new_data, do not return whether data has been lost (because conceptionally it
564  // hasn't)
565  if(!_receiver->getAccessModeFlags().has(AccessMode::wait_for_new_data)) {
566  return false;
567  }
568 
569  return !dataNotLost;
570  }
