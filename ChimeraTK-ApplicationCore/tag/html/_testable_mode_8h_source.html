274 template<typename T>
275 TestableMode::AccessorPair<T> TestableMode::decorate(
276 AccessorPair<T> other, const VariableNetworkNode& producer, const VariableNetworkNode& consumer) {
277 if(not _enabled) {
278 return other;
279 }
280
281 if(_debugDecorating) {
282 logger(Logger::Severity::debug, "TestableMode")
283 << " Decorating pair " << producer.getQualifiedName() << "[" << other.first->getId() << "] -> "
284 << consumer.getQualifiedName() << "[" << other.second->getId() << "]";
285 }
286
287 // create variable IDs
288 size_t varId = detail::TestableMode::getNextVariableId();
289 size_t varIdReturn;
290 AccessorPair<T> result;
291
292 if(producer.getDirection().withReturn) {
293 varIdReturn = detail::TestableMode::getNextVariableId();
294 }
295
296 // decorate the process variable if testable mode is enabled and mode is push-type
297 if(!producer.getDirection().withReturn) {
298 result.first = boost::make_shared<AccessorDecorator<T>>(*this, other.first, false, true, varId, varId);
299 result.second = boost::make_shared<AccessorDecorator<T>>(*this, other.second, true, false, varId, varId);
300 }
301 else {
302 result.first = boost::make_shared<AccessorDecorator<T>>(*this, other.first, true, true, varIdReturn, varId);
303 result.second = boost::make_shared<AccessorDecorator<T>>(*this, other.second, true, true, varId, varIdReturn);
304 }
305
306 // put the decorators into the list
307 auto& variable = _variables.at(varId);
308 variable.name = "Internal:" + producer.getQualifiedName();
309 if(consumer.getType() != NodeType::invalid) {
310 variable.name += "->" + consumer.getQualifiedName();
311 }
312 if(producer.getDirection().withReturn) {
313 auto& returnVariable = _variables.at(varIdReturn);
314 returnVariable.name = variable.name + " (return)";
315 }
316
317 return result;
318 }

322 template<typename T>
323 boost::shared_ptr<NDRegisterAccessor<T>> TestableMode::decorate(
324 boost::shared_ptr<NDRegisterAccessor<T>> other, DecoratorType direction, const std::string& name, size_t varId) {
325 if(not _enabled) {
326 return other;
327 }
328
329 if(_debugDecorating) {
330 logger(Logger::Severity::debug, "TestableMode")
331 << " Decorating single " << (direction == DecoratorType::READ ? "consumer " : "feeder ") << name << "["
332 << other->getId() << "]";
333 }
334
335 if(varId == 0) {
336 varId = detail::TestableMode::getNextVariableId();
337 }
338
339 _variables[varId].processVariable = other;
340 if(not name.empty()) {
341 _variables.at(varId).name = name;
342 }
343
344 auto pvarDec = boost::make_shared<AccessorDecorator<T>>(
345 *this, other, direction == DecoratorType::READ, direction == DecoratorType::WRITE, varId, varId);
346
347 return pvarDec;
348 }

356 template<typename UserType>
357 TestableMode::AccessorDecorator<UserType>::AccessorDecorator(detail::TestableMode& testableMode,
358 boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> accessor, bool handleRead, bool handleWrite,
359 size_t variableIdRead, size_t variableIdWrite)
360 : ChimeraTK::NDRegisterAccessorDecorator<UserType>(accessor), _handleRead(handleRead), _handleWrite(handleWrite),
361 _variableIdRead(variableIdRead), _variableIdWrite(variableIdWrite), _testableMode(testableMode) {
362 assert(_variableIdRead != 0);
363 assert(_variableIdWrite != 0);
364
365 // if receiving end, register for testable mode (stall detection)
366 if(this->isReadable() && handleRead) {
367 _testableMode._variables[_variableIdRead].processVariable = accessor;
368 assert(accessor->getAccessModeFlags().has(AccessMode::wait_for_new_data));
369 }
370
371 // if this decorating a bidirectional process variable, set the
372 // valueRejectCallback
373 auto bidir = boost::dynamic_pointer_cast<BidirectionalProcessArray<UserType>>(accessor);
374 if(bidir) {
375 bidir->setValueRejectCallback([this] { decrementCounter(); });
376 }
377 else {
378 assert(!(handleRead && handleWrite));
379 }
380 }

432 template<typename UserType>
433 void TestableMode::AccessorDecorator<UserType>::obtainLockAndDecrementCounter(bool hasNewData) {
434 if(!_testableMode.testLock()) {
435 _testableMode.lock("doReadTransfer " + this->getName());
436 }
437 if(!hasNewData) {
438 return;
439 }
440 auto& variable = _testableMode._variables.at(_variableIdRead);
441 if(variable.counter > 0) {
442 assert(_testableMode._counter > 0);
443 --_testableMode._counter;
444 --variable.counter;
445 if(_testableMode._enableDebug) {
446 logger(Logger::Severity::debug, "TestableMode")
447 << "TestableModeAccessorDecorator[name='" << this->getName() << "', id=" << _variableIdRead
448 << "]: testableMode.counter decreased, now at value " << _testableMode._counter << " / "
449 << variable.counter;
450 }
451 }
452 else {
453 if(_testableMode._enableDebug) {
454 logger(Logger::Severity::debug, "TestableMode")
455 << "TestableModeAccessorDecorator[name='" << this->getName() << "', id=" << _variableIdRead
456 << "]: testableMode.counter NOT decreased, was already at value " << _testableMode._counter << " / "
457 << variable.counter << "\n"
458 << variable.name;
459 }
460 }

481 template<typename UserType>
482 bool TestableMode::AccessorDecorator<UserType>::accountForWriteOperation(
483 const std::function<bool(void)>& writeOperation) {
484 bool dataLost = false;
485 if(!_testableMode.testLock()) {
486 // may happen if first write in thread is done before first blocking read
487 _testableMode.lock("write " + this->getName());
488 }
489
490 dataLost = writeOperation();
491
492 if(!dataLost) {
493 ++_testableMode._counter;
494 ++_testableMode._variables.at(_variableIdWrite).counter;
495 if(_testableMode._enableDebug) {
496 logger(Logger::Severity::debug, "TestableMode")
497 << "TestableModeAccessorDecorator::write[name='" << this->getName() << "', id=" << _variableIdWrite
498 << "]: testableMode.counter increased, now at value " << _testableMode._counter;
499 }
500 }
501 else {
502 if(_testableMode._enableDebug) {
503 logger(Logger::Severity::debug, "TestableMode")
504 << "TestableModeAccessorDecorator::write[name='" << this->getName() << "', id=" << _variableIdWrite
505 << "]: testableMode.counter not increased due to lost data";
506 }
507 }
508 return dataLost;
509 }
