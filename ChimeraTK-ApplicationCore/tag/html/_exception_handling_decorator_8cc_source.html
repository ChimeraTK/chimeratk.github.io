13 template<typename UserType>
14 ExceptionHandlingDecorator<UserType>::ExceptionHandlingDecorator(
15 boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> accessor, const VariableNetworkNode& networkNode)
16 : ChimeraTK::NDRegisterAccessorDecorator<UserType>(accessor), _direction(networkNode.getDirection()) {
17 const auto& deviceAlias = networkNode.getDeviceAlias();
18 const auto& registerName = networkNode.getRegisterName();
19
20 assert(Application::getInstance()._deviceManagerMap.count(deviceAlias) != 0);
21 auto deviceManager = Application::getInstance()._deviceManagerMap[deviceAlias];
22 _deviceManager = deviceManager;
23
24 // Consuming from the network means writing to the device what you consumed.
25 if(_direction.dir == VariableDirection::consuming) {
26 deviceManager->_writeRegisterPaths.emplace_back(registerName);
27
28 // writeable registers get a recoveryAccessor
29 // Notice: There will be write-accessors without recovery accessors in future (intentionally turned off by the
30 // application programmer). When this feature is added the VariableNetworkNode will get a new data member to
31 // indicate this.
32 auto nElements = networkNode.getNumberOfElements();
33
34 // The device in the deviceModule does not have a valid backend yet. It is set when open() is called, which has
35 // not happened yet. We have to get the backend from the application.
36 auto deviceBackend = Application::getInstance()._deviceManagerMap.at(deviceAlias)->getDevice().getBackend();
37
38 _recoveryAccessor = deviceBackend->getRegisterAccessor<UserType>(
39 registerName, nElements, 0, {}); // recovery accessors don't have wait_for_new_data
40 // version number and write order are still {nullptr} and 0 (i.e. invalid)
41 _recoveryHelper = boost::make_shared<RecoveryHelper>(_recoveryAccessor, VersionNumber(nullptr), 0);
42
43 // add recovery accessor to DeviceManager so the last known value is restored during device recovery, unless
44 // the data type is Void, in which case there is no value to recover and writing will likely trigger some unwanted
45 // action.
46 if(!std::is_same<UserType, ChimeraTK::Void>::value) {
47 deviceManager->addRecoveryAccessor(_recoveryHelper);
48 }
49 }
50 else if(_direction.dir == VariableDirection::feeding) {
51 deviceManager->_readRegisterPaths.emplace_back(registerName);
52 }
53 else {
54 throw ChimeraTK::logic_error("Invalid variable direction in " + networkNode.getRegisterName());
55 }
56 }

60 template<typename UserType>
61 void ExceptionHandlingDecorator<UserType>::doPreWrite(TransferType type, VersionNumber versionNumber) {
62 auto deviceManager = _deviceManager.lock();
63
64 /* For writeable accessors, copy data to the recoveryAccessor before performing the write.
65 * Otherwise, the decorated accessor may have swapped the data out of the user buffer already.
66 * This obtains a shared lock from the DeviceModule, hence, the regular writing happening here
67 * can be performed in shared mode of the mutex and accessors are not blocking each other.
68 * In case of recovery, the DeviceModule thread will take an exclusive lock so that this thread can not
69 * modify the recoveryAccessor's user buffer while data is written to the device.
70 */
71 {
72 _inhibitWriteTransfer = false;
73 _hasThrownLogicError = false;
74 _dataLostInPreviousWrite = false;
75 auto recoverylock{deviceManager->getRecoverySharedLock()};
76
77 if(_recoveryAccessor != nullptr) {
78 if(!_recoveryHelper->wasWritten && (_recoveryHelper->writeOrder != 0)) {
79 _dataLostInPreviousWrite = true;
80 }
81
82 // Access to _recoveryAccessor is only possible channel-wise
83 for(unsigned int ch = 0; ch < _recoveryAccessor->getNumberOfChannels(); ++ch) {
84 _recoveryAccessor->accessChannel(ch) = buffer_2D[ch];
85 }
86 _recoveryHelper->versionNumber = versionNumber;
87 _recoveryHelper->writeOrder = deviceManager->writeOrder();
88 _recoveryHelper->wasWritten = false;
89 }
90 else {
91 _hasThrownLogicError = true;
92 throw ChimeraTK::logic_error(
93 "ChimeraTK::ExceptionhandlingDecorator: Calling write() on a non-writeable accessor is not supported ");
94 }
95
96 boost::shared_lock<boost::shared_mutex> errorLock(deviceManager->_errorMutex);
97 if(deviceManager->_deviceHasError) {
98 _inhibitWriteTransfer = true;
99 return;
100 }
101
102 ++deviceManager->_synchronousTransferCounter; // must be under the lock
103
104 } // lock guard goes out of scope
105
106 // Now delegate call to the generic decorator, which swaps the buffer, without adding our exception handling with
107 // the generic transfer preWrite and postWrite are only delegated if the transfer is allowed.
108 ChimeraTK::NDRegisterAccessorDecorator<UserType>::doPreWrite(type, versionNumber);
109 }

113 template<typename UserType>
114 void ExceptionHandlingDecorator<UserType>::doPostWrite(TransferType type, VersionNumber versionNumber) {
115 auto deviceManager = _deviceManager.lock();
116
117 if(_hasThrownLogicError) {
118 // preWrite has not been delegated, so there is nothing to do here. Let
119 // postWrite() throw the active exception we have. Don't clear logic erros here.
120 return;
121 }
122 if(!_inhibitWriteTransfer) {
123 --deviceManager->_synchronousTransferCounter;
124 try {
125 ChimeraTK::NDRegisterAccessorDecorator<UserType>::doPostWrite(type, versionNumber);
126 {
127 auto recoverylock{deviceManager->getRecoverySharedLock()};
128 // the transfer was successful or doPostRead did not throw and we reach this point,
129 // so we matk these data as written
130 _recoveryHelper->wasWritten = true;
131 } // end scope for recovery lock
132 }
133 catch(ChimeraTK::runtime_error& e) {
134 auto description = std::string(e.what()) + " (seen by '" + _target->getName() + "')";
135 // Report exception to the exception backend. This would be done by the TransferElement base class only if
136 // we would let the exception through, hence we have to take care of this here.
137 this->_exceptionBackend->setException(description);
138 // Report exception to the DeviceModule
139 deviceManager->reportException(description);
140 }
141 }
142 assert(_activeException == nullptr);
143 }

147 template<typename UserType>
148 void ExceptionHandlingDecorator<UserType>::doPostRead(TransferType type, bool hasNewData) {
149 auto deviceManager = _deviceManager.lock();
150
151 // preRead has not been called when the transfer was not allowed. Don't call postRead in this case.
152 if(!_hasThrownToInhibitTransfer) {
153 try {
154 if(!TransferElement::_accessModeFlags.has(AccessMode::wait_for_new_data)) { // was as synchronous transfer
155 --deviceManager->_synchronousTransferCounter;
156 }
157 _target->setActiveException(this->_activeException);
158 _target->postRead(type, hasNewData);
159 if(hasNewData) {
160 // Reset the flag after a successful read.
161 // It is only reset if there was new data. A readNonBlocking on a faulty device is not different to a
162 // readNonBlocking on working device: There just is no new data. We only reset it on the next successful read
163 // with the initial value, otherwise keep the exception flag.
164 _hasReportedException = false;
165 }
166 }
167 catch(ChimeraTK::runtime_error& e) {
168 auto description = std::string(e.what()) + " (seen by '" + _target->getName() + "')";
169 // Report exception to the exception backend. This would be done by the TransferElement base class only if
170 // we would let the exception through, hence we have to take care of this here.
171 this->_exceptionBackend->setException(description);
172 // Report exception to the DeviceModule
173 deviceManager->reportException(description);
174 _hasReportedException = true;
175 }
176 }
177 else {
178 _activeException = nullptr;
179 }
180
181 if(_hasReportedException || _hasThrownToInhibitTransfer) {
182 _dataValidity = DataValidity::faulty;
183 // Note: This assertion does not hold
184 // See discussion in https://github.com/ChimeraTK/DeviceAccess/pull/178
185 // assert(_deviceModule->getExceptionVersionNumber() > _versionNumber);
186 if(deviceManager->getExceptionVersionNumber() > _versionNumber) {
187 _versionNumber = deviceManager->getExceptionVersionNumber();
188 }
189 }
190 else {
191 _dataValidity = _target->dataValidity();
192 // Note: This assertion does not hold
193 // See discussion in https://github.com/ChimeraTK/DeviceAccess/pull/178
194 // assert(_target->getVersionNumber() >= _versionNumber);
195 if(_target->getVersionNumber() > _versionNumber) {
196 _versionNumber = _target->getVersionNumber();
197 }
198 }
199
200 // only replace the user buffer if there really is new data
201 if(hasNewData) {
202 for(size_t i = 0; i < buffer_2D.size(); ++i) {
203 buffer_2D[i].swap(this->_target->accessChannel(static_cast<unsigned int>(i)));
204 }
205 }
206 assert(_activeException == nullptr);
207 }

211 template<typename UserType>
212 void ExceptionHandlingDecorator<UserType>::doPreRead(TransferType type) {
213 auto deviceManager = _deviceManager.lock();
214
215 _hasThrownToInhibitTransfer = false;
216
217 if(TransferElement::_versionNumber == VersionNumber(nullptr)) {
218 deviceManager->waitForInitialValues();
219 // we don't have to store the shared lock. Once we acquired it the deviceModule will never take it again.
220 }
221
222 if(!TransferElement::_accessModeFlags.has(AccessMode::wait_for_new_data)) {
223 boost::shared_lock<boost::shared_mutex> errorLock(deviceManager->_errorMutex);
224 if(deviceManager->_deviceHasError) {
225 _hasThrownToInhibitTransfer = true;
226 throw ChimeraTK::runtime_error("ExceptionHandlingDecorator has thrown to skip read transfer");
227 }
228 // must hold the errorMutex while modifying the counter, and it must not be given up
229 // after the decision to do the transfer until here
230 ++deviceManager->_synchronousTransferCounter;
231 }
232
233 ChimeraTK::NDRegisterAccessorDecorator<UserType>::doPreRead(type);
234 }

252 template<typename UserType>
253 template<typename Callable>
254 bool ExceptionHandlingDecorator<UserType>::genericWriteWrapper(Callable writeFunction) {
255 if(_inhibitWriteTransfer) {
256 return _dataLostInPreviousWrite;
257 }
258 bool transferReportsPreviousDataLost = false;
259 try {
260 transferReportsPreviousDataLost = writeFunction();
261 }
262 catch(ChimeraTK::runtime_error&) {
263 _activeException = std::current_exception();
264 }
265 return (transferReportsPreviousDataLost || _dataLostInPreviousWrite);
266 }
