19  bool UserInputValidator::validate(const ChimeraTK::TransferElementID& change) {
20  if(!change.isValid()) {
21  return validateAll();
22  }
23 
24  if(!_finalised) {
25  throw ChimeraTK::logic_error("Initial values were not validated");
26  }
27 
28  // We have downstream channels that signalized a change - invalidate all of our
29  if(_downstreamInvalidatingReturnChannels.count(change) > 0) {
30  _module->setCurrentVersionNumber({});
31  for(auto& v : _variableMap) {
32  v.second->reject(VariableBase::RejectionType::downstream);
33  }
34 
35  return false;
36  }
37 
38  if(!_validatorMap.count(change)) {
39  return false;
40  }
41 
42  for(auto* validator : _validatorMap.at(change)) {
43  if(!validator->isValidFunction()) {
44  _errorFunction(validator->errorMessage);
45  _variableMap.at(change)->reject(VariableBase::RejectionType::self);
46  return true;
47  }
48  }
49 
50  _variableMap.at(change)->accept();
51  return false;
52  }

56  bool UserInputValidator::validateAll() {
57  if(!_finalised) {
58  finalise();
59  }
60  bool rejected = false;

63  }
64  return rejected;

69  void UserInputValidator::finalise() {
70  if(_finalised) {
71  return;
72  }
73 
74  for(auto& accessor : _module->getAccessorListRecursive()) {
75  if(accessor.getDirection() == VariableDirection{VariableDirection::feeding, true} &&
76  accessor.getModel().getTags().count(std::string(UserInputValidator::tagValidatedVariable)) > 0) {
77  _downstreamInvalidatingReturnChannels.emplace(accessor.getAppAccessorNoType().getId());
78  }
79  }
80 
81  // Find longest path that is validated in our model, starting at this module
82 
83  // Step 1: Do a topological order of the tree of modules with a return channel, starting from this module
84  std::deque<Model::ApplicationModuleProxy> stack;
85  std::map<ApplicationModule*, int> distances;
86 
87  // Visitor that a) builds the order in stack and b) initialises the distance array used later to calculate
88  // the distance from this node to the entry in that array
89  auto orderVisitor = [&](auto proxy) {
90  if constexpr(Model::isApplicationModule(proxy)) {
91  stack.push_front(proxy);
92  distances.try_emplace(&proxy.getApplicationModule(), std::numeric_limits<int>::min());
93  }
94  };
95 
96  // March through the tree with post order to properly build the stack. Technically sort a tree that is larger
97  // that what we want to look at, because we cannot stop the visit if there is a PV access with return channel
98  // that does not correspond to another UserInputValidator. However these distances should then left
99  // uninitialised in the distance calculation below and not taken into account
100  _module->getModel().visit(orderVisitor, Model::visitOrderPost, Model::depthFirstSearch,
101  Model::keepPvAccesWithReturnChannel, Model::keepApplicationModules);
102 
103  // Step 2: From the topological sort of the subtree, calculate the distances from our module to the currently
104  // checked module.
105  distances[_module] = 0;
106  std::unordered_set<ApplicationModule*> downstreamModulesWithFeedback;
107 
108  auto downstreamModuleCollector = [&](auto proxy) {
109  if constexpr(Model::isApplicationModule(proxy)) {
110  downstreamModulesWithFeedback.insert(&proxy.getApplicationModule());
111  }
112  };
113 
114  auto connectingVariableVisitor = [&](auto proxy) {
115  if constexpr(Model::isVariable(proxy)) {
116  proxy.visit(downstreamModuleCollector, Model::adjacentOutSearch, Model::keepPvAccesWithReturnChannel,
117  Model::keepProcessVariables);
118  }
119  };
120 
121  for(const auto& stackEntry : stack) {
122  downstreamModulesWithFeedback.clear();
123 
124  // We need to find all connected modules to the module currently looked at. Unfortunately we need to
125  // do that with a double visit, because the connection via PV access edges is not directly between modules
126  // but through a variable Vertex. So we first jump to the Variable using adjacentOut and in that visitor
127  // collect the modules into the downstreamModulesWithFeedback set
128  stackEntry.visit(connectingVariableVisitor, Model::adjacentOutSearch, Model::keepApplicationModules,
129  Model::keepPvAccesWithReturnChannel);
130 
131  // The distances from this module to the module on the stack is then just updated to be either what it
132  // was or the distance from the currently looked-at stack entry + 1 (since we have equal weights for the
133  // edges)
134  for(auto* vtx : downstreamModulesWithFeedback) {
135  distances[vtx] = std::max(distances[vtx], distances[&stackEntry.getApplicationModule()] + 1);
136  }
137  }
138 
139  _validationDepth = 1 + std::max_element(distances.begin(), distances.end(), [](auto& a, auto& b) -> bool {
140  return a.second < b.second;
141  })->second;
142 
143  for(auto& v : _variableMap) {
144  v.second->setHistorySize(_validationDepth);
145  }
146 
147  _finalised = true;
148  }
