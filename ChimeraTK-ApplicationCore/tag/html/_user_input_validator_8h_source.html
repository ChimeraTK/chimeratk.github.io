219 template<typename... ACCESSORTYPES>
220 void UserInputValidator::add(
221 const std::string& errorMessage, const std::function<bool(void)>& isValidFunction, ACCESSORTYPES&... accessors) {
222 boost::fusion::list<ACCESSORTYPES&...> accessorList{accessors...};
223 static_assert(boost::fusion::size(accessorList) > 0, "Must specify at least one accessor!");
224 assert(isValidFunction != nullptr);
225
226 // create validator and store in list
227 _validators.emplace_back(isValidFunction, errorMessage);
228
229 // create map of accessors to validators, also add accessors/variables to list
230 boost::fusion::for_each(accessorList, [&](auto& accessor) {
231 addAccessorIfNeeded(accessor);
232 _validatorMap[accessor.getId()].push_back(&_validators.back());
233 });
234 }

238 template<typename UserType, template<typename> typename Accessor>
239 void UserInputValidator::setFallback(Accessor<UserType>& accessor, UserType value) {
240 addAccessorIfNeeded(accessor);
241 auto pv = std::dynamic_pointer_cast<Variable<UserType, Accessor>>(_variableMap.at(accessor.getId()));
242 assert(pv != nullptr);
243 if(pv->fallbackValue.size() != 1) {
244 throw ChimeraTK::logic_error(
245 "UserInputValidator::setFallback() with scalar value called for array-typed accessor '" + accessor.getName() +
246 "'.");
247 }
248 pv->fallbackValue[0] = value;
249 }

268 void UserInputValidator::addAccessorIfNeeded(Accessor<UserType>& accessor) {
269 if(_module == nullptr) {
270 _module = dynamic_cast<ApplicationModule*>(dynamic_cast<Module*>(accessor.getOwner())->findApplicationModule());
271 }
272 if(!_variableMap.count(accessor.getId())) {
273 accessor.addTag(std::string(tagValidatedVariable));
274 _variableMap[accessor.getId()] = std::make_shared<Variable<UserType, Accessor>>(accessor);
275 }

299 template<typename UserType, template<typename> typename Accessor>
300 void UserInputValidator::Variable<UserType, Accessor>::reject(RejectionType type) {
301 if(type == RejectionType::downstream && !lastAcceptedValue.empty()) {
302 lastAcceptedValue.pop_back();
303 }
304 if constexpr(std::is_same<Accessor<UserType>, ChimeraTK::ScalarPushInput<UserType>>::value ||
305 std::is_same<Accessor<UserType>, ChimeraTK::ScalarPushInputWB<UserType>>::value) {
306 if(lastAcceptedValue.empty()) {
307 accessor = fallbackValue[0];
308 }
309 else {
310 accessor = lastAcceptedValue.back()[0];
311 }
312 }
313 else {
314 if(lastAcceptedValue.empty()) {
315 accessor = fallbackValue;
316 }
317 else {
318 accessor = lastAcceptedValue.back();
319 }
320 }
321
322 if constexpr(std::is_same<Accessor<UserType>, ChimeraTK::ScalarPushInputWB<UserType>>::value ||
323 std::is_same<Accessor<UserType>, ChimeraTK::ArrayPushInputWB<UserType>>::value) {
324 accessor.write();
325 }
326 }

329 template<typename UserType, template<typename> typename Accessor>
330 void UserInputValidator::Variable<UserType, Accessor>::accept() {
331 if constexpr(std::is_same<Accessor<UserType>, ChimeraTK::ScalarPushInput<UserType>>::value ||
332 std::is_same<Accessor<UserType>, ChimeraTK::ScalarPushInputWB<UserType>>::value) {
333 auto savedValue = std::vector<UserType>(1);
334 savedValue[0] = accessor;
335 lastAcceptedValue.push_back(savedValue);
336 }
337 else {
338 auto savedValue = std::vector<UserType>(accessor.getNElements());
339 savedValue = accessor;
340 lastAcceptedValue.push_back(savedValue);
341 }
