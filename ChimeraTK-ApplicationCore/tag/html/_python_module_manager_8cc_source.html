44  void PythonModuleManager::init() {
45  if(_impl) {
46  return;
47  }
48 
49  _impl = std::make_unique<detail::PythonModuleManagerImpl>();
50  if(!_impl->statics) {
51  // Create impl, with scoped_interpreter etc.
52 
53  // The impl object outlives the PythonModuleManager, since we need to keep the interpreter instance alive until
54  // the end of the process, even across multiple Application instances (e.g. in tests). Otherwise certain Python
55  // modules like numpy and datetime lead to crashes when loaded multiple times.
56  _impl->statics = std::make_unique<detail::PythonModuleManagerStatics>();
57 
58  // boost::thread_interrupted does not have a what() function, so we cannot use py::register_exception() directly
59  // on it. Also py::set_error() seems to be unavailable for our version of pybind11, also there is this bug:
60  // https://github.com/pybind/pybind11/issues/4967
61  // As a work around, we roughly replicated what register_exception_impl() is doing in pybind11 2.11
62 
63  // Create the exception object and store it in a static py::object, so the (state-less) lambda passed to
64  // register_exception_translator() below can access it. As long as there is only one PythonModuleManager at a time
65  // this is perfectly fine.
66  _impl->statics->exceptionObject = {py::module::import("__main__"), "ThreadInterrupted"};
67 
68  py::register_exception_translator([](std::exception_ptr p) {
69  try {
70  if(p) std::rethrow_exception(p);
71  }
72  catch(const boost::thread_interrupted& e) {
73  detail::PythonModuleManagerImpl::statics->exceptionObject("Thread Interrupted");
74  }
75  });
76 
77  // Make sure all Python modules are imported that we need
78  py::exec("import threading, traceback, sys, gc");
79  }
80 
81  // create main group object functioning as "app" object on the Python side
82  _impl->mainGroup = std::make_unique<PyModuleGroup>(&Application::getInstance(), ".", "Root for Python Modules");
83 
84  // If the bindings have been created already set/replace the "app" object with the newly created main group. This
85  // happens when a previous instance of the PythonModuleManager in the same process has already loaded Python
86  // modules using the ApplicationCore Python bindings. Otherwise the Python bindings are not yet loaded at this
87  // point, so this assignment is done later in PythonModuleManager::setBindings().
88  if(_impl->statics->onMainGroupChangeCallback) {
89  _impl->statics->onMainGroupChangeCallback(_impl->mainGroup);
90  }
91 
92  // The scoped_interpreter keeps the GIL by default, so we have to release it here and do the locking explicitly.
93  // In the destructor of the PythonModuleManager (hence when the Application is being destroyed) we have to
94  // acquire the lock again to make sure static objects created internally by pybind11 can be destroyed while
95  // having the GIL (would otherwise lead to an error).
96  _impl->release = std::make_unique<py::gil_scoped_release>();
97  }

105  void PythonModuleManager::deinit() {
106  if(!_impl) {
107  return;
108  }
109 
110  // terminate all Python ApplicationModule threads
111  for(auto* mod : _impl->mainGroup->getSubmoduleListRecursive()) {
112  mod->terminate();
113  }
114 
115  // destroy the gil_scoped_release object, see comment in constructor when creating it
116  _impl->release.reset();
117 
118  // de-assign the app object (which points to the root module we are about to destroy)
119  if(_impl->statics->onMainGroupChangeCallback) {
120  _impl->statics->onMainGroupChangeCallback(std::unique_ptr<PyModuleGroup>{nullptr});
121  }
122 
123  // unload all Python modules, which will destroy all PythonApplicationModules etc. that have been constructed in
124  // Python code.
125  for(auto& mod : _impl->modules) {
126  auto modname = py::cast<std::string>(mod.attr("__name__"));
127  py::exec("sys.modules.pop('" + modname + "')");
128  }
129  _impl->modules.clear();
130  py::exec("gc.collect()");
131  }

143  void PythonModuleManager::createModules(Application& app) {
144  auto& config = app.getConfigReader();
145  for(auto& module : config.getModules("PythonModules")) {
146  init();
147 
148  auto name = config.get<std::string>("PythonModules/" + module + "/path");
149  py::gil_scoped_acquire gil;
150 
151  std::cout << "PythonModuleManager: Loading module " << name << std::endl;
152 
153  try {
154  py::object themod = py::module::import(name.c_str());
155  _impl->modules.emplace_back(std::move(themod));
156  }
157  catch(py::error_already_set& err) {
158  throw ChimeraTK::logic_error("Error loading Python module from " + name + ": " + err.what());
159  }
160  }
161  }
