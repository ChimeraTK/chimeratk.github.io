77 FeedingFanOut<UserType>::FeedingFanOut(std::string const& name, std::string const& unit,
78 std::string const& description, size_t numberOfElements, bool withReturn,
79 ConsumerImplementationPairs<UserType> const& consumerImplementationPairs)
80 : FanOut<UserType>(boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>>()),
81 // We pass default-constructed, empty AccessModeFlags, they may later be determined from _returnSlave
82 ChimeraTK::NDRegisterAccessor<UserType>("FeedingFanOut:" + name, AccessModeFlags{}, unit, description),
83 _withReturn(withReturn) {
84 ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D.resize(1);
85 ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0].resize(numberOfElements);
86
87 if(_withReturn) {
88 this->_readQueue = cppext::future_queue<void>(3);
89 this->_accessModeFlags = {AccessMode::wait_for_new_data};
90 }
91
92 // Add the consuming accessors
93 // TODO FanOut constructors and addSlave should get refactoring
94 for(auto el : consumerImplementationPairs) {
95 FeedingFanOut<UserType>::addSlave(el.first, el.second);

101 template<typename UserType>
102 void FeedingFanOut<UserType>::addSlave(
103 boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> slave, VariableNetworkNode& node) {
104 // check if array shape is compatible, unless the receiver is a trigger
105 // node, so no data is expected
106 if(slave->getNumberOfSamples() != 0 &&
107 (slave->getNumberOfChannels() != 1 || slave->getNumberOfSamples() != this->getNumberOfSamples())) {
108 std::string what = "FeedingFanOut::addSlave(): Trying to add a slave '" + slave->getName();
109 what += "' with incompatible array shape! Name of fan out: '" + this->getName() + "'";
110 throw ChimeraTK::logic_error(what);
111 }
112
113 // make sure slave is writeable
114 if(!slave->isWriteable()) {
115 throw ChimeraTK::logic_error("FeedingFanOut::addSlave() has been called "
116 "with a receiving implementation!");
117 }
118
119 // handle return channels
120 if(_withReturn) {
121 if(node.getDirection().withReturn) {
122 if(_returnSlave) {
123 throw ChimeraTK::logic_error("FeedingFanOut: Cannot add multiple slaves with return channel!");
124 }
125
126 // These assumptions should be guaranteed by the connection making code which created the PV
127 assert(slave->isReadable());
128 assert(slave->getAccessModeFlags().has(AccessMode::wait_for_new_data));
129
130 _returnSlave = slave;
131
132 // Set the readQeue from the return slave
133 // As this becomes the implemention of the feeding output, the flags are determined by that slave accessor
134 // If not _withReturn, the queue is not relevant because the feeding node is on output which is never read
135 this->_readQueue = _returnSlave->getReadQueue();
136 this->_accessModeFlags = _returnSlave->getAccessModeFlags();
137 }
138 }
139
140 // add the slave
141 FanOut<UserType>::_slaves.push_back(slave);
142 }

171 template<typename UserType>
172 void FeedingFanOut<UserType>::doPostRead(TransferType type, bool hasNewData) {
173 if(this->_disabled || !_returnSlave) {
174 return;
175 }
176 assert(_withReturn);
177
178 auto _ = cppext::finally([&] {
179 if(!hasNewData || TransferElement::_activeException) {
180 return;
181 }
182 _returnSlave->accessChannel(0).swap(ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0]);
183 // distribute return-channel update to the other slaves
184 for(auto& slave : FanOut<UserType>::_slaves) { // send out copies to slaves
185 if(slave == _returnSlave) {
186 continue;
187 }
188 if(slave->getNumberOfSamples() != 0) { // do not send copy if no data is expected (e.g. trigger)
189 slave->accessChannel(0) = ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0];
190 }
191 slave->writeDestructively(_returnSlave->getVersionNumber());
192 }
193 });
194
195 _returnSlave->postRead(type, hasNewData);
196
197 this->_versionNumber = _returnSlave->getVersionNumber();
198 this->_dataValidity = _returnSlave->dataValidity();
199 }

203 template<typename UserType>
204 void FeedingFanOut<UserType>::doPreWrite(TransferType, VersionNumber) {
205 if(this->_disabled) {
206 return;
207 }
208 for(auto& slave : FanOut<UserType>::_slaves) { // send out copies to slaves
209 if(slave->getNumberOfSamples() != 0) { // do not send copy if no data is expected (e.g. trigger)
210 if(slave == FanOut<UserType>::_slaves.front()) { // in case of first slave, swap instead of copy
211 slave->accessChannel(0).swap(ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0]);
212 }
213 else { // not the first slave: copy the data from the first slave
214 slave->accessChannel(0) = FanOut<UserType>::_slaves.front()->accessChannel(0);
215 }
216 }
217 slave->setDataValidity(this->dataValidity());
218 }
219
220 // Don't call pre-write on the slaves. Each slave has to do it's own exception handling, so we call the whole
221 // operation in doWriteTansfer(). To fulfill the TransferElement specification we would have to check the
222 // pre-conditions here so no logic error is thrown in the transfer phase (logic_errors are predictable and can
223 // always pre prevented. They should be thrown here already).
224 // FIXME: At the moment we can be lazy about it. logic_errors are not treated in ApplicationCore and the only
225 // effect is that the logic_error would be delayed after postRead() and terminate the application there, and not
226 // after the transfer. Advantage about being lazy: It safes a few virtual function calls.
227 }

231 template<typename UserType>
232 bool FeedingFanOut<UserType>::doWriteTransfer(ChimeraTK::VersionNumber versionNumber) {
233 if(this->_disabled) {
234 return false;
235 }
236 bool dataLost = false;
237 bool isFirst = true;
238 for(auto& slave : FanOut<UserType>::_slaves) {
239 bool ret;
240 if(isFirst) {
241 isFirst = false;
242 ret = slave->write(versionNumber);
243 }
244 else {
245 ret = slave->writeDestructively(versionNumber);
246 }
247 if(ret) {
248 dataLost = true;
249 }
250 }
251 return dataLost;
252 }
