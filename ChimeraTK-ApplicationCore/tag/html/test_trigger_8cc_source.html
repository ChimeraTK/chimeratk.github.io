528 BOOST_AUTO_TEST_CASE(testTriggerTransferGroup) {
529 std::cout << "***************************************************************"
530 "******************************************************"
531 << std::endl;
532 std::cout << "==> testTriggerTransferGroup" << std::endl;
533
534 ChimeraTK::BackendFactory::getInstance().setDMapFilePath("test.dmap");
535
536 TestApplication app;
537 auto pvManagers = ctk::createPVManager();
538 app.setPVManager(pvManagers.second);
539
540 ChimeraTK::Device dev;
541 dev.open(dummySdm.data());
542 auto backend = boost::dynamic_pointer_cast<TestTransferGroupDummy>(
543 ChimeraTK::BackendFactory::getInstance().createBackend(dummySdm.data()));
544 BOOST_CHECK(backend != nullptr);
545
546 app.initialise();
547 app.run();
548 app.testModule.mainLoopStarted.wait(); // make sure the module's mainLoop() is entered
549
550 // initialise values
551 app.testModule.consumingPush = 0;
552 app.testModule.consumingPush2 = 0;
553 app.testModule.consumingPush3 = 0;
554 dev.write("/REG1.DUMMY_WRITEABLE", 11);
555 dev.write("/REG2.DUMMY_WRITEABLE", 22);
556 dev.write("/REG3.DUMMY_WRITEABLE", 33);
557
558 // from the initial value transfer
559 CHECK_TIMEOUT(backend->numberOfTransfers == 1, 10000);
560
561 // trigger the transfer
562 app.testModule.theTrigger.write();
563 CHECK_TIMEOUT(backend->numberOfTransfers == 2, 10000);
564 BOOST_TEST(backend->lastBar == 0);
565 BOOST_TEST(backend->lastAddress == 0);
566
567 // We only explicitly connect the three registers in the app, but the connection code will also connect the other
568 // registers into the CS, hence we need to check for the full size
569 BOOST_TEST(backend->lastSizeInBytes == 32);
570
571 // check result
572 app.testModule.consumingPush.read();
573 app.testModule.consumingPush2.read();
574 app.testModule.consumingPush3.read();
575 BOOST_CHECK_EQUAL(app.testModule.consumingPush, 11);
576 BOOST_CHECK_EQUAL(app.testModule.consumingPush2, 22);
577 BOOST_CHECK_EQUAL(app.testModule.consumingPush3, 33);
578
579 // prepare a second transfer
580 dev.write("/REG1.DUMMY_WRITEABLE", 12);
581 dev.write("/REG2.DUMMY_WRITEABLE", 23);
582 dev.write("/REG3.DUMMY_WRITEABLE", 34);
583
584 // trigger the transfer
585 app.testModule.theTrigger.write();
586 CHECK_TIMEOUT(backend->numberOfTransfers == 3, 10000);
587 BOOST_TEST(backend->lastBar == 0);
588 BOOST_TEST(backend->lastAddress == 0);
589
590 // We only explicitly connect the three registers in the app, but the connection code will also connect the other
591 // registers into the CS, hence we need to check for the full size
592 BOOST_TEST(backend->lastSizeInBytes == 32);
593
594 // check result
595 app.testModule.consumingPush.read();
596 app.testModule.consumingPush2.read();
597 app.testModule.consumingPush3.read();
598 BOOST_CHECK_EQUAL(app.testModule.consumingPush, 12);
599 BOOST_CHECK_EQUAL(app.testModule.consumingPush2, 23);
600 BOOST_CHECK_EQUAL(app.testModule.consumingPush3, 34);
601
602 dev.close();
603 }
