126  template<typename Derived>
127  void InversionOfControlAccessor<Derived>::replace(Derived&& other) {
128  assert(static_cast<Derived*>(this)->_impl == nullptr && other._impl == nullptr);
129 
130  // remove accessor from owning module
131  if(getOwner() != nullptr) {
132  getOwner()->unregisterAccessor(_node);
133  }
134 
135  // remove node from model
136  if(getModel().isValid()) {
137  getModel().removeNode(_node);
138  }
139 
140  // transfer the node
141  _node = std::move(other._node);
142  other._node = VariableNetworkNode(); // Make sure the destructor of other sees an invalid node
143 
144  // update the app accesor pointer in the node
145  if(_node.getType() == NodeType::Application) {
146  _node.setAppAccessorPointer(static_cast<Derived*>(this));
147  }
148  else {
149  assert(_node.getType() == NodeType::invalid);
150  }
151  // Note: the accessor is registered by the VariableNetworkNode, so we don't have to re-register.
152  }

178  "InversionOfControlAccessor<> must be used in a curiously recurring template pattern!");
179 
180  auto path = Utilities::getPathName(name);

182 
184  auto addToOnwer = [&](auto& owner_casted) {
185  auto model = owner_casted.getModel();
186  if(!model.isValid()) {
187  // this happens e.g. for default-constructed owners and their sub-modules
188  return;
189  }
190  auto neighbourDir = model.visit(
191  Model::returnDirectory, Model::getNeighbourDirectory, Model::returnFirstHit(Model::DirectoryProxy{}));
192 
193  auto dir = neighbourDir.addDirectoryRecursive(Utilities::getPathName(name));
194  auto var = dir.addVariable(Utilities::getUnqualifiedName(name));
195 
196  model.addVariable(var, _node);
197  };
198  auto* owner_am = dynamic_cast<ApplicationModule*>(owner);
199  auto* owner_vg = dynamic_cast<VariableGroup*>(owner);
200  if(owner_am) {
201  addToOnwer(*owner_am);
202  }
203  else if(owner_vg) {
204  addToOnwer(*owner_vg);
205  }
206  else {
207  throw ChimeraTK::logic_error("Hierarchy error!?");
208  }
209 
210  owner->registerAccessor(_node);
211  }
