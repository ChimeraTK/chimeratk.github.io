463 template<typename ProxyType>
464 class NonOwningProxy {
465 public:
467 NonOwningProxy() = default;
468
470 // NOLINTNEXTLINE(google-explicit-constructor)
471 NonOwningProxy(const ProxyType& owningProxy)
472 : _vertex(owningProxy.isValid() ? owningProxy._d->vertex : nullptr),
473 _impl(owningProxy.isValid() ? owningProxy._d->impl : nullptr) {}
474
476 NonOwningProxy& operator=(const ProxyType& owningProxy) {
477 if(owningProxy.isValid()) {
478 _vertex = owningProxy._d->vertex;
479 _impl = owningProxy._d->impl;
480 }
481 else {
482 _vertex = nullptr;
483 _impl.reset();
484 }
485 return *this;
486 }
487
490 ProxyType lock() {
491 ProxyType p;
492 p._d = std::make_shared<Proxy::ProxyData>(_vertex, _impl.lock());
493 return p;
494 }
495
496 private:
497 Model::Vertex _vertex{};
498 std::weak_ptr<Impl> _impl;
499 };

575 // Make proxy object for the given vertex or return existing proxy from weak pointer.
576 // Note: No checks are made that the given vertex is the right one. Also it is not checked whether the proxy type
577 // is correct, which is used in cases where the vertex can have multiple roles (e.g. root vertex can be RootProxy,
578 // ModuleGroupProxy and DirectoryProxy).
579 template<typename PROXY>
580 PROXY makeProxy(Vertex vertex, const std::shared_ptr<Impl>& impl) const;
581
582 private:
583 // (Weak) pointer to Proxy, stored here (with the vertex) to make sure we use the same Proxy instance whenever
584 // we refer to this vertex. Since the VertexProperties are stored inside the Graph and the Proxy contains a
585 // shared pointer to the Graph, the pointer has to be weak to avoid a shared pointer loop.
586 // This pointer is basically used as a cache for visitProxy() resp. makeProxy(), which needs to be a const
587 // function. Hence this member variable must be made mutable.
588 mutable std::weak_ptr<Proxy::ProxyData> _proxy;

816 explicit constexpr OrSet(FILTER_LHS lhs, FILTER_RHS rhs) : FILTER_LHS(lhs), FILTER_RHS(rhs) {}
817
818 [[nodiscard]] bool evalVertexFilter(const typename FILTER_LHS::PropertiesType& e) const {
819 return FILTER_LHS::evalVertexFilter(e) || FILTER_RHS::evalVertexFilter(e);
820 };

823 return FILTER_LHS::evalEdgeFilter(e) || FILTER_RHS::evalEdgeFilter(e);
824 };

829 }

832 constexpr auto operator&&(const FILTER_NEXT_RHS& rhs) const {
833 return AndSet<OrSet<FILTER_LHS, FILTER_RHS>, FILTER_NEXT_RHS>(*this, rhs);
834 }

836 template<typename PROXY>

1486 // Helper class for Impl::visit()
1487 template<typename BASE, typename VISITOR, typename FILTER, bool RETURN_FIRST_HIT>
1488 struct VisitorHelper : public BASE {
1489 // Call given visitor functor for each vertex (at "discover_vertex"), stop when finishing vertex stopAfterVertex
1490 // (checked in "finish_vertex"). By default, stopAfterVertex is set to an out-of-range index so the search is
1491 // not stopped. The stopping is realised by throwing a DfsVisitor::StopException.
1492 explicit VisitorHelper(VISITOR& visitor, std::shared_ptr<Impl> impl, FILTER& filter, Vertex stopAfterVertex,
1493 ValueHolder<detail::VisitorReturnType<VISITOR, FILTER>>& rv, VisitOrder& visitOrder)
1494 : _visitor(visitor), _filter(filter), _stopAfterVertex(stopAfterVertex), _impl(std::move(impl)),
1495 _rv(rv), _visitOrder{visitOrder} {}
1496
1497 // This is a required function by boost::graph - disable naming check
1498 template<class Vertex, class Graph>
1499 // NOLINTNEXTLINE(readability-identifier-naming)
1500 void discover_vertex(Vertex v, Graph& g) {
1501 if(_visitOrder.type == VisitOrder::VisitOrderType::in) {
1502 doVisit(v, g);
1503 }
1504 }
1505
1506 // This is a required function by boost::graph - disable naming check
1507 template<class Vertex, class Graph>
1508 // NOLINTNEXTLINE(readability-identifier-naming)
1509 void finish_vertex(Vertex v, Graph& g) {
1510 if(_visitOrder.type == VisitOrder::VisitOrderType::post) {
1511 doVisit(v, g);
1512 }
1513 if(v == _stopAfterVertex) {
1514 throw StopException();
1515 }
1516 }
1517
1518 template<class Vertex, class Graph>
1519 void doVisit(Vertex v, Graph& g) {
1520 // apply vertex filter
1521 if(!_filter.evalVertexFilter(g[v])) {
1522 return;
1523 }
1524
1525 if constexpr(RETURN_FIRST_HIT) {
1526 if constexpr(!std::is_same<detail::VisitorReturnType<VISITOR, FILTER>, void>::value) {
1527 _rv.value = g[v].visitProxy(_visitor, v, _impl);
1528 }
1529 else {
1530 g[v].visitProxy(_visitor, v, _impl);
1531 }
1532 throw StopException();
1533 }
1534 else {
1535 g[v].visitProxy(_visitor, v, _impl);
1536 }
1537 }
1538
1539 auto getReturnValue() { return _rv.get(); }
1540

1542 class StopException : public std::exception {};
1543
1544 protected:
1545 VISITOR& _visitor;
1546 FILTER& _filter;
1547 Vertex _stopAfterVertex;
1548 std::shared_ptr<Impl> _impl;
1549 ValueHolder<detail::VisitorReturnType<VISITOR, FILTER>>& _rv;
1550 VisitOrder _visitOrder;
1551 };

1557 template<typename VISITOR, typename... Args>
1558 auto Impl::visit(Vertex startVertex, VISITOR visitor, Args... args) {
1559 // increase visiting level to prevent modification while iterating, decrease when leaving this function
1560 _graphVisitingLevel++;
1561 auto decrementVisitingLevel = cppext::finally([&] { _graphVisitingLevel--; });
1562
1563 auto filteredGraph = getFilteredGraph(args...);
1564
1565 auto vertexFilter = Model::getVertexFilter(args...);
1566
1567 // Execute search operation based on selected type
1568 auto typeHolder = Model::getSearchType<Args...>();
1569 using type = typename decltype(typeHolder)::type;
1570 constexpr bool isAdjacent =
1571 std::is_same<AdjacentSearch, type>::value || std::is_base_of<AdjacentSearch, type>::value;
1572 constexpr bool isAdjacentIn =
1573 std::is_same<AdjacentInSearch, type>::value || std::is_base_of<AdjacentInSearch, type>::value;
1574 constexpr bool isAdjacentOut =
1575 std::is_same<AdjacentOutSearch, type>::value || std::is_base_of<AdjacentOutSearch, type>::value;
1576 constexpr bool isDFS =
1577 std::is_same<DepthFirstSearch, type>::value || std::is_base_of<DepthFirstSearch, type>::value;
1578 constexpr bool isBFS =
1579 std::is_same<BreadthFirstSearch, type>::value || std::is_base_of<BreadthFirstSearch, type>::value;
1580
1581 constexpr bool continueDisjunct = Model::hasSearchOption<ContinueSearchDisjunctTrees, Args...>();
1582 constexpr bool returnFirst = Model::hasSearchOption<ReturnFirstHit, Args...>();
1583
1584 // Wrapper for visitor to filter wrong objecttypes in constexpr condition, so the actual visitor does not have
1585 // to work with all proxy types, just the ones which can get past the VisitorFilter.
1586 using VisitorReturnType = detail::VisitorReturnType<VISITOR, decltype(vertexFilter)>;
1587 auto visitorWrapper = [&](auto proxy) -> VisitorReturnType {
1588 if constexpr(vertexFilter.template constevalObjecttype<decltype(proxy)>()) {
1589 return visitor(proxy);
1590 }
1591 // The VertexFilter makes sure that this function is never called with the wrong type, hence if we end up here
1592 // there is a bug in the VertexFilter. Note that this code will get instantiated, just never executed.
1593 throw ChimeraTK::logic_error("We should never end up here...");
1594 };
1595
1596 if constexpr(isAdjacent || isAdjacentOut) {
1597 auto [start, end] = boost::out_edges(startVertex, filteredGraph);
1598 for(auto it = start; it != end; ++it) {
1599 // obtain target vertex of the current outgoing edge (i.e. vertex at the other end of the edge)
1600 auto vtx = target(*it, _graph);
1601
1602 // apply vertex filter
1603 if(!vertexFilter.evalVertexFilter(_graph[vtx])) {
1604 continue;
1605 }
1606
1607 if constexpr(returnFirst) {
1608 return filteredGraph[vtx].visitProxy(visitorWrapper, vtx, shared_from_this());
1609 }
1610 else {
1611 filteredGraph[vtx].visitProxy(visitorWrapper, vtx, shared_from_this());
1612 }
1613 }
1614 }
1615 if constexpr(isAdjacent || isAdjacentIn) {
1616 auto [start, end] = boost::in_edges(startVertex, filteredGraph);
1617 for(auto it = start; it != end; ++it) {
1618 // obtain source vertex of the current incoming edge (i.e. vertex at the other end of the edge)
1619 auto vtx = source(*it, _graph);
1620
1621 // apply vertex filter
1622 if(!vertexFilter.evalVertexFilter(_graph[vtx])) {
1623 continue;
1624 }
1625
1626 if constexpr(returnFirst) {
1627 return filteredGraph[vtx].visitProxy(visitorWrapper, vtx, shared_from_this());
1628 }
1629 else {
1630 filteredGraph[vtx].visitProxy(visitorWrapper, vtx, shared_from_this());
1631 }
1632 }
1633 }
1634 if constexpr(isDFS || isBFS) {
1635 // For DFS: By default stop the search when the start vertex is finished, unless configured otherwise
1636 // For BFS: The search anyway does not continue with disjunct parts, and stopping when the start vertex is
1637 // finished would stop too early (no recursion).
1638 Vertex stopVertex;
1639 if constexpr(continueDisjunct || isBFS) {
1640 stopVertex = std::numeric_limits<Vertex>::max();
1641 }
1642 else {
1643 stopVertex = startVertex;
1644 }
1645
1646 // create visitor object
1647 detail::ValueHolder<detail::VisitorReturnType<decltype(visitorWrapper), decltype(vertexFilter)>> rv;
1648 constexpr bool hasVisitOrderOption = Model::hasSearchOption<VisitOrder, Args...>();
1649 VisitOrder visitOrder;
1650 if constexpr(hasVisitOrderOption) {
1651 visitOrder = Model::getSearchOption<VisitOrder, Args...>(args...);
1652 }
1653 auto visitorObjectFactory = [&]() {
1654 if constexpr(isDFS) {
1655 return detail::VisitorHelper<boost::dfs_visitor<>, decltype(visitorWrapper), decltype(vertexFilter),
1656 returnFirst>(visitorWrapper, shared_from_this(), vertexFilter, stopVertex, rv, visitOrder);
1657 }
1658 else {
1659 return detail::VisitorHelper<boost::bfs_visitor<>, decltype(visitorWrapper), decltype(vertexFilter),
1660 returnFirst>(visitorWrapper, shared_from_this(), vertexFilter, stopVertex, rv, visitOrder);
1661 }
1662 };
1663 auto visitorObject = visitorObjectFactory();
1664
1665 // Prepare a color_map for use with depth_first_search resp. breadth_first_search algorithms.
1666 //
1667 // Default color_map works only for boost::vecS for the vertex container, but we are using boost::listS.
1668 // BOOST documentation is minimalistic in this point. The following solution has been found "experimentally".
1669 //
1670 // The color_map is used by the algorithms (boost::depth_first_search and boost::breadth_first_search) to
1671 // keep track of seen vertices. It is used to store a value of the type boost::default_color_type for each
1672 // vertex, so the map key needs to be some kind of identifier for the vertex.
1673 //
1674 // The default color_map uses the vertex index as a key, which is not possible, because there is no vertex
1675 // index when using boost::listS. The color_map argument has follow the concept "read/write property map".
1676 // associative_property_map implements this concept and can use a std::map as a basis.
1677 //
1678 // The type of the map key (Vertex) was found out more or less by trial and error.
1679 std::map<Vertex, boost::default_color_type> colors;
1680 boost::associative_property_map color_map(colors);
1681
1682 // The try-catch block is just a trick to terminate the search operation early (when stopVertex hs finished)
1683 try {
1684 if constexpr(isDFS) {
1685 boost::depth_first_search(filteredGraph, visitorObject, color_map, startVertex);
1686 }
1687 else {
1688 boost::queue<Vertex> buffer;
1689 boost::breadth_first_search(filteredGraph, startVertex, buffer, visitorObject, color_map);
1690 }
1691 }
1692 catch(typename decltype(visitorObject)::StopException&) {
1693 // Ignore this exception, as it is just to use to trick the boost::depth/breadth_first_search() to stop
1694 }
1695
1696 if constexpr(returnFirst) {
1697 return visitorObject.getReturnValue();
1698 }
1699 }
1700
1701 if constexpr(returnFirst) {
1702 // Nothing found but ReturnFirstHit option specified: return the not-found value
1703 return Model::getSearchOption<ReturnFirstHit, Args...>(args...).notFoundValue.get();
1704 }
1705 }

1709 template<typename VISITOR, typename PROXY>
1710 bool Impl::visitByPath(std::string_view path, VISITOR visitor, PROXY startProxy) {
1711 // Note: the _graphVisitingLevel is not incremented here, since this visitByPath() function does not use any
1712 // iterators directly. Only when using visit() internally, iterators are used, but visit() increments the counter
1713 // itself.
1714
1715 // remove any redundant "./" at the beginning
1716 while(boost::starts_with(path, "./")) {
1717 path = path.substr(2);
1718 }
1719
1720 // resolve reference to ourself
1721 if(path.empty() || path == ".") {
1722 visitor(startProxy);
1723 return true;
1724 }
1725
1726 // first component is one level up
1727 if(boost::starts_with(path, "../") || path == "..") {
1728 // remove the component from the path
1729 path = (path.size() > 2) ? path.substr(3) : "";
1730
1731 // delegate to our parent proxy
1732 bool found = false;
1733 visit(
1734 startProxy._d->vertex,
1735 [&](auto parent) {
1736 if constexpr(isDirectory(parent) || isRoot(parent)) {
1737 found = parent.visitByPath(path, visitor);
1738 }
1739 else {
1740 assert(false); // directory structure guarantees only directories and the root as parents
1741 }
1742 },
1743 getParent);
1744 return found;
1745 }
1746
1747 // first component refers to root
1748 if(boost::starts_with(path, "/")) {
1749 // remove the leading slash
1750 path = path.substr(1);
1751
1752 // delegate to root proxy
1753 return RootProxy::makeRootProxy(shared_from_this()).visitByPath(path, visitor);
1754 }
1755
1756 // first component is a child
1757 // split by first slash
1758 auto slash = path.find_first_of('/');
1759 auto childName = path.substr(0, slash);
1760 path = (slash < path.length() - 1) ? path.substr(slash + 1) : "";
1761
1762 // delegate to child proxy
1763 bool found = false;
1764 visit(
1765 startProxy._d->vertex,
1766 [&](auto child) {
1767 if constexpr(isDirectory(child) || isVariable(child)) {
1768 if(child.getName() == childName) {
1769 found = child.visitByPath(path, visitor);
1770 }
1771 }
1772 else {
1773 assert(false); // search predicate guarantees only directories and variables
1774 }
1775 },
1776 children);
1777 return found;
1778 }
