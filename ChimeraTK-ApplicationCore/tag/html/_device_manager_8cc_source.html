18  std::vector<VariableNetworkNode> DeviceManager::getNodesList() const {
19  std::vector<VariableNetworkNode> rv;
20 
21  // obtain register catalogue
22  auto catalog = _device.getRegisterCatalogue();
23 
24  // iterate catalogue, create VariableNetworkNode for all registers
25  for(const auto& reg : catalog) {
26  // ignore 2D registers
27  if(reg.getNumberOfDimensions() > 1) {
28  continue;
29  }
30 
31  // guess direction and determine update mode
32  VariableDirection direction{};
33  UpdateMode updateMode;
34  if(reg.isWriteable()) {
35  direction = {VariableDirection::consuming, false};
36  updateMode = UpdateMode::push;
37  }
38  else {
39  direction = {VariableDirection::feeding, false};
40  if(reg.getSupportedAccessModes().has(AccessMode::wait_for_new_data)) {
41  updateMode = UpdateMode::push;
42  }
43  else {
44  updateMode = UpdateMode::poll;
45  }
46  }
47 
48  // find minimum type required to represent data
49  const std::type_info& valTyp = reg.getDataDescriptor().minimumDataType().getAsTypeInfo();
50 
51  // create node and add to list
52  rv.emplace_back(reg.getRegisterName(), _deviceAliasOrCDD, reg.getRegisterName(), updateMode, direction, valTyp,
53  reg.getNumberOfElements());
54  }
55 
56  return rv;
57  }

61  void DeviceManager::reportException(std::string errMsg) {
62  if(_owner->getTestableMode().isEnabled()) {
63  assert(_owner->getTestableMode().testLock());
64  }
65 
66  // The error queue must only be modified when holding both mutexes (error mutex and testable mode mutex), because
67  // the testable mode counter must always be consistent with the content of the queue.
68  // To avoid deadlocks you must always first aquire the testable mode mutex if you need both.
69  // You can hold the error mutex without holding the testable mode mutex (for instance for checking the error
70  // predicate), but then you must not try to aquire the testable mode mutex!
71  boost::unique_lock<boost::shared_mutex> errorLock(_errorMutex);
72 
73  if(!_deviceHasError) { // only report new errors if the device does not have reported errors already
74  if(_errorQueue.push(std::move(errMsg))) {
75  if(_owner->getTestableMode().isEnabled()) {
76  ++_owner->getTestableMode()._counter;
77  }
78  } // else do nothing. There are plenty of errors reported already: The queue is full.
79  // set the error flag and notify the other threads
80  _deviceHasError = true;
81  _exceptionVersionNumber = {}; // generate a new exception version number
82  errorLock.unlock();
83  }
84  else {
85  errorLock.unlock();
86  }
87  }

98  void DeviceManager::mainLoop() {
99  Application::registerThread("DM_" + getName());
100  std::string error;
101 
102  // We have the testable mode lock. The device has not been initialised yet, but from now on the
103  // testableMode.deviceInitialisationCounter will take care or it
104  _testableModeReached = true;
105 
106  // flag whether the devices was opened+initialised for the first time
107  bool firstSuccess = true;
108 
109  while(true) {
110  // [Spec: 2.3.1] (Re)-open the device.
111  do {
112  _owner->getTestableMode().unlock("Wait before open/recover device");
113  usleep(500000);
114  boost::this_thread::interruption_point();
115  _owner->getTestableMode().lock("Attempt open/recover device");
116  try {
117  // The globalDeviceOpenMutex is a work around for backends which do not implement open() in a thread-safe
118  // manner. This seems to be the case for most backends currently, hence it was decided to implement this
119  // workaround for now (see #11478).
120  static std::mutex globalDeviceOpenMutex;
121  std::lock_guard<std::mutex> globalDeviceOpenLock(globalDeviceOpenMutex);
122  _device.open();
123  }
124  catch(ChimeraTK::runtime_error& e) {
125  assert(_deviceError._status != StatusOutput::Status::OK); // any error must already be reported...
126  if(std::string(_deviceError._message) != e.what()) {
127  ChimeraTK::logger(Logger::Severity::error, "Device " + _deviceAliasOrCDD) << e.what() << std::endl;
128  // set proper error message in very first attempt to open the device
129  setCurrentVersionNumber({});
130  _deviceError.write(StatusOutput::Status::FAULT, e.what());
131  }
132  continue; // should not be necessary because isFunctional() should return false. But no harm in leaving it in.
133  }
134  } while(!_device.isFunctional());
135 
136  boost::unique_lock<boost::shared_mutex> errorLock(_errorMutex);
137 
138  // [Spec: 2.3.3] Empty exception reporting queue.
139  while(_errorQueue.pop()) {
140  if(_owner->getTestableMode()._enabled) {
141  assert(_owner->getTestableMode()._counter > 0);
142  --_owner->getTestableMode()._counter;
143  }
144  }
145  errorLock.unlock(); // we don't need to hold the lock for now, but we will need it later
146 
147  for(auto& writeMe : _writeRegisterPaths) {
148  auto reg = _device.getOneDRegisterAccessor<std::string>(writeMe); // the user data type does not matter here.
149  if(!reg.isWriteable()) {
150  throw ChimeraTK::logic_error(std::string(writeMe) + " is not writeable!");
151  }
152  }
153 
154  for(auto& readMe : _readRegisterPaths) {
155  auto reg = _device.getOneDRegisterAccessor<std::string>(readMe); // the user data type does not matter here.
156  if(!reg.isReadable()) {
157  throw ChimeraTK::logic_error(std::string(readMe) + " is not readable!");
158  }
159  }
160 
161  // [Spec: 2.3.2] Run initialisation handlers
162  try {
163  for(auto& initHandler : _initialisationHandlers) {
164  initHandler(_device);
165  }
166  }
167  catch(ChimeraTK::runtime_error& e) {
168  assert(_deviceError._status != StatusOutput::Status::OK); // any error must already be reported...
169  // update error message, since it might have been changed...
170  if(std::string(_deviceError._message) != e.what()) {
171  ChimeraTK::logger(Logger::Severity::error, "Device " + _deviceAliasOrCDD) << e.what() << std::endl;
172  setCurrentVersionNumber({});
173  _deviceError.write(StatusOutput::Status::FAULT, e.what());
174  }
175  // Jump back to re-opening the device
176  continue;
177  }
178 
179  // Write all recovery accessors
180  // We are now entering the critical recovery section. It is protected by the recovery mutex until the
181  // deviceHasError flag has been cleared.
182  boost::unique_lock<boost::shared_mutex> recoveryLock(_recoveryMutex);
183  try {
184  // sort recovery helpers according to write order
185  _recoveryHelpers.sort([](boost::shared_ptr<RecoveryHelper>& a, boost::shared_ptr<RecoveryHelper>& b) {
186  return a->writeOrder < b->writeOrder;
187  });
188  for(auto& recoveryHelper : _recoveryHelpers) {
189  if(recoveryHelper->versionNumber != VersionNumber{nullptr}) {
190  recoveryHelper->accessor->write();
191  recoveryHelper->wasWritten = true;
192  }
193  }
194  }
195  catch(ChimeraTK::runtime_error& e) {
196  // update error message, since it might have been changed...
197  if(std::string(_deviceError._message) != e.what()) {
198  ChimeraTK::logger(Logger::Severity::error, "Device " + _deviceAliasOrCDD) << e.what() << std::endl;
199  setCurrentVersionNumber({});
200  _deviceError.write(StatusOutput::Status::FAULT, e.what());
201  }
202  // Jump back to re-opening the device
203  continue;
204  }
205 
206  errorLock.lock();
207  _deviceHasError = false;
208  errorLock.unlock();
209 
210  recoveryLock.unlock();
211 
212  // send the trigger that the device is available again
213  _device.activateAsyncRead();
214  if(_isHoldingInitialValueLatch) {
215  _isHoldingInitialValueLatch = false;
216  _initialValueLatch.count_down();
217  }
218 
219  // [Spec: 2.3.5] Reset exception state and wait for the next error to be reported.
220  _deviceError.writeOk();
221  deviceBecameFunctional.write();
222 
223  if(!firstSuccess) {
224  ChimeraTK::logger(Logger::Severity::info, "Device " + _deviceAliasOrCDD) << "Error cleared." << std::endl;
225  }
226  firstSuccess = false;
227 
228  // decrement special testable mode counter, was incremented manually above to make sure initialisation completes
229  // within one "application step"
230  if(Application::getInstance().getTestableMode()._enabled) {
231  --_owner->getTestableMode()._deviceInitialisationCounter;
232  }
233 
234  // [Spec: 2.3.8] Wait for an exception being reported by the ExceptionHandlingDecorators
235  // release the testable mode mutex for waiting for the exception.
236  _owner->getTestableMode().unlock("Wait for exception");
237 
238  // Do not modify the queue without holding the testable mode lock, because we also consistently have to modify
239  // the counter protected by that mutex.
240  // Just call wait(), not pop_wait().
241  boost::this_thread::interruption_point();
242  _errorQueue.wait();
243  boost::this_thread::interruption_point();
244 
245  _owner->getTestableMode().lock("Process exception");
246  // increment special testable mode counter to make sure the initialisation completes within one
247  // "application step"
248  if(Application::getInstance().getTestableMode()._enabled) {
249  ++_owner->getTestableMode()._deviceInitialisationCounter; // matched above with a decrement
250  }
251 
252  errorLock.lock(); // we need both locks to modify the queue
253 
254  auto popResult = _errorQueue.pop(error);
255  assert(popResult); // this if should always be true, otherwise the waiting did not work.
256  (void)popResult; // avoid warning in production build. g++5.4 does not support [[maybe_unused]] yet.
257  if(_owner->getTestableMode()._enabled) {
258  assert(_owner->getTestableMode()._counter > 0);
259  --_owner->getTestableMode()._counter;
260  }
261 
262  // [ExceptionHandling Spec: C.3.3.14] report exception to the control system
263  ChimeraTK::logger(Logger::Severity::error, "Device " + _deviceAliasOrCDD) << error << std::endl;
264  setCurrentVersionNumber({});
265  _deviceError.write(StatusOutput::Status::FAULT, error);
266 
267  // We must not hold the lock while waiting for the synchronousTransferCounter to go back to 0. Only release it
268  // after deviceError has been written, so the CircularDependencyDetector can read the error message from its
269  // thread for printing.
270  errorLock.unlock();
271 
272  // [ExceptionHandling Spec: C.3.3.15] Wait for all synchronous transfers to finish before starting recovery.
273  while(_synchronousTransferCounter > 0) {
274  usleep(1000);
275  }
276 
277  } // while(true)
278  }

282  void DeviceManager::prepare() {
283  // Set initial status to error
284  setCurrentVersionNumber({});
285  _deviceError.write(StatusOutput::Status::FAULT, "Attempting to open device...");
286 
287  // Increment special testable mode counter to make sure the initialisation completes within one
288  // "application step". Start with counter increased (device not initialised yet, wait).
289  // We can to this here without testable mode lock because the application is still single threaded.
290  if(Application::getInstance().getTestableMode()._enabled) {
291  ++_owner->getTestableMode()._deviceInitialisationCounter; // released and increased in handeException loop
292  }
293  }

328  // The DeviceManager does not process the device registers, and hence circular networks involving the DeviceManager
329  // are not truely circular. Hence no real circular network cecking is done here.
330

361  void DeviceManager::terminate() {
362  if(_moduleThread.joinable()) {
363  _moduleThread.interrupt();
364  // try joining the thread
365  while(!_moduleThread.try_join_for(boost::chrono::milliseconds(10))) {
366  // send boost interrupted exception through the _errorQueue
367  _errorQueue.push_exception(std::make_exception_ptr(boost::thread_interrupted()));
368 
369  // it may not suffice to send the exception once, as the exception might get overwritten in the queue, thus we
370  // repeat this until the thread was joined.
371  }
372  }
373  assert(!_moduleThread.joinable());
374  }
