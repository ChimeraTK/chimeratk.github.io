30  { \
31  std::chrono::steady_clock::time_point t0 = std::chrono::steady_clock::now(); \
32  while(!(condition)) { \
33  bool timeout_reached = (std::chrono::steady_clock::now() - t0) > std::chrono::milliseconds(maxMilliseconds); \

35  if(timeout_reached) break; \

37  } \

70  using ctk::VariableGroup::VariableGroup;
71  ctk::ScalarPushInput<int32_t> v1{this, "v1", "cm", "Input 1 for testing"};
72  ctk::ScalarPushInput<int32_t> v2{this, "/REG2", "cm", "Input 2 for testing"};
73  ctk::ScalarPushInput<int32_t> v3{this, "v3", "cm", "Input 3 for testing"};
74  ctk::ScalarPushInput<int32_t> v4{this, "v4", "cm", "Input 4 for testing"};
75  };
76  Inputs inputs{this, "inputs", "A group of inputs"};
77  ctk::ScalarOutput<int32_t> value{this, "/value", "cm", "The last value received from any of the inputs"};
78  ctk::ScalarOutput<uint32_t> index{
79  this, "index", "", "The index (1..4) of the input where the last value was received"};
80 
81  void prepare() override {
82  incrementDataFaultCounter(); // force all outputs to invalid
83  writeAll();
84  decrementDataFaultCounter(); // validity according to input validity
85  }
86 
87  void mainLoop() override {
88  auto group = inputs.readAnyGroup();
89  while(true) {
90  auto justRead = group.readAny();
91  if(inputs.v1.getId() == justRead) {
92  index = 1;
93  value = int32_t(inputs.v1);
94  }
95  else if(inputs.v2.getId() == justRead) {
96  index = 2;
97  value = int32_t(inputs.v2);
98  }
99  else if(inputs.v3.getId() == justRead) {
100  index = 3;
101  value = int32_t(inputs.v3);
102  }
103  else if(inputs.v4.getId() == justRead) {
104  index = 4;
105  value = int32_t(inputs.v4);
106  }
107  else {
108  index = 0;
109  value = 0;
110  }
111  usleep(10000); // wait some extra time to make sure we are really blocking
112  // the test procedure thread
113  index.write();
114  value.write();
115  }
116  }
117  };
