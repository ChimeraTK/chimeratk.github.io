168  template<typename UserType>
169  void ScalarAccessor<UserType>::writeIfDifferent(UserType newValue) {
170  // Need to get to the MetaDataPropagatingRegisterDecorator to obtain the last written data validity for this PV.
171  // The dynamic_cast is ok, since the MetaDataPropagatingRegisterDecorator is always the outermost accessor, cf.
172  // the data validity propagation specification, Section 2.5.1.
173  auto* targetMetaDataPropagatingDecorator =
174  dynamic_cast<MetaDataPropagatingRegisterDecorator<UserType>*>(this->get());
175  assert(targetMetaDataPropagatingDecorator != nullptr);
176 
177  // In contrast to ScalarRegisterAccessor::writeIfDifferent(), we must not set the data validity on the target
178  // accessor, since that would be interpreted by the MetaDataPropagatingRegisterDecorator as an application-induced
179  // forced fault state. This would result in invalidity lock-ups if this happens in a circular network. Hence the
180  // comparison of the data validity must also be done against the validity of the decorator's target accessor which
181  // corresponds to the last written data validity for this PV.
182  if(this->get()->accessData(0, 0) != newValue || this->getVersionNumber() == VersionNumber(nullptr) ||
183  targetMetaDataPropagatingDecorator->getTargetValidity() != this->getOwner()->getDataValidity()) {
184  setAndWrite(newValue);
185  }
186  }
