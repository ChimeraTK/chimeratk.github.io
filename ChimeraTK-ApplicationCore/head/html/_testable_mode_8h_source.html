273  template<typename T>
274  TestableMode::AccessorPair<T> TestableMode::decorate(
275  AccessorPair<T> other, const VariableNetworkNode& producer, const VariableNetworkNode& consumer) {
276  if(not _enabled) {
277  return other;
278  }
279 
280  if(_debugDecorating) {
281  logger(Logger::Severity::debug, "TestableMode")
282  << " Decorating pair " << producer.getQualifiedName() << "[" << other.first->getId() << "] -> "
283  << consumer.getQualifiedName() << "[" << other.second->getId() << "]";
284  }
285 
286  // create variable IDs
287  size_t varId = detail::TestableMode::getNextVariableId();
288  size_t varIdReturn;
289  AccessorPair<T> result;
290 
291  if(producer.getDirection().withReturn) {
292  varIdReturn = detail::TestableMode::getNextVariableId();
293  }
294 
295  // decorate the process variable if testable mode is enabled and mode is push-type
296  if(!producer.getDirection().withReturn) {
297  result.first = boost::make_shared<AccessorDecorator<T>>(*this, other.first, false, true, varId, varId);
298  result.second = boost::make_shared<AccessorDecorator<T>>(*this, other.second, true, false, varId, varId);
299  }
300  else {
301  result.first = boost::make_shared<AccessorDecorator<T>>(*this, other.first, true, true, varIdReturn, varId);
302  result.second = boost::make_shared<AccessorDecorator<T>>(*this, other.second, true, true, varId, varIdReturn);
303  }
304 
305  // put the decorators into the list
306  auto& variable = _variables.at(varId);
307  variable.name = "Internal:" + producer.getQualifiedName();
308  if(consumer.getType() != NodeType::invalid) {
309  variable.name += "->" + consumer.getQualifiedName();
310  }
311  if(producer.getDirection().withReturn) {
312  auto& returnVariable = _variables.at(varIdReturn);
313  returnVariable.name = variable.name + " (return)";
314  }
315 
316  return result;
317  }

321  template<typename T>
322  boost::shared_ptr<NDRegisterAccessor<T>> TestableMode::decorate(
323  boost::shared_ptr<NDRegisterAccessor<T>> other, DecoratorType direction, const std::string& name, size_t varId) {
324  if(not _enabled) {
325  return other;
326  }
327 
328  if(_debugDecorating) {
329  logger(Logger::Severity::debug, "TestableMode")
330  << " Decorating single " << (direction == DecoratorType::READ ? "consumer " : "feeder ") << name << "["
331  << other->getId() << "]";
332  }
333 
334  if(varId == 0) {
335  varId = detail::TestableMode::getNextVariableId();
336  }
337 
338  _variables[varId].processVariable = other;
339  if(not name.empty()) {
340  _variables.at(varId).name = name;
341  }
342 
343  auto pvarDec = boost::make_shared<AccessorDecorator<T>>(
344  *this, other, direction == DecoratorType::READ, direction == DecoratorType::WRITE, varId, varId);
345 
346  return pvarDec;
347  }

355  template<typename UserType>
356  TestableMode::AccessorDecorator<UserType>::AccessorDecorator(detail::TestableMode& testableMode,
357  boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> accessor, bool handleRead, bool handleWrite,
358  size_t variableIdRead, size_t variableIdWrite)
359  : ChimeraTK::NDRegisterAccessorDecorator<UserType>(accessor), _handleRead(handleRead), _handleWrite(handleWrite),
360  _variableIdRead(variableIdRead), _variableIdWrite(variableIdWrite), _testableMode(testableMode) {
361  assert(_variableIdRead != 0);
362  assert(_variableIdWrite != 0);
363 
364  // if receiving end, register for testable mode (stall detection)
365  if(this->isReadable() && handleRead) {
366  _testableMode._variables[_variableIdRead].processVariable = accessor;
367  assert(accessor->getAccessModeFlags().has(AccessMode::wait_for_new_data));
368  }
369 
370  // if this decorating a bidirectional process variable, set the
371  // valueRejectCallback
372  auto bidir = boost::dynamic_pointer_cast<BidirectionalProcessArray<UserType>>(accessor);
373  if(bidir) {
374  bidir->setValueRejectCallback([this] { decrementCounter(); });
375  }
376  else {
377  assert(!(handleRead && handleWrite));
378  }
379  }

431  template<typename UserType>
432  void TestableMode::AccessorDecorator<UserType>::obtainLockAndDecrementCounter(bool hasNewData) {
433  if(!_testableMode.testLock()) {
434  _testableMode.lock("doReadTransfer " + this->getName());
435  }
436  if(!hasNewData) {
437  return;
438  }
439  auto& variable = _testableMode._variables.at(_variableIdRead);
440  if(variable.counter > 0) {
441  assert(_testableMode._counter > 0);
442  --_testableMode._counter;
443  --variable.counter;
444  if(_testableMode._enableDebug) {
445  logger(Logger::Severity::debug, "TestableMode")
446  << "TestableModeAccessorDecorator[name='" << this->getName() << "', id=" << _variableIdRead
447  << "]: testableMode.counter decreased, now at value " << _testableMode._counter << " / "
448  << variable.counter;
449  }
450  }
451  else {
452  if(_testableMode._enableDebug) {
453  logger(Logger::Severity::debug, "TestableMode")
454  << "TestableModeAccessorDecorator[name='" << this->getName() << "', id=" << _variableIdRead
455  << "]: testableMode.counter NOT decreased, was already at value " << _testableMode._counter << " / "
456  << variable.counter << "\n"
457  << variable.name;
458  }
459  }
460  }

480  template<typename UserType>
481  bool TestableMode::AccessorDecorator<UserType>::accountForWriteOperation(
482  const std::function<bool(void)>& writeOperation) {
483  bool dataLost = false;
484  if(!_testableMode.testLock()) {
485  // may happen if first write in thread is done before first blocking read
486  _testableMode.lock("write " + this->getName());
487  }
488 
489  dataLost = writeOperation();
490 
491  if(!dataLost) {
492  ++_testableMode._counter;
493  ++_testableMode._variables.at(_variableIdWrite).counter;
494  if(_testableMode._enableDebug) {
495  logger(Logger::Severity::debug, "TestableMode")
496  << "TestableModeAccessorDecorator::write[name='" << this->getName() << "', id=" << _variableIdWrite
497  << "]: testableMode.counter increased, now at value " << _testableMode._counter;
498  }
499  }
500  else {
501  if(_testableMode._enableDebug) {
502  logger(Logger::Severity::debug, "TestableMode")
503  << "TestableModeAccessorDecorator::write[name='" << this->getName() << "', id=" << _variableIdWrite
504  << "]: testableMode.counter not increased due to lost data";
505  }
506  }
507  return dataLost;
508  }
