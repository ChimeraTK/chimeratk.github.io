43  std::string escapeName(const std::string& name, bool allowDotsAndSlashes) {
44  std::string myLegalChars(legalChars);
45 
46  // remove underscore from legal chars, as we use it as an escape character
47  assert(myLegalChars[myLegalChars.size() - 1] == '_');
48  myLegalChars.resize(myLegalChars.size() - 1);
49 
50  if(allowDotsAndSlashes) {
51  myLegalChars += "./";
52  }
53 
54  std::string name_stripped;
55  size_t i;
56  size_t iLast = 0;
57  while((i = name.find_first_not_of(myLegalChars, iLast)) != std::string::npos) {
58  name_stripped += name.substr(iLast, i - iLast);
59  name_stripped += (boost::format("_%03d") % int(name[i])).str();
60  iLast = i + 1;
61  }
62  name_stripped += name.substr(iLast);
63 
64  return name_stripped;
65  }

69  std::string unescapeName(const std::string& name_stripped) {
70  std::string name;
71  size_t i;
72  size_t iLast = 0;
73  while((i = name_stripped.find('_', iLast)) != std::string::npos) {
74  name += name_stripped.substr(iLast, i - iLast);
75  char code = char(std::stoi(name_stripped.substr(i + 1, 3)));
76  name += code;
77  iLast = i + 4;
78  }
79  name += name_stripped.substr(iLast);
80 
81  return name;
