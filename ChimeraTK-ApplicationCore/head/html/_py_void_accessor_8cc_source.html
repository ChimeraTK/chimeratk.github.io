44 void PyVoidAccessor::bind(py::module& m) {
45 // strictly speaking, py::nodelete is not necessary since we hand out instances only via factory function,
46 // but leave it here for consistentcy
47 py::class_<PyVoidAccessor, PyTransferElementBase, std::unique_ptr<PyVoidAccessor, py::nodelete>> scalaracc(
48 m, "VoidAccessor", py::buffer_protocol());
49 scalaracc.def(py::init<>())
50 .def("read", &PyVoidAccessor::read,
51 "Read the data from the device.\n\nIf AccessMode::wait_for_new_data was set, this function will block "
52 "until new data has arrived. Otherwise it still might block for a short time until the data transfer was "
53 "complete.")
54 .def("readNonBlocking", &PyVoidAccessor::readNonBlocking,
55 "Read the next value, if available in the input buffer.\n\nIf AccessMode::wait_for_new_data was set, this "
56 "function returns immediately and the return value indicated if a new value was available (true) or not "
57 "(false).\n\nIf AccessMode::wait_for_new_data was not set, this function is identical to read(), which "
58 "will still return quickly. Depending on the actual transfer implementation, the backend might need to "
59 "transfer data to obtain the current value before returning. Also this function is not guaranteed to be "
60 "lock free. The return value will be always true in this mode.")
61 .def("readLatest", &PyVoidAccessor::readLatest,
62 "Read the latest value, discarding any other update since the last read if present.\n\nOtherwise this "
63 "function is identical to readNonBlocking(), i.e. it will never wait for new values and it will return "
64 "whether a new value was available if AccessMode::wait_for_new_data is set.")
65 .def("write", &PyVoidAccessor::write,
66 "Write the data to device.\n\nThe return value is true, old data was lost on the write transfer (e.g. due "
67 "to an buffer overflow). In case of an unbuffered write transfer, the return value will always be false.")
68 .def("writeDestructively", &PyVoidAccessor::writeDestructively,
69 "Just like write(), but allows the implementation to destroy the content of the user buffer in the "
70 "process.\n\nThis is an optional optimisation, hence there is a default implementation which just calls "
71 "the normal doWriteTransfer(). In any case, the application must expect the user buffer of the "
72 "TransferElement to contain undefined data after calling this function.")
73 .def("getName", &PyVoidAccessor::getName, "Returns the name that identifies the process variable.")
74 .def("getUnit", &PyVoidAccessor::getUnit,
75 "Returns the engineering unit.\n\nIf none was specified, it will default to ' n./ a.'")
76 .def("getDescription", &PyVoidAccessor::getDescription, "Returns the description of this variable/register.")
77 .def("getValueType", &PyVoidAccessor::getValueType,
78 "Returns the std::type_info for the value type of this transfer element.\n\nThis can be used to determine "
79 "the type at runtime.")
80 .def("getVersionNumber", &PyVoidAccessor::getVersionNumber,
81 "Returns the version number that is associated with the last transfer (i.e. last read or write)")
82 .def("isReadOnly", &PyVoidAccessor::isReadOnly,
83 "Check if transfer element is read only, i.e. it is readable but not writeable.")
84 .def("isReadable", &PyVoidAccessor::isReadable, "Check if transfer element is readable.")
85 .def("isWriteable", &PyVoidAccessor::isWriteable, "Check if transfer element is writeable.")
86 .def("getId", &PyVoidAccessor::getId,
87 "Obtain unique ID for the actual implementation of this TransferElement.\n\nThis means that e.g. two "
88 "instances of VoidRegisterAccessor created by the same call to Device::getVoidRegisterAccessor() (e.g. "
89 "by copying the accessor to another using NDRegisterAccessorBridge::replace()) will have the same ID, "
90 "while two instances obtained by to difference calls to Device::getVoidRegisterAccessor() will have a "
91 "different ID even when accessing the very same register.")
92 .def("dataValidity", &PyVoidAccessor::dataValidity,
93 "Return current validity of the data.\n\nWill always return DataValidity.ok if the backend does not "
94 "support it")
95 .def("__repr__", &PyVoidAccessor::repr);
96
100 m.def(
101 "VoidInput",
102 [](VariableGroup& owner, const std::string& name, const std::string& description) {
103 return dynamic_cast<PyOwningObject&>(owner).make_child<PyVoidAccessor>(
104 VoidTypeTag<VoidInput>(), &owner, name, description);
105 },
106 py::return_value_policy::reference, "");
107

116 },
117 py::return_value_policy::reference, "");
