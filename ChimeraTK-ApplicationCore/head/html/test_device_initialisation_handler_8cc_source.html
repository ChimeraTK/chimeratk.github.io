60 BOOST_AUTO_TEST_CASE(testBasicInitialisation) {
61 std::cout << "testBasicInitialisation" << std::endl;
62 TestApplication app;
63
64 var1 = 0;
65 var2 = 0;
66 var3 = 0;
67
68 ctk::TestFacility test{app};
69 test.runApplication();
70 ctk::Device dummy;
71 dummy.open(deviceCDD.data());
72
73 // ********************************************************
74 // REQUIRED TEST 1: After opening the device is initialised
75 // ********************************************************
76 BOOST_CHECK_EQUAL(var1, 42);
77
78 var1 = 0;
79
80 // check that accessing an exception triggers a reconnection with re-initialisation
81 auto dummyBackend = boost::dynamic_pointer_cast<ctk::ExceptionDummy>(
82 ctk::BackendFactory::getInstance().createBackend(deviceCDD.data()));
83 dummyBackend->throwExceptionWrite = true;
84
85 auto reg2_cs = test.getScalar<int32_t>("/REG2");
86 reg2_cs = 19;
87 reg2_cs.write();
88 test.stepApplication(false);
89
90 BOOST_CHECK_EQUAL(var2, 0);
91 BOOST_CHECK_EQUAL(var1, 0);
92 dummyBackend->throwExceptionWrite = false; // now the device should work again and be re-initialised
93
94 reg2_cs = 20;
95 reg2_cs.write();
96 test.stepApplication();
97
98 BOOST_CHECK_EQUAL(dummy.read<int32_t>("/REG2"), 20);
99
100 // ****************************************************************
101 // REQUIRED TEST 2: After an exception the device is re-initialised
102 // ****************************************************************
103 BOOST_CHECK_EQUAL(var1, 42);
104 }

160 BOOST_AUTO_TEST_CASE(testInitialisationException) {
161 std::cout << "testInitialisationException" << std::endl;
162
163 var1 = 0;
164 var2 = 0;
165 var3 = 0;
166
167 throwInInitialisation = true;
168 TestApplication app;
169
170 app.dev.addInitialisationHandler(&initialiseReg2);
171 app.dev.addInitialisationHandler(&initialiseReg3);
172 // app.dev.connectTo(app.cs);
173 ctk::TestFacility test(app, false); // test facility without testable mode
174 ctk::Device dummy;
175 dummy.open(deviceCDD.data());
176
177 // We cannot use runApplication because the DeviceModule leaves the testable mode without variables in the queue,
178 // but has not finished error handling yet. In this special case we cannot make the programme continue, because
179 // stepApplication only works if the queues are not empty. We have to work with timeouts here (until someone comes
180 // up with a better idea)
181 app.run();
182 // app.dumpConnections();
183
184 CHECK_EQUAL_TIMEOUT(test.readScalar<int32_t>(ctk::RegisterPath("/Devices") /
185 ctk::Utilities::escapeName(deviceCDD.data(), false) / "status"),
186 1, 30000);
187 CHECK_EQUAL_TIMEOUT(test.readScalar<std::string>(ctk::RegisterPath("/Devices") /
188 ctk::Utilities::escapeName(deviceCDD.data(), false) / "status_message"),
189 exceptionMessage, 10000);
190
191 // Check that the execution of init handlers was stopped after the exception:
192 // initialiseReg2 and initialiseReg3 were not executed. As we already checked with timeout that the
193 // initialisation error has been reported, we know that the data was written to the device and don't need the timeout here.
194
195 BOOST_CHECK_EQUAL(var1, 42);
196 BOOST_CHECK_EQUAL(var2, 0);
197 BOOST_CHECK_EQUAL(var3, 0);
198
199 // recover the error
200 throwInInitialisation = false;
201
202 // wait until the device is reported to be OK again (check with timeout),
203 // then check the initialisation (again, no extra timeout needed because of the logic:
204 // success is only reported after successful init).
205 CHECK_EQUAL_TIMEOUT(test.readScalar<int32_t>(ctk::RegisterPath("/Devices") /
206 ctk::Utilities::escapeName(deviceCDD.data(), false) / "status"),
207 0, 10000);
208 // We use the macro here for convenience, it's a test, speed should not matter
209 // NOLINTNEXTLINE(readability-container-size-empty)
210 CHECK_EQUAL_TIMEOUT(test.readScalar<std::string>(ctk::RegisterPath("/Devices") /
211 ctk::Utilities::escapeName(deviceCDD.data(), false) / "status_message"),
212 "", 10000);
213
214 // initialisation should be correct now
215 BOOST_CHECK_EQUAL(var1, 42);
216 BOOST_CHECK_EQUAL(var2, 47);
217 BOOST_CHECK_EQUAL(var3, 52);
218
219 // now check that the initialisation error is also reportet when recovering
220 // Prepare registers to be initialised
221 var1 = 12;
222 var2 = 13;
223 var3 = 14;
224
225 // Make initialisation fail when executed, and then cause an error condition
226 throwInInitialisation = true;
227 auto dummyBackend = boost::dynamic_pointer_cast<ctk::ExceptionDummy>(
228 ctk::BackendFactory::getInstance().createBackend(deviceCDD.data()));
229 dummyBackend->throwExceptionWrite = true;
230
231 auto reg4_cs = test.getScalar<int32_t>("/REG4");
232 reg4_cs = 20;
233 reg4_cs.write();
234
235 CHECK_EQUAL_TIMEOUT(test.readScalar<int32_t>(ctk::RegisterPath("/Devices") /
236 ctk::Utilities::escapeName(deviceCDD.data(), false) / "status"),
237 1, 10000);
238 // First we see the message from the failing write
239 BOOST_CHECK(!test
240 .readScalar<std::string>(
241 ctk::RegisterPath("/Devices") / ctk::Utilities::escapeName(deviceCDD.data(), false) / "status_message")
242 .empty());
243 dummyBackend->throwExceptionWrite = false;
244 // Afterwards we see a message from the failing initialisation (which we can now distinguish from the original write
245 // exception because write does not throw any more)
246 CHECK_EQUAL_TIMEOUT(test.readScalar<std::string>(ctk::RegisterPath("/Devices") /
247 ctk::Utilities::escapeName(deviceCDD.data(), false) / "status_message"),
248 exceptionMessage, 10000);
249
250 // Now fix the initialisation error and check that the device comes up.
251 throwInInitialisation = false;
252 // Wait until the device is OK again
253 CHECK_EQUAL_TIMEOUT(test.readScalar<int32_t>(ctk::RegisterPath("/Devices") /
254 ctk::Utilities::escapeName(deviceCDD.data(), false) / "status"),
255 0, 10000);
256
257 // We use the macro here for convenience, it's a test, speed should not matter
258 // NOLINTNEXTLINE(readability-container-size-empty)
259 CHECK_EQUAL_TIMEOUT(test.readScalar<std::string>(ctk::RegisterPath("/Devices") /
260 ctk::Utilities::escapeName(deviceCDD.data(), false) / "status_message"),
261 "", 10000);
262 // Finally check that the 20 arrives on the device
263 CHECK_EQUAL_TIMEOUT(dummy.read<int32_t>("/REG4"), 20, 10000);
264 }

272 BOOST_AUTO_TEST_CASE(TestDeviceClosedInInitHandler) {
273 // Check that the device has been closed when the init handler is called.
274 std::cout << "TestDeviceClosedInInitHandler" << std::endl;
275
276 TestApplication app;
277 // Cache the opened state in the init handler in a variable. BOOST_CHECK is not threat safe and
278 // cannot directly be used in the handler.
279 bool isOpenedInInitHandler{
280 true}; // We expect false, so we set the starting value to true to know the test is sensitive.
281 app.dev.addInitialisationHandler([&](ctk::Device& d) { isOpenedInInitHandler = d.isOpened(); });
282
283 ctk::TestFacility testFacility(app);
284 testFacility.runApplication();
285 // The testFacility in testable mode guarantees that the device has been opened at this point. So we know the init
286 // handler with the test has been run at this point.
287 BOOST_CHECK(!isOpenedInInitHandler);
288 }
