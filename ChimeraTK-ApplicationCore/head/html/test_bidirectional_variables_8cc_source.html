27 /* Module which converts the input data from inches to centimetres - and the
28  * other way round for the return channel. In case of the return channel, the
29  * data is rounded downwards to integer inches and sent again forward. */
30 struct ModuleA : public ctk::ApplicationModule {
31  using ctk::ApplicationModule::ApplicationModule;
32 
33  ctk::ScalarPushInputWB<int> var1{this, "var1", "inches", "A length, for some reason rounded to integer"};
34  ctk::ScalarOutputPushRB<double> var2{this, "var2", "centimetres", "Same length converted to centimetres"};
35 
36  void prepare() override {
37  incrementDataFaultCounter(); // force all outputs to invalid
38  writeAll(); // write initial values
39  decrementDataFaultCounter(); // validity according to input validity
40  }
41 
42  void mainLoop() override {
43  auto group = readAnyGroup();
44  while(true) {
45  auto var = group.readAny();
46  if(var == var2.getId()) {
47  var1 = std::floor(var2 / 2.54);
48  var1.write();
49  }
50  var2 = var1 * 2.54;
51  var2.write();
52  }
53  }
54 };

58 /* Module which limits a value to stay below a maximum value. */
59 struct ModuleB : public ctk::ApplicationModule {
60  using ctk::ApplicationModule::ApplicationModule;
61 
62  ctk::ScalarPushInputWB<double> var2{this, "var2", "centimetres", "Some length, confined to a configurable range"};
63  ctk::ScalarPushInput<double> max{this, "max", "centimetres", "Maximum length"};
64  ctk::ScalarOutput<double> var3{this, "var3", "centimetres", "The limited length"};
65 
66  void prepare() override {
67  incrementDataFaultCounter(); // force all outputs to invalid
68  writeAll(); // write initial values
69  decrementDataFaultCounter(); // validity according to input validity
70  }
71 
72  void mainLoop() override {
73  auto group = readAnyGroup();
74  while(true) {
75  auto var = group.readAny();
76  bool write = var == var2.getId();
77  if(var2 > max) {
78  var2 = static_cast<double>(max);
79  var2.write();
80  write = true;
81  }
82  if(write) { // write only if var2 was received or the value was changed
83  // due to a reduced limit
84  var3 = static_cast<double>(var2);
85  var3.write();
86  }
87  }
88  }
89 };

114 struct ModuleFunnel : public ctk::ApplicationModule {
115  using ApplicationModule::ApplicationModule;
116 
117  ctk::ScalarPushInputWB<int> var1{this, "/var1", "", "Input with funneled return channel"};
118  ctk::ScalarOutput<int> var1out{this, "var1out", "", ""};
119  ctk::ScalarPushInput<int> var1in{this, "var1in", "", ""};
120 
121  void mainLoop() override {
122  // This module essentially splits up the forward and return channel of the PushInputWB "var1".
123  auto group = readAnyGroup();
124 
125  auto change = var1.getId();
126 
127  while(true) {
128  if(change == var1.getId()) {
129  var1out.setAndWrite(var1);
130  }
131  else if(change == var1in.getId()) {
132  var1.setAndWrite(var1in);
133  }
134 
135  change = group.readAny();
136  }
137  }
138 };

258  auto var3 = test.getScalar<double>("var3");
259  auto max = test.getScalar<double>("max");
260  test.runApplication();
261 
262  // set maximum in B, so that var1=49 is still below maximum but var2=50 is
263  // already above and rounding in ModuleB will change the value again
264  max = 49.5 * 2.54;
265  max.write();
266  test.stepApplication();
267 
268  // inject value which does not get limited
269  var1 = 49;
270  var1.write();
271  test.stepApplication();
272  var1_copied.read();
273  var2.read();
274  var3.read();
275  BOOST_CHECK_EQUAL(static_cast<int>(var1_copied), 49);
276  BOOST_CHECK_CLOSE(static_cast<double>(var2), 49 * 2.54, 0.001);
277  BOOST_CHECK_CLOSE(static_cast<double>(var3), 49 * 2.54, 0.001);
278  BOOST_CHECK(var1.readNonBlocking() == false); // nothing was sent through the return channel
279  BOOST_CHECK(var1_copied.readLatest() == false);
280  BOOST_CHECK(var2.readNonBlocking() == false);
281  BOOST_CHECK(var3.readNonBlocking() == false);
282 
283  // inject value which gets limited
284  var1 = 50;
285  var1.write();
286  test.stepApplication();
287  var1.read();
288  BOOST_CHECK_EQUAL(static_cast<int>(var1), 49);
289  var1_copied.read();
290  BOOST_CHECK_EQUAL(static_cast<int>(var1_copied), 50);
291  var1_copied.read();
292  BOOST_CHECK_EQUAL(static_cast<int>(var1_copied), 49);
293  var2.read();
294  BOOST_CHECK_CLOSE(static_cast<double>(var2), 50 * 2.54, 0.001);
295  var2.read();
296  BOOST_CHECK_CLOSE(static_cast<double>(var2), 49.5 * 2.54, 0.001);
297  var2.read();
298  BOOST_CHECK_CLOSE(static_cast<double>(var2), 49 * 2.54, 0.001);
299  var3.read();
300  BOOST_CHECK_CLOSE(static_cast<double>(var3), 49.5 * 2.54, 0.001);
301  var3.read();
302  BOOST_CHECK_CLOSE(static_cast<double>(var3), 49 * 2.54, 0.001);
303  BOOST_CHECK(var1.readNonBlocking() == false);
304  BOOST_CHECK(var1_copied.readLatest() == false);
305  BOOST_CHECK(var2.readNonBlocking() == false);
306  BOOST_CHECK(var3.readNonBlocking() == false);
307 
308  // change the limit so the current value gets changed
309  max = 48.5 * 2.54;
310  max.write();
311  test.stepApplication();
312  var1.read();
313  BOOST_CHECK_EQUAL(static_cast<int>(var1), 48);
314  var1_copied.read();
315  BOOST_CHECK_EQUAL(static_cast<int>(var1_copied), 48);
316  var2.read();
317  BOOST_CHECK_CLOSE(static_cast<double>(var2), 48.5 * 2.54, 0.001);
318  var2.read();
319  BOOST_CHECK_CLOSE(static_cast<double>(var2), 48 * 2.54, 0.001);
320  var3.read();
321  BOOST_CHECK_CLOSE(static_cast<double>(var3), 48.5 * 2.54, 0.001);
322  var3.read();
323  BOOST_CHECK_CLOSE(static_cast<double>(var3), 48 * 2.54, 0.001);
324  BOOST_CHECK(var1.readNonBlocking() == false);
325  BOOST_CHECK(var1_copied.readLatest() == false);
326  BOOST_CHECK(var2.readNonBlocking() == false);
327  BOOST_CHECK(var3.readNonBlocking() == false);
328 
329  // Run the following tests a couple of times, as they are testing for the
330  // absence of race conditions. This makes it more likely to find failures in a
331  // single run of the test
332  for(size_t i = 0; i < 10; ++i) {
333  // feed in some default values (so the tests can be executed multiple times
334  // in a row)
335  max = 48.5 * 2.54;
336  max.write();
337  test.stepApplication();
338  var1 = 50;
339  var1.write();
340  test.stepApplication();
341  var1.readLatest(); // empty the queues
342  var1_copied.readLatest();
343  var2.readLatest();
344  var3.readLatest();
345  BOOST_CHECK_EQUAL(static_cast<int>(var1), 48);
346  BOOST_CHECK_EQUAL(static_cast<int>(var1_copied), 48);
347  BOOST_CHECK_CLOSE(static_cast<double>(var2), 48 * 2.54, 0.001);
348  BOOST_CHECK_CLOSE(static_cast<double>(var3), 48 * 2.54, 0.001);
349  BOOST_CHECK(var1.readNonBlocking() == false);
350  BOOST_CHECK(var1_copied.readLatest() == false);
351  BOOST_CHECK(var2.readNonBlocking() == false);
352  BOOST_CHECK(var3.readNonBlocking() == false);
353 
354  // concurrent change of value and limit. Note: The final result must be
355  // deterministic, but which values are seen in between is subject to race
356  // conditions between the two concurrent updates. Thus we are using
357  // readLatest() in some cases here.
358  var1 = 30;
359  max = 25.5 * 2.54;
360  var1.write();
361  max.write();
362  test.stepApplication();
363  var1.read();
364  BOOST_CHECK_EQUAL(static_cast<int>(var1), 25);
365  var1_copied.read();
366  BOOST_CHECK_EQUAL(static_cast<int>(var1_copied), 30);
367  BOOST_CHECK(var1_copied.readLatest() == true);
368  BOOST_CHECK_EQUAL(static_cast<int>(var1_copied), 25);
369  BOOST_CHECK(var2.readLatest() == true);
370  BOOST_CHECK_CLOSE(static_cast<double>(var2), 25 * 2.54, 0.001);
371  BOOST_CHECK(var3.readLatest() == true);
372  BOOST_CHECK_CLOSE(static_cast<double>(var3), 25 * 2.54, 0.001);
373  BOOST_CHECK(var1.readNonBlocking() == false);
374  BOOST_CHECK(var1_copied.readLatest() == false);
375  BOOST_CHECK(var2.readNonBlocking() == false);
376  BOOST_CHECK(var3.readNonBlocking() == false);
377 
378  // concurrent change of value and limit - other order than before
379  var1 = 15;
380  max = 20.5 * 2.54;
381  max.write();
382  var1.write();
383  test.stepApplication();
384  var1_copied.read();
385  BOOST_CHECK_EQUAL(static_cast<int>(var1_copied), 15);
386  BOOST_CHECK(var2.readLatest() == true);
387  BOOST_CHECK_CLOSE(static_cast<double>(var2), 15 * 2.54, 0.001);
388  BOOST_CHECK(var3.readLatest() == true);

390  BOOST_CHECK(var1.readNonBlocking() == false);
391  BOOST_CHECK(var1_copied.readLatest() == false);
392  BOOST_CHECK(var2.readNonBlocking() == false);
393  BOOST_CHECK(var3.readNonBlocking() == false);
