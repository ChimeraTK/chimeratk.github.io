76 template<typename Derived>
77 InversionOfControlAccessor<Derived>::~InversionOfControlAccessor() {
78 if(getOwner() != nullptr) {
79 if(static_cast<Derived*>(this)->_impl != nullptr) {
80 auto* entity = getOwner();
81
82 if(entity != nullptr) {
83 auto* owner = dynamic_cast<Module*>(entity);
84 while(owner->getOwner() != nullptr) {
85 owner = dynamic_cast<Module*>(owner->getOwner());
86 }
87
88 auto* application = dynamic_cast<Application*>(owner);
89 assert(application != nullptr);
90
91 if(application->getLifeCycleState() == LifeCycleState::run) {
92 try {
93 throw ChimeraTK::logic_error(
94 "Variable has been destroyed with active connections while application is still running");
95 }
96 catch(ChimeraTK::logic_error&) {
97 std::terminate();
98 }
99 }
100 }
101 }
102 getOwner()->unregisterAccessor(_node);
103 }
104 if(getModel().isValid()) {
105 try {
106 getModel().removeNode(_node);
107 }
108 catch(ChimeraTK::logic_error& e) {
109 std::cerr << "ChimeraTK::logic_error caught: " << e.what() << std::endl;
110 std::terminate();
111 }
112 }
113 }

149 template<typename Derived>
150 void InversionOfControlAccessor<Derived>::replace(Derived&& other) {
151 if(static_cast<Derived*>(this)->_impl != nullptr || other._impl != nullptr) {
152 try {
153 throw ChimeraTK::logic_error(
154 "Variable has been destroyed with active connections while application is still running");
155 }
156 catch(ChimeraTK::logic_error& ex) {
157 std::terminate();
158 }
159 }
160
161 // remove accessor from owning module
162 if(getOwner() != nullptr) {
163 getOwner()->unregisterAccessor(_node);
164 }
165
166 // remove node from model
167 if(getModel().isValid()) {
168 getModel().removeNode(_node);
169 }
170
171 // transfer the node
172 _node = std::move(other._node);
173 other._node = VariableNetworkNode(); // Make sure the destructor of other sees an invalid node
174
175 // update the app accessor pointer in the node
176 if(_node.getType() == NodeType::Application) {
177 _node.setAppAccessorPointer(static_cast<Derived*>(this));
178 }
179 else {
180 assert(_node.getType() == NodeType::invalid);
181 }
182 // Note: the accessor is registered by the VariableNetworkNode, so we don't have to re-register.
183 }

202 template<typename Derived>
203 InversionOfControlAccessor<Derived>::InversionOfControlAccessor(Module* owner, const std::string& name,
204 VariableDirection direction, std::string unit, size_t nElements, UpdateMode mode, const std::string& description,
205 const std::type_info* valueType, const std::unordered_set<std::string>& tags)
206 : _node(owner, static_cast<Derived*>(this), ChimeraTK::Utilities::raiseIftrailingSlash(name, false), direction, unit,
207 nElements, mode, completeDescription(owner, description), valueType, tags) {
208 static_assert(std::is_base_of<InversionOfControlAccessor<Derived>, Derived>::value,
209 "InversionOfControlAccessor<> must be used in a curiously recurring template pattern!");
210
212 auto addToOnwer = [&](auto& owner_casted) {
213 auto model = owner_casted.getModel();
214 if(!model.isValid()) {
215 // this happens e.g. for default-constructed owners and their sub-modules
216 return;
217 }
218 auto neighbourDir = model.visit(
219 Model::returnDirectory, Model::getNeighbourDirectory, Model::returnFirstHit(Model::DirectoryProxy{}));
220
221 auto dir = neighbourDir.addDirectoryRecursive(Utilities::getPathName(name));
222 auto var = dir.addVariable(Utilities::getUnqualifiedName(name));
223
224 model.addVariable(var, _node);
225 };
226 auto* owner_am = dynamic_cast<ApplicationModule*>(owner);
227 auto* owner_vg = dynamic_cast<VariableGroup*>(owner);
228 if(owner_am) {
229 addToOnwer(*owner_am);
230 }
231 else if(owner_vg) {
232 addToOnwer(*owner_vg);
233 }
234 else {
235 throw ChimeraTK::logic_error("Trying to add " + name + " to " + owner->getQualifiedName() +
236 " which is neither an ApplicationModule nor a VariableGroup, but a " +
237 boost::core::demangled_name(typeid(owner)));
238 }
239
240 owner->registerAccessor(_node);
241 }
