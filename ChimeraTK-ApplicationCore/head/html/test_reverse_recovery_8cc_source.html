47BOOST_AUTO_TEST_CASE(testDirectThreadedFanOutWithReturn) {
48 std::cout << "testDirectThreadedFanOutWithReturn" << std::endl;
49
50 ctk::BackendFactory::getInstance().setDMapFilePath("testTagged.dmap");
51
52 TestApplication app;
53 ctk::DeviceModule devModule{&app, "taggedDevice", "/trigger"};
54
55 std::atomic<bool> up{false};
56
57 app.mod.doMainLoop = [&]() {
58 up = true;
59 up.notify_one();
60 };
61
62 ctk::TestFacility test(app, false);
63
64 ctk::Device dev;
65 dev.open("baseDevice");
66
67 // Initialize the device with some values
68 dev.write<int32_t>("/readWrite", 4);
69 dev.write<int32_t>("/writeOnlyRB.DUMMY_WRITEABLE", 8);
70 dev.write<int32_t>("/secondReadWrite", 16);
71
72 // Set initial values for the variables
73 test.setScalarDefault<int32_t>("/taggedReadWrite", 12);
74 test.setScalarDefault<int32_t>("/taggedWriteOnly", 24);
75 test.setScalarDefault<int32_t>("/untagged", 36);
76
77 test.runApplication();
78
79 // Wait for the device to become ready
80 CHECK_EQUAL_TIMEOUT(test.readScalar<int32_t>("/Devices/taggedDevice/status"), 0, 1000);
81
82 up.wait(false);
83
84 auto taggedReadWriteCs = test.getScalar<int32_t>("/taggedReadWrite");
85 auto taggedWriteOnlyCs = test.getScalar<int32_t>("/taggedWriteOnly");
86 auto untagged = test.getScalar<int32_t>("/untagged");
87
88 // Check that the values are still on the values we have written explicitly
89 // into the device, and not the initial values we configured above
90 BOOST_TEST(dev.read<int32_t>("/readWrite") == 4);
91 BOOST_TEST(dev.read<int32_t>("/writeOnlyRB") == 8);
92
93 // Check that instead those values have been propagated to the CS (where applicable)
94 CHECK_EQUAL_TIMEOUT((taggedReadWriteCs.readLatest(), int(taggedReadWriteCs)), 4, 2000);
95
96 // The untagged register should have received the initial value from the CS
97 BOOST_TEST(dev.read<int32_t>("/secondReadWrite") == 36);
98
99 // Just do normal operations
100 taggedReadWriteCs.setAndWrite(48);
101 taggedWriteOnlyCs.setAndWrite(96);
102 untagged.setAndWrite(128);
103
104 CHECK_EQUAL_TIMEOUT(dev.read<int32_t>("/readWrite"), 48, 2000);
105 CHECK_EQUAL_TIMEOUT(dev.read<int32_t>("/writeOnlyRB"), 96, 2000);
106 CHECK_EQUAL_TIMEOUT(dev.read<int32_t>("/secondReadWrite"), 128, 2000);
107
108 dev.write<int32_t>("/readWrite", 3);
109 dev.write<int32_t>("/writeOnlyRB.DUMMY_WRITEABLE", 7);
110 dev.write<int32_t>("/secondReadWrite", 15);
111 devModule.reportException("Trigger device recovery");
112
113 CHECK_EQUAL_TIMEOUT(test.readScalar<int32_t>("/Devices/taggedDevice/status"), 1, 1000);
114 // Wait for ApplicationCore to recover
115 CHECK_EQUAL_TIMEOUT(test.readScalar<int32_t>("/Devices/taggedDevice/status"), 0, 1000);
116
117 // The two tagged registers should keep their values, the untagged register should receive the value written before
118 CHECK_EQUAL_TIMEOUT(dev.read<int32_t>("/readWrite"), 3, 1000);
119 CHECK_EQUAL_TIMEOUT(dev.read<int32_t>("/writeOnlyRB"), 7, 1000);
120 CHECK_EQUAL_TIMEOUT(dev.read<int32_t>("/secondReadWrite"), 128, 1000);
121
122 // The read-write register should have propagated its value to the CS
123 CHECK_EQUAL_TIMEOUT((taggedReadWriteCs.readLatest(), int32_t(taggedReadWriteCs)), 3, 2000);
124
125 app.shutdown();
126}

142 TestApplication app;

177 app.shutdown();
