37 ApplicationModule& ApplicationModule::operator=(ApplicationModule&& other) noexcept {
38 assert(!_moduleThread.joinable()); // if the thread is already running, moving is no longer allowed!
39
40 // Keep the model as is (except from updating the pointers to the C++ objects). To do so, we have to hide it from
41 // unregisterModule() which is executed in Module::operator=(), because it would destroy the model.
42 Model::ApplicationModuleProxy model = std::move(other._model);
43 other._model = {};
44
45 VariableGroup::operator=(std::move(other));
46
47 if(model.isValid()) {
48 model.informMove(*this);
49 }
50 _model = model;
51
52 return *this;
53 }

65 void ApplicationModule::terminate() {
66 if(_moduleThread.joinable()) {
67 _moduleThread.interrupt();
68 // try joining the thread
69 while(!_moduleThread.try_join_for(boost::chrono::milliseconds(10))) {
70 // if thread is not yet joined, send interrupt() to all variables.
71 for(auto& var : getAccessorListRecursive()) {
72 auto el{var.getAppAccessorNoType().getHighLevelImplElement()};
73 el->interrupt();
74 }
75 // it may not suffice to send interrupt() once, as the exception might get
76 // overwritten in the queue, thus we repeat this until the thread was
77 // joined.
78 }
79 }
80 assert(!_moduleThread.joinable());
81 }

99 void ApplicationModule::mainLoopWrapper() {
100 Application::registerThread("AM_" + className()); // + getName());
101
102 // Acquire testable mode lock, so from this point on we are running only one user thread concurrently
103 Application::getInstance().getTestableMode().lock("start", true);
104
105 // Read all variables once to obtain the initial values from the devices and from the control system persistency
106 // layer. This is done in two steps, first for all poll-type variables and then for all push-types, because
107 // poll-type reads might trigger distribution of values to push-type variables via a ConsumingFanOut.
108 for(auto& variable : getAccessorListRecursive()) {
109 if(variable.getDirection().dir != VariableDirection::consuming) {
110 continue;
111 }
112 if(variable.getMode() == UpdateMode::poll) {
113 assert(!variable.getAppAccessorNoType().getHighLevelImplElement()->getAccessModeFlags().has(
114 AccessMode::wait_for_new_data));
115 Application::getInstance().getTestableMode().unlock("Initial value read for poll-type " + variable.getName());
116 Application::getInstance()._circularDependencyDetector.registerDependencyWait(variable);
117 variable.getAppAccessorNoType().read();
118 Application::getInstance()._circularDependencyDetector.unregisterDependencyWait(variable);
119 if(not Application::getInstance().getTestableMode().testLock()) {
120 // The lock may have already been acquired if the above read() goes to a ConsumingFanOut, which sends out
121 // the data to a slave decorated by a TestableModeAccessorDecorator. Hence we here must acquire the lock only
122 // if we do not have it.
123 Application::getInstance().getTestableMode().lock(
124 "Initial value read for poll-type " + variable.getName(), true);
125 }
126 }
127 }
128
129 for(auto& variable : getAccessorListRecursive()) {
130 // According to spec, readback values are not considered in the distribution of initial values
131 // However, if they are declared as providing reverse recovery, they have to be considered.
132 auto doNotSkipInIvDistribution = variable.getDirection().dir == VariableDirection::feeding &&
133 variable.getDirection().withReturn && variable.getTags().contains(ChimeraTK::SystemTags::reverseRecovery);
134
135 if(!doNotSkipInIvDistribution && variable.getDirection().dir != VariableDirection::consuming) {
136 continue;
137 }
138
139 if(variable.getMode() == UpdateMode::push) {
140 Application::getInstance().getTestableMode().unlock("Initial value read for push-type " + variable.getName());
141 Application::getInstance()._circularDependencyDetector.registerDependencyWait(variable);
142 // Will internally release and lock during the read, hence surround with lock/unlock
143 Application::getInstance().getTestableMode().lock(
144 "Initial value read for push-type " + variable.getName(), true);
145 variable.getAppAccessorNoType().read();
146 Application::getInstance().getTestableMode().unlock("Initial value read for push-type " + variable.getName());
147 Application::getInstance()._circularDependencyDetector.unregisterDependencyWait(variable);
148 Application::getInstance().getTestableMode().lock(
149 "Initial value read for push-type " + variable.getName(), true);
150 }
151 }
152
153 // We are holding the testable mode lock, so we are sure the mechanism will work now.
154 _testableModeReached = true;
155
156 // enter the main loop
157 mainLoop();
158 Application::getInstance().getTestableMode().unlock("terminate");
159 }

174 std::list<EntityOwner*> ApplicationModule::getInputModulesRecursively(std::list<EntityOwner*> startList) {
175 if(_recursionStopper.recursionDetected()) {
176 return startList;
177 }
178
179 // If this module is already in the list we found a circular dependency.
180 // Remember this for the next time the recursive scan calls this function
181 if(std::count(startList.begin(), startList.end(), this)) {
182 _recursionStopper.setRecursionDetected();
183 }
184
185 // Whether a circular dependency has been detected or not, we must loop all inputs and add this module to the list
186 // so the calling code sees the second instance and can also detect the circle. The reason why we have to scan all
187 // inputs even if a circle is detected is this:
188 // * A single input starts the scan by adding it's owning module. At this point not all inputs if that module are in
189 // the circular network.
190 // * When a circle is detected, it might only be one of multiple entangled circled. If we would break the recursion
191 // and not scan all the
192 // inputs this is sufficient to identify that the particular input is in a circle. But at this point we have to
193 // tell in which network it is and have to scan the complete network to calculate the correct hash value.
194
195 startList.push_back(this); // first we add this module to the start list. We will call all inputs with it.
196 std::list<EntityOwner*> returnList{
197 startList}; // prepare the return list. Deltas from the inputs will be added to it.
198 for(auto& accessor : this->getAccessorListRecursive()) {
199 // not consumed from network -> not an input, just continue
200 if(accessor.getDirection().dir != VariableDirection::consuming) {
201 continue;
202 }
203
204 // find the feeder in the network
205 auto proxy = accessor.getModel().visit(Model::returnApplicationModule, Model::keepApplicationModules,
206 Model::keepPvAccess, Model::adjacentInSearch, Model::returnFirstHit(Model::ApplicationModuleProxy{}));
207 if(!proxy.isValid()) {
208 continue;
209 }
210 auto& feedingModule = proxy.getApplicationModule();
211
212 auto thisInputsRecursiveModuleList = feedingModule.getInputModulesRecursively(startList);
213 // only add the modules that were added by the recursive search to the output list
214 assert(startList.size() <= thisInputsRecursiveModuleList.size());
215 auto copyStartIter = thisInputsRecursiveModuleList.begin();
216 std::advance(copyStartIter, startList.size());
217
218 returnList.insert(returnList.end(), copyStartIter, thisInputsRecursiveModuleList.end());
219 }
220 return returnList;
221 }

238 }

244 DataValidity ApplicationModule::getDataValidity() const {
245 if(_dataFaultCounter == 0) {
246 return DataValidity::ok;
247 }
248 if(_circularNetworkHash != 0) {
249 // In a circular dependency network, internal inputs are ignored.
250 // If all external inputs (including the ones from this module) are OK, the
251 // data validity is set to OK.
252 if(Application::getInstance()._circularNetworkInvalidityCounters[_circularNetworkHash] == 0) {
253 return DataValidity::ok;
254 }
255 }
256 // not a circular network or invalidity counter not 0 -> keep the faulty flag
257 return DataValidity::faulty;
258 }

266 if(!vg) {

268 // fails.
269 return;
270 }
271 _model.remove(*vg);
272 }
