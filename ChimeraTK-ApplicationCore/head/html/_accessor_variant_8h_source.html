35 /********************************************************************************************************************/
36 // Combined variant types that include all derived accessor types alongside the base accessor type.
37 //
38 // This avoids object slicing: when the variant stores (e.g.) a ScalarPollInput<UserType> instead of slicing
39 // it to ScalarAccessor<UserType>, std::visit dispatches to ScalarPollInput::read() (which calls readLatest())
40 // rather than the base ScalarAccessor::read(). The same applies to ArrayPollInput and the NoInitialValue
41 // variants.
42 //
43 // All types the factory functions might create must be included as alternatives in the variant.
44 /********************************************************************************************************************/
45
47 using ScalarAccessorVariant = typename detail::NestedVariant<UserTypeTemplateVariantNoVoid<ScalarAccessor>,
48 UserTypeTemplateVariantNoVoid<ScalarPollInput>, UserTypeTemplateVariantNoVoid<ScalarPushInput>,
