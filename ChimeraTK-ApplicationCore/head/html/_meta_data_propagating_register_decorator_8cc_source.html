12 template<typename T>
13 void MetaDataPropagatingRegisterDecorator<T>::doPostRead(TransferType type, bool hasNewData) {
14 NDRegisterAccessorDecorator<T, T>::doPostRead(type, hasNewData);
15
16 // update the version number
17 if(!_disableVersionNumberPropagation && _target->getAccessModeFlags().has(AccessMode::wait_for_new_data) &&
18 type == TransferType::read) {
19 _owner->setCurrentVersionNumber(this->getVersionNumber());
20 }
21
22 // Check if the data validity flag changed. If yes, propagate this information to the owning module and the application
23 if(_dataValidity != _lastValidity) {
24 if(_dataValidity == DataValidity::faulty) { // data validity changes to faulty
25 _owner->incrementDataFaultCounter();
26 // external input in a circular dependency network
27 if(_owner->getCircularNetworkHash() && !_isCircularInput) {
28 ++(Application::getInstance()._circularNetworkInvalidityCounters[_owner->getCircularNetworkHash()]);
29 }
30 }
31 else { // data validity changed to OK
32 _owner->decrementDataFaultCounter();
33 // external inpput in a circular dependency network
34 if(_owner->getCircularNetworkHash() && !_isCircularInput) {
35 --(Application::getInstance()._circularNetworkInvalidityCounters[_owner->getCircularNetworkHash()]);
36 }
37 }
38 _lastValidity = _dataValidity;
39 }
40 }

42 template<typename T>
43 void MetaDataPropagatingRegisterDecorator<T>::doPreWrite(TransferType type, VersionNumber versionNumber) {
44 // We cannot use NDRegisterAccessorDecorator<T> here because we need a different implementation of setting the
45 // target data validity. So we have a complete implemetation here.
46
47 if(_owner->getCircularNetworkHash() && _dataValidity != _lastValidity) {
48 // In circular dependency networks an output which actively has DataValidity::faulty set by the user logic is handled
49 // as if an external input was invalid -> increase or decrease the network's invalidity counter accordingly
50 if(_dataValidity == DataValidity::faulty) { // data validity changes to faulty
51 ++(Application::getInstance()._circularNetworkInvalidityCounters[_owner->getCircularNetworkHash()]);
52 }
53 else {
54 --(Application::getInstance()._circularNetworkInvalidityCounters[_owner->getCircularNetworkHash()]);
55 }
56 _lastValidity = _dataValidity;
57 }
58
59 // Now propagate the flag and the data to the target and perform the write
60 if(_dataValidity == DataValidity::faulty) { // the application has manualy set the validity to faulty
61 _target->setDataValidity(DataValidity::faulty);
62 }
63 else {
64 if(_direction.dir == VariableDirection::feeding && !_disableDataValidityPropagation) {
65 _target->setDataValidity(_owner->getDataValidity());
66 }
67 else {
68 // Do not propagate the owner's validity through the return channel. The user can still override this because
69 // of the override above
70 _target->setDataValidity(DataValidity::ok);
71 }
72 }
73
74 for(unsigned int i = 0; i < _target->getNumberOfChannels(); ++i) {
75 buffer_2D[i].swap(_target->accessChannel(i));
76 }
77 _target->preWrite(type, versionNumber);
78 }
