18 DeviceManager::DeviceManager(Application* application, const std::string& deviceAliasOrCDD)
19 : ApplicationModule(application, "/Devices/" + Utilities::escapeName(deviceAliasOrCDD, false), ""),
20 _device(deviceAliasOrCDD), _deviceAliasOrCDD(deviceAliasOrCDD), _owner{application} {
21 auto involvedBackends = _device.getInvolvedBackendIDs();
22 // Create a recovery group with barrier size 1.
23 _recoveryGroup = std::make_shared<RecoveryGroup>(involvedBackends, _owner);
24
25 // loop all already existing DeviceManagers and look for shared backends
26 int64_t recoveryGroupSize{1};
27 for(const auto& [alias, existingDeviceManager] : Application::getInstance().getDeviceManagerMap()) {
28 for(auto backendID : involvedBackends) {
29 if(existingDeviceManager->_recoveryGroup->recoveryBackendIDs.contains(backendID)) {
30 // Note: The next line modifies involvedBackends while iterating over it.
31 // This is only allowed because the iteration is terminated with a break below!
32 involvedBackends.merge(existingDeviceManager->_recoveryGroup->recoveryBackendIDs);
33 existingDeviceManager->_recoveryGroup = _recoveryGroup;
34 ++recoveryGroupSize;
35 break;
36 }
37 }
38 }
39
40 if(recoveryGroupSize > 1) {
41 // update the recovery group
42 _recoveryGroup->recoveryBackendIDs = involvedBackends;
43 // The barrier does not allow modification of the number of participants.
44 // We put a placement new to replace it. Before this, we have to call the constructor of the old instance.
45 _recoveryGroup->recoveryBarrier.~barrier();
46 new(&_recoveryGroup->recoveryBarrier) std::barrier(recoveryGroupSize);
47 }
48 }

52 std::vector<VariableNetworkNode> DeviceManager::getNodesList() const {
53 std::vector<VariableNetworkNode> rv;
54
55 // obtain register catalogue
56 auto catalog = _device.getRegisterCatalogue();
57
58 // iterate catalogue, create VariableNetworkNode for all registers
59 for(const auto& reg : catalog) {
60 // ignore 2D registers
61 if(reg.getNumberOfDimensions() > 1) {
62 continue;
63 }
64
65 // guess direction and determine update mode
66 VariableDirection direction{};
67 UpdateMode updateMode;
68 if(reg.isWriteable()) {
69 direction = {VariableDirection::consuming, false};
70 updateMode = UpdateMode::push;
71 }
72 else {
73 direction = {VariableDirection::feeding, false};
74 if(reg.getSupportedAccessModes().has(AccessMode::wait_for_new_data)) {
75 updateMode = UpdateMode::push;
76 }
77 else {
78 updateMode = UpdateMode::poll;
79 }
80 }
81
82 // find minimum type required to represent data
83 const auto* valTyp = &(reg.getDataDescriptor().minimumDataType().getAsTypeInfo());
84
85 // create node and add to list
86 rv.emplace_back(reg.getRegisterName(), _deviceAliasOrCDD, reg.getRegisterName(), updateMode, direction, *valTyp,
87 reg.getNumberOfElements());
88 }
89
90 return rv;
91 }

95 void DeviceManager::reportException(const std::string& errMsg) {
96 if(_owner->getTestableMode().isEnabled()) {
97 assert(_owner->getTestableMode().testLock());
98 }
99
100 // The error queue must only be modified when holding both mutexes (error mutex and testable mode mutex), because
101 // the testable mode counter must always be consistent with the content of the queue.
102 // To avoid deadlocks you must always first aquire the testable mode mutex if you need both.
103 // You can hold the error mutex without holding the testable mode mutex (for instance for checking the error
104 // predicate), but then you must not try to aquire the testable mode mutex!
105 boost::unique_lock<boost::shared_mutex> errorLock(_errorMutex);
106
107 if(!_deviceHasError) { // only report new errors if the device does not have reported errors already
108 if(_errorQueue.push(errMsg)) {
109 if(_owner->getTestableMode().isEnabled()) {
110 ++_owner->getTestableMode()._counter;
111 }
112 } // else do nothing. There are plenty of errors reported already: The queue is full.
113 // set the error flag and notify the other threads
114 _device.setException(errMsg);
115 _deviceHasError = true;
116 _exceptionVersionNumber = {}; // generate a new exception version number
117
118 // Release the error lock before notifying the other device managers in the recovery group.
119 // They will re-inform us, and holding the lock would deadlock trying to re-acquire this error mutex.
120 errorLock.unlock();
121
122 // Inform the other DeviceManagers in the recovery group.
123 for(auto const& [cdd, deviceManager] : _owner->getDeviceManagerMap()) {
124 if(deviceManager.get() == this) {
125 continue;
126 }
127 if(deviceManager->_recoveryGroup == _recoveryGroup) {
128 // Only append a device info to the recovery message if there is none.
129 auto deviceInfo =
130 (errMsg.find("[in device ") == std::string::npos) ? "[in device " + _deviceAliasOrCDD + "]" : "";
131 deviceManager->reportException(errMsg + deviceInfo);
132 }
133 }
134 }
135 }

146 void DeviceManager::mainLoop() {
147 // Whenever the loop is left, we have to drop the barrier so the other DeviceManagers are not blocked indefinitely.
148 // They could never terminate in this situation, and hence the whole application could never terminate.
149 try {
150 mainLoopImpl();
151 }
152 catch(...) {
153 _recoveryGroup->shutdown = true;
154 _recoveryGroup->recoveryBarrier.arrive_and_drop();
155 throw;
156 }
157 }

159 /********************************************************************************************************************/
160
161 void DeviceManager::mainLoopImpl() {
162 Application::registerThread("DM_" + getName());
163 std::string error;
164
165 // We have the testable mode lock. The device has not been initialised yet, but from now on the
166 // testableMode.deviceInitialisationCounter will take care or it
167 _testableModeReached = true;
168
169 // flag whether the devices was opened+initialised for the first time
170 bool firstSuccess = true;
171
172 while(true) {
173 // Sync point DETECTION:
174 // The manager has seen an error and (re)starts recovery. Wait until all
175 // involved DeviceManagers have seen it.
176 _recoveryGroup->waitForRecoveryStage(RecoveryGroup::RecoveryStage::DETECTION);
177 // Reset error stage to NO_ERROR. Contains a barrier to make sure all threads have seen it.
178 _recoveryGroup->resetErrorAtStage();
179
180 // Starting stage OPEN
181 // [Spec: 2.3.1] (Re)-open the device.
182 do {
183 _owner->getTestableMode().unlock("Wait before open/recover device");
184 usleep(500000);
185 boost::this_thread::interruption_point();
186 _owner->getTestableMode().lock("Attempt open/recover device");
187
188 try {
189 std::lock_guard<std::mutex> deviceOpenLock(_recoveryGroup->deviceOpenCloseMutex);
190 _device.open();
191 }
192 catch(ChimeraTK::runtime_error& e) {
193 assert(_deviceError._status != StatusOutput::Status::OK); // any error must already be reported...
194 if(std::string(_deviceError._message) != e.what()) {
195 ChimeraTK::logger(Logger::Severity::error, "Device " + _deviceAliasOrCDD) << e.what() << std::endl;
196 // set proper error message in very first attempt to open the device
197 setCurrentVersionNumber({});
198 _deviceError.write(StatusOutput::Status::FAULT, e.what());
199 }
200
201 continue; // should not be necessary because isFunctional() should return false. But no harm in leaving it in.
202 }
203 } while(!_device.isFunctional());
204
205 boost::unique_lock<boost::shared_mutex> errorLock(_errorMutex);
206
207 // [Spec: 2.3.3] Empty exception reporting queue.
208 while(_errorQueue.pop()) {
209 if(_owner->getTestableMode()._enabled) {
210 assert(_owner->getTestableMode()._counter > 0);
211 --_owner->getTestableMode()._counter;
212 }
213 }
214 errorLock.unlock(); // we don't need to hold the lock for now, but we will need it later
215
216 for(auto& writeMe : _writeRegisterPaths) {
217 auto reg = _device.getOneDRegisterAccessor<std::string>(writeMe); // the user data type does not matter here.
218 if(!reg.isWriteable()) {
219 _owner->getTestableMode().unlock("throwing" + std::string(writeMe) + " is not writeable!");
220 throw ChimeraTK::logic_error(std::string(writeMe) + " is not writeable!");
221 }
222 }
223
224 for(auto& readMe : _readRegisterPaths) {
225 auto reg = _device.getOneDRegisterAccessor<std::string>(readMe); // the user data type does not matter here.
226 if(!reg.isReadable()) {
227 _owner->getTestableMode().unlock("throwing" + std::string(readMe) + " is not readable!");
228 throw ChimeraTK::logic_error(std::string(readMe) + " is not readable!");
229 }
230 }
231
232 // Sync point (stage OPEN complete): Device opened. Synchronise before starting init scripts.
233 assert(_recoveryGroup->errorAtStage ==
234 RecoveryGroup::RecoveryStage::NO_ERROR); // no other thread must have modified the flag until here.
235
236 // no need to check the return value. No error reported in the OPEN stage.
237 _recoveryGroup->waitForRecoveryStage(RecoveryGroup::RecoveryStage::OPEN);
238
239 // Starting stage INIT_HANDLERS
240 // [Spec: 2.3.2] Run initialisation handlers
241 try {
242 for(auto& initHandler : _initialisationHandlers) {
243 {
244 // Hold the open/close lock while executing the init handler, so no other
245 // DeviceManager closes the device while the init handler is running.
246 std::lock_guard<std::mutex> openCloseLock(_recoveryGroup->deviceOpenCloseMutex);
247 _device.close();
248 initHandler(_device);
249 _device.open();
250 }
251 }
252 }
253 catch(ChimeraTK::runtime_error& e) {
254 assert(_deviceError._status != StatusOutput::Status::OK); // any error must already be reported...
255 // update error message, since it might have been changed...
256 if(std::string(_deviceError._message) != e.what()) {
257 ChimeraTK::logger(Logger::Severity::error, "Device " + _deviceAliasOrCDD) << e.what() << std::endl;
258 setCurrentVersionNumber({});
259 _deviceError.write(StatusOutput::Status::FAULT, e.what());
260 }
261 // Mark recovery as failed. All DeviceManagers will return to the beginning of the recovery after the next
262 // synchronisation point
263 _recoveryGroup->setErrorAtStage(RecoveryGroup::RecoveryStage::INIT_HANDLERS);
264 }
265 catch(...) {
266 // This will terminate this DeviceManager main loop
267 // Drop the testable mode lock before we bail out.
268 // The other DeviceManagers in this recovery group will be informed in the mainLoop's catch all block.
269 _owner->getTestableMode().unlock("ERROR in INIT_HANDLERS");
270 throw;
271 }
272
273 // Sync point (stage INIT_HANDLERS complete): Wait until all init scripts are done before writing recovery
274 // accessors.
275 if(!_recoveryGroup->waitForRecoveryStage(RecoveryGroup::RecoveryStage::INIT_HANDLERS)) {
276 // If another thread has already continued and set an error for recovery stage RECOVERY_ACCESSORS,
277 // waitForRecoveryStage(INIT_HANDLERS) will still return 'true', so all threads arrive at the
278 // barrier for stage RECOVERY_ACCESSORS.
279 // If there was error in stage INIT_HANDLERS, all threads will see it here and continue.
280 continue;
281 }
282
283 // Starting stage RECOVERY_ACCESSORS
284 // Write all recovery accessors
285 // We are now entering the critical recovery section. It is protected by the recovery mutex until the
286 // deviceHasError flag has been cleared.
287 boost::unique_lock<boost::shared_mutex> recoveryLock(_recoveryMutex);
288 try {
289 // sort recovery helpers according to write order
290 _recoveryHelpers.sort([](boost::shared_ptr<RecoveryHelper>& a, boost::shared_ptr<RecoveryHelper>& b) {
291 return a->writeOrder < b->writeOrder;
292 });
293 for(auto& recoveryHelper : _recoveryHelpers) {
294 if(recoveryHelper->versionNumber != VersionNumber{nullptr}) {
295 recoveryHelper->accessor->write();
296 recoveryHelper->wasWritten = true;
297 }
298 }
299 }
300 catch(ChimeraTK::runtime_error& e) {
301 // update error message, since it might have been changed...
302 if(std::string(_deviceError._message) != e.what()) {
303 ChimeraTK::logger(Logger::Severity::error, "Device " + _deviceAliasOrCDD) << e.what() << std::endl;
304 setCurrentVersionNumber({});
305 _deviceError.write(StatusOutput::Status::FAULT, e.what());
306 }
307 // Mark recovery as failed. All DeviceManagers will return to the beginning of the recovery after the next
308 // synchronisation point
309 _recoveryGroup->setErrorAtStage(RecoveryGroup::RecoveryStage::RECOVERY_ACCESSORS);
310 }
311 catch(...) {
312 // This will terminate this DeviceManager main loop
313 // Drop the testable mode lock before we bail out.
314 // The other DeviceManagers in this recovery group will be informed in the mainLoop's catch all block.
315 _owner->getTestableMode().unlock("ERROR in RECOVERY_ACCESSORS");
316 throw;
317 }
318
319 // Sync point (stage RECOVERY_ACCESSORS complete): All recovery accessors have been written.
320 if(!_recoveryGroup->waitForRecoveryStage(RecoveryGroup::RecoveryStage::RECOVERY_ACCESSORS)) {
321 // In case of error, jump back to the beginning of the recovery/open procedure
322 continue;
323 }
324
325 // complete the recovery, then wait for the exception
326 errorLock.lock();
327 _deviceHasError = false;
328 errorLock.unlock();
329
330 // Sync point (stage CLEAR_ERROR complete): All DeviceManagers have cleared the error flag.
331 // This barrier protects against exceptions being reported. From now on read/write operations can
332 // report exceptions again, and they will not be suppressed by reportException due to an existing error condition.
333 _recoveryGroup->waitForRecoveryStage(RecoveryGroup::RecoveryStage::CLEAR_ERROR);
334
335 recoveryLock.unlock();
336
337 // send the trigger that the device is available again
338 _device.activateAsyncRead();
339 if(_isHoldingInitialValueLatch) {
340 _isHoldingInitialValueLatch = false;
341 _initialValueLatch.count_down();
342 }
343
344 // [Spec: 2.3.5] Reset exception state and wait for the next error to be reported.
345 _deviceError.writeOk();
346 deviceBecameFunctional.write();
347
348 if(!firstSuccess) {
349 ChimeraTK::logger(Logger::Severity::info, "Device " + _deviceAliasOrCDD) << "Error cleared." << std::endl;
350 }
351 firstSuccess = false;
352
353 // decrement special testable mode counter, was incremented manually above to make sure initialisation completes
354 // within one "application step"
355 if(Application::getInstance().getTestableMode()._enabled) {
356 --_owner->getTestableMode()._deviceInitialisationCounter;
357 }
358
359 // [Spec: 2.3.8] Wait for an exception being reported by the ExceptionHandlingDecorators
360 // release the testable mode mutex for waiting for the exception.
361 _owner->getTestableMode().unlock("Wait for exception");
362
363 // Do not modify the queue without holding the testable mode lock, because we also consistently have to modify
364 // the counter protected by that mutex.
365 // Just call wait(), not pop_wait().
366 boost::this_thread::interruption_point();
367 _errorQueue.wait();
368 boost::this_thread::interruption_point();
369
370 _owner->getTestableMode().lock("Process exception");
371 // increment special testable mode counter to make sure the initialisation completes within one
372 // "application step"
373 if(Application::getInstance().getTestableMode()._enabled) {
374 ++_owner->getTestableMode()._deviceInitialisationCounter; // matched above with a decrement
375 }
376
377 errorLock.lock(); // we need both locks to modify the queue
378
379 auto popResult = _errorQueue.pop(error);
380 assert(popResult); // this if should always be true, otherwise the waiting did not work.
381 (void)popResult; // avoid warning in production build. g++5.4 does not support [[maybe_unused]] yet.
382 if(_owner->getTestableMode()._enabled) {
383 assert(_owner->getTestableMode()._counter > 0);
384 --_owner->getTestableMode()._counter;
385 }
386
387 // [ExceptionHandling Spec: C.3.3.14] report exception to the control system
388 ChimeraTK::logger(Logger::Severity::error, "Device " + _deviceAliasOrCDD) << error << std::endl;
389 setCurrentVersionNumber({});
390 _deviceError.write(StatusOutput::Status::FAULT, error);
391
392 // We must not hold the lock while waiting for the synchronousTransferCounter to go back to 0. Only release it
393 // after deviceError has been written, so the CircularDependencyDetector can read the error message from its
394 // thread for printing.
395 errorLock.unlock();
396
397 // [ExceptionHandling Spec: C.3.3.15] Wait for all synchronous transfers to finish before starting recovery.
398 while(_synchronousTransferCounter > 0) {
399 usleep(1000);
400 }
401
402 } // while(true)
403 }

407 void DeviceManager::prepare() {
408 // Set initial status to error
409 setCurrentVersionNumber({});
410 _deviceError.write(StatusOutput::Status::FAULT, "Attempting to open device...");
411
412 // Increment special testable mode counter to make sure the initialisation completes within one
413 // "application step". Start with counter increased (device not initialised yet, wait).
414 // We can to this here without testable mode lock because the application is still single threaded.
415 if(Application::getInstance().getTestableMode()._enabled) {
416 ++_owner->getTestableMode()._deviceInitialisationCounter; // released and increased in handeException loop
417 }
418 }

486 void DeviceManager::terminate() {
487 if(_moduleThread.joinable()) {
488 _moduleThread.interrupt();
489 // try joining the thread
490 while(!_moduleThread.try_join_for(boost::chrono::milliseconds(10))) {
491 // send boost interrupted exception through the _errorQueue
492 _errorQueue.push_exception(std::make_exception_ptr(boost::thread_interrupted()));
493
494 // it may not suffice to send the exception once, as the exception might get overwritten in the queue, thus we
495 // repeat this until the thread was joined.
496 }
497 }
498 assert(!_moduleThread.joinable());
499 }
