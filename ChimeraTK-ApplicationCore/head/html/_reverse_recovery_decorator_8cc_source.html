8 template<typename UserType>
9 ReverseRecoveryDecorator<UserType>::ReverseRecoveryDecorator(
10 boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> accessor, const VariableNetworkNode& networkNode)
11 : ExceptionHandlingDecorator<UserType>(std::move(accessor), networkNode) {
12 // Check if we are wrapping a push-type variable and forbid that
13 if(TransferElement::getAccessModeFlags().has(AccessMode::wait_for_new_data)) {
14 throw ChimeraTK::logic_error("Cannot use reverse recovery on push-type input");
15 }
16
17 // Set ourselves as wfnd:
18 TransferElement::_accessModeFlags.add(AccessMode::wait_for_new_data);
19
20 _recoveryHelper->notificationQueue = cppext::future_queue<void>(3);
21 _recoveryHelper->recoveryDirection = RecoveryHelper::Direction::fromDevice;
22
23 // Set the read queue as continuation of the notification queue
24 // The continuation will just trigger a read on the target accessor
25 this->_readQueue =
26 _recoveryHelper->notificationQueue.template then<void>([&, this]() { _target->read(); }, std::launch::deferred);
27 }
