14 void CircularDependencyDetector::registerDependencyWait(VariableNetworkNode& node) {
15 assert(node.getType() == NodeType::Application);
16 std::unique_lock<std::mutex> lock(_mutex);
17
18 auto* dependent = dynamic_cast<Module*>(node.getOwningModule())->findApplicationModule();
19
20 // register the waiting node in the map (used for the detectBlockedModules() thread). This is done also for Device
21 // variables etc.
22 _awaitedNodes[dependent] = node;
23
24 // checking of direct circular dependencies is only done for Application-type feeders
25 auto feedingModule = node.getModel().visit(Model::returnApplicationModule, Model::keepPvAccess,
26 Model::keepApplicationModules, Model::adjacentInSearch, Model::returnFirstHit(Model::ApplicationModuleProxy{}));
27 if(!feedingModule.isValid()) {
28 return;
29 }
30 Module* dependency = &feedingModule.getApplicationModule();
31
32 // If a module depends on itself, the detector would always detect a circular dependency, even if it is resolved
33 // by writing initial values in prepare(). Hence we do not check anything in this case.
34 if(dependent == dependency) {
35 return;
36 }
37
38 // Register dependent-dependency relation in map
39 _awaitedVariables[dependent] = node.getQualifiedName();
40 _waitMap[dependent] = dependency;
41
42 // check for circular dependencies
43 auto* depdep = dependency;
44 while(_waitMap.find(depdep) != _waitMap.end()) {
45 auto* depdep_prev = depdep;
46 depdep = _waitMap[depdep];
47 if(depdep == dependent) {
48 // The detected circular dependency might still resolve itself, because registerDependencyWait() is called even
49 // if initial values are already present in the queues.
50 for(size_t i = 0; i < 1000; ++i) {
51 // give other thread a chance to read their initial value
52 lock.unlock();
53 usleep(10000);
54 lock.lock();
55 // if the module depending on an initial value for "us" is no longer waiting for us to send an initial value,
56 // the circular dependency is resolved.
57 if(_waitMap.find(depdep_prev) == _waitMap.end() || _waitMap[depdep] != dependent) {
58 return;
59 }
60 }
61
62 std::cerr << "*** Circular dependency of ApplicationModules found while waiting for initial values!"
63 << std::endl;
64 std::cerr << std::endl;
65
66 std::cerr << dependent->getQualifiedNameWithType() << " waits for " << node.getQualifiedName()
67 << " from:" << std::endl;
68 auto* depdep2 = dependency;
69 while(_waitMap.find(depdep2) != _waitMap.end()) {
70 auto waitsFor = _awaitedVariables[depdep2];
71 std::cerr << depdep2->getQualifiedNameWithType();
72 if(depdep2 == dependent) {
73 break;
74 }
75 depdep2 = _waitMap[depdep2];
76 std::cerr << " waits for " << waitsFor << " from:" << std::endl;
77 }
78 std::cerr << "." << std::endl;
79
80 std::cerr << std::endl;
81 std::cerr
82 << "Please provide an initial value in the prepare() function of one of the involved ApplicationModules!"
83 << std::endl;
84
85 throw ChimeraTK::logic_error("Circular dependency of ApplicationModules while waiting for initial values");
86 }
87 // Give other threads a chance to add to the wait map
88 lock.unlock();
89 usleep(10000);
90 lock.lock();
91 }
92 }

96 void CircularDependencyDetector::printWaiters(std::ostream& stream) {
97 if(_waitMap.empty()) {
98 return;
99 }
100 stream << "The following modules are still waiting for initial values:" << std::endl;
101 for(auto& waiters : _waitMap) {
102 stream << waiters.first->getQualifiedNameWithType() << " waits for " << _awaitedVariables[waiters.first]
103 << " from " << waiters.second->getQualifiedNameWithType() << std::endl;
104 }
105 stream << "(end of list)" << std::endl;
106 }

110 void CircularDependencyDetector::unregisterDependencyWait(VariableNetworkNode& node) {
111 assert(node.getType() == NodeType::Application);
112 std::lock_guard<std::mutex> lock(_mutex);
113 auto* mod = dynamic_cast<Module*>(node.getOwningModule())->findApplicationModule();
114 _waitMap.erase(mod);
115 _awaitedVariables.erase(mod);
116 _awaitedNodes.erase(mod);
117 }

127 void CircularDependencyDetector::detectBlockedModules() {
128 auto& app = Application::getInstance();
129 while(true) {
130 // wait some time to slow down this check
131 boost::this_thread::sleep_for(boost::chrono::seconds(60));
132
133 // check for modules which did not yet reach their mainLoop
134 bool allModulesEnteredMainLoop = true;
135 for(auto* module : app.getSubmoduleListRecursive()) {
136 // Note: We are "abusing" this flag which was introduced for the testable mode. It actually just shows whether
137 // the mainLoop() has been called already (resp. will be called very soon). It is not depending on the
138 // testableMode. FIXME: Rename this flag!
139 if(module->hasReachedTestableMode()) {
140 continue;
141 }
142
143 // Found a module which is still waiting for initial values
144 allModulesEnteredMainLoop = false;
145
146 // require lock against concurrent accesses on the maps
147 std::lock_guard<std::mutex> lock(_mutex);
148
149 // Check if module has registered a dependency wait. If not, situation will either resolve soon or module will
150 // register a dependency wait soon. Both cases can be found in the next iteration.
151 if(_awaitedNodes.find(module) == _awaitedNodes.end()) {
152 continue;
153 }
154
155 auto* appModule = dynamic_cast<ApplicationModule*>(module);
156 if(!appModule) {
157 // only ApplicationModule can have hasReachedTestableMode() == true, so it should not happen
158 logger(Logger::Severity::warning, "CircularDependencyDetector")
159 << "found non-application module: " << module->getQualifiedNameWithType() << std::endl;
160 continue;
161 }
162
163 // Iteratively search for reason the module blocks
164 std::set<Module*> searchedModules;
165 std::function<void(const VariableNetworkNode& node)> iterativeSearch = [&](const VariableNetworkNode& node) {
166 auto visitor = [&](auto proxy) {
167 if constexpr(Model::isApplicationModule(proxy)) {
168 // fed by other ApplicationModule: check if that one is waiting, too
169 auto* feedingAppModule = dynamic_cast<ApplicationModule*>(&proxy.getApplicationModule());
170 if(feedingAppModule->hasReachedTestableMode()) {
171 // the feeding module has started its mainLoop(), but not sent us an initial value
172 // FIXME: There is a race condition, if the situation has right now resolved and the feeding module just
173 // not yet sent the initial value. Ideally we should wait another iteration before printing warnings!
174 if(_modulesWeHaveWarnedAbout.find(feedingAppModule) == _modulesWeHaveWarnedAbout.end()) {
175 _modulesWeHaveWarnedAbout.insert(feedingAppModule);
176 logger(Logger::Severity::warning, "CircularDependencyDetector")
177 << "Note: ApplicationModule " << appModule->getQualifiedNameWithType() << " is waiting for an "
178 << "initial value, because " << feedingAppModule->getQualifiedNameWithType()
179 << " has not yet sent one." << std::endl;
180 }
181 return;
182 }
183 if(_awaitedNodes.find(feedingAppModule) == _awaitedNodes.end()) {
184 // The other module is not right now waiting. It will either enter the mainLoop soon, or start
185 // waiting soon. Let's just wait another iteration.
186 return;
187 }
188 // The other module is right now waiting: continue iterative search
189 bool notYetSearched = searchedModules.insert(feedingAppModule).second;
190 if(notYetSearched) {
191 iterativeSearch(_awaitedNodes.at(feedingAppModule));
192 }
193 else {
194 if(_modulesWeHaveWarnedAbout.find(feedingAppModule) == _modulesWeHaveWarnedAbout.end()) {
195 _modulesWeHaveWarnedAbout.insert(feedingAppModule);
196 logger(Logger::Severity::warning, "CircularDependencyDetector")
197 << "Note: ApplicationModule " << appModule->getQualifiedNameWithType() << " and "
198 << feedingAppModule->getQualifiedNameWithType()
199 << " are both waiting, on initial value provided directly or indirectly by the other."
200 << std::endl;
201 }
202 }
203 }
204 else if constexpr(Model::isDeviceModule(proxy)) {
205 // fed by device
206 const auto& deviceName = proxy.getAliasOrCdd();
207 if(_devicesWeHaveWarnedAbout.find(deviceName) == _devicesWeHaveWarnedAbout.end()) {
208 _devicesWeHaveWarnedAbout.insert(deviceName);
209 auto myLog = logger(Logger::Severity::warning, "CircularDependencyDetector");
210 myLog << "Note: Still waiting for device " << deviceName << " to come up";
211 auto dm = Application::getInstance()._deviceManagerMap[deviceName];
212 std::unique_lock dmLock(dm->_errorMutex);
213 if(dm->_deviceHasError) {
214 myLog << " (" << std::string(dm->_deviceError._message) << ")";
215 }
216 myLog << "...";
217 }
218 }
219 else {
220 // fed by anything else?
221 logger(Logger::Severity::warning, "CircularDependencyDetector")
222 << "At least one ApplicationModule " << appModule->getQualifiedNameWithType() << " is waiting for an "
223 << "initial value from an unexpected source."
224 << "\n"
225 << "This is probably a BUG in the ChimeraTK framework.";
226 }
227 };
228
229 node.getModel().visit(visitor, Model::keepPvAccess, Model::adjacentInSearch);
230 };
231 iterativeSearch(_awaitedNodes.at(appModule));
232 }
233
234 // if all modules are in the mainLoop, stop this thread
235 if(allModulesEnteredMainLoop) {
236 break;
237 }
238 }
239
240 logger(Logger::Severity::info, "CircularDependencyDetector") << "All application modules are running.";
241
242 // free some memory
243 _waitMap.clear();
244 _awaitedVariables.clear();
245 _awaitedNodes.clear();
246 _modulesWeHaveWarnedAbout.clear();
247 _devicesWeHaveWarnedAbout.clear();
248 _otherThingsWeHaveWarnedAbout.clear();
249 }
