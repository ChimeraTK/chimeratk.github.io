30 /* The test module is writing to the device. It is the "module under test".
31  * This is the one whose variables are to be recovered. It is not the place the the

34 struct TestModule : public ctk::ApplicationModule {
35  TestModule(ctk::ModuleGroup* owner, const std::string& name, const std::string& description,
36  const std::unordered_set<std::string>& tags = {})
37  : ApplicationModule(owner, name, description, tags), mainLoopStarted(2) {}
38 
39  ctk::ScalarPushInput<int32_t> trigger{this, "trigger", "", "This is my trigger."};
40  ctk::ScalarOutput<int32_t> scalarOutput{this, "TO_DEV_SCALAR1", "", "Here I write a scalar"};
41  ctk::ArrayOutput<int32_t> arrayOutput{this, "TO_DEV_ARRAY1", "", 4, "Here I write an array"};
42 
43  // We do not use testable mode for this test, so we need this barrier to synchronise to the beginning of the
44  // mainLoop(). This is required to make sure the initial value propagation is done.
45  // execute this right after the Application::run():
46  // app.testModule.mainLoopStarted.wait(); // make sure the module's mainLoop() is entered
47  boost::barrier mainLoopStarted;
48 
49  void mainLoop() override {
50  mainLoopStarted.wait();
51 
52  while(true) {
53  scalarOutput = int32_t(trigger);
54  scalarOutput.write();
55  for(uint i = 0; i < 4; i++) {
56  arrayOutput[i] = int32_t(trigger);
57  }
58  arrayOutput.write();
59  trigger.read(); // read the blocking variable at the end so the initial values are propagated
60  }
61  }
62 };

136 BOOST_AUTO_TEST_CASE(testProcessVariableRecovery) {
137  std::cout << "testProcessVariableRecovery" << std::endl;
138  TestApplication app;
139 
140  ctk::TestFacility test{app, false};
141  // Write initial values manually since we do not use the testable mode.
142  // Otherwise the main loops never start.
143 
144  // initial value for the direct CS->DEV register
145  test.writeScalar("/TEST/TO_DEV_SCALAR2", 42);
146  std::vector<int32_t> array = {99, 99, 99, 99};
147  test.writeArray("/TEST/TO_DEV_ARRAY2", array);
148 
149  // initial value for the trigger
150  test.writeScalar("/TEST/trigger", 0);
151 
152  app.run();
153  app.module.mainLoopStarted.wait();
154 
155  ctk::Device dummy;
156  dummy.open(deviceCDD.data());
157 
158  // wait for the device to be opened successfully so the access to the dummy does not throw
159  // (as they use the same backend it now throws if there has been an exception somewhere else)
160  CHECK_EQUAL_TIMEOUT(test.readScalar<int32_t>(
161  std::string("/Devices/") + ctk::Utilities::escapeName(deviceCDD.data(), false) + "/status"),
162  0, 10000);
163 
164  // Check that the initial values are there.
165  CHECK_EQUAL_TIMEOUT(dummy.read<int32_t>("/TEST/TO_DEV_SCALAR2"), 42, 10000);
166  CHECK_EQUAL_TIMEOUT(dummy.read<int32_t>("/TEST/TO_DEV_ARRAY2", 1, 0)[0], 99, 10000);
167  CHECK_EQUAL_TIMEOUT(dummy.read<int32_t>("/TEST/TO_DEV_ARRAY2", 1, 1)[0], 99, 10000);
168  CHECK_EQUAL_TIMEOUT(dummy.read<int32_t>("/TEST/TO_DEV_ARRAY2", 1, 2)[0], 99, 10000);
169  CHECK_EQUAL_TIMEOUT(dummy.read<int32_t>("/TEST/TO_DEV_ARRAY2", 1, 3)[0], 99, 10000);
170 
171  // Update device register via application module.
172  auto trigger = test.getScalar<int32_t>("/TEST/trigger");
173  trigger = 100;
174  trigger.write();
175  // Check if the values are updated.
176  CHECK_EQUAL_TIMEOUT(dummy.read<int32_t>("/TEST/TO_DEV_SCALAR1"), 100, 10000);
177  CHECK_EQUAL_TIMEOUT(dummy.read<int32_t>("/TEST/TO_DEV_ARRAY1", 1, 0)[0], 100, 10000);
178  CHECK_EQUAL_TIMEOUT(dummy.read<int32_t>("/TEST/TO_DEV_ARRAY1", 1, 1)[0], 100, 10000);
179  CHECK_EQUAL_TIMEOUT(dummy.read<int32_t>("/TEST/TO_DEV_ARRAY1", 1, 2)[0], 100, 10000);
180  CHECK_EQUAL_TIMEOUT(dummy.read<int32_t>("/TEST/TO_DEV_ARRAY1", 1, 3)[0], 100, 10000);
181 
182  auto dummyBackend = boost::dynamic_pointer_cast<ctk::ExceptionDummy>(
183  ctk::BackendFactory::getInstance().createBackend(deviceCDD.data()));
184 
185  // Set the device to throw.
186  dummyBackend->throwExceptionOpen = true;
187 
188  // Set dummy registers to 0.
189  dummy.write<int32_t>("/CONSTANT/VAR32", 0);
190  dummy.write<int32_t>("/TEST/TO_DEV_SCALAR1", 0);
191  dummy.write<int32_t>("/TEST/TO_DEV_SCALAR2", 0);
192  array = {0, 0, 0, 0};
193  dummy.write("/TEST/TO_DEV_ARRAY1", array);
194  dummy.write("/TEST/TO_DEV_ARRAY2", array);
195 
196  CHECK_EQUAL_TIMEOUT(dummy.read<int32_t>("/CONSTANT/VAR32"), 0, 10000);
197  dummyBackend->throwExceptionWrite = true;
198  dummyBackend->throwExceptionRead = true;
199 
200  // Now we trigger the reading module. This should put the device into an error state
201  auto trigger2 = test.getVoid("/deviceTrigger");
202  trigger2.write();
203 
204  // Verify that the device is in error state.
205  CHECK_EQUAL_TIMEOUT(test.readScalar<int32_t>(ctk::RegisterPath("/Devices") /
206  ctk::Utilities::escapeName(deviceCDD.data(), false) / "status"),
207  1, 10000);
208 
209  // Set device back to normal.
210  dummyBackend->throwExceptionWrite = false;
211  dummyBackend->throwExceptionRead = false;
212  dummyBackend->throwExceptionOpen = false;
213  // Verify if the device is ready.
214  CHECK_EQUAL_TIMEOUT(test.readScalar<int32_t>(ctk::RegisterPath("/Devices") /
215  ctk::Utilities::escapeName(deviceCDD.data(), false) / "status"),
216  0, 10000);
217 
218  // Device should have the correct values now. Notice that we did not trigger the writer module!
219  BOOST_CHECK_EQUAL(dummy.read<int32_t>("/TEST/TO_DEV_SCALAR2"), 42);
220  BOOST_CHECK((dummy.read<int32_t>("/TEST/TO_DEV_ARRAY2", 0) == std::vector<int32_t>{99, 99, 99, 99}));
221 
222  BOOST_CHECK_EQUAL(dummy.read<int32_t>("/TEST/TO_DEV_SCALAR1"), 100);
223  BOOST_CHECK((dummy.read<int32_t>("/TEST/TO_DEV_ARRAY1", 0) == std::vector<int32_t>{100, 100, 100, 100}));
224 }
