310  std::list<EntityOwner*> VariableNetworkNode::scanForCircularDepencency() const {
311  // We are starting a new scan. Reset the indicator for already found circular dependencies.
312  detail::CircularDependencyDetectionRecursionStopper::startNewScan();
313 
314  if(getType() != NodeType::Application && getType() != NodeType::TriggerReceiver &&
315  getType() != NodeType::TriggerProvider) {
316  assert(getType() == NodeType::ControlSystem || getType() == NodeType::Device || getType() == NodeType::Constant);
317  return {};
318  }
319 
320  if(!getModel().isValid()) {
321  return {};
322  }
323 
324  // find the feeder of the network
325  auto amProxy = getModel().visit(Model::returnApplicationModule, Model::keepPvAccess, Model::keepApplicationModules,
326  Model::adjacentInSearch, Model::returnFirstHit(Model::ApplicationModuleProxy{}));
327  // CS modules and device modules don't have an owning module. They stop the circle anyway. So if either the feeder
328  // or the receiver (this) don't have an owning module, there is nothing to do here.
329  if(!amProxy.isValid()) {
330  return {};
331  }
332  assert(getDirection().dir == VariableDirection::consuming);
333 
334  Module* owningModule = &amProxy.getApplicationModule();
335 
336  // We do not put ourselves in the list right away. The called code will do this as well and detect a circle
337  // immediately, even if there is just a simple connection, we leave the marking of the already visited node
338  // to the recursive call.
339  auto inputModuleList = owningModule->getInputModulesRecursively({});
340 
341  auto nInstancesFound = std::count(inputModuleList.begin(), inputModuleList.end(), owningModule);
342  assert(nInstancesFound >= 1); // the start list must not have been deleted in the call
343  // The owning module has been found again when scanning inputs recursively -> There is a circular dependency
344  if(nInstancesFound > 1) {
345  // clean up the circular network we found and return it.
346  inputModuleList.sort();
347  inputModuleList.unique();
348 
349  // Remember that we are part of a circle, and of which circle
350  pdata->circularNetworkHash = boost::hash_range(inputModuleList.begin(), inputModuleList.end());
351  // we already did the assertion that the owning module is an application module above, so we can static cast here
352  auto* applicationModule = dynamic_cast<ApplicationModule*>(owningModule);
353  applicationModule->setCircularNetworkHash(pdata->circularNetworkHash);
354 
355  // Find the MetaDataPropagatingRegisterDecorator which is involved and set the _isCirularInput flag
356  auto internalTargetElements = getAppAccessorNoType().getInternalElements();
357  // This is a list of all the nested decorators, so we will find the right point to cast
358  for(auto& elem : internalTargetElements) {
359  auto flagProvider = boost::dynamic_pointer_cast<MetaDataPropagationFlagProvider>(elem);
360  if(flagProvider) {
361  flagProvider->_isCircularInput = true;
362  }
363  }
364 
365  return inputModuleList;
366  }
367 
368  // No circular network. Return an empty list.
369  return {};
370  }

423  void VariableNetworkNode::setAppAccessorConstImplementation(const VariableNetworkNode& feeder) const {
424  callForType(getValueType(), [&](auto t) {
425  using UserType = decltype(t);
426  auto impl = boost::dynamic_pointer_cast<ChimeraTK::NDRegisterAccessor<UserType>>(
427  boost::make_shared<ConstantAccessor<UserType>>(feeder.getConstantValue<UserType>(), getNumberOfElements(),
428  getMode() == UpdateMode::push ? AccessModeFlags{AccessMode::wait_for_new_data} : AccessModeFlags{}));
429 
430  if(getMode() == UpdateMode::push && getDirection().dir == VariableDirection::consuming) {
431  setAppAccessorImplementation<UserType>(Application::getInstance().getTestableMode().decorate(
432  impl, detail::TestableMode::DecoratorType::READ, "Constant"));
433  }
434  else {
435  setAppAccessorImplementation<UserType>(impl);
436  }
437  });
438  }
