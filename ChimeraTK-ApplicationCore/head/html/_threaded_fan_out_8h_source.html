134 template<typename UserType>
135 void ThreadedFanOut<UserType>::run() {
136 Application::registerThread("ThFO" + FanOut<UserType>::_impl->getName());
137 Application::getInstance().getTestableMode().lock("start");
138 _testableModeReached = true;
139
140 ChimeraTK::VersionNumber version{nullptr};
141 version = readInitialValues(FanOut<UserType>::_impl);
142 while(true) {
143 // send out copies to slaves
144 boost::this_thread::interruption_point();
145 auto validity = FanOut<UserType>::_impl->dataValidity();
146 for(auto& slave : FanOut<UserType>::_slaves) {
147 // do not send copy if no data is expected (e.g. trigger)
148 if(slave->getNumberOfSamples() != 0) {
149 slave->accessChannel(0) = FanOut<UserType>::_impl->accessChannel(0);
150 }
151 slave->setDataValidity(validity);
152 bool dataLoss = slave->writeDestructively(version);
153 if(dataLoss) {
154 Application::incrementDataLossCounter(slave->getName());
155 }
156 }
157 // receive data
158 boost::this_thread::interruption_point();
159 FanOut<UserType>::_impl->read();
160 version = FanOut<UserType>::_impl->getVersionNumber();
161 }
162 }

214 template<typename UserType>
215 void ThreadedFanOutWithReturn<UserType>::run() {
216 Application::registerThread("ThFO" + FanOut<UserType>::_impl->getName());
217 Application::getInstance().getTestableMode().lock("start");
218 _testableModeReached = true;
219
220 std::map<TransferElementID, boost::shared_ptr<NDRegisterAccessor<UserType>>> accessors;
221 for(auto& acc : FanOut<UserType>::_slaves) {
222 accessors[acc->getId()] = acc;
223 }
224 accessors[FanOut<UserType>::_impl->getId()] = FanOut<UserType>::_impl;
225
226 TransferElementID changedVariable = _initialValueProvider->getId();
227
228 auto version = readInitialValues(_initialValueProvider);
229
230 ReadAnyGroup group(_inputChannels.begin(), _inputChannels.end());
231
232 while(true) {
233 // send out copies to all receivers (slaves and return channel of feeding node)
234 for(auto& [id, accessor] : accessors) {
235 // do not feed back value to the accessor we got it from
236 if(id == changedVariable) {
237 continue;
238 }
239
240 // do not send copy if no data is expected (e.g. trigger)
241 if(accessor->getNumberOfSamples() != 0) {
242 accessor->accessChannel(0) = accessors[changedVariable]->accessChannel(0);
243 }
244
245 bool dataLoss = accessor->writeDestructively(version);
246
247 if(dataLoss) {
248 Application::incrementDataLossCounter(accessor->getName());
249 }
250 }
251
252 // receive data
253 boost::this_thread::interruption_point();
254 changedVariable = group.readAny();
255 boost::this_thread::interruption_point();
256
257 version = accessors[changedVariable]->getVersionNumber();
258 }
259 }
