126 template<typename UserType>
127 void ThreadedFanOut<UserType>::run() {
128 Application::registerThread("ThFO" + FanOut<UserType>::_impl->getName());
129 Application::getInstance().getTestableMode().lock("start");
130 _testableModeReached = true;
131
132 ChimeraTK::VersionNumber version{nullptr};
133 version = readInitialValues();
134 while(true) {
135 // send out copies to slaves
136 boost::this_thread::interruption_point();
137 auto validity = FanOut<UserType>::_impl->dataValidity();
138 for(auto& slave : FanOut<UserType>::_slaves) {
139 // do not send copy if no data is expected (e.g. trigger)
140 if(slave->getNumberOfSamples() != 0) {
141 slave->accessChannel(0) = FanOut<UserType>::_impl->accessChannel(0);
142 }
143 slave->setDataValidity(validity);
144 bool dataLoss = slave->writeDestructively(version);
145 if(dataLoss) {
146 Application::incrementDataLossCounter(slave->getName());
147 }
148 }
149 // receive data
150 boost::this_thread::interruption_point();
151 FanOut<UserType>::_impl->read();
152 version = FanOut<UserType>::_impl->getVersionNumber();
153 }
154 }

197 template<typename UserType>
198 void ThreadedFanOutWithReturn<UserType>::run() {
199 Application::registerThread("ThFO" + FanOut<UserType>::_impl->getName());
200 Application::getInstance().getTestableMode().lock("start");
201 _testableModeReached = true;
202
203 std::map<TransferElementID, boost::shared_ptr<NDRegisterAccessor<UserType>>> accessors;
204 for(auto& acc : FanOut<UserType>::_slaves) {
205 accessors[acc->getId()] = acc;
206 }
207 accessors[FanOut<UserType>::_impl->getId()] = FanOut<UserType>::_impl;
208
209 TransferElementID changedVariable = FanOut<UserType>::_impl->getId();
210 VersionNumber version{nullptr};
211
212 version = readInitialValues();
213
214 ReadAnyGroup group(_inputChannels.begin(), _inputChannels.end());
215
216 while(true) {
217 // send out copies to all receivers (slaves and return channel of feeding node)
218 for(auto& [id, accessor] : accessors) {
219 // do not feed back value to the accessor we got it from
220 if(id == changedVariable) {
221 continue;
222 }
223
224 // do not send copy if no data is expected (e.g. trigger)
225 if(accessor->getNumberOfSamples() != 0) {
226 accessor->accessChannel(0) = accessors[changedVariable]->accessChannel(0);
227 }
228
229 bool dataLoss = accessor->writeDestructively(version);
230 if(dataLoss) {
231 Application::incrementDataLossCounter(accessor->getName());
232 }
233 }
234
235 // receive data
236 boost::this_thread::interruption_point();
237 changedVariable = group.readAny();
238 boost::this_thread::interruption_point();
239
240 version = accessors[changedVariable]->getVersionNumber();
241 }
242 }
