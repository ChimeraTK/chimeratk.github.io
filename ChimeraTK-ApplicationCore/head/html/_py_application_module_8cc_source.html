25 void PyApplicationModule::run() {
26 {
27 py::gil_scoped_acquire gil;
28
29 auto locals = py::dict("theModule"_a = *this);
30 py::exec(R"(
31 def mainLoopWrapper2(self):
32 try:
33 self.mainLoopWrapper()
34 except ThreadInterrupted as e:
35 pass
36 except Exception as e:
37 print("Exception in module "+self.getName()+":")
38 traceback.print_exception(e)
39 theModule.mainLoopWrapper2 = mainLoopWrapper2.__get__(theModule)
40 )",
41 py::globals(), locals);
42 _myThread = py::eval("threading.Thread(target=theModule.mainLoopWrapper2)", py::globals(), locals);
43 Application::getInstance().getTestableMode().unlock("releaseForPythonModuleStart");
44 _myThread.attr("start")();
45 }
46 // must release GIL before acquiring testable mode lock, to avoid deadlock
47 Application::getInstance().getTestableMode().lock("acquireForPythonModuleStart");
48 }

52 void PyApplicationModule::terminate() {
53 ApplicationModule::terminate();
54
55 // The module was not started
56 if(!_myThread) {
57 return;
58 }
59
60 py::gil_scoped_acquire gil;
61 while(_myThread.attr("is_alive")().cast<bool>()) {
62 for(auto& var : getAccessorListRecursive()) {
63 auto el{var.getAppAccessorNoType().getHighLevelImplElement()};
64 el->interrupt();
65 }
66
67 _myThread.attr("join")(0.01);
68 }
69 }

73 void PyApplicationModule::bind(py::module& m) {
74 // on Python side, PythonApplicationModule derives from PyVariableGroup although on C++ side, it does not.
75 // So we specify inheritance on Python side by constructor args.
76 py::class_<PyApplicationModule, PythonApplicationModuleTrampoline, VariableGroup,
77 std::unique_ptr<PyApplicationModule, py::nodelete>>
78 cam(m, "ApplicationModule", py::multiple_inheritance());
79
80 cam.def(py::init([](ModuleGroup& owner, const std::string& name, const std::string& description,
81 const std::unordered_set<std::string>& tags) {
82 return dynamic_cast<PyOwningObject&>(owner).make_child<PythonApplicationModuleTrampoline>(
83 &owner, name, description, tags);
84 }),
85 py::return_value_policy::reference,
86 // doc and default args:
87 "", py::arg("owner"), py::arg("name"), py::arg("description"),
88 py::arg("tags") = std::unordered_set<std::string>{});
89
90 cam.def("mainLoopWrapper", &PyApplicationModule::mainLoopWrapper, py::call_guard<py::gil_scoped_release>())
91 .def("appConfig",
92 []([[maybe_unused]] PyApplicationModule& pam) { return PyConfigReader(PyApplicationModule::appConfig()); })
93 .def("getDataValidity", &PyApplicationModule::getDataValidity,
94 "Return the data validity flag.\n\nIf any This function will be called by all output accessors in their "
95 "write functions.")
96 .def("getCurrentVersionNumber", &PyApplicationModule::getCurrentVersionNumber,
97 "Return the current version number which has been received with the last push-type read operation.")
98 .def("setCurrentVersionNumber", &PyApplicationModule::setCurrentVersionNumber,
99 "Set the current version number.\n\nThis function is called by the push-type input accessors in their read "
100 "functions.",
101 py::arg("versionNumber"))
102 // TODO: Bind Model .def("getModel", &PythonApplicationModule::getModel)
103 .def("incrementDataFaultCounter", &PyApplicationModule::incrementDataFaultCounter,
104 "Set the data validity flag to fault and increment the fault counter.\n\nThis function will be called by "
105 "all input accessors when receiving the a faulty update if the previous update was ok. The caller of this "
106 "function must ensure that calls to this function are paired to a subsequent call to "
107 "decrementDataFaultCounter().")
108 .def("decrementDataFaultCounter", &PyApplicationModule::decrementDataFaultCounter,
109 "Decrement the fault counter and set the data validity flag to ok if the counter has reached 0.\n\nThis "
110 "function will be called by all input accessors when receiving the an ok update if the previous update was "
111 "faulty. The caller of this function must ensure that calles to this function are paired to a previous "
112 "call to incrementDataFaultCounter().")
113 .def("getDataFaultCounter", &PyApplicationModule::getDataFaultCounter,
114 "Get the Number of inputs which report DataValidity::faulty.")
115 .def(
116 "logger",
117 [](PyApplicationModule& module, Logger::Severity severity) {
118 // Get name of python implementation, not the C++ class, which would be PythonApplicationModule
119 auto pythonClassName = py::cast(&module).get_type().attr("__name__").cast<std::string>();
120 return PyLoggerStreamProxy(severity, pythonClassName);
121 },
122 "Convenicene function to obtain a logger stream with the given Severity.\n\nThe context string will be "
123 "obtained from the class name of the module.")
124 .def("disable", &PyApplicationModule::disable,
125 "Disable the module such that it is not part of the Application.");
126 }
