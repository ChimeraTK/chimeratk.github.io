24  void PyApplicationModule::run() {
25  {
26  py::gil_scoped_acquire gil;
27 
28  auto locals = py::dict("theModule"_a = *this);
29  py::exec(R"(
30  def mainLoopWrapper2(self):
31  try:
32  self.mainLoopWrapper()
33  except ThreadInterrupted as e:
34  pass
35  except Exception as e:
36  print("Exception in module "+self.getName()+":")
37  traceback.print_exception(e)
38  theModule.mainLoopWrapper2 = mainLoopWrapper2.__get__(theModule)
39  )",
40  py::globals(), locals);
41  _myThread = py::eval("threading.Thread(target=theModule.mainLoopWrapper2)", py::globals(), locals);
42  Application::getInstance().getTestableMode().unlock("releaseForPythonModuleStart");
43  _myThread.attr("start")();
44  }
45  // must release GIL before acquiring testable mode lock, to avoid deadlock
46  Application::getInstance().getTestableMode().lock("acquireForPythonModuleStart");
47  }

52  ApplicationModule::terminate();
53 
54  // The module was not started
55  if(!_myThread) {
56  return;
57  }
58 
59  py::gil_scoped_acquire gil;

63  if(el->getAccessModeFlags().has(AccessMode::wait_for_new_data)) {

66  }
