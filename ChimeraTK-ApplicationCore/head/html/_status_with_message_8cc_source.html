39 void StatusWithMessage::writeOkIfDifferent() {
40 if(_status != StatusOutput::Status::OK || _status.checkMetadataWriteDifference<int32_t>()) {
41 setOk();
42 writeAll();
43 }
44 // This assert makes sure the above "if" condition is sufficient. There is no way to set the status to OK and have
45 // a non-empty message string.

47 }
