181 template<typename UserType>
182 void ArrayAccessor<UserType>::writeIfDifferent(const std::vector<UserType>& newValue) {
183 // Need to get to the MetaDataPropagatingRegisterDecorator to obtain the last written data validity for this PV.
184 // The dynamic_cast is ok, since the MetaDataPropagatingRegisterDecorator is always the outermost accessor, cf.
185 // the data validity propagation specification, Section 2.5.1.
186 auto* targetMetaDataPropagatingDecorator =
187 dynamic_cast<MetaDataPropagatingRegisterDecorator<UserType>*>(this->get());
188 assert(targetMetaDataPropagatingDecorator != nullptr);
189
190 // In contrast to OneDRegisterAccessor::writeIfDifferent(), we must not set the data validity on the target
191 // accessor, since that would be interpreted by the MetaDataPropagatingRegisterDecorator as an application-induced
192 // forced fault state. This would result in invalidity lock-ups if this happens in a circular network. Hence the
193 // comparison of the data validity must also be done against the validity of the decorator's target accessor which
194 // corresponds to the last written data validity for this PV.
195 if(!std::equal(newValue.begin(), newValue.end(), this->get()->accessChannel(0).begin()) ||
196 this->getVersionNumber() == VersionNumber(nullptr) ||
197 targetMetaDataPropagatingDecorator->getTargetValidity() != this->getOwner()->getDataValidity()) {
198 setAndWrite(newValue);
199 }
200 }
