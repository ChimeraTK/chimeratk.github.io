1502  template<typename VISITOR, typename... Args>
1503  auto Impl::visit(Vertex startVertex, VISITOR visitor, Args... args) {
1504  // increase visiting level to prevent modification while iterating, decrease when leaving this function
1505  _graphVisitingLevel++;
1506  auto decrementVisitingLevel = cppext::finally([&] { _graphVisitingLevel--; });
1507 
1508  auto filteredGraph = getFilteredGraph(args...);
1509 
1510  auto vertexFilter = Model::getVertexFilter(args...);
1511 
1512  // Execute search operation based on selected type
1513  auto typeHolder = Model::getSearchType<Args...>();
1514  using type = typename decltype(typeHolder)::type;
1515  constexpr bool isAdjacent =
1516  std::is_same<AdjacentSearch, type>::value || std::is_base_of<AdjacentSearch, type>::value;
1517  constexpr bool isAdjacentIn =
1518  std::is_same<AdjacentInSearch, type>::value || std::is_base_of<AdjacentInSearch, type>::value;
1519  constexpr bool isAdjacentOut =
1520  std::is_same<AdjacentOutSearch, type>::value || std::is_base_of<AdjacentOutSearch, type>::value;
1521  constexpr bool isDFS =
1522  std::is_same<DepthFirstSearch, type>::value || std::is_base_of<DepthFirstSearch, type>::value;
1523  constexpr bool isBFS =
1524  std::is_same<BreadthFirstSearch, type>::value || std::is_base_of<BreadthFirstSearch, type>::value;
1525 
1526  constexpr bool continueDisjunct = Model::hasSearchOption<ContinueSearchDisjunctTrees, Args...>();
1527  constexpr bool returnFirst = Model::hasSearchOption<ReturnFirstHit, Args...>();
1528 
1529  // Wrapper for visitor to filter wrong objecttypes in constexpr condition, so the actual visitor does not have
1530  // to work with all proxy types, just the ones which can get past the VisitorFilter.
1531  using VisitorReturnType = detail::VisitorReturnType<VISITOR, decltype(vertexFilter)>;
1532  auto visitorWrapper = [&](auto proxy) -> VisitorReturnType {
1533  if constexpr(vertexFilter.template constevalObjecttype<decltype(proxy)>()) {
1534  return visitor(proxy);
1535  }
1536  // The VertexFilter makes sure that this function is never called with the wrong type, hence if we end up here
1537  // there is a bug in the VertexFilter. Note that this code will get instantiated, just never executed.
1538  throw ChimeraTK::logic_error("We should never end up here...");
1539  };
1540 
1541  if constexpr(isAdjacent || isAdjacentOut) {
1542  auto [start, end] = boost::out_edges(startVertex, filteredGraph);
1543  for(auto it = start; it != end; ++it) {
1544  // obtain target vertex of the current outgoing edge (i.e. vertex at the other end of the edge)
1545  auto vtx = target(*it, _graph);
1546 
1547  // apply vertex filter
1548  if(!vertexFilter.evalVertexFilter(_graph[vtx])) {
1549  continue;
1550  }
1551 
1552  if constexpr(returnFirst) {
1553  return filteredGraph[vtx].visitProxy(visitorWrapper, vtx, shared_from_this());
1554  }
1555  else {

1657  // iterators directly. Only when using visit() internally, iterators are used, but visit() increments the counter
1658  // itself.

1662  path = path.substr(2);
1663  }

1667  visitor(startProxy);
1668  return true;

1670

1672  if(boost::starts_with(path, "../") || path == "..") {
1673  // remove the component from the path
1674  path = (path.size() > 2) ? path.substr(3) : "";

1677  bool found = false;

1679  startProxy._d->vertex,

1690  }
1691

1694  // remove the leading slash
1695  path = path.substr(1);

1710  startProxy._d->vertex,

1716  }
