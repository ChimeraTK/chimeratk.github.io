23 bool TestableMode::Lock::tryLockFor(std::chrono::seconds timeout, bool shared) {
24 assert(!_ownsLock);
25 _isShared = shared;
26 if(shared) {
27 _ownsLock = _mutex.try_lock_shared_for(timeout);
28 return _ownsLock;
29 }
30 _ownsLock = _mutex.try_lock_for(timeout);
31 return _ownsLock;
32 }

95 void TestableMode::lock(const std::string& name, bool shared) {
96 // don't do anything if testable mode is not enabled
97 if(not _enabled) {
98 return;
99 }
100
101 // debug output if enabled (also prevent spamming the same message)
102 if(_enableDebug && _repeatingMutexOwner == 0) { // LCOV_EXCL_LINE
103 // (only cout)
104 logger(Logger::Severity::debug, "TestableMode")
105 << "Application::testableModeLock(): Thread " << threadName() // LCOV_EXCL_LINE (only cout)
106 << " tries to obtain lock for " << name; // LCOV_EXCL_LINE (only cout)
107 } // LCOV_EXCL_LINE (only cout)
108
109 // if last lock was obtained repeatedly by the same thread, sleep a short time
110 // before obtaining the lock to give the other threads a chance to get the
111 // lock first
112 if(_repeatingMutexOwner > 0) {
113 usleep(10000);
114 }
115
116 // obtain the lock
117 boost::thread::id lastSeen_lastOwner = _lastMutexOwner;
118 repeatTryLock:
119 auto success = getLockObject().tryLockFor(std::chrono::seconds(30), shared);
120 boost::thread::id currentLastOwner = _lastMutexOwner;
121 if(!success) {
122 if(currentLastOwner != lastSeen_lastOwner) {
123 lastSeen_lastOwner = currentLastOwner;
124 goto repeatTryLock;
125 }
126
127 std::cerr << "testableModeLock(): Thread " << threadName() // LCOV_EXCL_LINE
128 << " could not obtain lock for 30 seconds, presumably because " // LCOV_EXCL_LINE
129 << threadName(_lastMutexOwner) << " [" << pthreadId(_lastMutexOwner) << "] does not release it."
130 << std::endl; // LCOV_EXCL_LINE
131
132 // throw a specialised exception to make sure whoever catches it really knows what he does...
133 terminateTestStalled(); // LCOV_EXCL_LINE
134 } // LCOV_EXCL_LINE
135
136 // check if the last owner of the mutex was this thread, which may be a hint
137 // that no other thread is waiting for the lock
138 if(currentLastOwner == boost::this_thread::get_id()) {
139 // debug output if enabled
140 if(_enableDebug && _repeatingMutexOwner == 0) { // LCOV_EXCL_LINE
141 // (only cout)
142 std::cerr << "testableModeLock(): Thread " << threadName() // LCOV_EXCL_LINE (only cout)
143 << " repeatedly obtained lock successfully for " // LCOV_EXCL_LINE (only cout)
144 << name // LCOV_EXCL_LINE (only cout)
145 << ". Further messages will be suppressed." << std::endl; // LCOV_EXCL_LINE (only cout)
146 } // LCOV_EXCL_LINE (only cout)
147
148 // increase counter for stall detection
149 _repeatingMutexOwner++;
150
151 // detect stall: if the same thread got the mutex with no other thread
152 // obtaining it in between for one second, we assume no other thread is able
153 // to process data at this time. The test should fail in this case
154 if(_repeatingMutexOwner > 100) {
155 // print an informative message first, which lists also all variables
156 // currently containing unread data.
157 std::cerr << "*** Tests are stalled due to data which has been sent but not received.\n";
158 std::cerr
159 << " The following variables still contain unread values or had data loss due to a queue overflow:\n";
160 for(auto& pair : _variables) {
161 const auto& variable = pair.second;
162 if(variable.counter > 0) {
163 std::cerr << " - " << variable.name << " [" << variable.processVariable->getId() << "]";
164 // check if process variable still has data in the queue
165 try {
166 if(variable.processVariable->readNonBlocking()) {
167 std::cerr << " (unread data in queue)";
168 }
169 else {
170 std::cerr << " (data loss)";
171 }
172 }
173 catch(ChimeraTK::logic_error&) {
174 // if we receive a logic_error in readNonBlocking() it just means
175 // another thread is waiting on a TransferFuture of this variable,
176 // and we actually were not allowed to read...
177 std::cerr << " (data loss)";
178 }
179 std::cerr << "\n";
180 }
181 }
182 std::cerr << "(end of list)\n";
183 // Check for modules waiting for initial values (prints nothing if there are no such modules)
184 Application::getInstance()._circularDependencyDetector.printWaiters(std::cerr);
185 // throw a specialised exception to make sure whoever catches it really knows what he does...
186 terminateTestStalled();
187 }
188 }
189 else {
190 // last owner of the mutex was different: reset the counter and store the
191 // thread id
192 _repeatingMutexOwner = 0;
193 _lastMutexOwner = boost::this_thread::get_id();
194
195 // debug output if enabled
196 if(_enableDebug) { // LCOV_EXCL_LINE (only cout)
197 logger(Logger::Severity::debug, "TestableMode")
198 << "TestableMode::lock(): Thread " << threadName() // LCOV_EXCL_LINE (only cout)
199 << " obtained lock successfully for " << name; // LCOV_EXCL_LINE (only cout)
200 } // LCOV_EXCL_LINE (only cout)
201 }
202 }

206 void TestableMode::unlock(const std::string& name) {
207 if(not _enabled) {
208 return;
209 }
210 if(_enableDebug &&
211 (not _repeatingMutexOwner // LCOV_EXCL_LINE (only cout)
212 || boost::thread::id(_lastMutexOwner) != boost::this_thread::get_id())) { // LCOV_EXCL_LINE (only cout)
213 logger(Logger::Severity::debug, "TestableMode")
214 << "TestableMode::unlock(): Thread " << threadName() // LCOV_EXCL_LINE (only cout)
215 << " releases lock for " << name; // LCOV_EXCL_LINE (only cout)
216 }
217 // LCOV_EXCL_LINE (only cout)
218 getLockObject().unlock();
219 }

223 void TestableMode::step(bool waitForDeviceInitialisation) {
224 // testableMode.counter must be non-zero, otherwise there is no input for the application to process. It is also
225 // sufficient if testableMode.deviceInitialisationCounter is non-zero, if waitForDeviceInitialisation == true. In
226 // that case we only wait for the device initialisation to be completed.
227 if(_counter == 0 && (!waitForDeviceInitialisation || _deviceInitialisationCounter == 0)) {
228 throw ChimeraTK::logic_error("Application::stepApplication() called despite no input was provided "
229 "to the application to process!");
230 }
231 // let the application run until it has processed all data (i.e. the semaphore
232 // counter is 0)
233 size_t oldCounter = 0;
234 while(_counter > 0 || (waitForDeviceInitialisation && _deviceInitialisationCounter > 0)) {
235 if(_enableDebug && (oldCounter != _counter)) { // LCOV_EXCL_LINE (only cout)
236 logger(Logger::Severity::debug, "TestableMode")
237 << "Application::stepApplication(): testableMode.counter = " << _counter; // LCOV_EXCL_LINE (only cout)
238 oldCounter = _counter; // LCOV_EXCL_LINE (only cout)
239 }
240 unlock("stepApplication");
241 boost::this_thread::yield();
242 lock("stepApplication", false);
243 }
244 }

257 std::string TestableMode::threadName(const boost::thread::id& threadId) {
258 std::unique_lock<std::mutex> myLock(_threadNamesMutex);
259 if(auto const& it = _threadNames.find(threadId); it != _threadNames.end()) {
