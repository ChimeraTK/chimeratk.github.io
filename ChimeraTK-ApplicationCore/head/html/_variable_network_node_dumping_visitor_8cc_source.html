18  void VariableNetworkNodeDumpingVisitor::dispatch(const VariableNetworkNode& t) {
19  if(t.getType() == NodeType::Application) {
20  stream() << " type = Application ('" << t.getQualifiedName() << "')";
21  }
22  if(t.getType() == NodeType::ControlSystem) {
23  stream() << " type = ControlSystem ('" << t.getPublicName() << "')";
24  }
25  if(t.getType() == NodeType::Device) {
26  stream() << " type = Device (" << t.getDeviceAlias() << ": " << t.getRegisterName() << ")";
27  }
28  if(t.getType() == NodeType::TriggerReceiver) {
29  stream() << " type = TriggerReceiver";
30  }
31  if(t.getType() == NodeType::Constant) {
32  stream() << " type = Constant";
33  }
34  if(t.getType() == NodeType::invalid) {
35  stream() << " type = **invalid**";
36  }
37 
38  if(t.getMode() == UpdateMode::push) {
39  stream() << _separator << "pushing";
40  }
41  if(t.getMode() == UpdateMode::poll) {
42  stream() << _separator << "polling";
43  }
44  if(t.getDirection().withReturn) {
45  stream() << _separator << "with return";
46  }
47  stream() << _separator;
48 
49  stream() << "data type: " << boost::core::demangle(t.getValueType().name());
50  stream() << _separator;
51  stream() << "length: " << t.getNumberOfElements();
52  stream() << _separator;
53 
54  stream() << "[ptr: " << &(*(t.pdata)) << "]";
55  stream() << _separator;
56 
57  stream() << "tags: [";
58  bool first = true;
59  for(const auto& tag : t.getTags()) {
60  if(!first) {
61  stream() << ",";
62  }
63  stream() << tag;
64  first = false;
65  }
66  stream() << "]";
67  stream() << _separator;
68 
69  if(t.isCircularInput()) {
70  stream() << "circular dependency " << t.pdata->circularNetworkHash;
71  stream() << _separator;
72  }
73 
74  stream() << std::endl;
75  }
