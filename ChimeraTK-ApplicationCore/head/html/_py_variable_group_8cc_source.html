21 .def("getName", &VariableGroup::getName, "Get the name of the module instance.")

30 },
31 "Read all readable variables in the group.\n\nIf there are push-type variables in the group, this call "
32 "will block until all of the variables have received an update. All push-type variables are read first, "
33 "the poll-type variables are therefore updated with the latest values upon return. includeReturnChannels "
34 "determines whether return channels of *OutputRB accessors are included in the read.",
35 py::arg("includeReturnChannels") = false)
